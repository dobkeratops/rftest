   52  impl<T> Map<uint, T> for TrieMap<T> {
   53      /// Return a reference to the value corresponding to the key
   54      #[inline]
   55      fn find<'a>(&'a self, key: &uint) -> Option<&'a T> {
   56          let mut node: &'a TrieNode<T> = &self.root;
   57          let mut idx = 0;
   58          loop {
   59              match node.children[chunk(*key, idx)] {
   60                Internal(ref x) => node = &**x,
   61                External(stored, ref value) => {
   62                  if stored == *key {
   63                      return Some(value)
   64                  } else {
   65                      return None
   66                  }
   67                }
   68                Nothing => return None
   69              }
   70              idx += 1;
   71          }
   72      }
   73  }

   75  impl<T> MutableMap<uint, T> for TrieMap<T> {
   76      /// Return a mutable reference to the value corresponding to the key
   77      #[inline]
   78      fn find_mut<'a>(&'a mut self, key: &uint) -> Option<&'a mut T> {
   79          find_mut(&mut self.root.children[chunk(*key, 0)], *key, 1)
   80      }
   81  
   82      /// Insert a key-value pair from the map. If the key already had a value
   83      /// present in the map, that value is returned. Otherwise None is returned.
   84      fn swap(&mut self, key: uint, value: T) -> Option<T> {
   85          let ret = insert(&mut self.root.count,
   86                           &mut self.root.children[chunk(key, 0)],
   87                           key, value, 1);
   88          if ret.is_none() { self.length += 1 }
   89          ret
   90      }
   91  
   92      /// Removes a key from the map, returning the value at the key if the key
   93      /// was previously in the map.
   94      fn pop(&mut self, key: &uint) -> Option<T> {
   95          let ret = remove(&mut self.root.count,
   96                           &mut self.root.children[chunk(*key, 0)],
   97                           *key, 1);
   98          if ret.is_some() { self.length -= 1 }
   99          ret
  100      }
  101  }

  140  // FIXME #5846 we want to be able to choose between &x and &mut x
  141  // (with many different `x`) below, so we need to optionally pass mut
  142  // as a tt, but the only thing we can do with a `tt` is pass them to
  143  // other macros, so this takes the `& <mutability> <operand>` token
  144  // sequence and forces their evalutation as an expression. (see also
  145  // `item!` below.)
  146  macro_rules! addr { ($e:expr) => { $e } }
  147  
  148  macro_rules! bound {
  149      ($iterator_name:ident,
  150       // the current treemap
  151       self = $this:expr,
  152       // the key to look for
  153       key = $key:expr,
  154       // are we looking at the upper bound?
  155       is_upper = $upper:expr,
  156  
  157       // method names for slicing/iterating.
  158       slice_from = $slice_from:ident,
  159       iter = $iter:ident,
  160  
  161       // see the comment on `addr!`, this is just an optional mut, but
  162       // there's no 0-or-1 repeats yet.
  163       mutability = $($mut_:tt)*) => {
  164          {
  165              // # For `mut`
  166              // We need an unsafe pointer here because we are borrowing
  167              // mutable references to the internals of each of these
  168              // mutable nodes, while still using the outer node.
  169              //
  170              // However, we're allowed to flaunt rustc like this because we
  171              // never actually modify the "shape" of the nodes. The only
  172              // place that mutation is can actually occur is of the actual
  173              // values of the TrieMap (as the return value of the
  174              // iterator), i.e. we can never cause a deallocation of any
  175              // TrieNodes so the raw pointer is always valid.
  176              //
  177              // # For non-`mut`
  178              // We like sharing code so much that even a little unsafe won't
  179              // stop us.
  180              let this = $this;
  181              let mut node = addr!(& $($mut_)* this.root as * $($mut_)* TrieNode<T>);
  182  
  183              let key = $key;
  184  
  185              let mut it = unsafe {$iterator_name::new()};
  186              // everything else is zero'd, as we want.
  187              it.remaining_max = this.length;
  188  
  189              // this addr is necessary for the `Internal` pattern.
  190              addr!(loop {
  191                      let children = unsafe {addr!(& $($mut_)* (*node).children)};
  192                      // it.length is the current depth in the iterator and the
  193                      // current depth through the `uint` key we've traversed.
  194                      let child_id = chunk(key, it.length);
  195                      let (slice_idx, ret) = match children[child_id] {
  196                          Internal(ref $($mut_)* n) => {
  197                              node = addr!(& $($mut_)* **n as * $($mut_)* TrieNode<T>);
  198                              (child_id + 1, false)
  199                          }
  200                          External(stored, _) => {
  201                              (if stored < key || ($upper && stored == key) {
  202                                  child_id + 1
  203                              } else {
  204                                  child_id
  205                              }, true)
  206                          }
  207                          Nothing => {
  208                              (child_id + 1, true)
  209                          }
  210                      };
  211                      // push to the stack.
  212                      it.stack[it.length] = children.$slice_from(slice_idx).$iter();
  213                      it.length += 1;
  214                      if ret { return it }
  215                  })
  216          }
  217      }
  218  }

  382  impl<T> TrieNode<T> {
  383      fn each_reverse<'a>(&'a self, f: |&uint, &'a T| -> bool) -> bool {
  384          for elt in self.children.rev_iter() {
  385              match *elt {
  386                  Internal(ref x) => if !x.each_reverse(|i,t| f(i,t)) { return false },
  387                  External(k, ref v) => if !f(&k, v) { return false },
  388                  Nothing => ()
  389              }
  390          }
  391          true
  392      }
  393  }

  395  // if this was done via a trait, the key could be generic
  396  #[inline]
  397  fn chunk(n: uint, idx: uint) -> uint {
  398      let sh = uint::BITS - (SHIFT * (idx + 1));
  399      (n >> sh) & MASK
  400  }
  401  
  402  fn find_mut<'r, T>(child: &'r mut Child<T>, key: uint, idx: uint) -> Option<&'r mut T> {
  403      match *child {
  404          External(stored, ref mut value) if stored == key => Some(value),
  405          External(..) => None,
  406          Internal(ref mut x) => find_mut(&mut x.children[chunk(key, idx)], key, idx + 1),
  407          Nothing => None
  408      }
  409  }

  411  fn insert<T>(count: &mut uint, child: &mut Child<T>, key: uint, value: T,
  412               idx: uint) -> Option<T> {
  413      // we branch twice to avoid having to do the `replace` when we
  414      // don't need to; this is much faster, especially for keys that
  415      // have long shared prefixes.
  416      match *child {
  417          Nothing => {
  418              *count += 1;
  419              *child = External(key, value);
  420              return None;
  421          }
  422          Internal(ref mut x) => {
  423              return insert(&mut x.count, &mut x.children[chunk(key, idx)], key, value, idx + 1);
  424          }
  425          External(stored_key, ref mut stored_value) if stored_key == key => {
  426              // swap in the new value and return the old.
  427              return Some(mem::replace(stored_value, value));
  428          }
  429          _ => {}
  430      }
  431  
  432      // conflict, an external node with differing keys: we have to
  433      // split the node, so we need the old value by value; hence we
  434      // have to move out of `child`.
  435      match mem::replace(child, Nothing) {
  436          External(stored_key, stored_value) => {
  437              let mut new = ~TrieNode::new();
  438              insert(&mut new.count,
  439                     &mut new.children[chunk(stored_key, idx)],
  440                     stored_key, stored_value, idx + 1);
  441              let ret = insert(&mut new.count, &mut new.children[chunk(key, idx)],
  442                           key, value, idx + 1);
  443              *child = Internal(new);
  444              return ret;
  445          }
  446          _ => unreachable!()
  447      }
  448  }
  449  
  450  fn remove<T>(count: &mut uint, child: &mut Child<T>, key: uint,
  451               idx: uint) -> Option<T> {
  452      let (ret, this) = match *child {
  453        External(stored, _) if stored == key => {
  454          match mem::replace(child, Nothing) {
  455              External(_, value) => (Some(value), true),
  456              _ => fail!()
  457          }
  458        }
  459        External(..) => (None, false),
  460        Internal(ref mut x) => {
  461            let ret = remove(&mut x.count, &mut x.children[chunk(key, idx)],
  462                             key, idx + 1);
  463            (ret, x.count == 0)
  464        }
  465        Nothing => (None, false)
  466      };
  467  
  468      if this {
  469          *child = Nothing;
  470          *count -= 1;
  471      }
  472      return ret;
  473  }

  527          item!(impl<'a, T> Iterator<(uint, &'a $($mut_)* T)> for $name<'a, T> {
  528                  // you might wonder why we're not even trying to act within the
  529                  // rules, and are just manipulating raw pointers like there's no
  530                  // such thing as invalid pointers and memory unsafety. The
  531                  // reason is performance, without doing this we can get the
  532                  // bench_iter_large microbenchmark down to about 30000 ns/iter
  533                  // (using .unsafe_ref to index self.stack directly, 38000
  534                  // ns/iter with [] checked indexing), but this smashes that down
  535                  // to 13500 ns/iter.
  536                  //
  537                  // Fortunately, it's still safe...
  538                  //
  539                  // We have an invariant that every Internal node
  540                  // corresponds to one push to self.stack, and one pop,
  541                  // nested appropriately. self.stack has enough storage
  542                  // to store the maximum depth of Internal nodes in the
  543                  // trie (8 on 32-bit platforms, 16 on 64-bit).
  544                  fn next(&mut self) -> Option<(uint, &'a $($mut_)* T)> {
  545                      let start_ptr = self.stack.as_mut_ptr();
  546  
  547                      unsafe {
  548                          // write_ptr is the next place to write to the stack.
  549                          // invariant: start_ptr <= write_ptr < end of the
  550                          // vector.
  551                          let mut write_ptr = start_ptr.offset(self.length as int);
  552                          while write_ptr != start_ptr {
  553                              // indexing back one is safe, since write_ptr >
  554                              // start_ptr now.
  555                              match (*write_ptr.offset(-1)).next() {
  556                                  // exhausted this iterator (i.e. finished this
  557                                  // Internal node), so pop from the stack.
  558                                  //
  559                                  // don't bother clearing the memory, because the
  560                                  // next time we use it we'll've written to it
  561                                  // first.
  562                                  None => write_ptr = write_ptr.offset(-1),
  563                                  Some(child) => {
  564                                      addr!(match *child {
  565                                              Internal(ref $($mut_)* node) => {
  566                                                  // going down a level, so push
  567                                                  // to the stack (this is the
  568                                                  // write referenced above)
  569                                                  *write_ptr = node.children.$iter();
  570                                                  write_ptr = write_ptr.offset(1);
  571                                              }
  572                                              External(key, ref $($mut_)* value) => {
  573                                                  self.remaining_max -= 1;
  574                                                  if self.remaining_min > 0 {
  575                                                      self.remaining_min -= 1;
  576                                                  }
  577                                                  // store the new length of the
  578                                                  // stack, based on our current
  579                                                  // position.
  580                                                  self.length = (write_ptr as uint
  581                                                                 - start_ptr as uint) /
  582                                                      mem::size_of_val(&*write_ptr);
  583  
  584                                                  return Some((key, value));
  585                                              }
  586                                              Nothing => {}
  587                                          })
  588                                  }
  589                              }
  590                          }
  591                      }
  592                      return None;
  593                  }

  596                  fn size_hint(&self) -> (uint, Option<uint>) {
  597                      (self.remaining_min, Some(self.remaining_max))
  598                  }
  599              })
  600      }
  601  }

   11  //! A doubly-linked list with owned nodes.
   12  //!
   13  //! The DList allows pushing and popping elements at either end.
   14  //!
   15  //! DList implements the trait Deque. It should be imported with `use
   16  //! collections::deque::Deque`.
   17  
   18  
   19  // DList is constructed like a singly-linked list over the field `next`.
   20  // including the last link being None; each Node owns its `next` field.
   21  //
   22  // Backlinks over DList::prev are raw pointers that form a full chain in
   23  // the reverse direction.
   24  
   25  use std::cast;
   26  use std::mem::{replace, swap};
   27  use std::ptr;
   28  use std::iter::Rev;
   29  use std::iter;
   30  
   31  use deque::Deque;

   33  /// A doubly-linked list.
   34  pub struct DList<T> {
   35      length: uint,
   36      list_head: Link<T>,
   37      list_tail: Rawlink<Node<T>>,
   38  }
   39  
   40  type Link<T> = Option<~Node<T>>;
   41  struct Rawlink<T> { p: *mut T }
   42  
   43  struct Node<T> {
   44      next: Link<T>,
   45      prev: Rawlink<Node<T>>,
   46      value: T,
   47  }
   48  
   49  /// Double-ended DList iterator
   50  pub struct Items<'a, T> {
   51      head: &'a Link<T>,
   52      tail: Rawlink<Node<T>>,
   53      nelem: uint,
   54  }

   75  /// Rawlink is a type like Option<T> but for holding a raw pointer
   76  impl<T> Rawlink<T> {
   77      /// Like Option::None for Rawlink
   78      fn none() -> Rawlink<T> {
   79          Rawlink{p: ptr::mut_null()}
   80      }
   81  
   82      /// Like Option::Some for Rawlink
   83      fn some(n: &mut T) -> Rawlink<T> {
   84          Rawlink{p: n}
   85      }
   86  
   87      /// Convert the `Rawlink` into an Option value
   88      fn resolve_immut(&self) -> Option<&T> {
   89          unsafe { self.p.to_option() }
   90      }
   91  
   92      /// Convert the `Rawlink` into an Option value
   93      fn resolve(&mut self) -> Option<&mut T> {
   94          if self.p.is_null() {
   95              None
   96          } else {
   97              Some(unsafe { cast::transmute(self.p) })
   98          }
   99      }
  100  
  101      /// Return the `Rawlink` and replace with `Rawlink::none()`
  102      fn take(&mut self) -> Rawlink<T> {
  103          replace(self, Rawlink::none())
  104      }
  105  }

  114  impl<T> Node<T> {
  115      fn new(v: T) -> Node<T> {
  116          Node{value: v, next: None, prev: Rawlink::none()}
  117      }
  118  }
  119  
  120  /// Set the .prev field on `next`, then return `Some(next)`
  121  fn link_with_prev<T>(mut next: ~Node<T>, prev: Rawlink<Node<T>>) -> Link<T> {
  122      next.prev = prev;
  123      Some(next)
  124  }

  150  impl<T> DList<T> {
  151      /// Add a Node first in the list
  152      #[inline]
  153      fn push_front_node(&mut self, mut new_head: ~Node<T>) {
  154          match self.list_head {
  155              None => {
  156                  self.list_tail = Rawlink::some(new_head);
  157                  self.list_head = link_with_prev(new_head, Rawlink::none());
  158              }
  159              Some(ref mut head) => {
  160                  new_head.prev = Rawlink::none();
  161                  head.prev = Rawlink::some(new_head);
  162                  swap(head, &mut new_head);
  163                  head.next = Some(new_head);
  164              }
  165          }
  166          self.length += 1;
  167      }
  168  
  169      /// Remove the first Node and return it, or None if the list is empty
  170      #[inline]
  171      fn pop_front_node(&mut self) -> Option<~Node<T>> {
  172          self.list_head.take().map(|mut front_node| {
  173              self.length -= 1;
  174              match front_node.next.take() {
  175                  Some(node) => self.list_head = link_with_prev(node, Rawlink::none()),
  176                  None => self.list_tail = Rawlink::none()
  177              }
  178              front_node
  179          })
  180      }
  181  
  182      /// Add a Node last in the list
  183      #[inline]
  184      fn push_back_node(&mut self, mut new_tail: ~Node<T>) {
  185          match self.list_tail.resolve() {
  186              None => return self.push_front_node(new_tail),
  187              Some(tail) => {
  188                  self.list_tail = Rawlink::some(new_tail);
  189                  tail.next = link_with_prev(new_tail, Rawlink::some(tail));
  190              }
  191          }
  192          self.length += 1;
  193      }
  194  
  195      /// Remove the last Node and return it, or None if the list is empty
  196      #[inline]
  197      fn pop_back_node(&mut self) -> Option<~Node<T>> {
  198          self.list_tail.resolve().map_or(None, |tail| {
  199              self.length -= 1;
  200              self.list_tail = tail.prev;
  201              match tail.prev.resolve() {
  202                  None => self.list_head.take(),
  203                  Some(tail_prev) => tail_prev.next.take()
  204              }
  205          })
  206      }
  207  }

  209  impl<T> Deque<T> for DList<T> {
  210      /// Provide a reference to the front element, or None if the list is empty
  211      #[inline]
  212      fn front<'a>(&'a self) -> Option<&'a T> {
  213          self.list_head.as_ref().map(|head| &head.value)
  214      }
  215  
  216      /// Provide a mutable reference to the front element, or None if the list is empty
  217      #[inline]
  218      fn front_mut<'a>(&'a mut self) -> Option<&'a mut T> {
  219          self.list_head.as_mut().map(|head| &mut head.value)
  220      }
  221  
  222      /// Provide a reference to the back element, or None if the list is empty
  223      #[inline]
  224      fn back<'a>(&'a self) -> Option<&'a T> {
  225          let tmp = self.list_tail.resolve_immut(); // FIXME: #3511: shouldn't need variable
  226          tmp.as_ref().map(|tail| &tail.value)
  227      }
  228  
  229      /// Provide a mutable reference to the back element, or None if the list is empty
  230      #[inline]
  231      fn back_mut<'a>(&'a mut self) -> Option<&'a mut T> {
  232          let tmp: Option<&'a mut Node<T>> =
  233              self.list_tail.resolve(); // FIXME: #3511: shouldn't need variable
  234          tmp.map(|tail| &mut tail.value)
  235      }
  236  
  237      /// Add an element first in the list
  238      ///
  239      /// O(1)
  240      fn push_front(&mut self, elt: T) {
  241          self.push_front_node(~Node::new(elt))
  242      }
  243  
  244      /// Remove the first element and return it, or None if the list is empty
  245      ///
  246      /// O(1)
  247      fn pop_front(&mut self) -> Option<T> {
  248          self.pop_front_node().map(|~Node{value, ..}| value)
  249      }
  250  
  251      /// Add an element last in the list
  252      ///
  253      /// O(1)
  254      fn push_back(&mut self, elt: T) {
  255          self.push_back_node(~Node::new(elt))
  256      }
  257  
  258      /// Remove the last element and return it, or None if the list is empty
  259      ///
  260      /// O(1)
  261      fn pop_back(&mut self) -> Option<T> {
  262          self.pop_back_node().map(|~Node{value, ..}| value)
  263      }
  264  }

  266  impl<T> DList<T> {
  267      /// Create an empty DList
  268      #[inline]
  269      pub fn new() -> DList<T> {
  270          DList{list_head: None, list_tail: Rawlink::none(), length: 0}
  271      }
  272  
  273      /// Move the last element to the front of the list.
  274      ///
  275      /// If the list is empty, do nothing.
  276      #[inline]
  277      pub fn rotate_forward(&mut self) {
  278          self.pop_back_node().map(|tail| {
  279              self.push_front_node(tail)
  280          });
  281      }
  282  
  283      /// Move the first element to the back of the list.
  284      ///
  285      /// If the list is empty, do nothing.
  286      #[inline]
  287      pub fn rotate_backward(&mut self) {
  288          self.pop_front_node().map(|head| {
  289              self.push_back_node(head)
  290          });
  291      }
  292  
  293      /// Add all elements from `other` to the end of the list
  294      ///
  295      /// O(1)
  296      pub fn append(&mut self, mut other: DList<T>) {
  297          match self.list_tail.resolve() {
  298              None => *self = other,
  299              Some(tail) => {
  300                  // Carefully empty `other`.
  301                  let o_tail = other.list_tail.take();
  302                  let o_length = other.length;
  303                  match other.list_head.take() {
  304                      None => return,
  305                      Some(node) => {
  306                          tail.next = link_with_prev(node, self.list_tail);
  307                          self.list_tail = o_tail;
  308                          self.length += o_length;
  309                      }
  310                  }
  311              }
  312          }
  313      }
  314  
  315      /// Add all elements from `other` to the beginning of the list
  316      ///
  317      /// O(1)
  318      #[inline]
  319      pub fn prepend(&mut self, mut other: DList<T>) {
  320          swap(self, &mut other);
  321          self.append(other);
  322      }
  323  
  324      /// Insert `elt` before the first `x` in the list where `f(x, elt)` is true,
  325      /// or at the end.
  326      ///
  327      /// O(N)
  328      pub fn insert_when(&mut self, elt: T, f: |&T, &T| -> bool) {
  329          {
  330              let mut it = self.mut_iter();
  331              loop {
  332                  match it.peek_next() {
  333                      None => break,
  334                      Some(x) => if f(x, &elt) { break }
  335                  }
  336                  it.next();
  337              }
  338              it.insert_next(elt);
  339          }
  340      }
  341  
  342      /// Merge DList `other` into this DList, using the function `f`.
  343      /// Iterate the both DList with `a` from self and `b` from `other`, and
  344      /// put `a` in the result if `f(a, b)` is true, else `b`.
  345      ///
  346      /// O(max(N, M))
  347      pub fn merge(&mut self, mut other: DList<T>, f: |&T, &T| -> bool) {
  348          {
  349              let mut it = self.mut_iter();
  350              loop {
  351                  let take_a = match (it.peek_next(), other.front()) {
  352                      (_   , None) => return,
  353                      (None, _   ) => break,
  354                      (Some(ref mut x), Some(y)) => f(*x, y),
  355                  };
  356                  if take_a {
  357                      it.next();
  358                  } else {
  359                      it.insert_next_node(other.pop_front_node().unwrap());
  360                  }
  361              }
  362          }
  363          self.append(other);
  364      }

  423  #[unsafe_destructor]
  424  impl<T> Drop for DList<T> {
  425      fn drop(&mut self) {
  426          // Dissolve the dlist in backwards direction
  427          // Just dropping the list_head can lead to stack exhaustion
  428          // when length is >> 1_000_000
  429          let mut tail = self.list_tail;
  430          loop {
  431              match tail.resolve() {
  432                  None => break,
  433                  Some(prev) => {
  434                      prev.next.take(); // release ~Node<T>
  435                      tail = prev.prev;
  436                  }
  437              }
  438          }
  439          self.length = 0;
  440          self.list_head = None;
  441          self.list_tail = Rawlink::none();
  442      }
  443  }

  446  impl<'a, A> Iterator<&'a A> for Items<'a, A> {
  447      #[inline]
  448      fn next(&mut self) -> Option<&'a A> {
  449          if self.nelem == 0 {
  450              return None;
  451          }
  452          self.head.as_ref().map(|head| {
  453              self.nelem -= 1;
  454              self.head = &head.next;
  455              &head.value
  456          })
  457      }

  520  /// Allow mutating the DList while iterating
  521  pub trait ListInsertion<A> {
  522      /// Insert `elt` just after to the element most recently returned by `.next()`
  523      ///
  524      /// The inserted element does not appear in the iteration.
  525      fn insert_next(&mut self, elt: A);
  526  
  527      /// Provide a reference to the next element, without changing the iterator
  528      fn peek_next<'a>(&'a mut self) -> Option<&'a mut A>;
  529  }

  531  // private methods for MutItems
  532  impl<'a, A> MutItems<'a, A> {
  533      fn insert_next_node(&mut self, mut ins_node: ~Node<A>) {
  534          // Insert before `self.head` so that it is between the
  535          // previously yielded element and self.head.
  536          //
  537          // The inserted node will not appear in further iteration.
  538          match self.head.resolve() {
  539              None => { self.list.push_back_node(ins_node); }
  540              Some(node) => {
  541                  let prev_node = match node.prev.resolve() {
  542                      None => return self.list.push_front_node(ins_node),
  543                      Some(prev) => prev,
  544                  };
  545                  let node_own = prev_node.next.take_unwrap();
  546                  ins_node.next = link_with_prev(node_own, Rawlink::some(ins_node));
  547                  prev_node.next = link_with_prev(ins_node, Rawlink::some(prev_node));
  548                  self.list.length += 1;
  549              }
  550          }
  551      }
  552  }

  639      pub fn check_links<T>(list: &DList<T>) {
  640          let mut len = 0u;
  641          let mut last_ptr: Option<&Node<T>> = None;
  642          let mut node_ptr: &Node<T>;
  643          match list.list_head {
  644              None => { assert_eq!(0u, list.length); return }
  645              Some(ref node) => node_ptr = &**node,
  646          }
  647          loop {
  648              match (last_ptr, node_ptr.prev.resolve_immut()) {
  649                  (None   , None      ) => {}
  650                  (None   , _         ) => fail!("prev link for list_head"),
  651                  (Some(p), Some(pptr)) => {
  652                      assert_eq!(p as *Node<T>, pptr as *Node<T>);
  653                  }
  654                  _ => fail!("prev link is none, not good"),
  655              }
  656              match node_ptr.next {
  657                  Some(ref next) => {
  658                      last_ptr = Some(node_ptr);
  659                      node_ptr = &**next;
  660                      len += 1;
  661                  }
  662                  None => {
  663                      len += 1;
  664                      break;
  665                  }
  666              }
  667          }
  668          assert_eq!(len, list.length);
  669      }

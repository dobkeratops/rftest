  340                              };

  368              ///   - find required node using `traverse_left`/`traverse_right`
  369              ///     (current node is `Entries::node` field)
  370              ///   - complete initialization with `traverse_complete`

  373              /// `self.node` is None iteration will start from last

  375              #[inline]

  844                            key: &K) -> Option<V> {
  845      fn heir_swap<K: TotalOrd, V>(node: &mut ~TreeNode<K, V>,
  846                                   child: &mut Option<~TreeNode<K, V>>) {
  847          // *could* be done without recursion, but it won't borrow check
  848          for x in child.mut_iter() {
  849              if x.right.is_some() {
  850                  heir_swap(node, &mut x.right);
  851              } else {
  852                  swap(&mut node.key, &mut x.key);
  853                  swap(&mut node.value, &mut x.value);
  854              }
  855          }
  856      }
  857  
  858      match *node {
  859        None => {
  860          return None; // bottom of tree
  861        }
  862        Some(ref mut save) => {
  863          let (ret, rebalance) = match key.cmp(&save.key) {
  864            Less => (remove(&mut save.left, key), true),
  865            Greater => (remove(&mut save.right, key), true),
  866            Equal => {
  867              if save.left.is_some() {
  868                  if save.right.is_some() {
  869                      let mut left = save.left.take_unwrap();
  870                      if left.right.is_some() {
  871                          heir_swap(save, &mut left.right);
  872                      } else {
  873                          swap(&mut save.key, &mut left.key);
  874                          swap(&mut save.value, &mut left.value);
  875                      }
  876                      save.left = Some(left);
  877                      (remove(&mut save.left, key), true)
  878                  } else {
  879                      let new = save.left.take_unwrap();
  880                      let ~TreeNode{value, ..} = replace(save, new);
  881                      *save = save.left.take_unwrap();
  882                      (Some(value), true)
  883                  }
  884              } else if save.right.is_some() {
  885                  let new = save.right.take_unwrap();
  886                  let ~TreeNode{value, ..} = replace(save, new);
  887                  (Some(value), true)
  888              } else {
  889                  (None, false)
  890              }
  891            }
  892          };
  893  
  894          if rebalance {
  895              let left_level = save.left.as_ref().map_or(0, |x| x.level);
  896              let right_level = save.right.as_ref().map_or(0, |x| x.level);
  897  
  898              // re-balance, if necessary
  899              if left_level < save.level - 1 || right_level < save.level - 1 {
  900                  save.level -= 1;
  901  
  902                  if right_level > save.level {
  903                      for x in save.right.mut_iter() { x.level = save.level }
  904                  }
  905  
  906                  skew(save);
  907  
  908                  for right in save.right.mut_iter() {
  909                      skew(right);
  910                      for x in right.right.mut_iter() { skew(x) }
  911                  }
  912  
  913                  split(save);
  914                  for x in save.right.mut_iter() { split(x) }
  915              }
  916  
  917              return ret;
  918          }
  919        }
  920      }
  921      return match node.take() {
  922          Some(~TreeNode{value, ..}) => Some(value), None => fail!()
  923      };
  924  }

   50  // Lexicographical comparison
   51  fn lt<K: Ord + TotalOrd, V: Ord>(a: &TreeMap<K, V>,
   52                                   b: &TreeMap<K, V>) -> bool {
   53      // the Zip iterator is as long as the shortest of a and b.
   54      for ((key_a, value_a), (key_b, value_b)) in a.iter().zip(b.iter()) {
   55          if *key_a < *key_b { return true; }
   56          if *key_a > *key_b { return false; }
   57          if *value_a < *value_b { return true; }
   58          if *value_a > *value_b { return false; }
   59      }
   60  
   61      a.len() < b.len()
   62  }

   91  impl<K: TotalOrd, V> Map<K, V> for TreeMap<K, V> {
   92      /// Return a reference to the value corresponding to the key
   93      fn find<'a>(&'a self, key: &K) -> Option<&'a V> {
   94          let mut current: &'a Option<~TreeNode<K, V>> = &self.root;
   95          loop {
   96              match *current {
   97                Some(ref r) => {
   98                  match key.cmp(&r.key) {
   99                    Less => current = &r.left,
  100                    Greater => current = &r.right,
  101                    Equal => return Some(&r.value)
  102                  }
  103                }
  104                None => return None
  105              }
  106          }
  107      }

  117      /// Insert a key-value pair from the map. If the key already had a value

  123      }
  124

  188  macro_rules! bound_setup {
  189      // initialiser of the iterator to manipulate
  190      ($iter:expr,
  191       // whether we are looking for the lower or upper bound.
  192       $is_lower_bound:expr) => {
  193          {
  194              let mut iter = $iter;
  195              loop {
  196                  if !iter.node.is_null() {
  197                      let node_k = unsafe {&(*iter.node).key};
  198                      match k.cmp(node_k) {
  199                          Less => iter.traverse_left(),
  200                          Greater => iter.traverse_right(),
  201                          Equal => {
  202                              if $is_lower_bound {
  203                                  iter.traverse_complete();
  204                                  return iter;
  205                              } else {
  206                                  iter.traverse_right()
  207                              }
  208                          }
  209                      }
  210                  } else {
  211                      iter.traverse_complete();
  212                      return iter;
  213                  }
  214              }
  215          }
  216      }

  324  // FIXME #5846 we want to be able to choose between &x and &mut x
  325  // (with many different `x`) below, so we need to optionally pass mut
  326  // as a tt, but the only thing we can do with a `tt` is pass them to
  327  // other macros, so this takes the `& <mutability> <operand>` token
  328  // sequence and forces their evalutation as an expression.
  329  macro_rules! addr { ($e:expr) => { $e }}
  330  // putting an optional mut into type signatures
  331  macro_rules! item { ($i:item) => { $i }}

  333  macro_rules! define_iterator {
  334      ($name:ident,
  335       $rev_name:ident,
  336  
  337       // the function to go from &m Option<~TreeNode> to *m TreeNode
  338       deref = $deref:ident,
  339  
  340       // see comment on `addr!`, this is just an optional `mut`, but
  341       // there's no support for 0-or-1 repeats.
  342       addr_mut = $($addr_mut:tt)*
  343       ) => {
  344          // private methods on the forward iterator (item!() for the
  345          // addr_mut in the next_ return value)
  346          item!(impl<'a, K, V> $name<'a, K, V> {
  347              #[inline(always)]
  348              fn next_(&mut self, forward: bool) -> Option<(&'a K, &'a $($addr_mut)* V)> {
  349                  while !self.stack.is_empty() || !self.node.is_null() {
  350                      if !self.node.is_null() {
  351                          let node = unsafe {addr!(& $($addr_mut)* *self.node)};
  352                          {
  353                              let next_node = if forward {
  354                                  addr!(& $($addr_mut)* node.left)
  355                              } else {
  356                                  addr!(& $($addr_mut)* node.right)
  357                              };
  358                              self.node = $deref(next_node);
  359                          }
  360                          self.stack.push(node);
  361                      } else {
  362                          let node = self.stack.pop().unwrap();
  363                          let next_node = if forward {
  364                              addr!(& $($addr_mut)* node.right)
  365                          } else {
  366                              addr!(& $($addr_mut)* node.left)
  367                          };
  368                          self.node = $deref(next_node);
  369                          self.remaining_max -= 1;
  370                          if self.remaining_min > 0 {
  371                              self.remaining_min -= 1;
  372                          }
  373                          return Some((&node.key, addr!(& $($addr_mut)* node.value)));
  374                      }
  375                  }
  376                  None
  377              }

  379              /// traverse_left, traverse_right and traverse_complete are
  380              /// used to initialize Entries/MutEntries
  381              /// pointing to element inside tree structure.
  382              ///
  383              /// They should be used in following manner:
  384              ///   - create iterator using TreeMap::[mut_]iter_for_traversal
  385              ///   - find required node using `traverse_left`/`traverse_right`
  386              ///     (current node is `Entries::node` field)
  387              ///   - complete initialization with `traverse_complete`

  822  // Remove left horizontal link by rotating right
  823  fn skew<K: TotalOrd, V>(node: &mut ~TreeNode<K, V>) {
  824      if node.left.as_ref().map_or(false, |x| x.level == node.level) {
  825          let mut save = node.left.take_unwrap();
  826          swap(&mut node.left, &mut save.right); // save.right now None
  827          swap(node, &mut save);
  828          node.right = Some(save);
  829      }
  830  }

  832  // Remove dual horizontal link by rotating left and increasing level of
  833  // the parent
  834  fn split<K: TotalOrd, V>(node: &mut ~TreeNode<K, V>) {
  835      if node.right.as_ref().map_or(false,
  836        |x| x.right.as_ref().map_or(false, |y| y.level == node.level)) {
  837          let mut save = node.right.take_unwrap();
  838          swap(&mut node.right, &mut save.left); // save.left now None
  839          save.level += 1;
  840          swap(node, &mut save);
  841          node.left = Some(save);
  842      }
  843  }

  891                            key: &K) -> Option<V> {
  892      fn heir_swap<K: TotalOrd, V>(node: &mut ~TreeNode<K, V>,
  893                                   child: &mut Option<~TreeNode<K, V>>) {
  894          // *could* be done without recursion, but it won't borrow check
  895          for x in child.mut_iter() {
  896              if x.right.is_some() {
  897                  heir_swap(node, &mut x.right);
  898              } else {
  899                  swap(&mut node.key, &mut x.key);
  900                  swap(&mut node.value, &mut x.value);
  901              }
  902          }
  903      }
  904  
  905      match *node {
  906        None => {
  907          return None; // bottom of tree
  908        }
  909        Some(ref mut save) => {
  910          let (ret, rebalance) = match key.cmp(&save.key) {
  911            Less => (remove(&mut save.left, key), true),
  912            Greater => (remove(&mut save.right, key), true),
  913            Equal => {
  914              if save.left.is_some() {
  915                  if save.right.is_some() {
  916                      let mut left = save.left.take_unwrap();
  917                      if left.right.is_some() {
  918                          heir_swap(save, &mut left.right);
  919                      } else {
  920                          swap(&mut save.key, &mut left.key);
  921                          swap(&mut save.value, &mut left.value);
  922                      }
  923                      save.left = Some(left);
  924                      (remove(&mut save.left, key), true)
  925                  } else {
  926                      let new = save.left.take_unwrap();
  927                      let ~TreeNode{value, ..} = replace(save, new);
  928                      *save = save.left.take_unwrap();
  929                      (Some(value), true)
  930                  }
  931              } else if save.right.is_some() {
  932                  let new = save.right.take_unwrap();
  933                  let ~TreeNode{value, ..} = replace(save, new);
  934                  (Some(value), true)
  935              } else {
  936                  (None, false)
  937              }
  938            }
  939          };
  940  
  941          if rebalance {
  942              let left_level = save.left.as_ref().map_or(0, |x| x.level);
  943              let right_level = save.right.as_ref().map_or(0, |x| x.level);
  944  
  945              // re-balance, if necessary
  946              if left_level < save.level - 1 || right_level < save.level - 1 {
  947                  save.level -= 1;
  948  
  949                  if right_level > save.level {
  950                      for x in save.right.mut_iter() { x.level = save.level }
  951                  }
  952  
  953                  skew(save);
  954  
  955                  for right in save.right.mut_iter() {
  956                      skew(right);
  957                      for x in right.right.mut_iter() { skew(x) }
  958                  }
  959  
  960                  split(save);
  961                  for x in save.right.mut_iter() { split(x) }
  962              }
  963  
  964              return ret;
  965          }
  966        }
  967      }
  968      return match node.take() {
  969          Some(~TreeNode{value, ..}) => Some(value), None => fail!()
  970      };
  971  }

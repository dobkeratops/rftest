  44  impl<V> Map<uint, V> for SmallIntMap<V> {
  45      /// Return a reference to the value corresponding to the key
  46      fn find<'a>(&'a self, key: &uint) -> Option<&'a V> {
  47          if *key < self.v.len() {
  48              match *self.v.get(*key) {
  49                Some(ref value) => Some(value),
  50                None => None
  51              }
  52          } else {
  53              None
  54          }
  55      }
  56  }

  58  impl<V> MutableMap<uint, V> for SmallIntMap<V> {
  59      /// Return a mutable reference to the value corresponding to the key
  60      fn find_mut<'a>(&'a mut self, key: &uint) -> Option<&'a mut V> {
  61          if *key < self.v.len() {
  62              match *self.v.get_mut(*key) {
  63                Some(ref mut value) => Some(value),
  64                None => None
  65              }
  66          } else {
  67              None
  68          }
  69      }
  70  
  71      /// Insert a key-value pair into the map. An existing value for a
  72      /// key is replaced by the new value. Return true if the key did
  73      /// not already exist in the map.
  74      fn insert(&mut self, key: uint, value: V) -> bool {
  75          let exists = self.contains_key(&key);
  76          let len = self.v.len();
  77          if len <= key {
  78              self.v.grow_fn(key - len + 1, |_| None);
  79          }
  80          *self.v.get_mut(key) = Some(value);
  81          !exists
  82      }
  83  
  84      /// Remove a key-value pair from the map. Return true if the key
  85      /// was present in the map, otherwise false.
  86      fn remove(&mut self, key: &uint) -> bool {
  87          self.pop(key).is_some()
  88      }
  89  
  90      /// Insert a key-value pair from the map. If the key already had a value
  91      /// present in the map, that value is returned. Otherwise None is returned.
  92      fn swap(&mut self, key: uint, value: V) -> Option<V> {
  93          match self.find_mut(&key) {
  94              Some(loc) => { return Some(replace(loc, value)); }
  95              None => ()
  96          }
  97          self.insert(key, value);
  98          return None;
  99      }
 100  
 101      /// Removes a key from the map, returning the value at the key if the key
 102      /// was previously in the map.
 103      fn pop(&mut self, key: &uint) -> Option<V> {
 104          if *key >= self.v.len() {
 105              return None;
 106          }
 107          self.v.get_mut(*key).take()
 108      }
 109  }

 120      pub fn get<'a>(&'a self, key: &uint) -> &'a V {
 121          self.find(key).expect("key not present")
 122      }

 124      /// An iterator visiting all key-value pairs in ascending order by the keys.
 125      /// Iterator element type is (uint, &'r V)
 126      pub fn iter<'r>(&'r self) -> Entries<'r, V> {
 127          Entries {
 128              front: 0,
 129              back: self.v.len(),
 130              iter: self.v.iter()
 131          }
 132      }
 133  
 134      /// An iterator visiting all key-value pairs in ascending order by the keys,
 135      /// with mutable references to the values
 136      /// Iterator element type is (uint, &'r mut V)
 137      pub fn mut_iter<'r>(&'r mut self) -> MutEntries<'r, V> {
 138          MutEntries {
 139              front: 0,
 140              back: self.v.len(),
 141              iter: self.v.mut_iter()
 142          }
 143      }
 144  
 145      /// An iterator visiting all key-value pairs in descending order by the keys.
 146      /// Iterator element type is (uint, &'r V)
 147      pub fn rev_iter<'r>(&'r self) -> RevEntries<'r, V> {
 148          self.iter().rev()
 149      }
 150  
 151      /// An iterator visiting all key-value pairs in descending order by the keys,
 152      /// with mutable references to the values
 153      /// Iterator element type is (uint, &'r mut V)
 154      pub fn mut_rev_iter<'r>(&'r mut self) -> RevMutEntries <'r, V> {
 155          self.mut_iter().rev()
 156      }

 158      /// Empties the hash map, moving all values into the specified closure
 159      pub fn move_iter(&mut self)
 160          -> FilterMap<(uint, Option<V>), (uint, V),
 161                  Enumerate<vec::MoveItems<Option<V>>>>
 162      {
 163          let values = replace(&mut self.v, vec!());
 164          values.move_iter().enumerate().filter_map(|(i, v)| {
 165              v.map(|v| (i, v))
 166          })
 167      }
 168  }

 170  impl<V:Clone> SmallIntMap<V> {
 171      pub fn update_with_key(&mut self,
 172                             key: uint,
 173                             val: V,
 174                             ff: |uint, V, V| -> V)
 175                             -> bool {
 176          let new_val = match self.find(&key) {
 177              None => val,
 178              Some(orig) => ff(key, (*orig).clone(), val)
 179          };
 180          self.insert(key, new_val)
 181      }
 182  
 183      pub fn update(&mut self, key: uint, newval: V, ff: |V, V| -> V) -> bool {
 184          self.update_with_key(key, newval, |_k, v, v1| ff(v,v1))
 185      }
 186  }

 189  macro_rules! iterator {
 190      (impl $name:ident -> $elem:ty, $getter:ident) => {
 191          impl<'a, T> Iterator<$elem> for $name<'a, T> {
 192              #[inline]
 193              fn next(&mut self) -> Option<$elem> {
 194                  while self.front < self.back {
 195                      match self.iter.next() {
 196                          Some(elem) => {
 197                              if elem.is_some() {
 198                                  let index = self.front;
 199                                  self.front += 1;
 200                                  return Some((index, elem. $getter ()));
 201                              }
 202                          }
 203                          _ => ()
 204                      }
 205                      self.front += 1;
 206                  }
 207                  None
 208              }
 209  
 210              #[inline]
 211              fn size_hint(&self) -> (uint, Option<uint>) {
 212                  (0, Some(self.back - self.front))
 213              }
 214          }
 215      }
 216  }

 218  macro_rules! double_ended_iterator {
 219      (impl $name:ident -> $elem:ty, $getter:ident) => {
 220          impl<'a, T> DoubleEndedIterator<$elem> for $name<'a, T> {
 221              #[inline]
 222              fn next_back(&mut self) -> Option<$elem> {
 223                  while self.front < self.back {
 224                      match self.iter.next_back() {
 225                          Some(elem) => {
 226                              if elem.is_some() {
 227                                  self.back -= 1;
 228                                  return Some((self.back, elem. $getter ()));
 229                              }
 230                          }
 231                          _ => ()
 232                      }
 233                      self.back -= 1;
 234                  }
 235                  None
 236              }
 237          }
 238      }
 239  }

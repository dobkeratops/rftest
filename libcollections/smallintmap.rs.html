  27  impl<V> Container for SmallIntMap<V> {
  28      /// Return the number of elements in the map
  29      fn len(&self) -> uint {
  30          self.v.iter().count(|elt| elt.is_some())
  31      }
  32  
  33      /// Return true if there are no elements in the map
  34      fn is_empty(&self) -> bool {
  35          self.v.iter().all(|elt| elt.is_none())
  36      }
  37  }

  44  impl<V> Map<uint, V> for SmallIntMap<V> {
  45      /// Return a reference to the value corresponding to the key
  46      fn find<'a>(&'a self, key: &uint) -> Option<&'a V> {
  47          if *key < self.v.len() {
  48              match self.v[*key] {
  49                Some(ref value) => Some(value),
  50                None => None
  51              }
  52          } else {
  53              None
  54          }
  55      }
  56  }

  58  impl<V> MutableMap<uint, V> for SmallIntMap<V> {
  59      /// Return a mutable reference to the value corresponding to the key
  60      fn find_mut<'a>(&'a mut self, key: &uint) -> Option<&'a mut V> {
  61          if *key < self.v.len() {
  62              match self.v[*key] {
  63                Some(ref mut value) => Some(value),
  64                None => None
  65              }
  66          } else {
  67              None
  68          }
  69      }
  70  
  71      /// Insert a key-value pair into the map. An existing value for a
  72      /// key is replaced by the new value. Return true if the key did
  73      /// not already exist in the map.
  74      fn insert(&mut self, key: uint, value: V) -> bool {
  75          let exists = self.contains_key(&key);
  76          let len = self.v.len();
  77          if len <= key {
  78              self.v.grow_fn(key - len + 1, |_| None);
  79          }
  80          self.v[key] = Some(value);
  81          !exists
  82      }
  83  
  84      /// Remove a key-value pair from the map. Return true if the key
  85      /// was present in the map, otherwise false.
  86      fn remove(&mut self, key: &uint) -> bool {
  87          self.pop(key).is_some()
  88      }
  89  
  90      /// Insert a key-value pair from the map. If the key already had a value
  91      /// present in the map, that value is returned. Otherwise None is returned.
  92      fn swap(&mut self, key: uint, value: V) -> Option<V> {
  93          match self.find_mut(&key) {
  94              Some(loc) => { return Some(replace(loc, value)); }
  95              None => ()
  96          }
  97          self.insert(key, value);
  98          return None;
  99      }
 100  
 101      /// Removes a key from the map, returning the value at the key if the key
 102      /// was previously in the map.
 103      fn pop(&mut self, key: &uint) -> Option<V> {
 104          if *key >= self.v.len() {
 105              return None;
 106          }
 107          self.v[*key].take()
 108      }
 109  }

 111  impl<V> SmallIntMap<V> {
 112      /// Create an empty SmallIntMap
 113      pub fn new() -> SmallIntMap<V> { SmallIntMap{v: ~[]} }
 114  
 115      pub fn get<'a>(&'a self, key: &uint) -> &'a V {
 116          self.find(key).expect("key not present")
 117      }
 118  
 119      /// An iterator visiting all key-value pairs in ascending order by the keys.
 120      /// Iterator element type is (uint, &'r V)
 121      pub fn iter<'r>(&'r self) -> Entries<'r, V> {
 122          Entries {
 123              front: 0,
 124              back: self.v.len(),
 125              iter: self.v.iter()
 126          }
 127      }
 128  
 129      /// An iterator visiting all key-value pairs in ascending order by the keys,
 130      /// with mutable references to the values
 131      /// Iterator element type is (uint, &'r mut V)
 132      pub fn mut_iter<'r>(&'r mut self) -> MutEntries<'r, V> {
 133          MutEntries {
 134              front: 0,
 135              back: self.v.len(),
 136              iter: self.v.mut_iter()
 137          }
 138      }
 139  
 140      /// An iterator visiting all key-value pairs in descending order by the keys.
 141      /// Iterator element type is (uint, &'r V)
 142      pub fn rev_iter<'r>(&'r self) -> RevEntries<'r, V> {
 143          self.iter().rev()
 144      }
 145  
 146      /// An iterator visiting all key-value pairs in descending order by the keys,
 147      /// with mutable references to the values
 148      /// Iterator element type is (uint, &'r mut V)
 149      pub fn mut_rev_iter<'r>(&'r mut self) -> RevMutEntries <'r, V> {
 150          self.mut_iter().rev()
 151      }
 152  
 153      /// Empties the hash map, moving all values into the specified closure
 154      pub fn move_iter(&mut self)
 155          -> FilterMap<(uint, Option<V>), (uint, V),
 156                  Enumerate<slice::MoveItems<Option<V>>>>
 157      {
 158          let values = replace(&mut self.v, ~[]);
 159          values.move_iter().enumerate().filter_map(|(i, v)| {
 160              v.map(|v| (i, v))
 161          })
 162      }
 163  }

 165  impl<V:Clone> SmallIntMap<V> {
 166      pub fn update_with_key(&mut self,
 167                             key: uint,
 168                             val: V,
 169                             ff: |uint, V, V| -> V)
 170                             -> bool {
 171          let new_val = match self.find(&key) {
 172              None => val,
 173              Some(orig) => ff(key, (*orig).clone(), val)
 174          };
 175          self.insert(key, new_val)
 176      }
 177  
 178      pub fn update(&mut self, key: uint, newval: V, ff: |V, V| -> V) -> bool {
 179          self.update_with_key(key, newval, |_k, v, v1| ff(v,v1))
 180      }
 181  }

 184  macro_rules! iterator {
 185      (impl $name:ident -> $elem:ty, $getter:ident) => {
 186          impl<'a, T> Iterator<$elem> for $name<'a, T> {
 187              #[inline]
 188              fn next(&mut self) -> Option<$elem> {
 189                  while self.front < self.back {
 190                      match self.iter.next() {
 191                          Some(elem) => {
 192                              if elem.is_some() {
 193                                  let index = self.front;
 194                                  self.front += 1;
 195                                  return Some((index, elem. $getter ()));
 196                              }
 197                          }
 198                          _ => ()
 199                      }
 200                      self.front += 1;
 201                  }
 202                  None
 203              }
 204  
 205              #[inline]
 206              fn size_hint(&self) -> (uint, Option<uint>) {
 207                  (0, Some(self.back - self.front))
 208              }
 209          }
 210      }
 211  }

 213  macro_rules! double_ended_iterator {
 214      (impl $name:ident -> $elem:ty, $getter:ident) => {
 215          impl<'a, T> DoubleEndedIterator<$elem> for $name<'a, T> {
 216              #[inline]
 217              fn next_back(&mut self) -> Option<$elem> {
 218                  while self.front < self.back {
 219                      match self.iter.next_back() {
 220                          Some(elem) => {
 221                              if elem.is_some() {
 222                                  self.back -= 1;
 223                                  return Some((self.back, elem. $getter ()));
 224                              }
 225                          }
 226                          _ => ()
 227                      }
 228                      self.back -= 1;
 229                  }
 230                  None
 231              }
 232          }
 233      }
 234  }

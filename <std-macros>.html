179     //
180     // A scheme-style conditional that helps to improve code clarity in some instances when
181     // the `if`, `else if`, and `else` keywords obscure predicates undesirably.
182     //
183     // # Example
184     //
185     // ~~~
186     // let clamped =
187     //     if x > mx { mx }
188     //     else if x < mn { mn }
189     //     else { x };
190     // ~~~
191     //
192     // Using `cond!`, the above could be written as:
193     //
194     // ~~~
195     // let clamped = cond!(
196     //     (x > mx) { mx }
197     //     (x < mn) { mn }
198     //     _        { x  }
199     // );
200     // ~~~
201     //
202     // The optional default case is denoted by `_`.
203     //
204     macro_rules! cond (
205         ( $(($pred:expr) $body:block)+ _ $default:block ) => (
206             $(if $pred $body else)+
207             $default
208         );
209         // for if the default case was ommitted
210         ( $(($pred:expr) $body:block)+ ) => (
211             $(if $pred $body)else+
212         );
213     )

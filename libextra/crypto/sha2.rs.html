   18  // Sha-512 and Sha-256 use basically the same calculations which are implemented by these macros.
   19  // Inlining the calculations seems to result in better generated code.
   20  macro_rules! schedule_round( ($t:expr) => (
   21          W[$t] = sigma1(W[$t - 2]) + W[$t - 7] + sigma0(W[$t - 15]) + W[$t - 16];
   22      )
   23  )
   24  
   25  macro_rules! sha2_round(
   26      ($A:ident, $B:ident, $C:ident, $D:ident,
   27       $E:ident, $F:ident, $G:ident, $H:ident, $K:ident, $t:expr) => (
   28          {
   29              $H += sum1($E) + ch($E, $F, $G) + $K[$t] + W[$t];
   30              $D += $H;
   31              $H += sum0($A) + maj($A, $B, $C);
   32          }
   33      )
   34  )

   64      fn reset(&mut self, h: &[u64, ..8]) {
   65          self.H0 = h[0];
   66          self.H1 = h[1];
   67          self.H2 = h[2];
   68          self.H3 = h[3];
   69          self.H4 = h[4];
   70          self.H5 = h[5];
   71          self.H6 = h[6];
   72          self.H7 = h[7];
   73      }
   74  
   75      fn process_block(&mut self, data: &[u8]) {
   76          fn ch(x: u64, y: u64, z: u64) -> u64 {
   77              ((x & y) ^ ((!x) & z))
   78          }
   79  
   80          fn maj(x: u64, y: u64, z: u64) -> u64 {
   81              ((x & y) ^ (x & z) ^ (y & z))
   82          }
   83  
   84          fn sum0(x: u64) -> u64 {
   85              ((x << 36) | (x >> 28)) ^ ((x << 30) | (x >> 34)) ^ ((x << 25) | (x >> 39))
   86          }
   87  
   88          fn sum1(x: u64) -> u64 {
   89              ((x << 50) | (x >> 14)) ^ ((x << 46) | (x >> 18)) ^ ((x << 23) | (x >> 41))
   90          }
   91  
   92          fn sigma0(x: u64) -> u64 {
   93              ((x << 63) | (x >> 1)) ^ ((x << 56) | (x >> 8)) ^ (x >> 7)
   94          }
   95  
   96          fn sigma1(x: u64) -> u64 {
   97              ((x << 45) | (x >> 19)) ^ ((x << 3) | (x >> 61)) ^ (x >> 6)
   98          }
   99  
  100          let mut a = self.H0;
  101          let mut b = self.H1;
  102          let mut c = self.H2;
  103          let mut d = self.H3;
  104          let mut e = self.H4;
  105          let mut f = self.H5;
  106          let mut g = self.H6;
  107          let mut h = self.H7;
  108  
  109          let mut W = [0u64, ..80];
  110  
  111          read_u64v_be(W.mut_slice(0, 16), data);
  112  
  113          // Putting the message schedule inside the same loop as the round calculations allows for
  114          // the compiler to generate better code.
  115          do uint::range_step(0, 64, 8) |t| {
  116              schedule_round!(t + 16);
  117              schedule_round!(t + 17);
  118              schedule_round!(t + 18);
  119              schedule_round!(t + 19);
  120              schedule_round!(t + 20);
  121              schedule_round!(t + 21);
  122              schedule_round!(t + 22);
  123              schedule_round!(t + 23);
  124  
  125              sha2_round!(a, b, c, d, e, f, g, h, K64, t);
  126              sha2_round!(h, a, b, c, d, e, f, g, K64, t + 1);
  127              sha2_round!(g, h, a, b, c, d, e, f, K64, t + 2);
  128              sha2_round!(f, g, h, a, b, c, d, e, K64, t + 3);
  129              sha2_round!(e, f, g, h, a, b, c, d, K64, t + 4);
  130              sha2_round!(d, e, f, g, h, a, b, c, K64, t + 5);
  131              sha2_round!(c, d, e, f, g, h, a, b, K64, t + 6);
  132              sha2_round!(b, c, d, e, f, g, h, a, K64, t + 7);
  133              true
  134          };
  135  
  136          do uint::range_step(64, 80, 8) |t| {
  137              sha2_round!(a, b, c, d, e, f, g, h, K64, t);
  138              sha2_round!(h, a, b, c, d, e, f, g, K64, t + 1);
  139              sha2_round!(g, h, a, b, c, d, e, f, K64, t + 2);
  140              sha2_round!(f, g, h, a, b, c, d, e, K64, t + 3);
  141              sha2_round!(e, f, g, h, a, b, c, d, K64, t + 4);
  142              sha2_round!(d, e, f, g, h, a, b, c, K64, t + 5);
  143              sha2_round!(c, d, e, f, g, h, a, b, K64, t + 6);
  144              sha2_round!(b, c, d, e, f, g, h, a, K64, t + 7);
  145              true
  146          };
  147  
  148          self.H0 += a;
  149          self.H1 += b;
  150          self.H2 += c;
  151          self.H3 += d;
  152          self.H4 += e;
  153          self.H5 += f;

  184  // A structure that keeps track of the state of the Sha-512 operation and contains the logic
  185  // necessary to perform the final calculations.
  186  struct Engine512 {
  187      length_bits: (u64, u64),
  188      buffer: FixedBuffer128,
  189      state: Engine512State,
  190      finished: bool,
  191  }
  192  
  193  impl Engine512 {
  194      fn new(h: &[u64, ..8]) -> Engine512 {
  195          return Engine512 {
  196              length_bits: (0, 0),
  197              buffer: FixedBuffer128::new(),
  198              state: Engine512State::new(h),
  199              finished: false
  200          }
  201      }
  202  
  203      fn reset(&mut self, h: &[u64, ..8]) {
  204          self.length_bits = (0, 0);
  205          self.buffer.reset();
  206          self.state.reset(h);
  207          self.finished = false;
  208      }
  209  
  210      fn input(&mut self, input: &[u8]) {
  211          assert!(!self.finished)
  212          // Assumes that input.len() can be converted to u64 without overflow
  213          self.length_bits = shift_add_check_overflow_tuple(self.length_bits, input.len() as u64, 3);
  214          self.buffer.input(input, |input: &[u8]| { self.state.process_block(input) });
  215      }
  216  
  217      fn finish(&mut self) {
  218          if self.finished {
  219              return;
  220          }
  221  
  222          self.buffer.standard_padding(16, |input: &[u8]| { self.state.process_block(input) });
  223          match self.length_bits {
  224              (hi, low) => {
  225                  write_u64_be(self.buffer.next(8), hi);
  226                  write_u64_be(self.buffer.next(8), low);
  227              }
  228          }
  229          self.state.process_block(self.buffer.full_buffer());
  230  
  231          self.finished = true;
  232      }
  233  }

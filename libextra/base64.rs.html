  53 pub trait ToBase64 {

  59 impl<'self> ToBase64 for &'self [u8] {
  60     /**
  61      * Turn a vector of `u8` bytes into a base64 string.
  62      *
  63      * # Example
  64      *
  65      * ~~~ {.rust}
  66      * extern mod extra;
  67      * use extra::base64::{ToBase64, standard};
  68      *
  69      * fn main () {
  70      *     let str = [52,32].to_base64(standard);
  71      *     printfln!("%s", str);
  72      * }
  73      * ~~~
  74      */
  75     fn to_base64(&self, config: Config) -> ~str {
  76         let bytes = match config.char_set {
  77             Standard => STANDARD_CHARS,
  78             UrlSafe => URLSAFE_CHARS
  79         };
  80 
  81         let mut v: ~[u8] = ~[];
  82         let mut i = 0;
  83         let mut cur_length = 0;
  84         let len = self.len();
  85         while i < len - (len % 3) {
  86             match config.line_length {
  87                 Some(line_length) =>
  88                     if cur_length >= line_length {
  89                         v.push('\r' as u8);
  90                         v.push('\n' as u8);
  91                         cur_length = 0;
  92                     },
  93                 None => ()
  94             }
  95 
  96             let n = (self[i] as u32) << 16 |
  97                     (self[i + 1] as u32) << 8 |
  98                     (self[i + 2] as u32);
  99 
 100             // This 24-bit number gets separated into four 6-bit numbers.
 101             v.push(bytes[(n >> 18) & 63]);
 102             v.push(bytes[(n >> 12) & 63]);
 103             v.push(bytes[(n >> 6 ) & 63]);
 104             v.push(bytes[n & 63]);
 105 
 106             cur_length += 4;
 107             i += 3;
 108         }
 109 
 110         if len % 3 != 0 {
 111             match config.line_length {
 112                 Some(line_length) =>
 113                     if cur_length >= line_length {
 114                         v.push('\r' as u8);
 115                         v.push('\n' as u8);
 116                     },
 117                 None => ()
 118             }
 119         }
 120 
 121         // Heh, would be cool if we knew this was exhaustive
 122         // (the dream of bounded integer types)
 123         match len % 3 {
 124             0 => (),
 125             1 => {
 126                 let n = (self[i] as u32) << 16;
 127                 v.push(bytes[(n >> 18) & 63]);
 128                 v.push(bytes[(n >> 12) & 63]);
 129                 if config.pad {
 130                     v.push('=' as u8);
 131                     v.push('=' as u8);
 132                 }
 133             }
 134             2 => {
 135                 let n = (self[i] as u32) << 16 |
 136                     (self[i + 1u] as u32) << 8;
 137                 v.push(bytes[(n >> 18) & 63]);
 138                 v.push(bytes[(n >> 12) & 63]);
 139                 v.push(bytes[(n >> 6 ) & 63]);
 140                 if config.pad {
 141                     v.push('=' as u8);
 142                 }
 143             }
 144             _ => fail!("Algebra is broken, please alert the math police")
 145         }
 146 
 147         unsafe {
 148             str::raw::from_bytes_owned(v)
 149         }
 150     }
 151 }

 153 /// A trait for converting from base64 encoded values.
 154 pub trait FromBase64 {
 155     /// Converts the value of `self`, interpreted as base64 encoded data, into
 156     /// an owned vector of bytes, returning the vector.
 157     fn from_base64(&self) -> Result<~[u8], ~str>;
 158 }
 159 
 160 impl<'self> FromBase64 for &'self str {
 161     /**
 162      * Convert any base64 encoded string (literal, `@`, `&`, or `~`)
 163      * to the byte values it encodes.
 164      *
 165      * You can use the `from_bytes` function in `std::str`
 166      * to turn a `[u8]` into a string with characters corresponding to those
 167      * values.
 168      *
 169      * # Example
 170      *
 171      * This converts a string literal to base64 and back.
 172      *
 173      * ~~~ {.rust}
 174      * extern mod extra;
 175      * use extra::base64::{ToBase64, FromBase64, standard};
 176      * use std::str;
 177      *
 178      * fn main () {
 179      *     let hello_str = "Hello, World".to_base64(standard);
 180      *     printfln!("%s", hello_str);
 181      *     let bytes = hello_str.from_base64();
 182      *     printfln!("%?", bytes);
 183      *     let result_str = str::from_bytes(bytes);
 184      *     printfln!("%s", result_str);
 185      * }
 186      * ~~~
 187      */
 188     fn from_base64(&self) -> Result<~[u8], ~str> {
 189         let mut r = ~[];
 190         let mut buf: u32 = 0;
 191         let mut modulus = 0;
 192 
 193         let mut it = self.byte_iter().enumerate();
 194         for (idx, byte) in it {
 195             let val = byte as u32;
 196 
 197             match byte as char {
 198                 'A'..'Z' => buf |= val - 0x41,
 199                 'a'..'z' => buf |= val - 0x47,
 200                 '0'..'9' => buf |= val + 0x04,
 201                 '+'|'-' => buf |= 0x3E,
 202                 '/'|'_' => buf |= 0x3F,
 203                 '\r'|'\n' => loop,
 204                 '=' => break,
 205                 _ => return Err(fmt!("Invalid character '%c' at position %u",
 206                                      self.char_at(idx), idx))
 207             }
 208 
 209             buf <<= 6;
 210             modulus += 1;
 211             if modulus == 4 {
 212                 modulus = 0;
 213                 r.push((buf >> 22) as u8);
 214                 r.push((buf >> 14) as u8);
 215                 r.push((buf >> 6 ) as u8);
 216             }
 217         }
 218 
 219         for (idx, byte) in it {
 220             if (byte as char) != '=' {
 221                 return Err(fmt!("Invalid character '%c' at position %u",
 222                                 self.char_at(idx), idx));
 223             }
 224         }
 225 
 226         match modulus {
 227             2 => {
 228                 r.push((buf >> 10) as u8);
 229             }
 230             3 => {
 231                 r.push((buf >> 16) as u8);
 232                 r.push((buf >> 8 ) as u8);
 233             }
 234             0 => (),
 235             _ => return Err(~"Invalid Base64 length")
 236         }
 237 
 238         Ok(r)
 239     }
 240 }

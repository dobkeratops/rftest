  11 /*!
  12 A library for iterating through the lines in a series of
  13 files. Very similar to [the Python module of the same
  14 name](http://docs.python.org/3.3/library/fileinput.html).
  15 
  16 It allows the programmer to automatically take filenames from the
  17 command line arguments (via `input` and `input_state`), as well as
  18 specify them as a vector directly (`input_vec` and
  19 `input_vec_state`). The files are opened as necessary, so any files
  20 that can't be opened only cause an error when reached in the
  21 iteration.
  22 
  23 On the command line, `stdin` is represented by a filename of `-` (a
  24 single hyphen) and in the functions that take a vector directly
  25 (e.g. `input_vec`) it is represented by `None`. Note `stdin` is *not*
  26 reset once it has been finished, so attempting to iterate on `[None,
  27 None]` will only take input once unless `io::stdin().seek(0, SeekSet)`
  28 is called between.
  29 
  30 The `pathify` function handles converting a list of file paths as
  31 strings to the appropriate format, including the (optional) conversion
  32 of `"-"` to `stdin`.
  33 
  34 # Basic
  35 
  36 In many cases, one can use the `input_*` functions without having
  37 to handle any `FileInput` structs. E.g. a simple `cat` program
  38 
  39     for input |line| {
  40         io::println(line)
  41     }
  42 
  43 or a program that numbers lines after concatenating two files
  44 
  45     for input_vec_state(pathify([~"a.txt", ~"b.txt"])) |line, state| {
  46         io::println(fmt!("%u: %s", state.line_num,
  47                                    line));
  48     }
  49 
  50 The two `input_vec*` functions take a vec of file names (where empty
  51 means read from `stdin`), the other two functions use the command line
  52 arguments.
  53 
  54 # Advanced
  55 
  56 For more complicated uses (e.g. if one needs to pause iteration and
  57 resume it later), a `FileInput` instance can be constructed via the
  58 `from_vec`, `from_vec_raw` and `from_args` functions.
  59 
  60 Once created, the `each_line` (from the `std::io::ReaderUtil` trait)
  61 and `each_line_state` methods allow one to iterate on the lines; the
  62 latter provides more information about the position within the
  63 iteration to the caller.
  64 
  65 It is possible (and safe) to skip lines and files using the
  66 `read_line` and `next_file` methods. Also, `FileInput` implements
  67 `std::io::Reader`, and the state will be updated correctly while
  68 using any of those methods.
  69 
  70 E.g. the following program reads until an empty line, pauses for user
  71 input, skips the current file and then numbers the remaining lines
  72 (where the numbers are from the start of each file, rather than the
  73 total line count).
  74 
  75     let input = FileInput::from_vec(pathify([~"a.txt", ~"b.txt", ~"c.txt"],
  76                                              true));
  77 
  78     for input.each_line |line| {
  79         if line.is_empty() {
  80             break
  81         }
  82         io::println(line);
  83     }
  84 
  85     io::println("Continue?");
  86 
  87     if io::stdin().read_line() == ~"yes" {
  88         input.next_file(); // skip!
  89 
  90         for input.each_line_state |line, state| {
  91            io::println(fmt!("%u: %s", state.line_num_file,
  92                                       line))
  93         }
  94     }
  95 */

 104 /**
 105 A summary of the internal state of a `FileInput` object. `line_num`
 106 and `line_num_file` represent the number of lines read in total and in
 107 the current file respectively. `current_path` is `None` if the current
 108 file is `stdin`.
 109 */
 110 #[deriving(Clone)]
 111 pub struct FileInputState {
 112     current_path: Option<Path>,
 113     line_num: uint,
 114     line_num_file: uint
 115 }
 116 
 117 impl FileInputState {
 118     fn is_stdin(&self) -> bool {
 119         self.current_path.is_none()
 120     }
 121 
 122     fn is_first_line(&self) -> bool {
 123         self.line_num_file == 1
 124     }
 125 }

 127 struct FileInput_ {
 128     /**
 129     `Some(path)` is the file represented by `path`, `None` is
 130     `stdin`. Consumed as the files are read.
 131     */
 132     priv files: ~[Option<Path>],
 133     /**
 134     The current file: `Some(r)` for an open file, `None` before
 135     starting and after reading everything.
 136     */
 137     priv current_reader: Option<@io::Reader>,
 138     priv state: FileInputState,
 139 
 140     /**
 141     Used to keep track of whether we need to insert the newline at the
 142     end of a file that is missing it, which is needed to separate the
 143     last and first lines.
 144     */
 145     priv previous_was_newline: bool
 146 }

 168     }

 174     pub fn from_vec_raw(files: ~[Option<Path>])
 175                                          -> FileInput {

 191     /**

 201     priv fn current_file_eof(&self) -> bool {
 202         match self.fi.current_reader {
 203             None => false,
 204             Some(r) => r.eof()
 205         }
 206     }
 207 
 208     /**
 209     Skip to the next file in the queue. Can `fail` when opening
 210     a file.
 211 
 212     Returns `false` if there is no more files, and `true` when it
 213     successfully opens the next file.
 214     */
 215 
 216     pub fn next_file(&self) -> bool {
 217         // No more files
 218 
 219         if self.fi.files.is_empty() {
 220             self.fi.current_reader = None;
 221             return false;
 222         }
 223 
 224         let path_option = self.fi.files.shift();
 225         let file = match path_option {
 226             None => io::stdin(),
 227             Some(ref path) => io::file_reader(path).unwrap()
 228         };
 229 
 230         self.fi.current_reader = Some(file);
 231         self.fi.state.current_path = path_option;
 232         self.fi.state.line_num_file = 0;
 233         true
 234     }
 235 
 236     /**
 237     Attempt to open the next file if there is none currently open,
 238     or if the current one is EOF'd.
 239 
 240     Returns `true` if it had to move to the next file and did
 241     so successfully.
 242     */
 243     priv fn next_file_if_eof(&self) -> bool {
 244         match self.fi.current_reader {
 245             None => self.next_file(),
 246             Some(r) => {
 247                 if r.eof() {
 248                     self.next_file()
 249                 } else {
 250                     false
 251                 }
 252             }
 253         }
 254     }
 255 
 256     /**
 257     Apply `f` to each line successively, along with some state
 258     (line numbers and file names, see documentation for
 259     `FileInputState`). Otherwise identical to `lines_each`.
 260     */
 261     pub fn each_line_state(&self,
 262                             f: &fn(&str, FileInputState) -> bool) -> bool {
 263          self.each_line(|line| f(line, self.fi.state.clone()))
 264     }
 265 
 266 
 267     /**
 268     Retrieve the current `FileInputState` information.
 269     */
 270     pub fn state(&self) -> FileInputState {
 271         self.fi.state.clone()
 272     }
 273 }

 275 impl io::Reader for FileInput {
 276     fn read_byte(&self) -> int {
 277         loop {
 278             let stepped = self.next_file_if_eof();
 279 
 280             // if we moved to the next file, and the previous
 281             // character wasn't \n, then there is an unfinished line
 282             // from the previous file. This library models
 283             // line-by-line processing and the trailing line of the
 284             // previous file and the leading of the current file
 285             // should be considered different, so we need to insert a
 286             // fake line separator
 287             if stepped && !self.fi.previous_was_newline {
 288                 self.fi.state.line_num += 1;
 289                 self.fi.state.line_num_file += 1;
 290                 self.fi.previous_was_newline = true;
 291                 return '\n' as int;
 292             }
 293 
 294             match self.fi.current_reader {
 295                 None => return -1,
 296                 Some(r) => {
 297                     let b = r.read_byte();
 298 
 299                     if b < 0 {
 300                         loop;
 301                     }
 302 
 303                     if b == '\n' as int {
 304                         self.fi.state.line_num += 1;
 305                         self.fi.state.line_num_file += 1;
 306                         self.fi.previous_was_newline = true;
 307                     } else {
 308                         self.fi.previous_was_newline = false;
 309                     }
 310 
 311                     return b;
 312                 }
 313             }
 314         }
 315     }
 316     fn read(&self, buf: &mut [u8], len: uint) -> uint {
 317         let mut count = 0;
 318         while count < len {
 319             let b = self.read_byte();
 320             if b < 0 { break }
 321 
 322             buf[count] = b as u8;
 323             count += 1;
 324         }
 325 
 326         count
 327     }
 328     fn eof(&self) -> bool {
 329         // we've run out of files, and current_reader is either None or eof.
 330 
 331         self.fi.files.is_empty() &&
 332             match self.fi.current_reader { None => true, Some(r) => r.eof() }
 333 
 334     }
 335     fn seek(&self, offset: int, whence: io::SeekStyle) {
 336         match self.fi.current_reader {
 337             None => {},
 338             Some(r) => r.seek(offset, whence)
 339         }
 340     }
 341     fn tell(&self) -> uint {
 342         match self.fi.current_reader {
 343             None => 0,
 344             Some(r) => r.tell()
 345         }
 346     }
 347 }

 349 /**
 350 Convert a list of strings to an appropriate form for a `FileInput`
 351 instance. `stdin_hyphen` controls whether `-` represents `stdin` or
 352 a literal `-`.
 353 */
 354 // XXX: stupid, unclear name
 355 pub fn pathify(vec: &[~str], stdin_hyphen : bool) -> ~[Option<Path>] {
 356     vec.iter().transform(|str| {
 357         if stdin_hyphen && "-" == *str {
 358             None
 359         } else {
 360             Some(Path(*str))
 361         }
 362     }).collect()
 363 }

  11  /*!
  12  A library for iterating through the lines in a series of
  13  files. Very similar to [the Python module of the same
  14  name](http://docs.python.org/3.3/library/fileinput.html).
  15  
  16  It allows the programmer to automatically take filenames from the
  17  command line arguments (via `input` and `input_state`), as well as
  18  specify them as a vector directly (`input_vec` and
  19  `input_vec_state`). The files are opened as necessary, so any files
  20  that can't be opened only cause an error when reached in the
  21  iteration.
  22  
  23  On the command line, `stdin` is represented by a filename of `-` (a
  24  single hyphen) and in the functions that take a vector directly
  25  (e.g. `input_vec`) it is represented by `None`. Note `stdin` is *not*
  26  reset once it has been finished, so attempting to iterate on `[None,
  27  None]` will only take input once unless `io::stdin().seek(0, SeekSet)`
  28  is called between.
  29  
  30  The `make_path_option_vec` function handles converting a list of file paths as
  31  strings to the appropriate format, including the (optional) conversion
  32  of `"-"` to `stdin`.
  33  
  34  # Basic
  35  
  36  In many cases, one can use the `input_*` functions without having
  37  to handle any `FileInput` structs. E.g. a simple `cat` program
  38  
  39      for input |line| {
  40          io::println(line)
  41      }
  42  
  43  or a program that numbers lines after concatenating two files
  44  
  45      for input_vec_state(make_path_option_vec([~"a.txt", ~"b.txt"])) |line, state| {
  46          io::println(fmt!("%u: %s", state.line_num,
  47                                     line));
  48      }
  49  
  50  The two `input_vec*` functions take a vec of file names (where empty
  51  means read from `stdin`), the other two functions use the command line
  52  arguments.
  53  
  54  # Advanced
  55  
  56  For more complicated uses (e.g. if one needs to pause iteration and
  57  resume it later), a `FileInput` instance can be constructed via the
  58  `from_vec`, `from_vec_raw` and `from_args` functions.
  59  
  60  Once created, the `each_line` (from the `std::io::ReaderUtil` trait)
  61  and `each_line_state` methods allow one to iterate on the lines; the
  62  latter provides more information about the position within the
  63  iteration to the caller.
  64  
  65  It is possible (and safe) to skip lines and files using the
  66  `read_line` and `next_file` methods. Also, `FileInput` implements
  67  `std::io::Reader`, and the state will be updated correctly while
  68  using any of those methods.
  69  
  70  E.g. the following program reads until an empty line, pauses for user
  71  input, skips the current file and then numbers the remaining lines
  72  (where the numbers are from the start of each file, rather than the
  73  total line count).
  74  
  75      let input = FileInput::from_vec(pathify([~"a.txt", ~"b.txt", ~"c.txt"],
  76                                               true));
  77  
  78      for input.each_line |line| {
  79          if line.is_empty() {
  80              break
  81          }
  82          io::println(line);
  83      }
  84  
  85      io::println("Continue?");
  86  
  87      if io::stdin().read_line() == ~"yes" {
  88          input.next_file(); // skip!
  89  
  90          for input.each_line_state |line, state| {
  91             io::println(fmt!("%u: %s", state.line_num_file,
  92                                        line))
  93          }
  94      }
  95  */
  96

 104  /**
 105  A summary of the internal state of a `FileInput` object. `line_num`
 106  and `line_num_file` represent the number of lines read in total and in
 107  the current file respectively. `current_path` is `None` if the current
 108  file is `stdin`.
 109  */
 110  #[deriving(Clone)]
 111  pub struct FileInputState {
 112      current_path: Option<Path>,
 113      line_num: uint,
 114      line_num_file: uint
 115  }
 116  
 117  impl FileInputState {
 118      fn is_stdin(&self) -> bool {
 119          self.current_path.is_none()
 120      }
 121  
 122      fn is_first_line(&self) -> bool {
 123          self.line_num_file == 1
 124      }
 125  }

 127  struct FileInput_ {
 128      /**
 129      `Some(path)` is the file represented by `path`, `None` is
 130      `stdin`. Consumed as the files are read.
 131      */
 132      files: ~[Option<Path>],
 133      /**
 134      The current file: `Some(r)` for an open file, `None` before
 135      starting and after reading everything.
 136      */
 137      current_reader: Option<@io::Reader>,
 138      state: FileInputState,
 139  
 140      /**
 141      Used to keep track of whether we need to insert the newline at the
 142      end of a file that is missing it, which is needed to separate the
 143      last and first lines.
 144      */
 145      previous_was_newline: bool
 146  }

 161  impl FileInput {
 162      /**
 163      Create a `FileInput` object from a vec of files. An empty
 164      vec means lines are read from `stdin` (use `from_vec_raw` to stop
 165      this behaviour). Any occurrence of `None` represents `stdin`.
 166      */
 167      pub fn from_vec(files: ~[Option<Path>]) -> FileInput {
 168          FileInput::from_vec_raw(
 169              if files.is_empty() {
 170                  ~[None]
 171              } else {
 172                  files
 173              })
 174      }
 175  
 176      /**
 177      Identical to `from_vec`, but an empty `files` vec stays
 178      empty. (`None` is `stdin`.)
 179      */
 180      pub fn from_vec_raw(files: ~[Option<Path>])
 181                                           -> FileInput {
 182          FileInput{
 183              fi: @mut FileInput_ {
 184                  files: files,
 185                  current_reader: None,
 186                  state: FileInputState {
 187                      current_path: None,
 188                      line_num: 0,
 189                      line_num_file: 0
 190                  },
 191                  // there was no previous unended line
 192                  previous_was_newline: true
 193              }
 194          }
 195      }
 196  
 197      /**
 198      Create a `FileInput` object from the command line
 199      arguments. `"-"` represents `stdin`.
 200      */
 201      pub fn from_args() -> FileInput {
 202          let args = os::args();
 203          let pathed = make_path_option_vec(args.tail(), true);
 204          FileInput::from_vec(pathed)
 205      }
 206  
 207      fn current_file_eof(&self) -> bool {
 208          match self.fi.current_reader {
 209              None => false,
 210              Some(r) => r.eof()
 211          }
 212      }
 213  
 214      /**
 215      Skip to the next file in the queue. Can `fail` when opening
 216      a file.
 217  
 218      Returns `false` if there is no more files, and `true` when it
 219      successfully opens the next file.
 220      */
 221  
 222      pub fn next_file(&self) -> bool {
 223          // No more files
 224  
 225          if self.fi.files.is_empty() {
 226              self.fi.current_reader = None;
 227              return false;
 228          }
 229  
 230          let path_option = self.fi.files.shift();
 231          let file = match path_option {
 232              None => io::stdin(),
 233              Some(ref path) => io::file_reader(path).unwrap()
 234          };
 235  
 236          self.fi.current_reader = Some(file);
 237          self.fi.state.current_path = path_option;
 238          self.fi.state.line_num_file = 0;
 239          true
 240      }
 241  
 242      /**
 243      Attempt to open the next file if there is none currently open,
 244      or if the current one is EOF'd.
 245  
 246      Returns `true` if it had to move to the next file and did
 247      so successfully.
 248      */
 249      fn next_file_if_eof(&self) -> bool {
 250          match self.fi.current_reader {
 251              None => self.next_file(),
 252              Some(r) => {
 253                  if r.eof() {
 254                      self.next_file()
 255                  } else {
 256                      false
 257                  }
 258              }
 259          }
 260      }
 261  
 262      /**
 263      Apply `f` to each line successively, along with some state
 264      (line numbers and file names, see documentation for
 265      `FileInputState`). Otherwise identical to `lines_each`.
 266      */
 267      pub fn each_line_state(&self,
 268                              f: &fn(&str, FileInputState) -> bool) -> bool {
 269           self.each_line(|line| f(line, self.fi.state.clone()))
 270      }
 271  
 272  
 273      /**
 274      Retrieve the current `FileInputState` information.
 275      */
 276      pub fn state(&self) -> FileInputState {
 277          self.fi.state.clone()
 278      }
 279  }

 281  impl io::Reader for FileInput {
 282      fn read_byte(&self) -> int {
 283          loop {
 284              let stepped = self.next_file_if_eof();
 285  
 286              // if we moved to the next file, and the previous
 287              // character wasn't \n, then there is an unfinished line
 288              // from the previous file. This library models
 289              // line-by-line processing and the trailing line of the
 290              // previous file and the leading of the current file
 291              // should be considered different, so we need to insert a
 292              // fake line separator
 293              if stepped && !self.fi.previous_was_newline {
 294                  self.fi.state.line_num += 1;
 295                  self.fi.state.line_num_file += 1;
 296                  self.fi.previous_was_newline = true;
 297                  return '\n' as int;
 298              }
 299  
 300              match self.fi.current_reader {
 301                  None => return -1,
 302                  Some(r) => {
 303                      let b = r.read_byte();
 304  
 305                      if b < 0 {
 306                          loop;
 307                      }
 308  
 309                      if b == '\n' as int {
 310                          self.fi.state.line_num += 1;
 311                          self.fi.state.line_num_file += 1;
 312                          self.fi.previous_was_newline = true;
 313                      } else {
 314                          self.fi.previous_was_newline = false;
 315                      }
 316  
 317                      return b;
 318                  }
 319              }
 320          }
 321      }
 322      fn read(&self, buf: &mut [u8], len: uint) -> uint {
 323          let mut count = 0;
 324          while count < len {
 325              let b = self.read_byte();
 326              if b < 0 { break }
 327  
 328              buf[count] = b as u8;
 329              count += 1;
 330          }
 331  
 332          count
 333      }
 334      fn eof(&self) -> bool {
 335          // we've run out of files, and current_reader is either None or eof.
 336  
 337          self.fi.files.is_empty() &&
 338              match self.fi.current_reader { None => true, Some(r) => r.eof() }
 339  
 340      }
 341      fn seek(&self, offset: int, whence: io::SeekStyle) {
 342          match self.fi.current_reader {
 343              None => {},
 344              Some(r) => r.seek(offset, whence)
 345          }
 346      }
 347      fn tell(&self) -> uint {
 348          match self.fi.current_reader {
 349              None => 0,
 350              Some(r) => r.tell()
 351          }
 352      }
 353  }

 355  /**
 356  Convert a list of strings to an appropriate form for a `FileInput`
 357  instance. `stdin_hyphen` controls whether `-` represents `stdin` or
 358  a literal `-`.
 359  */
 360  pub fn make_path_option_vec(vec: &[~str], stdin_hyphen : bool) -> ~[Option<Path>] {
 361      vec.iter().map(|str| {
 362          if stdin_hyphen && "-" == *str {
 363              None
 364          } else {
 365              Some(Path(*str))
 366          }
 367      }).collect()
 368  }

 371  Iterate directly over the command line arguments (no arguments implies
 372  reading from `stdin`).
 373  
 374  Fails when attempting to read from a file that can't be opened.
 375  */
 376  pub fn input(f: &fn(&str) -> bool) -> bool {
 377      let i = FileInput::from_args();
 378      i.each_line(f)
 379  }
 380  
 381  /**
 382  Iterate directly over the command line arguments (no arguments
 383  implies reading from `stdin`) with the current state of the iteration
 384  provided at each call.
 385  
 386  Fails when attempting to read from a file that can't be opened.
 387  */
 388  pub fn input_state(f: &fn(&str, FileInputState) -> bool) -> bool {
 389      let i = FileInput::from_args();
 390      i.each_line_state(f)
 391  }
 392  
 393  /**
 394  Iterate over a vector of files (an empty vector implies just `stdin`).
 395  
 396  Fails when attempting to read from a file that can't be opened.
 397  */
 398  pub fn input_vec(files: ~[Option<Path>], f: &fn(&str) -> bool) -> bool {
 399      let i = FileInput::from_vec(files);
 400      i.each_line(f)
 401  }
 402  
 403  /**
 404  Iterate over a vector of files (an empty vector implies just `stdin`)
 405  with the current state of the iteration provided at each call.
 406  
 407  Fails when attempting to read from a file that can't be opened.
 408  */
 409  pub fn input_vec_state(files: ~[Option<Path>],
 410                         f: &fn(&str, FileInputState) -> bool) -> bool {
 411      let i = FileInput::from_vec(files);
 412      i.each_line_state(f)
 413  }

  124 // The default console test runner. It accepts the command line
  125 // arguments and a vector of test_descs.
  126 pub fn test_main(args: &[~str], tests: ~[TestDescAndFn]) {
  127     let opts =
  128         match parse_opts(args) {
  129           either::Left(o) => o,
  130           either::Right(m) => fail!(m)
  131         };
  132     if !run_tests_console(&opts, tests) { fail!("Some tests failed"); }
  133 }
  134 
  135 // A variant optimized for invocation with a static test vector.
  136 // This will fail (intentionally) when fed any dynamic tests, because
  137 // it is copying the static values out into a dynamic vector and cannot
  138 // copy dynamic values. It is doing this because from this point on
  139 // a ~[TestDescAndFn] is used in order to effect ownership-transfer
  140 // semantics into parallel test runners, which in turn requires a ~[]
  141 // rather than a &[].
  142 pub fn test_main_static(args: &[~str], tests: &[TestDescAndFn]) {
  143     let owned_tests = do tests.map |t| {
  144         match t.testfn {
  145             StaticTestFn(f) =>
  146             TestDescAndFn { testfn: StaticTestFn(f), desc: t.desc.clone() },
  147 
  148             StaticBenchFn(f) =>
  149             TestDescAndFn { testfn: StaticBenchFn(f), desc: t.desc.clone() },
  150 
  151             _ => {
  152                 fail!("non-static tests passed to test::test_main_static");
  153             }
  154         }
  155     };
  156     test_main(args, owned_tests)
  157 }

  190 fn usage(binary: &str, helpstr: &str) -> ! {
  191     let message = fmt!("Usage: %s [OPTIONS] [FILTER]", binary);
  192     println(groups::usage(message, optgroups()));
  193     println("");
  194     if helpstr == "help" {
  195         println("\
  196 The FILTER is matched against the name of all tests to run, and if any tests
  197 have a substring match, only those tests are run.
  198 
  199 By default, all tests are run in parallel. This can be altered with the
  200 RUST_THREADS environment variable when running tests (set it to 1).
  201 
  202 Test Attributes:
  203 
  204     #[test]        - Indicates a function is a test to be run. This function
  205                      takes no arguments.
  206     #[bench]       - Indicates a function is a benchmark to be run. This
  207                      function takes one argument (extra::test::BenchHarness).
  208     #[should_fail] - This function (also labeled with #[test]) will only pass if
  209                      the code causes a failure (an assertion failure or fail!)
  210     #[ignore]      - When applied to a function which is already attributed as a
  211                      test, then the test runner will ignore these tests during
  212                      normal test runs. Running with --ignored will run these
  213                      tests. This may also be written as #[ignore(cfg(...))] to
  214                      ignore the test on certain configurations.");
  215     }
  216     unsafe { libc::exit(0) }
  217 }

  219 // Parses command line arguments into test options
  220 pub fn parse_opts(args: &[~str]) -> OptRes {
  221     let args_ = args.tail();
  222     let matches =
  223         match groups::getopts(args_, optgroups()) {
  224           Ok(m) => m,
  225           Err(f) => return either::Right(getopts::fail_str(f))
  226         };
  227 
  228     if getopts::opt_present(&matches, "h") { usage(args[0], "h"); }
  229     if getopts::opt_present(&matches, "help") { usage(args[0], "help"); }
  230 
  231     let filter =
  232         if matches.free.len() > 0 {
  233             Some((matches).free[0].clone())
  234         } else {
  235             None
  236         };
  237 
  238     let run_ignored = getopts::opt_present(&matches, "ignored");
  239 
  240     let logfile = getopts::opt_maybe_str(&matches, "logfile");
  241     let logfile = logfile.map(|s| Path(*s));
  242 
  243     let run_benchmarks = getopts::opt_present(&matches, "bench");
  244     let run_tests = ! run_benchmarks ||
  245         getopts::opt_present(&matches, "test");
  246 
  247     let ratchet_metrics = getopts::opt_maybe_str(&matches, "ratchet-metrics");
  248     let ratchet_metrics = ratchet_metrics.map(|s| Path(*s));
  249 
  250     let ratchet_noise_percent =
  251         getopts::opt_maybe_str(&matches, "ratchet-noise-percent");
  252     let ratchet_noise_percent = ratchet_noise_percent.map(|s| f64::from_str(*s).get());
  253 
  254     let save_metrics = getopts::opt_maybe_str(&matches, "save-metrics");
  255     let save_metrics = save_metrics.map(|s| Path(*s));
  256 
  257     let test_opts = TestOpts {
  258         filter: filter,
  259         run_ignored: run_ignored,
  260         run_tests: run_tests,
  261         run_benchmarks: run_benchmarks,
  262         ratchet_metrics: ratchet_metrics,
  263         ratchet_noise_percent: ratchet_noise_percent,
  264         save_metrics: save_metrics,
  265         logfile: logfile
  266     };
  267 
  268     either::Left(test_opts)
  269 }

  441     pub fn write_metric_diff(&self, diff: &MetricDiff) {
  442         let mut noise = 0;
  443         let mut improved = 0;
  444         let mut regressed = 0;
  445         let mut added = 0;
  446         let mut removed = 0;
  447 
  448         for (k, v) in diff.iter() {
  449             match *v {
  450                 LikelyNoise => noise += 1,
  451                 MetricAdded => {
  452                     added += 1;
  453                     self.write_added();
  454                     self.out.write_line(fmt!(": %s", *k));
  455                 }
  456                 MetricRemoved => {
  457                     removed += 1;
  458                     self.write_removed();
  459                     self.out.write_line(fmt!(": %s", *k));
  460                 }
  461                 Improvement(pct) => {
  462                     improved += 1;
  463                     self.out.write_str(*k);
  464                     self.out.write_str(": ");
  465                     self.write_improved();
  466                     self.out.write_line(fmt!(" by %.2f%%", pct as float))
  467                 }
  468                 Regression(pct) => {
  469                     regressed += 1;
  470                     self.out.write_str(*k);
  471                     self.out.write_str(": ");
  472                     self.write_regressed();
  473                     self.out.write_line(fmt!(" by %.2f%%", pct as float))
  474                 }
  475             }
  476         }
  477         self.out.write_line(fmt!("result of ratchet: %u matrics added, %u removed, \
  478                                   %u improved, %u regressed, %u noise",
  479                                  added, removed, improved, regressed, noise));
  480         if regressed == 0 {
  481             self.out.write_line("updated ratchet file")
  482         } else {
  483             self.out.write_line("left ratchet file untouched")
  484         }
  485     }

  489                             ratchet_pct: Option<f64>) -> bool {

  497                     None => (),

  503                 self.write_metric_diff(&diff);
  504                 ok
  505             }
  506         };

  508         let test_success = self.failed == 0u;
  509         if !test_success {
  510             self.write_failures();
  511         }

  513         let success = ratchet_success && test_success;
  514 
  515         self.out.write_str("\ntest result: ");
  516         if success {
  517             // There's no parallelism at this point so it's safe to use color
  518             self.write_ok();
  519         } else {
  520             self.write_failed();
  521         }
  522         self.out.write_str(fmt!(". %u passed; %u failed; %u ignored; %u measured\n\n",
  523                                 self.passed, self.failed, self.ignored, self.measured));
  524         return success;
  525     }

  881     /// Compare against another MetricMap. Optionally compare all
  882     /// measurements in the maps using the provided `noise_pct` as a
  883     /// percentage of each value to consider noise. If `None`, each
  884     /// measurement's noise threshold is independently chosen as the
  885     /// maximum of that measurement's recorded noise quantity in either
  886     /// map.
  887     pub fn compare_to_old(&self, old: &MetricMap,
  888                           noise_pct: Option<f64>) -> MetricDiff {
  889         let mut diff : MetricDiff = TreeMap::new();
  890         for (k, vold) in old.iter() {
  891             let r = match self.find(k) {
  892                 None => MetricRemoved,
  893                 Some(v) => {
  894                     let delta = (v.value - vold.value);
  895                     let noise = match noise_pct {
  896                         None => f64::max(vold.noise.abs(), v.noise.abs()),
  897                         Some(pct) => vold.value * pct / 100.0
  898                     };
  899                     if delta.abs() <= noise {
  900                         LikelyNoise
  901                     } else {
  902                         let pct = delta.abs() / (vold.value).max(&f64::epsilon) * 100.0;
  903                         if vold.noise < 0.0 {
  904                             // When 'noise' is negative, it means we want
  905                             // to see deltas that go up over time, and can
  906                             // only tolerate slight negative movement.
  907                             if delta < 0.0 {
  908                                 Regression(pct)
  909                             } else {
  910                                 Improvement(pct)
  911                             }
  912                         } else {
  913                             // When 'noise' is positive, it means we want
  914                             // to see deltas that go down over time, and
  915                             // can only tolerate slight positive movements.
  916                             if delta < 0.0 {
  917                                 Improvement(pct)
  918                             } else {
  919                                 Regression(pct)
  920                             }
  921                         }
  922                     }
  923                 }
  924             };
  925             diff.insert((*k).clone(), r);
  926         }
  927         for (k, _) in self.iter() {
  928             if !diff.contains_key(k) {
  929                 diff.insert((*k).clone(), MetricAdded);
  930             }
  931         }
  932         diff
  933     }

  935     /// Insert a named `value` (+/- `noise`) metric into the map. The value
  936     /// must be non-negative. The `noise` indicates the uncertainty of the
  937     /// metric, which doubles as the "noise range" of acceptable
  938     /// pairwise-regressions on this named value, when comparing from one
  939     /// metric to the next using `compare_to_old`.
  940     ///
  941     /// If `noise` is positive, then it means this metric is of a value
  942     /// you want to see grow smaller, so a change larger than `noise` in the
  943     /// positive direction represents a regression.
  944     ///
  945     /// If `noise` is negative, then it means this metric is of a value
  946     /// you want to see grow larger, so a change larger than `noise` in the

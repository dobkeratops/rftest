  22 /// Calculates the md4 hash of the given slice of bytes, returning the 128-bit
  23 /// result as a quad of u32's
  24 pub fn md4(msg: &[u8]) -> Quad {

  26     // which performs shifts by 32 bits or more has undefined
  27     // results.
  28     let orig_len: u64 = (msg.len() * 8u) as u64;
  29 
  30     // pad message
  31     let mut msg = vec::append(msg.to_owned(), [0x80u8]);
  32     let mut bitlen = orig_len + 8u64;
  33     while (bitlen + 64u64) % 512u64 > 0u64 {
  34         msg.push(0u8);
  35         bitlen += 8u64;
  36     }
  37 
  38     // append length
  39     let mut i = 0u64;
  40     while i < 8u64 {
  41         msg.push((orig_len >> (i * 8u64)) as u8);
  42         i += 1u64;
  43     }
  44 
  45     let mut a = 0x67452301u32;
  46     let mut b = 0xefcdab89u32;
  47     let mut c = 0x98badcfeu32;
  48     let mut d = 0x10325476u32;

  50     fn rot(r: int, x: u32) -> u32 {
  51         let r = r as u32;

  53     }
  54 
  55     let mut i = 0u;
  56     let e = msg.len();

  58     while i < e {
  59         let (aa, bb, cc, dd) = (a, b, c, d);
  60

  64             x[j] = (msg[base] as u32) + (msg[base + 1u] as u32 << 8u32) +

  66                 (msg[base + 3u] as u32 << 24u32);

  69 
  70         let mut j = 0u;

  72             a = rot(3, a + ((b & c) | (!b & d)) + x[j]);
  73             j += 1u;
  74             d = rot(7, d + ((a & b) | (!a & c)) + x[j]);
  75             j += 1u;
  76             c = rot(11, c + ((d & a) | (!d & b)) + x[j]);
  77             j += 1u;
  78             b = rot(19, b + ((c & d) | (!c & a)) + x[j]);
  79             j += 1u;
  80         }
  81 
  82         let mut j = 0u;
  83         let q = 0x5a827999u32;
  84         while j < 4u {
  85             a = rot(3, a + ((b & c) | ((b & d) | (c & d))) + x[j] + q);
  86             d = rot(5, d + ((a & b) | ((a & c) | (b & c))) + x[j + 4u] + q);

  88             b = rot(13, b + ((c & d) | ((c & a) | (d & a))) + x[j + 12u] + q);

  90         }
  91 
  92         let mut j = 0u;

  95             let jj = if j > 2u { j - 3u } else { j };

  97             d = rot(9, d + (a ^ b ^ c) + x[jj + 8u] + q);
  98             c = rot(11, c + (d ^ a ^ b) + x[jj + 4u] + q);
  99             b = rot(15, b + (c ^ d ^ a) + x[jj + 12u] + q);
 100             j += 2u;
 101         }
 102 
 103         a += aa; b += bb; c += cc; d += dd;
 104         i += 64u;
 105     }
 106     return Quad {a: a, b: b, c: c, d: d};
 107 }

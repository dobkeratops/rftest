   36  /// A record specifying a time value in seconds and nanoseconds.
   37  #[deriving(Eq, Encodable, Decodable)]
   38  pub struct Timespec { sec: i64, nsec: i32 }
   39  
   40  /*
   41   * Timespec assumes that pre-epoch Timespecs have negative sec and positive
   42   * nsec fields. Darwin's and Linux's struct timespec functions handle pre-
   43   * epoch timestamps using a "two steps back, one step forward" representation,
   44   * though the man pages do not actually document this. For example, the time
   45   * -1.2 seconds before the epoch is represented by `Timespec { sec: -2_i64,
   46   * nsec: 800_000_000_i32 }`.
   47   */
   48  impl Timespec {
   49      pub fn new(sec: i64, nsec: i32) -> Timespec {
   50          assert!(nsec >= 0 && nsec < NSEC_PER_SEC);
   51          Timespec { sec: sec, nsec: nsec }
   52      }
   53  }

   55  impl Ord for Timespec {
   56      fn lt(&self, other: &Timespec) -> bool {
   57          self.sec < other.sec ||
   58              (self.sec == other.sec && self.nsec < other.nsec)
   59      }
   60  }

  236       *

  242              self.strftime("%Y-%m-%dT%H:%M:%SZ")
  243          } else {
  244              let s = self.strftime("%Y-%m-%dT%H:%M:%S");
  245              let sign = if self.tm_gmtoff > 0_i32 { '+' } else { '-' };
  246              let mut m = num::abs(self.tm_gmtoff) / 60_i32;
  247              let h = m / 60_i32;
  248              m -= h * 60_i32;
  249              s + fmt!("%c%02d:%02d", sign, h as int, m as int)
  250          }
  251      }
  252  }

  255      fn match_str(s: &str, pos: uint, needle: &str) -> bool {
  256          let mut i = pos;
  257          for ch in needle.byte_iter() {
  258              if s[i] != ch {
  259                  return false;
  260              }
  261              i += 1u;
  262          }
  263          return true;
  264      }

  266      fn match_strs(ss: &str, pos: uint, strs: &[(~str, i32)])
  267        -> Option<(i32, uint)> {
  268          let mut i = 0u;
  269          let len = strs.len();
  270          while i < len {
  271              match strs[i] { // can't use let due to let-pattern bugs
  272                  (ref needle, value) => {
  273                      if match_str(ss, pos, *needle) {
  274                          return Some((value, pos + needle.len()));
  275                      }
  276                  }
  277              }
  278              i += 1u;
  279          }
  280  
  281          None
  282      }

  284      fn match_digits(ss: &str, pos: uint, digits: uint, ws: bool)
  285        -> Option<(i32, uint)> {
  286          let mut pos = pos;
  287          let len = ss.len();
  288          let mut value = 0_i32;
  289  
  290          let mut i = 0u;
  291          while i < digits {
  292              if pos >= len {
  293                  return None;
  294              }
  295              let range = ss.char_range_at(pos);
  296              pos = range.next;
  297  
  298              match range.ch {
  299                '0' .. '9' => {
  300                  value = value * 10_i32 + (range.ch as i32 - '0' as i32);
  301                }
  302                ' ' if ws => (),
  303                _ => return None
  304              }
  305              i += 1u;
  306          }
  307  
  308          Some((value, pos))
  309      }

  312                               min: i32, max: i32) -> Option<(i32, uint)> {
  313          match match_digits(ss, pos, digits, ws) {
  314            Some((val, pos)) if val >= min && val <= max => {

  316            }

  319      }

  375              None => Err(~"Invalid month")
  376            },

  392              None => Err(~"Invalid month")

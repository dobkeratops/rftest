  54      pub fn unwrap(self) -> A {

  58          match state {
  59              Forced(v) => v,
  60              _ => fail!( "Logic error." ),
  61          }
  62      }

  64      pub fn get_ref<'a>(&'a mut self) -> &'a A {
  65          /*!
  66          * Executes the future's closure and then returns a borrowed
  67          * pointer to the result.  The borrowed pointer lasts as long as
  68          * the future.
  69          */
  70          match self.state {
  71              Forced(ref v) => return v,
  72              Evaluating => fail!("Recursive forcing of future!"),
  73              Pending(_) => {
  74                  match replace(&mut self.state, Evaluating) {
  75                      Forced(_) | Evaluating => fail!("Logic error."),
  76                      Pending(f) => {
  77                          self.state = Forced(f());
  78                          self.get_ref()
  79                      }
  80                  }
  81              }
  82          }
  83      }

 124      pub fn spawn(blk: ~fn() -> A) -> Future<A> {
 125          /*!
 126           * Create a future from a unique closure.

 129           * value of the future.

 132          let (port, chan) = oneshot();

 134          do task::spawn_with(chan) |chan| {

 136          }

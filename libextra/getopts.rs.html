   11 /*!
   12  * Simple getopt alternative.
   13  *
   14  * Construct a vector of options, either by using reqopt, optopt, and optflag
   15  * or by building them from components yourself, and pass them to getopts,
   16  * along with a vector of actual arguments (not including argv[0]). You'll
   17  * either get a failure code back, or a match. You'll have to verify whether
   18  * the amount of 'free' arguments in the match is what you expect. Use opt_*
   19  * accessors to get argument values out of the matches object.
   20  *
   21  * Single-character options are expected to appear on the command line with a
   22  * single preceding dash; multiple-character options are expected to be
   23  * proceeded by two dashes. Options that expect an argument accept their
   24  * argument following either a space or an equals sign. Single-character
   25  * options don't require the space.
   26  *
   27  * # Example
   28  *
   29  * The following example shows simple command line parsing for an application
   30  * that requires an input file to be specified, accepts an optional output
   31  * file name following -o, and accepts both -h and --help as optional flags.
   32  *
   33  * ```
   34  *    extern mod extra;
   35  *    use extra::getopts::*;
   36  *    use std::os;
   37  *
   38  *    fn do_work(in: &str, out: Option<~str>) {
   39  *        println(in);
   40  *        println(match out {
   41  *            Some(x) => x,
   42  *            None => ~"No Output"
   43  *        });
   44  *    }
   45  *
   46  *    fn print_usage(program: &str, _opts: &[Opt]) {
   47  *        printfln!("Usage: %s [options]", program);
   48  *        println("-o\t\tOutput");
   49  *        println("-h --help\tUsage");
   50  *    }
   51  *
   52  *    fn main() {
   53  *        let args = os::args();
   54  *
   55  *        let program = args[0].clone();
   56  *
   57  *        let opts = ~[
   58  *            optopt("o"),
   59  *            optflag("h"),
   60  *            optflag("help")
   61  *        ];
   62  *        let matches = match getopts(args.tail(), opts) {
   63  *            Ok(m) => { m }
   64  *            Err(f) => { fail!(fail_str(f)) }
   65  *        };
   66  *        if opt_present(&matches, "h") || opt_present(&matches, "help") {
   67  *            print_usage(program, opts);
   68  *            return;
   69  *        }
   70  *        let output = opt_maybe_str(&matches, "o");
   71  *        let input: &str = if !matches.free.is_empty() {
   72  *            matches.free[0].clone()
   73  *        } else {
   74  *            print_usage(program, opts);
   75  *            return;
   76  *        };
   77  *        do_work(input, output);
   78  *    }
   79  * ```
   80  */

  249 /**
  250  * Parse command line arguments according to the provided options
  251  *
  252  * On success returns `ok(Opt)`. Use functions such as `opt_present`
  253  * `opt_str`, etc. to interrogate results.  Returns `err(Fail_)` on failure.
  254  * Use <fail_str> to get an error message.
  255  */
  256 pub fn getopts(args: &[~str], opts: &[Opt]) -> Result {
  257     let n_opts = opts.len();
  258     fn f(_x: uint) -> ~[Optval] { return ~[]; }
  259     let mut vals = vec::from_fn(n_opts, f);
  260     let mut free: ~[~str] = ~[];
  261     let l = args.len();
  262     let mut i = 0;
  263     while i < l {
  264         let cur = args[i].clone();
  265         let curlen = cur.len();
  266         if !is_arg(cur) {
  267             free.push(cur);
  268         } else if cur == ~"--" {
  269             let mut j = i + 1;
  270             while j < l { free.push(args[j].clone()); j += 1; }
  271             break;
  272         } else {
  273             let mut names;
  274             let mut i_arg = None;
  275             if cur[1] == '-' as u8 {
  276                 let tail = cur.slice(2, curlen);
  277                 let tail_eq: ~[&str] = tail.split_iter('=').collect();
  278                 if tail_eq.len() <= 1 {
  279                     names = ~[Long(tail.to_owned())];
  280                 } else {
  281                     names =
  282                         ~[Long(tail_eq[0].to_owned())];
  283                     i_arg = Some(tail_eq[1].to_owned());
  284                 }
  285             } else {
  286                 let mut j = 1;
  287                 let mut last_valid_opt_id = None;
  288                 names = ~[];
  289                 while j < curlen {
  290                     let range = cur.char_range_at(j);
  291                     let opt = Short(range.ch);
  292 
  293                     /* In a series of potential options (eg. -aheJ), if we
  294                        see one which takes an argument, we assume all
  295                        subsequent characters make up the argument. This
  296                        allows options such as -L/usr/local/lib/foo to be
  297                        interpreted correctly
  298                     */
  299 
  300                     match find_opt(opts, opt.clone()) {
  301                       Some(id) => last_valid_opt_id = Some(id),
  302                       None => {
  303                         let arg_follows =
  304                             last_valid_opt_id.is_some() &&
  305                             match opts[last_valid_opt_id.unwrap()]
  306                               .hasarg {
  307 
  308                               Yes | Maybe => true,
  309                               No => false
  310                             };
  311                         if arg_follows && j < curlen {
  312                             i_arg = Some(cur.slice(j, curlen).to_owned());
  313                             break;
  314                         } else {
  315                             last_valid_opt_id = None;
  316                         }
  317                       }
  318                     }
  319                     names.push(opt);
  320                     j = range.next;
  321                 }
  322             }
  323             let mut name_pos = 0;
  324             for nm in names.iter() {
  325                 name_pos += 1;
  326                 let optid = match find_opt(opts, (*nm).clone()) {
  327                   Some(id) => id,
  328                   None => return Err(UnrecognizedOption(name_str(nm)))
  329                 };
  330                 match opts[optid].hasarg {
  331                   No => {
  332                     if !i_arg.is_none() {
  333                         return Err(UnexpectedArgument(name_str(nm)));
  334                     }
  335                     vals[optid].push(Given);
  336                   }
  337                   Maybe => {
  338                     if !i_arg.is_none() {
  339                         vals[optid].push(Val((i_arg.clone()).unwrap()));
  340                     } else if name_pos < names.len() ||
  341                                   i + 1 == l || is_arg(args[i + 1]) {
  342                         vals[optid].push(Given);
  343                     } else { i += 1; vals[optid].push(Val(args[i].clone())); }
  344                   }
  345                   Yes => {
  346                     if !i_arg.is_none() {
  347                         vals[optid].push(Val(i_arg.clone().unwrap()));
  348                     } else if i + 1 == l {
  349                         return Err(ArgumentMissing(name_str(nm)));
  350                     } else { i += 1; vals[optid].push(Val(args[i].clone())); }
  351                   }
  352                 }
  353             }
  354         }
  355         i += 1;
  356     }
  357     i = 0u;
  358     while i < n_opts {
  359         let n = vals[i].len();
  360         let occ = opts[i].occur;
  361         if occ == Req {
  362             if n == 0 {
  363                 return Err(OptionMissing(name_str(&(opts[i].name))));
  364             }
  365         }
  366         if occ != Multi {
  367             if n > 1 {
  368                 return Err(OptionDuplicated(name_str(&(opts[i].name))));
  369             }
  370         }
  371         i += 1;
  372     }
  373     return Ok(Matches {opts: opts.to_owned(),
  374                vals: vals,
  375                free: free});
  376 }

  730     /** Splits a string into substrings with possibly internal whitespace,
  731      *  each of them at most `lim` bytes long. The substrings have leading and trailing
  732      *  whitespace removed, and are only cut at whitespace boundaries.
  733      *
  734      *  Note: Function was moved here from `std::str` because this module is the only place that
  735      *  uses it, and because it was to specific for a general string function.
  736      *
  737      *  #Failure:
  738      *
  739      *  Fails during iteration if the string contains a non-whitespace
  740      *  sequence longer than the limit.
  741      */
  742     fn each_split_within<'a>(ss: &'a str,
  743                              lim: uint,
  744                              it: &fn(&'a str) -> bool) -> bool {
  745         // Just for fun, let's write this as an state machine:
  746 
  747         enum SplitWithinState {
  748             A,  // leading whitespace, initial state
  749             B,  // words
  750             C,  // internal and trailing whitespace
  751         }
  752         enum Whitespace {
  753             Ws, // current char is whitespace
  754             Cr  // current char is not whitespace
  755         }
  756         enum LengthLimit {
  757             UnderLim, // current char makes current substring still fit in limit
  758             OverLim   // current char makes current substring no longer fit in limit
  759         }
  760 
  761         let mut slice_start = 0;
  762         let mut last_start = 0;
  763         let mut last_end = 0;
  764         let mut state = A;
  765         let mut fake_i = ss.len();
  766         let mut lim = lim;
  767 
  768         let mut cont = true;
  769         let slice: &fn() = || { cont = it(ss.slice(slice_start, last_end)) };
  770 
  771         // if the limit is larger than the string, lower it to save cycles
  772         if (lim >= fake_i) {
  773             lim = fake_i;
  774         }
  775 
  776         let machine: &fn((uint, char)) -> bool = |(i, c)| {
  777             let whitespace = if ::std::char::is_whitespace(c) { Ws }       else { Cr };
  778             let limit      = if (i - slice_start + 1) <= lim  { UnderLim } else { OverLim };
  779 
  780             state = match (state, whitespace, limit) {
  781                 (A, Ws, _)        => { A }
  782                 (A, Cr, _)        => { slice_start = i; last_start = i; B }
  783 
  784                 (B, Cr, UnderLim) => { B }
  785                 (B, Cr, OverLim)  if (i - last_start + 1) > lim
  786                                 => fail!("word starting with %? longer than limit!",
  787                                         ss.slice(last_start, i + 1)),
  788                 (B, Cr, OverLim)  => { slice(); slice_start = last_start; B }
  789                 (B, Ws, UnderLim) => { last_end = i; C }
  790                 (B, Ws, OverLim)  => { last_end = i; slice(); A }
  791 
  792                 (C, Cr, UnderLim) => { last_start = i; B }
  793                 (C, Cr, OverLim)  => { slice(); slice_start = i; last_start = i; last_end = i; B }
  794                 (C, Ws, OverLim)  => { slice(); A }
  795                 (C, Ws, UnderLim) => { C }
  796             };
  797 
  798             cont
  799         };
  800 
  801         ss.iter().enumerate().advance(|x| machine(x));
  802 
  803         // Let the automaton 'run out' by supplying trailing whitespace
  804         while cont && match state { B | C => true, A => false } {
  805             machine((fake_i, ' '));
  806             fake_i += 1;
  807         }
  808         return cont;
  809     }

   57  fn escape_str(s: &str) -> ~str {
   58      let mut escaped = ~"\"";
   59      for c in s.iter() {
   60          match c {
   61            '"' => escaped.push_str("\\\""),
   62            '\\' => escaped.push_str("\\\\"),
   63            '\x08' => escaped.push_str("\\b"),
   64            '\x0c' => escaped.push_str("\\f"),
   65            '\n' => escaped.push_str("\\n"),
   66            '\r' => escaped.push_str("\\r"),
   67            '\t' => escaped.push_str("\\t"),
   68            _ => escaped.push_char(c),
   69          }
   70      };
   71  
   72      escaped.push_char('"');
   73  
   74      escaped
   75  }

   77  fn spaces(n: uint) -> ~str {
   78      let mut ss = ~"";
   79      do n.times {
   80          ss.push_str(" ");
   81      }
   82      return ss;
   83  }

  586

  614      fn parse_integer(&mut self) -> Result<float, Error> {
  615          let mut res = 0f;
  616  
  617          match self.ch {
  618            '0' => {
  619              self.bump();
  620  
  621              // There can be only one leading '0'.
  622              match self.ch {
  623                '0' .. '9' => return self.error(~"invalid number"),
  624                _ => ()
  625              }
  626            }
  627            '1' .. '9' => {

  629                  match self.ch {
  630                    '0' .. '9' => {
  631                      res *= 10f;
  632                      res += ((self.ch as int) - ('0' as int)) as float;
  633  
  634                      self.bump();
  635                    }
  636                    _ => break
  637                  }
  638              }
  639            }
  640            _ => return self.error(~"invalid number")
  641          }

  643          Ok(res)
  644      }

  646      fn parse_decimal(&mut self, res: float) -> Result<float, Error> {
  647          self.bump();
  648  
  649          // Make sure a digit follows the decimal place.
  650          match self.ch {
  651            '0' .. '9' => (),
  652            _ => return self.error(~"invalid number")
  653          }
  654  
  655          let mut res = res;
  656          let mut dec = 1f;
  657          while !self.eof() {
  658              match self.ch {
  659                '0' .. '9' => {
  660                  dec /= 10f;
  661                  res += (((self.ch as int) - ('0' as int)) as float) * dec;
  662  
  663                  self.bump();
  664                }
  665                _ => break
  666              }
  667          }
  668  
  669          Ok(res)
  670      }

  672      fn parse_exponent(&mut self, mut res: float) -> Result<float, Error> {
  673          self.bump();
  674  
  675          let mut exp = 0u;
  676          let mut neg_exp = false;
  677  
  678          match self.ch {
  679            '+' => self.bump(),
  680            '-' => { self.bump(); neg_exp = true; }
  681            _ => ()
  682          }
  683  
  684          // Make sure a digit follows the exponent place.
  685          match self.ch {
  686            '0' .. '9' => (),
  687            _ => return self.error(~"invalid number")
  688          }
  689  
  690          while !self.eof() {
  691              match self.ch {
  692                '0' .. '9' => {
  693                  exp *= 10u;
  694                  exp += (self.ch as uint) - ('0' as uint);
  695  
  696                  self.bump();
  697                }
  698                _ => break
  699              }
  700          }
  701  
  702          let exp = float::pow_with_uint(10u, exp);
  703          if neg_exp {
  704              res /= exp;
  705          } else {
  706              res *= exp;
  707          }
  708  
  709          Ok(res)
  710      }

  712      fn parse_str(&mut self) -> Result<~str, Error> {
  713          let mut escape = false;
  714          let mut res = ~"";
  715  
  716          loop {
  717              self.bump();
  718              if self.eof() {
  719                  return self.error(~"EOF while parsing string");
  720              }
  721  
  722              if (escape) {
  723                  match self.ch {
  724                    '"' => res.push_char('"'),
  725                    '\\' => res.push_char('\\'),
  726                    '/' => res.push_char('/'),
  727                    'b' => res.push_char('\x08'),
  728                    'f' => res.push_char('\x0c'),
  729                    'n' => res.push_char('\n'),
  730                    'r' => res.push_char('\r'),
  731                    't' => res.push_char('\t'),
  732                    'u' => {
  733                        // Parse \u1234.
  734                        let mut i = 0u;
  735                        let mut n = 0u;
  736                        while i < 4u {
  737                            match self.next_char() {
  738                              '0' .. '9' => {
  739                                n = n * 16u + (self.ch as uint)
  740                                            - ('0'     as uint);
  741                              },
  742                              'a' | 'A' => n = n * 16u + 10u,
  743                              'b' | 'B' => n = n * 16u + 11u,
  744                              'c' | 'C' => n = n * 16u + 12u,
  745                              'd' | 'D' => n = n * 16u + 13u,
  746                              'e' | 'E' => n = n * 16u + 14u,
  747                              'f' | 'F' => n = n * 16u + 15u,
  748                              _ => return self.error(
  749                                     ~"invalid \\u escape (unrecognized hex)")
  750                            }
  751                            i += 1u;
  752                        }
  753  
  754                        // Error out if we didn't parse 4 digits.
  755                        if i != 4u {
  756                            return self.error(
  757                              ~"invalid \\u escape (not four digits)");
  758                        }
  759  
  760                        res.push_char(n as char);
  761                    }
  762                    _ => return self.error(~"invalid escape")
  763                  }
  764                  escape = false;
  765              } else if self.ch == '\\' {
  766                  escape = true;
  767              } else {
  768                  if self.ch == '"' {
  769                      self.bump();
  770                      return Ok(res);
  771                  }
  772                  res.push_char(self.ch);
  773              }
  774          }
  775      }

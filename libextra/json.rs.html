   58 fn escape_str(s: &str) -> ~str {
   59     let mut escaped = ~"\"";
   60     for c in s.iter() {
   61         match c {
   62           '"' => escaped.push_str("\\\""),
   63           '\\' => escaped.push_str("\\\\"),
   64           '\x08' => escaped.push_str("\\b"),
   65           '\x0c' => escaped.push_str("\\f"),
   66           '\n' => escaped.push_str("\\n"),
   67           '\r' => escaped.push_str("\\r"),
   68           '\t' => escaped.push_str("\\t"),
   69           _ => escaped.push_char(c),
   70         }
   71     };
   72 
   73     escaped.push_char('"');
   74 
   75     escaped
   76 }

   78 fn spaces(n: uint) -> ~str {
   79     let mut ss = ~"";
   80     do n.times {
   81         ss.push_str(" ");
   82     }
   83     return ss;
   84 }

  615     fn parse_integer(&mut self) -> Result<float, Error> {
  616         let mut res = 0f;
  617 
  618         match self.ch {
  619           '0' => {
  620             self.bump();

  623             match self.ch {
  624               '0' .. '9' => return self.error(~"invalid number"),

  627           }
  628           '1' .. '9' => {

  638                 }
  639             }
  640           }

  647     fn parse_decimal(&mut self, res: float) -> Result<float, Error> {

  649 
  650         // Make sure a digit follows the decimal place.
  651         match self.ch {
  652           '0' .. '9' => (),

  654         }
  655

  657         let mut dec = 1f;

  659             match self.ch {

  661                 dec /= 10f;
  662                 res += (((self.ch as int) - ('0' as int)) as float) * dec;
  663 
  664                 self.bump();
  665               }

  667             }
  668         }

  673     fn parse_exponent(&mut self, mut res: float) -> Result<float, Error> {
  674         self.bump();
  675 
  676         let mut exp = 0u;
  677         let mut neg_exp = false;
  678 
  679         match self.ch {
  680           '+' => self.bump(),

  682           _ => ()
  683         }
  684 
  685         // Make sure a digit follows the exponent place.
  686         match self.ch {
  687           '0' .. '9' => (),
  688           _ => return self.error(~"invalid number")
  689         }
  690 
  691         while !self.eof() {
  692             match self.ch {
  693               '0' .. '9' => {
  694                 exp *= 10u;
  695                 exp += (self.ch as uint) - ('0' as uint);
  696 
  697                 self.bump();
  698               }
  699               _ => break
  700             }
  701         }
  702 
  703         let exp = float::pow_with_uint(10u, exp);
  704         if neg_exp {
  705             res /= exp;
  706         } else {
  707             res *= exp;
  708         }
  709 
  710         Ok(res)
  711     }

  713     fn parse_str(&mut self) -> Result<~str, Error> {
  714         let mut escape = false;
  715         let mut res = ~"";
  716 
  717         loop {
  718             self.bump();
  719             if self.eof() {
  720                 return self.error(~"EOF while parsing string");
  721             }
  722 
  723             if (escape) {
  724                 match self.ch {
  725                   '"' => res.push_char('"'),
  726                   '\\' => res.push_char('\\'),
  727                   '/' => res.push_char('/'),
  728                   'b' => res.push_char('\x08'),
  729                   'f' => res.push_char('\x0c'),
  730                   'n' => res.push_char('\n'),
  731                   'r' => res.push_char('\r'),
  732                   't' => res.push_char('\t'),
  733                   'u' => {
  734                       // Parse \u1234.
  735                       let mut i = 0u;
  736                       let mut n = 0u;
  737                       while i < 4u {
  738                           match self.next_char() {
  739                             '0' .. '9' => {
  740                               n = n * 16u + (self.ch as uint)
  741                                           - ('0'     as uint);
  742                             },
  743                             'a' | 'A' => n = n * 16u + 10u,
  744                             'b' | 'B' => n = n * 16u + 11u,
  745                             'c' | 'C' => n = n * 16u + 12u,
  746                             'd' | 'D' => n = n * 16u + 13u,
  747                             'e' | 'E' => n = n * 16u + 14u,
  748                             'f' | 'F' => n = n * 16u + 15u,
  749                             _ => return self.error(
  750                                    ~"invalid \\u escape (unrecognized hex)")
  751                           }
  752                           i += 1u;
  753                       }
  754 
  755                       // Error out if we didn't parse 4 digits.
  756                       if i != 4u {
  757                           return self.error(
  758                             ~"invalid \\u escape (not four digits)");
  759                       }
  760 
  761                       res.push_char(n as char);
  762                   }
  763                   _ => return self.error(~"invalid escape")
  764                 }
  765                 escape = false;
  766             } else if self.ch == '\\' {
  767                 escape = true;
  768             } else {
  769                 if self.ch == '"' {
  770                     self.bump();
  771                     return Ok(res);
  772                 }
  773                 res.push_char(self.ch);
  774             }
  775         }
  776     }

   70  fn encode_inner(s: &str, full_url: bool) -> ~str {
   71      do io::with_str_reader(s) |rdr| {
   72          let mut out = ~"";
   73  
   74          while !rdr.eof() {
   75              let ch = rdr.read_byte() as u8 as char;
   76              match ch {
   77                // unreserved:
   78                'A' .. 'Z' |
   79                'a' .. 'z' |
   80                '0' .. '9' |
   81                '-' | '.' | '_' | '~' => {
   82                  out.push_char(ch);
   83                }
   84                _ => {
   85                    if full_url {
   86                      match ch {
   87                        // gen-delims:
   88                        ':' | '/' | '?' | '#' | '[' | ']' | '@' |
   89  
   90                        // sub-delims:
   91                        '!' | '$' | '&' | '"' | '(' | ')' | '*' |
   92                        '+' | ',' | ';' | '=' => {
   93                          out.push_char(ch);
   94                        }
   95  
   96                        _ => out.push_str(fmt!("%%%X", ch as uint))
   97                      }
   98                  } else {
   99                      out.push_str(fmt!("%%%X", ch as uint));
  100                  }
  101                }
  102              }
  103          }
  104  
  105          out
  106      }
  107  }

  130  fn decode_inner(s: &str, full_url: bool) -> ~str {
  131      do io::with_str_reader(s) |rdr| {
  132          let mut out = ~"";
  133  
  134          while !rdr.eof() {
  135              match rdr.read_char() {
  136                '%' => {
  137                  let bytes = rdr.read_bytes(2u);
  138                  let ch = uint::parse_bytes(bytes, 16u).unwrap() as u8 as char;
  139  
  140                  if full_url {
  141                      // Only decode some characters:
  142                      match ch {
  143                        // gen-delims:
  144                        ':' | '/' | '?' | '#' | '[' | ']' | '@' |
  145  
  146                        // sub-delims:
  147                        '!' | '$' | '&' | '"' | '(' | ')' | '*' |
  148                        '+' | ',' | ';' | '=' => {
  149                          out.push_char('%');
  150                          out.push_char(bytes[0u] as char);
  151                          out.push_char(bytes[1u] as char);
  152                        }
  153  
  154                        ch => out.push_char(ch)
  155                      }
  156                  } else {
  157                        out.push_char(ch);
  158                  }
  159                }
  160                ch => out.push_char(ch)
  161              }
  162          }
  163  
  164          out
  165      }
  166  }

  203  /**
  204   * Encode a hashmap to the 'application/x-www-form-urlencoded' media type.
  205   */
  206  pub fn encode_form_urlencoded(m: &HashMap<~str, ~[~str]>) -> ~str {
  207      let mut out = ~"";
  208      let mut first = true;
  209  
  210      for (key, values) in m.iter() {
  211          let key = encode_plus(*key);
  212  
  213          for value in values.iter() {
  214              if first {
  215                  first = false;
  216              } else {
  217                  out.push_char('&');
  218                  first = false;
  219              }
  220  
  221              out.push_str(fmt!("%s=%s", key, encode_plus(*value)));
  222          }
  223      }
  224  
  225      out
  226  }

  228  /**
  229   * Decode a string encoded with the 'application/x-www-form-urlencoded' media
  230   * type into a hashmap.
  231   */
  232  pub fn decode_form_urlencoded(s: &[u8]) -> HashMap<~str, ~[~str]> {
  233      do io::with_bytes_reader(s) |rdr| {
  234          let mut m = HashMap::new();
  235          let mut key = ~"";
  236          let mut value = ~"";
  237          let mut parsing_key = true;
  238  
  239          while !rdr.eof() {
  240              match rdr.read_char() {
  241                  '&' | ';' => {
  242                      if key != ~"" && value != ~"" {
  243                          let mut values = match m.pop(&key) {
  244                              Some(values) => values,
  245                              None => ~[],
  246                          };
  247  
  248                          values.push(value);
  249                          m.insert(key, values);
  250                      }
  251  
  252                      parsing_key = true;
  253                      key = ~"";
  254                      value = ~"";
  255                  }
  256                  '=' => parsing_key = false,
  257                  ch => {
  258                      let ch = match ch {
  259                          '%' => {
  260                              let bytes = rdr.read_bytes(2u);
  261                              uint::parse_bytes(bytes, 16u).unwrap() as u8 as char
  262                          }
  263                          '+' => ' ',
  264                          ch => ch
  265                      };
  266  
  267                      if parsing_key {
  268                          key.push_char(ch)
  269                      } else {
  270                          value.push_char(ch)
  271                      }
  272                  }
  273              }
  274          }
  275  
  276          if key != ~"" && value != ~"" {
  277              let mut values = match m.pop(&key) {
  278                  Some(values) => values,
  279                  None => ~[],
  280              };
  281  
  282              values.push(value);
  283              m.insert(key, values);
  284          }
  285  
  286          m
  287      }
  288  }

  291  fn split_char_first(s: &str, c: char) -> (~str, ~str) {
  292      let len = s.len();
  293      let mut index = len;
  294      let mut mat = 0;
  295      do io::with_str_reader(s) |rdr| {
  296          let mut ch;
  297          while !rdr.eof() {
  298              ch = rdr.read_byte() as u8 as char;
  299              if ch == c {
  300                  // found a match, adjust markers
  301                  index = rdr.tell()-1;
  302                  mat = 1;
  303                  break;
  304              }
  305          }
  306      }
  307      if index+mat == len {
  308          return (s.slice(0, index).to_owned(), ~"");
  309      } else {
  310          return (s.slice(0, index).to_owned(),
  311               s.slice(index + mat, s.len()).to_owned());
  312      }
  313  }

  358  // returns the scheme and the rest of the url, or a parsing error
  359  pub fn get_scheme(rawurl: &str) -> Result<(~str, ~str), ~str> {
  360      for (i,c) in rawurl.iter().enumerate() {
  361          match c {
  362            'A' .. 'Z' | 'a' .. 'z' => loop,
  363            '0' .. '9' | '+' | '-' | '.' => {
  364              if i == 0 {
  365                  return Err(~"url: Scheme must begin with a letter.");
  366              }
  367              loop;
  368            }
  369            ':' => {
  370              if i == 0 {
  371                  return Err(~"url: Scheme cannot be empty.");
  372              } else {
  373                  return Ok((rawurl.slice(0,i).to_owned(),
  374                                  rawurl.slice(i+1,rawurl.len()).to_owned()));
  375              }
  376            }
  377            _ => {
  378              return Err(~"url: Invalid character in scheme.");
  379            }
  380          }
  381      };
  382      return Err(~"url: Scheme must be terminated with a colon.");
  383  }

  392  // returns userinfo, host, port, and unparsed part, or an error
  393  fn get_authority(rawurl: &str) ->
  394      Result<(Option<UserInfo>, ~str, Option<~str>, ~str), ~str> {
  395      if !rawurl.starts_with("//") {
  396          // there is no authority.
  397          return Ok((None, ~"", None, rawurl.to_str()));
  398      }
  399  
  400      enum State {
  401          Start, // starting state
  402          PassHostPort, // could be in user or port
  403          Ip6Port, // either in ipv6 host or port
  404          Ip6Host, // are in an ipv6 host
  405          InHost, // are in a host - may be ipv6, but don't know yet
  406          InPort // are in port
  407      }
  408  
  409      let len = rawurl.len();
  410      let mut st = Start;
  411      let mut input = Digit; // most restricted, start here.
  412  
  413      let mut userinfo = None;
  414      let mut host = ~"";
  415      let mut port = None;
  416  
  417      let mut colon_count = 0;
  418      let mut pos = 0;
  419      let mut begin = 2;
  420      let mut end = len;
  421  
  422      for (i,c) in rawurl.iter().enumerate() {
  423          if i < 2 { loop; } // ignore the leading //
  424  
  425          // deal with input class first
  426          match c {
  427            '0' .. '9' => (),
  428            'A' .. 'F' | 'a' .. 'f' => {
  429              if input == Digit {
  430                  input = Hex;
  431              }
  432            }
  433            'G' .. 'Z' | 'g' .. 'z' | '-' | '.' | '_' | '~' | '%' |
  434            '&' |'\'' | '(' | ')' | '+' | '!' | '*' | ',' | ';' | '=' => {
  435              input = Unreserved;
  436            }
  437            ':' | '@' | '?' | '#' | '/' => {
  438              // separators, don't change anything
  439            }
  440            _ => {
  441              return Err(~"Illegal character in authority");
  442            }
  443          }
  444  
  445          // now process states
  446          match c {
  447            ':' => {
  448              colon_count += 1;
  449              match st {
  450                Start => {
  451                  pos = i;
  452                  st = PassHostPort;
  453                }
  454                PassHostPort => {
  455                  // multiple colons means ipv6 address.
  456                  if input == Unreserved {
  457                      return Err(
  458                          ~"Illegal characters in IPv6 address.");
  459                  }
  460                  st = Ip6Host;
  461                }
  462                InHost => {
  463                  pos = i;
  464                  if input == Unreserved {
  465                      // must be port
  466                      host = rawurl.slice(begin, i).to_owned();
  467                      st = InPort;
  468                  } else {
  469                      // can't be sure whether this is an ipv6 address or a port
  470                      st = Ip6Port;
  471                  }
  472                }
  473                Ip6Port => {
  474                  if input == Unreserved {
  475                      return Err(~"Illegal characters in authority.");
  476                  }
  477                  st = Ip6Host;
  478                }
  479                Ip6Host => {
  480                  if colon_count > 7 {
  481                      host = rawurl.slice(begin, i).to_owned();
  482                      pos = i;
  483                      st = InPort;
  484                  }
  485                }
  486                _ => {
  487                  return Err(~"Invalid ':' in authority.");
  488                }
  489              }
  490              input = Digit; // reset input class
  491            }
  492  
  493            '@' => {
  494              input = Digit; // reset input class
  495              colon_count = 0; // reset count
  496              match st {
  497                Start => {
  498                  let user = rawurl.slice(begin, i).to_owned();
  499                  userinfo = Some(UserInfo::new(user, None));
  500                  st = InHost;
  501                }
  502                PassHostPort => {
  503                  let user = rawurl.slice(begin, pos).to_owned();
  504                  let pass = rawurl.slice(pos+1, i).to_owned();
  505                  userinfo = Some(UserInfo::new(user, Some(pass)));
  506                  st = InHost;
  507                }
  508                _ => {
  509                  return Err(~"Invalid '@' in authority.");
  510                }
  511              }
  512              begin = i+1;
  513            }
  514  
  515            '?' | '#' | '/' => {
  516              end = i;
  517              break;
  518            }
  519            _ => ()
  520          }
  521      }
  522  
  523      // finish up
  524      match st {
  525        Start => {
  526          host = rawurl.slice(begin, end).to_owned();
  527        }
  528        PassHostPort | Ip6Port => {
  529          if input != Digit {
  530              return Err(~"Non-digit characters in port.");
  531          }
  532          host = rawurl.slice(begin, pos).to_owned();
  533          port = Some(rawurl.slice(pos+1, end).to_owned());
  534        }
  535        Ip6Host | InHost => {
  536          host = rawurl.slice(begin, end).to_owned();
  537        }
  538        InPort => {
  539          if input != Digit {
  540              return Err(~"Non-digit characters in port.");
  541          }
  542          port = Some(rawurl.slice(pos+1, end).to_owned());
  543        }
  544      }
  545  
  546      let rest = rawurl.slice(end, len).to_owned();
  547      return Ok((userinfo, host, port, rest));
  548  }
  549  
  550  
  551  // returns the path and unparsed part of url, or an error
  552  fn get_path(rawurl: &str, authority: bool) ->
  553      Result<(~str, ~str), ~str> {
  554      let len = rawurl.len();
  555      let mut end = len;
  556      for (i,c) in rawurl.iter().enumerate() {
  557          match c {
  558            'A' .. 'Z' | 'a' .. 'z' | '0' .. '9' | '&' |'\'' | '(' | ')' | '.'
  559            | '@' | ':' | '%' | '/' | '+' | '!' | '*' | ',' | ';' | '='
  560            | '_' | '-' => {
  561              loop;
  562            }
  563            '?' | '#' => {
  564              end = i;
  565              break;
  566            }
  567            _ => return Err(~"Invalid character in path.")
  568          }
  569      }
  570  
  571      if authority {
  572          if end != 0 && !rawurl.starts_with("/") {
  573              return Err(~"Non-empty path must begin with\
  574                                 '/' in presence of authority.");
  575          }
  576      }
  577  
  578      return Ok((decode_component(rawurl.slice(0, end)),
  579                      rawurl.slice(end, len).to_owned()));
  580  }
  581  
  582  // returns the parsed query and the fragment, if present
  583  fn get_query_fragment(rawurl: &str) ->
  584      Result<(Query, Option<~str>), ~str> {
  585      if !rawurl.starts_with("?") {
  586          if rawurl.starts_with("#") {
  587              let f = decode_component(rawurl.slice(
  588                                                  1,
  589                                                  rawurl.len()));
  590              return Ok((~[], Some(f)));
  591          } else {
  592              return Ok((~[], None));
  593          }
  594      }
  595      let (q, r) = split_char_first(rawurl.slice(1, rawurl.len()), '#');
  596      let f = if r.len() != 0 {
  597          Some(decode_component(r)) } else { None };
  598      return Ok((query_from_str(q), f));
  599  }

  601  /**
  602   * Parse a `str` to a `url`
  603   *
  604   * # Arguments
  605   *
  606   * `rawurl` - a string representing a full url, including scheme.
  607   *
  608   * # Returns
  609   *
  610   * a `url` that contains the parsed representation of the url.
  611   *
  612   */
  613  
  614  pub fn from_str(rawurl: &str) -> Result<Url, ~str> {
  615      // scheme
  616      let (scheme, rest) = match get_scheme(rawurl) {
  617          Ok(val) => val,
  618          Err(e) => return Err(e),
  619      };
  620  
  621      // authority
  622      let (userinfo, host, port, rest) = match get_authority(rest) {
  623          Ok(val) => val,
  624          Err(e) => return Err(e),
  625      };
  626  
  627      // path
  628      let has_authority = if host == ~"" { false } else { true };
  629      let (path, rest) = match get_path(rest, has_authority) {
  630          Ok(val) => val,
  631          Err(e) => return Err(e),
  632      };
  633  
  634      // query and fragment
  635      let (query, fragment) = match get_query_fragment(rest) {
  636          Ok(val) => val,
  637          Err(e) => return Err(e),
  638      };
  639  
  640      Ok(Url::new(scheme, userinfo, host, port, path, query, fragment))
  641  }

  643  impl FromStr for Url {
  644      fn from_str(s: &str) -> Option<Url> {
  645          match from_str(s) {
  646              Ok(url) => Some(url),
  647              Err(_) => None
  648          }
  649      }
  650  }
  651  
  652  /**
  653   * Format a `url` as a string
  654   *
  655   * # Arguments
  656   *
  657   * `url` - a url.
  658   *
  659   * # Returns
  660   *
  661   * a `str` that contains the formatted url. Note that this will usually
  662   * be an inverse of `from_str` but might strip out unneeded separators.
  663   * for example, "http://somehost.com?", when parsed and formatted, will
  664   * result in just "http://somehost.com".
  665   *
  666   */
  667  pub fn to_str(url: &Url) -> ~str {
  668      let user = match url.user {
  669          Some(ref user) => userinfo_to_str(user),
  670          None => ~"",
  671      };
  672  
  673      let authority = if url.host.is_empty() {
  674          ~""
  675      } else {
  676          fmt!("//%s%s", user, url.host)
  677      };
  678  
  679      let query = if url.query.is_empty() {
  680          ~""
  681      } else {
  682          fmt!("?%s", query_to_str(&url.query))
  683      };
  684  
  685      let fragment = match url.fragment {
  686          Some(ref fragment) => fmt!("#%s", encode_component(*fragment)),
  687          None => ~"",
  688      };
  689  
  690      fmt!("%s:%s%s%s%s", url.scheme, authority, url.path, query, fragment)
  691  }
  692  
  693  impl ToStr for Url {
  694      fn to_str(&self) -> ~str {
  695          to_str(self)
  696      }
  697  }

  703  }

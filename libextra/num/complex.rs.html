  24 /// A complex number in Cartesian form.
  25 #[deriving(Eq,Clone)]
  26 pub struct Cmplx<T> {
  27     /// Real portion of the complex number
  28     re: T,
  29     /// Imaginary portion of the complex number
  30     im: T
  31 }
  32 
  33 pub type Complex = Cmplx<float>;
  34 pub type Complex32 = Cmplx<f32>;
  35 pub type Complex64 = Cmplx<f64>;

  37 impl<T: Clone + Num> Cmplx<T> {
  38     /// Create a new Cmplx
  39     #[inline]
  40     pub fn new(re: T, im: T) -> Cmplx<T> {
  41         Cmplx { re: re, im: im }
  42     }
  43 
  44     /**
  45     Returns the square of the norm (since `T` doesn't necessarily
  46     have a sqrt function), i.e. `re^2 + im^2`.
  47     */
  48     #[inline]
  49     pub fn norm_sqr(&self) -> T {
  50         self.re * self.re + self.im * self.im
  51     }
  52 
  53 
  54     /// Returns the complex conjugate. i.e. `re - i im`
  55     #[inline]
  56     pub fn conj(&self) -> Cmplx<T> {
  57         Cmplx::new(self.re.clone(), -self.im)
  58     }
  59 
  60 
  61     /// Multiplies `self` by the scalar `t`.
  62     #[inline]
  63     pub fn scale(&self, t: T) -> Cmplx<T> {
  64         Cmplx::new(self.re * t, self.im * t)
  65     }
  66 
  67     /// Divides `self` by the scalar `t`.
  68     #[inline]
  69     pub fn unscale(&self, t: T) -> Cmplx<T> {
  70         Cmplx::new(self.re / t, self.im / t)
  71     }
  72 
  73     /// Returns `1/self`
  74     #[inline]
  75     pub fn inv(&self) -> Cmplx<T> {
  76         let norm_sqr = self.norm_sqr();
  77         Cmplx::new(self.re / norm_sqr,
  78                     -self.im / norm_sqr)
  79     }
  80 }

  82 impl<T: Clone + Algebraic + Num> Cmplx<T> {
  83     /// Calculate |self|
  84     #[inline]
  85     pub fn norm(&self) -> T {
  86         self.re.hypot(&self.im)
  87     }
  88 }

  90 impl<T: Clone + Trigonometric + Algebraic + Num> Cmplx<T> {
  91     /// Calculate the principal Arg of self.
  92     #[inline]
  93     pub fn arg(&self) -> T {
  94         self.im.atan2(&self.re)
  95     }
  96     /// Convert to polar form (r, theta), such that `self = r * exp(i
  97     /// * theta)`
  98     #[inline]
  99     pub fn to_polar(&self) -> (T, T) {
 100         (self.norm(), self.arg())
 101     }
 102     /// Convert a polar representation into a complex number.
 103     #[inline]
 104     pub fn from_polar(r: &T, theta: &T) -> Cmplx<T> {
 105         Cmplx::new(r * theta.cos(), r * theta.sin())
 106     }
 107 }

 134 //   == [(a*c + b*d) / (c*c + d*d)] + i [(b*c - a*d) / (c*c + d*d)]
 135 impl<T: Clone + Num> Div<Cmplx<T>, Cmplx<T>> for Cmplx<T> {
 136     #[inline]
 137     fn div(&self, other: &Cmplx<T>) -> Cmplx<T> {
 138         let norm_sqr = other.norm_sqr();
 139         Cmplx::new((self.re*other.re + self.im*other.im) / norm_sqr,
 140                    (self.im*other.re - self.re*other.im) / norm_sqr)
 141     }
 142 }

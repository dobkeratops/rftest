  32  /**
  33   * Left fold
  34   *
  35   * Applies `f` to `u` and the first element in the list, then applies `f` to
  36   * the result of the previous call and the second element, and so on,
  37   * returning the accumulated result.
  38   *
  39   * # Arguments
  40   *
  41   * * ls - The list to fold
  42   * * z - The initial value
  43   * * f - The function to apply

  46      let mut accum: T = z;
  47      do iter(ls) |elt| { accum = f(&accum, elt);}
  48      accum

  51  /**
  52   * Search for an element that matches a given predicate
  53   *
  54   * Apply function `f` to each element of `v`, starting from the first.
  55   * When function `f` returns true then an option containing the element
  56   * is returned. If `f` matches no elements then none is returned.

  63              if f(hd) { return Some((*hd).clone()); }

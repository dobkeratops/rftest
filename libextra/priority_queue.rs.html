 129      // The implementations of siftup and siftdown use unsafe blocks in
 130      // order to move an element out of the vector (leaving behind a
 131      // zeroed element), shift along the others and move it back into the
 132      // vector over the junk element.  This reduces the constant factor
 133      // compared to using swaps, which involves twice as many moves.
 134      fn siftup(&mut self, start: uint, mut pos: uint) {
 135          unsafe {
 136              let new = replace(&mut self.data[pos], init());
 137  
 138              while pos > start {
 139                  let parent = (pos - 1) >> 1;
 140                  if new > self.data[parent] {
 141                      let x = replace(&mut self.data[parent], init());
 142                      move_val_init(&mut self.data[pos], x);
 143                      pos = parent;
 144                      loop
 145                  }
 146                  break
 147              }
 148              move_val_init(&mut self.data[pos], new);
 149          }
 150      }
 151  
 152      fn siftdown_range(&mut self, mut pos: uint, end: uint) {
 153          unsafe {
 154              let start = pos;
 155              let new = replace(&mut self.data[pos], init());
 156  
 157              let mut child = 2 * pos + 1;
 158              while child < end {
 159                  let right = child + 1;
 160                  if right < end && !(self.data[child] > self.data[right]) {
 161                      child = right;
 162                  }
 163                  let x = replace(&mut self.data[child], init());
 164                  move_val_init(&mut self.data[pos], x);
 165                  pos = child;
 166                  child = 2 * pos + 1;
 167              }
 168  
 169              move_val_init(&mut self.data[pos], new);
 170              self.siftup(start, pos);
 171          }
 172      }

 128      // The implementations of siftup and siftdown use unsafe blocks in
 129      // order to move an element out of the vector (leaving behind a
 130      // zeroed element), shift along the others and move it back into the
 131      // vector over the junk element.  This reduces the constant factor
 132      // compared to using swaps, which involves twice as many moves.
 133      fn siftup(&mut self, start: uint, mut pos: uint) {
 134          unsafe {
 135              let new = replace(&mut self.data[pos], init());
 136  
 137              while pos > start {
 138                  let parent = (pos - 1) >> 1;
 139                  if new > self.data[parent] {
 140                      let x = replace(&mut self.data[parent], init());
 141                      move_val_init(&mut self.data[pos], x);
 142                      pos = parent;
 143                      loop
 144                  }
 145                  break
 146              }
 147              move_val_init(&mut self.data[pos], new);
 148          }
 149      }
 150  
 151      fn siftdown_range(&mut self, mut pos: uint, end: uint) {
 152          unsafe {
 153              let start = pos;
 154              let new = replace(&mut self.data[pos], init());
 155  
 156              let mut child = 2 * pos + 1;
 157              while child < end {
 158                  let right = child + 1;
 159                  if right < end && !(self.data[child] > self.data[right]) {
 160                      child = right;
 161                  }
 162                  let x = replace(&mut self.data[child], init());
 163                  move_val_init(&mut self.data[pos], x);
 164                  pos = child;
 165                  child = 2 * pos + 1;
 166              }
 167  
 168              move_val_init(&mut self.data[pos], new);
 169              self.siftup(start, pos);
 170          }
 171      }

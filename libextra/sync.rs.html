   34  // Each waiting task receives on one of these.
   35  #[doc(hidden)]
   36  type WaitEnd = comm::PortOne<()>;
   37  #[doc(hidden)]
   38  type SignalEnd = comm::ChanOne<()>;
   39  // A doubly-ended queue of waiting tasks.
   40  #[doc(hidden)]
   41  struct WaitQueue { head: comm::Port<SignalEnd>,
   42                     tail: comm::Chan<SignalEnd> }
   43  
   44  impl WaitQueue {
   45      fn new() -> WaitQueue {
   46          let (block_head, block_tail) = comm::stream();
   47          WaitQueue { head: block_head, tail: block_tail }
   48      }
   49  
   50      // Signals one live task from the queue.
   51      fn signal(&self) -> bool {
   52          // The peek is mandatory to make sure recv doesn't block.
   53          if self.head.peek() {
   54              // Pop and send a wakeup signal. If the waiter was killed, its port
   55              // will have closed. Keep trying until we get a live task.
   56              if self.head.recv().try_send_deferred(()) {
   57                  true
   58              } else {
   59                  self.signal()
   60              }
   61          } else {
   62              false
   63          }
   64      }
   65  
   66      fn broadcast(&self) -> uint {
   67          let mut count = 0;
   68          while self.head.peek() {
   69              if self.head.recv().try_send_deferred(()) {
   70                  count += 1;
   71              }
   72          }
   73          count
   74      }
   75  
   76      fn wait_end(&self) -> WaitEnd {
   77          let (wait_end, signal_end) = comm::oneshot();
   78          self.tail.send_deferred(signal_end);
   79          wait_end
   80      }
   81  }

  103      pub fn acquire(&self) {
  104          unsafe {
  105              let mut waiter_nobe = None;
  106              do (**self).with |state| {
  107                  state.count -= 1;
  108                  if state.count < 0 {
  109                      // Create waiter nobe, enqueue ourself, and tell
  110                      // outer scope we need to block.
  111                      waiter_nobe = Some(state.waiters.wait_end());
  112                  }
  113              }
  114              // Uncomment if you wish to test for sem races. Not valgrind-friendly.
  115              /* do 1000.times { task::yield(); } */
  116              // Need to wait outside the exclusive.
  117              if waiter_nobe.is_some() {
  118                  let _ = waiter_nobe.unwrap().recv();
  119              }
  120          }
  121      }
  122  
  123      pub fn release(&self) {
  124          unsafe {
  125              do (**self).with |state| {
  126                  state.count += 1;
  127                  if state.count <= 0 {
  128                      state.waiters.signal();
  129                  }
  130              }
  131          }
  132      }
  133  
  134      pub fn access<U>(&self, blk: &fn() -> U) -> U {
  135          do task::unkillable {
  136              do (|| {
  137                  self.acquire();
  138                  unsafe {
  139                      do task::rekillable { blk() }
  140                  }
  141              }).finally {
  142                  self.release();
  143              }
  144          }
  145      }
  146  }

  153          do num_condvars.times {

  157      }
  158  }

  168  /// A mechanism for atomic-unlock-and-deschedule blocking and signalling.
  169  pub struct Condvar<'self> {
  170      // The 'Sem' object associated with this condvar. This is the one that's
  171      // atomically-unlocked-and-descheduled upon and reacquired during wakeup.
  172      priv sem: &'self Sem<~[WaitQueue]>,
  173      // This is (can be) an extra semaphore which is held around the reacquire
  174      // operation on the first one. This is only used in cvars associated with
  175      // rwlocks, and is needed to ensure that, when a downgrader is trying to
  176      // hand off the access lock (which would be the first field, here), a 2nd
  177      // writer waking up from a cvar wait can't race with a reader to steal it,
  178      // See the comment in write_cond for more detail.
  179      priv order: ReacquireOrderLock<'self>,
  180      // Make sure condvars are non-copyable.
  181      priv token: util::NonCopyable,
  182  }
  183  
  184  impl<'self> Condvar<'self> {
  185      /**
  186       * Atomically drop the associated lock, and block until a signal is sent.
  187       *
  188       * # Failure
  189       * A task which is killed (i.e., by linked failure with another task)
  190       * while waiting on a condition variable will wake up, fail, and unlock
  191       * the associated lock as it unwinds.
  192       */
  193      pub fn wait(&self) { self.wait_on(0) }
  194  
  195      /**
  196       * As wait(), but can specify which of multiple condition variables to
  197       * wait on. Only a signal_on() or broadcast_on() with the same condvar_id
  198       * will wake this thread.
  199       *
  200       * The associated lock must have been initialised with an appropriate
  201       * number of condvars. The condvar_id must be between 0 and num_condvars-1
  202       * or else this call will fail.
  203       *
  204       * wait() is equivalent to wait_on(0).
  205       */
  206      pub fn wait_on(&self, condvar_id: uint) {
  207          let mut WaitEnd = None;
  208          let mut out_of_bounds = None;
  209          do task::unkillable {
  210              // Release lock, 'atomically' enqueuing ourselves in so doing.
  211              unsafe {
  212                  do (**self.sem).with |state| {
  213                      if condvar_id < state.blocked.len() {
  214                          // Drop the lock.
  215                          state.count += 1;
  216                          if state.count <= 0 {
  217                              state.waiters.signal();
  218                          }
  219                          // Create waiter nobe, and enqueue ourself to
  220                          // be woken up by a signaller.
  221                          WaitEnd = Some(state.blocked[condvar_id].wait_end());
  222                      } else {
  223                          out_of_bounds = Some(state.blocked.len());
  224                      }
  225                  }
  226              }
  227  
  228              // If yield checks start getting inserted anywhere, we can be
  229              // killed before or after enqueueing. Deciding whether to
  230              // unkillably reacquire the lock needs to happen atomically
  231              // wrt enqueuing.
  232              do check_cvar_bounds(out_of_bounds, condvar_id, "cond.wait_on()") {
  233                  // Unconditionally "block". (Might not actually block if a
  234                  // signaller already sent -- I mean 'unconditionally' in contrast
  235                  // with acquire().)
  236                  do (|| {
  237                      unsafe {
  238                          do task::rekillable {
  239                              let _ = WaitEnd.take_unwrap().recv();
  240                          }
  241                      }
  242                  }).finally {
  243                      // Reacquire the condvar. Note this is back in the unkillable
  244                      // section; it needs to succeed, instead of itself dying.
  245                      match self.order {
  246                          Just(lock) => do lock.access {
  247                              self.sem.acquire();
  248                          },
  249                          Nothing => {
  250                              self.sem.acquire();
  251                          },
  252                      }
  253                  }
  254              }
  255          }
  256      }
  257  
  258      /// Wake up a blocked task. Returns false if there was no blocked task.
  259      pub fn signal(&self) -> bool { self.signal_on(0) }
  260  
  261      /// As signal, but with a specified condvar_id. See wait_on.
  262      pub fn signal_on(&self, condvar_id: uint) -> bool {
  263          unsafe {
  264              let mut out_of_bounds = None;
  265              let mut result = false;
  266              do (**self.sem).with |state| {
  267                  if condvar_id < state.blocked.len() {
  268                      result = state.blocked[condvar_id].signal();
  269                  } else {
  270                      out_of_bounds = Some(state.blocked.len());
  271                  }
  272              }
  273              do check_cvar_bounds(out_of_bounds, condvar_id, "cond.signal_on()") {
  274                  result
  275              }
  276          }
  277      }

  279      /// Wake up all blocked tasks. Returns the number of tasks woken.
  280      pub fn broadcast(&self) -> uint { self.broadcast_on(0) }
  281  
  282      /// As broadcast, but with a specified condvar_id. See wait_on.
  283      pub fn broadcast_on(&self, condvar_id: uint) -> uint {
  284          let mut out_of_bounds = None;
  285          let mut queue = None;
  286          unsafe {
  287              do (**self.sem).with |state| {
  288                  if condvar_id < state.blocked.len() {
  289                      // To avoid :broadcast_heavy, we make a new waitqueue,
  290                      // swap it out with the old one, and broadcast on the
  291                      // old one outside of the little-lock.
  292                      queue = Some(util::replace(&mut state.blocked[condvar_id],
  293                                                 WaitQueue::new()));
  294                  } else {
  295                      out_of_bounds = Some(state.blocked.len());
  296                  }
  297              }
  298              do check_cvar_bounds(out_of_bounds, condvar_id, "cond.signal_on()") {
  299                  let queue = queue.take_unwrap();
  300                  queue.broadcast()
  301              }
  302          }
  303      }
  304  }

  306  // Checks whether a condvar ID was out of bounds, and fails if so, or does
  307  // something else next on success.
  308  #[inline]
  309  #[doc(hidden)]
  310  fn check_cvar_bounds<U>(out_of_bounds: Option<uint>, id: uint, act: &str,
  311                          blk: &fn() -> U) -> U {
  312      match out_of_bounds {
  313          Some(0) =>
  314              fail!("%s with illegal ID %u - this lock has no condvars!", act, id),
  315          Some(length) =>
  316              fail!("%s with illegal ID %u - ID must be less than %u", act, id, length),
  317          None => blk()
  318      }
  319  }

  418  // NB: Wikipedia - Readers-writers_problem#The_third_readers-writers_problem
  419  
  420  #[doc(hidden)]
  421  struct RWLockInner {
  422      // You might ask, "Why don't you need to use an atomic for the mode flag?"
  423      // This flag affects the behaviour of readers (for plain readers, they
  424      // assert on it; for downgraders, they use it to decide which mode to
  425      // unlock for). Consider that the flag is only unset when the very last
  426      // reader exits; therefore, it can never be unset during a reader/reader
  427      // (or reader/downgrader) race.
  428      // By the way, if we didn't care about the assert in the read unlock path,
  429      // we could instead store the mode flag in write_downgrade's stack frame,
  430      // and have the downgrade tokens store a borrowed pointer to it.
  431      read_mode:  bool,
  432      // The only way the count flag is ever accessed is with xadd. Since it is
  433      // a read-modify-write operation, multiple xadds on different cores will
  434      // always be consistent with respect to each other, so a monotonic/relaxed
  435      // consistency ordering suffices (i.e., no extra barriers are needed).
  436      // FIXME(#6598): The atomics module has no relaxed ordering flag, so I use
  437      // acquire/release orderings superfluously. Change these someday.
  438      read_count: atomics::AtomicUint,
  439  }
  440  
  441  /**
  442   * A blocking, no-starvation, reader-writer lock with an associated condvar.
  443   *
  444   * # Failure
  445   * A task which fails while holding an rwlock will unlock the rwlock as it
  446   * unwinds.
  447   */
  448  pub struct RWLock {
  449      priv order_lock:  Semaphore,
  450      priv access_lock: Sem<~[WaitQueue]>,
  451      priv state:       UnsafeAtomicRcBox<RWLockInner>,
  452  }
  453  
  454  impl RWLock {
  455      /// Create a new rwlock, with one associated condvar.
  456      pub fn new() -> RWLock { RWLock::new_with_condvars(1) }
  457  
  458      /**
  459      * Create a new rwlock, with a specified number of associated condvars.
  460      * Similar to mutex_with_condvars.
  461      */
  462      pub fn new_with_condvars(num_condvars: uint) -> RWLock {
  463          let state = UnsafeAtomicRcBox::new(RWLockInner {
  464              read_mode:  false,
  465              read_count: atomics::AtomicUint::new(0),
  466          });
  467          RWLock { order_lock:  Semaphore::new(1),
  468                  access_lock: Sem::new_and_signal(1, num_condvars),
  469                  state:       state, }
  470      }
  471  
  472      /// Create a new handle to the rwlock.
  473      pub fn clone(&self) -> RWLock {
  474          RWLock { order_lock:  (&(self.order_lock)).clone(),
  475                   access_lock: Sem((*self.access_lock).clone()),
  476                   state:       self.state.clone() }
  477      }
  478  
  479      /**
  480       * Run a function with the rwlock in read mode. Calls to 'read' from other
  481       * tasks may run concurrently with this one.
  482       */
  483      pub fn read<U>(&self, blk: &fn() -> U) -> U {
  484          unsafe {
  485              do task::unkillable {
  486                  do (&self.order_lock).access {
  487                      let state = &mut *self.state.get();
  488                      let old_count = state.read_count.fetch_add(1, atomics::Acquire);
  489                      if old_count == 0 {
  490                          (&self.access_lock).acquire();
  491                          state.read_mode = true;
  492                      }
  493                  }
  494                  do (|| {
  495                      do task::rekillable { blk() }
  496                  }).finally {
  497                      let state = &mut *self.state.get();
  498                      assert!(state.read_mode);
  499                      let old_count = state.read_count.fetch_sub(1, atomics::Release);
  500                      assert!(old_count > 0);
  501                      if old_count == 1 {
  502                          state.read_mode = false;
  503                          // Note: this release used to be outside of a locked access
  504                          // to exclusive-protected state. If this code is ever
  505                          // converted back to such (instead of using atomic ops),
  506                          // this access MUST NOT go inside the exclusive access.
  507                          (&self.access_lock).release();
  508                      }
  509                  }
  510              }
  511          }
  512      }
  513  
  514      /**
  515       * Run a function with the rwlock in write mode. No calls to 'read' or
  516       * 'write' from other tasks will run concurrently with this one.
  517       */
  518      pub fn write<U>(&self, blk: &fn() -> U) -> U {
  519          unsafe {
  520              do task::unkillable {
  521                  (&self.order_lock).acquire();
  522                  do (&self.access_lock).access {
  523                      (&self.order_lock).release();
  524                      do task::rekillable {
  525                          blk()
  526                      }
  527                  }
  528              }
  529          }
  530      }
  531  
  532      /**
  533       * As write(), but also with a handle to a condvar. Waiting on this
  534       * condvar will allow readers and writers alike to take the rwlock before
  535       * the waiting task is signalled. (Note: a writer that waited and then
  536       * was signalled might reacquire the lock before other waiting writers.)
  537       */
  538      pub fn write_cond<U>(&self, blk: &fn(c: &Condvar) -> U) -> U {
  539          // It's important to thread our order lock into the condvar, so that
  540          // when a cond.wait() wakes up, it uses it while reacquiring the
  541          // access lock. If we permitted a waking-up writer to "cut in line",
  542          // there could arise a subtle race when a downgrader attempts to hand
  543          // off the reader cloud lock to a waiting reader. This race is tested
  544          // in arc.rs (test_rw_write_cond_downgrade_read_race) and looks like:
  545          // T1 (writer)              T2 (downgrader)             T3 (reader)
  546          // [in cond.wait()]
  547          //                          [locks for writing]
  548          //                          [holds access_lock]
  549          // [is signalled, perhaps by
  550          //  downgrader or a 4th thread]
  551          // tries to lock access(!)
  552          //                                                      lock order_lock
  553          //                                                      xadd read_count[0->1]
  554          //                                                      tries to lock access
  555          //                          [downgrade]
  556          //                          xadd read_count[1->2]
  557          //                          unlock access
  558          // Since T1 contended on the access lock before T3 did, it will steal
  559          // the lock handoff. Adding order_lock in the condvar reacquire path
  560          // solves this because T1 will hold order_lock while waiting on access,
  561          // which will cause T3 to have to wait until T1 finishes its write,
  562          // which can't happen until T2 finishes the downgrade-read entirely.
  563          // The astute reader will also note that making waking writers use the
  564          // order_lock is better for not starving readers.
  565          unsafe {
  566              do task::unkillable {
  567                  (&self.order_lock).acquire();
  568                  do (&self.access_lock).access_cond |cond| {
  569                      (&self.order_lock).release();
  570                      do task::rekillable {
  571                          let opt_lock = Just(&self.order_lock);
  572                          blk(&Condvar { sem: cond.sem, order: opt_lock,
  573                                         token: NonCopyable::new() })
  574                      }
  575                  }
  576              }
  577          }
  578      }
  579  
  580      /**
  581       * As write(), but with the ability to atomically 'downgrade' the lock;
  582       * i.e., to become a reader without letting other writers get the lock in
  583       * the meantime (such as unlocking and then re-locking as a reader would
  584       * do). The block takes a "write mode token" argument, which can be
  585       * transformed into a "read mode token" by calling downgrade(). Example:
  586       *
  587       * # Example
  588       *
  589       * ~~~ {.rust}
  590       * do lock.write_downgrade |mut write_token| {
  591       *     do write_token.write_cond |condvar| {
  592       *         ... exclusive access ...
  593       *     }
  594       *     let read_token = lock.downgrade(write_token);
  595       *     do read_token.read {
  596       *         ... shared access ...
  597       *     }
  598       * }
  599       * ~~~
  600       */
  601      pub fn write_downgrade<U>(&self, blk: &fn(v: RWLockWriteMode) -> U) -> U {
  602          // Implementation slightly different from the slicker 'write's above.
  603          // The exit path is conditional on whether the caller downgrades.
  604          do task::unkillable {
  605              (&self.order_lock).acquire();
  606              (&self.access_lock).acquire();
  607              (&self.order_lock).release();
  608              do (|| {
  609                  unsafe {
  610                      do task::rekillable {
  611                          blk(RWLockWriteMode { lock: self, token: NonCopyable::new() })
  612                      }
  613                  }
  614              }).finally {
  615                  let writer_or_last_reader;
  616                  // Check if we're releasing from read mode or from write mode.
  617                  let state = unsafe { &mut *self.state.get() };
  618                  if state.read_mode {
  619                      // Releasing from read mode.
  620                      let old_count = state.read_count.fetch_sub(1, atomics::Release);
  621                      assert!(old_count > 0);
  622                      // Check if other readers remain.
  623                      if old_count == 1 {
  624                          // Case 1: Writer downgraded & was the last reader
  625                          writer_or_last_reader = true;
  626                          state.read_mode = false;
  627                      } else {
  628                          // Case 2: Writer downgraded & was not the last reader
  629                          writer_or_last_reader = false;
  630                      }
  631                  } else {
  632                      // Case 3: Writer did not downgrade
  633                      writer_or_last_reader = true;
  634                  }
  635                  if writer_or_last_reader {
  636                      // Nobody left inside; release the "reader cloud" lock.
  637                      (&self.access_lock).release();
  638                  }
  639              }
  640          }
  641      }
  642  
  643      /// To be called inside of the write_downgrade block.
  644      pub fn downgrade<'a>(&self, token: RWLockWriteMode<'a>)
  645                           -> RWLockReadMode<'a> {
  646          if !borrow::ref_eq(self, token.lock) {
  647              fail!("Can't downgrade() with a different rwlock's write_mode!");
  648          }
  649          unsafe {
  650              do task::unkillable {
  651                  let state = &mut *self.state.get();
  652                  assert!(!state.read_mode);
  653                  state.read_mode = true;
  654                  // If a reader attempts to enter at this point, both the
  655                  // downgrader and reader will set the mode flag. This is fine.
  656                  let old_count = state.read_count.fetch_add(1, atomics::Release);
  657                  // If another reader was already blocking, we need to hand-off
  658                  // the "reader cloud" access lock to them.
  659                  if old_count != 0 {
  660                      // Guaranteed not to let another writer in, because
  661                      // another reader was holding the order_lock. Hence they
  662                      // must be the one to get the access_lock (because all
  663                      // access_locks are acquired with order_lock held). See
  664                      // the comment in write_cond for more justification.
  665                      (&self.access_lock).release();
  666                  }
  667              }
  668          }
  669          RWLockReadMode { lock: token.lock, token: NonCopyable::new() }
  670      }
  671  }
  672  
  673  /// The "write permission" token used for rwlock.write_downgrade().
  674  pub struct RWLockWriteMode<'self> { priv lock: &'self RWLock, priv token: NonCopyable }
  675  
  676  /// The "read permission" token used for rwlock.write_downgrade().
  677  pub struct RWLockReadMode<'self> { priv lock: &'self RWLock,
  678                                     priv token: NonCopyable }
  679  
  680  impl<'self> RWLockWriteMode<'self> {
  681      /// Access the pre-downgrade rwlock in write mode.
  682      pub fn write<U>(&self, blk: &fn() -> U) -> U { blk() }
  683      /// Access the pre-downgrade rwlock in write mode with a condvar.
  684      pub fn write_cond<U>(&self, blk: &fn(c: &Condvar) -> U) -> U {
  685          // Need to make the condvar use the order lock when reacquiring the
  686          // access lock. See comment in RWLock::write_cond for why.
  687          blk(&Condvar { sem:        &self.lock.access_lock,
  688                         order: Just(&self.lock.order_lock),
  689                         token: NonCopyable::new() })
  690      }
  691  }
  692  
  693  impl<'self> RWLockReadMode<'self> {
  694      /// Access the post-downgrade rwlock in read mode.
  695      pub fn read<U>(&self, blk: &fn() -> U) -> U { blk() }
  696  }

  29      /// Return the number of elements in the map

  31          self.v.iter().count(|elt| elt.is_some())
  32      }
  33  
  34      /// Return true if there are no elements in the map
  35      fn is_empty(&self) -> bool {
  36          self.v.iter().all(|elt| elt.is_none())
  37      }
  38  }

  45  impl<V> Map<uint, V> for SmallIntMap<V> {
  46      /// Return a reference to the value corresponding to the key
  47      fn find<'a>(&'a self, key: &uint) -> Option<&'a V> {
  48          if *key < self.v.len() {
  49              match self.v[*key] {
  50                Some(ref value) => Some(value),
  51                None => None
  52              }
  53          } else {
  54              None
  55          }
  56      }
  57  }

  59  impl<V> MutableMap<uint, V> for SmallIntMap<V> {
  60      /// Return a mutable reference to the value corresponding to the key
  61      fn find_mut<'a>(&'a mut self, key: &uint) -> Option<&'a mut V> {
  62          if *key < self.v.len() {
  63              match self.v[*key] {
  64                Some(ref mut value) => Some(value),
  65                None => None
  66              }
  67          } else {
  68              None
  69          }
  70      }
  71  
  72      /// Insert a key-value pair into the map. An existing value for a
  73      /// key is replaced by the new value. Return true if the key did
  74      /// not already exist in the map.
  75      fn insert(&mut self, key: uint, value: V) -> bool {
  76          let exists = self.contains_key(&key);
  77          let len = self.v.len();
  78          if len <= key {
  79              self.v.grow_fn(key - len + 1, |_| None);
  80          }
  81          self.v[key] = Some(value);
  82          !exists
  83      }
  84  
  85      /// Remove a key-value pair from the map. Return true if the key
  86      /// was present in the map, otherwise false.
  87      fn remove(&mut self, key: &uint) -> bool {
  88          self.pop(key).is_some()
  89      }
  90  
  91      /// Insert a key-value pair from the map. If the key already had a value
  92      /// present in the map, that value is returned. Otherwise None is returned.
  93      fn swap(&mut self, key: uint, value: V) -> Option<V> {
  94          match self.find_mut(&key) {
  95              Some(loc) => { return Some(replace(loc, value)); }
  96              None => ()
  97          }
  98          self.insert(key, value);
  99          return None;
 100      }
 101  
 102      /// Removes a key from the map, returning the value at the key if the key
 103      /// was previously in the map.
 104      fn pop(&mut self, key: &uint) -> Option<V> {
 105          if *key >= self.v.len() {
 106              return None;
 107          }
 108          self.v[*key].take()
 109      }
 110  }

 114      pub fn new() -> SmallIntMap<V> { SmallIntMap{v: ~[]} }

 116      pub fn get<'a>(&'a self, key: &uint) -> &'a V {
 117          self.find(key).expect("key not present")
 118      }

 120      /// An iterator visiting all key-value pairs in ascending order by the keys.
 121      /// Iterator element type is (uint, &'r V)
 122      pub fn iter<'r>(&'r self) -> SmallIntMapIterator<'r, V> {
 123          SmallIntMapIterator {
 124              front: 0,
 125              back: self.v.len(),
 126              iter: self.v.iter()
 127          }
 128      }
 129  
 130      /// An iterator visiting all key-value pairs in ascending order by the keys,
 131      /// with mutable references to the values
 132      /// Iterator element type is (uint, &'r mut V)
 133      pub fn mut_iter<'r>(&'r mut self) -> SmallIntMapMutIterator<'r, V> {
 134          SmallIntMapMutIterator {
 135              front: 0,
 136              back: self.v.len(),
 137              iter: self.v.mut_iter()
 138          }
 139      }
 140  
 141      /// An iterator visiting all key-value pairs in descending order by the keys.
 142      /// Iterator element type is (uint, &'r V)
 143      pub fn rev_iter<'r>(&'r self) -> SmallIntMapRevIterator<'r, V> {
 144          self.iter().invert()
 145      }
 146  
 147      /// An iterator visiting all key-value pairs in descending order by the keys,
 148      /// with mutable references to the values
 149      /// Iterator element type is (uint, &'r mut V)
 150      pub fn mut_rev_iter<'r>(&'r mut self) -> SmallIntMapMutRevIterator <'r, V> {
 151          self.mut_iter().invert()
 152      }
 153  
 154      /// Empties the hash map, moving all values into the specified closure
 155      pub fn move_iter(&mut self)
 156          -> FilterMap<(uint, Option<V>), (uint, V),
 157                  Enumerate<vec::MoveIterator<Option<V>>>>
 158      {
 159          let values = replace(&mut self.v, ~[]);
 160          values.move_iter().enumerate().filter_map(|(i, v)| {
 161              v.map_move(|v| (i, v))
 162          })
 163      }
 164  }

 166  impl<V:Clone> SmallIntMap<V> {
 167      pub fn update_with_key(&mut self, key: uint, val: V,
 168                             ff: &fn(uint, V, V) -> V) -> bool {
 169          let new_val = match self.find(&key) {
 170              None => val,
 171              Some(orig) => ff(key, (*orig).clone(), val)
 172          };
 173          self.insert(key, new_val)
 174      }
 175  
 176      pub fn update(&mut self, key: uint, newval: V, ff: &fn(V, V) -> V)
 177                    -> bool {
 178          self.update_with_key(key, newval, |_k, v, v1| ff(v,v1))
 179      }
 180  }

 183  macro_rules! iterator {
 184      (impl $name:ident -> $elem:ty, $getter:ident) => {
 185          impl<'self, T> Iterator<$elem> for $name<'self, T> {
 186              #[inline]
 187              fn next(&mut self) -> Option<$elem> {
 188                  while self.front < self.back {
 189                      match self.iter.next() {
 190                          Some(elem) => {
 191                              if elem.is_some() {
 192                                  let index = self.front;
 193                                  self.front += 1;
 194                                  return Some((index, elem. $getter ()));
 195                              }
 196                          }
 197                          _ => ()
 198                      }
 199                      self.front += 1;
 200                  }
 201                  None
 202              }
 203  
 204              #[inline]
 205              fn size_hint(&self) -> (uint, Option<uint>) {
 206                  (0, Some(self.back - self.front))
 207              }
 208          }
 209      }
 210  }

 212  macro_rules! double_ended_iterator {
 213      (impl $name:ident -> $elem:ty, $getter:ident) => {
 214          impl<'self, T> DoubleEndedIterator<$elem> for $name<'self, T> {
 215              #[inline]
 216              fn next_back(&mut self) -> Option<$elem> {
 217                  while self.front < self.back {
 218                      match self.iter.next_back() {
 219                          Some(elem) => {
 220                              if elem.is_some() {
 221                                  self.back -= 1;
 222                                  return Some((self.back, elem. $getter ()));
 223                              }
 224                          }
 225                          _ => ()
 226                      }
 227                      self.back -= 1;
 228                  }
 229                  None
 230              }
 231          }
 232      }
 233  }

 137  fn take_nonempty_prefix(rdr: @io::Reader,
 138                          ch: char,
 139                          pred: &fn(char) -> bool) -> (~str, char) {
 140      let mut buf = ~"";
 141      let mut ch = ch;
 142      while pred(ch) {
 143          buf.push_char(ch);

 153  fn take_num(rdr: @io::Reader, ch: char) -> (uint, char) {

 161  fn take_ident(rdr: @io::Reader, ch: char) -> (Identifier, char) {
 162      let (s,ch) = take_nonempty_prefix(rdr, ch, char::is_alphanumeric);
 163      if s.iter().all(char::is_digit) {
 164          match uint::from_str(s) {

 168      } else {
 169          (AlphaNumeric(s), ch)
 170      }
 171  }

 179  fn parse_reader(rdr: @io::Reader) -> Version {
 180      let (major, ch) = take_num(rdr, rdr.read_char());
 181      expect(ch, '.');
 182      let (minor, ch) = take_num(rdr, rdr.read_char());
 183      expect(ch, '.');
 184      let (patch, ch) = take_num(rdr, rdr.read_char());
 185  
 186      let mut pre = ~[];
 187      let mut build = ~[];
 188  
 189      let mut ch = ch;
 190      if ch == '-' {
 191          loop {
 192              let (id, c) = take_ident(rdr, rdr.read_char());
 193              pre.push(id);
 194              ch = c;
 195              if ch != '.' { break; }
 196          }
 197      }
 198  
 199      if ch == '+' {
 200          loop {
 201              let (id, c) = take_ident(rdr, rdr.read_char());
 202              build.push(id);
 203              ch = c;
 204              if ch != '.' { break; }
 205          }
 206      }
 207  
 208      Version {
 209          major: major,
 210          minor: minor,
 211          patch: patch,
 212          pre: pre,
 213          build: build,
 214      }
 215  }

 218  pub fn parse(s: &str) -> Option<Version> {
 219      if !s.is_ascii() {
 220          return None;
 221      }
 222      let s = s.trim();
 223      let mut bad = false;
 224      do bad_parse::cond.trap(|_| { debug!("bad"); bad = true }).inside {
 225          do io::with_str_reader(s) |rdr| {
 226              let v = parse_reader(rdr);
 227              if bad || v.to_str() != s.to_owned() {
 228                  None
 229              } else {
 230                  Some(v)
 231              }
 232          }
 233      }
 234  }

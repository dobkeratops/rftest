   11 //! An ordered map and set implemented as self-balancing binary search

   19 // This is implemented as an AA tree, which is a simplified variation of
   20 // a red-black tree where red (horizontal) nodes can only be added
   21 // as a right child. The time complexity is the same, and re-balancing
   22 // operations are more frequent but also cheaper.
   23 
   24 // Future improvements:
   25 
   26 // range search - O(log n) retrieval of an iterator from some key
   27

   29 //   * intersection: &
   30 //   * difference: -
   31 //   * symmetric difference: ^
   32 //   * union: |

   49 // Lexicographical comparison
   50 fn lt<K: Ord + TotalOrd, V: Ord>(a: &TreeMap<K, V>,
   51                                  b: &TreeMap<K, V>) -> bool {
   52     // the Zip iterator is as long as the shortest of a and b.
   53     for ((key_a, value_a), (key_b, value_b)) in a.iter().zip(b.iter()) {
   54         if *key_a < *key_b { return true; }
   55         if *key_a > *key_b { return false; }
   56         if *value_a < *value_b { return true; }
   57         if *value_a > *value_b { return false; }
   58     }
   59 
   60     a.len() < b.len()
   61 }

  237 impl<'self, K, V> TreeMapIterator<'self, K, V> {
  238     #[inline(always)]
  239     fn next_(&mut self, forward: bool) -> Option<(&'self K, &'self V)> {
  240         while !self.stack.is_empty() || self.node.is_some() {
  241             match *self.node {
  242               Some(ref x) => {
  243                 self.stack.push(x);
  244                 self.node = if forward { &x.left } else { &x.right };
  245               }
  246               None => {
  247                 let res = self.stack.pop();
  248                 self.node = if forward { &res.right } else { &res.left };
  249                 self.remaining_max -= 1;
  250                 if self.remaining_min > 0 {
  251                     self.remaining_min -= 1;
  252                 }
  253                 return Some((&res.key, &res.value));
  254               }
  255             }
  256         }
  257         None
  258     }
  259 }

  294 /// iter_traverse_left, iter_traverse_right and iter_traverse_complete are used to
  295 /// initialize TreeMapIterator pointing to element inside tree structure.
  296 ///
  297 /// They should be used in following manner:
  298 ///   - create iterator using TreeMap::iter_for_traversal
  299 ///   - find required node using `iter_traverse_left`/`iter_traverse_right`
  300 ///     (current node is `TreeMapIterator::node` field)
  301 ///   - complete initialization with `iter_traverse_complete`
  302 #[inline]
  303 fn iter_traverse_left<'a, K, V>(it: &mut TreeMapIterator<'a, K, V>) {
  304     let node = it.node.get_ref();
  305     it.stack.push(node);
  306     it.node = &node.left;
  307 }
  308 
  309 #[inline]
  310 fn iter_traverse_right<'a, K, V>(it: &mut TreeMapIterator<'a, K, V>) {
  311     it.node = &(it.node.get_ref().right);
  312 }

  314 /// iter_traverse_left, iter_traverse_right and iter_traverse_complete are used to
  315 /// initialize TreeMapIterator pointing to element inside tree structure.
  316 ///
  317 /// Completes traversal. Should be called before using iterator.
  318 /// Iteration will start from `self.node`.

  324     match *it.node {
  325         Some(ref n) => {
  326             it.stack.push(n);
  327             it.node = &none;
  328         }
  329         None => ()
  330     }
  331 }

  695 // Nodes keep track of their level in the tree, starting at 1 in the

  700     value: V,

  714 fn mutate_values<'r, K: TotalOrd, V>(node: &'r mut Option<~TreeNode<K, V>>,
  715                                      f: &fn(&'r K, &'r mut V) -> bool)
  716                                   -> bool {
  717     match *node {
  718       Some(~TreeNode{key: ref key, value: ref mut value, left: ref mut left,
  719                      right: ref mut right, _}) => {
  720         if !mutate_values(left,  |k,v| f(k,v)) { return false }
  721         if !f(key, value) { return false }
  722         if !mutate_values(right, |k,v| f(k,v)) { return false }
  723       }
  724       None => return false
  725     }
  726     true
  727 }

  729 // Remove left horizontal link by rotating right
  730 fn skew<K: TotalOrd, V>(node: &mut ~TreeNode<K, V>) {
  731     if node.left.map_default(false, |x| x.level == node.level) {
  732         let mut save = node.left.take_unwrap();
  733         swap(&mut node.left, &mut save.right); // save.right now None
  734         swap(node, &mut save);
  735         node.right = Some(save);
  736     }
  737 }

  739 // Remove dual horizontal link by rotating left and increasing level of
  740 // the parent
  741 fn split<K: TotalOrd, V>(node: &mut ~TreeNode<K, V>) {
  742     if node.right.map_default(false,
  743       |x| x.right.map_default(false, |y| y.level == node.level)) {
  744         let mut save = node.right.take_unwrap();
  745         swap(&mut node.right, &mut save.left); // save.left now None
  746         save.level += 1;
  747         swap(node, &mut save);
  748         node.left = Some(save);
  749     }
  750 }

  783           }
  784           Equal => {

  786             Some(replace(&mut save.value, value))
  787           }
  788         }

  797 fn remove<K: TotalOrd, V>(node: &mut Option<~TreeNode<K, V>>,
  798                           key: &K) -> Option<V> {
  799     fn heir_swap<K: TotalOrd, V>(node: &mut ~TreeNode<K, V>,
  800                                  child: &mut Option<~TreeNode<K, V>>) {
  801         // *could* be done without recursion, but it won't borrow check
  802         for x in child.mut_iter() {
  803             if x.right.is_some() {
  804                 heir_swap(node, &mut x.right);
  805             } else {
  806                 swap(&mut node.key, &mut x.key);
  807                 swap(&mut node.value, &mut x.value);
  808             }
  809         }
  810     }
  811 
  812     match *node {
  813       None => {
  814         return None; // bottom of tree
  815       }
  816       Some(ref mut save) => {
  817         let (ret, rebalance) = match key.cmp(&save.key) {
  818           Less => (remove(&mut save.left, key), true),
  819           Greater => (remove(&mut save.right, key), true),
  820           Equal => {
  821             if save.left.is_some() {
  822                 if save.right.is_some() {
  823                     let mut left = save.left.take_unwrap();
  824                     if left.right.is_some() {
  825                         heir_swap(save, &mut left.right);
  826                     } else {
  827                         swap(&mut save.key, &mut left.key);
  828                         swap(&mut save.value, &mut left.value);
  829                     }
  830                     save.left = Some(left);
  831                     (remove(&mut save.left, key), true)
  832                 } else {
  833                     let new = save.left.take_unwrap();
  834                     let ~TreeNode{value, _} = replace(save, new);
  835                     *save = save.left.take_unwrap();
  836                     (Some(value), true)
  837                 }
  838             } else if save.right.is_some() {
  839                 let new = save.right.take_unwrap();
  840                 let ~TreeNode{value, _} = replace(save, new);
  841                 (Some(value), true)
  842             } else {
  843                 (None, false)
  844             }
  845           }
  846         };
  847 
  848         if rebalance {
  849             let left_level = save.left.map_default(0, |x| x.level);
  850             let right_level = save.right.map_default(0, |x| x.level);
  851 
  852             // re-balance, if necessary
  853             if left_level < save.level - 1 || right_level < save.level - 1 {
  854                 save.level -= 1;
  855 
  856                 if right_level > save.level {
  857                     for x in save.right.mut_iter() { x.level = save.level }
  858                 }
  859 
  860                 skew(save);
  861 
  862                 for right in save.right.mut_iter() {
  863                     skew(right);
  864                     for x in right.right.mut_iter() { skew(x) }
  865                 }
  866 
  867                 split(save);
  868                 for x in save.right.mut_iter() { split(x) }
  869             }
  870 
  871             return ret;
  872         }

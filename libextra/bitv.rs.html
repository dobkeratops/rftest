   28 /// a mask that has a 1 for each defined bit in a small_bitv, assuming n bits
   29 #[inline]
   30 fn small_mask(nbits: uint) -> uint {
   31     (1 << nbits) - 1
   32 }

   39     #[inline]
   40     pub fn bits_op(&mut self,
   41                    right_bits: uint,
   42                    nbits: uint,
   43                    f: &fn(uint, uint) -> uint)
   44                    -> bool {
   45         let mask = small_mask(nbits);
   46         let old_b: uint = self.bits;
   47         let new_b = f(old_b, right_bits);
   48         self.bits = new_b;
   49         mask & old_b != mask & new_b
   50     }

  113 #[deriving(Clone)]
  114 struct BigBitv {
  115     storage: ~[uint]
  116 }

  118 /**
  119  * a mask that has a 1 for each defined bit in the nth element of a big_bitv,
  120  * assuming n bits.
  121  */
  122 #[inline]
  123 fn big_mask(nbits: uint, elem: uint) -> uint {
  124     let rmd = nbits % uint::bits;
  125     let nelems = nbits/uint::bits + if rmd == 0 {0} else {1};
  126 
  127     if elem < nelems - 1 || rmd == 0 {
  128         !0
  129     } else {
  130         (1 << rmd) - 1
  131     }
  132 }
  133 
  134 impl BigBitv {
  135     pub fn new(storage: ~[uint]) -> BigBitv {
  136         BigBitv {storage: storage}

  139     #[inline]
  140     pub fn process(&mut self,
  141                    b: &BigBitv,
  142                    nbits: uint,
  143                    op: &fn(uint, uint) -> uint)
  144                    -> bool {
  145         let len = b.storage.len();
  146         assert_eq!(self.storage.len(), len);
  147         let mut changed = false;
  148         for i in range(0, len) {
  149             let mask = big_mask(nbits, i);
  150             let w0 = self.storage[i] & mask;
  151             let w1 = b.storage[i] & mask;
  152             let w = op(w0, w1) & mask;
  153             if w0 != w {
  154                 changed = true;
  155                 self.storage[i] = w;
  156             }
  157         }
  158         changed
  159     }
  160 
  161     #[inline]
  162     pub fn each_storage(&mut self, op: &fn(v: &mut uint) -> bool) -> bool {
  163         range(0u, self.storage.len()).advance(|i| op(&mut self.storage[i]))
  164     }
  165 
  166     #[inline]
  167     pub fn negate(&mut self) { do self.each_storage |w| { *w = !*w; true }; }
  168 
  169     #[inline]
  170     pub fn union(&mut self, b: &BigBitv, nbits: uint) -> bool {
  171         self.process(b, nbits, |w1, w2| w1 | w2)
  172     }
  173 
  174     #[inline]
  175     pub fn intersect(&mut self, b: &BigBitv, nbits: uint) -> bool {
  176         self.process(b, nbits, |w1, w2| w1 & w2)
  177     }
  178 
  179     #[inline]
  180     pub fn become(&mut self, b: &BigBitv, nbits: uint) -> bool {
  181         self.process(b, nbits, |_, w| w)
  182     }
  183 
  184     #[inline]
  185     pub fn difference(&mut self, b: &BigBitv, nbits: uint) -> bool {
  186         self.process(b, nbits, |w1, w2| w1 & !w2)
  187     }
  188 
  189     #[inline]
  190     pub fn get(&self, i: uint) -> bool {
  191         let w = i / uint::bits;
  192         let b = i % uint::bits;
  193         let x = 1 & self.storage[w] >> b;
  194         x == 1
  195     }
  196 
  197     #[inline]
  198     pub fn set(&mut self, i: uint, x: bool) {
  199         let w = i / uint::bits;
  200         let b = i % uint::bits;
  201         let flag = 1 << b;
  202         self.storage[w] = if x { self.storage[w] | flag }
  203                           else { self.storage[w] & !flag };
  204     }
  205 
  206     #[inline]
  207     pub fn equals(&self, b: &BigBitv, nbits: uint) -> bool {
  208         let len = b.storage.len();
  209         do uint::iterate(0, len) |i| {
  210             let mask = big_mask(nbits, i);
  211             if mask & self.storage[i] != mask & b.storage[i] {
  212                 false
  213             } else {
  214                 true
  215             }
  216         }
  217     }

  225 /// The bitvector type
  226 #[deriving(Clone)]
  227 pub struct Bitv {
  228     /// Internal representation of the bit vector (small or large)
  229     rep: BitvVariant,
  230     /// The number of valid bits in the internal representation
  231     nbits: uint
  232 }

  234 fn die() -> ! {
  235     fail!("Tried to do operation on bit vectors with different sizes");
  236 }
  237 
  238 impl Bitv {
  239     #[inline]
  240     fn do_op(&mut self, op: Op, other: &Bitv) -> bool {
  241         if self.nbits != other.nbits {
  242             die();
  243         }
  244         match self.rep {
  245           Small(ref mut s) => match other.rep {
  246             Small(ref s1) => match op {
  247               Union      => s.union(s1,      self.nbits),
  248               Intersect  => s.intersect(s1,  self.nbits),
  249               Assign     => s.become(s1,     self.nbits),
  250               Difference => s.difference(s1, self.nbits)
  251             },
  252             Big(_) => die()
  253           },
  254           Big(ref mut s) => match other.rep {
  255             Small(_) => die(),
  256             Big(ref s1) => match op {
  257               Union      => s.union(s1,      self.nbits),
  258               Intersect  => s.intersect(s1,  self.nbits),
  259               Assign     => s.become(s1,     self.nbits),
  260               Difference => s.difference(s1, self.nbits)
  261             }
  262           }
  263         }
  264     }
  265 
  266 }

  270         let rep = if nbits <= uint::bits {
  271             Small(SmallBitv::new(if init {!0} else {0}))

  444     /**
  445      * Organise the bits into bytes, such that the first bit in the
  446      * bitv becomes the high-order bit of the first byte. If the
  447      * size of the bitv is not a multiple of 8 then trailing bits
  448      * will be filled-in with false/0
  449      */
  450     pub fn to_bytes(&self) -> ~[u8] {
  451         fn bit (bitv: &Bitv, byte: uint, bit: uint) -> u8 {
  452             let offset = byte * 8 + bit;
  453             if offset >= bitv.nbits {
  454                 0
  455             } else {
  456                 bitv[offset] as u8 << (7 - bit)
  457             }
  458         }
  459 
  460         let len = self.nbits/8 +
  461                   if self.nbits % 8 == 0 { 0 } else { 1 };
  462         vec::from_fn(len, |i|
  463             bit(self, i, 0) |
  464             bit(self, i, 1) |
  465             bit(self, i, 2) |
  466             bit(self, i, 3) |
  467             bit(self, i, 4) |

  689             self.bitv.storage.grow(other.capacity() / uint::bits, &0);
  690         }

  692             let old = self.bitv.storage[i];
  693             let new = f(old, w);
  694             self.bitv.storage[i] = new;
  695             self.size += nbits(new) - nbits(old);
  696         }
  697     }

  861 impl BitvSet {
  862     /// Visits each of the words that the two bit vectors (self and other)
  863     /// both have in common. The three yielded arguments are (bit location,
  864     /// w1, w2) where the bit location is the number of bits offset so far,
  865     /// and w1/w2 are the words coming from the two vectors self, other.
  866     fn common_iter<'a>(&'a self, other: &'a BitvSet)
  867         -> MapE<(uint,&uint),(uint,uint,uint), &'a ~[uint],Enumerate<vec::VecIterator<'a,uint>>> {
  868         let min = num::min(self.bitv.storage.len(),
  869                             other.bitv.storage.len());
  870         MapE{iter: self.bitv.storage.slice(0, min).iter().enumerate(),
  871              env: &other.bitv.storage,
  872              f: |(i, &w): (uint, &uint), o_store| (i * uint::bits, w, o_store[i])
  873         }
  874     }
  875 
  876     /// Visits each word in self or other that extends beyond the other. This
  877     /// will only iterate through one of the vectors, and it only iterates
  878     /// over the portion that doesn't overlap with the other one.
  879     ///
  880     /// The yielded arguments are a bool, the bit offset, and a word. The bool
  881     /// is true if the word comes from 'self', and false if it comes from
  882     /// 'other'.
  883     fn outlier_iter<'a>(&'a self, other: &'a BitvSet)
  884         -> MapE<(uint, &uint),(bool, uint, uint), uint, Enumerate<vec::VecIterator<'a, uint>>> {
  885         let len1 = self.bitv.storage.len();
  886         let len2 = other.bitv.storage.len();
  887         let min = num::min(len1, len2);
  888 
  889         if min < len1 {
  890             MapE{iter: self.bitv.storage.slice(min, len1).iter().enumerate(),
  891                  env: min,
  892                  f: |(i, &w): (uint, &uint), min| (true, (i + min) * uint::bits, w)
  893             }
  894         } else {
  895             MapE{iter: other.bitv.storage.slice(min, len2).iter().enumerate(),
  896                  env: min,
  897                  f: |(i, &w): (uint, &uint), min| (false, (i + min) * uint::bits, w)
  898             }
  899         }
  900     }
  901 }

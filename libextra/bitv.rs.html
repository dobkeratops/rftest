   23  struct SmallBitv {

   28  /// a mask that has a 1 for each defined bit in a small_bitv, assuming n bits
   29  #[inline]
   30  fn small_mask(nbits: uint) -> uint {
   31      (1 << nbits) - 1
   32  }

   39      #[inline]
   40      pub fn bits_op(&mut self,
   41                     right_bits: uint,
   42                     nbits: uint,
   43                     f: &fn(uint, uint) -> uint)
   44                     -> bool {
   45          let mask = small_mask(nbits);
   46          let old_b: uint = self.bits;
   47          let new_b = f(old_b, right_bits);
   48          self.bits = new_b;
   49          mask & old_b != mask & new_b
   50      }

  113  #[deriving(Clone)]
  114  struct BigBitv {
  115      storage: ~[uint]
  116  }
  117  
  118  /**
  119   * a mask that has a 1 for each defined bit in the nth element of a big_bitv,
  120   * assuming n bits.
  121   */
  122  #[inline]
  123  fn big_mask(nbits: uint, elem: uint) -> uint {
  124      let rmd = nbits % uint::bits;
  125      let nelems = nbits/uint::bits + if rmd == 0 {0} else {1};
  126  
  127      if elem < nelems - 1 || rmd == 0 {
  128          !0
  129      } else {
  130          (1 << rmd) - 1
  131      }
  132  }
  133  
  134  impl BigBitv {
  135      pub fn new(storage: ~[uint]) -> BigBitv {
  136          BigBitv {storage: storage}
  137      }
  138  
  139      #[inline]
  140      pub fn process(&mut self,
  141                     b: &BigBitv,
  142                     nbits: uint,
  143                     op: &fn(uint, uint) -> uint)
  144                     -> bool {
  145          let len = b.storage.len();
  146          assert_eq!(self.storage.len(), len);
  147          let mut changed = false;
  148          for (i, (a, b)) in self.storage.mut_iter()
  149                                 .zip(b.storage.iter())
  150                                 .enumerate() {
  151              let mask = big_mask(nbits, i);
  152              let w0 = *a & mask;
  153              let w1 = *b & mask;
  154              let w = op(w0, w1) & mask;
  155              if w0 != w {
  156                  changed = true;
  157                  *a = w;
  158              }
  159          }
  160          changed
  161      }
  162  
  163      #[inline]
  164      pub fn each_storage(&mut self, op: &fn(v: &mut uint) -> bool) -> bool {
  165          self.storage.mut_iter().advance(|elt| op(elt))
  166      }
  167  
  168      #[inline]
  169      pub fn negate(&mut self) { do self.each_storage |w| { *w = !*w; true }; }
  170  
  171      #[inline]
  172      pub fn union(&mut self, b: &BigBitv, nbits: uint) -> bool {
  173          self.process(b, nbits, |w1, w2| w1 | w2)
  174      }
  175  
  176      #[inline]
  177      pub fn intersect(&mut self, b: &BigBitv, nbits: uint) -> bool {
  178          self.process(b, nbits, |w1, w2| w1 & w2)
  179      }
  180  
  181      #[inline]
  182      pub fn become(&mut self, b: &BigBitv, nbits: uint) -> bool {
  183          self.process(b, nbits, |_, w| w)
  184      }
  185  
  186      #[inline]
  187      pub fn difference(&mut self, b: &BigBitv, nbits: uint) -> bool {
  188          self.process(b, nbits, |w1, w2| w1 & !w2)
  189      }
  190  
  191      #[inline]
  192      pub fn get(&self, i: uint) -> bool {
  193          let w = i / uint::bits;
  194          let b = i % uint::bits;
  195          let x = 1 & self.storage[w] >> b;
  196          x == 1
  197      }
  198  
  199      #[inline]
  200      pub fn set(&mut self, i: uint, x: bool) {
  201          let w = i / uint::bits;
  202          let b = i % uint::bits;
  203          let flag = 1 << b;
  204          self.storage[w] = if x { self.storage[w] | flag }
  205                            else { self.storage[w] & !flag };
  206      }
  207  
  208      #[inline]
  209      pub fn equals(&self, b: &BigBitv, nbits: uint) -> bool {
  210          for (i, elt) in b.storage.iter().enumerate() {
  211              let mask = big_mask(nbits, i);
  212              if mask & self.storage[i] != mask & *elt {
  213                  return false;
  214              }
  215          }
  216          true
  217      }
  218  }

  225  /// The bitvector type
  226  #[deriving(Clone)]
  227  pub struct Bitv {
  228      /// Internal representation of the bit vector (small or large)
  229      rep: BitvVariant,
  230      /// The number of valid bits in the internal representation
  231      nbits: uint
  232  }
  233  
  234  fn die() -> ! {
  235      fail!("Tried to do operation on bit vectors with different sizes");
  236  }
  237  
  238  impl Bitv {
  239      #[inline]
  240      fn do_op(&mut self, op: Op, other: &Bitv) -> bool {
  241          if self.nbits != other.nbits {
  242              die();
  243          }
  244          match self.rep {
  245            Small(ref mut s) => match other.rep {
  246              Small(ref s1) => match op {
  247                Union      => s.union(s1,      self.nbits),
  248                Intersect  => s.intersect(s1,  self.nbits),
  249                Assign     => s.become(s1,     self.nbits),
  250                Difference => s.difference(s1, self.nbits)
  251              },
  252              Big(_) => die()
  253            },
  254            Big(ref mut s) => match other.rep {
  255              Small(_) => die(),
  256              Big(ref s1) => match op {
  257                Union      => s.union(s1,      self.nbits),
  258                Intersect  => s.intersect(s1,  self.nbits),
  259                Assign     => s.become(s1,     self.nbits),
  260                Difference => s.difference(s1, self.nbits)
  261              }
  262            }
  263          }
  264      }
  265  
  266  }

  268  impl Bitv {
  269      pub fn new(nbits: uint, init: bool) -> Bitv {
  270          let rep = if nbits <= uint::bits {
  271              Small(SmallBitv::new(if init {!0} else {0}))
  272          }
  273          else {
  274              let nelems = nbits/uint::bits +
  275                           if nbits % uint::bits == 0 {0} else {1};
  276              let elem = if init {!0u} else {0u};
  277              let s = vec::from_elem(nelems, elem);
  278              Big(BigBitv::new(s))
  279          };
  280          Bitv {rep: rep, nbits: nbits}
  281      }

  444      /**
  445       * Organise the bits into bytes, such that the first bit in the
  446       * bitv becomes the high-order bit of the first byte. If the

  448       * will be filled-in with false/0
  449       */
  450      pub fn to_bytes(&self) -> ~[u8] {

  455              } else {
  456                  bitv[offset] as u8 << (7 - bit)
  457              }
  458          }

  461                    if self.nbits % 8 == 0 { 0 } else { 1 };
  462          vec::from_fn(len, |i|
  463              bit(self, i, 0) |
  464              bit(self, i, 1) |
  465              bit(self, i, 2) |
  466              bit(self, i, 3) |
  467              bit(self, i, 4) |

  675      #[inline]
  676      fn other_op(&mut self, other: &BitvSet, f: &fn(uint, uint) -> uint) {
  677          fn nbits(mut w: uint) -> uint {
  678              let mut bits = 0;
  679              for _ in range(0u, uint::bits) {
  680                  if w == 0 {
  681                      break;
  682                  }
  683                  bits += w & 1;
  684                  w >>= 1;
  685              }
  686              return bits;
  687          }

  689              self.bitv.storage.grow(other.capacity() / uint::bits, &0);
  690          }
  691          for (i, &w) in other.bitv.storage.iter().enumerate() {
  692              let old = self.bitv.storage[i];
  693              let new = f(old, w);
  694              self.bitv.storage[i] = new;
  695              self.size += nbits(new) - nbits(old);
  696          }
  697      }

  861  impl BitvSet {
  862      /// Visits each of the words that the two bit vectors (self and other)
  863      /// both have in common. The three yielded arguments are (bit location,
  864      /// w1, w2) where the bit location is the number of bits offset so far,
  865      /// and w1/w2 are the words coming from the two vectors self, other.
  866      fn common_iter<'a>(&'a self, other: &'a BitvSet)
  867          -> Map<'static, ((uint, &'a uint), &'a ~[uint]), (uint, uint, uint),
  868                 Zip<Enumerate<vec::VecIterator<'a, uint>>, Repeat<&'a ~[uint]>>> {
  869          let min = num::min(self.bitv.storage.len(), other.bitv.storage.len());
  870          self.bitv.storage.slice(0, min).iter().enumerate()
  871              .zip(Repeat::new(&other.bitv.storage))
  872              .map(|((i, &w), o_store)| (i * uint::bits, w, o_store[i]))
  873      }
  874  
  875      /// Visits each word in self or other that extends beyond the other. This
  876      /// will only iterate through one of the vectors, and it only iterates
  877      /// over the portion that doesn't overlap with the other one.
  878      ///
  879      /// The yielded arguments are a bool, the bit offset, and a word. The bool
  880      /// is true if the word comes from 'self', and false if it comes from
  881      /// 'other'.
  882      fn outlier_iter<'a>(&'a self, other: &'a BitvSet)
  883          -> Map<'static, ((uint, &'a uint), uint), (bool, uint, uint),
  884                 Zip<Enumerate<vec::VecIterator<'a, uint>>, Repeat<uint>>> {
  885          let slen = self.bitv.storage.len();
  886          let olen = other.bitv.storage.len();
  887  
  888          if olen < slen {
  889              self.bitv.storage.slice_from(olen).iter().enumerate()
  890                  .zip(Repeat::new(olen))
  891                  .map(|((i, &w), min)| (true, (i + min) * uint::bits, w))
  892          } else {
  893              other.bitv.storage.slice_from(slen).iter().enumerate()
  894                  .zip(Repeat::new(slen))
  895                  .map(|((i, &w), min)| (false, (i + min) * uint::bits, w))
  896          }
  897      }
  898  }

  900  pub struct BitvSetIterator<'self> {
  901      priv set: &'self BitvSet,
  902      priv next_idx: uint
  903  }

  905  impl<'self> Iterator<uint> for BitvSetIterator<'self> {
  906      #[inline]
  907      fn next(&mut self) -> Option<uint> {
  908          while self.next_idx < self.set.capacity() {
  909              let idx = self.next_idx;
  910              self.next_idx += 1;
  911  
  912              if self.set.contains(&idx) {
  913                  return Some(idx);
  914              }
  915          }
  916  
  917          return None;
  918      }
  919  
  920      fn size_hint(&self) -> (uint, Option<uint>) {
  921          (0, Some(self.set.capacity() - self.next_idx))
  922      }
  923  }

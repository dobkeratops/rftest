  11  //! Timers based on timerfd_create(2)
  12  //!
  13  //! On OSes which support timerfd_create, we can use these much more accurate
  14  //! timers over select() + a timeout (see timer_other.rs). This strategy still
  15  //! employs a worker thread which does the waiting on the timer fds (to send
  16  //! messages away).
  17  //!
  18  //! The worker thread in this implementation uses epoll(7) to block. It
  19  //! maintains a working set of *all* native timers in the process, along with a
  20  //! pipe file descriptor used to communicate that there is data available on the
  21  //! incoming channel to the worker thread. Timers send requests to update their
  22  //! timerfd settings to the worker thread (see the comment above 'oneshot' for
  23  //! why).
  24  //!
  25  //! As with timer_other, timers just using sleep() do not use the timerfd at
  26  //! all. They remove the timerfd from the worker thread and then invoke
  27  //! nanosleep() to block the calling thread.
  28  //!
  29  //! As with timer_other, all units in this file are in units of millseconds.
  30

  77      add(efd, input);

  98                  incoming = true;
  99              } else {

 101                  // drain the timerfd of how many times its fired
 102                  //
 103                  // FIXME: should this perform a send() this number of
 104                  //      times?

 108                          Some(i) => {

 113                      }
 114                  };

 116                      drop(list.remove(i));
 117                      del(efd, fd);
 118                  }
 119              }
 120          }

 123              match messages.try_recv() {
 124                  Data(NewTimer(fd, chan, one, timeval)) => {
 125                      // acknowledge we have the new channel, we will never send
 126                      // another message to the old channel
 127                      chan.send(());

 130                      // we need to add it to the epoll set.
 131                      match list.bsearch(|&(f, _, _)| f.cmp(&fd)) {
 132                          Some(i) => {
 133                              drop(mem::replace(&mut list[i], (fd, chan, one)));

 135                          None => {

 137                                  Some(i) => list.insert(i, (fd, chan, one)),

 140                              add(efd, fd);
 141                          }
 142                      }
 143  
 144                      // Update the timerfd's time value now that we have control
 145                      // of the timerfd

 147                          imp::timerfd_settime(fd, 0, &timeval, ptr::null())
 148                      };

 150                  }
 151  
 152                  Data(RemoveTimer(fd, chan)) => {
 153                      match list.bsearch(|&(f, _, _)| f.cmp(&fd)) {
 154                          Some(i) => {
 155                              drop(list.remove(i));
 156                              del(efd, fd);
 157                          }
 158                          None => {}
 159                      }
 160                      chan.send(());
 161                  }
 162  
 163                  Data(Shutdown) => {
 164                      assert!(list.len() == 0);
 165                      break 'outer;
 166                  }
 167  
 168                  _ => break,
 169              }
 170          }
 171      }

 183      pub fn sleep(ms: u64) {
 184          let mut to_sleep = libc::timespec {
 185              tv_sec: (ms / 1000) as libc::time_t,
 186              tv_nsec: ((ms % 1000) * 1000000) as libc::c_long,
 187          };

 189              if os::errno() as int != libc::EINTR as int {
 190                  fail!("failed to sleep, but not because of EINTR?");
 191              }
 192          }
 193      }

 211      // Periodic and oneshot channels are updated by updating the settings on the
 212      // corresopnding timerfd. The update is not performed on the thread calling
 213      // oneshot or period, but rather the helper epoll thread. The reason for
 214      // this is to avoid losing messages and avoid leaking messages across ports.
 215      //
 216      // By updating the timerfd on the helper thread, we're guaranteed that all
 217      // messages for a particular setting of the timer will be received by the
 218      // new channel/port pair rather than leaking old messages onto the new port
 219      // or leaking new messages onto the old port.
 220      //
 221      // We also wait for the remote thread to actually receive the new settings
 222      // before returning to guarantee the invariant that when oneshot() and
 223      // period() return that the old port will never receive any more messages.
 224  
 225      fn oneshot(&mut self, msecs: u64) -> Receiver<()> {
 226          let (tx, rx) = channel();
 227  
 228          let new_value = imp::itimerspec {
 229              it_interval: imp::timespec { tv_sec: 0, tv_nsec: 0 },
 230              it_value: imp::timespec {
 231                  tv_sec: (msecs / 1000) as libc::time_t,
 232                  tv_nsec: ((msecs % 1000) * 1000000) as libc::c_long,
 233              }
 234          };
 235          timer_helper::send(NewTimer(self.fd.fd(), tx, true, new_value));
 236          rx.recv();
 237          self.on_worker = true;
 238  
 239          return rx;
 240      }
 241  
 242      fn period(&mut self, msecs: u64) -> Receiver<()> {
 243          let (tx, rx) = channel();
 244  
 245          let spec = imp::timespec {
 246              tv_sec: (msecs / 1000) as libc::time_t,
 247              tv_nsec: ((msecs % 1000) * 1000000) as libc::c_long,
 248          };
 249          let new_value = imp::itimerspec { it_interval: spec, it_value: spec, };
 250          timer_helper::send(NewTimer(self.fd.fd(), tx, false, new_value));
 251          rx.recv();
 252          self.on_worker = true;
 253  
 254          return rx;
 255      }
 256  }

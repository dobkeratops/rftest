  25  /**
  26   * A value representing a child process.
  27   *
  28   * The lifetime of this value is linked to the lifetime of the actual
  29   * process - the Process destructor calls self.finish() which waits
  30   * for the process to terminate.
  31   */
  32  pub struct Process {
  33      /// The unique id of the process (this should never be negative).
  34      priv pid: pid_t,
  35  
  36      /// A handle to the process - on unix this will always be NULL, but on
  37      /// windows it will be a HANDLE to the process, which will prevent the
  38      /// pid being re-used until the handle is closed.
  39      priv handle: *(),
  40  
  41      /// None until finish() is called.
  42      priv exit_code: Option<p::ProcessExit>,
  43  }

  45  impl Process {
  46      /// Creates a new process using native process-spawning abilities provided
  47      /// by the OS. Operations on this process will be blocking instead of using
  48      /// the runtime for sleeping just this current task.
  49      ///
  50      /// # Arguments
  51      ///
  52      /// * prog - the program to run
  53      /// * args - the arguments to pass to the program, not including the program
  54      ///          itself
  55      /// * env - an optional environment to specify for the child process. If
  56      ///         this value is `None`, then the child will inherit the parent's
  57      ///         environment
  58      /// * cwd - an optionally specified current working directory of the child,
  59      ///         defaulting to the parent's current working directory
  60      /// * stdin, stdout, stderr - These optionally specified file descriptors
  61      ///     dictate where the stdin/out/err of the child process will go. If
  62      ///     these are `None`, then this module will bind the input/output to an
  63      ///     os pipe instead. This process takes ownership of these file
  64      ///     descriptors, closing them upon destruction of the process.
  65      pub fn spawn(config: p::ProcessConfig)
  66          -> Result<(Process, ~[Option<file::FileDesc>]), io::IoError>
  67      {
  68          // right now we only handle stdin/stdout/stderr.
  69          if config.extra_io.len() > 0 {
  70              return Err(super::unimpl());
  71          }
  72  
  73          fn get_io(io: p::StdioContainer, ret: &mut ~[Option<file::FileDesc>])
  74              -> (Option<os::Pipe>, c_int)
  75          {
  76              match io {
  77                  p::Ignored => { ret.push(None); (None, -1) }
  78                  p::InheritFd(fd) => { ret.push(None); (None, fd) }
  79                  p::CreatePipe(readable, _writable) => {
  80                      let pipe = os::pipe();
  81                      let (theirs, ours) = if readable {
  82                          (pipe.input, pipe.out)
  83                      } else {
  84                          (pipe.out, pipe.input)
  85                      };
  86                      ret.push(Some(file::FileDesc::new(ours, true)));
  87                      (Some(pipe), theirs)
  88                  }
  89              }
  90          }
  91  
  92          let mut ret_io = ~[];
  93          let (in_pipe, in_fd) = get_io(config.stdin, &mut ret_io);
  94          let (out_pipe, out_fd) = get_io(config.stdout, &mut ret_io);
  95          let (err_pipe, err_fd) = get_io(config.stderr, &mut ret_io);
  96  
  97          let env = config.env.map(|a| a.to_owned());
  98          let cwd = config.cwd.map(|a| Path::new(a));
  99          let res = spawn_process_os(config, env, cwd.as_ref(), in_fd, out_fd,
 100                                     err_fd);
 101  
 102          unsafe {
 103              for pipe in in_pipe.iter() { let _ = libc::close(pipe.input); }
 104              for pipe in out_pipe.iter() { let _ = libc::close(pipe.out); }
 105              for pipe in err_pipe.iter() { let _ = libc::close(pipe.out); }
 106          }
 107  
 108          match res {
 109              Ok(res) => {
 110                  Ok((Process { pid: res.pid, handle: res.handle, exit_code: None },
 111                      ret_io))
 112              }
 113              Err(e) => Err(e)
 114          }
 115      }

 125      fn wait(&mut self) -> p::ProcessExit {
 126          match self.exit_code {
 127              Some(code) => code,
 128              None => {
 129                  let code = waitpid(self.pid);
 130                  self.exit_code = Some(code);
 131                  code
 132              }
 133          }
 134      }
 135  
 136      fn kill(&mut self, signum: int) -> Result<(), io::IoError> {
 137          // if the process has finished, and therefore had waitpid called,
 138          // and we kill it, then on unix we might ending up killing a
 139          // newer process that happens to have the same (re-used) id
 140          match self.exit_code {
 141              Some(..) => return Err(io::IoError {
 142                  kind: io::OtherIoError,
 143                  desc: "can't kill an exited process",
 144                  detail: None,
 145              }),
 146              None => {}
 147          }
 148          return unsafe { killpid(self.pid, signum) };
 149      }
 150  }

 158  #[cfg(windows)]
 159  unsafe fn killpid(pid: pid_t, signal: int) -> Result<(), io::IoError> {
 160      let handle = libc::OpenProcess(libc::PROCESS_TERMINATE |
 161                                     libc::PROCESS_QUERY_INFORMATION,
 162                                     libc::FALSE, pid as libc::DWORD);
 163      if handle.is_null() {
 164          return Err(super::last_error())
 165      }
 166      let ret = match signal {
 167          // test for existence on signal 0
 168          0 => {
 169              let mut status = 0;
 170              let ret = libc::GetExitCodeProcess(handle, &mut status);
 171              if ret == 0 {
 172                  Err(super::last_error())
 173              } else if status != libc::STILL_ACTIVE {
 174                  Err(io::IoError {
 175                      kind: io::OtherIoError,
 176                      desc: "process no longer alive",
 177                      detail: None,
 178                  })
 179              } else {
 180                  Ok(())
 181              }
 182          }
 183          io::process::PleaseExitSignal | io::process::MustDieSignal => {
 184              let ret = libc::TerminateProcess(handle, 1);
 185              super::mkerr_winbool(ret)
 186          }
 187          _ => Err(io::IoError {
 188              kind: io::OtherIoError,
 189              desc: "unsupported signal on windows",
 190              detail: None,
 191          })
 192      };
 193      let _ = libc::CloseHandle(handle);
 194      return ret;
 195  }

 208  #[cfg(windows)]
 209  fn spawn_process_os(config: p::ProcessConfig,
 210                      env: Option<~[(~str, ~str)]>,
 211                      dir: Option<&Path>,
 212                      in_fd: c_int, out_fd: c_int,
 213                      err_fd: c_int) -> IoResult<SpawnProcessResult> {
 214      use std::libc::types::os::arch::extra::{DWORD, HANDLE, STARTUPINFO};
 215      use std::libc::consts::os::extra::{
 216          TRUE, FALSE,
 217          STARTF_USESTDHANDLES,
 218          INVALID_HANDLE_VALUE,
 219          DUPLICATE_SAME_ACCESS
 220      };
 221      use std::libc::funcs::extra::kernel32::{
 222          GetCurrentProcess,
 223          DuplicateHandle,
 224          CloseHandle,
 225          CreateProcessA
 226      };
 227      use std::libc::funcs::extra::msvcrt::get_osfhandle;
 228  
 229      use std::mem;
 230  
 231      if config.gid.is_some() || config.uid.is_some() {
 232          return Err(io::IoError {
 233              kind: io::OtherIoError,
 234              desc: "unsupported gid/uid requested on windows",
 235              detail: None,
 236          })
 237      }
 238  
 239      unsafe {
 240  
 241          let mut si = zeroed_startupinfo();
 242          si.cb = mem::size_of::<STARTUPINFO>() as DWORD;
 243          si.dwFlags = STARTF_USESTDHANDLES;
 244  
 245          let cur_proc = GetCurrentProcess();
 246  
 247          let orig_std_in = get_osfhandle(in_fd) as HANDLE;
 248          if orig_std_in == INVALID_HANDLE_VALUE as HANDLE {
 249              fail!("failure in get_osfhandle: {}", os::last_os_error());
 250          }
 251          if DuplicateHandle(cur_proc, orig_std_in, cur_proc, &mut si.hStdInput,
 252                             0, TRUE, DUPLICATE_SAME_ACCESS) == FALSE {
 253              fail!("failure in DuplicateHandle: {}", os::last_os_error());
 254          }
 255  
 256          let orig_std_out = get_osfhandle(out_fd) as HANDLE;
 257          if orig_std_out == INVALID_HANDLE_VALUE as HANDLE {
 258              fail!("failure in get_osfhandle: {}", os::last_os_error());
 259          }
 260          if DuplicateHandle(cur_proc, orig_std_out, cur_proc, &mut si.hStdOutput,
 261                             0, TRUE, DUPLICATE_SAME_ACCESS) == FALSE {
 262              fail!("failure in DuplicateHandle: {}", os::last_os_error());
 263          }
 264  
 265          let orig_std_err = get_osfhandle(err_fd) as HANDLE;
 266          if orig_std_err == INVALID_HANDLE_VALUE as HANDLE {
 267              fail!("failure in get_osfhandle: {}", os::last_os_error());
 268          }
 269          if DuplicateHandle(cur_proc, orig_std_err, cur_proc, &mut si.hStdError,
 270                             0, TRUE, DUPLICATE_SAME_ACCESS) == FALSE {
 271              fail!("failure in DuplicateHandle: {}", os::last_os_error());
 272          }
 273  
 274          let cmd = make_command_line(config.program, config.args);
 275          let mut pi = zeroed_process_information();
 276          let mut create_err = None;
 277  
 278          // stolen from the libuv code.
 279          let mut flags = 0;
 280          if config.detach {
 281              flags |= libc::DETACHED_PROCESS | libc::CREATE_NEW_PROCESS_GROUP;
 282          }
 283  
 284          with_envp(env, |envp| {
 285              with_dirp(dir, |dirp| {
 286                  cmd.with_c_str(|cmdp| {
 287                      let created = CreateProcessA(ptr::null(), cast::transmute(cmdp),
 288                                                   ptr::mut_null(), ptr::mut_null(), TRUE,
 289                                                   flags, envp, dirp, &mut si,
 290                                                   &mut pi);
 291                      if created == FALSE {
 292                          create_err = Some(super::last_error());
 293                      }
 294                  })
 295              })
 296          });
 297  
 298          assert!(CloseHandle(si.hStdInput) != 0);
 299          assert!(CloseHandle(si.hStdOutput) != 0);
 300          assert!(CloseHandle(si.hStdError) != 0);
 301  
 302          match create_err {
 303              Some(err) => return Err(err),
 304              None => {}
 305          }
 306  
 307          // We close the thread handle because we don't care about keeping the
 308          // thread id valid, and we aren't keeping the thread handle around to be
 309          // able to close it later. We don't close the process handle however
 310          // because std::we want the process id to stay valid at least until the
 311          // calling code closes the process handle.
 312          assert!(CloseHandle(pi.hThread) != 0);
 313  
 314          Ok(SpawnProcessResult {
 315              pid: pi.dwProcessId as pid_t,
 316              handle: pi.hProcess as *()
 317          })
 318      }
 319  }

 355  #[cfg(windows)]
 356  fn make_command_line(prog: &str, args: &[~str]) -> ~str {
 357      let mut cmd = ~"";
 358      append_arg(&mut cmd, prog);
 359      for arg in args.iter() {
 360          cmd.push_char(' ');
 361          append_arg(&mut cmd, *arg);
 362      }
 363      return cmd;
 364  
 365      fn append_arg(cmd: &mut ~str, arg: &str) {
 366          let quote = arg.chars().any(|c| c == ' ' || c == '\t');
 367          if quote {
 368              cmd.push_char('"');
 369          }
 370          for i in range(0u, arg.len()) {
 371              append_char_at(cmd, arg, i);
 372          }
 373          if quote {
 374              cmd.push_char('"');
 375          }
 376      }
 377  
 378      fn append_char_at(cmd: &mut ~str, arg: &str, i: uint) {
 379          match arg[i] as char {
 380              '"' => {
 381                  // Escape quotes.
 382                  cmd.push_str("\\\"");
 383              }
 384              '\\' => {
 385                  if backslash_run_ends_in_quote(arg, i) {
 386                      // Double all backslashes that are in runs before quotes.
 387                      cmd.push_str("\\\\");
 388                  } else {
 389                      // Pass other backslashes through unescaped.
 390                      cmd.push_char('\\');
 391                  }
 392              }
 393              c => {
 394                  cmd.push_char(c);
 395              }
 396          }
 397      }
 398  
 399      fn backslash_run_ends_in_quote(s: &str, mut i: uint) -> bool {
 400          while i < s.len() && s[i] as char == '\\' {
 401              i += 1;
 402          }
 403          return i < s.len() && s[i] as char == '"';
 404      }
 405  }

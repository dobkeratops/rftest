  25  /**
  26   * A value representing a child process.
  27   *
  28   * The lifetime of this value is linked to the lifetime of the actual
  29   * process - the Process destructor calls self.finish() which waits
  30   * for the process to terminate.
  31   */
  32  pub struct Process {
  33      /// The unique id of the process (this should never be negative).
  34      pid: pid_t,
  35  
  36      /// A handle to the process - on unix this will always be NULL, but on
  37      /// windows it will be a HANDLE to the process, which will prevent the
  38      /// pid being re-used until the handle is closed.
  39      handle: *(),
  40  
  41      /// None until finish() is called.
  42      exit_code: Option<p::ProcessExit>,
  43  
  44      /// Manually delivered signal
  45      exit_signal: Option<int>,
  46  }

  48  impl Process {
  49      /// Creates a new process using native process-spawning abilities provided
  50      /// by the OS. Operations on this process will be blocking instead of using
  51      /// the runtime for sleeping just this current task.
  52      ///
  53      /// # Arguments
  54      ///
  55      /// * prog - the program to run
  56      /// * args - the arguments to pass to the program, not including the program
  57      ///          itself
  58      /// * env - an optional environment to specify for the child process. If
  59      ///         this value is `None`, then the child will inherit the parent's
  60      ///         environment
  61      /// * cwd - an optionally specified current working directory of the child,
  62      ///         defaulting to the parent's current working directory
  63      /// * stdin, stdout, stderr - These optionally specified file descriptors
  64      ///     dictate where the stdin/out/err of the child process will go. If
  65      ///     these are `None`, then this module will bind the input/output to an
  66      ///     os pipe instead. This process takes ownership of these file
  67      ///     descriptors, closing them upon destruction of the process.
  68      pub fn spawn(config: p::ProcessConfig)
  69          -> Result<(Process, ~[Option<file::FileDesc>]), io::IoError>
  70      {
  71          // right now we only handle stdin/stdout/stderr.
  72          if config.extra_io.len() > 0 {
  73              return Err(super::unimpl());
  74          }

  76          fn get_io(io: p::StdioContainer, ret: &mut ~[Option<file::FileDesc>])
  77              -> (Option<os::Pipe>, c_int)
  78          {
  79              match io {
  80                  p::Ignored => { ret.push(None); (None, -1) }
  81                  p::InheritFd(fd) => { ret.push(None); (None, fd) }
  82                  p::CreatePipe(readable, _writable) => {
  83                      let pipe = os::pipe();
  84                      let (theirs, ours) = if readable {
  85                          (pipe.input, pipe.out)
  86                      } else {
  87                          (pipe.out, pipe.input)
  88                      };
  89                      ret.push(Some(file::FileDesc::new(ours, true)));
  90                      (Some(pipe), theirs)
  91                  }
  92              }
  93          }
  94  
  95          let mut ret_io = ~[];
  96          let (in_pipe, in_fd) = get_io(config.stdin, &mut ret_io);
  97          let (out_pipe, out_fd) = get_io(config.stdout, &mut ret_io);
  98          let (err_pipe, err_fd) = get_io(config.stderr, &mut ret_io);
  99  
 100          let env = config.env.map(|a| a.to_owned());
 101          let cwd = config.cwd.map(|a| Path::new(a));
 102          let res = spawn_process_os(config, env, cwd.as_ref(), in_fd, out_fd,
 103                                     err_fd);
 104  
 105          unsafe {
 106              for pipe in in_pipe.iter() { let _ = libc::close(pipe.input); }
 107              for pipe in out_pipe.iter() { let _ = libc::close(pipe.out); }
 108              for pipe in err_pipe.iter() { let _ = libc::close(pipe.out); }
 109          }
 110  
 111          match res {
 112              Ok(res) => {
 113                  Ok((Process {
 114                          pid: res.pid,
 115                          handle: res.handle,
 116                          exit_code: None,
 117                          exit_signal: None,
 118                      },
 119                      ret_io))
 120              }
 121              Err(e) => Err(e)
 122          }
 123      }

 133      fn wait(&mut self) -> p::ProcessExit {
 134          match self.exit_code {
 135              Some(code) => code,
 136              None => {
 137                  let code = waitpid(self.pid);
 138                  // On windows, waitpid will never return a signal. If a signal
 139                  // was successfully delivered to the process, however, we can
 140                  // consider it as having died via a signal.
 141                  let code = match self.exit_signal {
 142                      None => code,
 143                      Some(signal) if cfg!(windows) => p::ExitSignal(signal),
 144                      Some(..) => code,
 145                  };
 146                  self.exit_code = Some(code);
 147                  code
 148              }
 149          }
 150      }

 152      fn kill(&mut self, signum: int) -> Result<(), io::IoError> {
 153          // On linux (and possibly other unices), a process that has exited will
 154          // continue to accept signals because it is "defunct". The delivery of
 155          // signals will only fail once the child has been reaped. For this
 156          // reason, if the process hasn't exited yet, then we attempt to collect
 157          // their status with WNOHANG.
 158          if self.exit_code.is_none() {
 159              match waitpid_nowait(self.pid) {
 160                  Some(code) => { self.exit_code = Some(code); }
 161                  None => {}
 162              }
 163          }
 164  
 165          // if the process has finished, and therefore had waitpid called,
 166          // and we kill it, then on unix we might ending up killing a
 167          // newer process that happens to have the same (re-used) id
 168          match self.exit_code {
 169              Some(..) => return Err(io::IoError {
 170                  kind: io::OtherIoError,
 171                  desc: "can't kill an exited process",
 172                  detail: None,
 173              }),
 174              None => {}
 175          }
 176  
 177          // A successfully delivered signal that isn't 0 (just a poll for being
 178          // alive) is recorded for windows (see wait())
 179          match unsafe { killpid(self.pid, signum) } {
 180              Ok(()) if signum == 0 => Ok(()),
 181              Ok(()) => { self.exit_signal = Some(signum); Ok(()) }
 182              Err(e) => Err(e),
 183          }
 184      }
 185  }

 193  #[cfg(windows)]
 194  unsafe fn killpid(pid: pid_t, signal: int) -> Result<(), io::IoError> {
 195      let handle = libc::OpenProcess(libc::PROCESS_TERMINATE |
 196                                     libc::PROCESS_QUERY_INFORMATION,
 197                                     libc::FALSE, pid as libc::DWORD);
 198      if handle.is_null() {
 199          return Err(super::last_error())
 200      }
 201      let ret = match signal {
 202          // test for existence on signal 0
 203          0 => {
 204              let mut status = 0;
 205              let ret = libc::GetExitCodeProcess(handle, &mut status);
 206              if ret == 0 {
 207                  Err(super::last_error())
 208              } else if status != libc::STILL_ACTIVE {
 209                  Err(io::IoError {
 210                      kind: io::OtherIoError,
 211                      desc: "process no longer alive",
 212                      detail: None,
 213                  })
 214              } else {
 215                  Ok(())
 216              }
 217          }
 218          io::process::PleaseExitSignal | io::process::MustDieSignal => {
 219              let ret = libc::TerminateProcess(handle, 1);
 220              super::mkerr_winbool(ret)
 221          }
 222          _ => Err(io::IoError {
 223              kind: io::OtherIoError,
 224              desc: "unsupported signal on windows",
 225              detail: None,
 226          })
 227      };
 228      let _ = libc::CloseHandle(handle);
 229      return ret;
 230  }

 243  #[cfg(windows)]
 244  fn spawn_process_os(config: p::ProcessConfig,
 245                      env: Option<~[(~str, ~str)]>,
 246                      dir: Option<&Path>,
 247                      in_fd: c_int, out_fd: c_int,
 248                      err_fd: c_int) -> IoResult<SpawnProcessResult> {
 249      use libc::types::os::arch::extra::{DWORD, HANDLE, STARTUPINFO};
 250      use libc::consts::os::extra::{
 251          TRUE, FALSE,
 252          STARTF_USESTDHANDLES,
 253          INVALID_HANDLE_VALUE,
 254          DUPLICATE_SAME_ACCESS
 255      };
 256      use libc::funcs::extra::kernel32::{
 257          GetCurrentProcess,
 258          DuplicateHandle,
 259          CloseHandle,
 260          CreateProcessA
 261      };
 262      use libc::funcs::extra::msvcrt::get_osfhandle;
 263  
 264      use std::mem;
 265  
 266      if config.gid.is_some() || config.uid.is_some() {
 267          return Err(io::IoError {
 268              kind: io::OtherIoError,
 269              desc: "unsupported gid/uid requested on windows",
 270              detail: None,
 271          })
 272      }
 273  
 274      unsafe {
 275  
 276          let mut si = zeroed_startupinfo();
 277          si.cb = mem::size_of::<STARTUPINFO>() as DWORD;
 278          si.dwFlags = STARTF_USESTDHANDLES;
 279  
 280          let cur_proc = GetCurrentProcess();
 281  
 282          if in_fd != -1 {
 283              let orig_std_in = get_osfhandle(in_fd) as HANDLE;
 284              if orig_std_in == INVALID_HANDLE_VALUE as HANDLE {
 285                  fail!("failure in get_osfhandle: {}", os::last_os_error());
 286              }
 287              if DuplicateHandle(cur_proc, orig_std_in, cur_proc, &mut si.hStdInput,
 288                                 0, TRUE, DUPLICATE_SAME_ACCESS) == FALSE {
 289                  fail!("failure in DuplicateHandle: {}", os::last_os_error());
 290              }
 291          }
 292  
 293          if out_fd != -1 {
 294              let orig_std_out = get_osfhandle(out_fd) as HANDLE;
 295              if orig_std_out == INVALID_HANDLE_VALUE as HANDLE {
 296                  fail!("failure in get_osfhandle: {}", os::last_os_error());
 297              }
 298              if DuplicateHandle(cur_proc, orig_std_out, cur_proc, &mut si.hStdOutput,
 299                                 0, TRUE, DUPLICATE_SAME_ACCESS) == FALSE {
 300                  fail!("failure in DuplicateHandle: {}", os::last_os_error());
 301              }
 302          }
 303  
 304          if err_fd != -1 {
 305              let orig_std_err = get_osfhandle(err_fd) as HANDLE;
 306              if orig_std_err == INVALID_HANDLE_VALUE as HANDLE {
 307                  fail!("failure in get_osfhandle: {}", os::last_os_error());
 308              }
 309              if DuplicateHandle(cur_proc, orig_std_err, cur_proc, &mut si.hStdError,
 310                                 0, TRUE, DUPLICATE_SAME_ACCESS) == FALSE {
 311                  fail!("failure in DuplicateHandle: {}", os::last_os_error());
 312              }
 313          }
 314  
 315          let cmd = make_command_line(config.program, config.args);
 316          let mut pi = zeroed_process_information();
 317          let mut create_err = None;
 318  
 319          // stolen from the libuv code.
 320          let mut flags = 0;
 321          if config.detach {
 322              flags |= libc::DETACHED_PROCESS | libc::CREATE_NEW_PROCESS_GROUP;
 323          }
 324  
 325          with_envp(env, |envp| {
 326              with_dirp(dir, |dirp| {
 327                  cmd.with_c_str(|cmdp| {
 328                      let created = CreateProcessA(ptr::null(), cast::transmute(cmdp),
 329                                                   ptr::mut_null(), ptr::mut_null(), TRUE,
 330                                                   flags, envp, dirp, &mut si,
 331                                                   &mut pi);
 332                      if created == FALSE {
 333                          create_err = Some(super::last_error());
 334                      }
 335                  })
 336              })
 337          });
 338  
 339          if in_fd != -1 { assert!(CloseHandle(si.hStdInput) != 0); }
 340          if out_fd != -1 { assert!(CloseHandle(si.hStdOutput) != 0); }
 341          if err_fd != -1 { assert!(CloseHandle(si.hStdError) != 0); }
 342  
 343          match create_err {
 344              Some(err) => return Err(err),
 345              None => {}
 346          }
 347  
 348          // We close the thread handle because we don't care about keeping the
 349          // thread id valid, and we aren't keeping the thread handle around to be
 350          // able to close it later. We don't close the process handle however
 351          // because std::we want the process id to stay valid at least until the
 352          // calling code closes the process handle.
 353          assert!(CloseHandle(pi.hThread) != 0);
 354  
 355          Ok(SpawnProcessResult {
 356              pid: pi.dwProcessId as pid_t,
 357              handle: pi.hProcess as *()
 358          })
 359      }
 360  }

 396  #[cfg(windows)]
 397  fn make_command_line(prog: &str, args: &[~str]) -> ~str {
 398      let mut cmd = ~"";
 399      append_arg(&mut cmd, prog);
 400      for arg in args.iter() {
 401          cmd.push_char(' ');
 402          append_arg(&mut cmd, *arg);
 403      }
 404      return cmd;
 405  
 406      fn append_arg(cmd: &mut ~str, arg: &str) {
 407          let quote = arg.chars().any(|c| c == ' ' || c == '\t');
 408          if quote {
 409              cmd.push_char('"');
 410          }
 411          for i in range(0u, arg.len()) {
 412              append_char_at(cmd, arg, i);
 413          }
 414          if quote {
 415              cmd.push_char('"');
 416          }
 417      }
 418  
 419      fn append_char_at(cmd: &mut ~str, arg: &str, i: uint) {
 420          match arg[i] as char {
 421              '"' => {
 422                  // Escape quotes.
 423                  cmd.push_str("\\\"");
 424              }
 425              '\\' => {
 426                  if backslash_run_ends_in_quote(arg, i) {
 427                      // Double all backslashes that are in runs before quotes.
 428                      cmd.push_str("\\\\");
 429                  } else {
 430                      // Pass other backslashes through unescaped.
 431                      cmd.push_char('\\');
 432                  }
 433              }
 434              c => {
 435                  cmd.push_char(c);
 436              }
 437          }
 438      }
 439  
 440      fn backslash_run_ends_in_quote(s: &str, mut i: uint) -> bool {
 441          while i < s.len() && s[i] as char == '\\' {
 442              i += 1;
 443          }
 444          return i < s.len() && s[i] as char == '"';
 445      }
 446  }

 452                      in_fd: c_int, out_fd: c_int,
 453                      err_fd: c_int) -> IoResult<SpawnProcessResult> {
 454      use libc::funcs::posix88::unistd::{fork, dup2, close, chdir, execvp};
 455      use libc::funcs::bsd44::getdtablesize;
 456      use libc::c_ulong;

 476      unsafe fn set_cloexec(fd: c_int) {
 477          extern { fn ioctl(fd: c_int, req: c_ulong) -> c_int; }
 478  
 479          #[cfg(target_os = "macos")]
 480          #[cfg(target_os = "freebsd")]
 481          static FIOCLEX: c_ulong = 0x20006601;
 482          #[cfg(target_os = "linux")]
 483          #[cfg(target_os = "android")]
 484          static FIOCLEX: c_ulong = 0x5451;
 485  
 486          let ret = ioctl(fd, FIOCLEX);
 487          assert_eq!(ret, 0);
 488      }
 489  
 490      let dirp = dir.map(|p| p.to_c_str());
 491      let dirp = dirp.as_ref().map(|c| c.with_ref(|p| p)).unwrap_or(ptr::null());
 492  
 493      with_envp(env, proc(envp) {
 494          with_argv(config.program, config.args, proc(argv) unsafe {
 495              let pipe = os::pipe();
 496              let mut input = file::FileDesc::new(pipe.input, true);
 497              let mut output = file::FileDesc::new(pipe.out, true);
 498  
 499              set_cloexec(output.fd());
 500  
 501              let pid = fork();
 502              if pid < 0 {
 503                  fail!("failure in fork: {}", os::last_os_error());
 504              } else if pid > 0 {
 505                  drop(output);
 506                  let mut bytes = [0, ..4];
 507                  return match input.inner_read(bytes) {
 508                      Ok(4) => {
 509                          let errno = (bytes[0] << 24) as i32 |
 510                                      (bytes[1] << 16) as i32 |
 511                                      (bytes[2] <<  8) as i32 |
 512                                      (bytes[3] <<  0) as i32;
 513                          Err(io::IoError::from_errno(errno as uint, false))
 514                      }
 515                      Err(e) => {
 516                          assert!(e.kind == io::BrokenPipe ||
 517                                  e.kind == io::EndOfFile,
 518                                  "unexpected error: {}", e);
 519                          Ok(SpawnProcessResult {
 520                              pid: pid,
 521                              handle: ptr::null()
 522                          })
 523                      }
 524                      Ok(..) => fail!("short read on the cloexec pipe"),
 525                  };
 526              }
 527              // And at this point we've reached a special time in the life of the
 528              // child. The child must now be considered hamstrung and unable to
 529              // do anything other than syscalls really. Consider the following
 530              // scenario:
 531              //
 532              //      1. Thread A of process 1 grabs the malloc() mutex
 533              //      2. Thread B of process 1 forks(), creating thread C
 534              //      3. Thread C of process 2 then attempts to malloc()
 535              //      4. The memory of process 2 is the same as the memory of
 536              //         process 1, so the mutex is locked.
 537              //
 538              // This situation looks a lot like deadlock, right? It turns out
 539              // that this is what pthread_atfork() takes care of, which is
 540              // presumably implemented across platforms. The first thing that
 541              // threads to *before* forking is to do things like grab the malloc
 542              // mutex, and then after the fork they unlock it.
 543              //
 544              // Despite this information, libnative's spawn has been witnessed to
 545              // deadlock on both OSX and FreeBSD. I'm not entirely sure why, but
 546              // all collected backtraces point at malloc/free traffic in the
 547              // child spawned process.
 548              //
 549              // For this reason, the block of code below should contain 0
 550              // invocations of either malloc of free (or their related friends).
 551              //
 552              // As an example of not having malloc/free traffic, we don't close
 553              // this file descriptor by dropping the FileDesc (which contains an
 554              // allocation). Instead we just close it manually. This will never
 555              // have the drop glue anyway because this code never returns (the
 556              // child will either exec() or invoke libc::exit)
 557              let _ = libc::close(input.fd());
 558  
 559              fn fail(output: &mut file::FileDesc) -> ! {
 560                  let errno = os::errno();
 561                  let bytes = [
 562                      (errno << 24) as u8,
 563                      (errno << 16) as u8,
 564                      (errno <<  8) as u8,
 565                      (errno <<  0) as u8,
 566                  ];
 567                  assert!(output.inner_write(bytes).is_ok());
 568                  unsafe { libc::_exit(1) }
 569              }
 570  
 571              rustrt::rust_unset_sigprocmask();
 572  
 573              if in_fd == -1 {
 574                  let _ = libc::close(libc::STDIN_FILENO);
 575              } else if retry(|| dup2(in_fd, 0)) == -1 {
 576                  fail(&mut output);
 577              }
 578              if out_fd == -1 {
 579                  let _ = libc::close(libc::STDOUT_FILENO);
 580              } else if retry(|| dup2(out_fd, 1)) == -1 {
 581                  fail(&mut output);
 582              }
 583              if err_fd == -1 {
 584                  let _ = libc::close(libc::STDERR_FILENO);
 585              } else if retry(|| dup2(err_fd, 2)) == -1 {
 586                  fail(&mut output);
 587              }
 588              // close all other fds
 589              for fd in range(3, getdtablesize()).rev() {
 590                  if fd != output.fd() {
 591                      let _ = close(fd as c_int);
 592                  }
 593              }
 594  
 595              match config.gid {
 596                  Some(u) => {
 597                      if libc::setgid(u as libc::gid_t) != 0 {
 598                          fail(&mut output);
 599                      }
 600                  }
 601                  None => {}
 602              }
 603              match config.uid {
 604                  Some(u) => {
 605                      // When dropping privileges from root, the `setgroups` call will
 606                      // remove any extraneous groups. If we don't call this, then
 607                      // even though our uid has dropped, we may still have groups
 608                      // that enable us to do super-user things. This will fail if we
 609                      // aren't root, so don't bother checking the return value, this
 610                      // is just done as an optimistic privilege dropping function.
 611                      extern {
 612                          fn setgroups(ngroups: libc::c_int,
 613                                       ptr: *libc::c_void) -> libc::c_int;
 614                      }
 615                      let _ = setgroups(0, 0 as *libc::c_void);
 616  
 617                      if libc::setuid(u as libc::uid_t) != 0 {
 618                          fail(&mut output);
 619                      }
 620                  }
 621                  None => {}
 622              }
 623              if config.detach {
 624                  // Don't check the error of setsid because it fails if we're the
 625                  // process leader already. We just forked so it shouldn't return
 626                  // error, but ignore it anyway.
 627                  let _ = libc::setsid();
 628              }
 629              if !dirp.is_null() && chdir(dirp) == -1 {
 630                  fail(&mut output);
 631              }
 632              if !envp.is_null() {
 633                  set_environ(envp);
 634              }
 635              let _ = execvp(*argv, argv);
 636              fail(&mut output);
 637          })
 638      })
 639  }

 641  #[cfg(unix)]
 642  fn with_argv<T>(prog: &str, args: &[~str], cb: proc(**libc::c_char) -> T) -> T {
 643      use std::slice;
 644  
 645      // We can't directly convert `str`s into `*char`s, as someone needs to hold
 646      // a reference to the intermediary byte buffers. So first build an array to
 647      // hold all the ~[u8] byte strings.
 648      let mut tmps = slice::with_capacity(args.len() + 1);
 649  
 650      tmps.push(prog.to_c_str());
 651  
 652      for arg in args.iter() {
 653          tmps.push(arg.to_c_str());
 654      }
 655  
 656      // Next, convert each of the byte strings into a pointer. This is
 657      // technically unsafe as the caller could leak these pointers out of our
 658      // scope.
 659      let mut ptrs: Vec<_> = tmps.iter().map(|tmp| tmp.with_ref(|buf| buf)).collect();
 660  
 661      // Finally, make sure we add a null pointer.
 662      ptrs.push(ptr::null());
 663  
 664      cb(ptrs.as_ptr())
 665  }

 667  #[cfg(unix)]
 668  fn with_envp<T>(env: Option<~[(~str, ~str)]>, cb: proc(*c_void) -> T) -> T {
 669      use std::slice;
 670  
 671      // On posixy systems we can pass a char** for envp, which is a
 672      // null-terminated array of "k=v\n" strings. Like `with_argv`, we have to
 673      // have a temporary buffer to hold the intermediary `~[u8]` byte strings.
 674      match env {
 675          Some(env) => {
 676              let mut tmps = slice::with_capacity(env.len());
 677  
 678              for pair in env.iter() {
 679                  let kv = format!("{}={}", *pair.ref0(), *pair.ref1());
 680                  tmps.push(kv.to_c_str());
 681              }
 682  
 683              // Once again, this is unsafe.
 684              let mut ptrs: Vec<*libc::c_char> = tmps.iter()
 685                                                     .map(|tmp| tmp.with_ref(|buf| buf))
 686                                                     .collect();
 687              ptrs.push(ptr::null());
 688  
 689              cb(ptrs.as_ptr() as *c_void)
 690          }
 691          _ => cb(ptr::null())
 692      }
 693  }

 696  fn with_envp<T>(env: Option<~[(~str, ~str)]>, cb: |*mut c_void| -> T) -> T {
 697      // On win32 we pass an "environment block" which is not a char**, but
 698      // rather a concatenation of null-terminated k=v\0 sequences, with a final
 699      // \0 to terminate.
 700      match env {
 701          Some(env) => {
 702              let mut blk = ~[];
 703  
 704              for pair in env.iter() {
 705                  let kv = format!("{}={}", *pair.ref0(), *pair.ref1());
 706                  blk.push_all(kv.as_bytes());
 707                  blk.push(0);
 708              }
 709  
 710              blk.push(0);
 711  
 712              cb(blk.as_mut_ptr() as *mut c_void)
 713          }
 714          _ => cb(ptr::mut_null())
 715      }
 716  }

 738  #[cfg(unix)]
 739  fn translate_status(status: c_int) -> p::ProcessExit {
 740      #[cfg(target_os = "linux")]
 741      #[cfg(target_os = "android")]
 742      mod imp {
 743          pub fn WIFEXITED(status: i32) -> bool { (status & 0xff) == 0 }
 744          pub fn WEXITSTATUS(status: i32) -> i32 { (status >> 8) & 0xff }
 745          pub fn WTERMSIG(status: i32) -> i32 { status & 0x7f }
 746      }
 747  
 748      #[cfg(target_os = "macos")]
 749      #[cfg(target_os = "freebsd")]
 750      mod imp {
 751          pub fn WIFEXITED(status: i32) -> bool { (status & 0x7f) == 0 }
 752          pub fn WEXITSTATUS(status: i32) -> i32 { status >> 8 }
 753          pub fn WTERMSIG(status: i32) -> i32 { status & 0o177 }
 754      }
 755  
 756      if imp::WIFEXITED(status) {
 757          p::ExitStatus(imp::WEXITSTATUS(status) as int)
 758      } else {
 759          p::ExitSignal(imp::WTERMSIG(status) as int)
 760      }
 761  }

 763  /**
 764   * Waits for a process to exit and returns the exit code, failing
 765   * if there is no process with the specified id.
 766   *
 767   * Note that this is private to avoid race conditions on unix where if
 768   * a user calls waitpid(some_process.get_id()) then some_process.finish()
 769   * and some_process.destroy() and some_process.finalize() will then either
 770   * operate on a none-existent process or, even worse, on a newer process
 771   * with the same id.
 772   */
 773  fn waitpid(pid: pid_t) -> p::ProcessExit {
 774      return waitpid_os(pid);

 776      #[cfg(windows)]
 777      fn waitpid_os(pid: pid_t) -> p::ProcessExit {
 778          use libc::types::os::arch::extra::DWORD;
 779          use libc::consts::os::extra::{
 780              SYNCHRONIZE,
 781              PROCESS_QUERY_INFORMATION,
 782              FALSE,
 783              STILL_ACTIVE,
 784              INFINITE,
 785              WAIT_FAILED
 786          };
 787          use libc::funcs::extra::kernel32::{
 788              OpenProcess,
 789              GetExitCodeProcess,
 790              CloseHandle,
 791              WaitForSingleObject
 792          };
 793  
 794          unsafe {
 795  
 796              let process = OpenProcess(SYNCHRONIZE | PROCESS_QUERY_INFORMATION,
 797                                        FALSE,
 798                                        pid as DWORD);
 799              if process.is_null() {
 800                  fail!("failure in OpenProcess: {}", os::last_os_error());
 801              }
 802  
 803              loop {
 804                  let mut status = 0;
 805                  if GetExitCodeProcess(process, &mut status) == FALSE {
 806                      assert!(CloseHandle(process) != 0);
 807                      fail!("failure in GetExitCodeProcess: {}", os::last_os_error());
 808                  }
 809                  if status != STILL_ACTIVE {
 810                      assert!(CloseHandle(process) != 0);
 811                      return p::ExitStatus(status as int);
 812                  }
 813                  if WaitForSingleObject(process, INFINITE) == WAIT_FAILED {
 814                      assert!(CloseHandle(process) != 0);
 815                      fail!("failure in WaitForSingleObject: {}", os::last_os_error());
 816                  }
 817              }
 818          }
 819      }

 824          let mut status = 0 as c_int;

 826              -1 => fail!("unknown waitpid error: {}", super::last_error()),
 827              _ => translate_status(status),
 828          }
 829      }
 830  }

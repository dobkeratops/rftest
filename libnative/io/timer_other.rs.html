  11  //! Timers for non-linux/non-windows OSes
  12  //!
  13  //! This module implements timers with a worker thread, select(), and a lot of
  14  //! witchcraft that turns out to be horribly inaccurate timers. The unfortunate
  15  //! part is that I'm at a loss of what else to do one these OSes. This is also
  16  //! why linux has a specialized timerfd implementation and windows has its own
  17  //! implementation (they're more accurate than this one).
  18  //!
  19  //! The basic idea is that there is a worker thread that's communicated to via a
  20  //! channel and a pipe, the pipe is used by the worker thread in a select()
  21  //! syscall with a timeout. The timeout is the "next timer timeout" while the
  22  //! channel is used to send data over to the worker thread.
  23  //!
  24  //! Whenever the call to select() times out, then a channel receives a message.
  25  //! Whenever the call returns that the file descriptor has information, then the
  26  //! channel from timers is drained, enqueueing all incoming requests.
  27  //!
  28  //! The actual implementation of the helper thread is a sorted array of
  29  //! timers in terms of target firing date. The target is the absolute time at
  30  //! which the timer should fire. Timers are then re-enqueued after a firing if
  31  //! the repeat boolean is set.
  32  //!
  33  //! Naturally, all this logic of adding times and keeping track of
  34  //! relative/absolute time is a little lossy and not quite exact. I've done the
  35  //! best I could to reduce the amount of calls to 'now()', but there's likely
  36  //! still inaccuracies trickling in here and there.
  37  //!
  38  //! One of the tricky parts of this implementation is that whenever a timer is
  39  //! acted upon, it must cancel whatever the previous action was (if one is
  40  //! active) in order to act like the other implementations of this timer. In
  41  //! order to do this, the timer's inner pointer is transferred to the worker
  42  //! thread. Whenever the timer is modified, it first takes ownership back from
  43  //! the worker thread in order to modify the same data structure. This has the
  44  //! side effect of "cancelling" the previous requests while allowing a
  45  //! re-enqueueing later on.
  46  //!
  47  //! Note that all time units in this file are in *milliseconds*.
  48

  61  pub struct Timer {
  62      priv id: uint,
  63      priv inner: Option<~Inner>,
  64  }
  65  
  66  struct Inner {
  67      tx: Option<Sender<()>>,
  68      interval: u64,
  69      repeat: bool,
  70      target: u64,
  71      id: uint,
  72  }
  73  
  74  #[allow(visible_private_types)]
  75  pub enum Req {
  76      // Add a new timer to the helper thread.
  77      NewTimer(~Inner),
  78  
  79      // Remove a timer based on its id and then send it back on the channel
  80      // provided
  81      RemoveTimer(uint, Sender<~Inner>),
  82  
  83      // Shut down the loop and then ACK this channel once it's shut down
  84      Shutdown,
  85  }
  86  
  87  // returns the current time (in milliseconds)
  88  fn now() -> u64 {
  89      unsafe {
  90          let mut now: libc::timeval = mem::init();
  91          assert_eq!(imp::gettimeofday(&mut now, ptr::null()), 0);
  92          return (now.tv_sec as u64) * 1000 + (now.tv_usec as u64) / 1000;
  93      }
  94  }
  95  
  96  fn helper(input: libc::c_int, messages: Receiver<Req>) {
  97      let mut set: imp::fd_set = unsafe { mem::init() };
  98  
  99      let mut fd = FileDesc::new(input, true);
 100      let mut timeout: libc::timeval = unsafe { mem::init() };
 101  
 102      // active timers are those which are able to be selected upon (and it's a
 103      // sorted list, and dead timers are those which have expired, but ownership
 104      // hasn't yet been transferred back to the timer itself.
 105      let mut active: ~[~Inner] = ~[];
 106      let mut dead = ~[];
 107  
 108      // inserts a timer into an array of timers (sorted by firing time)
 109      fn insert(t: ~Inner, active: &mut ~[~Inner]) {
 110          match active.iter().position(|tm| tm.target > t.target) {
 111              Some(pos) => { active.insert(pos, t); }
 112              None => { active.push(t); }
 113          }
 114      }
 115  
 116      // signals the first requests in the queue, possible re-enqueueing it.
 117      fn signal(active: &mut ~[~Inner], dead: &mut ~[(uint, ~Inner)]) {
 118          let mut timer = match active.shift() {
 119              Some(timer) => timer, None => return
 120          };
 121          let tx = timer.tx.take_unwrap();
 122          if tx.try_send(()) && timer.repeat {
 123              timer.tx = Some(tx);
 124              timer.target += timer.interval;
 125              insert(timer, active);
 126          } else {
 127              drop(tx);
 128              dead.push((timer.id, timer));
 129          }
 130      }
 131  
 132      'outer: loop {
 133          let timeout = if active.len() == 0 {
 134              // Empty array? no timeout (wait forever for the next request)
 135              ptr::null()
 136          } else {
 137              let now = now();
 138              // If this request has already expired, then signal it and go
 139              // through another iteration
 140              if active[0].target <= now {
 141                  signal(&mut active, &mut dead);
 142                  continue;
 143              }
 144  
 145              // The actual timeout listed in the requests array is an
 146              // absolute date, so here we translate the absolute time to a
 147              // relative time.
 148              let tm = active[0].target - now;
 149              timeout.tv_sec = (tm / 1000) as libc::time_t;
 150              timeout.tv_usec = ((tm % 1000) * 1000) as libc::suseconds_t;
 151              &timeout as *libc::timeval
 152          };
 153  
 154          imp::fd_set(&mut set, input);
 155          match unsafe {
 156              imp::select(input + 1, &set, ptr::null(), ptr::null(), timeout)
 157          } {
 158              // timed out
 159              0 => signal(&mut active, &mut dead),
 160  
 161              // file descriptor write woke us up, we've got some new requests
 162              1 => {
 163                  loop {
 164                      match messages.try_recv() {
 165                          Data(Shutdown) => {
 166                              assert!(active.len() == 0);
 167                              break 'outer;
 168                          }
 169  
 170                          Data(NewTimer(timer)) => insert(timer, &mut active),
 171  
 172                          Data(RemoveTimer(id, ack)) => {
 173                              match dead.iter().position(|&(i, _)| id == i) {
 174                                  Some(i) => {
 175                                      let (_, i) = dead.remove(i).unwrap();
 176                                      ack.send(i);
 177                                      continue
 178                                  }
 179                                  None => {}
 180                              }
 181                              let i = active.iter().position(|i| i.id == id);
 182                              let i = i.expect("no timer found");
 183                              let t = active.remove(i).unwrap();
 184                              ack.send(t);
 185                          }
 186                          _ => break
 187                      }
 188                  }
 189  
 190                  // drain the file descriptor
 191                  let mut buf = [0];
 192                  assert_eq!(fd.inner_read(buf).unwrap(), 1);
 193              }
 194  
 195              -1 if os::errno() == libc::EINTR as int => {}
 196              n => fail!("helper thread failed in select() with error: {} ({})",
 197                         n, os::last_os_error())
 198          }
 199      }
 200  }

 202  impl Timer {
 203      pub fn new() -> IoResult<Timer> {
 204          timer_helper::boot(helper);
 205  
 206          static mut ID: atomics::AtomicUint = atomics::INIT_ATOMIC_UINT;
 207          let id = unsafe { ID.fetch_add(1, atomics::Relaxed) };
 208          Ok(Timer {
 209              id: id,
 210              inner: Some(~Inner {
 211                  tx: None,
 212                  interval: 0,
 213                  target: 0,
 214                  repeat: false,
 215                  id: id,
 216              })
 217          })
 218      }
 219  
 220      pub fn sleep(ms: u64) {
 221          let mut to_sleep = libc::timespec {
 222              tv_sec: (ms / 1000) as libc::time_t,
 223              tv_nsec: ((ms % 1000) * 1000000) as libc::c_long,
 224          };
 225          while unsafe { libc::nanosleep(&to_sleep, &mut to_sleep) } != 0 {
 226              if os::errno() as int != libc::EINTR as int {
 227                  fail!("failed to sleep, but not because of EINTR?");
 228              }
 229          }
 230      }
 231  
 232      fn inner(&mut self) -> ~Inner {
 233          match self.inner.take() {
 234              Some(i) => i,
 235              None => {
 236                  let (tx, rx) = channel();
 237                  timer_helper::send(RemoveTimer(self.id, tx));
 238                  rx.recv()
 239              }
 240          }
 241      }
 242  }
 243  
 244  impl rtio::RtioTimer for Timer {
 245      fn sleep(&mut self, msecs: u64) {
 246          let mut inner = self.inner();
 247          inner.tx = None; // cancel any previous request
 248          self.inner = Some(inner);
 249  
 250          Timer::sleep(msecs);
 251      }
 252  
 253      fn oneshot(&mut self, msecs: u64) -> Receiver<()> {
 254          let now = now();
 255          let mut inner = self.inner();
 256  
 257          let (tx, rx) = channel();
 258          inner.repeat = false;
 259          inner.tx = Some(tx);
 260          inner.interval = msecs;
 261          inner.target = now + msecs;
 262  
 263          timer_helper::send(NewTimer(inner));
 264          return rx;
 265      }
 266  
 267      fn period(&mut self, msecs: u64) -> Receiver<()> {
 268          let now = now();
 269          let mut inner = self.inner();
 270  
 271          let (tx, rx) = channel();
 272          inner.repeat = true;
 273          inner.tx = Some(tx);
 274          inner.interval = msecs;
 275          inner.target = now + msecs;
 276  
 277          timer_helper::send(NewTimer(inner));
 278          return rx;
 279      }
 280  }
 281  
 282  impl Drop for Timer {
 283      fn drop(&mut self) {
 284          self.inner = Some(self.inner());
 285      }
 286  }

  40  fn ip_to_inaddr(ip: ip::IpAddr) -> InAddr {
  41      match ip {
  42          ip::Ipv4Addr(a, b, c, d) => {
  43              InAddr(libc::in_addr {
  44                  s_addr: (d as u32 << 24) |
  45                          (c as u32 << 16) |
  46                          (b as u32 <<  8) |
  47                          (a as u32 <<  0)
  48              })
  49          }
  50          ip::Ipv6Addr(a, b, c, d, e, f, g, h) => {
  51              In6Addr(libc::in6_addr {
  52                  s6_addr: [
  53                      htons(a),
  54                      htons(b),
  55                      htons(c),
  56                      htons(d),
  57                      htons(e),
  58                      htons(f),
  59                      htons(g),
  60                      htons(h),
  61                  ]
  62              })
  63          }
  64      }
  65  }

  67  fn addr_to_sockaddr(addr: ip::SocketAddr) -> (libc::sockaddr_storage, uint) {
  68      unsafe {
  69          let storage: libc::sockaddr_storage = mem::init();
  70          let len = match ip_to_inaddr(addr.ip) {
  71              InAddr(inaddr) => {
  72                  let storage: *mut libc::sockaddr_in = cast::transmute(&storage);
  73                  (*storage).sin_family = libc::AF_INET as libc::sa_family_t;
  74                  (*storage).sin_port = htons(addr.port);
  75                  (*storage).sin_addr = inaddr;
  76                  mem::size_of::<libc::sockaddr_in>()
  77              }
  78              In6Addr(inaddr) => {
  79                  let storage: *mut libc::sockaddr_in6 = cast::transmute(&storage);
  80                  (*storage).sin6_family = libc::AF_INET6 as libc::sa_family_t;
  81                  (*storage).sin6_port = htons(addr.port);
  82                  (*storage).sin6_addr = inaddr;
  83                  mem::size_of::<libc::sockaddr_in6>()
  84              }
  85          };
  86          return (storage, len);
  87      }
  88  }

  90  fn socket(addr: ip::SocketAddr, ty: libc::c_int) -> IoResult<sock_t> {
  91      unsafe {
  92          let fam = match addr.ip {
  93              ip::Ipv4Addr(..) => libc::AF_INET,
  94              ip::Ipv6Addr(..) => libc::AF_INET6,
  95          };
  96          match libc::socket(fam, ty, 0) {
  97              -1 => Err(super::last_error()),
  98              fd => Ok(fd),
  99          }
 100      }
 101  }

 103  fn setsockopt<T>(fd: sock_t, opt: libc::c_int, val: libc::c_int,
 104                   payload: T) -> IoResult<()> {
 105      unsafe {
 106          let payload = &payload as *T as *libc::c_void;
 107          let ret = libc::setsockopt(fd, opt, val,
 108                                     payload,
 109                                     mem::size_of::<T>() as libc::socklen_t);
 110          if ret != 0 {
 111              Err(last_error())
 112          } else {
 113              Ok(())
 114          }
 115      }
 116  }

 134  fn sockname(fd: sock_t,
 135              f: extern "system" unsafe fn(sock_t, *mut libc::sockaddr,
 136                                           *mut libc::socklen_t) -> libc::c_int)
 137      -> IoResult<ip::SocketAddr>
 138  {
 139      let mut storage: libc::sockaddr_storage = unsafe { mem::init() };
 140      let mut len = mem::size_of::<libc::sockaddr_storage>() as libc::socklen_t;
 141      unsafe {
 142          let storage = &mut storage as *mut libc::sockaddr_storage;
 143          let ret = f(fd,
 144                      storage as *mut libc::sockaddr,
 145                      &mut len as *mut libc::socklen_t);
 146          if ret != 0 {
 147              return Err(last_error())
 148          }
 149      }
 150      return sockaddr_to_addr(&storage, len as uint);
 151  }

 153  pub fn sockaddr_to_addr(storage: &libc::sockaddr_storage,
 154                          len: uint) -> IoResult<ip::SocketAddr> {
 155      match storage.ss_family as libc::c_int {
 156          libc::AF_INET => {
 157              assert!(len as uint >= mem::size_of::<libc::sockaddr_in>());
 158              let storage: &libc::sockaddr_in = unsafe {
 159                  cast::transmute(storage)
 160              };
 161              let addr = storage.sin_addr.s_addr as u32;
 162              let a = (addr >>  0) as u8;
 163              let b = (addr >>  8) as u8;
 164              let c = (addr >> 16) as u8;
 165              let d = (addr >> 24) as u8;
 166              Ok(ip::SocketAddr {
 167                  ip: ip::Ipv4Addr(a, b, c, d),
 168                  port: ntohs(storage.sin_port),
 169              })
 170          }
 171          libc::AF_INET6 => {
 172              assert!(len as uint >= mem::size_of::<libc::sockaddr_in6>());
 173              let storage: &libc::sockaddr_in6 = unsafe {
 174                  cast::transmute(storage)
 175              };
 176              let a = ntohs(storage.sin6_addr.s6_addr[0]);
 177              let b = ntohs(storage.sin6_addr.s6_addr[1]);
 178              let c = ntohs(storage.sin6_addr.s6_addr[2]);
 179              let d = ntohs(storage.sin6_addr.s6_addr[3]);
 180              let e = ntohs(storage.sin6_addr.s6_addr[4]);
 181              let f = ntohs(storage.sin6_addr.s6_addr[5]);
 182              let g = ntohs(storage.sin6_addr.s6_addr[6]);
 183              let h = ntohs(storage.sin6_addr.s6_addr[7]);
 184              Ok(ip::SocketAddr {
 185                  ip: ip::Ipv6Addr(a, b, c, d, e, f, g, h),
 186                  port: ntohs(storage.sin6_port),
 187              })
 188          }
 189          _ => {
 190              Err(io::standard_error(io::OtherIoError))
 191          }
 192      }
 193  }

 247  impl TcpStream {
 248      pub fn connect(addr: ip::SocketAddr) -> IoResult<TcpStream> {
 249          unsafe {
 250              socket(addr, libc::SOCK_STREAM).and_then(|fd| {
 251                  let (addr, len) = addr_to_sockaddr(addr);
 252                  let addrp = &addr as *libc::sockaddr_storage;
 253                  let inner = Inner { fd: fd };
 254                  let ret = TcpStream { inner: UnsafeArc::new(inner) };
 255                  match retry(|| {
 256                      libc::connect(fd, addrp as *libc::sockaddr,
 257                                    len as libc::socklen_t)
 258                  }) {
 259                      -1 => Err(last_error()),
 260                      _ => Ok(ret),
 261                  }
 262              })
 263          }
 264      }

 269      }

 271      fn set_nodelay(&mut self, nodelay: bool) -> IoResult<()> {

 276      fn set_keepalive(&mut self, seconds: Option<uint>) -> IoResult<()> {
 277          let ret = setsockopt(self.fd(), libc::SOL_SOCKET, libc::SO_KEEPALIVE,
 278                               seconds.is_some() as libc::c_int);
 279          match seconds {
 280              Some(n) => ret.and_then(|()| self.set_tcp_keepalive(n)),
 281              None => ret,
 282          }
 283      }

 301  #[cfg(windows)] type wrlen = libc::c_int;
 302  #[cfg(not(windows))] type wrlen = libc::size_t;
 303  
 304  impl rtio::RtioTcpStream for TcpStream {
 305      fn read(&mut self, buf: &mut [u8]) -> IoResult<uint> {
 306          let ret = retry(|| {
 307              unsafe {
 308                  libc::recv(self.fd(),
 309                             buf.as_mut_ptr() as *mut libc::c_void,
 310                             buf.len() as wrlen,
 311                             0) as libc::c_int
 312              }
 313          });
 314          if ret == 0 {
 315              Err(io::standard_error(io::EndOfFile))
 316          } else if ret < 0 {
 317              Err(last_error())
 318          } else {
 319              Ok(ret as uint)
 320          }
 321      }
 322      fn write(&mut self, buf: &[u8]) -> IoResult<()> {
 323          let ret = keep_going(buf, |buf, len| unsafe {
 324              libc::send(self.fd(),
 325                         buf as *mut libc::c_void,
 326                         len as wrlen,
 327                         0) as i64
 328          });
 329          if ret < 0 {
 330              Err(super::last_error())
 331          } else {
 332              Ok(())
 333          }
 334      }
 335      fn peer_name(&mut self) -> IoResult<ip::SocketAddr> {
 336          sockname(self.fd(), libc::getpeername)
 337      }
 338      fn control_congestion(&mut self) -> IoResult<()> {
 339          self.set_nodelay(false)
 340      }
 341      fn nodelay(&mut self) -> IoResult<()> {
 342          self.set_nodelay(true)
 343      }
 344      fn keepalive(&mut self, delay_in_seconds: uint) -> IoResult<()> {
 345          self.set_keepalive(Some(delay_in_seconds))
 346      }
 347      fn letdie(&mut self) -> IoResult<()> {
 348          self.set_keepalive(None)
 349      }

 351      fn clone(&self) -> ~rtio::RtioTcpStream {
 352          ~TcpStream { inner: self.inner.clone() } as ~rtio::RtioTcpStream
 353      }
 354      fn close_write(&mut self) -> IoResult<()> {
 355          super::mkerr_libc(unsafe {
 356              libc::shutdown(self.fd(), libc::SHUT_WR)

 358      }
 359  }

 375  pub struct TcpListener {
 376      priv inner: UnsafeArc<Inner>,
 377  }

 379  impl TcpListener {
 380      pub fn bind(addr: ip::SocketAddr) -> IoResult<TcpListener> {
 381          unsafe {
 382              socket(addr, libc::SOCK_STREAM).and_then(|fd| {
 383                  let (addr, len) = addr_to_sockaddr(addr);
 384                  let addrp = &addr as *libc::sockaddr_storage;
 385                  let inner = Inner { fd: fd };
 386                  let ret = TcpListener { inner: UnsafeArc::new(inner) };
 387                  // On platforms with Berkeley-derived sockets, this allows
 388                  // to quickly rebind a socket, without needing to wait for
 389                  // the OS to clean up the previous one.
 390                  if cfg!(unix) {
 391                      match setsockopt(fd, libc::SOL_SOCKET,
 392                                       libc::SO_REUSEADDR,
 393                                       1 as libc::c_int) {
 394                          Err(n) => { return Err(n); },
 395                          Ok(..) => { }
 396                      }
 397                  }
 398                  match libc::bind(fd, addrp as *libc::sockaddr,
 399                                   len as libc::socklen_t) {
 400                      -1 => Err(last_error()),
 401                      _ => Ok(ret),
 402                  }
 403              })
 404          }
 405      }

 410      }

 418  }

 445              match retry(|| {

 447                               storagep as *mut libc::sockaddr,
 448                               &mut size as *mut libc::socklen_t) as libc::c_int
 449              }) as sock_t {
 450                  -1 => Err(last_error()),

 453          }

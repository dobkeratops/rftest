  69  fn addr_to_sockaddr(addr: ip::SocketAddr) -> (libc::sockaddr_storage, uint) {
  70      unsafe {
  71          let storage: libc::sockaddr_storage = mem::init();
  72          let len = match ip_to_inaddr(addr.ip) {
  73              InAddr(inaddr) => {
  74                  let storage: *mut libc::sockaddr_in = cast::transmute(&storage);
  75                  (*storage).sin_family = libc::AF_INET as libc::sa_family_t;
  76                  (*storage).sin_port = htons(addr.port);
  77                  (*storage).sin_addr = inaddr;
  78                  mem::size_of::<libc::sockaddr_in>()
  79              }
  80              In6Addr(inaddr) => {
  81                  let storage: *mut libc::sockaddr_in6 = cast::transmute(&storage);
  82                  (*storage).sin6_family = libc::AF_INET6 as libc::sa_family_t;
  83                  (*storage).sin6_port = htons(addr.port);
  84                  (*storage).sin6_addr = inaddr;
  85                  mem::size_of::<libc::sockaddr_in6>()
  86              }
  87          };
  88          return (storage, len);
  89      }
  90  }

 105  fn setsockopt<T>(fd: sock_t, opt: libc::c_int, val: libc::c_int,
 106                   payload: T) -> IoResult<()> {
 107      unsafe {
 108          let payload = &payload as *T as *libc::c_void;
 109          let ret = libc::setsockopt(fd, opt, val,
 110                                     payload,
 111                                     mem::size_of::<T>() as libc::socklen_t);
 112          if ret != 0 {
 113              Err(last_error())
 114          } else {
 115              Ok(())
 116          }
 117      }
 118  }

 120  fn getsockopt<T: Copy>(fd: sock_t, opt: libc::c_int,
 121                         val: libc::c_int) -> IoResult<T> {
 122      unsafe {
 123          let mut slot: T = mem::init();
 124          let mut len = mem::size_of::<T>() as libc::socklen_t;
 125          let ret = c::getsockopt(fd, opt, val,
 126                                  &mut slot as *mut _ as *mut _,
 127                                  &mut len);
 128          if ret != 0 {
 129              Err(last_error())
 130          } else {
 131              assert!(len as uint == mem::size_of::<T>());
 132              Ok(slot)
 133          }
 134      }
 135  }

 169  pub fn sockaddr_to_addr(storage: &libc::sockaddr_storage,
 170                          len: uint) -> IoResult<ip::SocketAddr> {
 171      match storage.ss_family as libc::c_int {
 172          libc::AF_INET => {
 173              assert!(len as uint >= mem::size_of::<libc::sockaddr_in>());
 174              let storage: &libc::sockaddr_in = unsafe {
 175                  cast::transmute(storage)
 176              };
 177              let addr = storage.sin_addr.s_addr as u32;
 178              let a = (addr >>  0) as u8;
 179              let b = (addr >>  8) as u8;
 180              let c = (addr >> 16) as u8;
 181              let d = (addr >> 24) as u8;
 182              Ok(ip::SocketAddr {
 183                  ip: ip::Ipv4Addr(a, b, c, d),
 184                  port: ntohs(storage.sin_port),
 185              })
 186          }
 187          libc::AF_INET6 => {
 188              assert!(len as uint >= mem::size_of::<libc::sockaddr_in6>());
 189              let storage: &libc::sockaddr_in6 = unsafe {
 190                  cast::transmute(storage)
 191              };
 192              let a = ntohs(storage.sin6_addr.s6_addr[0]);
 193              let b = ntohs(storage.sin6_addr.s6_addr[1]);
 194              let c = ntohs(storage.sin6_addr.s6_addr[2]);
 195              let d = ntohs(storage.sin6_addr.s6_addr[3]);
 196              let e = ntohs(storage.sin6_addr.s6_addr[4]);
 197              let f = ntohs(storage.sin6_addr.s6_addr[5]);
 198              let g = ntohs(storage.sin6_addr.s6_addr[6]);
 199              let h = ntohs(storage.sin6_addr.s6_addr[7]);
 200              Ok(ip::SocketAddr {
 201                  ip: ip::Ipv6Addr(a, b, c, d, e, f, g, h),
 202                  port: ntohs(storage.sin6_port),
 203              })
 204          }
 205          _ => {
 206              Err(io::standard_error(io::OtherIoError))
 207          }
 208      }
 209  }

 246      pub fn connect(addr: ip::SocketAddr,
 247                     timeout: Option<u64>) -> IoResult<TcpStream> {
 248          let fd = try!(socket(addr, libc::SOCK_STREAM));
 249          let (addr, len) = addr_to_sockaddr(addr);
 250          let inner = Inner { fd: fd };
 251          let ret = TcpStream { inner: UnsafeArc::new(inner) };
 252  
 253          let len = len as libc::socklen_t;
 254          let addrp = &addr as *_ as *libc::sockaddr;
 255          match timeout {
 256              Some(timeout) => {
 257                  try!(TcpStream::connect_timeout(fd, addrp, len, timeout));
 258                  Ok(ret)
 259              },
 260              None => {
 261                  match retry(|| unsafe { libc::connect(fd, addrp, len) }) {
 262                      -1 => Err(last_error()),
 263                      _ => Ok(ret),
 264                  }
 265              }
 266          }
 267      }

 269      // See http://developerweb.net/viewtopic.php?id=3196 for where this is
 270      // derived from.
 271      fn connect_timeout(fd: sock_t,
 272                         addrp: *libc::sockaddr,
 273                         len: libc::socklen_t,
 274                         timeout: u64) -> IoResult<()> {
 275          use std::os;
 276          #[cfg(unix)]    use INPROGRESS = libc::EINPROGRESS;
 277          #[cfg(windows)] use INPROGRESS = libc::WSAEINPROGRESS;
 278          #[cfg(unix)]    use WOULDBLOCK = libc::EWOULDBLOCK;
 279          #[cfg(windows)] use WOULDBLOCK = libc::WSAEWOULDBLOCK;
 280  
 281          // Make sure the call to connect() doesn't block
 282          try!(set_nonblocking(fd, true));
 283  
 284          let ret = match unsafe { libc::connect(fd, addrp, len) } {
 285              // If the connection is in progress, then we need to wait for it to
 286              // finish (with a timeout). The current strategy for doing this is
 287              // to use select() with a timeout.
 288              -1 if os::errno() as int == INPROGRESS as int ||
 289                    os::errno() as int == WOULDBLOCK as int => {
 290                  let mut set: c::fd_set = unsafe { mem::init() };
 291                  c::fd_set(&mut set, fd);
 292                  match await(fd, &mut set, timeout) {
 293                      0 => Err(io::IoError {
 294                          kind: io::TimedOut,
 295                          desc: "connection timed out",
 296                          detail: None,
 297                      }),
 298                      -1 => Err(last_error()),
 299                      _ => {
 300                          let err: libc::c_int = try!(
 301                              getsockopt(fd, libc::SOL_SOCKET, libc::SO_ERROR));
 302                          if err == 0 {
 303                              Ok(())
 304                          } else {
 305                              Err(io::IoError::from_errno(err as uint, true))
 306                          }
 307                      }
 308                  }
 309              }
 310  
 311              -1 => Err(last_error()),
 312              _ => Ok(()),
 313          };
 314  
 315          // be sure to turn blocking I/O back on
 316          try!(set_nonblocking(fd, false));
 317          return ret;
 318  
 319          #[cfg(unix)]
 320          fn set_nonblocking(fd: sock_t, nb: bool) -> IoResult<()> {
 321              let set = nb as libc::c_int;
 322              super::mkerr_libc(retry(|| unsafe { c::ioctl(fd, c::FIONBIO, &set) }))
 323          }
 324          #[cfg(windows)]
 325          fn set_nonblocking(fd: sock_t, nb: bool) -> IoResult<()> {
 326              let mut set = nb as libc::c_ulong;
 327              if unsafe { c::ioctlsocket(fd, c::FIONBIO, &mut set) != 0 } {
 328                  Err(last_error())
 329              } else {
 330                  Ok(())
 331              }
 332          }
 333  
 334          #[cfg(unix)]
 335          fn await(fd: sock_t, set: &mut c::fd_set, timeout: u64) -> libc::c_int {
 336              let start = ::io::timer::now();
 337              retry(|| unsafe {
 338                  // Recalculate the timeout each iteration (it is generally
 339                  // undefined what the value of the 'tv' is after select
 340                  // returns EINTR).
 341                  let timeout = timeout - (::io::timer::now() - start);
 342                  let tv = libc::timeval {
 343                      tv_sec: (timeout / 1000) as libc::time_t,
 344                      tv_usec: ((timeout % 1000) * 1000) as libc::suseconds_t,
 345                  };
 346                  c::select(fd + 1, ptr::null(), set as *mut _ as *_,
 347                            ptr::null(), &tv)
 348              })
 349          }
 350          #[cfg(windows)]
 351          fn await(_fd: sock_t, set: &mut c::fd_set, timeout: u64) -> libc::c_int {
 352              let tv = libc::timeval {
 353                  tv_sec: (timeout / 1000) as libc::time_t,
 354                  tv_usec: ((timeout % 1000) * 1000) as libc::suseconds_t,
 355              };
 356              unsafe { c::select(1, ptr::mut_null(), set, ptr::mut_null(), &tv) }
 357          }
 358      }

 399      fn read(&mut self, buf: &mut [u8]) -> IoResult<uint> {
 400          let ret = retry(|| {
 401              unsafe {
 402                  libc::recv(self.fd(),
 403                             buf.as_mut_ptr() as *mut libc::c_void,
 404                             buf.len() as wrlen,
 405                             0) as libc::c_int
 406              }
 407          });
 408          if ret == 0 {
 409              Err(io::standard_error(io::EndOfFile))
 410          } else if ret < 0 {
 411              Err(last_error())
 412          } else {
 413              Ok(ret as uint)
 414          }
 415      }

 418              libc::send(self.fd(),
 419                         buf as *mut libc::c_void,
 420                         len as wrlen,
 421                         0) as i64
 422          });
 423          if ret < 0 {
 424              Err(super::last_error())
 425          } else {
 426              Ok(())
 427          }
 428      }

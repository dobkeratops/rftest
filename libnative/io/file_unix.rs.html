  36  impl FileDesc {
  37      /// Create a `FileDesc` from an open C file descriptor.
  38      ///
  39      /// The `FileDesc` will take ownership of the specified file descriptor and
  40      /// close it upon destruction if the `close_on_drop` flag is true, otherwise
  41      /// it will not close the file descriptor when this `FileDesc` is dropped.
  42      ///
  43      /// Note that all I/O operations done on this object will be *blocking*, but
  44      /// they do not require the runtime to be active.
  45      pub fn new(fd: fd_t, close_on_drop: bool) -> FileDesc {
  46          FileDesc { inner: UnsafeArc::new(Inner {
  47              fd: fd,
  48              close_on_drop: close_on_drop
  49          }) }
  50      }
  51  
  52      // FIXME(#10465) these functions should not be public, but anything in
  53      //               native::io wanting to use them is forced to have all the
  54      //               rtio traits in scope
  55      pub fn inner_read(&mut self, buf: &mut [u8]) -> Result<uint, IoError> {
  56          let ret = retry(|| unsafe {
  57              libc::read(self.fd(),
  58                         buf.as_mut_ptr() as *mut libc::c_void,
  59                         buf.len() as libc::size_t) as libc::c_int
  60          });
  61          if ret == 0 {
  62              Err(io::standard_error(io::EndOfFile))
  63          } else if ret < 0 {
  64              Err(super::last_error())
  65          } else {
  66              Ok(ret as uint)
  67          }
  68      }
  69      pub fn inner_write(&mut self, buf: &[u8]) -> Result<(), IoError> {
  70          let ret = keep_going(buf, |buf, len| {
  71              unsafe {
  72                  libc::write(self.fd(), buf as *libc::c_void,
  73                              len as libc::size_t) as i64
  74              }
  75          });
  76          if ret < 0 {
  77              Err(super::last_error())
  78          } else {
  79              Ok(())
  80          }
  81      }

  87      }
  88  }

 200  impl Drop for Inner {
 201      fn drop(&mut self) {
 202          // closing stdio file handles makes no sense, so never do it. Also, note
 203          // that errors are ignored when closing a file descriptor. The reason
 204          // for this is that if an error occurs we don't actually know if the
 205          // file descriptor was closed or not, and if we retried (for something
 206          // like EINTR), we might close another valid file descriptor (opened
 207          // after we closed ours.
 208          if self.close_on_drop && self.fd > libc::STDERR_FILENO {
 209              let n = unsafe { libc::close(self.fd) };
 210              if n != 0 {
 211                  println!("error {} when closing file descriptor {}", n,
 212                           self.fd);
 213              }
 214          }
 215      }
 216  }

 315  pub fn open(path: &CString, fm: io::FileMode, fa: io::FileAccess)
 316          -> IoResult<FileDesc> {
 317      let flags = match fm {
 318          io::Open => 0,
 319          io::Append => libc::O_APPEND,
 320          io::Truncate => libc::O_TRUNC,
 321      };

 323      let (flags, mode) = match fa {
 324          io::Read => (flags | libc::O_RDONLY, 0),
 325          io::Write => (flags | libc::O_WRONLY | libc::O_CREAT,
 326                        libc::S_IRUSR | libc::S_IWUSR),
 327          io::ReadWrite => (flags | libc::O_RDWR | libc::O_CREAT,
 328                            libc::S_IRUSR | libc::S_IWUSR),
 329      };
 330  
 331      match retry(|| unsafe { libc::open(path.with_ref(|p| p), flags, mode) }) {
 332          -1 => Err(super::last_error()),

 334      }

 343  pub fn readdir(p: &CString) -> IoResult<~[Path]> {
 344      use std::libc::{dirent_t};
 345      use std::libc::{opendir, readdir_r, closedir};
 346  
 347      fn prune(root: &CString, dirs: ~[Path]) -> ~[Path] {
 348          let root = unsafe { CString::new(root.with_ref(|p| p), false) };
 349          let root = Path::new(root);
 350  
 351          dirs.move_iter().filter(|path| {
 352              path.as_vec() != bytes!(".") && path.as_vec() != bytes!("..")
 353          }).map(|path| root.join(path)).collect()
 354      }
 355  
 356      extern {
 357          fn rust_dirent_t_size() -> libc::c_int;
 358          fn rust_list_dir_val(ptr: *mut dirent_t) -> *libc::c_char;
 359      }
 360  
 361      let size = unsafe { rust_dirent_t_size() };
 362      let mut buf = Vec::<u8>::with_capacity(size as uint);
 363      let ptr = buf.as_mut_slice().as_mut_ptr() as *mut dirent_t;
 364  
 365      let dir_ptr = p.with_ref(|buf| unsafe { opendir(buf) });
 366  
 367      if dir_ptr as uint != 0 {
 368          let mut paths = ~[];
 369          let mut entry_ptr = 0 as *mut dirent_t;
 370          while unsafe { readdir_r(dir_ptr, ptr, &mut entry_ptr) == 0 } {
 371              if entry_ptr.is_null() { break }
 372              let cstr = unsafe {
 373                  CString::new(rust_list_dir_val(entry_ptr), false)
 374              };
 375              paths.push(Path::new(cstr));
 376          }
 377          assert_eq!(unsafe { closedir(dir_ptr) }, 0);
 378          Ok(prune(p, paths))
 379      } else {
 380          Err(super::last_error())
 381      }
 382  }

 413  pub fn readlink(p: &CString) -> IoResult<Path> {
 414      let p = p.with_ref(|p| p);
 415      let mut len = unsafe { libc::pathconf(p, libc::_PC_NAME_MAX) };
 416      if len == -1 {
 417          len = 1024; // FIXME: read PATH_MAX from C ffi?
 418      }
 419      let mut buf = slice::with_capacity::<u8>(len as uint);
 420      match retry(|| unsafe {
 421          libc::readlink(p, buf.as_ptr() as *mut libc::c_char,
 422                         len as libc::size_t) as libc::c_int

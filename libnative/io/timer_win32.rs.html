  11  //! Timers based on win32 WaitableTimers
  12  //!
  13  //! This implementation is meant to be used solely on windows. As with other
  14  //! implementations, there is a worker thread which is doing all the waiting on
  15  //! a large number of timers for all active timers in the system. This worker
  16  //! thread uses the select() equivalent, WaitForMultipleObjects. One of the
  17  //! objects being waited on is a signal into the worker thread to notify that
  18  //! the incoming channel should be looked at.
  19  //!
  20  //! Other than that, the implementation is pretty straightforward in terms of
  21  //! the other two implementations of timers with nothing *that* new showing up.
  22  
  23  use std::comm::Data;
  24  use std::libc;
  25  use std::ptr;
  26  use std::rt::rtio;
  27  
  28  use io::timer_helper;
  29  use io::IoResult;
  30  
  31  pub struct Timer {
  32      priv obj: libc::HANDLE,
  33      priv on_worker: bool,
  34  }
  35  
  36  pub enum Req {
  37      NewTimer(libc::HANDLE, Sender<()>, bool),
  38      RemoveTimer(libc::HANDLE, Sender<()>),
  39      Shutdown,
  40  }
  41  
  42  fn helper(input: libc::HANDLE, messages: Receiver<Req>) {
  43      let mut objs = ~[input];
  44      let mut chans = ~[];
  45  
  46      'outer: loop {
  47          let idx = unsafe {
  48              imp::WaitForMultipleObjects(objs.len() as libc::DWORD,
  49                                          objs.as_ptr(),
  50                                          0 as libc::BOOL,
  51                                          libc::INFINITE)
  52          };
  53  
  54          if idx == 0 {
  55              loop {
  56                  match messages.try_recv() {
  57                      Data(NewTimer(obj, c, one)) => {
  58                          objs.push(obj);
  59                          chans.push((c, one));
  60                      }
  61                      Data(RemoveTimer(obj, c)) => {
  62                          c.send(());
  63                          match objs.iter().position(|&o| o == obj) {
  64                              Some(i) => {
  65                                  drop(objs.remove(i));
  66                                  drop(chans.remove(i - 1));
  67                              }
  68                              None => {}
  69                          }
  70                      }
  71                      Data(Shutdown) => {
  72                          assert_eq!(objs.len(), 1);
  73                          assert_eq!(chans.len(), 0);
  74                          break 'outer;
  75                      }
  76                      _ => break
  77                  }
  78              }
  79          } else {
  80              let remove = {
  81                  match &chans[idx - 1] {
  82                      &(ref c, oneshot) => !c.try_send(()) || oneshot
  83                  }
  84              };
  85              if remove {
  86                  drop(objs.remove(idx as uint));
  87                  drop(chans.remove(idx as uint - 1));
  88              }
  89          }
  90      }
  91  }
  92  
  93  impl Timer {
  94      pub fn new() -> IoResult<Timer> {
  95          timer_helper::boot(helper);
  96  
  97          let obj = unsafe {
  98              imp::CreateWaitableTimerA(ptr::mut_null(), 0, ptr::null())
  99          };
 100          if obj.is_null() {
 101              Err(super::last_error())
 102          } else {
 103              Ok(Timer { obj: obj, on_worker: false, })
 104          }
 105      }
 106  
 107      pub fn sleep(ms: u64) {
 108          use std::rt::rtio::RtioTimer;
 109          let mut t = Timer::new().ok().expect("must allocate a timer!");
 110          t.sleep(ms);
 111      }
 112  
 113      fn remove(&mut self) {
 114          if !self.on_worker { return }
 115  
 116          let (tx, rx) = channel();
 117          timer_helper::send(RemoveTimer(self.obj, tx));
 118          rx.recv();
 119  
 120          self.on_worker = false;
 121      }
 122  }
 123  
 124  impl rtio::RtioTimer for Timer {
 125      fn sleep(&mut self, msecs: u64) {
 126          self.remove();
 127  
 128          // there are 10^6 nanoseconds in a millisecond, and the parameter is in
 129          // 100ns intervals, so we multiply by 10^4.
 130          let due = -(msecs * 10000) as libc::LARGE_INTEGER;
 131          assert_eq!(unsafe {
 132              imp::SetWaitableTimer(self.obj, &due, 0, ptr::null(),
 133                                    ptr::mut_null(), 0)
 134          }, 1);
 135  
 136          let _ = unsafe { imp::WaitForSingleObject(self.obj, libc::INFINITE) };
 137      }
 138  
 139      fn oneshot(&mut self, msecs: u64) -> Receiver<()> {
 140          self.remove();
 141          let (tx, rx) = channel();
 142  
 143          // see above for the calculation
 144          let due = -(msecs * 10000) as libc::LARGE_INTEGER;
 145          assert_eq!(unsafe {
 146              imp::SetWaitableTimer(self.obj, &due, 0, ptr::null(),
 147                                    ptr::mut_null(), 0)
 148          }, 1);
 149  
 150          timer_helper::send(NewTimer(self.obj, tx, true));
 151          self.on_worker = true;
 152          return rx;
 153      }
 154  
 155      fn period(&mut self, msecs: u64) -> Receiver<()> {
 156          self.remove();
 157          let (tx, rx) = channel();
 158  
 159          // see above for the calculation
 160          let due = -(msecs * 10000) as libc::LARGE_INTEGER;
 161          assert_eq!(unsafe {
 162              imp::SetWaitableTimer(self.obj, &due, msecs as libc::LONG,
 163                                    ptr::null(), ptr::mut_null(), 0)
 164          }, 1);
 165  
 166          timer_helper::send(NewTimer(self.obj, tx, false));
 167          self.on_worker = true;
 168  
 169          return rx;
 170      }
 171  }
 172  
 173  impl Drop for Timer {
 174      fn drop(&mut self) {
 175          self.remove();
 176          assert!(unsafe { libc::CloseHandle(self.obj) != 0 });
 177      }
 178  }

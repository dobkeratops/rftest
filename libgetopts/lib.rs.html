   11  //! Simple getopt alternative.
   12  //!
   13  //! Construct a vector of options, either by using `reqopt`, `optopt`, and `optflag`
   14  //! or by building them from components yourself, and pass them to `getopts`,
   15  //! along with a vector of actual arguments (not including `argv[0]`). You'll
   16  //! either get a failure code back, or a match. You'll have to verify whether
   17  //! the amount of 'free' arguments in the match is what you expect. Use `opt_*`
   18  //! accessors to get argument values out of the matches object.
   19  //!
   20  //! Single-character options are expected to appear on the command line with a
   21  //! single preceding dash; multiple-character options are expected to be
   22  //! proceeded by two dashes. Options that expect an argument accept their
   23  //! argument following either a space or an equals sign. Single-character
   24  //! options don't require the space.
   25  //!
   26  //! # Example
   27  //!
   28  //! The following example shows simple command line parsing for an application
   29  //! that requires an input file to be specified, accepts an optional output
   30  //! file name following `-o`, and accepts both `-h` and `--help` as optional flags.
   31  //!
   32  //! ~~~{.rust}
   33  //! extern crate getopts;
   34  //! use getopts::{optopt,optflag,getopts,OptGroup};
   35  //! use std::os;
   36  //!
   37  //! fn do_work(inp: &str, out: Option<~str>) {
   38  //!     println!("{}", inp);
   39  //!     match out {
   40  //!         Some(x) => println!("{}", x),
   41  //!         None => println!("No Output"),
   42  //!     }
   43  //! }
   44  //!
   45  //! fn print_usage(program: &str, _opts: &[OptGroup]) {
   46  //!     println!("Usage: {} [options]", program);
   47  //!     println!("-o\t\tOutput");
   48  //!     println!("-h --help\tUsage");
   49  //! }
   50  //!
   51  //! fn main() {
   52  //!     let args = os::args();
   53  //!
   54  //!     let program = args[0].clone();
   55  //!
   56  //!     let opts = ~[
   57  //!         optopt("o", "", "set output file name", "NAME"),
   58  //!         optflag("h", "help", "print this help menu")
   59  //!     ];
   60  //!     let matches = match getopts(args.tail(), opts) {
   61  //!         Ok(m) => { m }
   62  //!         Err(f) => { fail!(f.to_err_msg()) }
   63  //!     };
   64  //!     if matches.opt_present("h") {
   65  //!         print_usage(program, opts);
   66  //!         return;
   67  //!     }
   68  //!     let output = matches.opt_str("o");
   69  //!     let input: &str = if !matches.free.is_empty() {
   70  //!         (*matches.free.get(0)).clone()
   71  //!     } else {
   72  //!         print_usage(program, opts);
   73  //!         return;
   74  //!     };
   75  //!     do_work(input, output);
   76  //! }
   77  //! ~~~

  514  /// Parse command line arguments according to the provided options.
  515  ///
  516  /// On success returns `Ok(Opt)`. Use methods such as `opt_present`
  517  /// `opt_str`, etc. to interrogate results.  Returns `Err(Fail_)` on failure.
  518  /// Use `to_err_msg` to get an error message.
  519  pub fn getopts(args: &[~str], optgrps: &[OptGroup]) -> Result {
  520      let opts: Vec<Opt> = optgrps.iter().map(|x| x.long_to_short()).collect();
  521      let n_opts = opts.len();
  522  
  523      fn f(_x: uint) -> Vec<Optval> { return Vec::new(); }
  524  
  525      let mut vals = Vec::from_fn(n_opts, f);
  526      let mut free: Vec<~str> = Vec::new();
  527      let l = args.len();
  528      let mut i = 0;
  529      while i < l {
  530          let cur = args[i].clone();
  531          let curlen = cur.len();
  532          if !is_arg(cur) {
  533              free.push(cur);
  534          } else if cur == ~"--" {
  535              let mut j = i + 1;
  536              while j < l { free.push(args[j].clone()); j += 1; }
  537              break;
  538          } else {
  539              let mut names;
  540              let mut i_arg = None;
  541              if cur[1] == '-' as u8 {
  542                  let tail = cur.slice(2, curlen);
  543                  let tail_eq: Vec<&str> = tail.split('=').collect();
  544                  if tail_eq.len() <= 1 {
  545                      names = vec!(Long(tail.to_owned()));
  546                  } else {
  547                      names =
  548                          vec!(Long((*tail_eq.get(0)).to_owned()));
  549                      i_arg = Some((*tail_eq.get(1)).to_owned());
  550                  }
  551              } else {
  552                  let mut j = 1;
  553                  let mut last_valid_opt_id = None;
  554                  names = Vec::new();
  555                  while j < curlen {
  556                      let range = cur.char_range_at(j);
  557                      let opt = Short(range.ch);
  558  
  559                      /* In a series of potential options (eg. -aheJ), if we
  560                         see one which takes an argument, we assume all
  561                         subsequent characters make up the argument. This
  562                         allows options such as -L/usr/local/lib/foo to be
  563                         interpreted correctly
  564                      */
  565  
  566                      match find_opt(opts.as_slice(), opt.clone()) {
  567                        Some(id) => last_valid_opt_id = Some(id),
  568                        None => {
  569                          let arg_follows =
  570                              last_valid_opt_id.is_some() &&
  571                              match opts.get(last_valid_opt_id.unwrap())
  572                                .hasarg {
  573  
  574                                Yes | Maybe => true,
  575                                No => false
  576                              };
  577                          if arg_follows && j < curlen {
  578                              i_arg = Some(cur.slice(j, curlen).to_owned());
  579                              break;
  580                          } else {
  581                              last_valid_opt_id = None;
  582                          }
  583                        }
  584                      }
  585                      names.push(opt);
  586                      j = range.next;
  587                  }
  588              }
  589              let mut name_pos = 0;
  590              for nm in names.iter() {
  591                  name_pos += 1;
  592                  let optid = match find_opt(opts.as_slice(), (*nm).clone()) {
  593                    Some(id) => id,
  594                    None => return Err(UnrecognizedOption(nm.to_str()))
  595                  };
  596                  match opts.get(optid).hasarg {
  597                    No => {
  598                      if !i_arg.is_none() {
  599                          return Err(UnexpectedArgument(nm.to_str()));
  600                      }
  601                      vals.get_mut(optid).push(Given);
  602                    }
  603                    Maybe => {
  604                      if !i_arg.is_none() {
  605                          vals.get_mut(optid)
  606                              .push(Val((i_arg.clone())
  607                              .unwrap()));
  608                      } else if name_pos < names.len() ||
  609                                    i + 1 == l || is_arg(args[i + 1]) {
  610                          vals.get_mut(optid).push(Given);
  611                      } else {
  612                          i += 1;
  613                          vals.get_mut(optid).push(Val(args[i].clone()));
  614                      }
  615                    }
  616                    Yes => {
  617                      if !i_arg.is_none() {
  618                          vals.get_mut(optid).push(Val(i_arg.clone().unwrap()));
  619                      } else if i + 1 == l {
  620                          return Err(ArgumentMissing(nm.to_str()));
  621                      } else {
  622                          i += 1;
  623                          vals.get_mut(optid).push(Val(args[i].clone()));
  624                      }
  625                    }
  626                  }
  627              }
  628          }
  629          i += 1;
  630      }
  631      i = 0u;
  632      while i < n_opts {
  633          let n = vals.get(i).len();
  634          let occ = opts.get(i).occur;
  635          if occ == Req {
  636              if n == 0 {
  637                  return Err(OptionMissing(opts.get(i).name.to_str()));
  638              }
  639          }
  640          if occ != Multi {
  641              if n > 1 {
  642                  return Err(OptionDuplicated(opts.get(i).name.to_str()));
  643              }
  644          }
  645          i += 1;
  646      }
  647      Ok(Matches {
  648          opts: opts,
  649          vals: vals,
  650          free: free
  651      })
  652  }

  782  /// Splits a string into substrings with possibly internal whitespace,
  783  /// each of them at most `lim` bytes long. The substrings have leading and trailing
  784  /// whitespace removed, and are only cut at whitespace boundaries.
  785  ///
  786  /// Note: Function was moved here from `std::str` because this module is the only place that
  787  /// uses it, and because it was to specific for a general string function.
  788  ///
  789  /// #Failure:
  790  ///
  791  /// Fails during iteration if the string contains a non-whitespace
  792  /// sequence longer than the limit.
  793  fn each_split_within<'a>(ss: &'a str, lim: uint, it: |&'a str| -> bool)
  794                       -> bool {
  795      // Just for fun, let's write this as a state machine:
  796  
  797      enum SplitWithinState {
  798          A,  // leading whitespace, initial state
  799          B,  // words
  800          C,  // internal and trailing whitespace
  801      }
  802      enum Whitespace {
  803          Ws, // current char is whitespace
  804          Cr  // current char is not whitespace
  805      }
  806      enum LengthLimit {
  807          UnderLim, // current char makes current substring still fit in limit
  808          OverLim   // current char makes current substring no longer fit in limit
  809      }
  810  
  811      let mut slice_start = 0;
  812      let mut last_start = 0;
  813      let mut last_end = 0;
  814      let mut state = A;
  815      let mut fake_i = ss.len();
  816      let mut lim = lim;
  817  
  818      let mut cont = true;
  819  
  820      // if the limit is larger than the string, lower it to save cycles
  821      if lim >= fake_i {
  822          lim = fake_i;
  823      }
  824  
  825      let machine: |&mut bool, (uint, char)| -> bool = |cont, (i, c)| {
  826          let whitespace = if ::std::char::is_whitespace(c) { Ws }       else { Cr };
  827          let limit      = if (i - slice_start + 1) <= lim  { UnderLim } else { OverLim };
  828  
  829          state = match (state, whitespace, limit) {
  830              (A, Ws, _)        => { A }
  831              (A, Cr, _)        => { slice_start = i; last_start = i; B }
  832  
  833              (B, Cr, UnderLim) => { B }
  834              (B, Cr, OverLim)  if (i - last_start + 1) > lim
  835                              => fail!("word starting with {} longer than limit!",
  836                                      ss.slice(last_start, i + 1)),
  837              (B, Cr, OverLim)  => {
  838                  *cont = it(ss.slice(slice_start, last_end));
  839                  slice_start = last_start;
  840                  B
  841              }
  842              (B, Ws, UnderLim) => {
  843                  last_end = i;
  844                  C
  845              }
  846              (B, Ws, OverLim)  => {
  847                  last_end = i;
  848                  *cont = it(ss.slice(slice_start, last_end));
  849                  A
  850              }
  851  
  852              (C, Cr, UnderLim) => {
  853                  last_start = i;
  854                  B
  855              }
  856              (C, Cr, OverLim)  => {
  857                  *cont = it(ss.slice(slice_start, last_end));
  858                  slice_start = i;
  859                  last_start = i;
  860                  last_end = i;
  861                  B
  862              }
  863              (C, Ws, OverLim)  => {
  864                  *cont = it(ss.slice(slice_start, last_end));
  865                  A
  866              }
  867              (C, Ws, UnderLim) => {
  868                  C
  869              }
  870          };
  871  
  872          *cont
  873      };
  874  
  875      ss.char_indices().advance(|x| machine(&mut cont, x));
  876  
  877      // Let the automaton 'run out' by supplying trailing whitespace
  878      while cont && match state { B | C => true, A => false } {
  879          machine(&mut cont, (fake_i, ' '));
  880          fake_i += 1;
  881      }
  882      return cont;
  883  }

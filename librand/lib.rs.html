  11  /*!
  12  Utilities for random number generation
  13  
  14  The key functions are `random()` and `Rng::gen()`. These are polymorphic
  15  and so can be used to generate any type that implements `Rand`. Type inference
  16  means that often a simple call to `rand::random()` or `rng.gen()` will
  17  suffice, but sometimes an annotation is required, e.g. `rand::random::<f64>()`.
  18  
  19  See the `distributions` submodule for sampling random numbers from
  20  distributions like normal and exponential.
  21  
  22  # Task-local RNG
  23  
  24  There is built-in support for a RNG associated with each task stored
  25  in task-local storage. This RNG can be accessed via `task_rng`, or
  26  used implicitly via `random`. This RNG is normally randomly seeded
  27  from an operating-system source of randomness, e.g. `/dev/urandom` on
  28  Unix systems, and will automatically reseed itself from this source
  29  after generating 32 KiB of random data.
  30  
  31  # Cryptographic security
  32  
  33  An application that requires random numbers for cryptographic purposes
  34  should prefer `OSRng`, which reads randomness from one of the source
  35  that the operating system provides (e.g. `/dev/urandom` on
  36  Unixes). The other random number generators provided by this module
  37  are either known to be insecure (`XorShiftRng`), or are not verified
  38  to be secure (`IsaacRng`, `Isaac64Rng` and `StdRng`).
  39  
  40  *Note*: on Linux, `/dev/random` is more secure than `/dev/urandom`,
  41  but it is a blocking RNG, and will wait until it has determined that
  42  it has collected enough entropy to fulfill a request for random
  43  data. It can be used with the `Rng` trait provided by this module by
  44  opening the file and passing it to `reader::ReaderRng`. Since it
  45  blocks, `/dev/random` should only be used to retrieve small amounts of
  46  randomness.
  47  
  48  # Examples
  49  
  50  ```rust
  51  use rand::Rng;
  52  
  53  let mut rng = rand::task_rng();
  54  if rng.gen() { // bool
  55      println!("int: {}, uint: {}", rng.gen::<int>(), rng.gen::<uint>())
  56  }
  57  ```
  58  
  59  ```rust
  60  let tuple_ptr = rand::random::<~(f64, char)>();
  61  println!("{:?}", tuple_ptr)
  62  ```
  63  */

 105  /// A random number generator.
 106  pub trait Rng {
 107      /// Return the next random u32.
 108      ///
 109      /// This rarely needs to be called directly, prefer `r.gen()` to
 110      /// `r.next_u32()`.
 111      // FIXME #7771: Should be implemented in terms of next_u64
 112      fn next_u32(&mut self) -> u32;
 113  
 114      /// Return the next random u64.
 115      ///
 116      /// By default this is implemented in terms of `next_u32`. An
 117      /// implementation of this trait must provide at least one of
 118      /// these two methods. Similarly to `next_u32`, this rarely needs
 119      /// to be called directly, prefer `r.gen()` to `r.next_u64()`.
 120      fn next_u64(&mut self) -> u64 {
 121          (self.next_u32() as u64 << 32) | (self.next_u32() as u64)
 122      }
 123  
 124      /// Fill `dest` with random data.
 125      ///
 126      /// This has a default implementation in terms of `next_u64` and
 127      /// `next_u32`, but should be overridden by implementations that
 128      /// offer a more efficient solution than just calling those
 129      /// methods repeatedly.
 130      ///
 131      /// This method does *not* have a requirement to bear any fixed
 132      /// relationship to the other methods, for example, it does *not*
 133      /// have to result in the same output as progressively filling
 134      /// `dest` with `self.gen::<u8>()`, and any such behaviour should
 135      /// not be relied upon.
 136      ///
 137      /// This method should guarantee that `dest` is entirely filled
 138      /// with new data, and may fail if this is impossible
 139      /// (e.g. reading past the end of a file that is being used as the
 140      /// source of randomness).
 141      ///
 142      /// # Example
 143      ///
 144      /// ```rust
 145      /// use rand::{task_rng, Rng};
 146      ///
 147      /// let mut v = [0u8, .. 13579];
 148      /// task_rng().fill_bytes(v);
 149      /// println!("{:?}", v);
 150      /// ```
 151      fn fill_bytes(&mut self, dest: &mut [u8]) {
 152          // this could, in theory, be done by transmuting dest to a
 153          // [u64], but this is (1) likely to be undefined behaviour for
 154          // LLVM, (2) has to be very careful about alignment concerns,
 155          // (3) adds more `unsafe` that needs to be checked, (4)
 156          // probably doesn't give much performance gain if
 157          // optimisations are on.
 158          let mut count = 0;
 159          let mut num = 0;
 160          for byte in dest.mut_iter() {
 161              if count == 0 {
 162                  // we could micro-optimise here by generating a u32 if
 163                  // we only need a few more bytes to fill the vector
 164                  // (i.e. at most 4).
 165                  num = self.next_u64();
 166                  count = 8;
 167              }
 168  
 169              *byte = (num & 0xff) as u8;
 170              num >>= 8;
 171              count -= 1;
 172          }
 173      }

 175      /// Return a random value of a `Rand` type.
 176      ///
 177      /// # Example
 178      ///
 179      /// ```rust
 180      /// use rand::{task_rng, Rng};
 181      ///
 182      /// let mut rng = task_rng();
 183      /// let x: uint = rng.gen();
 184      /// println!("{}", x);
 185      /// println!("{:?}", rng.gen::<(f64, bool)>());
 186      /// ```
 187      #[inline(always)]
 188      fn gen<T: Rand>(&mut self) -> T {
 189          Rand::rand(self)
 190      }

 192      /// Return a random vector of the specified length.
 193      ///
 194      /// # Example
 195      ///
 196      /// ```rust
 197      /// use rand::{task_rng, Rng};
 198      ///
 199      /// let mut rng = task_rng();

 201      /// println!("{}", x);
 202      /// println!("{}", rng.gen_vec::<(f64, bool)>(5));
 203      /// ```
 204      fn gen_vec<T: Rand>(&mut self, len: uint) -> Vec<T> {

 206      }

 208      /// Generate a random value in the range [`low`, `high`). Fails if
 209      /// `low >= high`.
 210      ///
 211      /// This is a convenience wrapper around
 212      /// `distributions::Range`. If this function will be called
 213      /// repeatedly with the same arguments, one should use `Range`, as
 214      /// that will amortize the computations that allow for perfect
 215      /// uniformity, as they only happen on initialization.
 216      ///
 217      /// # Example
 218      ///
 219      /// ```rust
 220      /// use rand::{task_rng, Rng};
 221      ///
 222      /// let mut rng = task_rng();
 223      /// let n: uint = rng.gen_range(0u, 10);
 224      /// println!("{}", n);
 225      /// let m: f64 = rng.gen_range(-40.0, 1.3e5);
 226      /// println!("{}", m);
 227      /// ```
 228      fn gen_range<T: Ord + SampleRange>(&mut self, low: T, high: T) -> T {
 229          assert!(low < high, "Rng.gen_range called with low >= high");
 230          Range::new(low, high).ind_sample(self)
 231      }

 233      /// Return a bool with a 1 in n chance of true
 234      ///
 235      /// # Example
 236      ///
 237      /// ```rust
 238      /// use rand::{task_rng, Rng};
 239      ///
 240      /// let mut rng = task_rng();
 241      /// println!("{:b}", rng.gen_weighted_bool(3));
 242      /// ```
 243      fn gen_weighted_bool(&mut self, n: uint) -> bool {
 244          n == 0 || self.gen_range(0, n) == 0
 245      }

 247      /// Return a random string of the specified length composed of
 248      /// A-Z,a-z,0-9.
 249      ///
 250      /// # Example
 251      ///
 252      /// ```rust
 253      /// use rand::{task_rng, Rng};
 254      ///

 256      /// ```
 257      fn gen_ascii_str(&mut self, len: uint) -> ~str {
 258          static GEN_ASCII_STR_CHARSET: &'static [u8] = bytes!("ABCDEFGHIJKLMNOPQRSTUVWXYZ\
 259                                                               abcdefghijklmnopqrstuvwxyz\

 261          let mut s = str::with_capacity(len);
 262          for _ in range(0, len) {
 263              s.push_char(self.choose(GEN_ASCII_STR_CHARSET) as char)
 264          }
 265          s
 266      }

 268      /// Choose an item randomly, failing if `values` is empty.
 269      fn choose<T: Clone>(&mut self, values: &[T]) -> T {
 270          self.choose_option(values).expect("Rng.choose: `values` is empty").clone()
 271      }

 273      /// Choose `Some(&item)` randomly, returning `None` if values is
 274      /// empty.
 275      ///
 276      /// # Example
 277      ///
 278      /// ```rust
 279      /// use rand::{task_rng, Rng};
 280      ///
 281      /// let choices = [1, 2, 4, 8, 16, 32];
 282      /// let mut rng = task_rng();
 283      /// println!("{:?}", rng.choose_option(choices));
 284      /// println!("{:?}", rng.choose_option(choices.slice_to(0)));
 285      /// ```
 286      fn choose_option<'a, T>(&mut self, values: &'a [T]) -> Option<&'a T> {
 287          if values.is_empty() {
 288              None
 289          } else {
 290              Some(&values[self.gen_range(0u, values.len())])
 291          }
 292      }

 294      /// Shuffle a mutable slice in place.
 295      ///
 296      /// # Example
 297      ///
 298      /// ```rust
 299      /// use rand::{task_rng, Rng};
 300      ///
 301      /// let mut rng = task_rng();
 302      /// let mut y = [1,2,3];
 303      /// rng.shuffle(y);
 304      /// println!("{}", y.as_slice());
 305      /// rng.shuffle(y);
 306      /// println!("{}", y.as_slice());
 307      /// ```
 308      fn shuffle<T>(&mut self, values: &mut [T]) {
 309          let mut i = values.len();
 310          while i >= 2u {
 311              // invariant: elements with index >= i have been locked in place.
 312              i -= 1u;
 313              // lock element i in place.
 314              values.swap(i, self.gen_range(0u, i + 1u));
 315          }
 316      }

 324      /// Randomly sample up to `n` elements from an iterator.
 325      ///
 326      /// # Example
 327      ///
 328      /// ```rust
 329      /// use rand::{task_rng, Rng};
 330      ///
 331      /// let mut rng = task_rng();
 332      /// let sample = rng.sample(range(1, 100), 5);
 333      /// println!("{}", sample);
 334      /// ```
 335      fn sample<A, T: Iterator<A>>(&mut self, iter: T, n: uint) -> Vec<A> {
 336          let mut reservoir = Vec::with_capacity(n);
 337          for (i, elem) in iter.enumerate() {
 338              if i < n {
 339                  reservoir.push(elem);
 340                  continue
 341              }
 342  
 343              let k = self.gen_range(0, i + 1);
 344              if k < reservoir.len() {
 345                  *reservoir.get_mut(k) = elem
 346              }
 347          }
 348          reservoir
 349      }
 350  }

 409  impl StdRng {
 410      /// Create a randomly seeded instance of `StdRng`.
 411      ///
 412      /// This is a very expensive operation as it has to read
 413      /// randomness from the operating system and use this in an
 414      /// expensive seeding operation. If one is only generating a small
 415      /// number of random numbers, or doesn't need the utmost speed for
 416      /// generating each number, `task_rng` and/or `random` may be more
 417      /// appropriate.
 418      ///
 419      /// Reading the randomness from the OS may fail, and any error is
 420      /// propagated via the `IoResult` return value.
 421      #[cfg(not(target_word_size="64"))]
 422      pub fn new() -> IoResult<StdRng> {

 467  ///

 469  /// consideration for cryptography or security. If you require a specifically
 470  /// seeded `Rng` for consistency over time you should pick one algorithm and
 471  /// create the `Rng` yourself.
 472  ///
 473  /// This will read randomness from the operating system to seed the

 475  pub fn weak_rng() -> XorShiftRng {

 482  /// An Xorshift[1] random number
 483  /// generator.
 484  ///
 485  /// The Xorshift algorithm is not suitable for cryptographic purposes
 486  /// but is very fast. If you do not know for sure that it fits your
 487  /// requirements, use a more secure one such as `IsaacRng` or `OSRng`.
 488  ///
 489  /// [1]: Marsaglia, George (July 2003). ["Xorshift
 490  /// RNGs"](http://www.jstatsoft.org/v08/i14/paper). *Journal of
 491  /// Statistical Software*. Vol. 8 (Issue 14).
 492  pub struct XorShiftRng {
 493      x: u32,
 494      y: u32,
 495      z: u32,
 496      w: u32,
 497  }
 498  
 499  impl Rng for XorShiftRng {
 500      #[inline]
 501      fn next_u32(&mut self) -> u32 {
 502          let x = self.x;
 503          let t = x ^ (x << 11);
 504          self.x = self.y;
 505          self.y = self.z;
 506          self.z = self.w;
 507          let w = self.w;
 508          self.w = w ^ (w >> 19) ^ (t ^ (t >> 8));
 509          self.w
 510      }
 511  }
 512  
 513  impl SeedableRng<[u32, .. 4]> for XorShiftRng {
 514      /// Reseed an XorShiftRng. This will fail if `seed` is entirely 0.
 515      fn reseed(&mut self, seed: [u32, .. 4]) {
 516          assert!(!seed.iter().all(|&x| x == 0),
 517                  "XorShiftRng.reseed called with an all zero seed.");
 518  
 519          self.x = seed[0];
 520          self.y = seed[1];
 521          self.z = seed[2];
 522          self.w = seed[3];
 523      }
 524  
 525      /// Create a new XorShiftRng. This will fail if `seed` is entirely 0.
 526      fn from_seed(seed: [u32, .. 4]) -> XorShiftRng {
 527          assert!(!seed.iter().all(|&x| x == 0),
 528                  "XorShiftRng::from_seed called with an all zero seed.");
 529  
 530          XorShiftRng {
 531              x: seed[0],
 532              y: seed[1],
 533              z: seed[2],
 534              w: seed[3]
 535          }
 536      }
 537  }

 539  impl XorShiftRng {
 540      /// Create an xor shift random number generator with a random seed.
 541      pub fn new() -> IoResult<XorShiftRng> {
 542          let mut s = [0u8, ..16];
 543          let mut r = try!(OSRng::new());
 544          loop {
 545              r.fill_bytes(s);
 546  
 547              if !s.iter().all(|x| *x == 0) {
 548                  break;
 549              }
 550          }
 551          let s: [u32, ..4] = unsafe { cast::transmute(s) };
 552          Ok(SeedableRng::from_seed(s))
 553      }
 554  }

 556  /// Controls how the task-local RNG is reseeded.
 557  struct TaskRngReseeder;
 558  
 559  impl reseeding::Reseeder<StdRng> for TaskRngReseeder {
 560      fn reseed(&mut self, rng: &mut StdRng) {
 561          *rng = match StdRng::new() {
 562              Ok(r) => r,
 563              Err(e) => fail!("could not reseed task_rng: {}", e)
 564          }
 565      }
 566  }
 567  static TASK_RNG_RESEED_THRESHOLD: uint = 32_768;
 568  type TaskRngInner = reseeding::ReseedingRng<StdRng, TaskRngReseeder>;
 569  /// The task-local RNG.
 570  pub struct TaskRng {
 571      // This points into TLS (specifically, it points to the endpoint
 572      // of a ~ stored in TLS, to make it robust against TLS moving
 573      // things internally) and so this struct cannot be legally
 574      // transferred between tasks *and* it's unsafe to deallocate the
 575      // RNG other than when a task is finished.
 576      //
 577      // The use of unsafe code here is OK if the invariants above are
 578      // satisfied; and it allows us to avoid (unnecessarily) using a
 579      // GC'd or RC'd pointer.
 580      rng: *mut TaskRngInner,
 581      marker: marker::NoSend,
 582  }
 583  
 584  // used to make space in TLS for a random number generator
 585  local_data_key!(TASK_RNG_KEY: ~TaskRngInner)
 586  
 587  /// Retrieve the lazily-initialized task-local random number
 588  /// generator, seeded by the system. Intended to be used in method
 589  /// chaining style, e.g. `task_rng().gen::<int>()`.
 590  ///
 591  /// The RNG provided will reseed itself from the operating system
 592  /// after generating a certain amount of randomness.
 593  ///
 594  /// The internal RNG used is platform and architecture dependent, even
 595  /// if the operating system random number generator is rigged to give
 596  /// the same sequence always. If absolute consistency is required,
 597  /// explicitly select an RNG, e.g. `IsaacRng` or `Isaac64Rng`.
 598  pub fn task_rng() -> TaskRng {
 599      local_data::get_mut(TASK_RNG_KEY, |rng| match rng {
 600          None => {
 601              let r = match StdRng::new() {
 602                  Ok(r) => r,
 603                  Err(e) => fail!("could not initialize task_rng: {}", e)
 604              };
 605              let mut rng = ~reseeding::ReseedingRng::new(r,
 606                                                          TASK_RNG_RESEED_THRESHOLD,
 607                                                          TaskRngReseeder);
 608              let ptr = &mut *rng as *mut TaskRngInner;
 609  
 610              local_data::set(TASK_RNG_KEY, rng);
 611  
 612              TaskRng { rng: ptr, marker: marker::NoSend }
 613          }
 614          Some(rng) => TaskRng { rng: &mut **rng, marker: marker::NoSend }
 615      })
 616  }

 620          unsafe { (*self.rng).next_u32() }
 621      }

 625      }

 628      fn fill_bytes(&mut self, bytes: &mut [u8]) {

 630      }
 631  }

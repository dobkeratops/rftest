  21  /// A random number generator that uses the ISAAC algorithm[1].
  22  ///
  23  /// The ISAAC algorithm is generally accepted as suitable for
  24  /// cryptographic purposes, but this implementation has not be
  25  /// verified as such. Prefer a generator like `OSRng` that defers to
  26  /// the operating system for cases that need high security.
  27  ///
  28  /// [1]: Bob Jenkins, [*ISAAC: A fast cryptographic random number
  29  /// generator*](http://www.burtleburtle.net/bob/rand/isaacafa.html)
  30  pub struct IsaacRng {
  31      priv cnt: u32,
  32      priv rsl: [u32, .. RAND_SIZE],
  33      priv mem: [u32, .. RAND_SIZE],
  34      priv a: u32,
  35      priv b: u32,
  36      priv c: u32
  37  }
  38  static EMPTY: IsaacRng = IsaacRng {
  39      cnt: 0,
  40      rsl: [0, .. RAND_SIZE],
  41      mem: [0, .. RAND_SIZE],
  42      a: 0, b: 0, c: 0
  43  };

  55              })

  57

  59          rng
  60      }

  70      /// Initialises `self`. If `use_rsl` is true, then use the current value
  71      /// of `rsl` as a seed, otherwise construct one algorithmically (not
  72      /// randomly).
  73      fn init(&mut self, use_rsl: bool) {
  74          let mut a = 0x9e3779b9;
  75          let mut b = a;
  76          let mut c = a;
  77          let mut d = a;
  78          let mut e = a;
  79          let mut f = a;
  80          let mut g = a;
  81          let mut h = a;
  82  
  83          macro_rules! mix(
  84              () => {{
  85                  a^=b<<11; d+=a; b+=c;
  86                  b^=c>>2;  e+=b; c+=d;
  87                  c^=d<<8;  f+=c; d+=e;
  88                  d^=e>>16; g+=d; e+=f;
  89                  e^=f<<10; h+=e; f+=g;
  90                  f^=g>>4;  a+=f; g+=h;
  91                  g^=h<<8;  b+=g; h+=a;
  92                  h^=a>>9;  c+=h; a+=b;
  93              }}
  94          );
  95  
  96          for _ in range(0, 4) { mix!(); }
  97  
  98          if use_rsl {
  99              macro_rules! memloop (
 100                  ($arr:expr) => {{
 101                      for i in range_step(0u32, RAND_SIZE, 8) {
 102                          a+=$arr[i  ]; b+=$arr[i+1];
 103                          c+=$arr[i+2]; d+=$arr[i+3];
 104                          e+=$arr[i+4]; f+=$arr[i+5];
 105                          g+=$arr[i+6]; h+=$arr[i+7];
 106                          mix!();
 107                          self.mem[i  ]=a; self.mem[i+1]=b;
 108                          self.mem[i+2]=c; self.mem[i+3]=d;
 109                          self.mem[i+4]=e; self.mem[i+5]=f;
 110                          self.mem[i+6]=g; self.mem[i+7]=h;
 111                      }
 112                  }}
 113              );
 114  
 115              memloop!(self.rsl);
 116              memloop!(self.mem);
 117          } else {
 118              for i in range_step(0u32, RAND_SIZE, 8) {
 119                  mix!();
 120                  self.mem[i  ]=a; self.mem[i+1]=b;
 121                  self.mem[i+2]=c; self.mem[i+3]=d;
 122                  self.mem[i+4]=e; self.mem[i+5]=f;
 123                  self.mem[i+6]=g; self.mem[i+7]=h;
 124              }
 125          }
 126  
 127          self.isaac();
 128      }

 130      /// Refills the output buffer (`self.rsl`)
 131      #[inline]
 132      fn isaac(&mut self) {
 133          self.c += 1;
 134          // abbreviations
 135          let mut a = self.a;
 136          let mut b = self.b + self.c;
 137  
 138          static MIDPOINT: uint = RAND_SIZE as uint / 2;
 139  
 140          macro_rules! ind (($x:expr) => {
 141              self.mem[($x >> 2) & (RAND_SIZE - 1)]
 142          });
 143          macro_rules! rngstep(
 144              ($j:expr, $shift:expr) => {{
 145                  let base = $j;
 146                  let mix = if $shift < 0 {
 147                      a >> -$shift as uint
 148                  } else {
 149                      a << $shift as uint
 150                  };
 151  
 152                  let x = self.mem[base  + mr_offset];
 153                  a = (a ^ mix) + self.mem[base + m2_offset];
 154                  let y = ind!(x) + a + b;
 155                  self.mem[base + mr_offset] = y;
 156  
 157                  b = ind!(y >> RAND_SIZE_LEN) + x;
 158                  self.rsl[base + mr_offset] = b;
 159              }}
 160          );
 161  
 162          let r = [(0, MIDPOINT), (MIDPOINT, 0)];
 163          for &(mr_offset, m2_offset) in r.iter() {
 164              for i in range_step(0u, MIDPOINT, 4) {
 165                  rngstep!(i + 0, 13);
 166                  rngstep!(i + 1, -6);
 167                  rngstep!(i + 2, 2);
 168                  rngstep!(i + 3, -16);
 169              }
 170          }
 171  
 172          self.a = a;
 173          self.b = b;
 174          self.cnt = RAND_SIZE;
 175      }
 176  }

 178  impl Rng for IsaacRng {
 179      #[inline]
 180      fn next_u32(&mut self) -> u32 {
 181          if self.cnt == 0 {
 182              // make some more numbers
 183              self.isaac();
 184          }
 185          self.cnt -= 1;
 186          self.rsl[self.cnt]
 187      }
 188  }

 190  impl<'a> SeedableRng<&'a [u32]> for IsaacRng {
 191      fn reseed(&mut self, seed: &'a [u32]) {
 192          // make the seed into [seed[0], seed[1], ..., seed[seed.len()
 193          // - 1], 0, 0, ...], to fill rng.rsl.
 194          let seed_iter = seed.iter().map(|&x| x).chain(Repeat::new(0u32));
 195  
 196          for (rsl_elem, seed_elem) in self.rsl.mut_iter().zip(seed_iter) {
 197              *rsl_elem = seed_elem;
 198          }
 199          self.cnt = 0;
 200          self.a = 0;
 201          self.b = 0;
 202          self.c = 0;
 203  
 204          self.init(true);
 205      }

 207      /// Create an ISAAC random number generator with a seed. This can
 208      /// be any length, although the maximum number of elements used is
 209      /// 256 and any more will be silently ignored. A generator
 210      /// constructed with a given seed will generate the same sequence
 211      /// of values as all other generators constructed with that seed.
 212      fn from_seed(seed: &'a [u32]) -> IsaacRng {
 213          let mut rng = EMPTY;
 214          rng.reseed(seed);
 215          rng
 216      }
 217  }

 333      /// Refills the output buffer (`self.rsl`)
 334      fn isaac64(&mut self) {
 335          self.c += 1;
 336          // abbreviations
 337          let mut a = self.a;
 338          let mut b = self.b + self.c;
 339          static MIDPOINT: uint =  RAND_SIZE_64 / 2;
 340          static MP_VEC: [(uint, uint), .. 2] = [(0,MIDPOINT), (MIDPOINT, 0)];
 341          macro_rules! ind (
 342              ($x:expr) => {
 343                  *self.mem.unsafe_ref(($x as uint >> 3) & (RAND_SIZE_64 - 1))
 344              }
 345          );
 346          macro_rules! rngstep(
 347              ($j:expr, $shift:expr) => {{
 348                  let base = base + $j;
 349                  let mix = a ^ (if $shift < 0 {
 350                      a >> -$shift as uint
 351                  } else {
 352                      a << $shift as uint
 353                  });
 354                  let mix = if $j == 0 {!mix} else {mix};
 355  
 356                  unsafe {
 357                      let x = *self.mem.unsafe_ref(base + mr_offset);
 358                      a = mix + *self.mem.unsafe_ref(base + m2_offset);
 359                      let y = ind!(x) + a + b;
 360                      self.mem.unsafe_set(base + mr_offset, y);
 361  
 362                      b = ind!(y >> RAND_SIZE_64_LEN) + x;
 363                      self.rsl.unsafe_set(base + mr_offset, b);
 364                  }
 365              }}
 366          );
 367  
 368          for &(mr_offset, m2_offset) in MP_VEC.iter() {
 369              for base in range(0, MIDPOINT / 4).map(|i| i * 4) {
 370                  rngstep!(0, 21);
 371                  rngstep!(1, -5);
 372                  rngstep!(2, 12);
 373                  rngstep!(3, -33);
 374              }
 375          }
 376  
 377          self.a = a;
 378          self.b = b;
 379          self.cnt = RAND_SIZE_64;
 380      }
 381  }

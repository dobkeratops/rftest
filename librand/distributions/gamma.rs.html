  19  /// The Gamma distribution `Gamma(shape, scale)` distribution.
  20  ///
  21  /// The density function of this distribution is
  22  ///
  23  /// ```notrust
  24  /// f(x) =  x^(k - 1) * exp(-x / Î¸) / (Î(k) * Î¸^k)
  25  /// ```
  26  ///
  27  /// where `Î` is the Gamma function, `k` is the shape and `Î¸` is the
  28  /// scale and both `k` and `Î¸` are strictly positive.
  29  ///
  30  /// The algorithm used is that described by Marsaglia & Tsang 2000[1],
  31  /// falling back to directly sampling from an Exponential for `shape
  32  /// == 1`, and using the boosting technique described in [1] for
  33  /// `shape < 1`.
  34  ///
  35  /// # Example
  36  ///
  37  /// ```rust
  38  /// use rand::distributions::{IndependentSample, Gamma};
  39  ///
  40  /// let gamma = Gamma::new(2.0, 5.0);
  41  /// let v = gamma.ind_sample(&mut rand::task_rng());
  42  /// println!("{} is from a Gamma(2, 5) distribution", v);
  43  /// ```
  44  ///
  45  /// [1]: George Marsaglia and Wai Wan Tsang. 2000. "A Simple Method
  46  /// for Generating Gamma Variables" *ACM Trans. Math. Softw.* 26, 3
  47  /// (September 2000),
  48  /// 363-372. DOI:[10.1145/358407.358414](http://doi.acm.org/10.1145/358407.358414)
  49  pub enum Gamma {
  50      priv Large(GammaLargeShape),

 150      fn ind_sample<R: Rng>(&self, rng: &mut R) -> f64 {
 151          loop {
 152              let StandardNormal(x) = rng.gen::<StandardNormal>();
 153              let v_cbrt = 1.0 + self.c * x;

 159              let Open01(u) = rng.gen::<Open01<f64>>();
 160  
 161              let x_sqr = x * x;
 162              if u < 1.0 - 0.0331 * x_sqr * x_sqr ||
 163                  num::ln(u) < 0.5 * x_sqr + self.d * (1.0 - v + num::ln(v)) {
 164                  return self.d * v * self.scale
 165              }
 166          }

 170  /// The chi-squared distribution `ÏÂ²(k)`, where `k` is the degrees of
 171  /// freedom.
 172  ///
 173  /// For `k > 0` integral, this distribution is the sum of the squares
 174  /// of `k` independent standard normal random variables. For other
 175  /// `k`, this uses the equivalent characterisation `ÏÂ²(k) = Gamma(k/2,
 176  /// 2)`.
 177  ///
 178  /// # Example
 179  ///
 180  /// ```rust
 181  /// use rand::distributions::{ChiSquared, IndependentSample};
 182  ///
 183  /// let chi = ChiSquared::new(11.0);
 184  /// let v = chi.ind_sample(&mut rand::task_rng());
 185  /// println!("{} is from a ÏÂ²(11) distribution", v)
 186  /// ```
 187  pub enum ChiSquared {
 188      // k == 1, Gamma(alpha, ..) is particularly slow for alpha < 1,
 189      // e.g. when alpha = 1/2 as it would be for this case, so special-
 190      // casing and using the definition of N(0,1)^2 is faster.
 191      priv DoFExactlyOne,
 192      priv DoFAnythingElse(Gamma)
 193  }
 194  
 195  impl ChiSquared {
 196      /// Create a new chi-squared distribution with degrees-of-freedom
 197      /// `k`. Fails if `k < 0`.
 198      pub fn new(k: f64) -> ChiSquared {
 199          if k == 1.0 {
 200              DoFExactlyOne
 201          } else {
 202              assert!(k > 0.0, "ChiSquared::new called with `k` < 0");
 203              DoFAnythingElse(Gamma::new(0.5 * k, 2.0))
 204          }
 205      }
 206  }

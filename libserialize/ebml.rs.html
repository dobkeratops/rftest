  101      pub struct Res {
  102          val: uint,
  103          next: uint
  104      }
  105  
  106      #[inline(never)]
  107      fn vuint_at_slow(data: &[u8], start: uint) -> Res {
  108          let a = data[start];
  109          if a & 0x80u8 != 0u8 {
  110              return Res {val: (a & 0x7fu8) as uint, next: start + 1u};
  111          }
  112          if a & 0x40u8 != 0u8 {
  113              return Res {val: ((a & 0x3fu8) as uint) << 8u |
  114                          (data[start + 1u] as uint),
  115                      next: start + 2u};
  116          }
  117          if a & 0x20u8 != 0u8 {
  118              return Res {val: ((a & 0x1fu8) as uint) << 16u |
  119                          (data[start + 1u] as uint) << 8u |
  120                          (data[start + 2u] as uint),
  121                      next: start + 3u};
  122          }
  123          if a & 0x10u8 != 0u8 {
  124              return Res {val: ((a & 0x0fu8) as uint) << 24u |
  125                          (data[start + 1u] as uint) << 16u |
  126                          (data[start + 2u] as uint) << 8u |
  127                          (data[start + 3u] as uint),
  128                      next: start + 4u};
  129          }
  130          fail!("vint too big");
  131      }
  132  
  133      pub fn vuint_at(data: &[u8], start: uint) -> Res {
  134          use std::mem::from_be32;
  135  
  136          if data.len() - start < 4 {
  137              return vuint_at_slow(data, start);
  138          }
  139  
  140          // Lookup table for parsing EBML Element IDs as per http://ebml.sourceforge.net/specs/
  141          // The Element IDs are parsed by reading a big endian u32 positioned at data[start].
  142          // Using the four most significant bits of the u32 we lookup in the table below how the
  143          // element ID should be derived from it.
  144          //
  145          // The table stores tuples (shift, mask) where shift is the number the u32 should be right
  146          // shifted with and mask is the value the right shifted value should be masked with.
  147          // If for example the most significant bit is set this means it's a class A ID and the u32
  148          // should be right shifted with 24 and masked with 0x7f. Therefore we store (24, 0x7f) at
  149          // index 0x8 - 0xF (four bit numbers where the most significant bit is set).
  150          //
  151          // By storing the number of shifts and masks in a table instead of checking in order if
  152          // the most significant bit is set, the second most significant bit is set etc. we can
  153          // replace up to three "and+branch" with a single table lookup which gives us a measured
  154          // speedup of around 2x on x86_64.
  155          static SHIFT_MASK_TABLE: [(u32, u32), ..16] = [
  156              (0, 0x0), (0, 0x0fffffff),
  157              (8, 0x1fffff), (8, 0x1fffff),
  158              (16, 0x3fff), (16, 0x3fff), (16, 0x3fff), (16, 0x3fff),
  159              (24, 0x7f), (24, 0x7f), (24, 0x7f), (24, 0x7f),
  160              (24, 0x7f), (24, 0x7f), (24, 0x7f), (24, 0x7f)
  161          ];
  162  
  163          unsafe {
  164              let ptr = data.as_ptr().offset(start as int) as *i32;
  165              let val = from_be32(*ptr) as u32;
  166  
  167              let i = (val >> 28u) as uint;
  168              let (shift, mask) = SHIFT_MASK_TABLE[i];
  169              Res {
  170                  val: ((val >> shift) & mask) as uint,
  171                  next: start + (((32 - shift) >> 3) as uint)
  172              }
  173          }
  174      }

  180      pub fn doc_at<'a>(data: &'a [u8], start: uint) -> TaggedDoc<'a> {
  181          let elt_tag = vuint_at(data, start);
  182          let elt_size = vuint_at(data, elt_tag.next);
  183          let end = elt_size.next + elt_size.val;
  184          TaggedDoc {
  185              tag: elt_tag.val,
  186              doc: Doc { data: data, start: elt_size.next, end: end }
  187          }
  188      }

  190      pub fn maybe_get_doc<'a>(d: Doc<'a>, tg: uint) -> Option<Doc<'a>> {
  191          let mut pos = d.start;
  192          while pos < d.end {
  193              let elt_tag = vuint_at(d.data, pos);
  194              let elt_size = vuint_at(d.data, elt_tag.next);
  195              pos = elt_size.next + elt_size.val;
  196              if elt_tag.val == tg {
  197                  return Some(Doc { data: d.data, start: elt_size.next,
  198                                    end: pos });
  199              }
  200          }
  201          None
  202      }

  214      pub fn docs<'a>(d: Doc<'a>, it: |uint, Doc<'a>| -> bool) -> bool {
  215          let mut pos = d.start;
  216          while pos < d.end {
  217              let elt_tag = vuint_at(d.data, pos);
  218              let elt_size = vuint_at(d.data, elt_tag.next);
  219              pos = elt_size.next + elt_size.val;
  220              let doc = Doc { data: d.data, start: elt_size.next, end: pos };
  221              if !it(elt_tag.val, doc) {

  223              }

  288          fn _check_label(&mut self, lbl: &str) {
  289              if self.pos < self.parent.end {
  290                  let TaggedDoc { tag: r_tag, doc: r_doc } =
  291                      doc_at(self.parent.data, self.pos);
  292  
  293                  if r_tag == (EsLabel as uint) {
  294                      self.pos = r_doc.end;
  295                      let str = r_doc.as_str_slice();
  296                      if lbl != str {
  297                          fail!("Expected label {} but found {}", lbl, str);
  298                      }
  299                  }
  300              }

  303          fn next_doc(&mut self, exp_tag: EbmlEncoderTag) -> Doc<'doc> {
  304              debug!(". next_doc(exp_tag={:?})", exp_tag);
  305              if self.pos >= self.parent.end {
  306                  fail!("no more documents in current node!");
  307              }
  308              let TaggedDoc { tag: r_tag, doc: r_doc } =
  309                  doc_at(self.parent.data, self.pos);
  310              debug!("self.parent={}-{} self.pos={} r_tag={} r_doc={}-{}",
  311                     self.parent.start,
  312                     self.parent.end,
  313                     self.pos,
  314                     r_tag,
  315                     r_doc.start,
  316                     r_doc.end);
  317              if r_tag != (exp_tag as uint) {
  318                  fail!("expected EBML doc with tag {:?} but found tag {:?}",
  319                         exp_tag, r_tag);
  320              }
  321              if r_doc.end > self.parent.end {
  322                  fail!("invalid EBML, child extends to {:#x}, parent to {:#x}",
  323                        r_doc.end, self.parent.end);
  324              }
  325              self.pos = r_doc.end;
  326              r_doc
  327          }

  329          fn push_doc<T>(&mut self, exp_tag: EbmlEncoderTag,
  330                         f: |&mut Decoder<'doc>| -> T) -> T {
  331              let d = self.next_doc(exp_tag);
  332              let old_parent = self.parent;
  333              let old_pos = self.pos;
  334              self.parent = d;
  335              self.pos = d.start;
  336              let r = f(self);
  337              self.parent = old_parent;
  338              self.pos = old_pos;
  339              r
  340          }

  348          pub fn read_opaque<R>(&mut self, op: |&mut Decoder<'doc>, Doc| -> R) -> R {
  349              let doc = self.next_doc(EsOpaque);

  353              self.pos = doc.start;
  354

  356

  358              self.pos = old_pos;

  360          }

  606      // ebml writing
  607      pub struct Encoder<'a> {
  608          // FIXME(#5665): this should take a trait object. Note that if you
  609          //               delete this comment you should consider removing the
  610          //               unwrap()'s below of the results of the calls to
  611          //               write(). We're guaranteed that writing into a MemWriter
  612          //               won't fail, but this is not true for all I/O streams in
  613          //               general.
  614          writer: &'a mut MemWriter,
  615          priv size_positions: ~[uint],
  616          last_error: io::IoResult<()>,
  617      }
  618  
  619      fn write_sized_vuint(w: &mut MemWriter, n: uint, size: uint) {
  620          match size {
  621              1u => w.write(&[0x80u8 | (n as u8)]),
  622              2u => w.write(&[0x40u8 | ((n >> 8_u) as u8), n as u8]),
  623              3u => w.write(&[0x20u8 | ((n >> 16_u) as u8), (n >> 8_u) as u8,
  624                              n as u8]),
  625              4u => w.write(&[0x10u8 | ((n >> 24_u) as u8), (n >> 16_u) as u8,
  626                              (n >> 8_u) as u8, n as u8]),
  627              _ => fail!("vint to write too big: {}", n)
  628          }.unwrap()
  629      }
  630  
  631      fn write_vuint(w: &mut MemWriter, n: uint) {
  632          if n < 0x7f_u { write_sized_vuint(w, n, 1u); return; }
  633          if n < 0x4000_u { write_sized_vuint(w, n, 2u); return; }
  634          if n < 0x200000_u { write_sized_vuint(w, n, 3u); return; }
  635          if n < 0x10000000_u { write_sized_vuint(w, n, 4u); return; }
  636          fail!("vint to write too big: {}", n);
  637      }

  659          pub fn start_tag(&mut self, tag_id: uint) {
  660              debug!("Start tag {}", tag_id);
  661  
  662              // Write the enum ID:
  663              write_vuint(self.writer, tag_id);
  664  
  665              // Write a placeholder four-byte size.
  666              self.size_positions.push(try!(self.writer.tell()) as uint);
  667              let zeroes: &[u8] = &[0u8, 0u8, 0u8, 0u8];
  668              try!(self.writer.write(zeroes));
  669          }
  670  
  671          pub fn end_tag(&mut self) {
  672              let last_size_pos = self.size_positions.pop().unwrap();
  673              let cur_pos = try!(self.writer.tell());
  674              try!(self.writer.seek(last_size_pos as i64, io::SeekSet));
  675              let size = cur_pos as uint - last_size_pos - 4;
  676              write_sized_vuint(self.writer, size, 4u);
  677              try!(self.writer.seek(cur_pos as i64, io::SeekSet));
  678  
  679              debug!("End tag (size = {})", size);
  680          }

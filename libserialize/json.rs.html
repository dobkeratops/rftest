   20  # What is JSON?
   21  
   22  JSON (JavaScript Object Notation) is a way to write data in Javascript.
   23  Like XML it allows one to encode structured data in a text format that can be read by humans easily.
   24  Its native compatibility with JavaScript and its simple syntax make it used widely.
   25  
   26  Json data are encoded in a form of "key":"value".
   27  Data types that can be encoded are JavaScript types :
   28  boolean (`true` or `false`), number (`f64`), string, array, object, null.
   29  An object is a series of string keys mapping to values, in `"key": value` format.
   30  Arrays are enclosed in square brackets ([ ... ]) and objects in curly brackets ({ ... }).
   31  A simple JSON document encoding a person, his/her age, address and phone numbers could look like:
   32  
   33  ```ignore
   34  {
   35      "FirstName": "John",
   36      "LastName": "Doe",
   37      "Age": 43,
   38      "Address": {
   39          "Street": "Downing Street 10",
   40          "City": "London",
   41          "Country": "Great Britain"
   42      },
   43      "PhoneNumbers": [
   44          "+44 1234567",
   45          "+44 2345678"
   46      ]
   47  }
   48  ```

   50  # Rust Type-based Encoding and Decoding
   51  
   52  Rust provides a mechanism for low boilerplate encoding & decoding
   53  of values to and from JSON via the serialization API.
   54  To be able to encode a piece of data, it must implement the `serialize::Encodable` trait.
   55  To be able to decode a piece of data, it must implement the `serialize::Decodable` trait.
   56  The Rust compiler provides an annotation to automatically generate
   57  the code for these traits: `#[deriving(Decodable, Encodable)]`
   58  
   59  To encode using Encodable :
   60  
   61  ```rust
   62  use std::io;
   63  use serialize::{json, Encodable};
   64  
   65   #[deriving(Encodable)]
   66   pub struct TestStruct   {
   67      data_str: ~str,
   68   }
   69  
   70  fn main() {
   71      let to_encode_object = TestStruct{data_str:~"example of string to encode"};
   72      let mut m = io::MemWriter::new();
   73      {
   74          let mut encoder = json::Encoder::new(&mut m as &mut std::io::Writer);
   75          to_encode_object.encode(&mut encoder);
   76      }
   77  }
   78  ```
   79  
   80  Two wrapper functions are provided to encode a Encodable object
   81  into a string (~str) or buffer (~[u8]): `str_encode(&m)` and `buffer_encode(&m)`.
   82  
   83  ```rust
   84  use serialize::json;
   85  let to_encode_object = ~"example of string to encode";
   86  let encoded_str: ~str = json::Encoder::str_encode(&to_encode_object);
   87  ```
   88  
   89  JSON API provide an enum `json::Json` and a trait `ToJson` to encode object.
   90  The trait `ToJson` encode object into a container `json::Json` and the API provide writer
   91  to encode them into a stream or a string ...
   92  
   93  When using `ToJson` the `Encodable` trait implementation is not mandatory.
   94  
   95  A basic `ToJson` example using a TreeMap of attribute name / attribute value:
   96  
   97  
   98  ```rust
   99  extern crate collections;
  100  extern crate serialize;
  101  
  102  use serialize::json;
  103  use serialize::json::ToJson;
  104  use collections::TreeMap;
  105  
  106  pub struct MyStruct  {
  107      attr1: u8,
  108      attr2: ~str,
  109  }
  110  
  111  impl ToJson for MyStruct {
  112      fn to_json( &self ) -> json::Json {
  113          let mut d = ~TreeMap::new();
  114          d.insert(~"attr1", self.attr1.to_json());
  115          d.insert(~"attr2", self.attr2.to_json());
  116          json::Object(d)
  117      }
  118  }
  119  
  120  fn main() {
  121      let test2: MyStruct = MyStruct {attr1: 1, attr2:~"test"};
  122      let tjson: json::Json = test2.to_json();
  123      let json_str: ~str = tjson.to_str();
  124  }
  125  ```
  126  
  127  To decode a JSON string using `Decodable` trait :
  128  
  129  ```rust
  130  extern crate serialize;
  131  use serialize::{json, Decodable};
  132  
  133  #[deriving(Decodable)]
  134  pub struct MyStruct  {
  135       attr1: u8,
  136       attr2: ~str,
  137  }
  138  
  139  fn main() {
  140      let json_str_to_decode: ~str =
  141              ~"{\"attr1\":1,\"attr2\":\"toto\"}";
  142      let json_object = json::from_str(json_str_to_decode);
  143      let mut decoder = json::Decoder::new(json_object.unwrap());
  144      let decoded_object: MyStruct = Decodable::decode(&mut decoder); // create the final object
  145  }

  311      /// Creates a new JSON encoder whose output will be written to the writer
  312      /// specified.
  313      pub fn new<'a>(wr: &'a mut io::Writer) -> Encoder<'a> {
  314          Encoder { wr: wr, error: Ok(()) }
  315      }
  316  
  317      /// Encode the specified struct into a json [u8]
  318      pub fn buffer_encode<T:Encodable<Encoder<'a>>>(to_encode_object: &T) -> ~[u8]  {
  319         //Serialize the object in a string using a writer
  320          let mut m = MemWriter::new();
  321          {
  322              let mut encoder = Encoder::new(&mut m as &mut io::Writer);
  323              to_encode_object.encode(&mut encoder);
  324          }
  325          m.unwrap()
  326      }
  327  
  328      /// Encode the specified struct into a json str
  329      pub fn str_encode<T:Encodable<Encoder<'a>>>(to_encode_object: &T) -> ~str  {
  330          let buff:~[u8] = Encoder::buffer_encode(to_encode_object);
  331          str::from_utf8_owned(buff).unwrap()
  332      }
  333  }

  25  impl<'a> ToHex for &'a [u8] {
  26      /**
  27       * Turn a vector of `u8` bytes into a hexadecimal string.
  28       *
  29       * # Example
  30       *
  31       * ```rust
  32       * extern crate serialize;
  33       * use serialize::hex::ToHex;
  34       *
  35       * fn main () {
  36       *     let str = [52,32].to_hex();
  37       *     println!("{}", str);
  38       * }
  39       * ```
  40       */
  41      fn to_hex(&self) -> ~str {
  42          let mut v = slice::with_capacity(self.len() * 2);
  43          for &byte in self.iter() {
  44              v.push(CHARS[(byte >> 4) as uint]);
  45              v.push(CHARS[(byte & 0xf) as uint]);
  46          }
  47  
  48          unsafe {
  49              str::raw::from_utf8_owned(v)
  50          }
  51      }
  52  }

  79  impl<'a> FromHex for &'a str {
  80      /**
  81       * Convert any hexadecimal encoded string (literal, `@`, `&`, or `~`)
  82       * to the byte values it encodes.
  83       *
  84       * You can use the `from_utf8_owned` function in `std::str`
  85       * to turn a `[u8]` into a string with characters corresponding to those
  86       * values.
  87       *
  88       * # Example
  89       *
  90       * This converts a string literal to hexadecimal and back.
  91       *
  92       * ```rust
  93       * extern crate serialize;
  94       * use serialize::hex::{FromHex, ToHex};
  95       * use std::str;
  96       *
  97       * fn main () {
  98       *     let hello_str = "Hello, World".as_bytes().to_hex();
  99       *     println!("{}", hello_str);
 100       *     let bytes = hello_str.from_hex().unwrap();
 101       *     println!("{:?}", bytes);
 102       *     let result_str = str::from_utf8_owned(bytes).unwrap();
 103       *     println!("{}", result_str);
 104       * }
 105       * ```
 106       */
 107      fn from_hex(&self) -> Result<~[u8], FromHexError> {
 108          // This may be an overestimate if there is any whitespace
 109          let mut b = slice::with_capacity(self.len() / 2);
 110          let mut modulus = 0;
 111          let mut buf = 0u8;
 112  
 113          for (idx, byte) in self.bytes().enumerate() {
 114              buf <<= 4;
 115  
 116              match byte as char {
 117                  'A'..'F' => buf |= byte - ('A' as u8) + 10,
 118                  'a'..'f' => buf |= byte - ('a' as u8) + 10,
 119                  '0'..'9' => buf |= byte - ('0' as u8),
 120                  ' '|'\r'|'\n'|'\t' => {
 121                      buf >>= 4;
 122                      continue
 123                  }
 124                  _ => return Err(InvalidHexCharacter(self.char_at(idx), idx)),
 125              }
 126  
 127              modulus += 1;
 128              if modulus == 2 {
 129                  modulus = 0;
 130                  b.push(buf);
 131              }
 132          }
 133  
 134          match modulus {
 135              0 => Ok(b),
 136              _ => Err(InvalidHexLength),
 137          }
 138      }
 139  }

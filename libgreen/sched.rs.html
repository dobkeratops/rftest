   30  /// A scheduler is responsible for coordinating the execution of Tasks
   31  /// on a single thread. The scheduler runs inside a slightly modified
   32  /// Rust Task. When not running this task is stored in the scheduler
   33  /// struct. The scheduler struct acts like a baton, all scheduling
   34  /// actions are transfers of the baton.
   35  ///
   36  /// FIXME: This creates too many callbacks to run_sched_once, resulting
   37  /// in too much allocation and too many events.
   38  pub struct Scheduler {
   39      /// ID number of the pool that this scheduler is a member of. When
   40      /// reawakening green tasks, this is used to ensure that tasks aren't
   41      /// reawoken on the wrong pool of schedulers.
   42      pool_id: uint,
   43      /// There are N work queues, one per scheduler.
   44      work_queue: deque::Worker<~GreenTask>,
   45      /// Work queues for the other schedulers. These are created by
   46      /// cloning the core work queues.
   47      work_queues: ~[deque::Stealer<~GreenTask>],
   48      /// The queue of incoming messages from other schedulers.
   49      /// These are enqueued by SchedHandles after which a remote callback
   50      /// is triggered to handle the message.
   51      message_queue: msgq::Consumer<SchedMessage>,
   52      /// Producer used to clone sched handles from
   53      message_producer: msgq::Producer<SchedMessage>,
   54      /// A shared list of sleeping schedulers. We'll use this to wake
   55      /// up schedulers when pushing work onto the work queue.
   56      sleeper_list: SleeperList,
   57      /// Indicates that we have previously pushed a handle onto the
   58      /// SleeperList but have not yet received the Wake message.
   59      /// Being `true` does not necessarily mean that the scheduler is
   60      /// not active since there are multiple event sources that may
   61      /// wake the scheduler. It just prevents the scheduler from pushing
   62      /// multiple handles onto the sleeper list.
   63      sleepy: bool,
   64      /// A flag to indicate we've received the shutdown message and should
   65      /// no longer try to go to sleep, but exit instead.
   66      no_sleep: bool,
   67      stack_pool: StackPool,
   68      /// The scheduler runs on a special task. When it is not running
   69      /// it is stored here instead of the work queue.
   70      sched_task: Option<~GreenTask>,
   71      /// An action performed after a context switch on behalf of the
   72      /// code running before the context switch
   73      cleanup_job: Option<CleanupJob>,
   74      /// If the scheduler shouldn't run some tasks, a friend to send
   75      /// them to.
   76      friend_handle: Option<SchedHandle>,
   77      /// Should this scheduler run any task, or only pinned tasks?
   78      run_anything: bool,
   79      /// A fast XorShift rng for scheduler use
   80      rng: XorShiftRng,
   81      /// A togglable idle callback
   82      idle_callback: Option<~PausableIdleCallback>,
   83      /// A countdown that starts at a random value and is decremented
   84      /// every time a yield check is performed. When it hits 0 a task
   85      /// will yield.
   86      yield_check_count: uint,
   87      /// A flag to tell the scheduler loop it needs to do some stealing
   88      /// in order to introduce randomness as part of a yield
   89      steal_for_yield: bool,
   90      /// Bookkeeping for the number of tasks which are currently running around
   91      /// inside this pool of schedulers
   92      task_state: TaskState,
   93  
   94      // n.b. currently destructors of an object are run in top-to-bottom in order
   95      //      of field declaration. Due to its nature, the pausable idle callback
   96      //      must have some sort of handle to the event loop, so it needs to get
   97      //      destroyed before the event loop itself. For this reason, we destroy
   98      //      the event loop last to ensure that any unsafe references to it are
   99      //      destroyed before it's actually destroyed.
  100  
  101      /// The event loop used to drive the scheduler and perform I/O
  102      event_loop: ~EventLoop,
  103  }

  175      // FIXME: This may eventually need to be refactored so that
  176      // the scheduler itself doesn't have to call event_loop.run.
  177      // That will be important for embedding the runtime into external
  178      // event loops.

  180      // Take a main task to run, and a scheduler to run it in. Create a
  181      // scheduler task and bootstrap into it.
  182      pub fn bootstrap(mut ~self) {
  183  
  184          // Build an Idle callback.
  185          let cb = ~SchedRunner as ~Callback;
  186          self.idle_callback = Some(self.event_loop.pausable_idle_callback(cb));
  187  
  188          // Create a task for the scheduler with an empty context.
  189          let sched_task = GreenTask::new_typed(Some(Coroutine::empty()),
  190                                                TypeSched);
  191  
  192          // Before starting our first task, make sure the idle callback
  193          // is active. As we do not start in the sleep state this is
  194          // important.
  195          self.idle_callback.get_mut_ref().resume();
  196  
  197          // Now, as far as all the scheduler state is concerned, we are inside
  198          // the "scheduler" context. The scheduler immediately hands over control
  199          // to the event loop, and this will only exit once the event loop no
  200          // longer has any references (handles or I/O objects).
  201          rtdebug!("starting scheduler {}", self.sched_id());
  202          let mut sched_task = self.run(sched_task);
  203  
  204          // Close the idle callback.
  205          let mut sched = sched_task.sched.take_unwrap();
  206          sched.idle_callback.take();
  207          // Make one go through the loop to run the close callback.
  208          let mut stask = sched.run(sched_task);
  209  
  210          // Now that we are done with the scheduler, clean up the
  211          // scheduler task. Do so by removing it from TLS and manually
  212          // cleaning up the memory it uses. As we didn't actually call
  213          // task.run() on the scheduler task we never get through all
  214          // the cleanup code it runs.
  215          rtdebug!("stopping scheduler {}", stask.sched.get_ref().sched_id());
  216  
  217          // Should not have any messages
  218          let message = stask.sched.get_mut_ref().message_queue.pop();
  219          rtassert!(match message { msgq::Empty => true, _ => false });
  220  
  221          stask.task.get_mut_ref().destroyed = true;
  222      }

  224      // This does not return a scheduler, as the scheduler is placed
  225      // inside the task.
  226      pub fn run(mut ~self, stask: ~GreenTask) -> ~GreenTask {
  227  
  228          // This is unsafe because we need to place the scheduler, with
  229          // the event_loop inside, inside our task. But we still need a
  230          // mutable reference to the event_loop to give it the "run"
  231          // command.
  232          unsafe {
  233              let event_loop: *mut ~EventLoop = &mut self.event_loop;
  234              // Our scheduler must be in the task before the event loop
  235              // is started.
  236              stask.put_with_sched(self);
  237              (*event_loop).run();
  238          }
  239  
  240          //  This is a serious code smell, but this function could be done away
  241          //  with if necessary. The ownership of `stask` was transferred into
  242          //  local storage just before the event loop ran, so it is possible to
  243          //  transmute `stask` as a uint across the running of the event loop to
  244          //  re-acquire ownership here.
  245          //
  246          // This would involve removing the Task from TLS, removing the runtime,
  247          // forgetting the runtime, and then putting the task into `stask`. For
  248          // now, because we have `GreenTask::convert`, I chose to take this
  249          // method for cleanliness. This function is *not* a fundamental reason
  250          // why this function should exist.
  251          GreenTask::convert(Local::take())
  252      }
  253  
  254      // * Execution Functions - Core Loop Logic
  255  
  256      // This function is run from the idle callback on the uv loop, indicating
  257      // that there are no I/O events pending. When this function returns, we will
  258      // fall back to epoll() in the uv event loop, waiting for more things to
  259      // happen. We may come right back off epoll() if the idle callback is still
  260      // active, in which case we're truly just polling to see if I/O events are
  261      // complete.
  262      //
  263      // The model for this function is to execute as much work as possible while
  264      // still fairly considering I/O tasks. Falling back to epoll() frequently is
  265      // often quite expensive, so we attempt to avoid it as much as possible. If
  266      // we have any active I/O on the event loop, then we're forced to fall back
  267      // to epoll() in order to provide fairness, but as long as we're doing work
  268      // and there's no active I/O, we can continue to do work.
  269      //
  270      // If we try really hard to do some work, but no work is available to be
  271      // done, then we fall back to epoll() to block this thread waiting for more
  272      // work (instead of busy waiting).
  273      fn run_sched_once(mut ~self, stask: ~GreenTask) {
  274          // Make sure that we're not lying in that the `stask` argument is indeed
  275          // the scheduler task for this scheduler.
  276          assert!(self.sched_task.is_none());
  277  
  278          // Assume that we need to continue idling unless we reach the
  279          // end of this function without performing an action.
  280          self.idle_callback.get_mut_ref().resume();
  281  
  282          // First we check for scheduler messages, these are higher
  283          // priority than regular tasks.
  284          let (mut sched, mut stask, mut did_work) =
  285              self.interpret_message_queue(stask, DontTryTooHard);
  286  
  287          // After processing a message, we consider doing some more work on the
  288          // event loop. The "keep going" condition changes after the first
  289          // iteration becase we don't want to spin here infinitely.
  290          //
  291          // Once we start doing work we can keep doing work so long as the
  292          // iteration does something. Note that we don't want to starve the
  293          // message queue here, so each iteration when we're done working we
  294          // check the message queue regardless of whether we did work or not.
  295          let mut keep_going = !did_work || !sched.event_loop.has_active_io();
  296          while keep_going {
  297              let (a, b, c) = match sched.do_work(stask) {
  298                  (sched, task, false) => {
  299                      sched.interpret_message_queue(task, GiveItYourBest)
  300                  }
  301                  (sched, task, true) => {
  302                      let (sched, task, _) =
  303                          sched.interpret_message_queue(task, GiveItYourBest);
  304                      (sched, task, true)
  305                  }
  306              };
  307              sched = a;
  308              stask = b;
  309              did_work = c;
  310  
  311              // We only keep going if we managed to do something productive and
  312              // also don't have any active I/O. If we didn't do anything, we
  313              // should consider going to sleep, and if we have active I/O we need
  314              // to poll for completion.
  315              keep_going = did_work && !sched.event_loop.has_active_io();
  316          }
  317  
  318          // If we ever did some work, then we shouldn't put our scheduler
  319          // entirely to sleep just yet. Leave the idle callback active and fall
  320          // back to epoll() to see what's going on.
  321          if did_work {
  322              return stask.put_with_sched(sched);
  323          }
  324  
  325          // If we got here then there was no work to do.
  326          // Generate a SchedHandle and push it to the sleeper list so
  327          // somebody can wake us up later.
  328          if !sched.sleepy && !sched.no_sleep {
  329              rtdebug!("scheduler has no work to do, going to sleep");
  330              sched.sleepy = true;
  331              let handle = sched.make_handle();
  332              sched.sleeper_list.push(handle);
  333              // Since we are sleeping, deactivate the idle callback.
  334              sched.idle_callback.get_mut_ref().pause();
  335          } else {
  336              rtdebug!("not sleeping, already doing so or no_sleep set");
  337              // We may not be sleeping, but we still need to deactivate
  338              // the idle callback.
  339              sched.idle_callback.get_mut_ref().pause();
  340          }
  341  
  342          // Finished a cycle without using the Scheduler. Place it back
  343          // in TLS.
  344          stask.put_with_sched(sched);
  345      }
  346  
  347      // This function returns None if the scheduler is "used", or it
  348      // returns the still-available scheduler. At this point all
  349      // message-handling will count as a turn of work, and as a result
  350      // return None.
  351      fn interpret_message_queue(mut ~self, stask: ~GreenTask,
  352                                 effort: EffortLevel)
  353              -> (~Scheduler, ~GreenTask, bool)
  354      {
  355  
  356          let msg = if effort == DontTryTooHard {
  357              self.message_queue.casual_pop()
  358          } else {
  359              // When popping our message queue, we could see an "inconsistent"
  360              // state which means that we *should* be able to pop data, but we
  361              // are unable to at this time. Our options are:
  362              //
  363              //  1. Spin waiting for data
  364              //  2. Ignore this and pretend we didn't find a message
  365              //
  366              // If we choose route 1, then if the pusher in question is currently
  367              // pre-empted, we're going to take up our entire time slice just
  368              // spinning on this queue. If we choose route 2, then the pusher in
  369              // question is still guaranteed to make a send() on its async
  370              // handle, so we will guaranteed wake up and see its message at some
  371              // point.
  372              //
  373              // I have chosen to take route #2.
  374              match self.message_queue.pop() {
  375                  msgq::Data(t) => Some(t),
  376                  msgq::Empty | msgq::Inconsistent => None
  377              }
  378          };
  379  
  380          match msg {
  381              Some(PinnedTask(task)) => {
  382                  let mut task = task;
  383                  task.give_home(HomeSched(self.make_handle()));
  384                  let (sched, task) = self.resume_task_immediately(stask, task);
  385                  (sched, task, true)
  386              }
  387              Some(TaskFromFriend(task)) => {
  388                  rtdebug!("got a task from a friend. lovely!");
  389                  let (sched, task) =
  390                      self.process_task(stask, task,
  391                                        Scheduler::resume_task_immediately_cl);
  392                  (sched, task, true)
  393              }
  394              Some(RunOnce(task)) => {
  395                  // bypass the process_task logic to force running this task once
  396                  // on this home scheduler. This is often used for I/O (homing).
  397                  let (sched, task) = self.resume_task_immediately(stask, task);
  398                  (sched, task, true)
  399              }
  400              Some(Wake) => {
  401                  self.sleepy = false;
  402                  (self, stask, true)
  403              }
  404              Some(Shutdown) => {
  405                  rtdebug!("shutting down");
  406                  if self.sleepy {
  407                      // There may be an outstanding handle on the
  408                      // sleeper list.  Pop them all to make sure that's
  409                      // not the case.
  410                      loop {
  411                          match self.sleeper_list.pop() {
  412                              Some(handle) => {
  413                                  let mut handle = handle;
  414                                  handle.send(Wake);
  415                              }
  416                              None => break
  417                          }
  418                      }
  419                  }
  420                  // No more sleeping. After there are no outstanding
  421                  // event loop references we will shut down.
  422                  self.no_sleep = true;
  423                  self.sleepy = false;
  424                  (self, stask, true)
  425              }
  426              Some(NewNeighbor(neighbor)) => {
  427                  self.work_queues.push(neighbor);
  428                  (self, stask, false)
  429              }
  430              None => (self, stask, false)
  431          }
  432      }

  434      fn do_work(mut ~self,
  435                 stask: ~GreenTask) -> (~Scheduler, ~GreenTask, bool) {
  436          rtdebug!("scheduler calling do work");
  437          match self.find_work() {
  438              Some(task) => {
  439                  rtdebug!("found some work! running the task");
  440                  let (sched, task) =
  441                      self.process_task(stask, task,
  442                                        Scheduler::resume_task_immediately_cl);
  443                  (sched, task, true)
  444              }
  445              None => {
  446                  rtdebug!("no work was found, returning the scheduler struct");
  447                  (self, stask, false)
  448              }
  449          }
  450      }
  451  
  452      // Workstealing: In this iteration of the runtime each scheduler
  453      // thread has a distinct work queue. When no work is available
  454      // locally, make a few attempts to steal work from the queues of
  455      // other scheduler threads. If a few steals fail we end up in the
  456      // old "no work" path which is fine.
  457  
  458      // First step in the process is to find a task. This function does
  459      // that by first checking the local queue, and if there is no work
  460      // there, trying to steal from the remote work queues.
  461      fn find_work(&mut self) -> Option<~GreenTask> {
  462          rtdebug!("scheduler looking for work");
  463          if !self.steal_for_yield {
  464              match self.work_queue.pop() {
  465                  Some(task) => {
  466                      rtdebug!("found a task locally");
  467                      return Some(task)
  468                  }
  469                  None => {
  470                      rtdebug!("scheduler trying to steal");
  471                      return self.try_steals();
  472                  }
  473              }
  474          } else {
  475              // During execution of the last task, it performed a 'yield',
  476              // so we're doing some work stealing in order to introduce some
  477              // scheduling randomness. Otherwise we would just end up popping
  478              // that same task again. This is pretty lame and is to work around
  479              // the problem that work stealing is not designed for 'non-strict'
  480              // (non-fork-join) task parallelism.
  481              self.steal_for_yield = false;
  482              match self.try_steals() {
  483                  Some(task) => {
  484                      rtdebug!("stole a task after yielding");
  485                      return Some(task);
  486                  }
  487                  None => {
  488                      rtdebug!("did not steal a task after yielding");
  489                      // Back to business
  490                      return self.find_work();
  491                  }
  492              }
  493          }
  494      }
  495  
  496      // Try stealing from all queues the scheduler knows about. This
  497      // naive implementation can steal from our own queue or from other
  498      // special schedulers.
  499      fn try_steals(&mut self) -> Option<~GreenTask> {
  500          let work_queues = &mut self.work_queues;
  501          let len = work_queues.len();
  502          let start_index = self.rng.gen_range(0, len);
  503          for index in range(0, len).map(|i| (i + start_index) % len) {
  504              match work_queues[index].steal() {
  505                  deque::Data(task) => {
  506                      rtdebug!("found task by stealing");
  507                      return Some(task)
  508                  }
  509                  _ => ()
  510              }
  511          };
  512          rtdebug!("giving up on stealing");
  513          return None;
  514      }

  516      // * Task Routing Functions - Make sure tasks send up in the right
  517      // place.

  519      fn process_task(mut ~self, cur: ~GreenTask,
  520                      mut next: ~GreenTask,
  521                      schedule_fn: SchedulingFn) -> (~Scheduler, ~GreenTask) {
  522          rtdebug!("processing a task");
  523  
  524          match next.take_unwrap_home() {
  525              HomeSched(home_handle) => {
  526                  if home_handle.sched_id != self.sched_id() {
  527                      rtdebug!("sending task home");
  528                      next.give_home(HomeSched(home_handle));
  529                      Scheduler::send_task_home(next);
  530                      (self, cur)
  531                  } else {
  532                      rtdebug!("running task here");
  533                      next.give_home(HomeSched(home_handle));
  534                      schedule_fn(self, cur, next)
  535                  }
  536              }
  537              AnySched if self.run_anything => {
  538                  rtdebug!("running anysched task here");
  539                  next.give_home(AnySched);
  540                  schedule_fn(self, cur, next)
  541              }
  542              AnySched => {
  543                  rtdebug!("sending task to friend");
  544                  next.give_home(AnySched);
  545                  self.send_to_friend(next);
  546                  (self, cur)
  547              }
  548          }
  549      }

  551      fn send_task_home(task: ~GreenTask) {
  552          let mut task = task;
  553          match task.take_unwrap_home() {
  554              HomeSched(mut home_handle) => home_handle.send(PinnedTask(task)),
  555              AnySched => rtabort!("error: cannot send anysched task home"),
  556          }
  557      }
  558  
  559      /// Take a non-homed task we aren't allowed to run here and send
  560      /// it to the designated friend scheduler to execute.
  561      fn send_to_friend(&mut self, task: ~GreenTask) {
  562          rtdebug!("sending a task to friend");
  563          match self.friend_handle {
  564              Some(ref mut handle) => {
  565                  handle.send(TaskFromFriend(task));
  566              }
  567              None => {
  568                  rtabort!("tried to send task to a friend but scheduler has no friends");
  569              }
  570          }
  571      }

  574      ///
  575      /// Pushes the task onto the work stealing queue and tells the
  576      /// event loop to run it later. Always use this instead of pushing
  577      /// to the work queue directly.
  578      pub fn enqueue_task(&mut self, task: ~GreenTask) {
  579

  581          assert!(!task.is_sched());
  582          self.work_queue.push(task);
  583          match self.idle_callback {
  584              Some(ref mut idle) => idle.resume(),
  585              None => {} // allow enqueuing before the scheduler starts
  586          }
  587  
  588          // We've made work available. Notify a
  589          // sleeping scheduler.
  590  
  591          match self.sleeper_list.casual_pop() {
  592              Some(handle) => {
  593                  let mut handle = handle;
  594                  handle.send(Wake)
  595              }
  596              None => { (/* pass */) }
  597          };
  598      }

  600      // * Core Context Switching Functions
  601  
  602      // The primary function for changing contexts. In the current
  603      // design the scheduler is just a slightly modified GreenTask, so
  604      // all context swaps are from GreenTask to GreenTask. The only difference
  605      // between the various cases is where the inputs come from, and
  606      // what is done with the resulting task. That is specified by the
  607      // cleanup function f, which takes the scheduler and the
  608      // old task as inputs.
  609  
  610      pub fn change_task_context(mut ~self,
  611                                 current_task: ~GreenTask,
  612                                 mut next_task: ~GreenTask,
  613                                 f: |&mut Scheduler, ~GreenTask|) -> ~GreenTask {
  614          let f_opaque = ClosureConverter::from_fn(f);
  615  
  616          let current_task_dupe = &*current_task as *GreenTask;
  617  
  618          // The current task is placed inside an enum with the cleanup
  619          // function. This enum is then placed inside the scheduler.
  620          self.cleanup_job = Some(CleanupJob::new(current_task, f_opaque));
  621  
  622          // The scheduler is then placed inside the next task.
  623          next_task.sched = Some(self);
  624  
  625          // However we still need an internal mutable pointer to the
  626          // original task. The strategy here was "arrange memory, then
  627          // get pointers", so we crawl back up the chain using
  628          // transmute to eliminate borrowck errors.
  629          unsafe {
  630  
  631              let sched: &mut Scheduler =
  632                  cast::transmute_mut_region(*next_task.sched.get_mut_ref());
  633  
  634              let current_task: &mut GreenTask = match sched.cleanup_job {
  635                  Some(CleanupJob { task: ref mut task, .. }) => &mut **task,
  636                  None => rtabort!("no cleanup job")
  637              };
  638  
  639              let (current_task_context, next_task_context) =
  640                  Scheduler::get_contexts(current_task, next_task);
  641  
  642              // Done with everything - put the next task in TLS. This
  643              // works because due to transmute the borrow checker
  644              // believes that we have no internal pointers to
  645              // next_task.
  646              cast::forget(next_task);
  647  
  648              // The raw context swap operation. The next action taken
  649              // will be running the cleanup job from the context of the
  650              // next task.
  651              Context::swap(current_task_context, next_task_context);
  652          }
  653  
  654          // When the context swaps back to this task we immediately
  655          // run the cleanup job, as expected by the previously called
  656          // swap_contexts function.
  657          let mut current_task: ~GreenTask = unsafe {
  658              cast::transmute(current_task_dupe)
  659          };
  660          current_task.sched.get_mut_ref().run_cleanup_job();
  661  
  662          // See the comments in switch_running_tasks_and_then for why a lock
  663          // is acquired here. This is the resumption points and the "bounce"
  664          // that it is referring to.
  665          unsafe {
  666              let _guard = current_task.nasty_deschedule_lock.lock();
  667          }
  668          return current_task;
  669      }
  670  
  671      // Returns a mutable reference to both contexts involved in this
  672      // swap. This is unsafe - we are getting mutable internal
  673      // references to keep even when we don't own the tasks. It looks
  674      // kinda safe because we are doing transmutes before passing in
  675      // the arguments.
  676      pub fn get_contexts<'a>(current_task: &mut GreenTask, next_task: &mut GreenTask) ->
  677          (&'a mut Context, &'a mut Context) {
  678          let current_task_context =
  679              &mut current_task.coroutine.get_mut_ref().saved_context;
  680          let next_task_context =
  681                  &mut next_task.coroutine.get_mut_ref().saved_context;
  682          unsafe {
  683              (cast::transmute_mut_region(current_task_context),
  684               cast::transmute_mut_region(next_task_context))
  685          }
  686      }

  690      pub fn resume_task_immediately(~self, cur: ~GreenTask,
  691                                     next: ~GreenTask) -> (~Scheduler, ~GreenTask) {

  694              assert!(sched.sched_task.is_none());

  706      /// Block a running task, context switch to the scheduler, then pass the
  707      /// blocked task to a closure.
  708      ///
  709      /// # Safety note
  710      ///
  711      /// The closure here is a *stack* closure that lives in the
  712      /// running task.  It gets transmuted to the scheduler's lifetime
  713      /// and called while the task is blocked.
  714      ///
  715      /// This passes a Scheduler pointer to the fn after the context switch
  716      /// in order to prevent that fn from performing further scheduling operations.
  717      /// Doing further scheduling could easily result in infinite recursion.
  718      ///
  719      /// Note that if the closure provided relinquishes ownership of the
  720      /// BlockedTask, then it is possible for the task to resume execution before
  721      /// the closure has finished executing. This would naturally introduce a
  722      /// race if the closure and task shared portions of the environment.
  723      ///
  724      /// This situation is currently prevented, or in other words it is
  725      /// guaranteed that this function will not return before the given closure
  726      /// has returned.
  727      pub fn deschedule_running_task_and_then(mut ~self,
  728                                              cur: ~GreenTask,
  729                                              f: |&mut Scheduler, BlockedTask|) {
  730          // Trickier - we need to get the scheduler task out of self
  731          // and use it as the destination.
  732          let stask = self.sched_task.take_unwrap();
  733          // Otherwise this is the same as below.
  734          self.switch_running_tasks_and_then(cur, stask, f)
  735      }

  737      pub fn switch_running_tasks_and_then(~self,
  738                                           cur: ~GreenTask,
  739                                           next: ~GreenTask,
  740                                           f: |&mut Scheduler, BlockedTask|) {
  741          // And here comes one of the sad moments in which a lock is used in a
  742          // core portion of the rust runtime. As always, this is highly
  743          // undesirable, so there's a good reason behind it.
  744          //
  745          // There is an excellent outline of the problem in issue #8132, and it's
  746          // summarized in that `f` is executed on a sched task, but its
  747          // environment is on the previous task. If `f` relinquishes ownership of
  748          // the BlockedTask, then it may introduce a race where `f` is using the
  749          // environment as well as the code after the 'deschedule' block.
  750          //
  751          // The solution we have chosen to adopt for now is to acquire a

  760          let cur = self.change_task_context(cur, next, |sched, mut task| {

  815      pub fn yield_now(mut ~self, cur: ~GreenTask) {

  828                  sched.enqueue_task(task);
  829              });
  830              cur.put()
  831          }

  834      pub fn maybe_yield(mut ~self, cur: ~GreenTask) {
  835          // It's possible for sched tasks to possibly call this function, and it
  836          // just means that they're likely sending on channels (which
  837          // occasionally call this function). Sched tasks follow different paths
  838          // when executing yield_now(), which may possibly trip the assertion
  839          // below. For this reason, we just have sched tasks bail out soon.
  840          //
  841          // Sched tasks have no need to yield anyway because as soon as they
  842          // return they'll yield to other threads by falling back to the event
  843          // loop. Additionally, we completely control sched tasks, so we can make
  844          // sure that they never execute more than enough code.
  845          if cur.is_sched() {
  846              return cur.put_with_sched(self)
  847          }
  848  
  849          // The number of times to do the yield check before yielding, chosen
  850          // arbitrarily.
  851          rtassert!(self.yield_check_count > 0);
  852          self.yield_check_count -= 1;

  854              self.yield_now(cur);
  855          } else {
  856              cur.put_with_sched(self);
  857          }
  858      }

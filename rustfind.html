441 fn lookup_def_node_of_node(dc:&DocContext,node:&AstNode, node_spans:&FNodeInfoMap, node_def_node:&HashMap<ast::NodeId,ast::def_id>)->Option<ast::NodeId> {
442     
443     match *node {
444         astnode_expr(e)=>match e.node {
445             // handle methods-calls
446             ast::expr_method_call(ref id,ref receiver,ref ident,ref ty_params,ref arg_exprs,ref call_sugar)=>{
447                 let rec_ty_node= astnode_expr(*receiver).ty_node_id();
448                 let rec_ty_node1= dc.tycx.node_types.find(&(*id as uint));
449 
450                 match dc.ca.maps.method_map.find(&e.id) {
451                     None=> {},//logi!("no method map entry for",e.id),
452                     Some(mme)=>{
453                         //logi!("Method Map entry for",e.id);
454                         match mme.origin {
455                             typeck::method_static(def_id)=> 
456                                 return Some(def_id.node),
457                             typeck::method_trait(def_id,_,_)=>
458                                 return Some(def_id.node),
459                             typeck::method_param(mp)=>{
460                                 match dc.tycx.trait_method_def_ids.find(&mp.trait_id) {
461                                     None=>{},
462                                     Some(method_def_ids)=>{
463                                         return Some(method_def_ids[mp.method_num].node)
464                                     }
465                                 }
466                             }
467                         }
468                     }
469                 }
470             },
471             // handle struct-fields? "object.field"
472             ast::expr_field(ref object_expr,ref ident,ref ty_params)=>{
473                 // we want the type of the object..
474                 let obj_ty=dc.tycx.node_types.find(&(object_expr.id as uint));
475                 let tydef=auto_deref_ty(ty::get(*obj_ty.unwrap()));
476                 match tydef.sty {

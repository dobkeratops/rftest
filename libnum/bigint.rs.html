  215  impl Sub<BigUint, BigUint> for BigUint {
  216      fn sub(&self, other: &BigUint) -> BigUint {
  217          let new_len = cmp::max(self.data.len(), other.data.len());
  218          let zeros = ZERO_VEC.iter().cycle();
  219          let (a, b) = (self.data.iter().chain(zeros.clone()), other.data.iter().chain(zeros));
  220  
  221          let mut borrow = 0;
  222          let diff: Vec<BigDigit> =  a.take(new_len).zip(b).map(|(ai, bi)| {
  223              let (hi, lo) = BigDigit::from_uint(
  224                  BigDigit::base + (*ai as uint) - (*bi as uint) - (borrow as uint)
  225              );
  226              /*
  227              hi * (base) + lo == 1*(base) + ai - bi - borrow
  228              => ai - bi - borrow < 0 <=> hi == 0
  229              */
  230              borrow = if hi == 0 { 1 } else { 0 };
  231              lo
  232          }).collect();
  233  
  234          assert_eq!(borrow, 0);     // <=> assert!((self >= other));
  235          return BigUint::new(diff);
  236      }
  237  }

  239  impl Mul<BigUint, BigUint> for BigUint {
  240      fn mul(&self, other: &BigUint) -> BigUint {
  241          if self.is_zero() || other.is_zero() { return Zero::zero(); }
  242  
  243          let (s_len, o_len) = (self.data.len(), other.data.len());
  244          if s_len == 1 { return mul_digit(other, self.data.as_slice()[0]);  }
  245          if o_len == 1 { return mul_digit(self,  other.data.as_slice()[0]); }
  246  
  247          // Using Karatsuba multiplication
  248          // (a1 * base + a0) * (b1 * base + b0)
  249          // = a1*b1 * base^2 +
  250          //   (a1*b1 + a0*b0 - (a1-b0)*(b1-a0)) * base +
  251          //   a0*b0
  252          let half_len = cmp::max(s_len, o_len) / 2;
  253          let (s_hi, s_lo) = cut_at(self,  half_len);
  254          let (o_hi, o_lo) = cut_at(other, half_len);
  255  
  256          let ll = s_lo * o_lo;
  257          let hh = s_hi * o_hi;
  258          let mm = {
  259              let (s1, n1) = sub_sign(s_hi, s_lo);
  260              let (s2, n2) = sub_sign(o_hi, o_lo);
  261              match (s1, s2) {
  262                  (Equal, _) | (_, Equal) => hh + ll,
  263                  (Less, Greater) | (Greater, Less) => hh + ll + (n1 * n2),
  264                  (Less, Less) | (Greater, Greater) => hh + ll - (n1 * n2)
  265              }
  266          };
  267  
  268          return ll + mm.shl_unit(half_len) + hh.shl_unit(half_len * 2);
  269

  271          fn mul_digit(a: &BigUint, n: BigDigit) -> BigUint {
  272              if n == 0 { return Zero::zero(); }
  273              if n == 1 { return (*a).clone(); }
  274  
  275              let mut carry = 0;
  276              let mut prod: Vec<BigDigit> = a.data.iter().map(|ai| {
  277                  let (hi, lo) = BigDigit::from_uint(
  278                      (*ai as uint) * (n as uint) + (carry as uint)
  279                  );
  280                  carry = hi;
  281                  lo
  282              }).collect();
  283              if carry != 0 { prod.push(carry); }
  284              return BigUint::new(prod);
  285          }

  378      fn div_mod_floor(&self, other: &BigUint) -> (BigUint, BigUint) {
  379          if other.is_zero() { fail!() }
  380          if self.is_zero() { return (Zero::zero(), Zero::zero()); }
  381          if *other == One::one() { return ((*self).clone(), Zero::zero()); }
  382  
  383          match self.cmp(other) {
  384              Less    => return (Zero::zero(), (*self).clone()),
  385              Equal   => return (One::one(), Zero::zero()),
  386              Greater => {} // Do nothing
  387          }
  388  
  389          let mut shift = 0;
  390          let mut n = *other.data.last().unwrap();
  391          while n < (1 << BigDigit::bits - 2) {
  392              n <<= 1;
  393              shift += 1;
  394          }
  395          assert!(shift < BigDigit::bits);
  396          let (d, m) = div_mod_floor_inner(self << shift, other << shift);
  397          return (d, m >> shift);
  398  
  399  
  400          fn div_mod_floor_inner(a: BigUint, b: BigUint) -> (BigUint, BigUint) {
  401              let mut m = a;
  402              let mut d: BigUint = Zero::zero();
  403              let mut n = 1;
  404              while m >= b {
  405                  let (d0, d_unit, b_unit) = div_estimate(&m, &b, n);
  406                  let mut d0 = d0;
  407                  let mut prod = b * d0;
  408                  while prod > m {
  409                      // FIXME(#5992): assignment operator overloads
  410                      // d0 -= d_unit
  411                      d0   = d0 - d_unit;
  412                      // FIXME(#5992): assignment operator overloads
  413                      // prod -= b_unit;
  414                      prod = prod - b_unit
  415                  }
  416                  if d0.is_zero() {
  417                      n = 2;
  418                      continue;
  419                  }
  420                  n = 1;
  421                  // FIXME(#5992): assignment operator overloads
  422                  // d += d0;
  423                  d = d + d0;

  464       *

 1324          // Generate a random BigUint...
 1325          let biguint = self.gen_biguint(bit_size);

 1328              // ...except that if the BigUint is zero, we need to try

 1335                  Zero
 1336              }
 1337          } else if self.gen() {
 1338              Plus
 1339          } else {
 1340              Minus
 1341          };

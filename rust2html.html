248 fn text_here_is(line:&str, pos:uint,reftext:&str, color:int)->int {
249     if pos+reftext.len()+2>=line.len() { return 0}
250     if pos>0 {
251         if is_alphanumeric(line[pos-1] as char) { return 0 }// must be word start
252     }
253     if is_alphanumeric(line[pos+reftext.len()] as char) { return 0 }// must be word start
254     let mut i=0;
255     while i<reftext.len() {
256         if reftext[i]!=line[i+pos] {return 0}
257         i+=1;
258     }
259 
260     return color;
261 }

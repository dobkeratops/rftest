  46 // Traverse the crate, collecting all the test functions, eliding any
  47 // existing main functions, and synthesizing a main test harness
  48 pub fn modify_for_testing(sess: session::Session,
  49                           crate: @ast::Crate)
  50                        -> @ast::Crate {
  51     // We generate the test harness when building in the 'test'
  52     // configuration, either with the '--test' or '--cfg test'
  53     // command line options.
  54     let should_test = attr::contains_name(crate.config, "test");
  55 
  56     if should_test {
  57         generate_test_harness(sess, crate)
  58     } else {
  59         strip_test_functions(crate)
  60     }
  61 }

  63 fn generate_test_harness(sess: session::Session,
  64                          crate: @ast::Crate)
  65                       -> @ast::Crate {
  66     let cx: @mut TestCtxt = @mut TestCtxt {
  67         sess: sess,
  68         crate: crate,
  69         ext_cx: ExtCtxt::new(sess.parse_sess, sess.opts.cfg.clone()),
  70         path: ~[],
  71         testfns: ~[]
  72     };
  73 
  74     let ext_cx = cx.ext_cx;
  75     ext_cx.bt_push(ExpnInfo {
  76         call_site: dummy_sp(),
  77         callee: NameAndSpan {
  78             name: @"test",
  79             span: None
  80         }
  81     });
  82 
  83     let precursor = @fold::AstFoldFns {
  84         fold_crate: |a,b| fold_crate(cx, a, b),
  85         fold_item: |a,b| fold_item(cx, a, b),
  86         fold_mod: |a,b| fold_mod(cx, a, b),.. *fold::default_ast_fold()};
  87 
  88     let fold = fold::make_fold(precursor);
  89     let res = @fold.fold_crate(&*crate);
  90     ext_cx.bt_pop();
  91     return res;
  92 }

 148 fn fold_item(cx: @mut TestCtxt, i: @ast::item, fld: @fold::ast_fold)
 149           -> Option<@ast::item> {
 150     cx.path.push(i.ident);
 151     debug!("current path: %s",
 152            ast_util::path_name_i(cx.path.clone()));
 153 
 154     if is_test_fn(cx, i) || is_bench_fn(i) {
 155         match i.node {
 156           ast::item_fn(_, purity, _, _, _) if purity == ast::unsafe_fn => {

 161           }

 164             let test = Test {
 165                 span: i.span,
 166                 path: cx.path.clone(),
 167                 bench: is_bench_fn(i),

 169                 should_fail: should_fail(i)
 170             };
 171             cx.testfns.push(test);

 178     cx.path.pop();

 182 fn is_test_fn(cx: @mut TestCtxt, i: @ast::item) -> bool {

 185     fn has_test_signature(i: @ast::item) -> bool {
 186         match &i.node {
 187           &ast::item_fn(ref decl, _, _, ref generics, _) => {
 188             let no_output = match decl.output.node {
 189                 ast::ty_nil => true,
 190                 _ => false
 191             };
 192             decl.inputs.is_empty()
 193                 && no_output
 194                 && !generics.is_parameterized()
 195           }
 196           _ => false
 197         }
 198     }
 199 
 200     if has_test_attr && !has_test_signature(i) {
 201         let sess = cx.sess;
 202         sess.span_err(
 203             i.span,
 204             "functions used as tests must have signature fn() -> ()."
 205         );
 206     }

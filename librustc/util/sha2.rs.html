  20  /// Write a u32 into a vector, which must be 4 bytes long. The value is written in big-endian
  21  /// format.
  22  fn write_u32_be(dst: &mut[u8], input: u32) {
  23      use std::cast::transmute;
  24      use std::mem::to_be32;
  25      assert!(dst.len() == 4);
  26      unsafe {
  27          let x: *mut i32 = transmute(dst.unsafe_mut_ref(0));
  28          *x = to_be32(input as i32);
  29      }
  30  }

  32  /// Read a vector of bytes into a vector of u32s. The values are read in big-endian format.
  33  fn read_u32v_be(dst: &mut[u32], input: &[u8]) {
  34      use std::cast::transmute;
  35      use std::mem::to_be32;
  36      assert!(dst.len() * 4 == input.len());
  37      unsafe {
  38          let mut x: *mut i32 = transmute(dst.unsafe_mut_ref(0));
  39          let mut y: *i32 = transmute(input.unsafe_ref(0));
  40          for _ in range(0, dst.len()) {
  41              *x = to_be32(*y);
  42              x = x.offset(1);
  43              y = y.offset(1);
  44          }
  45      }
  46  }

  48  trait ToBits {
  49      /// Convert the value in bytes to the number of bits, a tuple where the 1st item is the
  50      /// high-order value and the 2nd item is the low order value.
  51      fn to_bits(self) -> (Self, Self);
  52  }
  53  
  54  impl ToBits for u64 {
  55      fn to_bits(self) -> (u64, u64) {
  56          return (self >> 61, self << 3);
  57      }
  58  }

  60  /// Adds the specified number of bytes to the bit count. fail!() if this would cause numeric
  61  /// overflow.
  62  fn add_bytes_to_bits<T: Int + CheckedAdd + ToBits>(bits: T, bytes: T) -> T {
  63      let (new_high_bits, new_low_bits) = bytes.to_bits();
  64  
  65      if new_high_bits > Zero::zero() {
  66          fail!("numeric overflow occured.")
  67      }
  68  
  69      match bits.checked_add(&new_low_bits) {
  70          Some(x) => return x,
  71          None => fail!("numeric overflow occured.")
  72      }
  73  }

  75  /// A FixedBuffer, likes its name implies, is a fixed size buffer. When the buffer becomes full, it
  76  /// must be processed. The input() method takes care of processing and then clearing the buffer
  77  /// automatically. However, other methods do not and require the caller to process the buffer. Any
  78  /// method that modifies the buffer directory or provides the caller with bytes that can be modified
  79  /// results in those bytes being marked as used by the buffer.
  80  trait FixedBuffer {
  81      /// Input a vector of bytes. If the buffer becomes full, process it with the provided
  82      /// function and then clear the buffer.
  83      fn input(&mut self, input: &[u8], func: |&[u8]|);
  84  
  85      /// Reset the buffer.
  86      fn reset(&mut self);
  87  
  88      /// Zero the buffer up until the specified index. The buffer position currently must not be
  89      /// greater than that index.
  90      fn zero_until(&mut self, idx: uint);
  91  
  92      /// Get a slice of the buffer of the specified size. There must be at least that many bytes
  93      /// remaining in the buffer.
  94      fn next<'s>(&'s mut self, len: uint) -> &'s mut [u8];
  95  
  96      /// Get the current buffer. The buffer must already be full. This clears the buffer as well.
  97      fn full_buffer<'s>(&'s mut self) -> &'s [u8];
  98  
  99      /// Get the current position of the buffer.
 100      fn position(&self) -> uint;
 101  
 102      /// Get the number of bytes remaining in the buffer until it is full.
 103      fn remaining(&self) -> uint;
 104  
 105      /// Get the size of the buffer
 106      fn size(&self) -> uint;
 107  }

 109  /// A FixedBuffer of 64 bytes useful for implementing Sha256 which has a 64 byte blocksize.
 110  struct FixedBuffer64 {
 111      buffer: [u8, ..64],
 112      buffer_idx: uint,
 113  }
 114  
 115  impl FixedBuffer64 {
 116      /// Create a new FixedBuffer64
 117      fn new() -> FixedBuffer64 {
 118          return FixedBuffer64 {
 119              buffer: [0u8, ..64],
 120              buffer_idx: 0
 121          };
 122      }
 123  }
 124  
 125  impl FixedBuffer for FixedBuffer64 {
 126      fn input(&mut self, input: &[u8], func: |&[u8]|) {
 127          let mut i = 0;
 128  
 129          let size = self.size();
 130  
 131          // If there is already data in the buffer, copy as much as we can into it and process
 132          // the data if the buffer becomes full.
 133          if self.buffer_idx != 0 {
 134              let buffer_remaining = size - self.buffer_idx;
 135              if input.len() >= buffer_remaining {
 136                      copy_memory(
 137                          self.buffer.mut_slice(self.buffer_idx, size),
 138                          input.slice_to(buffer_remaining));
 139                  self.buffer_idx = 0;
 140                  func(self.buffer);
 141                  i += buffer_remaining;
 142              } else {
 143                  copy_memory(
 144                      self.buffer.mut_slice(self.buffer_idx, self.buffer_idx + input.len()),
 145                      input);
 146                  self.buffer_idx += input.len();
 147                  return;
 148              }
 149          }
 150  
 151          // While we have at least a full buffer size chunks's worth of data, process that data
 152          // without copying it into the buffer
 153          while input.len() - i >= size {
 154              func(input.slice(i, i + size));
 155              i += size;
 156          }
 157  
 158          // Copy any input data into the buffer. At this point in the method, the ammount of
 159          // data left in the input vector will be less than the buffer size and the buffer will
 160          // be empty.
 161          let input_remaining = input.len() - i;
 162          copy_memory(
 163              self.buffer.mut_slice(0, input_remaining),
 164              input.slice_from(i));
 165          self.buffer_idx += input_remaining;
 166      }
 167  
 168      fn reset(&mut self) {
 169          self.buffer_idx = 0;
 170      }
 171  
 172      fn zero_until(&mut self, idx: uint) {
 173          assert!(idx >= self.buffer_idx);
 174          self.buffer.mut_slice(self.buffer_idx, idx).set_memory(0);
 175          self.buffer_idx = idx;
 176      }
 177  
 178      fn next<'s>(&'s mut self, len: uint) -> &'s mut [u8] {
 179          self.buffer_idx += len;
 180          return self.buffer.mut_slice(self.buffer_idx - len, self.buffer_idx);
 181      }
 182  
 183      fn full_buffer<'s>(&'s mut self) -> &'s [u8] {
 184          assert!(self.buffer_idx == 64);
 185          self.buffer_idx = 0;
 186          return self.buffer.slice_to(64);
 187      }
 188  
 189      fn position(&self) -> uint { self.buffer_idx }
 190  
 191      fn remaining(&self) -> uint { 64 - self.buffer_idx }
 192  
 193      fn size(&self) -> uint { 64 }
 194  }

 196  /// The StandardPadding trait adds a method useful for Sha256 to a FixedBuffer struct.
 197  trait StandardPadding {
 198      /// Add padding to the buffer. The buffer must not be full when this method is called and is
 199      /// guaranteed to have exactly rem remaining bytes when it returns. If there are not at least
 200      /// rem bytes available, the buffer will be zero padded, processed, cleared, and then filled
 201      /// with zeros again until only rem bytes are remaining.
 202      fn standard_padding(&mut self, rem: uint, func: |&[u8]|);
 203  }
 204  
 205  impl <T: FixedBuffer> StandardPadding for T {
 206      fn standard_padding(&mut self, rem: uint, func: |&[u8]|) {
 207          let size = self.size();
 208  
 209          self.next(1)[0] = 128;
 210  
 211          if self.remaining() < rem {
 212              self.zero_until(size);
 213              func(self.full_buffer());
 214          }
 215  
 216          self.zero_until(size - rem);
 217      }
 218  }

 220  /// The Digest trait specifies an interface common to digest functions, such as SHA-1 and the SHA-2
 221  /// family of digest functions.
 222  pub trait Digest {
 223      /// Provide message data.
 224      ///
 225      /// # Arguments
 226      ///
 227      /// * input - A vector of message data
 228      fn input(&mut self, input: &[u8]);
 229  
 230      /// Retrieve the digest result. This method may be called multiple times.
 231      ///
 232      /// # Arguments
 233      ///
 234      /// * out - the vector to hold the result. Must be large enough to contain output_bits().
 235      fn result(&mut self, out: &mut [u8]);
 236  
 237      /// Reset the digest. This method must be called after result() and before supplying more
 238      /// data.
 239      fn reset(&mut self);
 240  
 241      /// Get the output size in bits.
 242      fn output_bits(&self) -> uint;
 243  
 244      /// Convenience function that feeds a string into a digest.
 245      ///
 246      /// # Arguments
 247      ///
 248      /// * `input` The string to feed into the digest
 249      fn input_str(&mut self, input: &str) {
 250          self.input(input.as_bytes());
 251      }
 252  
 253      /// Convenience function that retrieves the result of a digest as a
 254      /// newly allocated vec of bytes.
 255      fn result_bytes(&mut self) -> Vec<u8> {
 256          let mut buf = Vec::from_elem((self.output_bits()+7)/8, 0u8);
 257          self.result(buf.as_mut_slice());
 258          buf
 259      }
 260  
 261      /// Convenience function that retrieves the result of a digest as a
 262      /// ~str in hexadecimal format.
 263      fn result_str(&mut self) -> ~str {
 264          self.result_bytes().as_slice().to_hex()
 265      }
 266  }

 306      fn process_block(&mut self, data: &[u8]) {
 307          fn ch(x: u32, y: u32, z: u32) -> u32 {
 308              ((x & y) ^ ((!x) & z))
 309          }
 310  
 311          fn maj(x: u32, y: u32, z: u32) -> u32 {
 312              ((x & y) ^ (x & z) ^ (y & z))
 313          }
 314  
 315          fn sum0(x: u32) -> u32 {
 316              ((x >> 2) | (x << 30)) ^ ((x >> 13) | (x << 19)) ^ ((x >> 22) | (x << 10))
 317          }
 318  
 319          fn sum1(x: u32) -> u32 {
 320              ((x >> 6) | (x << 26)) ^ ((x >> 11) | (x << 21)) ^ ((x >> 25) | (x << 7))
 321          }
 322  
 323          fn sigma0(x: u32) -> u32 {
 324              ((x >> 7) | (x << 25)) ^ ((x >> 18) | (x << 14)) ^ (x >> 3)
 325          }
 326  
 327          fn sigma1(x: u32) -> u32 {
 328              ((x >> 17) | (x << 15)) ^ ((x >> 19) | (x << 13)) ^ (x >> 10)
 329          }

 342          // Sha-512 and Sha-256 use basically the same calculations which are implemented
 343          // by these macros. Inlining the calculations seems to result in better generated code.
 344          macro_rules! schedule_round( ($t:expr) => (
 345                  w[$t] = sigma1(w[$t - 2]) + w[$t - 7] + sigma0(w[$t - 15]) + w[$t - 16];
 346                  )
 347          )
 348  
 349          macro_rules! sha2_round(
 350              ($A:ident, $B:ident, $C:ident, $D:ident,
 351               $E:ident, $F:ident, $G:ident, $H:ident, $K:ident, $t:expr) => (
 352                  {
 353                      $H += sum1($E) + ch($E, $F, $G) + $K[$t] + w[$t];
 354                      $D += $H;
 355                      $H += sum0($A) + maj($A, $B, $C);
 356                  }
 357               )
 358          )

 363          // the compiler to generate better code.

 372              schedule_round!(t + 23);

   61  pub fn explain_region_and_span(cx: &ctxt, region: ty::Region)
   62                              -> (~str, Option<Span>) {
   63      return match region {
   64        ReScope(node_id) => {
   65          match cx.map.find(node_id) {
   66            Some(ast_map::NodeBlock(ref blk)) => {
   67              explain_span(cx, "block", blk.span)
   68            }
   69            Some(ast_map::NodeExpr(expr)) => {
   70              match expr.node {
   71                ast::ExprCall(..) => explain_span(cx, "call", expr.span),
   72                ast::ExprMethodCall(..) => {
   73                  explain_span(cx, "method call", expr.span)
   74                },
   75                ast::ExprMatch(..) => explain_span(cx, "match", expr.span),
   76                _ => explain_span(cx, "expression", expr.span)
   77              }
   78            }
   79            Some(ast_map::NodeStmt(stmt)) => {
   80                explain_span(cx, "statement", stmt.span)
   81            }
   82            Some(ast_map::NodeItem(it)) if (match it.node {
   83                  ast::ItemFn(..) => true, _ => false}) => {
   84                explain_span(cx, "function body", it.span)
   85            }
   86            Some(_) | None => {
   87              // this really should not happen
   88              (format!("unknown scope: {}.  Please report a bug.", node_id),
   89               None)
   90            }
   91          }
   92        }
   93  
   94        ReFree(ref fr) => {
   95          let prefix = match fr.bound_region {
   96            BrAnon(idx) => format!("the anonymous lifetime \\#{} defined on",
   97                                 idx + 1),
   98            BrFresh(_) => format!("an anonymous lifetime defined on"),
   99            _ => format!("the lifetime {} as defined on",
  100                      bound_region_ptr_to_str(cx, fr.bound_region))
  101          };
  102  
  103          match cx.map.find(fr.scope_id) {
  104            Some(ast_map::NodeBlock(ref blk)) => {
  105              let (msg, opt_span) = explain_span(cx, "block", blk.span);
  106              (format!("{} {}", prefix, msg), opt_span)
  107            }
  108            Some(ast_map::NodeItem(it)) if match it.node {
  109                  ast::ItemImpl(..) => true, _ => false} => {
  110              let (msg, opt_span) = explain_span(cx, "impl", it.span);
  111              (format!("{} {}", prefix, msg), opt_span)
  112            }
  113            Some(_) | None => {
  114              // this really should not happen
  115              (format!("{} node {}", prefix, fr.scope_id), None)
  116            }
  117          }
  118        }
  119  
  120        ReStatic => { (~"the static lifetime", None) }
  121  
  122        ReEmpty => { (~"the empty lifetime", None) }
  123  
  124        // I believe these cases should not occur (except when debugging,
  125        // perhaps)
  126        ty::ReInfer(_) | ty::ReEarlyBound(..) | ty::ReLateBound(..) => {
  127          (format!("lifetime {:?}", region), None)
  128        }
  129      };

  131      fn explain_span(cx: &ctxt, heading: &str, span: Span)
  132          -> (~str, Option<Span>) {
  133          let lo = cx.sess.codemap().lookup_char_pos_adj(span.lo);
  134          (format!("the {} at {}:{}", heading,
  135                lo.line, lo.col.to_uint()), Some(span))
  136      }
  137  }

  143  pub fn bound_region_to_str(cx: &ctxt,
  144                             prefix: &str, space: bool,
  145                             br: BoundRegion) -> ~str {
  146      let space_str = if space { " " } else { "" };
  147  
  148      if cx.sess.verbose() {
  149          return format!("{}{}{}", prefix, br.repr(cx), space_str);
  150      }
  151  
  152      match br {
  153          BrNamed(_, name)   => format!("{}'{}{}", prefix,
  154                                        token::get_name(name), space_str),
  155          BrAnon(_)           => prefix.to_str(),
  156          BrFresh(_)          => prefix.to_str(),
  157      }
  158  }

  161  // you should use `explain_region()` or, better yet,

  163  pub fn region_ptr_to_str(cx: &ctxt, region: Region) -> ~str {
  164      region_to_str(cx, "&", true, region)
  165  }

  167  pub fn region_to_str(cx: &ctxt, prefix: &str, space: bool, region: Region) -> ~str {
  168      let space_str = if space { " " } else { "" };
  169  
  170      if cx.sess.verbose() {
  171          return format!("{}{}{}", prefix, region.repr(cx), space_str);
  172      }
  173  
  174      // These printouts are concise.  They do not contain all the information
  175      // the user might want to diagnose an error, but there is basically no way
  176      // to fit that into a short string.  Hence the recommendation to use
  177      // `explain_region()` or `note_and_explain_region()`.
  178      match region {
  179          ty::ReScope(_) => prefix.to_str(),
  180          ty::ReEarlyBound(_, _, name) => token::get_name(name).get().to_str(),
  181          ty::ReLateBound(_, br) => bound_region_to_str(cx, prefix, space, br),
  182          ty::ReFree(ref fr) => bound_region_to_str(cx, prefix, space, fr.bound_region),
  183          ty::ReInfer(ReSkolemized(_, br)) => {
  184              bound_region_to_str(cx, prefix, space, br)
  185          }
  186          ty::ReInfer(ReVar(_)) => prefix.to_str(),
  187          ty::ReStatic => format!("{}'static{}", prefix, space_str),
  188          ty::ReEmpty => format!("{}'<empty>{}", prefix, space_str)
  189      }
  190  }

  254      fn bare_fn_to_str(cx: &ctxt,
  255                        purity: ast::Purity,
  256                        abi: abi::Abi,
  257                        ident: Option<ast::Ident>,
  258                        sig: &ty::FnSig)
  259                        -> ~str {
  260          let mut s = if abi == abi::Rust {
  261              ~""
  262          } else {
  263              format!("extern {} ", abi.to_str())
  264          };
  265  
  266          match purity {
  267              ast::ImpureFn => {}
  268              _ => {
  269                  s.push_str(purity.to_str());
  270                  s.push_char(' ');
  271              }
  272          };
  273  
  274          s.push_str("fn");
  275  
  276          match ident {
  277              Some(i) => {
  278                  s.push_char(' ');
  279                  s.push_str(token::get_ident(i).get());
  280              }
  281              _ => { }
  282          }
  283  
  284          push_sig_to_str(cx, &mut s, '(', ')', sig);
  285  
  286          return s;
  287      }
  288      fn closure_to_str(cx: &ctxt, cty: &ty::ClosureTy) -> ~str {
  289          let is_proc =
  290              (cty.sigil, cty.onceness) == (ast::OwnedSigil, ast::Once);
  291          let is_borrowed_closure = cty.sigil == ast::BorrowedSigil;
  292  
  293          let mut s = if is_proc || is_borrowed_closure {
  294              ~""
  295          } else {
  296              cty.sigil.to_str()
  297          };
  298  
  299          match (cty.sigil, cty.region) {
  300              (ast::ManagedSigil, ty::ReStatic) |
  301              (ast::OwnedSigil, ty::ReStatic) => {}
  302  
  303              (_, region) => {
  304                  s.push_str(region_to_str(cx, "", true, region));
  305              }
  306          }
  307  
  308          match cty.purity {
  309              ast::ImpureFn => {}
  310              _ => {
  311                  s.push_str(cty.purity.to_str());
  312                  s.push_char(' ');
  313              }
  314          };
  315  
  316          if is_proc {
  317              s.push_str("proc");
  318          } else {
  319              match cty.onceness {
  320                  ast::Many => {}
  321                  ast::Once => {
  322                      s.push_str(cty.onceness.to_str());
  323                      s.push_char(' ');
  324                  }
  325              };
  326  
  327              if !is_borrowed_closure {
  328                  s.push_str("fn");
  329              }
  330          }
  331  
  332          if !is_borrowed_closure {
  333              // Print bounds before `fn` if this is not a borrowed closure.
  334              if !cty.bounds.is_empty() {
  335                  s.push_str(":");
  336                  s.push_str(cty.bounds.repr(cx));
  337              }
  338  
  339              push_sig_to_str(cx, &mut s, '(', ')', &cty.sig);
  340          } else {
  341              // Print bounds after the signature if this is a borrowed closure.
  342              push_sig_to_str(cx, &mut s, '|', '|', &cty.sig);
  343  
  344              if is_borrowed_closure {
  345                  if !cty.bounds.is_empty() {
  346                      s.push_str(":");
  347                      s.push_str(cty.bounds.repr(cx));
  348                  }
  349              }
  350          }
  351  
  352          return s;
  353      }
  354      fn push_sig_to_str(cx: &ctxt,
  355                         s: &mut ~str,
  356                         bra: char,
  357                         ket: char,
  358                         sig: &ty::FnSig) {
  359          s.push_char(bra);
  360          let strs: Vec<~str> = sig.inputs.iter().map(|a| fn_input_to_str(cx, *a)).collect();
  361          s.push_str(strs.connect(", "));
  362          if sig.variadic {
  363              s.push_str(", ...");
  364          }
  365          s.push_char(ket);
  366  
  367          if ty::get(sig.output).sty != ty_nil {
  368              s.push_str(" -> ");
  369              if ty::type_is_bot(sig.output) {
  370                  s.push_char('!');
  371              } else {
  372                  s.push_str(ty_to_str(cx, sig.output));
  373              }
  374          }
  375      }
  376  
  377      // if there is an id, print that instead of the structural type:
  378      /*for def_id in ty::type_def_id(typ).iter() {
  379          // note that this typedef cannot have type parameters
  380          return ty::item_path_str(cx, *def_id);
  381      }*/
  382  
  383      // pretty print the structural type representation:
  384      return match ty::get(typ).sty {
  385        ty_nil => ~"()",
  386        ty_bot => ~"!",
  387        ty_bool => ~"bool",
  388        ty_char => ~"char",
  389        ty_int(ast::TyI) => ~"int",
  390        ty_int(t) => ast_util::int_ty_to_str(t),
  391        ty_uint(ast::TyU) => ~"uint",
  392        ty_uint(t) => ast_util::uint_ty_to_str(t),
  393        ty_float(t) => ast_util::float_ty_to_str(t),
  394        ty_box(typ) => ~"@" + ty_to_str(cx, typ),
  395        ty_uniq(typ) => ~"~" + ty_to_str(cx, typ),
  396        ty_ptr(ref tm) => ~"*" + mt_to_str(cx, tm),
  397        ty_rptr(r, ref tm) => {
  398          region_ptr_to_str(cx, r) + mt_to_str(cx, tm)
  399        }
  400        ty_tup(ref elems) => {
  401          let strs: Vec<~str> = elems.iter().map(|elem| ty_to_str(cx, *elem)).collect();
  402          ~"(" + strs.connect(",") + ")"
  403        }
  404        ty_closure(ref f) => {
  405            closure_to_str(cx, *f)
  406        }
  407        ty_bare_fn(ref f) => {
  408            bare_fn_to_str(cx, f.purity, f.abi, None, &f.sig)
  409        }
  410        ty_infer(infer_ty) => infer_ty.to_str(),
  411        ty_err => ~"[type error]",
  412        ty_param(param_ty {idx: id, def_id: did}) => {
  413            let ident = match cx.ty_param_defs.borrow().find(&did.node) {
  414                Some(def) => token::get_ident(def.ident).get().to_str(),
  415                // This should not happen...
  416                None => format!("BUG[{:?}]", id)
  417            };
  418            if !cx.sess.verbose() {
  419                ident
  420            } else {
  421              format!("{}:{:?}", ident, did)
  422            }
  423        }
  424        ty_self(..) => ~"Self",
  425        ty_enum(did, ref substs) | ty_struct(did, ref substs) => {
  426          let base = ty::item_path_str(cx, did);
  427          parameterized(cx,
  428                        base,
  429                        &substs.regions,
  430                        substs.tps.as_slice(),
  431                        did,
  432                        false)
  433        }
  434        ty_trait(~ty::TyTrait {
  435            def_id: did, ref substs, store: s, mutability: mutbl, ref bounds
  436        }) => {
  437          let base = ty::item_path_str(cx, did);
  438          let ty = parameterized(cx, base, &substs.regions,
  439                                 substs.tps.as_slice(), did, true);
  440          let bound_sep = if bounds.is_empty() { "" } else { ":" };
  441          let bound_str = bounds.repr(cx);
  442          format!("{}{}{}{}{}", trait_store_to_str(cx, s), mutability_to_str(mutbl), ty,
  443                             bound_sep, bound_str)
  444        }
  445        ty_vec(ref mt, vs) => {
  446          vstore_ty_to_str(cx, mt, vs)
  447        }
  448        ty_str(vs) => format!("{}{}", vstore_to_str(cx, vs), "str")
  449      }
  450  }
  451  
  452  pub fn parameterized(cx: &ctxt,
  453                       base: &str,
  454                       regions: &ty::RegionSubsts,
  455                       tps: &[ty::t],
  456                       did: ast::DefId,
  457                       is_trait: bool) -> ~str {
  458  
  459      let mut strs = Vec::new();
  460      match *regions {
  461          ty::ErasedRegions => { }
  462          ty::NonerasedRegions(ref regions) => {
  463              for &r in regions.iter() {
  464                  strs.push(region_to_str(cx, "", false, r))
  465              }
  466          }
  467      }
  468  
  469      let generics = if is_trait {
  470          ty::lookup_trait_def(cx, did).generics.clone()
  471      } else {
  472          ty::lookup_item_type(cx, did).generics
  473      };
  474      let ty_params = generics.type_param_defs();
  475      let has_defaults = ty_params.last().map_or(false, |def| def.default.is_some());
  476      let num_defaults = if has_defaults {
  477          // We should have a borrowed version of substs instead of cloning.
  478          let mut substs = ty::substs {
  479              tps: Vec::from_slice(tps),
  480              regions: regions.clone(),
  481              self_ty: None
  482          };
  483          ty_params.iter().zip(tps.iter()).rev().take_while(|&(def, &actual)| {
  484              substs.tps.pop();
  485              match def.default {
  486                  Some(default) => ty::subst(cx, &substs, default) == actual,
  487                  None => false
  488              }
  489          }).len()
  490      } else {
  491          0
  492      };
  493  
  494      for t in tps.slice_to(tps.len() - num_defaults).iter() {
  495          strs.push(ty_to_str(cx, *t))
  496      }
  497  
  498      if strs.len() > 0u {
  499          format!("{}<{}>", base, strs.connect(","))
  500      } else {
  501          format!("{}", base)
  502      }
  503  }

  507      if s.len() >= 32u { s = s.slice(0u, 32u).to_owned(); }
  508      return s;
  509  }

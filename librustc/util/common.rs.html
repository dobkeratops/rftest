  64  pub fn loop_query(b: &ast::Block, p: @fn(&ast::expr_) -> bool) -> bool {
  65      let rs = @mut false;
  66      let visit_expr: @fn(@ast::expr,
  67                          (@mut bool,
  68                           oldvisit::vt<@mut bool>)) = |e, (flag, v)| {
  69          *flag |= p(&e.node);
  70          match e.node {
  71            // Skip inner loops, since a break in the inner loop isn't a
  72            // break inside the outer loop
  73            ast::expr_loop(*) | ast::expr_while(*) => {}
  74            _ => oldvisit::visit_expr(e, (flag, v))
  75          }
  76      };
  77      let v = oldvisit::mk_vt(@oldvisit::Visitor {
  78          visit_expr: visit_expr,
  79          .. *oldvisit::default_visitor()});
  80      oldvisit::visit_block(b, (rs, v));
  81      return *rs;
  82  }

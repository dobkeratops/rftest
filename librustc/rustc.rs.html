 138  pub fn describe_warnings() {
 139      use extra::sort::Sort;
 140      println("
 141  Available lint options:
 142      -W <foo>           Warn about <foo>
 143      -A <foo>           Allow <foo>
 144      -D <foo>           Deny <foo>
 145      -F <foo>           Forbid <foo> (deny, and deny all overrides)
 146  ");
 147  
 148      let lint_dict = lint::get_lint_dict();
 149      let mut lint_dict = lint_dict.move_iter()
 150                                   .map(|(k, v)| (v, k))
 151                                   .collect::<~[(lint::LintSpec, &'static str)]>();
 152      lint_dict.qsort();
 153  
 154      let mut max_key = 0;
 155      for &(_, name) in lint_dict.iter() {
 156          max_key = num::max(name.len(), max_key);
 157      }
 158      fn padded(max: uint, s: &str) -> ~str {
 159          str::from_bytes(vec::from_elem(max - s.len(), ' ' as u8)) + s
 160      }
 161      println("\nAvailable lint checks:\n");
 162      printfln!("    %s  %7.7s  %s",
 163                padded(max_key, "name"), "default", "meaning");
 164      printfln!("    %s  %7.7s  %s\n",
 165                padded(max_key, "----"), "-------", "-------");
 166      for (spec, name) in lint_dict.move_iter() {
 167          let name = name.replace("_", "-");
 168          printfln!("    %s  %7.7s  %s",
 169                    padded(max_key, name),
 170                    lint::level_to_str(spec.default),
 171                    spec.desc);
 172      }
 173      io::println("");
 174  }

 188  pub fn run_compiler(args: &~[~str], demitter: diagnostic::Emitter) {
 189      // Don't display log spew by default. Can override with RUST_LOG.
 190      ::std::logging::console_off();
 191  
 192      let mut args = (*args).clone();
 193      let binary = args.shift().to_managed();
 194  
 195      if args.is_empty() { usage(binary); return; }
 196  
 197      let matches =
 198          &match getopts::groups::getopts(args, optgroups()) {
 199            Ok(m) => m,
 200            Err(f) => {
 201              early_error(demitter, getopts::fail_str(f));
 202            }
 203          };
 204  
 205      if opt_present(matches, "h") || opt_present(matches, "help") {
 206          usage(binary);
 207          return;
 208      }
 209  
 210      // Display the available lint options if "-W help" or only "-W" is given.
 211      let lint_flags = vec::append(getopts::opt_strs(matches, "W"),
 212                                   getopts::opt_strs(matches, "warn"));
 213  
 214      let show_lint_options = lint_flags.iter().any(|x| x == &~"help") ||
 215          (opt_present(matches, "W") && lint_flags.is_empty());
 216  
 217      if show_lint_options {
 218          describe_warnings();
 219          return;
 220      }
 221  
 222      let r = getopts::opt_strs(matches, "Z");
 223      if r.iter().any(|x| x == &~"help") {
 224          describe_debug_flags();
 225          return;
 226      }
 227  
 228      if getopts::opt_maybe_str(matches, "passes") == Some(~"list") {
 229          back::passes::list_passes();
 230          return;
 231      }
 232  
 233      if opt_present(matches, "v") || opt_present(matches, "version") {
 234          version(binary);
 235          return;
 236      }
 237      let input = match matches.free.len() {
 238        0u => early_error(demitter, ~"no input filename given"),
 239        1u => {
 240          let ifile = matches.free[0].as_slice();
 241          if "-" == ifile {
 242              let src = str::from_bytes(io::stdin().read_whole_stream());
 243              str_input(src.to_managed())
 244          } else {
 245              file_input(Path(ifile))
 246          }
 247        }
 248        _ => early_error(demitter, ~"multiple input filenames provided")
 249      };
 250  
 251      let sopts = build_session_options(binary, matches, demitter);
 252      let sess = build_session(sopts, demitter);
 253      let odir = getopts::opt_maybe_str(matches, "out-dir").map_move(|o| Path(o));
 254      let ofile = getopts::opt_maybe_str(matches, "o").map_move(|o| Path(o));
 255      let cfg = build_configuration(sess, binary, &input);
 256      let pretty = do getopts::opt_default(matches, "pretty", "normal").map_move |a| {
 257          parse_pretty(sess, a)
 258      };
 259      match pretty {
 260        Some::<pp_mode>(ppm) => {
 261          pretty_print_input(sess, cfg, &input, ppm);
 262          return;
 263        }
 264        None::<pp_mode> => {/* continue */ }
 265      }
 266      let ls = opt_present(matches, "ls");
 267      if ls {
 268          match input {
 269            file_input(ref ifile) => {
 270              list_metadata(sess, &(*ifile), io::stdout());
 271            }
 272            str_input(_) => {
 273              early_error(demitter, ~"can not list metadata for stdin");
 274            }
 275          }
 276          return;
 277      }
 278  
 279      compile_input(sess, cfg, &input, &odir, &ofile);
 280  }

 285      done,

 288  /*
 289  This is a sanity check that any failure of the compiler is performed
 290  through the diagnostic module and reported properly - we shouldn't be calling
 291  plain-old-fail on any execution path that might be taken. Since we have
 292  console logging off by default, hitting a plain fail statement would make the
 293  compiler silently exit, which would be terrible.
 294  
 295  This method wraps the compiler in a subtask and injects a function into the
 296  diagnostic emitter which records when we hit a fatal error. If the task
 297  fails without recording a fatal error then we've encountered a compiler
 298  bug and need to present an error.
 299  */
 300  pub fn monitor(f: ~fn(diagnostic::Emitter)) {
 301      use std::comm::*;
 302  
 303      // XXX: This is a hack for newsched since it doesn't support split stacks.
 304      // rustc needs a lot of stack!
 305      static STACK_SIZE: uint = 6000000;
 306  
 307      let (p, ch) = stream();
 308      let ch = SharedChan::new(ch);
 309      let ch_capture = ch.clone();
 310      let mut task_builder = task::task();
 311      task_builder.supervised();
 312  
 313      // XXX: Hacks on hacks. If the env is trying to override the stack size
 314      // then *don't* set it explicitly.
 315      if os::getenv("RUST_MIN_STACK").is_none() {
 316          task_builder.opts.stack_size = Some(STACK_SIZE);
 317      }
 318  
 319      match do task_builder.try {
 320          let ch = ch_capture.clone();
 321          let ch_capture = ch.clone();
 322          // The 'diagnostics emitter'. Every error, warning, etc. should
 323          // go through this function.
 324          let demitter: @fn(Option<(@codemap::CodeMap, codemap::span)>,
 325                            &str,
 326                            diagnostic::level) =
 327                            |cmsp, msg, lvl| {
 328              if lvl == diagnostic::fatal {
 329                  ch_capture.send(fatal);
 330              }
 331              diagnostic::emit(cmsp, msg, lvl);
 332          };
 333  
 334          struct finally {
 335              ch: SharedChan<monitor_msg>,
 336          }
 337  
 338          impl Drop for finally {
 339              fn drop(&self) { self.ch.send(done); }
 340          }
 341  
 342          let _finally = finally { ch: ch };
 343  
 344          f(demitter);
 345  
 346          // Due reasons explain in #7732, if there was a jit execution context it
 347          // must be consumed and passed along to our parent task.
 348          back::link::jit::consume_engine()
 349      } {
 350          result::Ok(_) => { /* fallthrough */ }
 351          result::Err(_) => {
 352              // Task failed without emitting a fatal diagnostic
 353              if p.recv() == done {
 354                  diagnostic::emit(
 355                      None,
 356                      diagnostic::ice_msg("unexpected failure"),
 357                      diagnostic::error);
 358  
 359                  let xs = [
 360                      ~"the compiler hit an unexpected failure path. \
 361                       this is a bug",
 362                      ~"try running with RUST_LOG=rustc=1,::rt::backtrace \
 363                       to get further details and report the results \
 364                       to github.com/mozilla/rust/issues"
 365                  ];
 366                  for note in xs.iter() {
 367                      diagnostic::emit(None, *note, diagnostic::note)
 368                  }
 369              }
 370              // Fail so the process returns a failure code
 371              fail!();
 372          }
 373      }
 374  }

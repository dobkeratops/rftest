  33  // Any local node that may call something in its body block should be
  34  // explored. For example, if it's a live NodeItem that is a
  35  // function, then we should explore its block to check for codes that
  36  // may need to be marked as live.
  37  fn should_explore(tcx: &ty::ctxt, def_id: ast::DefId) -> bool {
  38      if !is_local(def_id) {
  39          return false;
  40      }
  41  
  42      match tcx.map.find(def_id.node) {
  43          Some(ast_map::NodeItem(..))
  44          | Some(ast_map::NodeMethod(..))
  45          | Some(ast_map::NodeForeignItem(..))
  46          | Some(ast_map::NodeTraitMethod(..)) => true,
  47          _ => false
  48      }
  49  }

 129      fn mark_live_symbols(&mut self) {

 205  // This visitor seeds items that
 206  //   1) We want to explicitly consider as live:
 207  //     * Item annotated with #[allow(dead_code)]
 208  //         - This is done so that if we want to suppress warnings for a
 209  //           group of dead functions, we only have to annotate the "root".
 210  //           For example, if both `f` and `g` are dead and `f` calls `g`,
 211  //           then annotating `f` with `#[allow(dead_code)]` will suppress
 212  //           warning for both `f` and `g`.
 213  //     * Item annotated with #[lang=".."]
 214  //         - This is because lang items are always callable from elsewhere.
 215  //   or

 322  impl<'a> DeadVisitor<'a> {
 323      // id := node id of an item's definition.
 324      // ctor_id := `Some` if the item is a struct_ctor (tuple struct),

 328      // DefMap maps the ExprPath of a struct_ctor to the node referred by
 329      // `ctor_id`. On the other hand, in a statement like
 330      // `type <ident> <generics> = <ty>;` where <ty> refers to a struct_ctor,

 333                        ctor_id: Option<ast::NodeId>) -> bool {

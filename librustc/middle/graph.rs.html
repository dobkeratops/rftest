  11  /*!
  12  
  13  A graph module for use in dataflow, region resolution, and elsewhere.
  14  
  15  # Interface details
  16  
  17  You customize the graph by specifying a "node data" type `N` and an
  18  "edge data" type `E`. You can then later gain access (mutable or
  19  immutable) to these "user-data" bits. Currently, you can only add
  20  nodes or edges to the graph. You cannot remove or modify them once
  21  added. This could be changed if we have a need.
  22  
  23  # Implementation details
  24  
  25  The main tricky thing about this code is the way that edges are
  26  stored. The edges are stored in a central array, but they are also
  27  threaded onto two linked lists for each node, one for incoming edges
  28  and one for outgoing edges. Note that every edge is a member of some
  29  incoming list and some outgoing list.  Basically you can load the
  30  first index of the linked list from the node data structures (the
  31  field `first_edge`) and then, for each edge, load the next index from
  32  the field `next_edge`). Each of those fields is an array that should
  33  be indexed by the direction (see the type `Direction`).
  34  
  35  */

 230      pub fn each_adjacent_edge(&self,
 231                                node: NodeIndex,
 232                                dir: Direction,
 233                                f: |EdgeIndex, &Edge<E>| -> bool)
 234                                -> bool {
 235          //! Iterates over all edges adjacent to the node `node`
 236          //! in the direction `dir` (either `Outgoing` or `Incoming)
 237  
 238          let mut edge_idx = self.first_adjacent(node, dir);
 239          while edge_idx != InvalidEdgeIndex {
 240              let edge = self.edges.get(edge_idx.get());
 241              if !f(edge_idx, edge) {
 242                  return false;
 243              }
 244              edge_idx = edge.next_edge[dir.repr];
 245          }
 246          return true;
 247      }

 250      // Fixed-point iteration
 251      //
 252      // A common use for graphs in our compiler is to perform
 253      // fixed-point iteration. In this case, each edge represents a
 254      // constaint, and the nodes themselves are associated with
 255      // variables or other bitsets. This method facilitates such a
 256      // computation.
 257  
 258      pub fn iterate_until_fixed_point(&self,
 259                                       op: |iter_index: uint,
 260                                            edge_index: EdgeIndex,
 261                                            edge: &Edge<E>|
 262                                            -> bool) {
 263          let mut iteration = 0;
 264          let mut changed = true;
 265          while changed {
 266              changed = false;
 267              iteration += 1;
 268              for (i, edge) in self.edges.iter().enumerate() {
 269                  changed |= op(iteration, EdgeIndex(i), edge);
 270              }
 271          }
 272      }
 273  }

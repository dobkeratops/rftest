  11  /*!
  12  
  13  A graph module for use in dataflow, region resolution, and elsewhere.
  14  
  15  # Interface details
  16  
  17  You customize the graph by specifying a "node data" type `N` and an
  18  "edge data" type `E`. You can then later gain access (mutable or
  19  immutable) to these "user-data" bits. Currently, you can only add
  20  nodes or edges to the graph. You cannot remove or modify them once
  21  added. This could be changed if we have a need.
  22  
  23  # Implementation details
  24  
  25  The main tricky thing about this code is the way that edges are
  26  stored. The edges are stored in a central array, but they are also
  27  threaded onto two linked lists for each node, one for incoming edges
  28  and one for outgoing edges. Note that every edge is a member of some
  29  incoming list and some outgoing list.  Basically you can load the
  30  first index of the linked list from the node data structures (the
  31  field `first_edge`) and then, for each edge, load the next index from
  32  the field `next_edge`). Each of those fields is an array that should
  33  be indexed by the direction (see the type `Direction`).
  34  
  35  */

 214      pub fn each_adjacent_edge(&self,
 215                                node: NodeIndex,
 216                                dir: Direction,
 217                                f: &fn(EdgeIndex, &Edge<E>) -> bool) -> bool {
 218          //! Iterates over all edges adjacent to the node `node`
 219          //! in the direction `dir` (either `Outgoing` or `Incoming)
 220  
 221          let mut edge_idx = self.first_adjacent(node, dir);
 222          while edge_idx != InvalidEdgeIndex {
 223              let edge = &self.edges[*edge_idx];
 224              if !f(edge_idx, edge) {
 225                  return false;
 226              }
 227              edge_idx = edge.next_edge[dir.repr];
 228          }
 229          return true;
 230      }

 232      ///////////////////////////////////////////////////////////////////////////
 233      // Fixed-point iteration
 234      //
 235      // A common use for graphs in our compiler is to perform
 236      // fixed-point iteration. In this case, each edge represents a
 237      // constaint, and the nodes themselves are associated with
 238      // variables or other bitsets. This method facilitates such a
 239      // computation.
 240  
 241      pub fn iterate_until_fixed_point(&self,
 242                                       op: &fn(iter_index: uint,
 243                                               edge_index: EdgeIndex,
 244                                               edge: &Edge<E>) -> bool) {
 245          let mut iteration = 0;
 246          let mut changed = true;
 247          while changed {
 248              changed = false;
 249              iteration += 1;
 250              for (i, edge) in self.edges.iter().enumerate() {
 251                  changed |= op(iteration, EdgeIndex(i), edge);
 252              }
 253          }
 254      }
 255  }

 260      while i < n {

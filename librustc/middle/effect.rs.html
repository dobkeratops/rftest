  56      fn require_unsafe(&mut self, span: Span, description: &str) {
  57          match self.context.unsafe_context {
  58              SafeContext => {
  59                  // Report an error.
  60                  self.tcx.sess.span_err(span,
  61                                    fmt!("%s requires unsafe function or block",
  62                                         description))
  63              }
  64              UnsafeBlock(block_id) => {
  65                  // OK, but record this.
  66                  debug!("effect: recording unsafe block as used: %?", block_id);
  67                  let _ = self.tcx.used_unsafe.insert(block_id);
  68              }
  69              UnsafeFn => {}
  70          }
  71      }
  72  }

  11  // Finds items that are externally reachable, to determine which items
  12  // need to have their metadata (and possibly their AST) serialized.
  13  // All items that can be referred to through an exported name are
  14  // reachable, and when a reachable thing is inline or generic, it
  15  // makes all other generics or inline functions that it references
  16  // reachable as well.

  30  // Returns true if the given set of attributes contains the `#[inline]`
  31  // attribute.
  32  fn attributes_specify_inlining(attrs: &[Attribute]) -> bool {
  33      attr::contains_name(attrs, "inline")
  34  }
  35  
  36  // Returns true if the given set of generics implies that the item it's
  37  // associated with must be inlined.
  38  fn generics_require_inlining(generics: &Generics) -> bool {
  39      !generics.ty_params.is_empty()
  40  }
  41  
  42  // Returns true if the given item must be inlined because it may be
  43  // monomorphized or it was marked with `#[inline]`. This will only return
  44  // true for functions.
  45  fn item_might_be_inlined(item: @item) -> bool {
  46      if attributes_specify_inlining(item.attrs) {
  47          return true
  48      }
  49  
  50      match item.node {
  51          item_fn(_, _, _, ref generics, _) => {
  52              generics_require_inlining(generics)
  53          }
  54          _ => false,
  55      }
  56  }
  57  
  58  // Returns true if the given type method must be inlined because it may be
  59  // monomorphized or it was marked with `#[inline]`.
  60  fn ty_method_might_be_inlined(ty_method: &TypeMethod) -> bool {
  61      attributes_specify_inlining(ty_method.attrs) ||
  62          generics_require_inlining(&ty_method.generics)
  63  }
  64  
  65  // Returns true if the given trait method must be inlined because it may be
  66  // monomorphized or it was marked with `#[inline]`.
  67  fn trait_method_might_be_inlined(trait_method: &trait_method) -> bool {
  68      match *trait_method {
  69          required(ref ty_method) => ty_method_might_be_inlined(ty_method),
  70          provided(_) => true
  71      }
  72  }

  74  // The context we're in. If we're in a public context, then public symbols are
  75  // marked reachable. If we're in a private context, then only trait
  76  // implementations are marked reachable.
  77  #[deriving(Clone, Eq)]
  78  enum PrivacyContext {
  79      PublicContext,
  80      PrivateContext,
  81  }
  82  
  83  // Information needed while computing reachability.
  84  struct ReachableContext {
  85      // The type context.
  86      tcx: ty::ctxt,
  87      // The method map, which links node IDs of method call expressions to the
  88      // methods they've been resolved to.
  89      method_map: typeck::method_map,
  90      // The set of items which must be exported in the linkage sense.
  91      reachable_symbols: @mut HashSet<NodeId>,
  92      // A worklist of item IDs. Each item ID in this worklist will be inlined
  93      // and will be scanned for further references.
  94      worklist: @mut ~[NodeId],
  95  }
  96  
  97  struct ReachableVisitor {
  98      reachable_symbols: @mut HashSet<NodeId>,
  99      worklist: @mut ~[NodeId],
 100  }
 101  
 102  impl Visitor<PrivacyContext> for ReachableVisitor {
 103  
 104      fn visit_item(&mut self, item:@item, privacy_context:PrivacyContext) {
 105  
 106                  match item.node {
 107                      item_fn(*) => {
 108                          if privacy_context == PublicContext {
 109                              self.reachable_symbols.insert(item.id);
 110                          }
 111                          if item_might_be_inlined(item) {
 112                              self.worklist.push(item.id)
 113                          }
 114                      }
 115                      item_struct(ref struct_def, _) => {
 116                          match struct_def.ctor_id {
 117                              Some(ctor_id) if
 118                                      privacy_context == PublicContext => {
 119                                  self.reachable_symbols.insert(ctor_id);
 120                              }
 121                              Some(_) | None => {}
 122                          }
 123                      }
 124                      item_enum(ref enum_def, _) => {
 125                          if privacy_context == PublicContext {
 126                              for variant in enum_def.variants.iter() {
 127                                  self.reachable_symbols.insert(variant.node.id);
 128                              }
 129                          }
 130                      }
 131                      item_impl(ref generics, ref trait_ref, _, ref methods) => {
 132                          // XXX(pcwalton): We conservatively assume any methods
 133                          // on a trait implementation are reachable, when this
 134                          // is not the case. We could be more precise by only
 135                          // treating implementations of reachable or cross-
 136                          // crate traits as reachable.
 137  
 138                          let should_be_considered_public = |method: @method| {
 139                              (method.vis == public &&
 140                                      privacy_context == PublicContext) ||
 141                                      trait_ref.is_some()
 142                          };
 143  
 144                          // Mark all public methods as reachable.
 145                          for &method in methods.iter() {
 146                              if should_be_considered_public(method) {
 147                                  self.reachable_symbols.insert(method.id);
 148                              }
 149                          }
 150  
 151                          if generics_require_inlining(generics) {
 152                              // If the impl itself has generics, add all public
 153                              // symbols to the worklist.
 154                              for &method in methods.iter() {
 155                                  if should_be_considered_public(method) {
 156                                      self.worklist.push(method.id)
 157                                  }
 158                              }
 159                          } else {
 160                              // Otherwise, add only public methods that have
 161                              // generics to the worklist.
 162                              for method in methods.iter() {
 163                                  let generics = &method.generics;
 164                                  let attrs = &method.attrs;
 165                                  if generics_require_inlining(generics) ||
 166                                          attributes_specify_inlining(*attrs) ||
 167                                          should_be_considered_public(*method) {
 168                                      self.worklist.push(method.id)
 169                                  }
 170                              }
 171                          }
 172                      }
 173                      item_trait(_, _, ref trait_methods) => {
 174                          // Mark all provided methods as reachable.
 175                          if privacy_context == PublicContext {
 176                              for trait_method in trait_methods.iter() {
 177                                  match *trait_method {
 178                                      provided(method) => {
 179                                          self.reachable_symbols.insert(method.id);
 180                                          self.worklist.push(method.id)
 181                                      }
 182                                      required(_) => {}
 183                                  }
 184                              }
 185                          }
 186                      }
 187                      _ => {}
 188                  }
 189  
 190                  if item.vis == public && privacy_context == PublicContext {
 191                      visit::walk_item(self, item, PublicContext)
 192                  } else {
 193                      visit::walk_item(self, item, PrivateContext)
 194                  }
 195      }
 196  
 197  }

 199  struct MarkSymbolVisitor {
 200      worklist: @mut ~[NodeId],
 201      method_map: typeck::method_map,
 202      tcx: ty::ctxt,
 203      reachable_symbols: @mut HashSet<NodeId>,
 204  }
 205  
 206  impl Visitor<()> for MarkSymbolVisitor {
 207  
 208      fn visit_expr(&mut self, expr:@Expr, _:()) {
 209  
 210                  match expr.node {
 211                      ExprPath(_) => {
 212                          let def = match self.tcx.def_map.find(&expr.id) {
 213                              Some(&def) => def,
 214                              None => {
 215                                  self.tcx.sess.span_bug(expr.span,
 216                                                    "def ID not in def map?!")
 217                              }
 218                          };
 219  
 220                          let def_id = def_id_of_def(def);
 221                          if ReachableContext::
 222                                  def_id_represents_local_inlined_item(self.tcx,
 223                                                                       def_id) {
 224                              self.worklist.push(def_id.node)
 225                          }
 226                          self.reachable_symbols.insert(def_id.node);
 227                      }
 228                      ExprMethodCall(*) => {
 229                          match self.method_map.find(&expr.id) {
 230                              Some(&typeck::method_map_entry {
 231                                  origin: typeck::method_static(def_id),
 232                                  _
 233                              }) => {
 234                                  if ReachableContext::
 235                                      def_id_represents_local_inlined_item(
 236                                          self.tcx,
 237                                          def_id) {
 238                                      self.worklist.push(def_id.node)
 239                                  }
 240                                  self.reachable_symbols.insert(def_id.node);
 241                              }
 242                              Some(_) => {}
 243                              None => {
 244                                  self.tcx.sess.span_bug(expr.span,
 245                                                    "method call expression \
 246                                                     not in method map?!")
 247                              }
 248                          }
 249                      }
 250                      _ => {}
 251                  }
 252  
 253                  visit::walk_expr(self, expr, ())
 254      }
 255  }

 257  impl ReachableContext {
 258      // Creates a new reachability computation context.
 259      fn new(tcx: ty::ctxt, method_map: typeck::method_map)
 260             -> ReachableContext {
 261          ReachableContext {
 262              tcx: tcx,
 263              method_map: method_map,
 264              reachable_symbols: @mut HashSet::new(),
 265              worklist: @mut ~[],
 266          }
 267      }
 268  
 269      // Step 1: Mark all public symbols, and add all public symbols that might
 270      // be inlined to a worklist.
 271      fn mark_public_symbols(&self, crate: @Crate) {
 272          let reachable_symbols = self.reachable_symbols;
 273          let worklist = self.worklist;
 274  
 275          let mut visitor = ReachableVisitor {
 276              reachable_symbols: reachable_symbols,
 277              worklist: worklist,
 278          };
 279  
 280  
 281          visit::walk_crate(&mut visitor, crate, PublicContext);
 282      }
 283  
 284      // Returns true if the given def ID represents a local item that is
 285      // eligible for inlining and false otherwise.
 286      fn def_id_represents_local_inlined_item(tcx: ty::ctxt, def_id: DefId)
 287                                              -> bool {
 288          if def_id.crate != LOCAL_CRATE {
 289              return false
 290          }
 291  
 292          let node_id = def_id.node;
 293          match tcx.items.find(&node_id) {
 294              Some(&ast_map::node_item(item, _)) => {
 295                  match item.node {
 296                      item_fn(*) => item_might_be_inlined(item),
 297                      _ => false,
 298                  }
 299              }
 300              Some(&ast_map::node_trait_method(trait_method, _, _)) => {
 301                  match *trait_method {
 302                      required(_) => false,
 303                      provided(_) => true,
 304                  }
 305              }
 306              Some(&ast_map::node_method(method, impl_did, _)) => {
 307                  if generics_require_inlining(&method.generics) ||
 308                          attributes_specify_inlining(method.attrs) {
 309                      true
 310                  } else {
 311                      // Check the impl. If the generics on the self type of the
 312                      // impl require inlining, this method does too.
 313                      assert!(impl_did.crate == LOCAL_CRATE);
 314                      match tcx.items.find(&impl_did.node) {
 315                          Some(&ast_map::node_item(item, _)) => {
 316                              match item.node {
 317                                  item_impl(ref generics, _, _, _) => {
 318                                      generics_require_inlining(generics)
 319                                  }
 320                                  _ => false
 321                              }
 322                          }
 323                          Some(_) => {
 324                              tcx.sess.span_bug(method.span,
 325                                                "method is not inside an \
 326                                                 impl?!")
 327                          }
 328                          None => {
 329                              tcx.sess.span_bug(method.span,
 330                                                "the impl that this method is \
 331                                                 supposedly inside of doesn't \
 332                                                 exist in the AST map?!")
 333                          }
 334                      }
 335                  }
 336              }
 337              Some(_) => false,
 338              None => false   // This will happen for default methods.
 339          }
 340      }
 341  
 342      // Helper function to set up a visitor for `propagate()` below.
 343      fn init_visitor(&self) -> MarkSymbolVisitor {
 344          let (worklist, method_map) = (self.worklist, self.method_map);
 345          let (tcx, reachable_symbols) = (self.tcx, self.reachable_symbols);
 346  
 347          MarkSymbolVisitor {
 348              worklist: worklist,
 349              method_map: method_map,
 350              tcx: tcx,
 351              reachable_symbols: reachable_symbols,
 352          }
 353      }
 354  
 355      // Step 2: Mark all symbols that the symbols on the worklist touch.
 356      fn propagate(&self) {
 357          let mut visitor = self.init_visitor();
 358          let mut scanned = HashSet::new();
 359          while self.worklist.len() > 0 {
 360              let search_item = self.worklist.pop();
 361              if scanned.contains(&search_item) {
 362                  loop
 363              }
 364              scanned.insert(search_item);
 365              self.reachable_symbols.insert(search_item);
 366  
 367              // Find the AST block corresponding to the item and visit it,
 368              // marking all path expressions that resolve to something
 369              // interesting.
 370              match self.tcx.items.find(&search_item) {
 371                  Some(&ast_map::node_item(item, _)) => {
 372                      match item.node {
 373                          item_fn(_, _, _, _, ref search_block) => {
 374                              visit::walk_block(&mut visitor, search_block, ())
 375                          }
 376                          _ => {
 377                              self.tcx.sess.span_bug(item.span,
 378                                                     "found non-function item \
 379                                                      in worklist?!")
 380                          }
 381                      }
 382                  }
 383                  Some(&ast_map::node_trait_method(trait_method, _, _)) => {
 384                      match *trait_method {
 385                          required(ref ty_method) => {
 386                              self.tcx.sess.span_bug(ty_method.span,
 387                                                     "found required method in \
 388                                                      worklist?!")
 389                          }
 390                          provided(ref method) => {
 391                              visit::walk_block(&mut visitor, &method.body, ())
 392                          }
 393                      }
 394                  }
 395                  Some(&ast_map::node_method(ref method, _, _)) => {
 396                      visit::walk_block(&mut visitor, &method.body, ())
 397                  }
 398                  Some(_) => {
 399                      let ident_interner = token::get_ident_interner();
 400                      let desc = ast_map::node_id_to_str(self.tcx.items,
 401                                                         search_item,
 402                                                         ident_interner);
 403                      self.tcx.sess.bug(fmt!("found unexpected thingy in \
 404                                              worklist: %s",
 405                                              desc))
 406                  }
 407                  None => {
 408                      self.tcx.sess.bug(fmt!("found unmapped ID in worklist: \
 409                                              %d",
 410                                             search_item))
 411                  }
 412              }
 413          }
 414      }
 415  
 416      // Step 3: Mark all destructors as reachable.
 417      //
 418      // XXX(pcwalton): This is a conservative overapproximation, but fixing
 419      // this properly would result in the necessity of computing *type*
 420      // reachability, which might result in a compile time loss.
 421      fn mark_destructors_reachable(&self) {
 422          for (_, destructor_def_id) in self.tcx.destructor_for_type.iter() {
 423              if destructor_def_id.crate == LOCAL_CRATE {
 424                  self.reachable_symbols.insert(destructor_def_id.node);
 425              }
 426          }
 427      }
 428  }

 430  pub fn find_reachable(tcx: ty::ctxt,
 431                        method_map: typeck::method_map,
 432                        crate: @Crate)
 433                        -> @mut HashSet<NodeId> {
 434      // XXX(pcwalton): We only need to mark symbols that are exported. But this
 435      // is more complicated than just looking at whether the symbol is `pub`,
 436      // because it might be the target of a `pub use` somewhere. For now, I
 437      // think we are fine, because you can't `pub use` something that wasn't
 438      // exported due to the bug whereby `use` only looks through public
 439      // modules even if you're inside the module the `use` appears in. When
 440      // this bug is fixed, however, this code will need to be updated. Probably
 441      // the easiest way to fix this (although a conservative overapproximation)
 442      // is to have the name resolution pass mark all targets of a `pub use` as
 443      // "must be reachable".
 444  
 445      let reachable_context = ReachableContext::new(tcx, method_map);
 446  
 447      // Step 1: Mark all public symbols, and add all public symbols that might
 448      // be inlined to a worklist.

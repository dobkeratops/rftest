 161      // Creates a new reachability computation context.
 162      fn new(tcx: &'a ty::ctxt, method_map: typeck::MethodMap) -> ReachableContext<'a> {
 163          ReachableContext {

 165              method_map: method_map,
 166              reachable_symbols: NodeSet::new(),
 167              worklist: Vec::new(),
 168          }
 169      }
 170  
 171      // Returns true if the given def ID represents a local item that is
 172      // eligible for inlining and false otherwise.
 173      fn def_id_represents_local_inlined_item(&self, def_id: ast::DefId) -> bool {
 174          if def_id.krate != ast::LOCAL_CRATE {
 175              return false
 176          }
 177  
 178          let node_id = def_id.node;
 179          match self.tcx.map.find(node_id) {
 180              Some(ast_map::NodeItem(item)) => {
 181                  match item.node {
 182                      ast::ItemFn(..) => item_might_be_inlined(item),
 183                      _ => false,
 184                  }
 185              }
 186              Some(ast_map::NodeTraitMethod(trait_method)) => {
 187                  match *trait_method {
 188                      ast::Required(_) => false,
 189                      ast::Provided(_) => true,
 190                  }
 191              }
 192              Some(ast_map::NodeMethod(method)) => {
 193                  if generics_require_inlining(&method.generics) ||
 194                          attributes_specify_inlining(method.attrs.as_slice()) {
 195                      true
 196                  } else {
 197                      let impl_did = self.tcx.map.get_parent_did(node_id);
 198                      // Check the impl. If the generics on the self type of the
 199                      // impl require inlining, this method does too.
 200                      assert!(impl_did.krate == ast::LOCAL_CRATE);
 201                      match self.tcx.map.expect_item(impl_did.node).node {
 202                          ast::ItemImpl(ref generics, _, _, _) => {
 203                              generics_require_inlining(generics)
 204                          }
 205                          _ => false
 206                      }
 207                  }
 208              }
 209              Some(_) => false,
 210              None => false   // This will happen for default methods.
 211          }
 212      }

 214      // Step 2: Mark all symbols that the symbols on the worklist touch.
 215      fn propagate(&mut self) {
 216          let mut scanned = HashSet::new();
 217          loop {
 218              if self.worklist.len() == 0 {
 219                  break
 220              }
 221              let search_item = self.worklist.pop().unwrap();
 222              if scanned.contains(&search_item) {
 223                  continue
 224              }
 225  
 226              scanned.insert(search_item);
 227              match self.tcx.map.find(search_item) {
 228                  Some(ref item) => self.propagate_node(item, search_item),
 229                  None if search_item == ast::CRATE_NODE_ID => {}
 230                  None => {
 231                      self.tcx.sess.bug(format!("found unmapped ID in worklist: \
 232                                                 {}",
 233                                                search_item))
 234                  }
 235              }
 236          }
 237      }

 239      fn propagate_node(&mut self, node: &ast_map::Node,
 240                        search_item: ast::NodeId) {
 241          if !self.tcx.sess.building_library.get() {
 242              // If we are building an executable, then there's no need to flag
 243              // anything as external except for `extern fn` types. These
 244              // functions may still participate in some form of native interface,
 245              // but all other rust-only interfaces can be private (they will not

 248                  ast_map::NodeItem(item) => {

 250                          ast::ItemFn(_, ast::ExternFn, _, _, _) => {
 251                              self.reachable_symbols.insert(search_item);
 252                          }
 253                          _ => {}
 254                      }
 255                  }
 256                  _ => {}
 257              }

 262              // all reachable nodes from this one.
 263              self.reachable_symbols.insert(search_item);

 265  
 266          match *node {
 267              ast_map::NodeItem(item) => {
 268                  match item.node {
 269                      ast::ItemFn(_, _, _, _, search_block) => {
 270                          if item_might_be_inlined(item) {

 274

 277                      ast::ItemStatic(..) => {
 278                          if attr::contains_name(item.attrs.as_slice(),
 279                                                 "address_insignificant") {
 280                              self.reachable_symbols.remove(&search_item);
 281                          }
 282                      }
 283  
 284                      // These are normal, nothing reachable about these
 285                      // inherently and their children are already in the
 286                      // worklist, as determined by the privacy pass
 287                      ast::ItemTy(..) |
 288                      ast::ItemMod(..) | ast::ItemForeignMod(..) |
 289                      ast::ItemImpl(..) | ast::ItemTrait(..) |
 290                      ast::ItemStruct(..) | ast::ItemEnum(..) => {}
 291  
 292                      _ => {
 293                          self.tcx.sess.span_bug(item.span,
 294                                                 "found non-function item \
 295                                                  in worklist?!")
 296                      }
 297                  }
 298              }
 299              ast_map::NodeTraitMethod(trait_method) => {
 300                  match *trait_method {
 301                      ast::Required(..) => {
 302                          // Keep going, nothing to get exported
 303                      }
 304                      ast::Provided(ref method) => {
 305                          visit::walk_block(self, method.body, ())
 306                      }
 307                  }
 308              }
 309              ast_map::NodeMethod(method) => {
 310                  let did = self.tcx.map.get_parent_did(search_item);
 311                  if method_might_be_inlined(self.tcx, method, did) {
 312                      visit::walk_block(self, method.body, ())
 313                  }
 314              }
 315              // Nothing to recurse on for these

 317              ast_map::NodeVariant(_) |
 318              ast_map::NodeStructCtor(_) => {}
 319              _ => {

 324          }
 325      }

 328      //

 341  pub fn find_reachable(tcx: &ty::ctxt,
 342                        method_map: typeck::MethodMap,
 343                        exported_items: &privacy::ExportedItems)

 346  
 347      // Step 1: Seed the worklist with all nodes which were found to be public as
 348      //         a result of the privacy pass along with all local lang items. If
 349      //         other crates link to us, they're going to expect to be able to
 350      //         use the lang items, so we need to be sure to mark them as
 351      //         exported.
 352      for &id in exported_items.iter() {
 353          reachable_context.worklist.push(id);
 354      }
 355      for (_, item) in tcx.lang_items.items() {

 358                  reachable_context.worklist.push(did.node);
 359              }
 360              _ => {}
 361          }
 362      }
 363  
 364      // Step 2: Mark all symbols that the symbols on the worklist touch.
 365      reachable_context.propagate();
 366  
 367      // Step 3: Mark all destructors as reachable.
 368      reachable_context.mark_destructors_reachable();
 369  
 370      // Return the set of reachable symbols.
 371      reachable_context.reachable_symbols
 372  }

 108  pub fn check_expr(v: &mut CheckCrateVisitor,
 109                    sess: Session,
 110                    def_map: resolve::DefMap,
 111                    method_map: typeck::method_map,
 112                    tcx: ty::ctxt,
 113                    e: @Expr,
 114                    is_const: bool) {
 115      if is_const {
 116          match e.node {
 117            ExprUnary(_, UnDeref, _) => { }
 118            ExprUnary(_, UnBox(_), _) | ExprUnary(_, UnUniq, _) => {
 119              sess.span_err(e.span,
 120                            "disallowed operator in constant expression");
 121              return;
 122            }
 123            ExprLit(@codemap::Spanned {node: lit_str(_), _}) => { }
 124            ExprBinary(*) | ExprUnary(*) => {
 125              if method_map.contains_key(&e.id) {
 126                  sess.span_err(e.span, "user-defined operators are not \
 127                                         allowed in constant expressions");
 128              }
 129            }
 130            ExprLit(_) => (),
 131            ExprCast(_, _) => {
 132              let ety = ty::expr_ty(tcx, e);
 133              if !ty::type_is_numeric(ety) && !ty::type_is_unsafe_ptr(ety) {
 134                  sess.span_err(e.span, ~"can not cast to `" +
 135                                ppaux::ty_to_str(tcx, ety) +
 136                                "` in a constant expression");
 137              }
 138            }
 139            ExprPath(ref pth) => {
 140              // NB: In the future you might wish to relax this slightly
 141              // to handle on-demand instantiation of functions via
 142              // foo::<bar> in a const. Currently that is only done on
 143              // a path in trans::callee that only works in block contexts.
 144              if !pth.segments.iter().all(|segment| segment.types.is_empty()) {
 145                  sess.span_err(
 146                      e.span, "paths in constants may only refer to \
 147                               items without type parameters");
 148              }
 149              match def_map.find(&e.id) {
 150                Some(&DefStatic(*)) |
 151                Some(&DefFn(_, _)) |
 152                Some(&DefVariant(_, _, _)) |
 153                Some(&DefStruct(_)) => { }
 154  
 155                Some(&def) => {
 156                  debug!("(checking const) found bad def: %?", def);
 157                  sess.span_err(
 158                      e.span,
 159                      "paths in constants may only refer to \
 160                       constants or functions");
 161                }
 162                None => {
 163                  sess.span_bug(e.span, "unbound path in const?!");
 164                }
 165              }
 166            }
 167            ExprCall(callee, _, NoSugar) => {
 168              match def_map.find(&callee.id) {
 169                  Some(&DefStruct(*)) => {}    // OK.
 170                  Some(&DefVariant(*)) => {}    // OK.
 171                  _ => {
 172                      sess.span_err(
 173                          e.span,
 174                          "function calls in constants are limited to \
 175                           struct and enum constructors");
 176                  }
 177              }
 178            }
 179            ExprParen(e) => { check_expr(v, sess, def_map, method_map,
 180                                          tcx, e, is_const); }
 181            ExprVstore(_, ExprVstoreSlice) |
 182            ExprVec(_, MutImmutable) |
 183            ExprAddrOf(MutImmutable, _) |
 184            ExprField(*) |
 185            ExprIndex(*) |
 186            ExprTup(*) |
 187            ExprRepeat(*) |
 188            ExprStruct(*) => { }
 189            ExprAddrOf(*) => {

 191                      e.span,

 199            }
 200          }
 201      }

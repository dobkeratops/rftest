  58  fn stack_check_item(v: &mut StackCheckVisitor,
  59                      item: @ast::item,
  60                      in_cx: Context) {
  61      match item.node {
  62          ast::item_fn(_, ast::extern_fn, _, _, _) => {
  63              // an extern fn is already being called from C code...
  64              let new_cx = Context {safe_stack: true};
  65              visit::walk_item(v, item, new_cx);
  66          }
  67          ast::item_fn(*) => {
  68              let safe_stack = fixed_stack_segment(item.attrs);
  69              let new_cx = Context {safe_stack: safe_stack};
  70              visit::walk_item(v, item, new_cx);
  71          }
  72          ast::item_impl(_, _, _, ref methods) => {
  73              // visit_method() would make this nicer
  74              for &method in methods.iter() {
  75                  let safe_stack = fixed_stack_segment(method.attrs);
  76                  let new_cx = Context {safe_stack: safe_stack};
  77                  visit::walk_method_helper(v, method, new_cx);
  78              }
  79          }
  80          ast::item_trait(_, _, ref methods) => {
  81              for method in methods.iter() {
  82                  match *method {
  83                      ast::provided(@ref method) => {
  84                          let safe_stack = fixed_stack_segment(method.attrs);
  85                          let new_cx = Context {safe_stack: safe_stack};
  86                          visit::walk_method_helper(v, method, new_cx);
  87                      }
  88                      ast::required(*) => ()
  89                  }
  90              }
  91          }
  92          _ => {
  93              visit::walk_item(v, item, in_cx);

 155  fn call_to_extern_fn(v: &mut StackCheckVisitor, callee: @ast::Expr) {
 156      // Permit direct calls to extern fns that are annotated with

 158      match callee.node {

 162                      match v.tcx.items.find(&id.node) {
 163                          Some(&ast_map::node_foreign_item(item, _, _, _)) => {
 164                              if attr::contains_name(item.attrs, "rust_stack") {
 165                                  return;
 166                              }
 167                          }
 168                          _ => {}
 169                      }
 170                  }
 171                  _ => {}
 172              }
 173          }
 174          _ => {}

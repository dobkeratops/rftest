  11  // Detecting language items.
  12  //
  13  // Language items are items that represent concepts intrinsic to the language
  14  // itself. Examples are:
  15  //
  16  // * Traits that specify "kinds"; e.g. "Freeze", "Send".
  17  //
  18  // * Traits that represent operators; e.g. "Add", "Sub", "Index".
  19  //
  20  // * Functions called by the compiler itself.
  21

 296      items: LanguageItems,
 297  
 298      crate: &'self Crate,
 299      session: Session,
 300  
 301      item_refs: HashMap<@str, uint>,
 302  }

 304  struct LanguageItemVisitor<'self> {
 305      this: *mut LanguageItemCollector<'self>,
 306  }
 307  
 308  impl<'self> Visitor<()> for LanguageItemVisitor<'self> {
 309  
 310      fn visit_item(&mut self, item:@item, _:()) {
 311  
 312                  for attribute in item.attrs.iter() {
 313                      unsafe {
 314                          (*self.this).match_and_collect_meta_item(
 315                              local_def(item.id),
 316                              attribute.node.value
 317                          );
 318                      }
 319                  }
 320  
 321          visit::walk_item(self, item, ());
 322      }
 323  }

 384      pub fn match_and_collect_meta_item(&mut self,
 385                                         item_def_id: DefId,
 386                                         meta_item: &MetaItem) {
 387          match meta_item.name_str_pair() {
 388              Some((key, value)) => {
 389                  self.match_and_collect_item(item_def_id, key, value);
 390              }
 391              None => {} // skip
 392          }
 393      }

 395      pub fn collect_item(&mut self, item_index: uint, item_def_id: DefId) {
 396          // Check for duplicates.
 397          match self.items.items[item_index] {
 398              Some(original_def_id) if original_def_id != item_def_id => {
 399                  self.session.err(fmt!("duplicate entry for `%s`",
 400                                        LanguageItems::item_name(item_index)));
 401              }
 402              Some(_) | None => {
 403                  // OK.
 404              }
 405          }
 406  
 407          // Matched.
 408          self.items.items[item_index] = Some(item_def_id);
 409      }

 411      pub fn match_and_collect_item(&mut self,
 412                                    item_def_id: DefId,
 413                                    key: &str,
 414                                    value: @str) {
 415          if "lang" != key {
 416              return;    // Didn't match.
 417          }
 418  
 419          let item_index = self.item_refs.find(&value).map_move(|x| *x);
 420          // prevent borrow checker from considering   ^~~~~~~~~~~
 421          // self to be borrowed (annoying)
 422  
 423          match item_index {
 424              Some(item_index) => {
 425                  self.collect_item(item_index, item_def_id);
 426              }
 427              None => {
 428                  // Didn't match.
 429                  return;
 430              }
 431          }
 432      }

 438      }

 440      pub fn collect_external_language_items(&mut self) {
 441          let crate_store = self.session.cstore;
 442          do iter_crate_data(crate_store) |crate_number, _crate_metadata| {
 443              do each_lang_item(crate_store, crate_number)
 444                      |node_id, item_index| {
 445                  let def_id = DefId { crate: crate_number, node: node_id };
 446                  self.collect_item(item_index, def_id);
 447                  true
 448              };
 449          }
 450      }

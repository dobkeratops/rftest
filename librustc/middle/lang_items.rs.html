  11  // Detecting language items.
  12  //
  13  // Language items are items that represent concepts intrinsic to the language
  14  // itself. Examples are:
  15  //
  16  // * Traits that specify "kinds"; e.g. "Freeze", "Send".
  17  //
  18  // * Traits that represent operators; e.g. "Add", "Sub", "Index".
  19  //
  20  // * Functions called by the compiler itself.
  21

 361      pub fn match_and_collect_meta_item(&mut self,
 362                                         item_def_id: def_id,
 363                                         meta_item: &MetaItem) {
 364          match meta_item.name_str_pair() {
 365              Some((key, value)) => {
 366                  self.match_and_collect_item(item_def_id, key, value);
 367              }
 368              None => {} // skip
 369          }
 370      }

 372      pub fn collect_item(&mut self, item_index: uint, item_def_id: def_id) {
 373          // Check for duplicates.
 374          match self.items.items[item_index] {
 375              Some(original_def_id) if original_def_id != item_def_id => {
 376                  self.session.err(fmt!("duplicate entry for `%s`",
 377                                        LanguageItems::item_name(item_index)));
 378              }
 379              Some(_) | None => {
 380                  // OK.
 381              }
 382          }
 383  
 384          // Matched.
 385          self.items.items[item_index] = Some(item_def_id);
 386      }

 388      pub fn match_and_collect_item(&mut self,
 389                                    item_def_id: def_id,
 390                                    key: &str,
 391                                    value: @str) {
 392          if "lang" != key {
 393              return;    // Didn't match.
 394          }
 395  
 396          let item_index = self.item_refs.find(&value).map_move(|x| *x);
 397          // prevent borrow checker from considering   ^~~~~~~~~~~
 398          // self to be borrowed (annoying)
 399  
 400          match item_index {
 401              Some(item_index) => {
 402                  self.collect_item(item_index, item_def_id);
 403              }
 404              None => {
 405                  // Didn't match.
 406                  return;
 407              }
 408          }
 409      }

 411      pub fn collect_local_language_items(&mut self) {
 412          let this: *mut LanguageItemCollector = &mut *self;
 413          visit_crate(self.crate, ((), mk_simple_visitor(@SimpleVisitor {
 414              visit_item: |item| {
 415                  for attribute in item.attrs.iter() {
 416                      unsafe {
 417                          (*this).match_and_collect_meta_item(
 418                              local_def(item.id),
 419                              attribute.node.value
 420                          );
 421                      }
 422                  }
 423              },
 424              .. *default_simple_visitor()
 425          })));
 426      }

 428      pub fn collect_external_language_items(&mut self) {
 429          let crate_store = self.session.cstore;
 430          do iter_crate_data(crate_store) |crate_number, _crate_metadata| {
 431              do each_lang_item(crate_store, crate_number)
 432                      |node_id, item_index| {
 433                  let def_id = def_id { crate: crate_number, node: node_id };
 434                  self.collect_item(item_index, def_id);
 435                  true
 436              };
 437          }
 438      }

 440      pub fn collect(&mut self) {
 441          self.collect_local_language_items();
 442          self.collect_external_language_items();
 443      }
 444  }

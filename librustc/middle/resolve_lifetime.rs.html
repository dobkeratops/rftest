 161      fn count_early_params(&self) -> uint {
 162          /*!
 163           * Counts the number of early-bound parameters that are in
 164           * scope.  Used when checking methods: the early-bound
 165           * lifetime parameters declared on the method are assigned
 166           * indices that come after the indices from the type.  Given
 167           * something like `impl<'a> Foo { ... fn bar<'b>(...) }`
 168           * then `'a` gets index 0 and `'b` gets index 1.
 169           */

 171          match *self {
 172              RootScope => 0,
 173              EarlyScope(base, lifetimes, _) => base + lifetimes.len(),
 174              LateScope(_, _, s) => s.count_early_params(),
 175              BlockScope(_, _) => 0,
 176          }
 177      }
 178  }

 180  impl<'a> LifetimeContext<'a> {
 181      /// Visits self by adding a scope and handling recursive walk over the contents with `walk`.
 182      fn visit_fn_decl(&mut self,
 183                       n: ast::NodeId,
 184                       generics: &ast::Generics,
 185                       scope: Scope,
 186                       walk: |&mut LifetimeContext, Scope|) {
 187          /*!
 188           * Handles visiting fns and methods. These are a bit
 189           * complicated because we must distinguish early- vs late-bound
 190           * lifetime parameters. We do this by checking which lifetimes
 191           * appear within type bounds; those are early bound lifetimes,
 192           * and the rest are late bound.
 193           *
 194           * For example:
 195           *
 196           *    fn foo<'a,'b,'c,T:Trait<'b>>(...)
 197           *
 198           * Here `'a` and `'c` are late bound but `'b` is early
 199           * bound. Note that early- and late-bound lifetimes may be
 200           * interspersed together.
 201           *
 202           * If early bound lifetimes are present, we separate them into
 203           * their own list (and likewise for late bound). They will be
 204           * numbered sequentially, starting from the lowest index that
 205           * is already in scope (for a fn item, that will be 0, but for
 206           * a method it might not be). Late bound lifetimes are
 207           * resolved by name and associated with a binder id (`n`), so
 208           * the ordering is not important there.
 209           */
 210  
 211          self.check_lifetime_names(&generics.lifetimes);
 212  
 213          let early_count = scope.count_early_params();
 214          let referenced_idents = free_lifetimes(&generics.ty_params);
 215          debug!("pushing fn scope id={} due to fn item/method\
 216                 referenced_idents={:?} \
 217                 early_count={}",
 218                 n,
 219                 referenced_idents.map(lifetime_show),
 220                 early_count);
 221          if referenced_idents.is_empty() {
 222              let scope1 = LateScope(n, &generics.lifetimes, scope);
 223              walk(self, &scope1)
 224          } else {
 225              let (early, late) = generics.lifetimes.clone().partition(
 226                  |l| referenced_idents.iter().any(|&i| i == l.name));
 227  
 228              let scope1 = EarlyScope(early_count, &early, scope);
 229              let scope2 = LateScope(n, &late, &scope1);
 230  
 231              walk(self, &scope2);
 232          }
 233          debug!("popping fn scope id={} due to fn item/method", n);
 234      }

 236      fn resolve_lifetime_ref(&mut self,
 237                              lifetime_ref: &ast::Lifetime,
 238                              scope: Scope) {
 239          // Walk up the scope chain, tracking the number of fn scopes
 240          // that we pass through, until we find a lifetime with the
 241          // given name or we run out of scopes. If we encounter a code
 242          // block, then the lifetime is not bound but free, so switch
 243          // over to `resolve_free_lifetime_ref()` to complete the
 244          // search.
 245          let mut depth = 0;
 246          let mut scope = scope;
 247          loop {
 248              match *scope {
 249                  BlockScope(id, s) => {
 250                      return self.resolve_free_lifetime_ref(id, lifetime_ref, s);
 251                  }
 252  
 253                  RootScope => {
 254                      break;
 255                  }
 256  
 257                  EarlyScope(base, lifetimes, s) => {
 258                      match search_lifetimes(lifetimes, lifetime_ref) {
 259                          Some((offset, decl_id)) => {
 260                              let index = base + offset;
 261                              let def = ast::DefEarlyBoundRegion(index, decl_id);
 262                              self.insert_lifetime(lifetime_ref, def);
 263                              return;
 264                          }
 265                          None => {
 266                              depth += 1;
 267                              scope = s;
 268                          }
 269                      }
 270                  }
 271  
 272                  LateScope(binder_id, lifetimes, s) => {
 273                      match search_lifetimes(lifetimes, lifetime_ref) {
 274                          Some((_index, decl_id)) => {
 275                              let def = ast::DefLateBoundRegion(binder_id, depth, decl_id);
 276                              self.insert_lifetime(lifetime_ref, def);
 277                              return;
 278                          }
 279  
 280                          None => {
 281                              depth += 1;
 282                              scope = s;
 283                          }
 284                      }
 285                  }
 286              }
 287          }
 288  
 289          self.unresolved_lifetime_ref(lifetime_ref);
 290      }

 292      fn resolve_free_lifetime_ref(&mut self,
 293                                   scope_id: ast::NodeId,
 294                                   lifetime_ref: &ast::Lifetime,
 295                                   scope: Scope) {

 297          // until we encounter a scope that contains the named lifetime
 298          // or we run out of scopes.
 299          let mut scope_id = scope_id;

 303              match *scope {
 304                  BlockScope(id, s) => {
 305                      scope_id = id;
 306                      scope = s;
 307                  }
 308  
 309                  RootScope => {
 310                      break;

  65  fn check_expr(cx: &mut MatchCheckCtxt, ex: &Expr) {
  66      visit::walk_expr(cx, ex, ());
  67      match ex.node {
  68        ExprMatch(scrut, ref arms) => {
  69          // First, check legality of move bindings.
  70          for arm in arms.iter() {
  71              check_legality_of_move_bindings(cx,
  72                                              arm.guard.is_some(),
  73                                              arm.pats.as_slice());
  74          }
  75  
  76          check_arms(cx, arms.as_slice());
  77          /* Check for exhaustiveness */
  78           // Check for empty enum, because is_useful only works on inhabited
  79           // types.
  80         let pat_ty = node_id_to_type(cx.tcx, scrut.id);
  81         if (*arms).is_empty() {
  82             if !type_is_empty(cx.tcx, pat_ty) {
  83                 // We know the type is inhabited, so this must be wrong
  84                 cx.tcx.sess.span_err(ex.span, format!("non-exhaustive patterns: \
  85                              type {} is non-empty",
  86                              ty_to_str(cx.tcx, pat_ty)));
  87             }
  88             // If the type *is* empty, it's vacuously exhaustive
  89             return;
  90         }
  91         match ty::get(pat_ty).sty {
  92            ty_enum(did, _) => {
  93                if (*enum_variants(cx.tcx, did)).is_empty() &&
  94                      (*arms).is_empty() {
  95  
  96                 return;
  97              }
  98            }
  99            _ => { /* We assume only enum types can be uninhabited */ }
 100         }
 101  
 102         let pats: Vec<@Pat> = arms.iter()
 103                                 .filter_map(unguarded_pat)
 104                                 .flat_map(|pats| pats.move_iter())
 105                                 .collect();
 106         if pats.is_empty() {
 107             cx.tcx.sess.span_err(ex.span, "non-exhaustive patterns");
 108         } else {
 109             check_exhaustive(cx, ex.span, pats);
 110         }
 111       }
 112       _ => ()
 113      }
 114  }

 116  // Check for unreachable patterns
 117  fn check_arms(cx: &MatchCheckCtxt, arms: &[Arm]) {
 118      let mut seen = Vec::new();
 119      for arm in arms.iter() {
 120          for pat in arm.pats.iter() {
 121  
 122              // Check that we do not match against a static NaN (#6804)
 123              let pat_matches_nan: |&Pat| -> bool = |p| {
 124                  let opt_def = cx.tcx.def_map.borrow().find_copy(&p.id);
 125                  match opt_def {
 126                      Some(DefStatic(did, false)) => {
 127                          let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 128                          match eval_const_expr(cx.tcx, const_expr) {
 129                              const_float(f) if f.is_nan() => true,
 130                              _ => false
 131                          }
 132                      }
 133                      _ => false
 134                  }
 135              };
 136  
 137              walk_pat(*pat, |p| {
 138                  if pat_matches_nan(p) {
 139                      cx.tcx.sess.span_warn(p.span, "unmatchable NaN in pattern, \
 140                                                     use the is_nan method in a guard instead");
 141                  }
 142                  true
 143              });
 144  
 145              let v = vec!(*pat);
 146              match is_useful(cx, &seen, v.as_slice()) {
 147                not_useful => {
 148                  cx.tcx.sess.span_err(pat.span, "unreachable pattern");
 149                }
 150                _ => ()
 151              }
 152              if arm.guard.is_none() { seen.push(v); }
 153          }
 154      }
 155  }

 158      match p.node {
 159        PatIdent(_, _, Some(s)) => { raw_pat(s) }
 160        _ => { p }
 161      }
 162  }

 164  fn check_exhaustive(cx: &MatchCheckCtxt, sp: Span, pats: Vec<@Pat> ) {
 165      assert!((!pats.is_empty()));
 166      let ext = match is_useful(cx, &pats.map(|p| vec!(*p)), [wild()]) {
 167          not_useful => {
 168              // This is good, wildcard pattern isn't reachable
 169              return;
 170          }
 171          useful_ => None,
 172          useful(ty, ref ctor) => {
 173              match ty::get(ty).sty {
 174                  ty::ty_bool => {
 175                      match *ctor {
 176                          val(const_bool(true)) => Some(~"true"),
 177                          val(const_bool(false)) => Some(~"false"),
 178                          _ => None
 179                      }
 180                  }
 181                  ty::ty_enum(id, _) => {
 182                      let vid = match *ctor {
 183                          variant(id) => id,
 184                          _ => fail!("check_exhaustive: non-variant ctor"),
 185                      };
 186                      let variants = ty::enum_variants(cx.tcx, id);
 187  
 188                      match variants.iter().find(|v| v.id == vid) {
 189                          Some(v) => Some(token::get_ident(v.name).get().to_str()),
 190                          None => {
 191                              fail!("check_exhaustive: bad variant in ctor")
 192                          }
 193                      }
 194                  }
 195                  ty::ty_unboxed_vec(..) | ty::ty_vec(..) => {
 196                      match *ctor {
 197                          vec(n) => Some(format!("vectors of length {}", n)),
 198                          _ => None
 199                      }
 200                  }
 201                  _ => None
 202              }
 203          }
 204      };
 205      let msg = ~"non-exhaustive patterns" + match ext {
 206          Some(ref s) => format!(": {} not covered",  *s),
 207          None => ~""
 208      };
 209      cx.tcx.sess.span_err(sp, msg);
 210  }

 214  #[deriving(Clone)]
 215  enum useful {
 216      useful(ty::t, ctor),
 217      useful_,
 218      not_useful,
 219  }
 220  
 221  #[deriving(Clone, Eq)]
 222  enum ctor {
 223      single,
 224      variant(DefId),
 225      val(const_val),
 226      range(const_val, const_val),
 227      vec(uint)
 228  }

 230  // Algorithm from http://moscova.inria.fr/~maranget/papers/warn/index.html
 231  //
 232  // Whether a vector `v` of patterns is 'useful' in relation to a set of such
 233  // vectors `m` is defined as there being a set of inputs that will match `v`
 234  // but not any of the sets in `m`.
 235  //
 236  // This is used both for reachability checking (if a pattern isn't useful in
 237  // relation to preceding patterns, it is not reachable) and exhaustiveness
 238  // checking (if a wildcard pattern is useful in relation to a matrix, the
 239  // matrix isn't exhaustive).
 240  
 241  // Note: is_useful doesn't work on empty types, as the paper notes.
 242  // So it assumes that v is non-empty.
 243  fn is_useful(cx: &MatchCheckCtxt, m: &matrix, v: &[@Pat]) -> useful {
 244      if m.len() == 0u {
 245          return useful_;
 246      }
 247      if m.get(0).len() == 0u {
 248          return not_useful
 249      }
 250      let real_pat = match m.iter().find(|r| r.get(0).id != 0) {
 251        Some(r) => *r.get(0), None => v[0]
 252      };
 253      let left_ty = if real_pat.id == 0 { ty::mk_nil() }
 254                    else { ty::node_id_to_type(cx.tcx, real_pat.id) };
 255  
 256      match pat_ctor_id(cx, v[0]) {
 257        None => {
 258          match missing_ctor(cx, m, left_ty) {
 259            None => {
 260              match ty::get(left_ty).sty {
 261                ty::ty_bool => {
 262                  match is_useful_specialized(cx, m, v,
 263                                              val(const_bool(true)),
 264                                              0u, left_ty){
 265                    not_useful => {
 266                      is_useful_specialized(cx, m, v,
 267                                            val(const_bool(false)),
 268                                            0u, left_ty)
 269                    }
 270                    ref u => (*u).clone(),
 271                  }
 272                }
 273                ty::ty_enum(eid, _) => {
 274                  for va in (*ty::enum_variants(cx.tcx, eid)).iter() {
 275                      match is_useful_specialized(cx, m, v, variant(va.id),
 276                                                  va.args.len(), left_ty) {
 277                        not_useful => (),
 278                        ref u => return (*u).clone(),
 279                      }
 280                  }
 281                  not_useful
 282                }
 283                ty::ty_vec(_, ty::vstore_fixed(n)) => {
 284                  is_useful_specialized(cx, m, v, vec(n), n, left_ty)
 285                }
 286                ty::ty_unboxed_vec(..) | ty::ty_vec(..) => {
 287                  let max_len = m.iter().rev().fold(0, |max_len, r| {
 288                    match r.get(0).node {
 289                      PatVec(ref before, _, ref after) => {
 290                        cmp::max(before.len() + after.len(), max_len)
 291                      }
 292                      _ => max_len
 293                    }
 294                  });
 295                  for n in iter::range(0u, max_len + 1) {
 296                    match is_useful_specialized(cx, m, v, vec(n), n, left_ty) {
 297                      not_useful => (),
 298                      ref u => return (*u).clone(),
 299                    }
 300                  }
 301                  not_useful
 302                }
 303                _ => {
 304                  let arity = ctor_arity(cx, &single, left_ty);
 305                  is_useful_specialized(cx, m, v, single, arity, left_ty)
 306                }
 307              }
 308            }
 309            Some(ref ctor) => {
 310              match is_useful(cx,
 311                              &m.iter().filter_map(|r| {
 312                                  default(cx, r.as_slice())
 313                              }).collect::<matrix>(),
 314                              v.tail()) {
 315                useful_ => useful(left_ty, (*ctor).clone()),
 316                ref u => (*u).clone(),
 317              }
 318            }
 319          }
 320        }
 321        Some(ref v0_ctor) => {
 322          let arity = ctor_arity(cx, v0_ctor, left_ty);
 323          is_useful_specialized(cx, m, v, (*v0_ctor).clone(), arity, left_ty)
 324        }
 325      }
 326  }

 328  fn is_useful_specialized(cx: &MatchCheckCtxt,
 329                               m: &matrix,
 330                               v: &[@Pat],
 331                               ctor: ctor,
 332                               arity: uint,
 333                               lty: ty::t)
 334                               -> useful {
 335      let ms = m.iter().filter_map(|r| {
 336          specialize(cx, r.as_slice(), &ctor, arity, lty)
 337      }).collect::<matrix>();
 338      let could_be_useful = is_useful(
 339          cx, &ms, specialize(cx, v, &ctor, arity, lty).unwrap().as_slice());
 340      match could_be_useful {
 341        useful_ => useful(lty, ctor),
 342        ref u => (*u).clone(),
 343      }
 344  }

 346  fn pat_ctor_id(cx: &MatchCheckCtxt, p: @Pat) -> Option<ctor> {
 347      let pat = raw_pat(p);
 348      match pat.node {
 349        PatWild | PatWildMulti => { None }
 350        PatIdent(_, _, _) | PatEnum(_, _) => {
 351          let opt_def = cx.tcx.def_map.borrow().find_copy(&pat.id);
 352          match opt_def {
 353            Some(DefVariant(_, id, _)) => Some(variant(id)),
 354            Some(DefStatic(did, false)) => {
 355              let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 356              Some(val(eval_const_expr(cx.tcx, const_expr)))
 357            }
 358            _ => None
 359          }
 360        }
 361        PatLit(expr) => { Some(val(eval_const_expr(cx.tcx, expr))) }
 362        PatRange(lo, hi) => {
 363          Some(range(eval_const_expr(cx.tcx, lo), eval_const_expr(cx.tcx, hi)))
 364        }
 365        PatStruct(..) => {
 366          match cx.tcx.def_map.borrow().find(&pat.id) {
 367            Some(&DefVariant(_, id, _)) => Some(variant(id)),
 368            _ => Some(single)
 369          }
 370        }
 371        PatUniq(_) | PatTup(_) | PatRegion(..) => {
 372          Some(single)
 373        }
 374        PatVec(ref before, slice, ref after) => {
 375          match slice {
 376            Some(_) => None,
 377            None => Some(vec(before.len() + after.len()))
 378          }
 379        }
 380      }
 381  }

 383  fn is_wild(cx: &MatchCheckCtxt, p: @Pat) -> bool {
 384      let pat = raw_pat(p);
 385      match pat.node {
 386        PatWild | PatWildMulti => { true }
 387        PatIdent(_, _, _) => {
 388          match cx.tcx.def_map.borrow().find(&pat.id) {
 389            Some(&DefVariant(_, _, _)) | Some(&DefStatic(..)) => { false }
 390            _ => { true }
 391          }
 392        }
 393        _ => { false }
 394      }
 395  }

 397  fn missing_ctor(cx: &MatchCheckCtxt,
 398                      m: &matrix,
 399                      left_ty: ty::t)
 400                   -> Option<ctor> {
 401      match ty::get(left_ty).sty {
 402        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(..) | ty::ty_tup(_) |
 403        ty::ty_struct(..) => {
 404          for r in m.iter() {
 405              if !is_wild(cx, *r.get(0)) { return None; }
 406          }
 407          return Some(single);
 408        }
 409        ty::ty_enum(eid, _) => {
 410          let mut found = Vec::new();
 411          for r in m.iter() {
 412              let r = pat_ctor_id(cx, *r.get(0));
 413              for id in r.iter() {
 414                  if !found.contains(id) {
 415                      found.push((*id).clone());
 416                  }
 417              }
 418          }
 419          let variants = ty::enum_variants(cx.tcx, eid);
 420          if found.len() != (*variants).len() {
 421              for v in (*variants).iter() {
 422                  if !found.iter().any(|x| x == &(variant(v.id))) {
 423                      return Some(variant(v.id));
 424                  }
 425              }
 426              fail!();
 427          } else { None }
 428        }
 429        ty::ty_nil => None,
 430        ty::ty_bool => {
 431          let mut true_found = false;
 432          let mut false_found = false;
 433          for r in m.iter() {
 434              match pat_ctor_id(cx, *r.get(0)) {
 435                None => (),
 436                Some(val(const_bool(true))) => true_found = true,
 437                Some(val(const_bool(false))) => false_found = true,
 438                _ => fail!("impossible case")
 439              }
 440          }
 441          if true_found && false_found { None }
 442          else if true_found { Some(val(const_bool(false))) }
 443          else { Some(val(const_bool(true))) }
 444        }
 445        ty::ty_vec(_, ty::vstore_fixed(n)) => {
 446          let mut missing = true;
 447          let mut wrong = false;
 448          for r in m.iter() {
 449            match r.get(0).node {
 450              PatVec(ref before, ref slice, ref after) => {
 451                let count = before.len() + after.len();
 452                if (count < n && slice.is_none()) || count > n {
 453                  wrong = true;
 454                }
 455                if count == n || (count < n && slice.is_some()) {
 456                  missing = false;
 457                }
 458              }
 459              _ => {}
 460            }
 461          }
 462          match (wrong, missing) {
 463            (true, _) => Some(vec(n)), // should be compile-time error
 464            (_, true) => Some(vec(n)),
 465            _         => None
 466          }
 467        }
 468        ty::ty_unboxed_vec(..) | ty::ty_vec(..) => {
 469  
 470          // Find the lengths and slices of all vector patterns.
 471          let mut vec_pat_lens = m.iter().filter_map(|r| {
 472              match r.get(0).node {
 473                  PatVec(ref before, ref slice, ref after) => {
 474                      Some((before.len() + after.len(), slice.is_some()))
 475                  }
 476                  _ => None
 477              }
 478          }).collect::<Vec<(uint, bool)> >();
 479  
 480          // Sort them by length such that for patterns of the same length,
 481          // those with a destructured slice come first.
 482          vec_pat_lens.sort_by(|&(len1, slice1), &(len2, slice2)| {
 483                      if len1 == len2 {
 484                          slice2.cmp(&slice1)
 485                      } else {
 486                          len1.cmp(&len2)
 487                      }
 488                  });
 489          vec_pat_lens.dedup();
 490  
 491          let mut found_slice = false;
 492          let mut next = 0;
 493          let mut missing = None;
 494          for &(length, slice) in vec_pat_lens.iter() {
 495              if length != next {
 496                  missing = Some(next);
 497                  break;
 498              }
 499              if slice {
 500                  found_slice = true;
 501                  break;
 502              }
 503              next += 1;
 504          }
 505  
 506          // We found patterns of all lengths within <0, next), yet there was no
 507          // pattern with a slice - therefore, we report vec(next) as missing.
 508          if !found_slice {
 509              missing = Some(next);
 510          }
 511          match missing {
 512            Some(k) => Some(vec(k)),
 513            None => None
 514          }
 515        }
 516        _ => Some(single)
 517      }
 518  }

 520  fn ctor_arity(cx: &MatchCheckCtxt, ctor: &ctor, ty: ty::t) -> uint {
 521      match ty::get(ty).sty {
 522        ty::ty_tup(ref fs) => fs.len(),
 523        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(..) => 1u,
 524        ty::ty_enum(eid, _) => {
 525            let id = match *ctor { variant(id) => id,
 526            _ => fail!("impossible case") };
 527          match ty::enum_variants(cx.tcx, eid).iter().find(|v| v.id == id ) {
 528              Some(v) => v.args.len(),
 529              None => fail!("impossible case")
 530          }
 531        }
 532        ty::ty_struct(cid, _) => ty::lookup_struct_fields(cx.tcx, cid).len(),
 533        ty::ty_unboxed_vec(..) | ty::ty_vec(..) => {
 534          match *ctor {
 535            vec(n) => n,
 536            _ => 0u
 537          }
 538        }
 539        _ => 0u
 540      }
 541  }

 551  fn specialize(cx: &MatchCheckCtxt,
 552                    r: &[@Pat],
 553                    ctor_id: &ctor,
 554                    arity: uint,
 555                    left_ty: ty::t)
 556                 -> Option<Vec<@Pat> > {
 557      // Sad, but I can't get rid of this easily
 558      let r0 = (*raw_pat(r[0])).clone();
 559      match r0 {
 560          Pat{id: pat_id, node: n, span: pat_span} =>
 561              match n {
 562              PatWild => {
 563                  Some(vec::append(Vec::from_elem(arity, wild()), r.tail()))
 564              }
 565              PatWildMulti => {
 566                  Some(vec::append(Vec::from_elem(arity, wild_multi()),
 567                                      r.tail()))
 568              }
 569              PatIdent(_, _, _) => {
 570                  let opt_def = cx.tcx.def_map.borrow().find_copy(&pat_id);
 571                  match opt_def {
 572                      Some(DefVariant(_, id, _)) => {
 573                          if variant(id) == *ctor_id {
 574                              Some(Vec::from_slice(r.tail()))
 575                          } else {
 576                              None
 577                          }
 578                      }
 579                      Some(DefStatic(did, _)) => {
 580                          let const_expr =
 581                              lookup_const_by_id(cx.tcx, did).unwrap();
 582                          let e_v = eval_const_expr(cx.tcx, const_expr);
 583                          let match_ = match *ctor_id {
 584                              val(ref v) => {
 585                                  match compare_const_vals(&e_v, v) {
 586                                      Some(val1) => (val1 == 0),
 587                                      None => {
 588                                          cx.tcx.sess.span_err(pat_span,
 589                                              "mismatched types between arms");
 590                                          false
 591                                      }
 592                                  }
 593                              },
 594                              range(ref c_lo, ref c_hi) => {
 595                                  let m1 = compare_const_vals(c_lo, &e_v);
 596                                  let m2 = compare_const_vals(c_hi, &e_v);
 597                                  match (m1, m2) {
 598                                      (Some(val1), Some(val2)) => {
 599                                          (val1 >= 0 && val2 <= 0)
 600                                      }
 601                                      _ => {
 602                                          cx.tcx.sess.span_err(pat_span,
 603                                              "mismatched types between ranges");
 604                                          false
 605                                      }
 606                                  }
 607                              }
 608                              single => true,
 609                              _ => fail!("type error")
 610                          };
 611                          if match_ {
 612                              Some(Vec::from_slice(r.tail()))
 613                          } else {
 614                              None
 615                          }
 616                      }
 617                      _ => {
 618                          Some(
 619                              vec::append(
 620                                  Vec::from_elem(arity, wild()),
 621                                  r.tail()
 622                              )
 623                          )
 624                      }
 625                  }
 626              }
 627              PatEnum(_, args) => {
 628                  let def = cx.tcx.def_map.borrow().get_copy(&pat_id);
 629                  match def {
 630                      DefStatic(did, _) => {
 631                          let const_expr =
 632                              lookup_const_by_id(cx.tcx, did).unwrap();
 633                          let e_v = eval_const_expr(cx.tcx, const_expr);
 634                          let match_ = match *ctor_id {
 635                              val(ref v) =>
 636                                  match compare_const_vals(&e_v, v) {
 637                                      Some(val1) => (val1 == 0),
 638                                      None => {
 639                                          cx.tcx.sess.span_err(pat_span,
 640                                              "mismatched types between arms");
 641                                          false
 642                                      }
 643                                  },
 644                              range(ref c_lo, ref c_hi) => {
 645                                  let m1 = compare_const_vals(c_lo, &e_v);
 646                                  let m2 = compare_const_vals(c_hi, &e_v);
 647                                  match (m1, m2) {
 648                                      (Some(val1), Some(val2)) => (val1 >= 0 && val2 <= 0),
 649                                      _ => {
 650                                          cx.tcx.sess.span_err(pat_span,
 651                                              "mismatched types between ranges");
 652                                          false
 653                                      }
 654                                  }
 655                              }
 656                              single => true,
 657                              _ => fail!("type error")
 658                          };
 659                          if match_ {
 660                              Some(Vec::from_slice(r.tail()))
 661                          } else {
 662                              None
 663                          }
 664                      }
 665                      DefVariant(_, id, _) if variant(id) == *ctor_id => {
 666                          let args = match args {
 667                              Some(args) => args.iter().map(|x| *x).collect(),
 668                              None => Vec::from_elem(arity, wild())
 669                          };
 670                          Some(vec::append(args, r.tail()))
 671                      }
 672                      DefVariant(_, _, _) => None,
 673  
 674                      DefFn(..) |
 675                      DefStruct(..) => {
 676                          let new_args;
 677                          match args {
 678                              Some(args) => {
 679                                  new_args = args.iter().map(|x| *x).collect()
 680                              }
 681                              None => new_args = Vec::from_elem(arity, wild())
 682                          }
 683                          Some(vec::append(new_args, r.tail()))
 684                      }
 685                      _ => None
 686                  }
 687              }
 688              PatStruct(_, ref pattern_fields, _) => {
 689                  // Is this a struct or an enum variant?
 690                  let def = cx.tcx.def_map.borrow().get_copy(&pat_id);
 691                  match def {
 692                      DefVariant(_, variant_id, _) => {
 693                          if variant(variant_id) == *ctor_id {
 694                              let struct_fields = ty::lookup_struct_fields(cx.tcx, variant_id);
 695                              let args = struct_fields.map(|sf| {
 696                                  match pattern_fields.iter().find(|f| f.ident.name == sf.name) {
 697                                      Some(f) => f.pat,
 698                                      _ => wild()
 699                                  }
 700                              });
 701                              Some(vec::append(args, r.tail()))
 702                          } else {
 703                              None
 704                          }
 705                      }
 706                      _ => {
 707                          // Grab the class data that we care about.
 708                          let class_fields;
 709                          let class_id;
 710                          match ty::get(left_ty).sty {
 711                              ty::ty_struct(cid, _) => {
 712                                  class_id = cid;
 713                                  class_fields =
 714                                      ty::lookup_struct_fields(cx.tcx,
 715                                                               class_id);
 716                              }
 717                              _ => {
 718                                  cx.tcx.sess.span_bug(
 719                                      pat_span,
 720                                      format!("struct pattern resolved to {}, \
 721                                            not a struct",
 722                                           ty_to_str(cx.tcx, left_ty)));
 723                              }
 724                          }
 725                          let args = class_fields.iter().map(|class_field| {
 726                              match pattern_fields.iter().find(|f|
 727                                              f.ident.name == class_field.name) {
 728                                  Some(f) => f.pat,
 729                                  _ => wild()
 730                              }
 731                          }).collect();
 732                          Some(vec::append(args, r.tail()))
 733                      }
 734                  }
 735              }
 736              PatTup(args) => {
 737                  Some(vec::append(args.iter().map(|x| *x).collect(), r.tail()))
 738              }
 739              PatUniq(a) | PatRegion(a) => {
 740                  Some(vec::append(vec!(a), r.tail()))
 741              }
 742              PatLit(expr) => {
 743                  let e_v = eval_const_expr(cx.tcx, expr);
 744                  let match_ = match *ctor_id {
 745                      val(ref v) => {
 746                          match compare_const_vals(&e_v, v) {
 747                              Some(val1) => val1 == 0,
 748                              None => {
 749                                  cx.tcx.sess.span_err(pat_span,
 750                                      "mismatched types between arms");
 751                                  false
 752                              }
 753                          }
 754                      },
 755                      range(ref c_lo, ref c_hi) => {
 756                          let m1 = compare_const_vals(c_lo, &e_v);
 757                          let m2 = compare_const_vals(c_hi, &e_v);
 758                          match (m1, m2) {
 759                              (Some(val1), Some(val2)) => (val1 >= 0 && val2 <= 0),
 760                              _ => {
 761                                  cx.tcx.sess.span_err(pat_span,
 762                                      "mismatched types between ranges");
 763                                  false
 764                              }
 765                          }
 766                      }
 767                      single => true,
 768                      _ => fail!("type error")
 769                  };
 770                  if match_ {
 771                      Some(Vec::from_slice(r.tail()))
 772                  } else {
 773                      None
 774                  }
 775              }
 776              PatRange(lo, hi) => {
 777                  let (c_lo, c_hi) = match *ctor_id {
 778                      val(ref v) => ((*v).clone(), (*v).clone()),
 779                      range(ref lo, ref hi) => ((*lo).clone(), (*hi).clone()),
 780                      single => return Some(Vec::from_slice(r.tail())),
 781                      _ => fail!("type error")
 782                  };
 783                  let v_lo = eval_const_expr(cx.tcx, lo);
 784                  let v_hi = eval_const_expr(cx.tcx, hi);
 785  
 786                  let m1 = compare_const_vals(&c_lo, &v_lo);
 787                  let m2 = compare_const_vals(&c_hi, &v_hi);
 788                  match (m1, m2) {
 789                      (Some(val1), Some(val2)) if val1 >= 0 && val2 <= 0 => {
 790                          Some(Vec::from_slice(r.tail()))
 791                      },
 792                      (Some(_), Some(_)) => None,
 793                      _ => {
 794                          cx.tcx.sess.span_err(pat_span,
 795                              "mismatched types between ranges");
 796                          None
 797                      }
 798                  }
 799              }
 800              PatVec(before, slice, after) => {
 801                  match *ctor_id {
 802                      vec(_) => {
 803                          let num_elements = before.len() + after.len();
 804                          if num_elements < arity && slice.is_some() {
 805                              let mut result = Vec::new();
 806                              for pat in before.iter() {
 807                                  result.push((*pat).clone());
 808                              }
 809                              for _ in iter::range(0, arity - num_elements) {
 810                                  result.push(wild())
 811                              }
 812                              for pat in after.iter() {
 813                                  result.push((*pat).clone());
 814                              }
 815                              for pat in r.tail().iter() {
 816                                  result.push((*pat).clone());
 817                              }
 818                              Some(result)
 819                          } else if num_elements == arity {
 820                              let mut result = Vec::new();
 821                              for pat in before.iter() {
 822                                  result.push((*pat).clone());
 823                              }
 824                              for pat in after.iter() {
 825                                  result.push((*pat).clone());
 826                              }
 827                              for pat in r.tail().iter() {
 828                                  result.push((*pat).clone());
 829                              }
 830                              Some(result)
 831                          } else {
 832                              None
 833                          }
 834                      }
 835                      _ => None
 836                  }
 837              }
 838          }
 839      }
 840  }

 876  fn is_refutable(cx: &MatchCheckCtxt, pat: &Pat) -> bool {
 877      let opt_def = cx.tcx.def_map.borrow().find_copy(&pat.id);
 878      match opt_def {
 879        Some(DefVariant(enum_id, _, _)) => {
 880          if ty::enum_variants(cx.tcx, enum_id).len() != 1u {
 881              return true;
 882          }
 883        }
 884        Some(DefStatic(..)) => return true,
 885        _ => ()
 886      }
 887  
 888      match pat.node {
 889        PatUniq(sub) | PatRegion(sub) | PatIdent(_, _, Some(sub)) => {
 890          is_refutable(cx, sub)
 891        }
 892        PatWild | PatWildMulti | PatIdent(_, _, None) => { false }
 893        PatLit(lit) => {
 894            match lit.node {
 895              ExprLit(lit) => {
 896                  match lit.node {
 897                      LitNil => false,    // `()`
 898                      _ => true,
 899                  }
 900              }
 901              _ => true,
 902            }
 903        }
 904        PatRange(_, _) => { true }
 905        PatStruct(_, ref fields, _) => {
 906          fields.iter().any(|f| is_refutable(cx, f.pat))
 907        }
 908        PatTup(ref elts) => {
 909          elts.iter().any(|elt| is_refutable(cx, *elt))
 910        }
 911        PatEnum(_, Some(ref args)) => {
 912          args.iter().any(|a| is_refutable(cx, *a))
 913        }
 914        PatEnum(_,_) => { false }
 915        PatVec(..) => { true }
 916      }

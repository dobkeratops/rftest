  64  fn check_expr(cx: &mut MatchCheckCtxt, ex: &Expr) {
  65      visit::walk_expr(cx, ex, ());
  66      match ex.node {
  67        ExprMatch(scrut, ref arms) => {
  68          // First, check legality of move bindings.
  69          for arm in arms.iter() {
  70              check_legality_of_move_bindings(cx,
  71                                              arm.guard.is_some(),
  72                                              arm.pats.as_slice());
  73          }
  74  
  75          check_arms(cx, arms.as_slice());
  76          /* Check for exhaustiveness */
  77           // Check for empty enum, because is_useful only works on inhabited
  78           // types.
  79         let pat_ty = node_id_to_type(cx.tcx, scrut.id);
  80         if (*arms).is_empty() {
  81             if !type_is_empty(cx.tcx, pat_ty) {
  82                 // We know the type is inhabited, so this must be wrong
  83                 cx.tcx.sess.span_err(ex.span, format!("non-exhaustive patterns: \
  84                              type {} is non-empty",
  85                              ty_to_str(cx.tcx, pat_ty)));
  86             }
  87             // If the type *is* empty, it's vacuously exhaustive
  88             return;
  89         }
  90         match ty::get(pat_ty).sty {
  91            ty_enum(did, _) => {
  92                if (*enum_variants(cx.tcx, did)).is_empty() &&
  93                      (*arms).is_empty() {
  94  
  95                 return;
  96              }
  97            }
  98            _ => { /* We assume only enum types can be uninhabited */ }
  99         }
 100  
 101         let pats: Vec<@Pat> = arms.iter()
 102                                 .filter_map(unguarded_pat)
 103                                 .flat_map(|pats| pats.move_iter())
 104                                 .collect();
 105         if pats.is_empty() {
 106             cx.tcx.sess.span_err(ex.span, "non-exhaustive patterns");
 107         } else {
 108             check_exhaustive(cx, ex.span, pats);
 109         }
 110       }
 111       _ => ()
 112      }
 113  }

 115  // Check for unreachable patterns
 116  fn check_arms(cx: &MatchCheckCtxt, arms: &[Arm]) {
 117      let mut seen = Vec::new();
 118      for arm in arms.iter() {
 119          for pat in arm.pats.iter() {
 120  
 121              // Check that we do not match against a static NaN (#6804)
 122              let pat_matches_nan: |&Pat| -> bool = |p| {
 123                  let opt_def = cx.tcx.def_map.borrow().find_copy(&p.id);
 124                  match opt_def {
 125                      Some(DefStatic(did, false)) => {
 126                          let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 127                          match eval_const_expr(cx.tcx, const_expr) {
 128                              const_float(f) if f.is_nan() => true,
 129                              _ => false
 130                          }
 131                      }
 132                      _ => false
 133                  }
 134              };
 135  
 136              walk_pat(*pat, |p| {
 137                  if pat_matches_nan(p) {
 138                      cx.tcx.sess.span_warn(p.span, "unmatchable NaN in pattern, \
 139                                                     use the is_nan method in a guard instead");
 140                  }
 141                  true
 142              });
 143  
 144              let v = vec!(*pat);
 145              match is_useful(cx, &seen, v.as_slice()) {
 146                not_useful => {
 147                  cx.tcx.sess.span_err(pat.span, "unreachable pattern");
 148                }
 149                _ => ()
 150              }
 151              if arm.guard.is_none() { seen.push(v); }
 152          }
 153      }
 154  }

 156  fn raw_pat(p: @Pat) -> @Pat {
 157      match p.node {
 158        PatIdent(_, _, Some(s)) => { raw_pat(s) }
 159        _ => { p }
 160      }
 161  }

 163  fn check_exhaustive(cx: &MatchCheckCtxt, sp: Span, pats: Vec<@Pat> ) {
 164      assert!((!pats.is_empty()));
 165      let ext = match is_useful(cx, &pats.iter().map(|p| vec!(*p)).collect(), [wild()]) {
 166          not_useful => {
 167              // This is good, wildcard pattern isn't reachable
 168              return;
 169          }
 170          useful_ => None,
 171          useful(ty, ref ctor) => {
 172              match ty::get(ty).sty {
 173                  ty::ty_bool => {
 174                      match *ctor {
 175                          val(const_bool(true)) => Some(~"true"),
 176                          val(const_bool(false)) => Some(~"false"),
 177                          _ => None
 178                      }
 179                  }
 180                  ty::ty_enum(id, _) => {
 181                      let vid = match *ctor {
 182                          variant(id) => id,
 183                          _ => fail!("check_exhaustive: non-variant ctor"),
 184                      };
 185                      let variants = ty::enum_variants(cx.tcx, id);
 186  
 187                      match variants.iter().find(|v| v.id == vid) {
 188                          Some(v) => Some(token::get_ident(v.name).get().to_str()),
 189                          None => {
 190                              fail!("check_exhaustive: bad variant in ctor")
 191                          }
 192                      }
 193                  }
 194                  ty::ty_vec(..) => {
 195                      match *ctor {
 196                          vec(n) => Some(format!("vectors of length {}", n)),
 197                          _ => None
 198                      }
 199                  }
 200                  _ => None
 201              }
 202          }
 203      };
 204      let msg = ~"non-exhaustive patterns" + match ext {
 205          Some(ref s) => format!(": {} not covered",  *s),
 206          None => ~""
 207      };
 208      cx.tcx.sess.span_err(sp, msg);
 209  }

 213  #[deriving(Clone)]
 214  enum useful {
 215      useful(ty::t, ctor),
 216      useful_,
 217      not_useful,
 218  }
 219  
 220  #[deriving(Clone, Eq)]
 221  enum ctor {
 222      single,
 223      variant(DefId),
 224      val(const_val),
 225      range(const_val, const_val),
 226      vec(uint)
 227  }
 228  
 229  // Algorithm from http://moscova.inria.fr/~maranget/papers/warn/index.html
 230  //
 231  // Whether a vector `v` of patterns is 'useful' in relation to a set of such
 232  // vectors `m` is defined as there being a set of inputs that will match `v`
 233  // but not any of the sets in `m`.
 234  //
 235  // This is used both for reachability checking (if a pattern isn't useful in
 236  // relation to preceding patterns, it is not reachable) and exhaustiveness
 237  // checking (if a wildcard pattern is useful in relation to a matrix, the
 238  // matrix isn't exhaustive).
 239  
 240  // Note: is_useful doesn't work on empty types, as the paper notes.
 241  // So it assumes that v is non-empty.
 242  fn is_useful(cx: &MatchCheckCtxt, m: &matrix, v: &[@Pat]) -> useful {
 243      if m.len() == 0u {
 244          return useful_;
 245      }
 246      if m.get(0).len() == 0u {
 247          return not_useful
 248      }
 249      let real_pat = match m.iter().find(|r| r.get(0).id != 0) {
 250        Some(r) => *r.get(0), None => v[0]
 251      };
 252      let left_ty = if real_pat.id == 0 { ty::mk_nil() }
 253                    else { ty::node_id_to_type(cx.tcx, real_pat.id) };
 254  
 255      match pat_ctor_id(cx, v[0]) {
 256        None => {
 257          match missing_ctor(cx, m, left_ty) {
 258            None => {
 259              match ty::get(left_ty).sty {
 260                ty::ty_bool => {
 261                  match is_useful_specialized(cx, m, v,
 262                                              val(const_bool(true)),
 263                                              0u, left_ty){
 264                    not_useful => {
 265                      is_useful_specialized(cx, m, v,
 266                                            val(const_bool(false)),
 267                                            0u, left_ty)
 268                    }
 269                    ref u => (*u).clone(),
 270                  }
 271                }
 272                ty::ty_enum(eid, _) => {
 273                  for va in (*ty::enum_variants(cx.tcx, eid)).iter() {
 274                      match is_useful_specialized(cx, m, v, variant(va.id),
 275                                                  va.args.len(), left_ty) {
 276                        not_useful => (),
 277                        ref u => return (*u).clone(),
 278                      }
 279                  }
 280                  not_useful
 281                }
 282                ty::ty_vec(_, ty::vstore_fixed(n)) => {
 283                  is_useful_specialized(cx, m, v, vec(n), n, left_ty)
 284                }
 285                ty::ty_vec(..) => {
 286                  let max_len = m.iter().rev().fold(0, |max_len, r| {
 287                    match r.get(0).node {
 288                      PatVec(ref before, _, ref after) => {
 289                        cmp::max(before.len() + after.len(), max_len)
 290                      }
 291                      _ => max_len
 292                    }
 293                  });
 294                  for n in iter::range(0u, max_len + 1) {
 295                    match is_useful_specialized(cx, m, v, vec(n), n, left_ty) {
 296                      not_useful => (),
 297                      ref u => return (*u).clone(),
 298                    }
 299                  }
 300                  not_useful
 301                }
 302                _ => {
 303                  let arity = ctor_arity(cx, &single, left_ty);
 304                  is_useful_specialized(cx, m, v, single, arity, left_ty)
 305                }
 306              }
 307            }
 308            Some(ref ctor) => {
 309              match is_useful(cx,
 310                              &m.iter().filter_map(|r| {
 311                                  default(cx, r.as_slice())
 312                              }).collect::<matrix>(),
 313                              v.tail()) {
 314                useful_ => useful(left_ty, (*ctor).clone()),
 315                ref u => (*u).clone(),
 316              }
 317            }
 318          }
 319        }
 320        Some(ref v0_ctor) => {
 321          let arity = ctor_arity(cx, v0_ctor, left_ty);
 322          is_useful_specialized(cx, m, v, (*v0_ctor).clone(), arity, left_ty)
 323        }
 324      }
 325  }

 327  fn is_useful_specialized(cx: &MatchCheckCtxt,
 328                               m: &matrix,
 329                               v: &[@Pat],
 330                               ctor: ctor,
 331                               arity: uint,
 332                               lty: ty::t)
 333                               -> useful {
 334      let ms = m.iter().filter_map(|r| {
 335          specialize(cx, r.as_slice(), &ctor, arity, lty)
 336      }).collect::<matrix>();
 337      let could_be_useful = is_useful(
 338          cx, &ms, specialize(cx, v, &ctor, arity, lty).unwrap().as_slice());
 339      match could_be_useful {
 340        useful_ => useful(lty, ctor),
 341        ref u => (*u).clone(),
 342      }
 343  }

 345  fn pat_ctor_id(cx: &MatchCheckCtxt, p: @Pat) -> Option<ctor> {
 346      let pat = raw_pat(p);
 347      match pat.node {
 348        PatWild | PatWildMulti => { None }
 349        PatIdent(_, _, _) | PatEnum(_, _) => {
 350          let opt_def = cx.tcx.def_map.borrow().find_copy(&pat.id);
 351          match opt_def {
 352            Some(DefVariant(_, id, _)) => Some(variant(id)),
 353            Some(DefStatic(did, false)) => {
 354              let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 355              Some(val(eval_const_expr(cx.tcx, const_expr)))
 356            }
 357            _ => None
 358          }
 359        }
 360        PatLit(expr) => { Some(val(eval_const_expr(cx.tcx, expr))) }
 361        PatRange(lo, hi) => {
 362          Some(range(eval_const_expr(cx.tcx, lo), eval_const_expr(cx.tcx, hi)))
 363        }
 364        PatStruct(..) => {
 365          match cx.tcx.def_map.borrow().find(&pat.id) {
 366            Some(&DefVariant(_, id, _)) => Some(variant(id)),
 367            _ => Some(single)
 368          }
 369        }
 370        PatUniq(_) | PatTup(_) | PatRegion(..) => {
 371          Some(single)
 372        }
 373        PatVec(ref before, slice, ref after) => {
 374          match slice {
 375            Some(_) => None,
 376            None => Some(vec(before.len() + after.len()))
 377          }
 378        }
 379      }
 380  }

 382  fn is_wild(cx: &MatchCheckCtxt, p: @Pat) -> bool {
 383      let pat = raw_pat(p);
 384      match pat.node {
 385        PatWild | PatWildMulti => { true }
 386        PatIdent(_, _, _) => {
 387          match cx.tcx.def_map.borrow().find(&pat.id) {
 388            Some(&DefVariant(_, _, _)) | Some(&DefStatic(..)) => { false }
 389            _ => { true }
 390          }
 391        }
 392        _ => { false }
 393      }
 394  }

 396  fn missing_ctor(cx: &MatchCheckCtxt,
 397                      m: &matrix,
 398                      left_ty: ty::t)
 399                   -> Option<ctor> {
 400      match ty::get(left_ty).sty {
 401        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(..) | ty::ty_tup(_) |
 402        ty::ty_struct(..) => {
 403          for r in m.iter() {
 404              if !is_wild(cx, *r.get(0)) { return None; }
 405          }
 406          return Some(single);
 407        }
 408        ty::ty_enum(eid, _) => {
 409          let mut found = Vec::new();
 410          for r in m.iter() {
 411              let r = pat_ctor_id(cx, *r.get(0));
 412              for id in r.iter() {
 413                  if !found.contains(id) {
 414                      found.push((*id).clone());
 415                  }
 416              }
 417          }
 418          let variants = ty::enum_variants(cx.tcx, eid);
 419          if found.len() != (*variants).len() {
 420              for v in (*variants).iter() {
 421                  if !found.iter().any(|x| x == &(variant(v.id))) {
 422                      return Some(variant(v.id));
 423                  }
 424              }
 425              fail!();
 426          } else { None }
 427        }
 428        ty::ty_nil => None,
 429        ty::ty_bool => {
 430          let mut true_found = false;
 431          let mut false_found = false;
 432          for r in m.iter() {
 433              match pat_ctor_id(cx, *r.get(0)) {
 434                None => (),
 435                Some(val(const_bool(true))) => true_found = true,
 436                Some(val(const_bool(false))) => false_found = true,
 437                _ => fail!("impossible case")
 438              }
 439          }
 440          if true_found && false_found { None }
 441          else if true_found { Some(val(const_bool(false))) }
 442          else { Some(val(const_bool(true))) }
 443        }
 444        ty::ty_vec(_, ty::vstore_fixed(n)) => {
 445          let mut missing = true;
 446          let mut wrong = false;
 447          for r in m.iter() {
 448            match r.get(0).node {
 449              PatVec(ref before, ref slice, ref after) => {
 450                let count = before.len() + after.len();
 451                if (count < n && slice.is_none()) || count > n {
 452                  wrong = true;
 453                }
 454                if count == n || (count < n && slice.is_some()) {
 455                  missing = false;
 456                }
 457              }
 458              _ => {}
 459            }
 460          }
 461          match (wrong, missing) {
 462            (true, _) => Some(vec(n)), // should be compile-time error
 463            (_, true) => Some(vec(n)),
 464            _         => None
 465          }
 466        }
 467        ty::ty_vec(..) => {
 468  
 469          // Find the lengths and slices of all vector patterns.
 470          let mut vec_pat_lens = m.iter().filter_map(|r| {
 471              match r.get(0).node {
 472                  PatVec(ref before, ref slice, ref after) => {
 473                      Some((before.len() + after.len(), slice.is_some()))
 474                  }
 475                  _ => None
 476              }
 477          }).collect::<Vec<(uint, bool)> >();
 478  
 479          // Sort them by length such that for patterns of the same length,
 480          // those with a destructured slice come first.
 481          vec_pat_lens.sort_by(|&(len1, slice1), &(len2, slice2)| {
 482                      if len1 == len2 {
 483                          slice2.cmp(&slice1)
 484                      } else {
 485                          len1.cmp(&len2)
 486                      }
 487                  });
 488          vec_pat_lens.dedup();
 489  
 490          let mut found_slice = false;
 491          let mut next = 0;
 492          let mut missing = None;
 493          for &(length, slice) in vec_pat_lens.iter() {
 494              if length != next {
 495                  missing = Some(next);
 496                  break;
 497              }
 498              if slice {
 499                  found_slice = true;
 500                  break;
 501              }
 502              next += 1;
 503          }
 504  
 505          // We found patterns of all lengths within <0, next), yet there was no
 506          // pattern with a slice - therefore, we report vec(next) as missing.
 507          if !found_slice {
 508              missing = Some(next);
 509          }
 510          match missing {
 511            Some(k) => Some(vec(k)),
 512            None => None
 513          }
 514        }
 515        _ => Some(single)
 516      }
 517  }

 519  fn ctor_arity(cx: &MatchCheckCtxt, ctor: &ctor, ty: ty::t) -> uint {
 520      match ty::get(ty).sty {
 521        ty::ty_tup(ref fs) => fs.len(),
 522        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(..) => 1u,
 523        ty::ty_enum(eid, _) => {
 524            let id = match *ctor { variant(id) => id,
 525            _ => fail!("impossible case") };
 526          match ty::enum_variants(cx.tcx, eid).iter().find(|v| v.id == id ) {
 527              Some(v) => v.args.len(),
 528              None => fail!("impossible case")
 529          }
 530        }
 531        ty::ty_struct(cid, _) => ty::lookup_struct_fields(cx.tcx, cid).len(),
 532        ty::ty_vec(..) => {
 533          match *ctor {
 534            vec(n) => n,
 535            _ => 0u
 536          }
 537        }
 538        _ => 0u
 539      }
 540  }

 550  fn specialize(cx: &MatchCheckCtxt,
 551                    r: &[@Pat],
 552                    ctor_id: &ctor,
 553                    arity: uint,
 554                    left_ty: ty::t)
 555                 -> Option<Vec<@Pat> > {
 556      // Sad, but I can't get rid of this easily
 557      let r0 = (*raw_pat(r[0])).clone();
 558      match r0 {
 559          Pat{id: pat_id, node: n, span: pat_span} =>
 560              match n {
 561              PatWild => {
 562                  Some(Vec::from_elem(arity, wild()).append(r.tail()))
 563              }
 564              PatWildMulti => {
 565                  Some(Vec::from_elem(arity, wild_multi()).append(r.tail()))
 566              }
 567              PatIdent(_, _, _) => {
 568                  let opt_def = cx.tcx.def_map.borrow().find_copy(&pat_id);
 569                  match opt_def {
 570                      Some(DefVariant(_, id, _)) => {
 571                          if variant(id) == *ctor_id {
 572                              Some(Vec::from_slice(r.tail()))
 573                          } else {
 574                              None
 575                          }
 576                      }
 577                      Some(DefStatic(did, _)) => {
 578                          let const_expr =
 579                              lookup_const_by_id(cx.tcx, did).unwrap();
 580                          let e_v = eval_const_expr(cx.tcx, const_expr);
 581                          let match_ = match *ctor_id {
 582                              val(ref v) => {
 583                                  match compare_const_vals(&e_v, v) {
 584                                      Some(val1) => (val1 == 0),
 585                                      None => {
 586                                          cx.tcx.sess.span_err(pat_span,
 587                                              "mismatched types between arms");
 588                                          false
 589                                      }
 590                                  }
 591                              },
 592                              range(ref c_lo, ref c_hi) => {
 593                                  let m1 = compare_const_vals(c_lo, &e_v);
 594                                  let m2 = compare_const_vals(c_hi, &e_v);
 595                                  match (m1, m2) {
 596                                      (Some(val1), Some(val2)) => {
 597                                          (val1 >= 0 && val2 <= 0)
 598                                      }
 599                                      _ => {
 600                                          cx.tcx.sess.span_err(pat_span,
 601                                              "mismatched types between ranges");
 602                                          false
 603                                      }
 604                                  }
 605                              }
 606                              single => true,
 607                              _ => fail!("type error")
 608                          };
 609                          if match_ {
 610                              Some(Vec::from_slice(r.tail()))
 611                          } else {
 612                              None
 613                          }
 614                      }
 615                      _ => {
 616                          Some(Vec::from_elem(arity, wild()).append(r.tail()))
 617                      }
 618                  }
 619              }
 620              PatEnum(_, args) => {
 621                  let def = cx.tcx.def_map.borrow().get_copy(&pat_id);
 622                  match def {
 623                      DefStatic(did, _) => {
 624                          let const_expr =
 625                              lookup_const_by_id(cx.tcx, did).unwrap();
 626                          let e_v = eval_const_expr(cx.tcx, const_expr);
 627                          let match_ = match *ctor_id {
 628                              val(ref v) =>
 629                                  match compare_const_vals(&e_v, v) {
 630                                      Some(val1) => (val1 == 0),
 631                                      None => {
 632                                          cx.tcx.sess.span_err(pat_span,
 633                                              "mismatched types between arms");
 634                                          false
 635                                      }
 636                                  },
 637                              range(ref c_lo, ref c_hi) => {
 638                                  let m1 = compare_const_vals(c_lo, &e_v);
 639                                  let m2 = compare_const_vals(c_hi, &e_v);
 640                                  match (m1, m2) {
 641                                      (Some(val1), Some(val2)) => (val1 >= 0 && val2 <= 0),
 642                                      _ => {
 643                                          cx.tcx.sess.span_err(pat_span,
 644                                              "mismatched types between ranges");
 645                                          false
 646                                      }
 647                                  }
 648                              }
 649                              single => true,
 650                              _ => fail!("type error")
 651                          };
 652                          if match_ {
 653                              Some(Vec::from_slice(r.tail()))
 654                          } else {
 655                              None
 656                          }
 657                      }
 658                      DefVariant(_, id, _) if variant(id) == *ctor_id => {
 659                          let args = match args {
 660                              Some(args) => args.iter().map(|x| *x).collect(),
 661                              None => Vec::from_elem(arity, wild())
 662                          };
 663                          Some(args.append(r.tail()))
 664                      }
 665                      DefVariant(_, _, _) => None,
 666  
 667                      DefFn(..) |
 668                      DefStruct(..) => {
 669                          let new_args;
 670                          match args {
 671                              Some(args) => {
 672                                  new_args = args.iter().map(|x| *x).collect()
 673                              }
 674                              None => new_args = Vec::from_elem(arity, wild())
 675                          }
 676                          Some(new_args.append(r.tail()))
 677                      }
 678                      _ => None
 679                  }
 680              }
 681              PatStruct(_, ref pattern_fields, _) => {
 682                  // Is this a struct or an enum variant?
 683                  let def = cx.tcx.def_map.borrow().get_copy(&pat_id);
 684                  match def {
 685                      DefVariant(_, variant_id, _) => {
 686                          if variant(variant_id) == *ctor_id {
 687                              let struct_fields = ty::lookup_struct_fields(cx.tcx, variant_id);
 688                              let args = struct_fields.iter().map(|sf| {
 689                                  match pattern_fields.iter().find(|f| f.ident.name == sf.name) {
 690                                      Some(f) => f.pat,
 691                                      _ => wild()
 692                                  }
 693                              }).collect::<Vec<_>>();
 694                              Some(args.append(r.tail()))
 695                          } else {
 696                              None
 697                          }
 698                      }
 699                      _ => {
 700                          // Grab the class data that we care about.
 701                          let class_fields;
 702                          let class_id;
 703                          match ty::get(left_ty).sty {
 704                              ty::ty_struct(cid, _) => {
 705                                  class_id = cid;
 706                                  class_fields =
 707                                      ty::lookup_struct_fields(cx.tcx,
 708                                                               class_id);
 709                              }
 710                              _ => {
 711                                  cx.tcx.sess.span_bug(
 712                                      pat_span,
 713                                      format!("struct pattern resolved to {}, \
 714                                            not a struct",
 715                                           ty_to_str(cx.tcx, left_ty)));
 716                              }
 717                          }
 718                          let args = class_fields.iter().map(|class_field| {
 719                              match pattern_fields.iter().find(|f|
 720                                              f.ident.name == class_field.name) {
 721                                  Some(f) => f.pat,
 722                                  _ => wild()
 723                              }
 724                          }).collect::<Vec<_>>();
 725                          Some(args.append(r.tail()))
 726                      }
 727                  }
 728              }
 729              PatTup(args) => {
 730                  Some(args.iter().map(|x| *x).collect::<Vec<_>>().append(r.tail()))
 731              }
 732              PatUniq(a) | PatRegion(a) => {
 733                  Some((vec!(a)).append(r.tail()))
 734              }
 735              PatLit(expr) => {
 736                  let e_v = eval_const_expr(cx.tcx, expr);
 737                  let match_ = match *ctor_id {
 738                      val(ref v) => {
 739                          match compare_const_vals(&e_v, v) {
 740                              Some(val1) => val1 == 0,
 741                              None => {
 742                                  cx.tcx.sess.span_err(pat_span,
 743                                      "mismatched types between arms");
 744                                  false
 745                              }
 746                          }
 747                      },
 748                      range(ref c_lo, ref c_hi) => {
 749                          let m1 = compare_const_vals(c_lo, &e_v);
 750                          let m2 = compare_const_vals(c_hi, &e_v);
 751                          match (m1, m2) {
 752                              (Some(val1), Some(val2)) => (val1 >= 0 && val2 <= 0),
 753                              _ => {
 754                                  cx.tcx.sess.span_err(pat_span,
 755                                      "mismatched types between ranges");
 756                                  false
 757                              }
 758                          }
 759                      }
 760                      single => true,
 761                      _ => fail!("type error")
 762                  };
 763                  if match_ {
 764                      Some(Vec::from_slice(r.tail()))
 765                  } else {
 766                      None
 767                  }
 768              }
 769              PatRange(lo, hi) => {
 770                  let (c_lo, c_hi) = match *ctor_id {
 771                      val(ref v) => ((*v).clone(), (*v).clone()),
 772                      range(ref lo, ref hi) => ((*lo).clone(), (*hi).clone()),
 773                      single => return Some(Vec::from_slice(r.tail())),
 774                      _ => fail!("type error")
 775                  };
 776                  let v_lo = eval_const_expr(cx.tcx, lo);
 777                  let v_hi = eval_const_expr(cx.tcx, hi);
 778  
 779                  let m1 = compare_const_vals(&c_lo, &v_lo);
 780                  let m2 = compare_const_vals(&c_hi, &v_hi);
 781                  match (m1, m2) {
 782                      (Some(val1), Some(val2)) if val1 >= 0 && val2 <= 0 => {
 783                          Some(Vec::from_slice(r.tail()))
 784                      },
 785                      (Some(_), Some(_)) => None,
 786                      _ => {
 787                          cx.tcx.sess.span_err(pat_span,
 788                              "mismatched types between ranges");
 789                          None
 790                      }
 791                  }
 792              }
 793              PatVec(before, slice, after) => {
 794                  match *ctor_id {
 795                      vec(_) => {
 796                          let num_elements = before.len() + after.len();
 797                          if num_elements < arity && slice.is_some() {
 798                              let mut result = Vec::new();
 799                              for pat in before.iter() {
 800                                  result.push((*pat).clone());
 801                              }
 802                              for _ in iter::range(0, arity - num_elements) {
 803                                  result.push(wild())
 804                              }
 805                              for pat in after.iter() {
 806                                  result.push((*pat).clone());
 807                              }
 808                              for pat in r.tail().iter() {
 809                                  result.push((*pat).clone());
 810                              }
 811                              Some(result)
 812                          } else if num_elements == arity {
 813                              let mut result = Vec::new();
 814                              for pat in before.iter() {
 815                                  result.push((*pat).clone());
 816                              }
 817                              for pat in after.iter() {
 818                                  result.push((*pat).clone());
 819                              }
 820                              for pat in r.tail().iter() {
 821                                  result.push((*pat).clone());
 822                              }
 823                              Some(result)
 824                          } else {
 825                              None
 826                          }
 827                      }
 828                      _ => None
 829                  }
 830              }
 831          }
 832      }
 833  }

 869  fn is_refutable(cx: &MatchCheckCtxt, pat: &Pat) -> bool {
 870      let opt_def = cx.tcx.def_map.borrow().find_copy(&pat.id);
 871      match opt_def {
 872        Some(DefVariant(enum_id, _, _)) => {
 873          if ty::enum_variants(cx.tcx, enum_id).len() != 1u {
 874              return true;
 875          }
 876        }
 877        Some(DefStatic(..)) => return true,
 878        _ => ()
 879      }
 880  
 881      match pat.node {
 882        PatUniq(sub) | PatRegion(sub) | PatIdent(_, _, Some(sub)) => {
 883          is_refutable(cx, sub)
 884        }
 885        PatWild | PatWildMulti | PatIdent(_, _, None) => { false }
 886        PatLit(lit) => {
 887            match lit.node {
 888              ExprLit(lit) => {
 889                  match lit.node {
 890                      LitNil => false,    // `()`
 891                      _ => true,
 892                  }
 893              }
 894              _ => true,
 895            }
 896        }
 897        PatRange(_, _) => { true }
 898        PatStruct(_, ref fields, _) => {
 899          fields.iter().any(|f| is_refutable(cx, f.pat))
 900        }
 901        PatTup(ref elts) => {
 902          elts.iter().any(|elt| is_refutable(cx, *elt))
 903        }
 904        PatEnum(_, Some(ref args)) => {
 905          args.iter().any(|a| is_refutable(cx, *a))
 906        }
 907        PatEnum(_,_) => { false }
 908        PatVec(..) => { true }
 909      }
 910  }

 914  fn check_legality_of_move_bindings(cx: &MatchCheckCtxt,
 915                                         has_guard: bool,

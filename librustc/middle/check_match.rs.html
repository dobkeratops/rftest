  37                     method_map: method_map,
  38                     moves_map: moves::MovesMap,
  39                     crate: &Crate) {
  40      let cx = @MatchCheckCtxt {tcx: tcx,
  41                                method_map: method_map,
  42                                moves_map: moves_map};
  43      oldvisit::visit_crate(crate, ((), oldvisit::mk_vt(@oldvisit::Visitor {
  44          visit_expr: |a,b| check_expr(cx, a, b),
  45          visit_local: |a,b| check_local(cx, a, b),
  46          visit_fn: |kind, decl, body, sp, id, (e, v)|
  47              check_fn(cx, kind, decl, body, sp, id, (e, v)),
  48          .. *oldvisit::default_visitor::<()>()

  53  pub fn check_expr(cx: @MatchCheckCtxt,
  54                    ex: @expr,
  55                    (s, v): ((), oldvisit::vt<()>)) {
  56      oldvisit::visit_expr(ex, (s, v));
  57      match ex.node {
  58        expr_match(scrut, ref arms) => {
  59          // First, check legality of move bindings.
  60          for arm in arms.iter() {
  61              check_legality_of_move_bindings(cx,
  62                                              arm.guard.is_some(),
  63                                              arm.pats);
  64          }
  65  
  66          check_arms(cx, *arms);
  67          /* Check for exhaustiveness */
  68           // Check for empty enum, because is_useful only works on inhabited
  69           // types.
  70         let pat_ty = node_id_to_type(cx.tcx, scrut.id);
  71         if (*arms).is_empty() {
  72             if !type_is_empty(cx.tcx, pat_ty) {
  73                 // We know the type is inhabited, so this must be wrong
  74                 cx.tcx.sess.span_err(ex.span, fmt!("non-exhaustive patterns: \
  75                              type %s is non-empty",
  76                              ty_to_str(cx.tcx, pat_ty)));
  77             }
  78             // If the type *is* empty, it's vacuously exhaustive
  79             return;
  80         }
  81         match ty::get(pat_ty).sty {
  82            ty_enum(did, _) => {
  83                if (*enum_variants(cx.tcx, did)).is_empty() &&
  84                      (*arms).is_empty() {
  85  
  86                 return;
  87              }
  88            }
  89            _ => { /* We assume only enum types can be uninhabited */ }
  90         }
  91         let arms = arms.iter().filter_map(unguarded_pat).collect::<~[~[@pat]]>().concat_vec();
  92         if arms.is_empty() {
  93             cx.tcx.sess.span_err(ex.span, "non-exhaustive patterns");
  94         } else {
  95             check_exhaustive(cx, ex.span, arms);
  96         }
  97       }
  98       _ => ()
  99      }
 100  }

 102  // Check for unreachable patterns
 103  pub fn check_arms(cx: &MatchCheckCtxt, arms: &[arm]) {
 104      let mut seen = ~[];
 105      for arm in arms.iter() {
 106          for pat in arm.pats.iter() {
 107  
 108              // Check that we do not match against a static NaN (#6804)
 109              let pat_matches_nan: &fn(@pat) -> bool = |p| {
 110                  match cx.tcx.def_map.find(&p.id) {
 111                      Some(&def_static(did, false)) => {
 112                          let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 113                          match eval_const_expr(cx.tcx, const_expr) {
 114                              const_float(f) if f.is_NaN() => true,
 115                              _ => false
 116                          }
 117                      }
 118                      _ => false
 119                  }
 120              };
 121              do walk_pat(*pat) |p| {
 122                  if pat_matches_nan(p) {
 123                      cx.tcx.sess.span_warn(p.span, "unmatchable NaN in pattern, \
 124                                                     use the is_NaN method in a guard instead");
 125                  }
 126                  true
 127              };
 128  
 129              let v = ~[*pat];
 130              match is_useful(cx, &seen, v) {
 131                not_useful => {
 132                  cx.tcx.sess.span_err(pat.span, "unreachable pattern");
 133                }
 134                _ => ()
 135              }
 136              if arm.guard.is_none() { seen.push(v); }
 137          }
 138      }
 139  }

 142      match p.node {
 143        pat_ident(_, _, Some(s)) => { raw_pat(s) }
 144        _ => { p }
 145      }
 146  }

 148  pub fn check_exhaustive(cx: &MatchCheckCtxt, sp: span, pats: ~[@pat]) {
 149      assert!((!pats.is_empty()));
 150      let ext = match is_useful(cx, &pats.map(|p| ~[*p]), [wild()]) {
 151          not_useful => {
 152              // This is good, wildcard pattern isn't reachable
 153              return;
 154          }
 155          useful_ => None,
 156          useful(ty, ref ctor) => {
 157              match ty::get(ty).sty {
 158                  ty::ty_bool => {
 159                      match (*ctor) {
 160                          val(const_bool(true)) => Some(@"true"),
 161                          val(const_bool(false)) => Some(@"false"),
 162                          _ => None
 163                      }
 164                  }
 165                  ty::ty_enum(id, _) => {
 166                      let vid = match *ctor {
 167                          variant(id) => id,
 168                          _ => fail!("check_exhaustive: non-variant ctor"),
 169                      };
 170                      let variants = ty::enum_variants(cx.tcx, id);
 171  
 172                      match variants.iter().find(|v| v.id == vid) {
 173                          Some(v) => Some(cx.tcx.sess.str_of(v.name)),
 174                          None => {
 175                              fail!("check_exhaustive: bad variant in ctor")
 176                          }
 177                      }
 178                  }
 179                  ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 180                      match *ctor {
 181                          vec(n) => Some(fmt!("vectors of length %u", n).to_managed()),
 182                          _ => None
 183                      }
 184                  }
 185                  _ => None
 186              }
 187          }
 188      };
 189      let msg = ~"non-exhaustive patterns" + match ext {
 190          Some(ref s) => fmt!(": %s not covered",  *s),
 191          None => ~""
 192      };
 193      cx.tcx.sess.span_err(sp, msg);
 194  }

 198  pub enum useful { useful(ty::t, ctor), useful_, not_useful }
 199  
 200  #[deriving(Eq)]
 201  pub enum ctor {
 202      single,
 203      variant(def_id),
 204      val(const_val),
 205      range(const_val, const_val),
 206      vec(uint)
 207  }

 209  // Algorithm from http://moscova.inria.fr/~maranget/papers/warn/index.html
 210  //
 211  // Whether a vector `v` of patterns is 'useful' in relation to a set of such
 212  // vectors `m` is defined as there being a set of inputs that will match `v`
 213  // but not any of the sets in `m`.
 214  //
 215  // This is used both for reachability checking (if a pattern isn't useful in
 216  // relation to preceding patterns, it is not reachable) and exhaustiveness
 217  // checking (if a wildcard pattern is useful in relation to a matrix, the
 218  // matrix isn't exhaustive).
 219  
 220  // Note: is_useful doesn't work on empty types, as the paper notes.
 221  // So it assumes that v is non-empty.
 222  pub fn is_useful(cx: &MatchCheckCtxt, m: &matrix, v: &[@pat]) -> useful {
 223      if m.len() == 0u { return useful_; }
 224      if m[0].len() == 0u { return not_useful; }
 225      let real_pat = match m.iter().find(|r| r[0].id != 0) {
 226        Some(r) => r[0], None => v[0]
 227      };
 228      let left_ty = if real_pat.id == 0 { ty::mk_nil() }
 229                    else { ty::node_id_to_type(cx.tcx, real_pat.id) };
 230  
 231      match pat_ctor_id(cx, v[0]) {
 232        None => {
 233          match missing_ctor(cx, m, left_ty) {
 234            None => {
 235              match ty::get(left_ty).sty {
 236                ty::ty_bool => {
 237                  match is_useful_specialized(cx, m, v,
 238                                              val(const_bool(true)),
 239                                              0u, left_ty){
 240                    not_useful => {
 241                      is_useful_specialized(cx, m, v,
 242                                            val(const_bool(false)),
 243                                            0u, left_ty)
 244                    }
 245                    ref u => *u,
 246                  }
 247                }
 248                ty::ty_enum(eid, _) => {
 249                  for va in (*ty::enum_variants(cx.tcx, eid)).iter() {
 250                      match is_useful_specialized(cx, m, v, variant(va.id),
 251                                                  va.args.len(), left_ty) {
 252                        not_useful => (),
 253                        ref u => return *u,
 254                      }
 255                  }
 256                  not_useful
 257                }
 258                ty::ty_evec(_, ty::vstore_fixed(n)) => {
 259                  is_useful_specialized(cx, m, v, vec(n), n, left_ty)
 260                }
 261                ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 262                  let max_len = do m.rev_iter().fold(0) |max_len, r| {
 263                    match r[0].node {
 264                      pat_vec(ref before, _, ref after) => {
 265                        num::max(before.len() + after.len(), max_len)
 266                      }
 267                      _ => max_len
 268                    }
 269                  };
 270                  for n in iterator::range(0u, max_len + 1) {
 271                    match is_useful_specialized(cx, m, v, vec(n), n, left_ty) {
 272                      not_useful => (),
 273                      ref u => return *u,
 274                    }
 275                  }
 276                  not_useful
 277                }
 278                _ => {
 279                  let arity = ctor_arity(cx, &single, left_ty);
 280                  is_useful_specialized(cx, m, v, single, arity, left_ty)
 281                }
 282              }
 283            }
 284            Some(ref ctor) => {
 285              match is_useful(cx,
 286                              &m.iter().filter_map(|r| default(cx, *r)).collect::<matrix>(),
 287                              v.tail()) {
 288                useful_ => useful(left_ty, *ctor),
 289                ref u => *u,
 290              }
 291            }
 292          }
 293        }
 294        Some(ref v0_ctor) => {
 295          let arity = ctor_arity(cx, v0_ctor, left_ty);
 296          is_useful_specialized(cx, m, v, *v0_ctor, arity, left_ty)
 297        }
 298      }
 299  }

 301  pub fn is_useful_specialized(cx: &MatchCheckCtxt,
 302                               m: &matrix,
 303                               v: &[@pat],
 304                               ctor: ctor,
 305                               arity: uint,
 306                               lty: ty::t)
 307                            -> useful {
 308      let ms = m.iter().filter_map(|r| specialize(cx, *r, &ctor, arity, lty)).collect::<matrix>();
 309      let could_be_useful = is_useful(
 310          cx, &ms, specialize(cx, v, &ctor, arity, lty).unwrap());
 311      match could_be_useful {
 312        useful_ => useful(lty, ctor),
 313        ref u => *u,
 314      }
 315  }

 317  pub fn pat_ctor_id(cx: &MatchCheckCtxt, p: @pat) -> Option<ctor> {
 318      let pat = raw_pat(p);
 319      match pat.node {
 320        pat_wild => { None }
 321        pat_ident(_, _, _) | pat_enum(_, _) => {
 322          match cx.tcx.def_map.find(&pat.id) {
 323            Some(&def_variant(_, id)) => Some(variant(id)),
 324            Some(&def_static(did, false)) => {
 325              let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 326              Some(val(eval_const_expr(cx.tcx, const_expr)))
 327            }
 328            _ => None
 329          }
 330        }
 331        pat_lit(expr) => { Some(val(eval_const_expr(cx.tcx, expr))) }
 332        pat_range(lo, hi) => {
 333          Some(range(eval_const_expr(cx.tcx, lo), eval_const_expr(cx.tcx, hi)))
 334        }
 335        pat_struct(*) => {
 336          match cx.tcx.def_map.find(&pat.id) {
 337            Some(&def_variant(_, id)) => Some(variant(id)),
 338            _ => Some(single)
 339          }
 340        }
 341        pat_box(_) | pat_uniq(_) | pat_tup(_) | pat_region(*) => {
 342          Some(single)
 343        }
 344        pat_vec(ref before, slice, ref after) => {
 345          match slice {
 346            Some(_) => None,
 347            None => Some(vec(before.len() + after.len()))
 348          }
 349        }
 350      }
 351  }

 353  pub fn is_wild(cx: &MatchCheckCtxt, p: @pat) -> bool {
 354      let pat = raw_pat(p);
 355      match pat.node {
 356        pat_wild => { true }
 357        pat_ident(_, _, _) => {
 358          match cx.tcx.def_map.find(&pat.id) {
 359            Some(&def_variant(_, _)) | Some(&def_static(*)) => { false }
 360            _ => { true }
 361          }
 362        }
 363        _ => { false }
 364      }
 365  }

 367  pub fn missing_ctor(cx: &MatchCheckCtxt,
 368                      m: &matrix,
 369                      left_ty: ty::t)
 370                   -> Option<ctor> {
 371      match ty::get(left_ty).sty {
 372        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(*) | ty::ty_tup(_) |
 373        ty::ty_struct(*) => {
 374          for r in m.iter() {
 375              if !is_wild(cx, r[0]) { return None; }
 376          }
 377          return Some(single);
 378        }
 379        ty::ty_enum(eid, _) => {
 380          let mut found = ~[];
 381          for r in m.iter() {
 382              let r = pat_ctor_id(cx, r[0]);
 383              for id in r.iter() {
 384                  if !found.contains(id) {
 385                      found.push(*id);
 386                  }
 387              }
 388          }
 389          let variants = ty::enum_variants(cx.tcx, eid);
 390          if found.len() != (*variants).len() {
 391              for v in (*variants).iter() {
 392                  if !found.iter().any(|x| x == &(variant(v.id))) {
 393                      return Some(variant(v.id));
 394                  }
 395              }
 396              fail!();
 397          } else { None }
 398        }
 399        ty::ty_nil => None,
 400        ty::ty_bool => {
 401          let mut true_found = false;
 402          let mut false_found = false;
 403          for r in m.iter() {
 404              match pat_ctor_id(cx, r[0]) {
 405                None => (),
 406                Some(val(const_bool(true))) => true_found = true,
 407                Some(val(const_bool(false))) => false_found = true,
 408                _ => fail!("impossible case")
 409              }
 410          }
 411          if true_found && false_found { None }
 412          else if true_found { Some(val(const_bool(false))) }
 413          else { Some(val(const_bool(true))) }
 414        }
 415        ty::ty_evec(_, ty::vstore_fixed(n)) => {
 416          let mut missing = true;
 417          let mut wrong = false;
 418          for r in m.iter() {
 419            match r[0].node {
 420              pat_vec(ref before, ref slice, ref after) => {
 421                let count = before.len() + after.len();
 422                if (count < n && slice.is_none()) || count > n {
 423                  wrong = true;
 424                }
 425                if count == n || (count < n && slice.is_some()) {
 426                  missing = false;
 427                }
 428              }
 429              _ => {}
 430            }
 431          }
 432          match (wrong, missing) {
 433            (true, _) => Some(vec(n)), // should be compile-time error
 434            (_, true) => Some(vec(n)),
 435            _         => None
 436          }
 437        }
 438        ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 439  
 440          // Find the lengths and slices of all vector patterns.
 441          let vec_pat_lens = do m.iter().filter_map |r| {
 442              match r[0].node {
 443                  pat_vec(ref before, ref slice, ref after) => {
 444                      Some((before.len() + after.len(), slice.is_some()))
 445                  }
 446                  _ => None
 447              }
 448          }.collect::<~[(uint, bool)]>();
 449  
 450          // Sort them by length such that for patterns of the same length,
 451          // those with a destructured slice come first.
 452          let mut sorted_vec_lens = sort::merge_sort(vec_pat_lens,
 453              |&(len1, slice1), &(len2, slice2)| {
 454                  if len1 == len2 {
 455                      slice1 > slice2
 456                  } else {
 457                      len1 <= len2
 458                  }
 459              }
 460          );
 461          sorted_vec_lens.dedup();
 462  
 463          let mut found_slice = false;
 464          let mut next = 0;
 465          let mut missing = None;
 466          for &(length, slice) in sorted_vec_lens.iter() {
 467              if length != next {
 468                  missing = Some(next);
 469                  break;
 470              }
 471              if slice {
 472                  found_slice = true;
 473                  break;
 474              }
 475              next += 1;
 476          }
 477  
 478          // We found patterns of all lengths within <0, next), yet there was no
 479          // pattern with a slice - therefore, we report vec(next) as missing.
 480          if !found_slice {
 481              missing = Some(next);
 482          }
 483          match missing {
 484            Some(k) => Some(vec(k)),
 485            None => None
 486          }
 487        }
 488        _ => Some(single)
 489      }
 490  }

 492  pub fn ctor_arity(cx: &MatchCheckCtxt, ctor: &ctor, ty: ty::t) -> uint {
 493      match ty::get(ty).sty {
 494        ty::ty_tup(ref fs) => fs.len(),
 495        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(*) => 1u,
 496        ty::ty_enum(eid, _) => {
 497            let id = match *ctor { variant(id) => id,
 498            _ => fail!("impossible case") };
 499          match ty::enum_variants(cx.tcx, eid).iter().find(|v| v.id == id ) {
 500              Some(v) => v.args.len(),
 501              None => fail!("impossible case")
 502          }
 503        }
 504        ty::ty_struct(cid, _) => ty::lookup_struct_fields(cx.tcx, cid).len(),
 505        ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 506          match *ctor {
 507            vec(n) => n,
 508            _ => 0u
 509          }
 510        }
 511        _ => 0u
 512      }
 513  }

 519  pub fn specialize(cx: &MatchCheckCtxt,

 521                    ctor_id: &ctor,

 523                    left_ty: ty::t)

 529              match n {
 530              pat_wild => {
 531                  Some(vec::append(vec::from_elem(arity, wild()), r.tail()))
 532              }

 549                                      Some(val1) => (val1 == 0),
 550                                      None => {

 554                                      }
 555                                  }
 556                              },

 561                                      (Some(val1), Some(val2)) => {

 565                                          cx.tcx.sess.span_err(pat_span,
 566                                              "mismatched types between ranges");
 567                                          false
 568                                      }
 569                                  }
 570                              }
 571                              single => true,
 572                              _ => fail!("type error")
 573                          };
 574                          if match_ {
 575                              Some(r.tail().to_owned())
 576                          } else {
 577                              None
 578                          }
 579                      }
 580                      _ => {
 581                          Some(
 582                              vec::append(
 583                                  vec::from_elem(arity, wild()),
 584                                  r.tail()
 585                              )
 586                          )
 587                      }
 588                  }
 589              }

 597                              val(ref v) =>
 598                                  match compare_const_vals(&e_v, v) {
 599                                      Some(val1) => (val1 == 0),
 600                                      None => {
 601                                          cx.tcx.sess.span_err(pat_span,
 602                                              "mismatched types between arms");
 603                                          false
 604                                      }
 605                                  },
 606                              range(ref c_lo, ref c_hi) => {
 607                                  let m1 = compare_const_vals(c_lo, &e_v);
 608                                  let m2 = compare_const_vals(c_hi, &e_v);
 609                                  match (m1, m2) {
 610                                      (Some(val1), Some(val2)) => (val1 >= 0 && val2 <= 0),
 611                                      _ => {
 612                                          cx.tcx.sess.span_err(pat_span,
 613                                              "mismatched types between ranges");
 614                                          false
 615                                      }
 616                                  }
 617                              }
 618                              single => true,
 619                              _ => fail!("type error")
 620                          };
 621                          if match_ {
 622                              Some(r.tail().to_owned())
 623                          } else {
 624                              None
 625                          }
 626                      }
 627                      def_variant(_, id) if variant(id) == *ctor_id => {
 628                          let args = match args {
 629                              Some(args) => args,
 630                              None => vec::from_elem(arity, wild())
 631                          };
 632                          Some(vec::append(args, r.tail()))
 633                      }
 634                      def_variant(_, _) => None,
 635  
 636                      def_fn(*) |
 637                      def_struct(*) => {
 638                          // FIXME #4731: Is this right? --pcw
 639                          let new_args;
 640                          match args {
 641                              Some(args) => new_args = args,
 642                              None => new_args = vec::from_elem(arity, wild())
 643                          }
 644                          Some(vec::append(new_args, r.tail()))
 645                      }
 646                      _ => None
 647                  }
 648              }
 649              pat_struct(_, ref flds, _) => {
 650                  // Is this a struct or an enum variant?

 652                      def_variant(_, variant_id) => {

 656                                  match flds.iter().find(|f|
 657                                                  f.ident == ty_field.ident) {
 658                                      Some(f) => f.pat,
 659                                      _ => wild()
 660                                  }

 664                              None
 665                          }
 666                      }

 672                              ty::ty_struct(cid, _) => {
 673                                  class_id = cid;

 675                                      ty::lookup_struct_fields(cx.tcx,
 676                                                               class_id);
 677                              }
 678                              _ => {
 679                                  cx.tcx.sess.span_bug(
 680                                      pat_span,
 681                                      fmt!("struct pattern resolved to %s, \
 682                                            not a struct",
 683                                           ty_to_str(cx.tcx, left_ty)));
 684                              }
 685                          }

 687                              match flds.iter().find(|f|
 688                                              f.ident == class_field.ident) {
 689                                  Some(f) => f.pat,
 690                                  _ => wild()
 691                              }
 692                          }).collect();
 693                          Some(vec::append(args, r.tail()))
 694                      }
 695                  }
 696              }
 697              pat_tup(args) => Some(vec::append(args, r.tail())),
 698              pat_box(a) | pat_uniq(a) | pat_region(a) => {
 699                  Some(vec::append(~[a], r.tail()))
 700              }
 701              pat_lit(expr) => {
 702                  let e_v = eval_const_expr(cx.tcx, expr);
 703                  let match_ = match *ctor_id {
 704                      val(ref v) => {
 705                          match compare_const_vals(&e_v, v) {
 706                              Some(val1) => val1 == 0,
 707                              None => {
 708                                  cx.tcx.sess.span_err(pat_span,
 709                                      "mismatched types between arms");
 710                                  false
 711                              }
 712                          }
 713                      },
 714                      range(ref c_lo, ref c_hi) => {
 715                          let m1 = compare_const_vals(c_lo, &e_v);
 716                          let m2 = compare_const_vals(c_hi, &e_v);
 717                          match (m1, m2) {
 718                              (Some(val1), Some(val2)) => (val1 >= 0 && val2 <= 0),
 719                              _ => {
 720                                  cx.tcx.sess.span_err(pat_span,
 721                                      "mismatched types between ranges");
 722                                  false
 723                              }
 724                          }
 725                      }
 726                      single => true,
 727                      _ => fail!("type error")
 728                  };
 729                  if match_ { Some(r.tail().to_owned()) } else { None }
 730              }
 731              pat_range(lo, hi) => {
 732                  let (c_lo, c_hi) = match *ctor_id {
 733                      val(ref v) => (*v, *v),
 734                      range(ref lo, ref hi) => (*lo, *hi),
 735                      single => return Some(r.tail().to_owned()),
 736                      _ => fail!("type error")
 737                  };
 738                  let v_lo = eval_const_expr(cx.tcx, lo);
 739                  let v_hi = eval_const_expr(cx.tcx, hi);
 740  
 741                  let m1 = compare_const_vals(&c_lo, &v_lo);
 742                  let m2 = compare_const_vals(&c_hi, &v_hi);
 743                  match (m1, m2) {
 744                      (Some(val1), Some(val2)) if val1 >= 0 && val2 <= 0 => {
 745                          Some(r.tail().to_owned())
 746                      },
 747                      (Some(_), Some(_)) => None,
 748                      _ => {
 749                          cx.tcx.sess.span_err(pat_span,
 750                              "mismatched types between ranges");
 751                          None
 752                      }
 753                  }
 754              }
 755              pat_vec(before, slice, after) => {
 756                  match *ctor_id {
 757                      vec(_) => {
 758                          let num_elements = before.len() + after.len();

 760                              Some(vec::append(
 761                                  vec::concat(&[
 762                                      before,
 763                                      vec::from_elem(
 764                                          arity - num_elements, wild()),
 765                                      after
 766                                  ]),
 767                                  r.tail()
 768                              ))
 769                          } else if num_elements == arity {
 770                              Some(vec::append(
 771                                  vec::append(before, after),
 772                                  r.tail()
 773                              ))
 774                          } else {
 775                              None
 776                          }
 777                      }
 778                      _ => None
 779                  }
 780              }
 781          }
 782      }

 820  pub fn is_refutable(cx: &MatchCheckCtxt, pat: &pat) -> bool {
 821      match cx.tcx.def_map.find(&pat.id) {
 822        Some(&def_variant(enum_id, _)) => {
 823          if ty::enum_variants(cx.tcx, enum_id).len() != 1u {
 824              return true;
 825          }
 826        }
 827        Some(&def_static(*)) => return true,
 828        _ => ()
 829      }
 830  
 831      match pat.node {
 832        pat_box(sub) | pat_uniq(sub) | pat_region(sub) |
 833        pat_ident(_, _, Some(sub)) => {
 834          is_refutable(cx, sub)
 835        }
 836        pat_wild | pat_ident(_, _, None) => { false }
 837        pat_lit(@expr {node: expr_lit(@spanned { node: lit_nil, _}), _}) => {
 838          // "()"
 839          false
 840        }
 841        pat_lit(_) | pat_range(_, _) => { true }
 842        pat_struct(_, ref fields, _) => {
 843          fields.iter().any(|f| is_refutable(cx, f.pat))
 844        }
 845        pat_tup(ref elts) => {
 846          elts.iter().any(|elt| is_refutable(cx, *elt))
 847        }
 848        pat_enum(_, Some(ref args)) => {
 849          args.iter().any(|a| is_refutable(cx, *a))
 850        }
 851        pat_enum(_,_) => { false }
 852        pat_vec(*) => { true }
 853      }
 854  }

 856  // Legality of move bindings checking
 857  
 858  pub fn check_legality_of_move_bindings(cx: &MatchCheckCtxt,
 859                                         has_guard: bool,
 860                                         pats: &[@pat]) {
 861      let tcx = cx.tcx;
 862      let def_map = tcx.def_map;
 863      let mut by_ref_span = None;
 864      let mut any_by_move = false;
 865      for pat in pats.iter() {
 866          do pat_bindings(def_map, *pat) |bm, id, span, _path| {
 867              match bm {
 868                  bind_by_ref(_) => {
 869                      by_ref_span = Some(span);
 870                  }
 871                  bind_infer => {
 872                      if cx.moves_map.contains(&id) {
 873                          any_by_move = true;
 874                      }
 875                  }
 876              }
 877          }
 878      }
 879  
 880      let check_move: &fn(@pat, Option<@pat>) = |p, sub| {
 881          // check legality of moving out of the enum
 882          if sub.is_some() {
 883              tcx.sess.span_err(
 884                  p.span,
 885                  "cannot bind by-move with sub-bindings");
 886          } else if has_guard {
 887              tcx.sess.span_err(
 888                  p.span,
 889                  "cannot bind by-move into a pattern guard");
 890          } else if by_ref_span.is_some() {
 891              tcx.sess.span_err(
 892                  p.span,
 893                  "cannot bind by-move and by-ref \
 894                   in the same pattern");
 895              tcx.sess.span_note(
 896                  by_ref_span.unwrap(),
 897                  "by-ref binding occurs here");
 898          }
 899      };
 900  
 901      if !any_by_move { return; } // pointless micro-optimization
 902      for pat in pats.iter() {

  67  pub fn check_expr(v: &mut CheckMatchVisitor,
  68                    cx: @MatchCheckCtxt,
  69                    ex: @Expr,
  70                    s: ()) {
  71      visit::walk_expr(v, ex, s);
  72      match ex.node {
  73        ExprMatch(scrut, ref arms) => {
  74          // First, check legality of move bindings.
  75          for arm in arms.iter() {
  76              check_legality_of_move_bindings(cx,
  77                                              arm.guard.is_some(),
  78                                              arm.pats);
  79          }
  80  
  81          check_arms(cx, *arms);
  82          /* Check for exhaustiveness */
  83           // Check for empty enum, because is_useful only works on inhabited
  84           // types.
  85         let pat_ty = node_id_to_type(cx.tcx, scrut.id);
  86         if (*arms).is_empty() {
  87             if !type_is_empty(cx.tcx, pat_ty) {
  88                 // We know the type is inhabited, so this must be wrong
  89                 cx.tcx.sess.span_err(ex.span, fmt!("non-exhaustive patterns: \
  90                              type %s is non-empty",
  91                              ty_to_str(cx.tcx, pat_ty)));
  92             }
  93             // If the type *is* empty, it's vacuously exhaustive
  94             return;
  95         }
  96         match ty::get(pat_ty).sty {
  97            ty_enum(did, _) => {
  98                if (*enum_variants(cx.tcx, did)).is_empty() &&
  99                      (*arms).is_empty() {
 100  
 101                 return;
 102              }
 103            }
 104            _ => { /* We assume only enum types can be uninhabited */ }
 105         }
 106         let arms = arms.iter().filter_map(unguarded_pat).collect::<~[~[@Pat]]>().concat_vec();
 107         if arms.is_empty() {
 108             cx.tcx.sess.span_err(ex.span, "non-exhaustive patterns");
 109         } else {
 110             check_exhaustive(cx, ex.span, arms);
 111         }
 112       }
 113       _ => ()
 114      }
 115  }

 117  // Check for unreachable patterns
 118  pub fn check_arms(cx: &MatchCheckCtxt, arms: &[Arm]) {
 119      let mut seen = ~[];
 120      for arm in arms.iter() {
 121          for pat in arm.pats.iter() {
 122  
 123              // Check that we do not match against a static NaN (#6804)
 124              let pat_matches_nan: &fn(@Pat) -> bool = |p| {
 125                  match cx.tcx.def_map.find(&p.id) {
 126                      Some(&DefStatic(did, false)) => {
 127                          let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 128                          match eval_const_expr(cx.tcx, const_expr) {
 129                              const_float(f) if f.is_nan() => true,
 130                              _ => false
 131                          }
 132                      }
 133                      _ => false
 134                  }
 135              };
 136              do walk_pat(*pat) |p| {
 137                  if pat_matches_nan(p) {
 138                      cx.tcx.sess.span_warn(p.span, "unmatchable NaN in pattern, \
 139                                                     use the is_nan method in a guard instead");
 140                  }
 141                  true
 142              };
 143  
 144              let v = ~[*pat];
 145              match is_useful(cx, &seen, v) {
 146                not_useful => {
 147                  cx.tcx.sess.span_err(pat.span, "unreachable pattern");
 148                }
 149                _ => ()
 150              }
 151              if arm.guard.is_none() { seen.push(v); }
 152          }
 153      }
 154  }

 159        _ => { p }
 160      }
 161  }

 163  pub fn check_exhaustive(cx: &MatchCheckCtxt, sp: Span, pats: ~[@Pat]) {
 164      assert!((!pats.is_empty()));
 165      let ext = match is_useful(cx, &pats.map(|p| ~[*p]), [wild()]) {
 166          not_useful => {
 167              // This is good, wildcard pattern isn't reachable
 168              return;
 169          }
 170          useful_ => None,
 171          useful(ty, ref ctor) => {
 172              match ty::get(ty).sty {
 173                  ty::ty_bool => {
 174                      match (*ctor) {
 175                          val(const_bool(true)) => Some(@"true"),
 176                          val(const_bool(false)) => Some(@"false"),
 177                          _ => None
 178                      }
 179                  }
 180                  ty::ty_enum(id, _) => {
 181                      let vid = match *ctor {
 182                          variant(id) => id,
 183                          _ => fail!("check_exhaustive: non-variant ctor"),
 184                      };
 185                      let variants = ty::enum_variants(cx.tcx, id);
 186  
 187                      match variants.iter().find(|v| v.id == vid) {
 188                          Some(v) => Some(cx.tcx.sess.str_of(v.name)),
 189                          None => {
 190                              fail!("check_exhaustive: bad variant in ctor")
 191                          }
 192                      }
 193                  }
 194                  ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 195                      match *ctor {
 196                          vec(n) => Some(fmt!("vectors of length %u", n).to_managed()),
 197                          _ => None
 198                      }
 199                  }
 200                  _ => None
 201              }
 202          }
 203      };
 204      let msg = ~"non-exhaustive patterns" + match ext {
 205          Some(ref s) => fmt!(": %s not covered",  *s),
 206          None => ~""
 207      };
 208      cx.tcx.sess.span_err(sp, msg);
 209  }

 213  pub enum useful { useful(ty::t, ctor), useful_, not_useful }
 214  
 215  #[deriving(Eq)]
 216  pub enum ctor {
 217      single,
 218      variant(DefId),
 219      val(const_val),
 220      range(const_val, const_val),
 221      vec(uint)
 222  }
 223  
 224  // Algorithm from http://moscova.inria.fr/~maranget/papers/warn/index.html
 225  //
 226  // Whether a vector `v` of patterns is 'useful' in relation to a set of such
 227  // vectors `m` is defined as there being a set of inputs that will match `v`
 228  // but not any of the sets in `m`.
 229  //
 230  // This is used both for reachability checking (if a pattern isn't useful in
 231  // relation to preceding patterns, it is not reachable) and exhaustiveness
 232  // checking (if a wildcard pattern is useful in relation to a matrix, the
 233  // matrix isn't exhaustive).
 234  
 235  // Note: is_useful doesn't work on empty types, as the paper notes.
 236  // So it assumes that v is non-empty.
 237  pub fn is_useful(cx: &MatchCheckCtxt, m: &matrix, v: &[@Pat]) -> useful {
 238      if m.len() == 0u { return useful_; }
 239      if m[0].len() == 0u { return not_useful; }
 240      let real_pat = match m.iter().find(|r| r[0].id != 0) {
 241        Some(r) => r[0], None => v[0]
 242      };
 243      let left_ty = if real_pat.id == 0 { ty::mk_nil() }
 244                    else { ty::node_id_to_type(cx.tcx, real_pat.id) };
 245  
 246      match pat_ctor_id(cx, v[0]) {
 247        None => {
 248          match missing_ctor(cx, m, left_ty) {
 249            None => {
 250              match ty::get(left_ty).sty {
 251                ty::ty_bool => {
 252                  match is_useful_specialized(cx, m, v,
 253                                              val(const_bool(true)),
 254                                              0u, left_ty){
 255                    not_useful => {
 256                      is_useful_specialized(cx, m, v,
 257                                            val(const_bool(false)),
 258                                            0u, left_ty)
 259                    }
 260                    ref u => *u,
 261                  }
 262                }
 263                ty::ty_enum(eid, _) => {
 264                  for va in (*ty::enum_variants(cx.tcx, eid)).iter() {
 265                      match is_useful_specialized(cx, m, v, variant(va.id),
 266                                                  va.args.len(), left_ty) {
 267                        not_useful => (),
 268                        ref u => return *u,
 269                      }
 270                  }
 271                  not_useful
 272                }
 273                ty::ty_evec(_, ty::vstore_fixed(n)) => {
 274                  is_useful_specialized(cx, m, v, vec(n), n, left_ty)
 275                }
 276                ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 277                  let max_len = do m.rev_iter().fold(0) |max_len, r| {
 278                    match r[0].node {
 279                      PatVec(ref before, _, ref after) => {
 280                        num::max(before.len() + after.len(), max_len)
 281                      }
 282                      _ => max_len
 283                    }
 284                  };
 285                  for n in iter::range(0u, max_len + 1) {
 286                    match is_useful_specialized(cx, m, v, vec(n), n, left_ty) {
 287                      not_useful => (),
 288                      ref u => return *u,
 289                    }
 290                  }
 291                  not_useful
 292                }
 293                _ => {
 294                  let arity = ctor_arity(cx, &single, left_ty);
 295                  is_useful_specialized(cx, m, v, single, arity, left_ty)
 296                }
 297              }
 298            }
 299            Some(ref ctor) => {
 300              match is_useful(cx,
 301                              &m.iter().filter_map(|r| default(cx, *r)).collect::<matrix>(),
 302                              v.tail()) {
 303                useful_ => useful(left_ty, *ctor),
 304                ref u => *u,
 305              }
 306            }
 307          }
 308        }
 309        Some(ref v0_ctor) => {
 310          let arity = ctor_arity(cx, v0_ctor, left_ty);
 311          is_useful_specialized(cx, m, v, *v0_ctor, arity, left_ty)
 312        }
 313      }
 314  }

 316  pub fn is_useful_specialized(cx: &MatchCheckCtxt,
 317                               m: &matrix,
 318                               v: &[@Pat],
 319                               ctor: ctor,
 320                               arity: uint,
 321                               lty: ty::t)
 322                            -> useful {
 323      let ms = m.iter().filter_map(|r| specialize(cx, *r, &ctor, arity, lty)).collect::<matrix>();
 324      let could_be_useful = is_useful(
 325          cx, &ms, specialize(cx, v, &ctor, arity, lty).unwrap());
 326      match could_be_useful {
 327        useful_ => useful(lty, ctor),
 328        ref u => *u,
 329      }
 330  }

 332  pub fn pat_ctor_id(cx: &MatchCheckCtxt, p: @Pat) -> Option<ctor> {
 333      let pat = raw_pat(p);
 334      match pat.node {
 335        PatWild => { None }
 336        PatIdent(_, _, _) | PatEnum(_, _) => {
 337          match cx.tcx.def_map.find(&pat.id) {
 338            Some(&DefVariant(_, id, _)) => Some(variant(id)),
 339            Some(&DefStatic(did, false)) => {
 340              let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 341              Some(val(eval_const_expr(cx.tcx, const_expr)))
 342            }
 343            _ => None
 344          }
 345        }
 346        PatLit(expr) => { Some(val(eval_const_expr(cx.tcx, expr))) }
 347        PatRange(lo, hi) => {
 348          Some(range(eval_const_expr(cx.tcx, lo), eval_const_expr(cx.tcx, hi)))
 349        }
 350        PatStruct(*) => {
 351          match cx.tcx.def_map.find(&pat.id) {
 352            Some(&DefVariant(_, id, _)) => Some(variant(id)),
 353            _ => Some(single)
 354          }
 355        }
 356        PatBox(_) | PatUniq(_) | PatTup(_) | PatRegion(*) => {
 357          Some(single)
 358        }
 359        PatVec(ref before, slice, ref after) => {
 360          match slice {
 361            Some(_) => None,
 362            None => Some(vec(before.len() + after.len()))
 363          }
 364        }
 365      }
 366  }

 368  pub fn is_wild(cx: &MatchCheckCtxt, p: @Pat) -> bool {
 369      let pat = raw_pat(p);
 370      match pat.node {
 371        PatWild => { true }
 372        PatIdent(_, _, _) => {
 373          match cx.tcx.def_map.find(&pat.id) {
 374            Some(&DefVariant(_, _, _)) | Some(&DefStatic(*)) => { false }
 375            _ => { true }
 376          }
 377        }
 378        _ => { false }
 379      }
 380  }

 382  pub fn missing_ctor(cx: &MatchCheckCtxt,
 383                      m: &matrix,
 384                      left_ty: ty::t)
 385                   -> Option<ctor> {
 386      match ty::get(left_ty).sty {
 387        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(*) | ty::ty_tup(_) |
 388        ty::ty_struct(*) => {
 389          for r in m.iter() {
 390              if !is_wild(cx, r[0]) { return None; }
 391          }
 392          return Some(single);
 393        }
 394        ty::ty_enum(eid, _) => {
 395          let mut found = ~[];
 396          for r in m.iter() {
 397              let r = pat_ctor_id(cx, r[0]);
 398              for id in r.iter() {
 399                  if !found.contains(id) {
 400                      found.push(*id);
 401                  }
 402              }
 403          }
 404          let variants = ty::enum_variants(cx.tcx, eid);
 405          if found.len() != (*variants).len() {
 406              for v in (*variants).iter() {
 407                  if !found.iter().any(|x| x == &(variant(v.id))) {
 408                      return Some(variant(v.id));
 409                  }
 410              }
 411              fail!();
 412          } else { None }
 413        }
 414        ty::ty_nil => None,
 415        ty::ty_bool => {
 416          let mut true_found = false;
 417          let mut false_found = false;
 418          for r in m.iter() {
 419              match pat_ctor_id(cx, r[0]) {
 420                None => (),
 421                Some(val(const_bool(true))) => true_found = true,
 422                Some(val(const_bool(false))) => false_found = true,
 423                _ => fail!("impossible case")
 424              }
 425          }
 426          if true_found && false_found { None }
 427          else if true_found { Some(val(const_bool(false))) }
 428          else { Some(val(const_bool(true))) }
 429        }
 430        ty::ty_evec(_, ty::vstore_fixed(n)) => {
 431          let mut missing = true;
 432          let mut wrong = false;
 433          for r in m.iter() {
 434            match r[0].node {
 435              PatVec(ref before, ref slice, ref after) => {
 436                let count = before.len() + after.len();
 437                if (count < n && slice.is_none()) || count > n {
 438                  wrong = true;
 439                }
 440                if count == n || (count < n && slice.is_some()) {
 441                  missing = false;
 442                }
 443              }
 444              _ => {}
 445            }
 446          }
 447          match (wrong, missing) {
 448            (true, _) => Some(vec(n)), // should be compile-time error
 449            (_, true) => Some(vec(n)),
 450            _         => None
 451          }
 452        }
 453        ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 454  
 455          // Find the lengths and slices of all vector patterns.
 456          let vec_pat_lens = do m.iter().filter_map |r| {
 457              match r[0].node {
 458                  PatVec(ref before, ref slice, ref after) => {
 459                      Some((before.len() + after.len(), slice.is_some()))
 460                  }
 461                  _ => None
 462              }
 463          }.collect::<~[(uint, bool)]>();
 464  
 465          // Sort them by length such that for patterns of the same length,
 466          // those with a destructured slice come first.
 467          let mut sorted_vec_lens = sort::merge_sort(vec_pat_lens,
 468              |&(len1, slice1), &(len2, slice2)| {
 469                  if len1 == len2 {
 470                      slice1 > slice2
 471                  } else {
 472                      len1 <= len2
 473                  }
 474              }
 475          );
 476          sorted_vec_lens.dedup();
 477  
 478          let mut found_slice = false;
 479          let mut next = 0;
 480          let mut missing = None;
 481          for &(length, slice) in sorted_vec_lens.iter() {
 482              if length != next {
 483                  missing = Some(next);
 484                  break;
 485              }
 486              if slice {
 487                  found_slice = true;
 488                  break;
 489              }
 490              next += 1;
 491          }
 492  
 493          // We found patterns of all lengths within <0, next), yet there was no
 494          // pattern with a slice - therefore, we report vec(next) as missing.
 495          if !found_slice {
 496              missing = Some(next);
 497          }
 498          match missing {
 499            Some(k) => Some(vec(k)),
 500            None => None
 501          }
 502        }
 503        _ => Some(single)
 504      }
 505  }

 507  pub fn ctor_arity(cx: &MatchCheckCtxt, ctor: &ctor, ty: ty::t) -> uint {
 508      match ty::get(ty).sty {
 509        ty::ty_tup(ref fs) => fs.len(),
 510        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(*) => 1u,
 511        ty::ty_enum(eid, _) => {
 512            let id = match *ctor { variant(id) => id,
 513            _ => fail!("impossible case") };
 514          match ty::enum_variants(cx.tcx, eid).iter().find(|v| v.id == id ) {
 515              Some(v) => v.args.len(),
 516              None => fail!("impossible case")
 517          }
 518        }
 519        ty::ty_struct(cid, _) => ty::lookup_struct_fields(cx.tcx, cid).len(),
 520        ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 521          match *ctor {
 522            vec(n) => n,
 523            _ => 0u
 524          }
 525        }
 526        _ => 0u
 527      }
 528  }

 734                              _ => {

 796          }

 836  pub fn is_refutable(cx: &MatchCheckCtxt, pat: &Pat) -> bool {
 837      match cx.tcx.def_map.find(&pat.id) {
 838        Some(&DefVariant(enum_id, _, _)) => {
 839          if ty::enum_variants(cx.tcx, enum_id).len() != 1u {
 840              return true;
 841          }
 842        }
 843        Some(&DefStatic(*)) => return true,
 844        _ => ()
 845      }
 846  
 847      match pat.node {
 848        PatBox(sub) | PatUniq(sub) | PatRegion(sub) |
 849        PatIdent(_, _, Some(sub)) => {
 850          is_refutable(cx, sub)
 851        }
 852        PatWild | PatIdent(_, _, None) => { false }
 853        PatLit(@Expr {node: ExprLit(@Spanned { node: lit_nil, _}), _}) => {
 854          // "()"
 855          false
 856        }
 857        PatLit(_) | PatRange(_, _) => { true }

 859          fields.iter().any(|f| is_refutable(cx, f.pat))
 860        }
 861        PatTup(ref elts) => {
 862          elts.iter().any(|elt| is_refutable(cx, *elt))
 863        }
 864        PatEnum(_, Some(ref args)) => {
 865          args.iter().any(|a| is_refutable(cx, *a))
 866        }
 867        PatEnum(_,_) => { false }
 868        PatVec(*) => { true }
 869      }
 870  }

 874  pub fn check_legality_of_move_bindings(cx: &MatchCheckCtxt,
 875                                         has_guard: bool,
 876                                         pats: &[@Pat]) {
 877      let tcx = cx.tcx;
 878      let def_map = tcx.def_map;
 879      let mut by_ref_span = None;
 880      let mut any_by_move = false;
 881      for pat in pats.iter() {
 882          do pat_bindings(def_map, *pat) |bm, id, span, _path| {
 883              match bm {
 884                  BindByRef(_) => {
 885                      by_ref_span = Some(span);
 886                  }
 887                  BindInfer => {
 888                      if cx.moves_map.contains(&id) {
 889                          any_by_move = true;
 890                      }
 891                  }
 892              }
 893          }
 894      }
 895  
 896      let check_move: &fn(@Pat, Option<@Pat>) = |p, sub| {
 897          // check legality of moving out of the enum
 898  
 899          // x @ Foo(*) is legal, but x @ Foo(y) isn't.
 900          if sub.map_move_default(false, |p| pat_contains_bindings(def_map, p)) {
 901              tcx.sess.span_err(
 902                  p.span,
 903                  "cannot bind by-move with sub-bindings");
 904          } else if has_guard {
 905              tcx.sess.span_err(
 906                  p.span,
 907                  "cannot bind by-move into a pattern guard");
 908          } else if by_ref_span.is_some() {

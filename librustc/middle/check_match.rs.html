  53  pub fn check_expr(cx: @MatchCheckCtxt,
  54                    ex: @expr,
  55                    (s, v): ((), oldvisit::vt<()>)) {
  56      oldvisit::visit_expr(ex, (s, v));
  57      match ex.node {
  58        expr_match(scrut, ref arms) => {
  59          // First, check legality of move bindings.
  60          for arm in arms.iter() {
  61              check_legality_of_move_bindings(cx,
  62                                              arm.guard.is_some(),
  63                                              arm.pats);
  64          }
  65  
  66          check_arms(cx, *arms);
  67          /* Check for exhaustiveness */
  68           // Check for empty enum, because is_useful only works on inhabited
  69           // types.
  70         let pat_ty = node_id_to_type(cx.tcx, scrut.id);
  71         if (*arms).is_empty() {
  72             if !type_is_empty(cx.tcx, pat_ty) {
  73                 // We know the type is inhabited, so this must be wrong
  74                 cx.tcx.sess.span_err(ex.span, fmt!("non-exhaustive patterns: \
  75                              type %s is non-empty",
  76                              ty_to_str(cx.tcx, pat_ty)));
  77             }
  78             // If the type *is* empty, it's vacuously exhaustive
  79             return;
  80         }
  81         match ty::get(pat_ty).sty {
  82            ty_enum(did, _) => {
  83                if (*enum_variants(cx.tcx, did)).is_empty() &&
  84                      (*arms).is_empty() {
  85  
  86                 return;
  87              }
  88            }
  89            _ => { /* We assume only enum types can be uninhabited */ }
  90         }
  91         let arms = arms.iter().filter_map(unguarded_pat).collect::<~[~[@pat]]>().concat_vec();
  92         if arms.is_empty() {
  93             cx.tcx.sess.span_err(ex.span, "non-exhaustive patterns");
  94         } else {
  95             check_exhaustive(cx, ex.span, arms);
  96         }
  97       }
  98       _ => ()
  99      }
 100  }

 102  // Check for unreachable patterns
 103  pub fn check_arms(cx: &MatchCheckCtxt, arms: &[arm]) {
 104      let mut seen = ~[];
 105      for arm in arms.iter() {
 106          for pat in arm.pats.iter() {
 107  
 108              // Check that we do not match against a static NaN (#6804)
 109              let pat_matches_nan: &fn(@pat) -> bool = |p| {
 110                  match cx.tcx.def_map.find(&p.id) {
 111                      Some(&def_static(did, false)) => {
 112                          let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 113                          match eval_const_expr(cx.tcx, const_expr) {
 114                              const_float(f) if f.is_NaN() => true,
 115                              _ => false
 116                          }
 117                      }
 118                      _ => false
 119                  }
 120              };
 121              do walk_pat(*pat) |p| {
 122                  if pat_matches_nan(p) {
 123                      cx.tcx.sess.span_warn(p.span, "unmatchable NaN in pattern, \
 124                                                     use the is_NaN method in a guard instead");
 125                  }
 126                  true
 127              };
 128  
 129              let v = ~[*pat];
 130              match is_useful(cx, &seen, v) {
 131                not_useful => {
 132                  cx.tcx.sess.span_err(pat.span, "unreachable pattern");
 133                }
 134                _ => ()
 135              }
 136              if arm.guard.is_none() { seen.push(v); }
 137          }
 138      }
 139  }

 145      }
 146  }

 148  pub fn check_exhaustive(cx: &MatchCheckCtxt, sp: span, pats: ~[@pat]) {
 149      assert!((!pats.is_empty()));
 150      let ext = match is_useful(cx, &pats.map(|p| ~[*p]), [wild()]) {
 151          not_useful => {
 152              // This is good, wildcard pattern isn't reachable
 153              return;
 154          }
 155          useful_ => None,
 156          useful(ty, ref ctor) => {
 157              match ty::get(ty).sty {
 158                  ty::ty_bool => {
 159                      match (*ctor) {
 160                          val(const_bool(true)) => Some(@"true"),
 161                          val(const_bool(false)) => Some(@"false"),
 162                          _ => None
 163                      }
 164                  }
 165                  ty::ty_enum(id, _) => {
 166                      let vid = match *ctor {
 167                          variant(id) => id,
 168                          _ => fail!("check_exhaustive: non-variant ctor"),
 169                      };
 170                      let variants = ty::enum_variants(cx.tcx, id);
 171  
 172                      match variants.iter().find(|v| v.id == vid) {
 173                          Some(v) => Some(cx.tcx.sess.str_of(v.name)),
 174                          None => {
 175                              fail!("check_exhaustive: bad variant in ctor")
 176                          }
 177                      }
 178                  }
 179                  ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 180                      match *ctor {
 181                          vec(n) => Some(fmt!("vectors of length %u", n).to_managed()),
 182                          _ => None
 183                      }
 184                  }
 185                  _ => None
 186              }
 187          }
 188      };
 189      let msg = ~"non-exhaustive patterns" + match ext {
 190          Some(ref s) => fmt!(": %s not covered",  *s),
 191          None => ~""
 192      };
 193      cx.tcx.sess.span_err(sp, msg);
 194  }

 198  pub enum useful { useful(ty::t, ctor), useful_, not_useful }
 199  
 200  #[deriving(Eq)]
 201  pub enum ctor {
 202      single,
 203      variant(def_id),
 204      val(const_val),
 205      range(const_val, const_val),
 206      vec(uint)
 207  }

 209  // Algorithm from http://moscova.inria.fr/~maranget/papers/warn/index.html
 210  //
 211  // Whether a vector `v` of patterns is 'useful' in relation to a set of such
 212  // vectors `m` is defined as there being a set of inputs that will match `v`
 213  // but not any of the sets in `m`.
 214  //
 215  // This is used both for reachability checking (if a pattern isn't useful in
 216  // relation to preceding patterns, it is not reachable) and exhaustiveness
 217  // checking (if a wildcard pattern is useful in relation to a matrix, the
 218  // matrix isn't exhaustive).
 219  
 220  // Note: is_useful doesn't work on empty types, as the paper notes.
 221  // So it assumes that v is non-empty.
 222  pub fn is_useful(cx: &MatchCheckCtxt, m: &matrix, v: &[@pat]) -> useful {
 223      if m.len() == 0u { return useful_; }
 224      if m[0].len() == 0u { return not_useful; }
 225      let real_pat = match m.iter().find(|r| r[0].id != 0) {
 226        Some(r) => r[0], None => v[0]
 227      };
 228      let left_ty = if real_pat.id == 0 { ty::mk_nil() }
 229                    else { ty::node_id_to_type(cx.tcx, real_pat.id) };
 230  
 231      match pat_ctor_id(cx, v[0]) {
 232        None => {
 233          match missing_ctor(cx, m, left_ty) {
 234            None => {
 235              match ty::get(left_ty).sty {
 236                ty::ty_bool => {
 237                  match is_useful_specialized(cx, m, v,
 238                                              val(const_bool(true)),
 239                                              0u, left_ty){
 240                    not_useful => {
 241                      is_useful_specialized(cx, m, v,
 242                                            val(const_bool(false)),
 243                                            0u, left_ty)
 244                    }
 245                    ref u => *u,
 246                  }
 247                }
 248                ty::ty_enum(eid, _) => {
 249                  for va in (*ty::enum_variants(cx.tcx, eid)).iter() {
 250                      match is_useful_specialized(cx, m, v, variant(va.id),
 251                                                  va.args.len(), left_ty) {
 252                        not_useful => (),
 253                        ref u => return *u,
 254                      }
 255                  }
 256                  not_useful
 257                }
 258                ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 259                  let max_len = do m.rev_iter().fold(0) |max_len, r| {
 260                    match r[0].node {
 261                      pat_vec(ref before, _, ref after) => {
 262                        num::max(before.len() + after.len(), max_len)
 263                      }
 264                      _ => max_len
 265                    }
 266                  };
 267                  for n in iterator::range(0u, max_len + 1) {
 268                    match is_useful_specialized(cx, m, v, vec(n), n, left_ty) {
 269                      not_useful => (),
 270                      ref u => return *u,
 271                    }
 272                  }
 273                  not_useful
 274                }
 275                _ => {
 276                  let arity = ctor_arity(cx, &single, left_ty);
 277                  is_useful_specialized(cx, m, v, single, arity, left_ty)
 278                }
 279              }
 280            }
 281            Some(ref ctor) => {
 282              match is_useful(cx,
 283                              &m.iter().filter_map(|r| default(cx, *r)).collect::<matrix>(),
 284                              v.tail()) {
 285                useful_ => useful(left_ty, *ctor),
 286                ref u => *u,
 287              }
 288            }
 289          }
 290        }
 291        Some(ref v0_ctor) => {
 292          let arity = ctor_arity(cx, v0_ctor, left_ty);
 293          is_useful_specialized(cx, m, v, *v0_ctor, arity, left_ty)
 294        }
 295      }
 296  }

 298  pub fn is_useful_specialized(cx: &MatchCheckCtxt,
 299                               m: &matrix,
 300                               v: &[@pat],
 301                               ctor: ctor,
 302                               arity: uint,
 303                               lty: ty::t)
 304                            -> useful {
 305      let ms = m.iter().filter_map(|r| specialize(cx, *r, &ctor, arity, lty)).collect::<matrix>();
 306      let could_be_useful = is_useful(
 307          cx, &ms, specialize(cx, v, &ctor, arity, lty).unwrap());
 308      match could_be_useful {
 309        useful_ => useful(lty, ctor),
 310        ref u => *u,
 311      }
 312  }

 314  pub fn pat_ctor_id(cx: &MatchCheckCtxt, p: @pat) -> Option<ctor> {
 315      let pat = raw_pat(p);
 316      match pat.node {
 317        pat_wild => { None }
 318        pat_ident(_, _, _) | pat_enum(_, _) => {
 319          match cx.tcx.def_map.find(&pat.id) {
 320            Some(&def_variant(_, id)) => Some(variant(id)),
 321            Some(&def_static(did, false)) => {
 322              let const_expr = lookup_const_by_id(cx.tcx, did).unwrap();
 323              Some(val(eval_const_expr(cx.tcx, const_expr)))
 324            }
 325            _ => None
 326          }
 327        }
 328        pat_lit(expr) => { Some(val(eval_const_expr(cx.tcx, expr))) }
 329        pat_range(lo, hi) => {
 330          Some(range(eval_const_expr(cx.tcx, lo), eval_const_expr(cx.tcx, hi)))
 331        }
 332        pat_struct(*) => {
 333          match cx.tcx.def_map.find(&pat.id) {
 334            Some(&def_variant(_, id)) => Some(variant(id)),
 335            _ => Some(single)
 336          }
 337        }
 338        pat_box(_) | pat_uniq(_) | pat_tup(_) | pat_region(*) => {
 339          Some(single)
 340        }
 341        pat_vec(ref before, slice, ref after) => {
 342          match slice {
 343            Some(_) => None,
 344            None => Some(vec(before.len() + after.len()))
 345          }
 346        }
 347      }
 348  }

 350  pub fn is_wild(cx: &MatchCheckCtxt, p: @pat) -> bool {
 351      let pat = raw_pat(p);
 352      match pat.node {
 353        pat_wild => { true }
 354        pat_ident(_, _, _) => {
 355          match cx.tcx.def_map.find(&pat.id) {
 356            Some(&def_variant(_, _)) | Some(&def_static(*)) => { false }
 357            _ => { true }
 358          }
 359        }
 360        _ => { false }
 361      }
 362  }

 364  pub fn missing_ctor(cx: &MatchCheckCtxt,
 365                      m: &matrix,
 366                      left_ty: ty::t)
 367                   -> Option<ctor> {
 368      match ty::get(left_ty).sty {
 369        ty::ty_box(_) | ty::ty_uniq(_) | ty::ty_rptr(*) | ty::ty_tup(_) |
 370        ty::ty_struct(*) => {
 371          for r in m.iter() {
 372              if !is_wild(cx, r[0]) { return None; }
 373          }
 374          return Some(single);
 375        }
 376        ty::ty_enum(eid, _) => {
 377          let mut found = ~[];
 378          for r in m.iter() {
 379              let r = pat_ctor_id(cx, r[0]);
 380              for id in r.iter() {
 381                  if !found.contains(id) {
 382                      found.push(*id);
 383                  }
 384              }
 385          }
 386          let variants = ty::enum_variants(cx.tcx, eid);
 387          if found.len() != (*variants).len() {
 388              for v in (*variants).iter() {
 389                  if !found.iter().any(|x| x == &(variant(v.id))) {
 390                      return Some(variant(v.id));
 391                  }
 392              }
 393              fail!();
 394          } else { None }
 395        }
 396        ty::ty_nil => None,
 397        ty::ty_bool => {
 398          let mut true_found = false;
 399          let mut false_found = false;
 400          for r in m.iter() {
 401              match pat_ctor_id(cx, r[0]) {
 402                None => (),
 403                Some(val(const_bool(true))) => true_found = true,
 404                Some(val(const_bool(false))) => false_found = true,
 405                _ => fail!("impossible case")
 406              }
 407          }
 408          if true_found && false_found { None }
 409          else if true_found { Some(val(const_bool(false))) }
 410          else { Some(val(const_bool(true))) }
 411        }
 412        ty::ty_unboxed_vec(*) | ty::ty_evec(*) => {
 413  
 414          // Find the lengths and slices of all vector patterns.
 415          let vec_pat_lens = do m.iter().filter_map |r| {
 416              match r[0].node {
 417                  pat_vec(ref before, ref slice, ref after) => {
 418                      Some((before.len() + after.len(), slice.is_some()))
 419                  }
 420                  _ => None
 421              }
 422          }.collect::<~[(uint, bool)]>();
 423  
 424          // Sort them by length such that for patterns of the same length,
 425          // those with a destructured slice come first.
 426          let mut sorted_vec_lens = sort::merge_sort(vec_pat_lens,
 427              |&(len1, slice1), &(len2, slice2)| {
 428                  if len1 == len2 {
 429                      slice1 > slice2
 430                  } else {
 431                      len1 <= len2
 432                  }
 433              }
 434          );
 435          sorted_vec_lens.dedup();
 436  
 437          let mut found_slice = false;
 438          let mut next = 0;

   11  //! A 'lint' check is a kind of miscellaneous constraint that a user _might_
   12  //! want to enforce, but might reasonably want to permit as well, on a
   13  //! module-by-module basis. They contrast with static constraints enforced by
   14  //! other phases of the compiler, which are generally required to hold in order
   15  //! to compile the program at all.
   16  //!
   17  //! The lint checking is all consolidated into one pass which runs just before
   18  //! translation to LLVM bytecode. Throughout compilation, lint warnings can be
   19  //! added via the `add_lint` method on the Session structure. This requires a
   20  //! span and an id of the node that the lint is being added to. The lint isn't
   21  //! actually emitted at that time because it is unknown what the actual lint
   22  //! level at that location is.
   23  //!
   24  //! To actually emit lint warnings/errors, a separate pass is used just before
   25  //! translation. A context keeps track of the current state of all lint levels.
   26  //! Upon entering a node of the ast which can modify the lint settings, the
   27  //! previous lint state is pushed onto a stack and the ast is then recursed
   28  //! upon.  As the ast is traversed, this keeps track of the current lint level
   29  //! for all lint attributes.
   30  //!
   31  //! To add a new lint warning, all you need to do is to either invoke `add_lint`
   32  //! on the session at the appropriate time, or write a few linting functions and
   33  //! modify the Context visitor appropriately. If you're adding lints from the
   34  //! Context itself, span_lint should be used instead of add_lint.

  450  impl<'a> Context<'a> {
  451      fn get_level(&self, lint: Lint) -> level {
  452          match self.cur.find(&(lint as uint)) {
  453            Some(&(lvl, _)) => lvl,
  454            None => allow
  455          }
  456      }
  457  
  458      fn get_source(&self, lint: Lint) -> LintSource {
  459          match self.cur.find(&(lint as uint)) {
  460            Some(&(_, src)) => src,
  461            None => Default
  462          }
  463      }
  464  
  465      fn set_level(&mut self, lint: Lint, level: level, src: LintSource) {
  466          if level == allow {
  467              self.cur.remove(&(lint as uint));
  468          } else {
  469              self.cur.insert(lint as uint, (level, src));
  470          }
  471      }
  472  
  473      fn lint_to_str(&self, lint: Lint) -> &'static str {
  474          for (k, v) in self.dict.iter() {
  475              if v.lint == lint {
  476                  return *k;
  477              }
  478          }
  479          fail!("unregistered lint {:?}", lint);
  480      }
  481  
  482      fn span_lint(&self, lint: Lint, span: Span, msg: &str) {
  483          let (level, src) = match self.cur.find(&(lint as uint)) {
  484              None => { return }
  485              Some(&(warn, src)) => (self.get_level(Warnings), src),
  486              Some(&pair) => pair,
  487          };
  488          if level == allow { return }
  489  
  490          let mut note = None;
  491          let msg = match src {
  492              Default => {
  493                  format!("{}, \\#[{}({})] on by default", msg,
  494                      level_to_str(level), self.lint_to_str(lint))
  495              },
  496              CommandLine => {
  497                  format!("{} [-{} {}]", msg,
  498                      match level {
  499                          warn => 'W', deny => 'D', forbid => 'F',
  500                          allow => fail!()
  501                      }, self.lint_to_str(lint).replace("_", "-"))
  502              },
  503              Node(src) => {
  504                  note = Some(src);
  505                  msg.to_str()
  506              }
  507          };
  508          match level {
  509              warn =>          { self.tcx.sess.span_warn(span, msg); }
  510              deny | forbid => { self.tcx.sess.span_err(span, msg);  }
  511              allow => fail!(),
  512          }
  513  
  514          for &span in note.iter() {
  515              self.tcx.sess.span_note(span, "lint level defined here");
  516          }
  517      }

  519      /**
  520       * Merge the lints specified by any lint attributes into the
  521       * current lint context, call the provided function, then reset the
  522       * lints in effect to their previous state.
  523       */
  524      fn with_lint_attrs(&mut self,
  525                         attrs: &[ast::Attribute],
  526                         f: |&mut Context|) {
  527          // Parse all of the lint attributes, and then add them all to the
  528          // current dictionary of lint information. Along the way, keep a history
  529          // of what we changed so we can roll everything back after invoking the
  530          // specified closure
  531          let mut pushed = 0u;
  532          each_lint(&self.tcx.sess, attrs, |meta, level, lintname| {
  533              match self.dict.find_equiv(&lintname) {
  534                  None => {
  535                      self.span_lint(
  536                          UnrecognizedLint,
  537                          meta.span,
  538                          format!("unknown `{}` attribute: `{}`",
  539                          level_to_str(level), lintname));
  540                  }
  541                  Some(lint) => {
  542                      let lint = lint.lint;
  543                      let now = self.get_level(lint);
  544                      if now == forbid && level != forbid {
  545                          self.tcx.sess.span_err(meta.span,
  546                          format!("{}({}) overruled by outer forbid({})",
  547                          level_to_str(level),
  548                          lintname, lintname));
  549                      } else if now != level {
  550                          let src = self.get_source(lint);
  551                          self.lint_stack.push((lint, now, src));
  552                          pushed += 1;
  553                          self.set_level(lint, level, Node(meta.span));
  554                      }
  555                  }
  556              }
  557              true
  558          });
  559  
  560          let old_is_doc_hidden = self.is_doc_hidden;
  561          self.is_doc_hidden =
  562              self.is_doc_hidden ||
  563              attrs.iter()
  564                   .any(|attr| {
  565                       attr.name().equiv(&("doc")) &&
  566                       match attr.meta_item_list() {
  567                           None => false,
  568                           Some(l) => {
  569                               attr::contains_name(l.as_slice(), "hidden")
  570                           }
  571                       }
  572                   });
  573  
  574          f(self);
  575  
  576          // rollback
  577          self.is_doc_hidden = old_is_doc_hidden;
  578          for _ in range(0, pushed) {
  579              let (lint, lvl, src) = self.lint_stack.pop().unwrap();
  580              self.set_level(lint, lvl, src);
  581          }
  582      }

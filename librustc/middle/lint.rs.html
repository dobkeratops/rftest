   38  /**
   39   * A 'lint' check is a kind of miscellaneous constraint that a user _might_
   40   * want to enforce, but might reasonably want to permit as well, on a
   41   * module-by-module basis. They contrast with static constraints enforced by
   42   * other phases of the compiler, which are generally required to hold in order
   43   * to compile the program at all.
   44   *
   45   * The lint checking is all consolidated into one pass which runs just before
   46   * translation to LLVM bytecode. Throughout compilation, lint warnings can be
   47   * added via the `add_lint` method on the Session structure. This requires a
   48   * span and an id of the node that the lint is being added to. The lint isn't
   49   * actually emitted at that time because it is unknown what the actual lint
   50   * level at that location is.
   51   *
   52   * To actually emit lint warnings/errors, a separate pass is used just before
   53   * translation. A context keeps track of the current state of all lint levels.
   54   * Upon entering a node of the ast which can modify the lint settings, the
   55   * previous lint state is pushed onto a stack and the ast is then recursed upon.
   56   * As the ast is traversed, this keeps track of the current lint level for all
   57   * lint attributes.
   58   *
   59   * At each node of the ast which can modify lint attributes, all known lint
   60   * passes are also applied.  Each lint pass is a visit::Visitor implementator.
   61   * The visitors are constructed via the lint_*() functions below. There are
   62   * also some lint checks which operate directly on ast nodes (such as
   63   * @ast::item), and those are organized as check_item_*(). Each visitor added
   64   * to the lint context is modified to stop once it reaches a node which could
   65   * alter the lint levels. This means that everything is looked at once and
   66   * only once by every lint pass.
   67   *
   68   * With this all in place, to add a new lint warning, all you need to do is to
   69   * either invoke `add_lint` on the session at the appropriate time, or write a
   70   * lint pass in this module which is just an ast visitor. The context used when
   71   * traversing the ast has a `span_lint` method which only needs the span of the
   72   * item that's being warned about.
   73   */

  366  enum AnyVisitor {
  367      // This is a pair so every visitor can visit every node. When a lint pass
  368      // is registered, another visitor is created which stops at all items
  369      // which can alter the attributes of the ast. This "item stopping visitor"
  370      // is the second element of the pair, while the original visitor is the
  371      // first element. This means that when visiting a node, the original
  372      // recursive call can use the original visitor's method, although the
  373      // recursing visitor supplied to the method is the item stopping visitor.
  374      OldVisitor(@mut OuterLint, @mut InnerLint),
  375      NewVisitor(@mut visit::Visitor<()>),
  376  }

  384      tcx: ty::ctxt,
  385      // Just a simple flag if we're currently recursing into a trait
  386      // implementation. This is only used by the lint_missing_doc() pass
  387      in_trait_impl: bool,
  388      // Another flag for doc lint emissions. Does some parent of the current node
  389      // have the doc(hidden) attribute? Treating this as allow(missing_doc) would
  390      // play badly with forbid(missing_doc) when it shouldn't.
  391      doc_hidden: bool,
  392      // When recursing into an attributed node of the ast which modifies lint
  393      // levels, this stack keeps track of the previous lint levels of whatever
  394      // was modified.
  395      lint_stack: ~[(lint, level, LintSource)],
  396      // Each of these visitors represents a lint pass. A number of the lint
  397      // attributes are registered by adding a visitor to iterate over the ast.
  398      // Others operate directly on @ast::item structures (or similar). Finally,
  399      // others still are added to the Session object via `add_lint`, and these
  400      // are all passed with the lint_session visitor.
  401      visitors: ~[AnyVisitor],
  402  }

  405      fn get_level(&self, lint: lint) -> level {
  406          match self.curr.find(&(lint as uint)) {
  407            Some(&(lvl, _)) => lvl,
  408            None => allow
  409          }
  410      }
  411  
  412      fn get_source(&self, lint: lint) -> LintSource {
  413          match self.curr.find(&(lint as uint)) {
  414            Some(&(_, src)) => src,
  415            None => Default
  416          }
  417      }
  418  
  419      fn set_level(&mut self, lint: lint, level: level, src: LintSource) {
  420          if level == allow {
  421              self.curr.remove(&(lint as uint));
  422          } else {
  423              self.curr.insert(lint as uint, (level, src));
  424          }
  425      }

  427      fn lint_to_str(&self, lint: lint) -> &'static str {
  428          for (k, v) in self.dict.iter() {
  429              if v.lint == lint {
  430                  return *k;
  431              }
  432          }
  433          fail!("unregistered lint %?", lint);
  434      }
  435  
  436      fn span_lint(&self, lint: lint, span: Span, msg: &str) {
  437          let (level, src) = match self.curr.find(&(lint as uint)) {
  438              None => { return }
  439              Some(&(warn, src)) => (self.get_level(warnings), src),
  440              Some(&pair) => pair,
  441          };
  442          if level == allow { return }
  443  
  444          let mut note = None;
  445          let msg = match src {
  446              Default | CommandLine => {
  447                  fmt!("%s [-%c %s%s]", msg, match level {
  448                          warn => 'W', deny => 'D', forbid => 'F',
  449                          allow => fail!()
  450                      }, self.lint_to_str(lint).replace("_", "-"),
  451                      if src == Default { " (default)" } else { "" })
  452              },
  453              Node(src) => {
  454                  note = Some(src);
  455                  msg.to_str()
  456              }
  457          };
  458          match level {
  459              warn =>          { self.tcx.sess.span_warn(span, msg); }
  460              deny | forbid => { self.tcx.sess.span_err(span, msg);  }
  461              allow => fail!(),
  462          }

  470       * Merge the lints specified by any lint attributes into the
  471       * current lint context, call the provided function, then reset the
  472       * lints in effect to their previous state.
  473       */
  474      fn with_lint_attrs(@mut self, attrs: &[ast::Attribute], f: &fn()) {
  475          // Parse all of the lint attributes, and then add them all to the
  476          // current dictionary of lint information. Along the way, keep a history
  477          // of what we changed so we can roll everything back after invoking the
  478          // specified closure
  479          let mut pushed = 0u;
  480          do each_lint(self.tcx.sess, attrs) |meta, level, lintname| {
  481              match self.dict.find_equiv(&lintname) {
  482                  None => {
  483                      self.span_lint(
  484                          unrecognized_lint,
  485                          meta.span,
  486                          fmt!("unknown `%s` attribute: `%s`",
  487                          level_to_str(level), lintname));
  488                  }
  489                  Some(lint) => {
  490                      let lint = lint.lint;
  491                      let now = self.get_level(lint);
  492                      if now == forbid && level != forbid {
  493                          self.tcx.sess.span_err(meta.span,
  494                          fmt!("%s(%s) overruled by outer forbid(%s)",
  495                          level_to_str(level),
  496                          lintname, lintname));
  497                      } else if now != level {
  498                          let src = self.get_source(lint);
  499                          self.lint_stack.push((lint, now, src));
  500                          pushed += 1;
  501                          self.set_level(lint, level, Node(meta.span));
  502                      }
  503                  }
  504              }
  505              true
  506          };
  507  
  508          // detect doc(hidden)
  509          let mut doc_hidden = do attrs.iter().any |attr| {
  510              "doc" == attr.name() &&
  511                  match attr.meta_item_list() {
  512                      Some(l) => attr::contains_name(l, "hidden"),
  513                      None    => false // not of the form #[doc(...)]
  514                  }
  515          };
  516  
  517          if doc_hidden && !self.doc_hidden {
  518              self.doc_hidden = true;
  519          } else {
  520              doc_hidden = false;
  521          }
  522  
  523          f();
  524  
  525          // rollback
  526          if doc_hidden && self.doc_hidden {
  527              self.doc_hidden = false;
  528          }
  529          do pushed.times {
  530              let (lint, lvl, src) = self.lint_stack.pop();
  531              self.set_level(lint, lvl, src);
  532          }
  533      }

  543      fn process(@mut self, n: AttributedNode) {

  545          // pair instead of just one visitor.
  546          match n {
  547              Item(it) => {

  551                              orig.process_item(it, self);

  555                              let new_visitor = new_visitor;
  556                              new_visitor.visit_item(it, ());
  557                          }
  558                      }
  559                  }
  560              }
  561              Crate(c) => {
  562                  for visitor in self.visitors.iter() {
  563                      match *visitor {
  564                          OldVisitor(_, stopping) => {
  565                              stopping.descend_crate(c, self)
  566                          }
  567                          NewVisitor(new_visitor) => {
  568                              let mut new_visitor = new_visitor;
  569                              visit::walk_crate(&mut new_visitor, c, ())
  570                          }
  571                      }
  572                  }
  573              }
  574              // Can't use visit::walk_method_helper because the
  575              // item_stopping_visitor has overridden visit_fn(&fk_method(... ))
  576              // to be a no-op, so manually invoke visit_fn.
  577              Method(m) => {
  578                  for visitor in self.visitors.iter() {

  580                          OldVisitor(orig, stopping) => {
  581                              let fk = visit::fk_method(m.ident, &m.generics, m);

  584                          }
  585                          NewVisitor(new_visitor) => {
  586                              let fk = visit::fk_method(m.ident,
  587                                                        &m.generics,
  588                                                        m);
  589                              let new_visitor = new_visitor;
  590                              new_visitor.visit_fn(&fk,
  591                                                   &m.decl,
  592                                                   &m.body,
  593                                                   m.span,
  594                                                   m.id,
  595                                                   ())
  596                          }
  597                      }
  598                  }
  599              }
  600          }
  601      }
  602  }

  604  pub fn each_lint(sess: session::Session,
  605                   attrs: &[ast::Attribute],
  606                   f: &fn(@ast::MetaItem, level, @str) -> bool) -> bool {

  613                  ast::MetaList(_, ref metas) => metas,
  614                  _ => {
  615                      sess.span_err(meta.span, "malformed lint attribute");
  616                      loop;
  617                  }
  618              };
  619              for meta in metas.iter() {
  620                  match meta.node {
  621                      ast::MetaWord(lintname) => {
  622                          if !f(*meta, level, lintname) {
  623                              return false;
  624                          }
  625                      }
  626                      _ => {
  627                          sess.span_err(meta.span, "malformed lint attribute");
  628                      }
  629                  }
  630              }
  631          }
  632      }
  633      true
  634  }

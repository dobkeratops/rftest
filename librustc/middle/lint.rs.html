   36 /**
   37  * A 'lint' check is a kind of miscellaneous constraint that a user _might_
   38  * want to enforce, but might reasonably want to permit as well, on a
   39  * module-by-module basis. They contrast with static constraints enforced by
   40  * other phases of the compiler, which are generally required to hold in order
   41  * to compile the program at all.
   42  *
   43  * The lint checking is all consolidated into one pass which runs just before
   44  * translation to LLVM bytecode. Throughout compilation, lint warnings can be
   45  * added via the `add_lint` method on the Session structure. This requires a
   46  * span and an id of the node that the lint is being added to. The lint isn't
   47  * actually emitted at that time because it is unknown what the actual lint
   48  * level at that location is.
   49  *
   50  * To actually emit lint warnings/errors, a separate pass is used just before
   51  * translation. A context keeps track of the current state of all lint levels.
   52  * Upon entering a node of the ast which can modify the lint settings, the
   53  * previous lint state is pushed onto a stack and the ast is then recursed upon.
   54  * As the ast is traversed, this keeps track of the current lint level for all
   55  * lint attributes.
   56  *
   57  * At each node of the ast which can modify lint attributes, all known lint
   58  * passes are also applied.  Each lint pass is an oldvisit::vt<()> structure.
   59  * The visitors are constructed via the lint_*() functions below. There are
   60  * also some lint checks which operate directly on ast nodes (such as
   61  * @ast::item), and those are organized as check_item_*(). Each visitor added
   62  * to the lint context is modified to stop once it reaches a node which could
   63  * alter the lint levels. This means that everything is looked at once and
   64  * only once by every lint pass.
   65  *
   66  * With this all in place, to add a new lint warning, all you need to do is to
   67  * either invoke `add_lint` on the session at the appropriate time, or write a
   68  * lint pass in this module which is just an ast visitor. The context used when
   69  * traversing the ast has a `span_lint` method which only needs the span of the
   70  * item that's being warned about.
   71  */

  295 enum AnyVisitor {
  296     // This is a pair so every visitor can visit every node. When a lint pass
  297     // is registered, another visitor is created which stops at all items
  298     // which can alter the attributes of the ast. This "item stopping visitor"
  299     // is the second element of the pair, while the original visitor is the
  300     // first element. This means that when visiting a node, the original
  301     // recursive call can use the original visitor's method, although the
  302     // recursing visitor supplied to the method is the item stopping visitor.
  303     OldVisitor(oldvisit::vt<@mut Context>, oldvisit::vt<@mut Context>),
  304     NewVisitor(@visit::Visitor<()>),
  305 }

  307 struct Context {
  308     // All known lint modes (string versions)
  309     dict: @LintDict,
  310     // Current levels of each lint warning
  311     curr: SmallIntMap<(level, LintSource)>,
  312     // context we're checking in (used to access fields like sess)
  313     tcx: ty::ctxt,
  314     // Just a simple flag if we're currently recursing into a trait
  315     // implementation. This is only used by the lint_missing_doc() pass
  316     in_trait_impl: bool,
  317     // Another flag for doc lint emissions. Does some parent of the current node
  318     // have the doc(hidden) attribute? Treating this as allow(missing_doc) would
  319     // play badly with forbid(missing_doc) when it shouldn't.
  320     doc_hidden: bool,
  321     // When recursing into an attributed node of the ast which modifies lint
  322     // levels, this stack keeps track of the previous lint levels of whatever
  323     // was modified.
  324     lint_stack: ~[(lint, level, LintSource)],
  325     // Each of these visitors represents a lint pass. A number of the lint
  326     // attributes are registered by adding a visitor to iterate over the ast.
  327     // Others operate directly on @ast::item structures (or similar). Finally,
  328     // others still are added to the Session object via `add_lint`, and these
  329     // are all passed with the lint_session visitor.
  330     visitors: ~[AnyVisitor],
  331 }

  350             self.curr.remove(&(lint as uint));

  354     }

  365     fn span_lint(&self, lint: lint, span: span, msg: &str) {
  366         let (level, src) = match self.curr.find(&(lint as uint)) {
  367             None => { return }
  368             Some(&(warn, src)) => (self.get_level(warnings), src),
  369             Some(&pair) => pair,
  370         };
  371         if level == allow { return }
  372 
  373         let mut note = None;
  374         let msg = match src {
  375             Default | CommandLine => {
  376                 fmt!("%s [-%c %s%s]", msg, match level {
  377                         warn => 'W', deny => 'D', forbid => 'F',
  378                         allow => fail!()
  379                     }, self.lint_to_str(lint).replace("_", "-"),
  380                     if src == Default { " (default)" } else { "" })
  381             },
  382             Node(src) => {
  383                 note = Some(src);
  384                 msg.to_str()
  385             }
  386         };
  387         match level {
  388             warn =>          { self.tcx.sess.span_warn(span, msg); }
  389             deny | forbid => { self.tcx.sess.span_err(span, msg);  }
  390             allow => fail!(),
  391         }
  392 
  393         for &span in note.iter() {
  394             self.tcx.sess.span_note(span, "lint level defined here");
  395         }
  396     }

  398     /**
  399      * Merge the lints specified by any lint attributes into the
  400      * current lint context, call the provided function, then reset the
  401      * lints in effect to their previous state.
  402      */
  403     fn with_lint_attrs(@mut self, attrs: &[ast::Attribute], f: &fn()) {
  404         // Parse all of the lint attributes, and then add them all to the
  405         // current dictionary of lint information. Along the way, keep a history
  406         // of what we changed so we can roll everything back after invoking the
  407         // specified closure
  408         let mut pushed = 0u;
  409         do each_lint(self.tcx.sess, attrs) |meta, level, lintname| {
  410             match self.dict.find_equiv(&lintname) {
  411                 None => {
  412                     self.span_lint(
  413                         unrecognized_lint,
  414                         meta.span,
  415                         fmt!("unknown `%s` attribute: `%s`",
  416                         level_to_str(level), lintname));
  417                 }
  418                 Some(lint) => {
  419                     let lint = lint.lint;
  420                     let now = self.get_level(lint);
  421                     if now == forbid && level != forbid {
  422                         self.tcx.sess.span_err(meta.span,
  423                         fmt!("%s(%s) overruled by outer forbid(%s)",
  424                         level_to_str(level),
  425                         lintname, lintname));
  426                     } else if now != level {
  427                         let src = self.get_source(lint);
  428                         self.lint_stack.push((lint, now, src));
  429                         pushed += 1;
  430                         self.set_level(lint, level, Node(meta.span));
  431                     }
  432                 }
  433             }
  434             true
  435         };
  436 
  437         // detect doc(hidden)
  438         let mut doc_hidden = do attrs.iter().any |attr| {
  439             "doc" == attr.name() &&
  440                 match attr.meta_item_list() {
  441                     Some(l) => attr::contains_name(l, "hidden"),
  442                     None    => false // not of the form #[doc(...)]
  443                 }
  444         };
  445 
  446         if doc_hidden && !self.doc_hidden {
  447             self.doc_hidden = true;
  448         } else {
  449             doc_hidden = false;
  450         }
  451 
  452         f();
  453 
  454         // rollback
  455         if doc_hidden && self.doc_hidden {
  456             self.doc_hidden = false;
  457         }
  458         do pushed.times {
  459             let (lint, lvl, src) = self.lint_stack.pop();
  460             self.set_level(lint, lvl, src);
  461         }
  462     }

  472     fn process(@mut self, n: AttributedNode) {
  473         // see comment of the `visitors` field in the struct for why there's a
  474         // pair instead of just one visitor.
  475         match n {
  476             Item(it) => {
  477                 for visitor in self.visitors.iter() {
  478                     match *visitor {
  479                         OldVisitor(orig, stopping) => {

  608 fn lint_type_limits() -> oldvisit::vt<@mut Context> {
  609     fn is_valid<T:cmp::Ord>(binop: ast::binop, v: T,
  610             min: T, max: T) -> bool {
  611         match binop {
  612             ast::lt => v <= max,
  613             ast::le => v < max,
  614             ast::gt => v >= min,
  615             ast::ge => v > min,
  616             ast::eq | ast::ne => v >= min && v <= max,
  617             _ => fail!()
  618         }
  619     }
  620 
  621     fn rev_binop(binop: ast::binop) -> ast::binop {
  622         match binop {
  623             ast::lt => ast::gt,
  624             ast::le => ast::ge,
  625             ast::gt => ast::lt,
  626             ast::ge => ast::le,
  627             _ => binop
  628         }
  629     }
  630 
  631     // for int & uint, be conservative with the warnings, so that the
  632     // warnings are consistent between 32- and 64-bit platforms
  633     fn int_ty_range(int_ty: ast::int_ty) -> (i64, i64) {
  634         match int_ty {
  635             ast::ty_i =>    (i64::min_value,        i64::max_value),
  636             ast::ty_char => (u32::min_value as i64, u32::max_value as i64),
  637             ast::ty_i8 =>   (i8::min_value  as i64, i8::max_value  as i64),
  638             ast::ty_i16 =>  (i16::min_value as i64, i16::max_value as i64),
  639             ast::ty_i32 =>  (i32::min_value as i64, i32::max_value as i64),
  640             ast::ty_i64 =>  (i64::min_value,        i64::max_value)
  641         }
  642     }
  643 
  644     fn uint_ty_range(uint_ty: ast::uint_ty) -> (u64, u64) {
  645         match uint_ty {
  646             ast::ty_u =>   (u64::min_value,         u64::max_value),
  647             ast::ty_u8 =>  (u8::min_value   as u64, u8::max_value   as u64),
  648             ast::ty_u16 => (u16::min_value  as u64, u16::max_value  as u64),
  649             ast::ty_u32 => (u32::min_value  as u64, u32::max_value  as u64),

  652     }

  654     fn check_limits(cx: &Context,
  655                     binop: ast::binop,
  656                     l: @ast::expr,
  657                     r: @ast::expr)
  658                     -> bool {
  659         let (lit, expr, swap) = match (&l.node, &r.node) {
  660             (&ast::expr_lit(_), _) => (l, r, true),
  661             (_, &ast::expr_lit(_)) => (r, l, false),
  662             _ => return true
  663         };
  664         // Normalize the binop so that the literal is always on the RHS in
  665         // the comparison
  666         let norm_binop = if swap {
  667             rev_binop(binop)
  668         } else {
  669             binop
  670         };
  671         match ty::get(ty::expr_ty(cx.tcx, expr)).sty {
  672             ty::ty_int(int_ty) => {
  673                 let (min, max) = int_ty_range(int_ty);
  674                 let lit_val: i64 = match lit.node {
  675                     ast::expr_lit(@li) => match li.node {
  676                         ast::lit_int(v, _) => v,
  677                         ast::lit_uint(v, _) => v as i64,
  678                         ast::lit_int_unsuffixed(v) => v,
  679                         _ => return true
  680                     },
  681                     _ => fail!()
  682                 };
  683                 is_valid(norm_binop, lit_val, min, max)
  684             }
  685             ty::ty_uint(uint_ty) => {
  686                 let (min, max): (u64, u64) = uint_ty_range(uint_ty);
  687                 let lit_val: u64 = match lit.node {
  688                     ast::expr_lit(@li) => match li.node {
  689                         ast::lit_int(v, _) => v as u64,
  690                         ast::lit_uint(v, _) => v,
  691                         ast::lit_int_unsuffixed(v) => v as u64,
  692                         _ => return true
  693                     },
  694                     _ => fail!()
  695                 };
  696                 is_valid(norm_binop, lit_val, min, max)
  697             }
  698             _ => true
  699         }
  700     }

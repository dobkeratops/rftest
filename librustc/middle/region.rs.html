  39 /**
  40 The region maps encode information about region relationships.
  41 
  42 - `scope_map` maps from:
  43   - an expression to the expression or block encoding the maximum
  44     (static) lifetime of a value produced by that expression.  This is
  45     generally the innermost call, statement, match, or block.
  46   - a variable or binding id to the block in which that variable is declared.
  47 - `free_region_map` maps from:
  48   - a free region `a` to a list of free regions `bs` such that
  49     `a <= b for all b in bs`
  50   - the free region map is populated during type check as we check
  51     each function. See the function `relate_free_regions` for
  52     more information.
  53 - `cleanup_scopes` includes scopes where trans cleanups occur
  54   - this is intended to reflect the current state of trans, not
  55     necessarily how I think things ought to work
  56 */
  57 pub struct RegionMaps {
  58     priv scope_map: HashMap<ast::NodeId, ast::NodeId>,
  59     priv free_region_map: HashMap<FreeRegion, ~[FreeRegion]>,
  60     priv cleanup_scopes: HashSet<ast::NodeId>
  61 }
  62 
  63 #[deriving(Clone)]
  64 pub struct Context {
  65     sess: Session,
  66     def_map: resolve::DefMap,
  67 
  68     // Generated maps:
  69     region_maps: @mut RegionMaps,
  70 
  71     // Scope where variables should be parented to
  72     var_parent: Option<ast::NodeId>,
  73 
  74     // Innermost enclosing expression
  75     parent: Option<ast::NodeId>,
  76 }

  79     pub fn relate_free_regions(&mut self, sub: FreeRegion, sup: FreeRegion) {
  80         match self.free_region_map.find_mut(&sub) {
  81             Some(sups) => {
  82                 if !sups.iter().any(|x| x == &sup) {
  83                     sups.push(sup);
  84                 }
  85                 return;
  86             }
  87             None => {}
  88         }
  89 
  90         debug!("relate_free_regions(sub=%?, sup=%?)", sub, sup);
  91 
  92         self.free_region_map.insert(sub, ~[sup]);
  93     }

 100     }

 117     pub fn encl_scope(&self, id: ast::NodeId) -> ast::NodeId {

 130     pub fn cleanup_scope(&self, expr_id: ast::NodeId) -> ast::NodeId {
 131         //! Returns the scope when temps in expr will be cleaned up
 132 
 133         let mut id = self.encl_scope(expr_id);
 134         while !self.cleanup_scopes.contains(&id) {
 135             id = self.encl_scope(id);
 136         }
 137         return id;
 138     }

 152     pub fn is_subscope_of(&self,
 153                           subscope: ast::NodeId,
 154                           superscope: ast::NodeId)
 155                           -> bool {
 156         /*!
 157          * Returns true if `subscope` is equal to or is lexically
 158          * nested inside `superscope` and false otherwise.
 159          */
 160 
 161         let mut s = subscope;
 162         while superscope != s {
 163             match self.scope_map.find(&s) {
 164                 None => {
 165                     debug!("is_subscope_of(%?, %?, s=%?)=false",
 166                            subscope, superscope, s);
 167 
 168                     return false;
 169                 }
 170                 Some(&scope) => s = scope
 171             }
 172         }
 173 
 174         debug!("is_subscope_of(%?, %?)=true",
 175                subscope, superscope);
 176 
 177         return true;
 178     }

 180     pub fn sub_free_region(&self, sub: FreeRegion, sup: FreeRegion) -> bool {
 181         /*!
 182          * Determines whether two free regions have a subregion relationship
 183          * by walking the graph encoded in `free_region_map`.  Note that
 184          * it is possible that `sub != sup` and `sub <= sup` and `sup <= sub`
 185          * (that is, the user can give two different names to the same lifetime).
 186          */
 187 
 188         if sub == sup {
 189             return true;
 190         }
 191 
 192         // Do a little breadth-first-search here.  The `queue` list
 193         // doubles as a way to detect if we've seen a particular FR
 194         // before.  Note that we expect this graph to be an *extremely
 195         // shallow* tree.
 196         let mut queue = ~[sub];
 197         let mut i = 0;
 198         while i < queue.len() {
 199             match self.free_region_map.find(&queue[i]) {
 200                 Some(parents) => {
 201                     for parent in parents.iter() {
 202                         if *parent == sup {
 203                             return true;
 204                         }
 205 
 206                         if !queue.iter().any(|x| x == parent) {
 207                             queue.push(*parent);
 208                         }
 209                     }
 210                 }
 211                 None => {}
 212             }
 213             i += 1;
 214         }
 215         return false;
 216     }

 218     pub fn is_subregion_of(&self,
 219                            sub_region: ty::Region,
 220                            super_region: ty::Region)
 221                            -> bool {
 222         /*!
 223          * Determines whether one region is a subregion of another.  This is
 224          * intended to run *after inference* and sadly the logic is somewhat
 225          * duplicated with the code in infer.rs.
 226          */
 227 
 228         debug!("is_subregion_of(sub_region=%?, super_region=%?)",
 229                sub_region, super_region);
 230 
 231         sub_region == super_region || {
 232             match (sub_region, super_region) {
 233                 (_, ty::re_static) => {
 234                     true
 235                 }
 236 
 237                 (ty::re_scope(sub_scope), ty::re_scope(super_scope)) => {
 238                     self.is_subscope_of(sub_scope, super_scope)
 239                 }
 240 
 241                 (ty::re_scope(sub_scope), ty::re_free(ref fr)) => {
 242                     self.is_subscope_of(sub_scope, fr.scope_id)
 243                 }
 244 
 245                 (ty::re_free(sub_fr), ty::re_free(super_fr)) => {
 246                     self.sub_free_region(sub_fr, super_fr)
 247                 }
 248 
 249                 _ => {
 250                     false
 251                 }
 252             }
 253         }
 254     }

 256     pub fn nearest_common_ancestor(&self,
 257                                    scope_a: ast::NodeId,
 258                                    scope_b: ast::NodeId)
 259                                    -> Option<ast::NodeId> {
 260         /*!
 261          * Finds the nearest common ancestor (if any) of two scopes.  That
 262          * is, finds the smallest scope which is greater than or equal to
 263          * both `scope_a` and `scope_b`.
 264          */
 265 
 266         if scope_a == scope_b { return Some(scope_a); }
 267 
 268         let a_ancestors = ancestors_of(self, scope_a);
 269         let b_ancestors = ancestors_of(self, scope_b);
 270         let mut a_index = a_ancestors.len() - 1u;
 271         let mut b_index = b_ancestors.len() - 1u;
 272 
 273         // Here, ~[ab]_ancestors is a vector going from narrow to broad.
 274         // The end of each vector will be the item where the scope is
 275         // defined; if there are any common ancestors, then the tails of
 276         // the vector will be the same.  So basically we want to walk
 277         // backwards from the tail of each vector and find the first point
 278         // where they diverge.  If one vector is a suffix of the other,
 279         // then the corresponding scope is a superscope of the other.
 280 
 281         if a_ancestors[a_index] != b_ancestors[b_index] {
 282             return None;
 283         }
 284 
 285         loop {
 286             // Loop invariant: a_ancestors[a_index] == b_ancestors[b_index]
 287             // for all indices between a_index and the end of the array
 288             if a_index == 0u { return Some(scope_a); }
 289             if b_index == 0u { return Some(scope_b); }
 290             a_index -= 1u;
 291             b_index -= 1u;
 292             if a_ancestors[a_index] != b_ancestors[b_index] {
 293                 return Some(a_ancestors[a_index + 1u]);
 294             }
 295         }
 296 
 297         fn ancestors_of(this: &RegionMaps, scope: ast::NodeId)
 298             -> ~[ast::NodeId]
 299         {
 300             // debug!("ancestors_of(scope=%d)", scope);
 301             let mut result = ~[scope];
 302             let mut scope = scope;
 303             loop {
 304                 match this.scope_map.find(&scope) {
 305                     None => return result,
 306                     Some(&superscope) => {
 307                         result.push(superscope);
 308                         scope = superscope;
 309                     }
 310                 }
 311                 // debug!("ancestors_of_loop(scope=%d)", scope);
 312             }
 313         }
 314     }
 315 }

 350 fn resolve_stmt(stmt: @ast::stmt,
 351                 (cx, visitor): (Context, oldvisit::vt<Context>)) {
 352     match stmt.node {
 353         ast::stmt_decl(*) => {
 354             oldvisit::visit_stmt(stmt, (cx, visitor));
 355         }
 356         ast::stmt_expr(_, stmt_id) |
 357         ast::stmt_semi(_, stmt_id) => {
 358             parent_to_expr(cx, stmt_id, stmt.span);
 359             let expr_cx = Context {parent: Some(stmt_id), ..cx};
 360             oldvisit::visit_stmt(stmt, (expr_cx, visitor));
 361         }
 362         ast::stmt_mac(*) => cx.sess.bug("unexpanded macro")
 363     }

 366 fn resolve_expr(expr: @ast::expr,
 367                 (cx, visitor): (Context, oldvisit::vt<Context>)) {
 368     parent_to_expr(cx, expr.id, expr.span);
 369 
 370     let mut new_cx = cx;
 371     new_cx.parent = Some(expr.id);
 372     match expr.node {
 373         ast::expr_assign_op(*) | ast::expr_index(*) | ast::expr_binary(*) |
 374         ast::expr_unary(*) | ast::expr_call(*) | ast::expr_method_call(*) => {
 375             // FIXME(#6268) Nested method calls
 376             //
 377             // The lifetimes for a call or method call look as follows:
 378             //
 379             // call.id
 380             // - arg0.id
 381             // - ...
 382             // - argN.id
 383             // - call.callee_id
 384             //
 385             // The idea is that call.callee_id represents *the time when
 386             // the invoked function is actually running* and call.id
 387             // represents *the time to prepare the arguments and make the
 388             // call*.  See the section "Borrows in Calls" borrowck/doc.rs
 389             // for an extended explanantion of why this distinction is
 390             // important.
 391             //
 392             // parent_to_expr(new_cx, expr.callee_id);
 393         }
 394 
 395         ast::expr_match(*) => {
 396             new_cx.var_parent = Some(expr.id);
 397         }
 398 
 399         _ => {}
 400     };
 401 
 402 
 403     oldvisit::visit_expr(expr, (new_cx, visitor));
 404 }

 420 fn resolve_fn(fk: &oldvisit::fn_kind,
 421               decl: &ast::fn_decl,
 422               body: &ast::Block,
 423               sp: span,
 424               id: ast::NodeId,
 425               (cx, visitor): (Context,
 426                               oldvisit::vt<Context>)) {
 427     debug!("region::resolve_fn(id=%?, \
 428                                span=%?, \
 429                                body.id=%?, \
 430                                cx.parent=%?)",
 431            id,
 432            cx.sess.codemap.span_to_str(sp),
 433            body.id,
 434            cx.parent);

 436     // The arguments and `self` are parented to the body of the fn.
 437     let decl_cx = Context {parent: Some(body.id),
 438                            var_parent: Some(body.id),
 439                            ..cx};
 440     match *fk {
 441         oldvisit::fk_method(_, _, method) => {
 442             cx.region_maps.record_parent(method.self_id, body.id);
 443         }
 444         _ => {}
 445     }

 449     // or it continues with the inherited scope (closures).
 450     let body_cx = match *fk {
 451         oldvisit::fk_item_fn(*) |
 452         oldvisit::fk_method(*) => {
 453             Context {parent: None, var_parent: None, ..cx}
 454         }
 455         oldvisit::fk_anon(*) |
 456         oldvisit::fk_fn_block(*) => {
 457             cx
 458         }
 459     };
 460     (visitor.visit_block)(body, (body_cx, visitor));
 461 }

 463 pub fn resolve_crate(sess: Session,
 464                      def_map: resolve::DefMap,
 465                      crate: &ast::Crate) -> @mut RegionMaps
 466 {
 467     let region_maps = @mut RegionMaps {
 468         scope_map: HashMap::new(),
 469         free_region_map: HashMap::new(),
 470         cleanup_scopes: HashSet::new(),
 471     };
 472     let cx = Context {sess: sess,
 473                       def_map: def_map,
 474                       region_maps: region_maps,
 475                       parent: None,

 492 // ___________________________________________________________________________
 493 // Determining region parameterization
 494 //
 495 // Infers which type defns must be region parameterized---this is done
 496 // by scanning their contents to see whether they reference a region
 497 // type, directly or indirectly.  This is a fixed-point computation.
 498 //
 499 // We do it in two passes.  First we walk the AST and construct a map
 500 // from each type defn T1 to other defns which make use of it.  For example,
 501 // if we have a type like:
 502 //
 503 //    type S = *int;
 504 //    type T = S;
 505 //
 506 // Then there would be a map entry from S to T.  During the same walk,
 507 // we also construct add any types that reference regions to a set and
 508 // a worklist.  We can then process the worklist, propagating indirect
 509 // dependencies until a fixed point is reached.
 510 
 511 pub type region_paramd_items = @mut HashMap<ast::NodeId, region_variance>;
 512 
 513 #[deriving(Eq)]
 514 pub struct region_dep {
 515     ambient_variance: region_variance,
 516     id: ast::NodeId
 517 }
 518 
 519 pub struct DetermineRpCtxt {
 520     sess: Session,
 521     ast_map: ast_map::map,
 522     def_map: resolve::DefMap,
 523     region_paramd_items: region_paramd_items,
 524     dep_map: @mut HashMap<ast::NodeId, @mut ~[region_dep]>,
 525     worklist: ~[ast::NodeId],
 526 
 527     // the innermost enclosing item id
 528     item_id: ast::NodeId,
 529 
 530     // true when we are within an item but not within a method.
 531     // see long discussion on region_is_relevant().
 532     anon_implies_rp: bool,
 533 
 534     // encodes the context of the current type; invariant if
 535     // mutable, covariant otherwise
 536     ambient_variance: region_variance,
 537 }

 539 pub fn join_variance(variance1: region_variance,
 540                      variance2: region_variance)
 541                   -> region_variance {
 542     match (variance1, variance2) {
 543       (rv_invariant, _) => {rv_invariant}
 544       (_, rv_invariant) => {rv_invariant}
 545       (rv_covariant, rv_contravariant) => {rv_invariant}
 546       (rv_contravariant, rv_covariant) => {rv_invariant}
 547       (rv_covariant, rv_covariant) => {rv_covariant}
 548       (rv_contravariant, rv_contravariant) => {rv_contravariant}
 549     }
 550 }
 551 
 552 /// Combines the ambient variance with the variance of a
 553 /// particular site to yield the final variance of the reference.
 554 ///
 555 /// Example: if we are checking function arguments then the ambient
 556 /// variance is contravariant.  If we then find a `&'r T` pointer, `r`
 557 /// appears in a co-variant position.  This implies that this
 558 /// occurrence of `r` is contra-variant with respect to the current
 559 /// item, and hence the function returns `rv_contravariant`.
 560 pub fn add_variance(ambient_variance: region_variance,
 561                     variance: region_variance)
 562                  -> region_variance {
 563     match (ambient_variance, variance) {
 564       (rv_invariant, _) => rv_invariant,
 565       (_, rv_invariant) => rv_invariant,
 566       (rv_covariant, c) => c,
 567       (c, rv_covariant) => c,
 568       (rv_contravariant, rv_contravariant) => rv_covariant
 569     }
 570 }
 571 
 572 impl DetermineRpCtxt {
 573     pub fn add_variance(&self, variance: region_variance) -> region_variance {
 574         add_variance(self.ambient_variance, variance)
 575     }
 576 
 577     /// Records that item `id` is region-parameterized with the
 578     /// variance `variance`.  If `id` was already parameterized, then
 579     /// the new variance is joined with the old variance.
 580     pub fn add_rp(&mut self, id: ast::NodeId, variance: region_variance) {
 581         assert!(id != 0);
 582         let old_variance = self.region_paramd_items.find(&id).
 583                                 map_consume(|x| *x);
 584         let joined_variance = match old_variance {
 585           None => variance,
 586           Some(v) => join_variance(v, variance)
 587         };
 588 
 589         debug!("add_rp() variance for %s: %? == %? ^ %?",
 590                ast_map::node_id_to_str(self.ast_map, id,
 591                                        token::get_ident_interner()),
 592                joined_variance, old_variance, variance);
 593 
 594         if Some(joined_variance) != old_variance {
 595             let region_paramd_items = self.region_paramd_items;
 596             region_paramd_items.insert(id, joined_variance);
 597             self.worklist.push(id);
 598         }
 599     }
 600 
 601     /// Indicates that the region-parameterization of the current item
 602     /// is dependent on the region-parameterization of the item
 603     /// `from`.  Put another way, it indicates that the current item
 604     /// contains a value of type `from`, so if `from` is
 605     /// region-parameterized, so is the current item.
 606     pub fn add_dep(&mut self, from: ast::NodeId) {
 607         debug!("add dependency from %d -> %d (%s -> %s) with variance %?",
 608                from, self.item_id,
 609                ast_map::node_id_to_str(self.ast_map, from,
 610                                        token::get_ident_interner()),
 611                ast_map::node_id_to_str(self.ast_map, self.item_id,
 612                                        token::get_ident_interner()),
 613                self.ambient_variance);
 614         let vec = do self.dep_map.find_or_insert_with(from) |_| {
 615             @mut ~[]
 616         };
 617         let dep = region_dep {
 618             ambient_variance: self.ambient_variance,
 619             id: self.item_id
 620         };
 621         if !vec.iter().any(|x| x == &dep) { vec.push(dep); }
 622     }
 623 
 624     // Determines whether a reference to a region that appears in the
 625     // AST implies that the enclosing type is region-parameterized (RP).
 626     // This point is subtle.  Here are some examples to make it more
 627     // concrete.
 628     //
 629     // 1. impl foo for &int { ... }
 630     // 2. impl foo for &'self int { ... }
 631     // 3. impl foo for bar { fn m(@self) -> &'self int { ... } }
 632     // 4. impl foo for bar { fn m(&self) -> &'self int { ... } }
 633     // 5. impl foo for bar { fn m(&self) -> &int { ... } }
 634     //
 635     // In case 1, the anonymous region is being referenced,
 636     // but it appears in a context where the anonymous region
 637     // resolves to self, so the impl foo is RP.
 638     //
 639     // In case 2, the self parameter is written explicitly.
 640     //
 641     // In case 3, the method refers to the region `self`, so that
 642     // implies that the impl must be region parameterized.  (If the
 643     // type bar is not region parameterized, that is an error, because
 644     // the self region is effectively unconstrained, but that is
 645     // detected elsewhere).
 646     //
 647     // In case 4, the method refers to the region `self`, but the
 648     // `self` region is bound by the `&self` receiver, and so this
 649     // does not require that `bar` be RP.
 650     //
 651     // In case 5, the anonymous region is referenced, but it
 652     // bound by the method, so it does not refer to self.  This impl
 653     // need not be region parameterized.
 654     //
 655     // Normally, & or &self implies that the enclosing item is RP.
 656     // However, within a function, & is always bound.  Within a method
 657     // with &self type, &self is also bound.  We detect those last two
 658     // cases via flags (anon_implies_rp and self_implies_rp) that are
 659     // true when the anon or self region implies RP.
 660     pub fn region_is_relevant(&self, r: &Option<ast::Lifetime>) -> bool {
 661         match r {
 662             &None => {
 663                 self.anon_implies_rp
 664             }
 665             &Some(ref l) if l.ident == special_idents::statik => {
 666                 false
 667             }
 668             &Some(ref l) if l.ident == special_idents::self_ => {
 669                 true
 670             }
 671             &Some(_) => {
 672                 false
 673             }
 674         }
 675     }

 677     pub fn with(@mut self,
 678                 item_id: ast::NodeId,
 679                 anon_implies_rp: bool,
 680                 f: &fn()) {
 681         let old_item_id = self.item_id;
 682         let old_anon_implies_rp = self.anon_implies_rp;
 683         self.item_id = item_id;
 684         self.anon_implies_rp = anon_implies_rp;
 685         debug!("with_item_id(%d, %b)",
 686                item_id,
 687                anon_implies_rp);
 688         let _i = ::util::common::indenter();
 689         f();
 690         self.item_id = old_item_id;
 691         self.anon_implies_rp = old_anon_implies_rp;
 692     }

 701     }

 741 fn determine_rp_in_ty(ty: &ast::Ty,
 742                       (cx, visitor): (@mut DetermineRpCtxt,
 743                                       oldvisit::vt<@mut DetermineRpCtxt>)) {
 744     // we are only interested in types that will require an item to
 745     // be region-parameterized.  if cx.item_id is zero, then this type
 746     // is not a member of a type defn nor is it a constitutent of an
 747     // impl etc.  So we can ignore it and its components.
 748     if cx.item_id == 0 { return; }
 749 
 750     // if this type directly references a region pointer like &'r ty,
 751     // add to the worklist/set.  Note that &'r ty is contravariant with
 752     // respect to &r, because &'r ty can be used whereever a *smaller*
 753     // region is expected (and hence is a supertype of those
 754     // locations)
 755     let sess = cx.sess;
 756     match ty.node {
 757         ast::ty_rptr(ref r, _) => {
 758             debug!("referenced rptr type %s",
 759                    pprust::ty_to_str(ty, sess.intr()));
 760 
 761             if cx.region_is_relevant(r) {
 762                 let rv = cx.add_variance(rv_contravariant);
 763                 cx.add_rp(cx.item_id, rv)
 764             }
 765         }
 766 
 767         ast::ty_closure(ref f) => {
 768             debug!("referenced fn type: %s",
 769                    pprust::ty_to_str(ty, sess.intr()));
 770             match f.region {
 771                 Some(_) => {
 772                     if cx.region_is_relevant(&f.region) {
 773                         let rv = cx.add_variance(rv_contravariant);
 774                         cx.add_rp(cx.item_id, rv)
 775                     }
 776                 }
 777                 None => {
 778                     if f.sigil == ast::BorrowedSigil && cx.anon_implies_rp {
 779                         let rv = cx.add_variance(rv_contravariant);
 780                         cx.add_rp(cx.item_id, rv)
 781                     }
 782                 }
 783             }
 784         }
 785 
 786         _ => {}
 787     }
 788 
 789     // if this references another named type, add the dependency
 790     // to the dep_map.  If the type is not defined in this crate,
 791     // then check whether it is region-parameterized and consider
 792     // that as a direct dependency.
 793     match ty.node {
 794       ast::ty_path(ref path, _, id) => {
 795         match cx.def_map.find(&id) {
 796           Some(&ast::def_ty(did)) |
 797           Some(&ast::def_trait(did)) |
 798           Some(&ast::def_struct(did)) => {
 799             if did.crate == ast::LOCAL_CRATE {
 800                 if cx.region_is_relevant(&path.rp) {
 801                     cx.add_dep(did.node);
 802                 }
 803             } else {
 804                 let cstore = sess.cstore;
 805                 match csearch::get_region_param(cstore, did) {
 806                   None => {}
 807                   Some(variance) => {
 808                     debug!("reference to external, rp'd type %s",
 809                            pprust::ty_to_str(ty, sess.intr()));
 810                     if cx.region_is_relevant(&path.rp) {
 811                         let rv = cx.add_variance(variance);
 812                         cx.add_rp(cx.item_id, rv)
 813                     }
 814                   }
 815                 }
 816             }
 817           }
 818           _ => {}
 819         }
 820       }
 821       _ => {}
 822     }
 823 
 824     match ty.node {
 825       ast::ty_box(ref mt) | ast::ty_uniq(ref mt) | ast::ty_vec(ref mt) |
 826       ast::ty_rptr(_, ref mt) | ast::ty_ptr(ref mt) => {
 827         visit_mt(mt, (cx, visitor));
 828       }
 829 
 830       ast::ty_path(ref path, _, _) => {
 831         // type parameters are---for now, anyway---always invariant
 832         do cx.with_ambient_variance(rv_invariant) {
 833             for tp in path.types.iter() {
 834                 (visitor.visit_ty)(tp, (cx, visitor));
 835             }
 836         }
 837       }
 838 
 839       ast::ty_closure(@ast::TyClosure {decl: ref decl, _}) |
 840       ast::ty_bare_fn(@ast::TyBareFn {decl: ref decl, _}) => {
 841         // fn() binds the & region, so do not consider &T types that
 842         // appear *inside* a fn() type to affect the enclosing item:
 843         do cx.with(cx.item_id, false) {
 844             // parameters are contravariant
 845             do cx.with_ambient_variance(rv_contravariant) {
 846                 for a in decl.inputs.iter() {
 847                     (visitor.visit_ty)(&a.ty, (cx, visitor));
 848                 }
 849             }
 850             (visitor.visit_ty)(&decl.output, (cx, visitor));
 851         }
 852       }
 853 
 854       _ => {
 855         oldvisit::visit_ty(ty, (cx, visitor));
 856       }
 857     }

 859     fn visit_mt(mt: &ast::mt,
 860                 (cx, visitor): (@mut DetermineRpCtxt,
 861                                 oldvisit::vt<@mut DetermineRpCtxt>)) {
 862         // mutability is invariant
 863         if mt.mutbl == ast::m_mutbl {
 864             do cx.with_ambient_variance(rv_invariant) {
 865                 (visitor.visit_ty)(mt.ty, (cx, visitor));
 866             }
 867         } else {
 868             (visitor.visit_ty)(mt.ty, (cx, visitor));
 869         }
 870     }
 871 }

 880 pub fn determine_rp_in_crate(sess: Session,
 881                              ast_map: ast_map::map,
 882                              def_map: resolve::DefMap,
 883                              crate: &ast::Crate)
 884                           -> region_paramd_items {
 885     let cx = @mut DetermineRpCtxt {
 886         sess: sess,
 887         ast_map: ast_map,
 888         def_map: def_map,
 889         region_paramd_items: @mut HashMap::new(),
 890         dep_map: @mut HashMap::new(),
 891         worklist: ~[],
 892         item_id: 0,
 893         anon_implies_rp: false,
 894         ambient_variance: rv_covariant
 895     };
 896 
 897     // Gather up the base set, worklist and dep_map
 898     let visitor = oldvisit::mk_vt(@oldvisit::Visitor {
 899         visit_fn: determine_rp_in_fn,
 900         visit_item: determine_rp_in_item,
 901         visit_ty: determine_rp_in_ty,
 902         visit_ty_method: determine_rp_in_ty_method,
 903         visit_struct_field: determine_rp_in_struct_field,
 904         .. *oldvisit::default_visitor()
 905     });
 906     oldvisit::visit_crate(crate, (cx, visitor));
 907

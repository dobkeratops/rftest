  37  /**
  38  The region maps encode information about region relationships.
  39  
  40  - `scope_map` maps from a scope id to the enclosing scope id; this is
  41    usually corresponding to the lexical nesting, though in the case of
  42    closures the parent scope is the innermost conditinal expression or repeating
  43    block
  44  
  45  - `var_map` maps from a variable or binding id to the block in which
  46    that variable is declared.
  47  
  48  - `free_region_map` maps from a free region `a` to a list of free
  49    regions `bs` such that `a <= b for all b in bs`
  50    - the free region map is populated during type check as we check
  51      each function. See the function `relate_free_regions` for
  52      more information.
  53  
  54  - `rvalue_scopes` includes entries for those expressions whose cleanup
  55    scope is larger than the default. The map goes from the expression
  56    id to the cleanup scope id. For rvalues not present in this table,
  57    the appropriate cleanup scope is the innermost enclosing statement,
  58    conditional expression, or repeating block (see `terminating_scopes`).
  59  
  60  - `terminating_scopes` is a set containing the ids of each statement,
  61    or conditional/repeating expression. These scopes are calling "terminating
  62    scopes" because, when attempting to find the scope of a temporary, by
  63    default we search up the enclosing scopes until we encounter the
  64    terminating scope. A conditional/repeating
  65    expression is one which is not guaranteed to execute exactly once
  66    upon entering the parent scope. This could be because the expression
  67    only executes conditionally, such as the expression `b` in `a && b`,
  68    or because the expression may execute many times, such as a loop
  69    body. The reason that we distinguish such expressions is that, upon
  70    exiting the parent scope, we cannot statically know how many times
  71    the expression executed, and thus if the expression creates
  72    temporaries we cannot know statically how many such temporaries we
  73    would have to cleanup. Therefore we ensure that the temporaries never
  74    outlast the conditional/repeating expression, preventing the need
  75    for dynamic checks and/or arbitrary amounts of stack space.
  76  */
  77  pub struct RegionMaps {
  78      priv scope_map: RefCell<NodeMap<ast::NodeId>>,
  79      priv var_map: RefCell<NodeMap<ast::NodeId>>,
  80      priv free_region_map: RefCell<HashMap<FreeRegion, Vec<FreeRegion> >>,
  81      priv rvalue_scopes: RefCell<NodeMap<ast::NodeId>>,
  82      priv terminating_scopes: RefCell<HashSet<ast::NodeId>>,
  83  }

 102      pub fn relate_free_regions(&self, sub: FreeRegion, sup: FreeRegion) {
 103          match self.free_region_map.borrow_mut().find_mut(&sub) {
 104              Some(sups) => {
 105                  if !sups.iter().any(|x| x == &sup) {
 106                      sups.push(sup);
 107                  }
 108                  return;
 109              }
 110              None => {}
 111          }
 112  
 113          debug!("relate_free_regions(sub={:?}, sup={:?})", sub, sup);
 114          self.free_region_map.borrow_mut().insert(sub, vec!(sup));
 115      }

 148          //! Returns the narrowest scope that encloses `id`, if any.
 149          self.scope_map.borrow().find(&id).map(|x| *x)
 150      }

 152      pub fn encl_scope(&self, id: ast::NodeId) -> ast::NodeId {
 153          //! Returns the narrowest scope that encloses `id`, if any.
 154          match self.scope_map.borrow().find(&id) {
 155              Some(&r) => r,
 156              None => { fail!("no enclosing scope for id {}", id); }
 157          }
 158      }

 161          /*!
 162           * Returns the lifetime of the local variable `var_id`
 163           */
 164          match self.var_map.borrow().find(&var_id) {
 165              Some(&r) => r,
 166              None => { fail!("no enclosing scope for id {}", var_id); }
 167          }
 168      }

 170      pub fn temporary_scope(&self, expr_id: ast::NodeId) -> Option<ast::NodeId> {
 171          //! Returns the scope when temp created by expr_id will be cleaned up
 172  
 173          // check for a designated rvalue scope
 174          match self.rvalue_scopes.borrow().find(&expr_id) {
 175              Some(&s) => {
 176                  debug!("temporary_scope({}) = {} [custom]", expr_id, s);
 177                  return Some(s);
 178              }
 179              None => { }
 180          }
 181  
 182          // else, locate the innermost terminating scope
 183          // if there's one. Static items, for instance, won't
 184          // have an enclusing scope, hence no scope will be
 185          // returned.
 186          let mut id = match self.opt_encl_scope(expr_id) {
 187              Some(i) => i,
 188              None => { return None; }
 189          };
 190  
 191          while !self.terminating_scopes.borrow().contains(&id) {
 192              match self.opt_encl_scope(id) {
 193                  Some(p) => {
 194                      id = p;
 195                  }
 196                  None => {
 197                      debug!("temporary_scope({}) = None", expr_id);
 198                      return None;
 199                  }
 200              }
 201          }
 202          debug!("temporary_scope({}) = {} [enclosing]", expr_id, id);
 203          return Some(id);
 204      }

 218      pub fn scopes_intersect(&self, scope1: ast::NodeId, scope2: ast::NodeId)
 219                              -> bool {
 220          self.is_subscope_of(scope1, scope2) ||
 221          self.is_subscope_of(scope2, scope1)
 222      }

 224      pub fn is_subscope_of(&self,
 225                            subscope: ast::NodeId,
 226                            superscope: ast::NodeId)
 227                            -> bool {
 228          /*!
 229           * Returns true if `subscope` is equal to or is lexically
 230           * nested inside `superscope` and false otherwise.
 231           */
 232  
 233          let mut s = subscope;
 234          while superscope != s {
 235              match self.scope_map.borrow().find(&s) {
 236                  None => {
 237                      debug!("is_subscope_of({}, {}, s={})=false",
 238                             subscope, superscope, s);
 239  
 240                      return false;
 241                  }
 242                  Some(&scope) => s = scope
 243              }
 244          }
 245  
 246          debug!("is_subscope_of({}, {})=true",
 247                 subscope, superscope);
 248  
 249          return true;
 250      }

 252      pub fn sub_free_region(&self, sub: FreeRegion, sup: FreeRegion) -> bool {
 253          /*!
 254           * Determines whether two free regions have a subregion relationship
 255           * by walking the graph encoded in `free_region_map`.  Note that
 256           * it is possible that `sub != sup` and `sub <= sup` and `sup <= sub`
 257           * (that is, the user can give two different names to the same lifetime).
 258           */
 259  
 260          if sub == sup {
 261              return true;
 262          }
 263  
 264          // Do a little breadth-first-search here.  The `queue` list
 265          // doubles as a way to detect if we've seen a particular FR
 266          // before.  Note that we expect this graph to be an *extremely
 267          // shallow* tree.
 268          let mut queue = vec!(sub);
 269          let mut i = 0;
 270          while i < queue.len() {
 271              match self.free_region_map.borrow().find(queue.get(i)) {
 272                  Some(parents) => {
 273                      for parent in parents.iter() {
 274                          if *parent == sup {
 275                              return true;
 276                          }
 277  
 278                          if !queue.iter().any(|x| x == parent) {
 279                              queue.push(*parent);
 280                          }
 281                      }
 282                  }
 283                  None => {}
 284              }
 285              i += 1;
 286          }
 287          return false;
 288      }

 290      pub fn is_subregion_of(&self,
 291                             sub_region: ty::Region,
 292                             super_region: ty::Region)
 293                             -> bool {
 294          /*!
 295           * Determines whether one region is a subregion of another.  This is
 296           * intended to run *after inference* and sadly the logic is somewhat
 297           * duplicated with the code in infer.rs.
 298           */
 299  
 300          debug!("is_subregion_of(sub_region={:?}, super_region={:?})",
 301                 sub_region, super_region);
 302  
 303          sub_region == super_region || {
 304              match (sub_region, super_region) {
 305                  (_, ty::ReStatic) => {
 306                      true
 307                  }
 308  
 309                  (ty::ReScope(sub_scope), ty::ReScope(super_scope)) => {
 310                      self.is_subscope_of(sub_scope, super_scope)
 311                  }
 312  
 313                  (ty::ReScope(sub_scope), ty::ReFree(ref fr)) => {
 314                      self.is_subscope_of(sub_scope, fr.scope_id)
 315                  }
 316  
 317                  (ty::ReFree(sub_fr), ty::ReFree(super_fr)) => {
 318                      self.sub_free_region(sub_fr, super_fr)
 319                  }
 320  
 321                  _ => {
 322                      false
 323                  }
 324              }
 325          }
 326      }

 328      pub fn nearest_common_ancestor(&self,
 329                                     scope_a: ast::NodeId,
 330                                     scope_b: ast::NodeId)
 331                                     -> Option<ast::NodeId> {
 332          /*!
 333           * Finds the nearest common ancestor (if any) of two scopes.  That
 334           * is, finds the smallest scope which is greater than or equal to
 335           * both `scope_a` and `scope_b`.
 336           */
 337  
 338          if scope_a == scope_b { return Some(scope_a); }
 339  
 340          let a_ancestors = ancestors_of(self, scope_a);
 341          let b_ancestors = ancestors_of(self, scope_b);
 342          let mut a_index = a_ancestors.len() - 1u;
 343          let mut b_index = b_ancestors.len() - 1u;
 344  
 345          // Here, ~[ab]_ancestors is a vector going from narrow to broad.
 346          // The end of each vector will be the item where the scope is
 347          // defined; if there are any common ancestors, then the tails of
 348          // the vector will be the same.  So basically we want to walk
 349          // backwards from the tail of each vector and find the first point
 350          // where they diverge.  If one vector is a suffix of the other,
 351          // then the corresponding scope is a superscope of the other.
 352  
 353          if *a_ancestors.get(a_index) != *b_ancestors.get(b_index) {
 354              return None;
 355          }
 356  
 357          loop {
 358              // Loop invariant: a_ancestors[a_index] == b_ancestors[b_index]
 359              // for all indices between a_index and the end of the array
 360              if a_index == 0u { return Some(scope_a); }
 361              if b_index == 0u { return Some(scope_b); }
 362              a_index -= 1u;
 363              b_index -= 1u;
 364              if *a_ancestors.get(a_index) != *b_ancestors.get(b_index) {
 365                  return Some(*a_ancestors.get(a_index + 1u));
 366              }
 367          }
 368  
 369          fn ancestors_of(this: &RegionMaps, scope: ast::NodeId)
 370              -> Vec<ast::NodeId> {
 371              // debug!("ancestors_of(scope={})", scope);
 372              let mut result = vec!(scope);
 373              let mut scope = scope;
 374              loop {
 375                  match this.scope_map.borrow().find(&scope) {
 376                      None => return result,
 377                      Some(&superscope) => {
 378                          result.push(superscope);
 379                          scope = superscope;
 380                      }
 381                  }
 382                  // debug!("ancestors_of_loop(scope={})", scope);
 383              }
 384          }
 385      }
 386  }

 392                          _sp: Span) {

 396  }

 398  /// Records the lifetime of a local variable as `cx.var_parent`
 399  fn record_var_lifetime(visitor: &mut RegionResolutionVisitor,
 400                         cx: Context,
 401                         var_id: ast::NodeId,
 402                         _sp: Span) {
 403      match cx.var_parent {
 404          Some(parent_id) => {
 405              visitor.region_maps.record_var_scope(var_id, parent_id);
 406          }
 407          None => {

 415  fn resolve_block(visitor: &mut RegionResolutionVisitor,
 416                   blk: &ast::Block,
 417                   cx: Context) {
 418      debug!("resolve_block(blk.id={})", blk.id);
 419  
 420      // Record the parent of this block.
 421      record_superlifetime(visitor, cx, blk.id, blk.span);
 422  
 423      // We treat the tail expression in the block (if any) somewhat
 424      // differently from the statements. The issue has to do with
 425      // temporary lifetimes. If the user writes:
 426      //
 427      //   {
 428      //     ... (&foo()) ...
 429      //   }
 430      //
 431  
 432      let subcx = Context {var_parent: Some(blk.id), parent: Some(blk.id)};
 433      visit::walk_block(visitor, blk, subcx);
 434  }

 436  fn resolve_arm(visitor: &mut RegionResolutionVisitor,
 437                 arm: &ast::Arm,
 438                 cx: Context) {

 481  fn resolve_expr(visitor: &mut RegionResolutionVisitor,
 482                  expr: &ast::Expr,
 483                  cx: Context) {
 484      debug!("resolve_expr(expr.id={})", expr.id);
 485  
 486      record_superlifetime(visitor, cx, expr.id, expr.span);
 487  
 488      let mut new_cx = cx;
 489      new_cx.parent = Some(expr.id);
 490      match expr.node {
 491          // Conditional or repeating scopes are always terminating
 492          // scopes, meaning that temporaries cannot outlive them.
 493          // This ensures fixed size stacks.
 494  
 495          ast::ExprBinary(ast::BiAnd, _, r) |
 496          ast::ExprBinary(ast::BiOr, _, r) => {
 497              // For shortcircuiting operators, mark the RHS as a terminating
 498              // scope since it only executes conditionally.
 499              visitor.region_maps.mark_as_terminating_scope(r.id);
 500          }
 501  
 502          ast::ExprIf(_, then, Some(otherwise)) => {
 503              visitor.region_maps.mark_as_terminating_scope(then.id);
 504              visitor.region_maps.mark_as_terminating_scope(otherwise.id);
 505          }
 506  
 507          ast::ExprIf(expr, then, None) => {
 508              visitor.region_maps.mark_as_terminating_scope(expr.id);
 509              visitor.region_maps.mark_as_terminating_scope(then.id);
 510          }
 511  
 512          ast::ExprLoop(body, _) => {
 513              visitor.region_maps.mark_as_terminating_scope(body.id);
 514          }
 515  
 516          ast::ExprWhile(expr, body) => {
 517              visitor.region_maps.mark_as_terminating_scope(expr.id);
 518              visitor.region_maps.mark_as_terminating_scope(body.id);
 519          }
 520  
 521          ast::ExprMatch(..) => {
 522              new_cx.var_parent = Some(expr.id);
 523          }
 524  
 525          ast::ExprAssignOp(..) | ast::ExprIndex(..) |
 526          ast::ExprUnary(..) | ast::ExprCall(..) | ast::ExprMethodCall(..) => {
 527              // FIXME(#6268) Nested method calls
 528              //
 529              // The lifetimes for a call or method call look as follows:
 530              //
 531              // call.id
 532              // - arg0.id
 533              // - ...
 534              // - argN.id
 535              // - call.callee_id
 536              //
 537              // The idea is that call.callee_id represents *the time when
 538              // the invoked function is actually running* and call.id
 539              // represents *the time to prepare the arguments and make the
 540              // call*.  See the section "Borrows in Calls" borrowck/doc.rs
 541              // for an extended explanantion of why this distinction is
 542              // important.
 543              //
 544              // record_superlifetime(new_cx, expr.callee_id);
 545          }
 546  
 547          _ => {}
 548      };
 549  
 550  
 551      visit::walk_expr(visitor, expr, new_cx);
 552  }

 554  fn resolve_local(visitor: &mut RegionResolutionVisitor,
 555                   local: &ast::Local,
 556                   cx: Context) {
 557      debug!("resolve_local(local.id={},local.init={})",
 558             local.id,local.init.is_some());
 559  
 560      let blk_id = match cx.var_parent {
 561          Some(id) => id,
 562          None => {
 563              visitor.sess.span_bug(
 564                  local.span,
 565                  "local without enclosing block");
 566          }
 567      };
 568  
 569      // For convenience in trans, associate with the local-id the var
 570      // scope that will be used for any bindings declared in this
 571      // pattern.
 572      visitor.region_maps.record_var_scope(local.id, blk_id);
 573  
 574      // As an exception to the normal rules governing temporary
 575      // lifetimes, initializers in a let have a temporary lifetime
 576      // of the enclosing block. This means that e.g. a program
 577      // like the following is legal:
 578      //
 579      //     let ref x = HashMap::new();
 580      //
 581      // Because the hash map will be freed in the enclosing block.
 582      //
 583      // We express the rules more formally based on 3 grammars (defined
 584      // fully in the helpers below that implement them):
 585      //
 586      // 1. `E&`, which matches expressions like `&<rvalue>` that
 587      //    own a pointer into the stack.
 588      //
 589      // 2. `P&`, which matches patterns like `ref x` or `(ref x, ref
 590      //    y)` that produce ref bindings into the value they are
 591      //    matched against or something (at least partially) owned by
 592      //    the value they are matched against. (By partially owned,
 593      //    I mean that creating a binding into a ref-counted or managed value
 594      //    would still count.)
 595      //
 596      // 3. `ET`, which matches both rvalues like `foo()` as well as lvalues
 597      //    based on rvalues like `foo().x[2].y`.
 598      //
 599      // A subexpression `<rvalue>` that appears in a let initializer
 600      // `let pat [: ty] = expr` has an extended temporary lifetime if
 601      // any of the following conditions are met:
 602      //
 603      // A. `pat` matches `P&` and `expr` matches `ET`
 604      //    (covers cases where `pat` creates ref bindings into an rvalue
 605      //     produced by `expr`)
 606      // B. `ty` is a borrowed pointer and `expr` matches `ET`
 607      //    (covers cases where coercion creates a borrow)
 608      // C. `expr` matches `E&`
 609      //    (covers cases `expr` borrows an rvalue that is then assigned
 610      //     to memory (at least partially) owned by the binding)
 611      //
 612      // Here are some examples hopefully giving an intution where each
 613      // rule comes into play and why:
 614      //
 615      // Rule A. `let (ref x, ref y) = (foo().x, 44)`. The rvalue `(22, 44)`
 616      // would have an extended lifetime, but not `foo()`.
 617      //
 618      // Rule B. `let x: &[...] = [foo().x]`. The rvalue `[foo().x]`
 619      // would have an extended lifetime, but not `foo()`.
 620      //
 621      // Rule C. `let x = &foo().x`. The rvalue ``foo()` would have extended
 622      // lifetime.
 623      //
 624      // In some cases, multiple rules may apply (though not to the same
 625      // rvalue). For example:
 626      //
 627      //     let ref x = [&a(), &b()];
 628      //
 629      // Here, the expression `[...]` has an extended lifetime due to rule
 630      // A, but the inner rvalues `a()` and `b()` have an extended lifetime
 631      // due to rule C.
 632      //
 633      // FIXME(#6308) -- Note that `[]` patterns work more smoothly post-DST.
 634  
 635      match local.init {
 636          Some(expr) => {
 637              record_rvalue_scope_if_borrow_expr(visitor, expr, blk_id);
 638  
 639              if is_binding_pat(local.pat) || is_borrowed_ty(local.ty) {
 640                  record_rvalue_scope(visitor, expr, blk_id);
 641              }
 642          }
 643  
 644          None => { }
 645      }
 646  
 647      visit::walk_local(visitor, local, cx);
 648  
 649      fn is_binding_pat(pat: &ast::Pat) -> bool {
 650          /*!
 651           * True if `pat` match the `P&` nonterminal:
 652           *
 653           *     P& = ref X
 654           *        | StructName { ..., P&, ... }
 655           *        | VariantName(..., P&, ...)
 656           *        | [ ..., P&, ... ]
 657           *        | ( ..., P&, ... )
 658           *        | ~P&
 659           *        | box P&
 660           */
 661  
 662          match pat.node {
 663              ast::PatIdent(ast::BindByRef(_), _, _) => true,
 664  
 665              ast::PatStruct(_, ref field_pats, _) => {
 666                  field_pats.iter().any(|fp| is_binding_pat(fp.pat))
 667              }
 668  
 669              ast::PatVec(ref pats1, ref pats2, ref pats3) => {
 670                  pats1.iter().any(|&p| is_binding_pat(p)) ||
 671                  pats2.iter().any(|&p| is_binding_pat(p)) ||
 672                  pats3.iter().any(|&p| is_binding_pat(p))
 673              }
 674  
 675              ast::PatEnum(_, Some(ref subpats)) |
 676              ast::PatTup(ref subpats) => {
 677                  subpats.iter().any(|&p| is_binding_pat(p))
 678              }
 679  
 680              ast::PatUniq(subpat) => {
 681                  is_binding_pat(subpat)
 682              }
 683  
 684              _ => false,
 685          }
 686      }

 688      fn is_borrowed_ty(ty: &ast::Ty) -> bool {
 689          /*!
 690           * True if `ty` is a borrowed pointer type
 691           * like `&int` or `&[...]`.
 692           */
 693  
 694          match ty.node {
 695              ast::TyRptr(..) => true,
 696              _ => false
 697          }
 698      }

 700      fn record_rvalue_scope_if_borrow_expr(visitor: &mut RegionResolutionVisitor,
 701                                            expr: &ast::Expr,
 702                                            blk_id: ast::NodeId) {
 703          /*!
 704           * If `expr` matches the `E&` grammar, then records an extended
 705           * rvalue scope as appropriate:
 706           *
 707           *     E& = & ET
 708           *        | StructName { ..., f: E&, ... }
 709           *        | [ ..., E&, ... ]
 710           *        | ( ..., E&, ... )
 711           *        | {...; E&}
 712           *        | ~E&
 713           *        | E& as ...
 714           *        | ( E& )
 715           */
 716  
 717          match expr.node {
 718              ast::ExprAddrOf(_, subexpr) => {
 719                  record_rvalue_scope_if_borrow_expr(visitor, subexpr, blk_id);
 720                  record_rvalue_scope(visitor, subexpr, blk_id);
 721              }
 722              ast::ExprStruct(_, ref fields, _) => {
 723                  for field in fields.iter() {
 724                      record_rvalue_scope_if_borrow_expr(
 725                          visitor, field.expr, blk_id);
 726                  }
 727              }
 728              ast::ExprVstore(subexpr, _) => {
 729                  visitor.region_maps.record_rvalue_scope(subexpr.id, blk_id);
 730                  record_rvalue_scope_if_borrow_expr(visitor, subexpr, blk_id);
 731              }
 732              ast::ExprVec(ref subexprs, _) |
 733              ast::ExprTup(ref subexprs) => {
 734                  for &subexpr in subexprs.iter() {
 735                      record_rvalue_scope_if_borrow_expr(
 736                          visitor, subexpr, blk_id);
 737                  }
 738              }
 739              ast::ExprUnary(ast::UnUniq, subexpr) => {
 740                  record_rvalue_scope_if_borrow_expr(visitor, subexpr, blk_id);
 741              }
 742              ast::ExprCast(subexpr, _) |
 743              ast::ExprParen(subexpr) => {
 744                  record_rvalue_scope_if_borrow_expr(visitor, subexpr, blk_id)
 745              }
 746              ast::ExprBlock(ref block) => {
 747                  match block.expr {
 748                      Some(subexpr) => {
 749                          record_rvalue_scope_if_borrow_expr(
 750                              visitor, subexpr, blk_id);
 751                      }
 752                      None => { }
 753                  }
 754              }
 755              _ => {
 756              }
 757          }
 758      }

 760      fn record_rvalue_scope<'a>(visitor: &mut RegionResolutionVisitor,
 761                                 expr: &'a ast::Expr,
 762                                 blk_id: ast::NodeId) {
 763          /*!
 764           * Applied to an expression `expr` if `expr` -- or something
 765           * owned or partially owned by `expr` -- is going to be
 766           * indirectly referenced by a variable in a let statement. In
 767           * that case, the "temporary lifetime" or `expr` is extended
 768           * to be the block enclosing the `let` statement.
 769           *
 770           * More formally, if `expr` matches the grammar `ET`, record
 771           * the rvalue scope of the matching `<rvalue>` as `blk_id`:
 772           *
 773           *     ET = *ET
 774           *        | ET[...]
 775           *        | ET.f
 776           *        | (ET)
 777           *        | <rvalue>
 778           *
 779           * Note: ET is intended to match "rvalues or
 780           * lvalues based on rvalues".
 781           */
 782  
 783          let mut expr = expr;
 784          loop {
 785              // Note: give all the expressions matching `ET` with the
 786              // extended temporary lifetime, not just the innermost rvalue,
 787              // because in trans if we must compile e.g. `*rvalue()`
 788              // into a temporary, we request the temporary scope of the
 789              // outer expression.
 790              visitor.region_maps.record_rvalue_scope(expr.id, blk_id);
 791  
 792              match expr.node {
 793                  ast::ExprAddrOf(_, ref subexpr) |
 794                  ast::ExprUnary(ast::UnDeref, ref subexpr) |
 795                  ast::ExprField(ref subexpr, _, _) |
 796                  ast::ExprIndex(ref subexpr, _) |
 797                  ast::ExprParen(ref subexpr) => {
 798                      let subexpr: &'a @Expr = subexpr; // FIXME(#11586)
 799                      expr = &**subexpr;
 800                  }
 801                  _ => {
 802                      return;
 803                  }
 804              }
 805          }
 806      }
 807  }

 814      visit::walk_item(visitor, item, new_cx);

 817  fn resolve_fn(visitor: &mut RegionResolutionVisitor,
 818                fk: &FnKind,
 819                decl: &ast::FnDecl,
 820                body: &ast::Block,
 821                sp: Span,
 822                id: ast::NodeId,
 823                cx: Context) {
 824      debug!("region::resolve_fn(id={}, \
 825                                 span={:?}, \
 826                                 body.id={}, \
 827                                 cx.parent={})",
 828             id,
 829             visitor.sess.codemap().span_to_str(sp),
 830             body.id,
 831             cx.parent);
 832  
 833      visitor.region_maps.mark_as_terminating_scope(body.id);
 834  
 835      // The arguments and `self` are parented to the body of the fn.
 836      let decl_cx = Context {parent: Some(body.id),
 837                             var_parent: Some(body.id)};
 838      visit::walk_fn_decl(visitor, decl, decl_cx);
 839  
 840      // The body of the fn itself is either a root scope (top-level fn)
 841      // or it continues with the inherited scope (closures).
 842      let body_cx = match *fk {
 843          visit::FkItemFn(..) | visit::FkMethod(..) => {
 844              Context {parent: None, var_parent: None, ..cx}
 845          }
 846          visit::FkFnBlock(..) => {
 847              // FIXME(#3696) -- at present we are place the closure body
 848              // within the region hierarchy exactly where it appears lexically.
 849              // This is wrong because the closure may live longer
 850              // than the enclosing expression. We should probably fix this,
 851              // but the correct fix is a bit subtle, and I am also not sure
 852              // that the present approach is unsound -- it may not permit
 853              // any illegal programs. See issue for more details.
 854              cx
 855          }
 856      };
 857      visitor.visit_block(body, body_cx);
 858  }

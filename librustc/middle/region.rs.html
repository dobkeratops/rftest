  42  /**
  43  The region maps encode information about region relationships.
  44  
  45  - `scope_map` maps from:
  46    - an expression to the expression or block encoding the maximum
  47      (static) lifetime of a value produced by that expression.  This is
  48      generally the innermost call, statement, match, or block.
  49    - a variable or binding id to the block in which that variable is declared.
  50  - `free_region_map` maps from:
  51    - a free region `a` to a list of free regions `bs` such that
  52      `a <= b for all b in bs`
  53    - the free region map is populated during type check as we check
  54      each function. See the function `relate_free_regions` for
  55      more information.
  56  - `cleanup_scopes` includes scopes where trans cleanups occur
  57    - this is intended to reflect the current state of trans, not
  58      necessarily how I think things ought to work
  59  */
  60  pub struct RegionMaps {
  61      priv scope_map: HashMap<ast::NodeId, ast::NodeId>,
  62      priv free_region_map: HashMap<FreeRegion, ~[FreeRegion]>,
  63      priv cleanup_scopes: HashSet<ast::NodeId>
  64  }
  65

  72      region_maps: @mut RegionMaps,
  73  
  74      // Scope where variables should be parented to
  75      var_parent: Option<ast::NodeId>,
  76  
  77      // Innermost enclosing expression
  78      parent: Option<ast::NodeId>,
  79  }

  82      pub fn relate_free_regions(&mut self, sub: FreeRegion, sup: FreeRegion) {
  83          match self.free_region_map.find_mut(&sub) {
  84              Some(sups) => {
  85                  if !sups.iter().any(|x| x == &sup) {
  86                      sups.push(sup);
  87                  }
  88                  return;
  89              }
  90              None => {}
  91          }
  92  
  93          debug!("relate_free_regions(sub=%?, sup=%?)", sub, sup);
  94  
  95          self.free_region_map.insert(sub, ~[sup]);
  96      }

  98      pub fn record_parent(&mut self, sub: ast::NodeId, sup: ast::NodeId) {
  99          debug!("record_parent(sub=%?, sup=%?)", sub, sup);
 100          assert!(sub != sup);
 101  
 102          self.scope_map.insert(sub, sup);
 103      }

 105      pub fn record_cleanup_scope(&mut self, scope_id: ast::NodeId) {

 107          //! from within regionck.  We wait until regionck because we do
 108          //! not know which operators are overloaded until that point,

 112      }

 114      pub fn opt_encl_scope(&self, id: ast::NodeId) -> Option<ast::NodeId> {
 115          //! Returns the narrowest scope that encloses `id`, if any.
 116  
 117          self.scope_map.find(&id).map_move(|x| *x)
 118      }

 120      pub fn encl_scope(&self, id: ast::NodeId) -> ast::NodeId {
 121          //! Returns the narrowest scope that encloses `id`, if any.
 122  
 123          match self.scope_map.find(&id) {
 124              Some(&r) => r,
 125              None => { fail!("No enclosing scope for id %?", id); }
 126          }
 127      }

 129      pub fn is_cleanup_scope(&self, scope_id: ast::NodeId) -> bool {
 130          self.cleanup_scopes.contains(&scope_id)
 131      }

 133      pub fn cleanup_scope(&self, expr_id: ast::NodeId) -> ast::NodeId {
 134          //! Returns the scope when temps in expr will be cleaned up
 135  
 136          let mut id = self.encl_scope(expr_id);
 137          while !self.cleanup_scopes.contains(&id) {
 138              id = self.encl_scope(id);
 139          }
 140          return id;
 141      }

 149      pub fn scopes_intersect(&self, scope1: ast::NodeId, scope2: ast::NodeId)
 150                              -> bool {
 151          self.is_subscope_of(scope1, scope2) ||
 152          self.is_subscope_of(scope2, scope1)
 153      }

 155      pub fn is_subscope_of(&self,
 156                            subscope: ast::NodeId,
 157                            superscope: ast::NodeId)
 158                            -> bool {
 159          /*!
 160           * Returns true if `subscope` is equal to or is lexically
 161           * nested inside `superscope` and false otherwise.
 162           */
 163  
 164          let mut s = subscope;
 165          while superscope != s {
 166              match self.scope_map.find(&s) {
 167                  None => {
 168                      debug!("is_subscope_of(%?, %?, s=%?)=false",
 169                             subscope, superscope, s);
 170  
 171                      return false;
 172                  }
 173                  Some(&scope) => s = scope
 174              }
 175          }
 176  
 177          debug!("is_subscope_of(%?, %?)=true",
 178                 subscope, superscope);
 179  
 180          return true;
 181      }

 183      pub fn sub_free_region(&self, sub: FreeRegion, sup: FreeRegion) -> bool {
 184          /*!
 185           * Determines whether two free regions have a subregion relationship
 186           * by walking the graph encoded in `free_region_map`.  Note that
 187           * it is possible that `sub != sup` and `sub <= sup` and `sup <= sub`
 188           * (that is, the user can give two different names to the same lifetime).
 189           */
 190  
 191          if sub == sup {
 192              return true;
 193          }
 194  
 195          // Do a little breadth-first-search here.  The `queue` list
 196          // doubles as a way to detect if we've seen a particular FR
 197          // before.  Note that we expect this graph to be an *extremely
 198          // shallow* tree.
 199          let mut queue = ~[sub];
 200          let mut i = 0;
 201          while i < queue.len() {
 202              match self.free_region_map.find(&queue[i]) {
 203                  Some(parents) => {
 204                      for parent in parents.iter() {
 205                          if *parent == sup {
 206                              return true;
 207                          }
 208  
 209                          if !queue.iter().any(|x| x == parent) {
 210                              queue.push(*parent);
 211                          }
 212                      }
 213                  }
 214                  None => {}
 215              }
 216              i += 1;
 217          }
 218          return false;
 219      }

 221      pub fn is_subregion_of(&self,
 222                             sub_region: ty::Region,
 223                             super_region: ty::Region)
 224                             -> bool {
 225          /*!
 226           * Determines whether one region is a subregion of another.  This is
 227           * intended to run *after inference* and sadly the logic is somewhat
 228           * duplicated with the code in infer.rs.
 229           */
 230  
 231          debug!("is_subregion_of(sub_region=%?, super_region=%?)",
 232                 sub_region, super_region);
 233  
 234          sub_region == super_region || {
 235              match (sub_region, super_region) {
 236                  (_, ty::re_static) => {
 237                      true
 238                  }
 239  
 240                  (ty::re_scope(sub_scope), ty::re_scope(super_scope)) => {
 241                      self.is_subscope_of(sub_scope, super_scope)
 242                  }
 243  
 244                  (ty::re_scope(sub_scope), ty::re_free(ref fr)) => {
 245                      self.is_subscope_of(sub_scope, fr.scope_id)
 246                  }
 247  
 248                  (ty::re_free(sub_fr), ty::re_free(super_fr)) => {
 249                      self.sub_free_region(sub_fr, super_fr)
 250                  }
 251  
 252                  _ => {
 253                      false
 254                  }
 255              }
 256          }
 257      }

 259      pub fn nearest_common_ancestor(&self,
 260                                     scope_a: ast::NodeId,
 261                                     scope_b: ast::NodeId)
 262                                     -> Option<ast::NodeId> {
 263          /*!
 264           * Finds the nearest common ancestor (if any) of two scopes.  That
 265           * is, finds the smallest scope which is greater than or equal to
 266           * both `scope_a` and `scope_b`.
 267           */
 268  
 269          if scope_a == scope_b { return Some(scope_a); }
 270  
 271          let a_ancestors = ancestors_of(self, scope_a);
 272          let b_ancestors = ancestors_of(self, scope_b);
 273          let mut a_index = a_ancestors.len() - 1u;
 274          let mut b_index = b_ancestors.len() - 1u;
 275  
 276          // Here, ~[ab]_ancestors is a vector going from narrow to broad.
 277          // The end of each vector will be the item where the scope is
 278          // defined; if there are any common ancestors, then the tails of
 279          // the vector will be the same.  So basically we want to walk
 280          // backwards from the tail of each vector and find the first point
 281          // where they diverge.  If one vector is a suffix of the other,
 282          // then the corresponding scope is a superscope of the other.
 283  
 284          if a_ancestors[a_index] != b_ancestors[b_index] {
 285              return None;
 286          }
 287  
 288          loop {
 289              // Loop invariant: a_ancestors[a_index] == b_ancestors[b_index]
 290              // for all indices between a_index and the end of the array
 291              if a_index == 0u { return Some(scope_a); }
 292              if b_index == 0u { return Some(scope_b); }
 293              a_index -= 1u;
 294              b_index -= 1u;
 295              if a_ancestors[a_index] != b_ancestors[b_index] {
 296                  return Some(a_ancestors[a_index + 1u]);
 297              }
 298          }
 299  
 300          fn ancestors_of(this: &RegionMaps, scope: ast::NodeId)
 301              -> ~[ast::NodeId]
 302          {
 303              // debug!("ancestors_of(scope=%d)", scope);
 304              let mut result = ~[scope];
 305              let mut scope = scope;
 306              loop {
 307                  match this.scope_map.find(&scope) {
 308                      None => return result,
 309                      Some(&superscope) => {
 310                          result.push(superscope);
 311                          scope = superscope;
 312                      }
 313                  }
 314                  // debug!("ancestors_of_loop(scope=%d)", scope);
 315              }
 316          }
 317      }
 318  }

 356  fn resolve_stmt(visitor: &mut RegionResolutionVisitor,
 357                  stmt: @ast::Stmt,
 358                  cx: Context) {
 359      match stmt.node {
 360          ast::StmtDecl(*) => {
 361              visit::walk_stmt(visitor, stmt, cx);
 362          }
 363          ast::StmtExpr(_, stmt_id) |
 364          ast::StmtSemi(_, stmt_id) => {
 365              parent_to_expr(cx, stmt_id, stmt.span);
 366              let expr_cx = Context {parent: Some(stmt_id), ..cx};
 367              visit::walk_stmt(visitor, stmt, expr_cx);

 369          ast::StmtMac(*) => cx.sess.bug("unexpanded macro")
 370      }

 373  fn resolve_expr(visitor: &mut RegionResolutionVisitor,
 374                  expr: @ast::Expr,
 375                  cx: Context) {
 376      parent_to_expr(cx, expr.id, expr.span);
 377  
 378      let mut new_cx = cx;
 379      new_cx.parent = Some(expr.id);
 380      match expr.node {
 381          ast::ExprAssignOp(*) | ast::ExprIndex(*) | ast::ExprBinary(*) |
 382          ast::ExprUnary(*) | ast::ExprCall(*) | ast::ExprMethodCall(*) => {
 383              // FIXME(#6268) Nested method calls
 384              //
 385              // The lifetimes for a call or method call look as follows:
 386              //
 387              // call.id
 388              // - arg0.id
 389              // - ...
 390              // - argN.id
 391              // - call.callee_id
 392              //
 393              // The idea is that call.callee_id represents *the time when
 394              // the invoked function is actually running* and call.id
 395              // represents *the time to prepare the arguments and make the
 396              // call*.  See the section "Borrows in Calls" borrowck/doc.rs
 397              // for an extended explanantion of why this distinction is
 398              // important.
 399              //
 400              // parent_to_expr(new_cx, expr.callee_id);
 401          }
 402  
 403          ast::ExprMatch(*) => {
 404              new_cx.var_parent = Some(expr.id);
 405          }
 406  
 407          _ => {}
 408      };
 409  
 410  
 411      visit::walk_expr(visitor, expr, new_cx);
 412  }

 430  fn resolve_fn(visitor: &mut RegionResolutionVisitor,
 431                fk: &visit::fn_kind,
 432                decl: &ast::fn_decl,
 433                body: &ast::Block,
 434                sp: Span,
 435                id: ast::NodeId,
 436                cx: Context) {
 437      debug!("region::resolve_fn(id=%?, \
 438                                 span=%?, \
 439                                 body.id=%?, \
 440                                 cx.parent=%?)",
 441             id,
 442             cx.sess.codemap.span_to_str(sp),
 443             body.id,
 444             cx.parent);
 445  
 446      // The arguments and `self` are parented to the body of the fn.
 447      let decl_cx = Context {parent: Some(body.id),
 448                             var_parent: Some(body.id),
 449                             ..cx};
 450      match *fk {
 451          visit::fk_method(_, _, method) => {
 452              cx.region_maps.record_parent(method.self_id, body.id);
 453          }
 454          _ => {}
 455      }
 456      visit::walk_fn_decl(visitor, decl, decl_cx);
 457  
 458      // The body of the fn itself is either a root scope (top-level fn)
 459      // or it continues with the inherited scope (closures).
 460      let body_cx = match *fk {
 461          visit::fk_item_fn(*) |
 462          visit::fk_method(*) => {

 464          }

 466          visit::fk_fn_block(*) => {
 467              cx
 468          }
 469      };
 470      visitor.visit_block(body, body_cx);
 471  }

 524  // ___________________________________________________________________________
 525  // Determining region parameterization
 526  //
 527  // Infers which type defns must be region parameterized---this is done
 528  // by scanning their contents to see whether they reference a region
 529  // type, directly or indirectly.  This is a fixed-point computation.
 530  //
 531  // We do it in two passes.  First we walk the AST and construct a map
 532  // from each type defn T1 to other defns which make use of it.  For example,
 533  // if we have a type like:
 534  //
 535  //    type S = *int;
 536  //    type T = S;
 537  //
 538  // Then there would be a map entry from S to T.  During the same walk,
 539  // we also construct add any types that reference regions to a set and
 540  // a worklist.  We can then process the worklist, propagating indirect
 541  // dependencies until a fixed point is reached.
 542  
 543  pub type region_paramd_items = @mut HashMap<ast::NodeId, region_variance>;
 544  
 545  #[deriving(Eq)]
 546  pub struct region_dep {
 547      ambient_variance: region_variance,
 548      id: ast::NodeId
 549  }
 550  
 551  pub struct DetermineRpCtxt {
 552      sess: Session,
 553      ast_map: ast_map::map,
 554      def_map: resolve::DefMap,
 555      region_paramd_items: region_paramd_items,
 556      dep_map: @mut HashMap<ast::NodeId, @mut ~[region_dep]>,
 557      worklist: ~[ast::NodeId],
 558  
 559      // the innermost enclosing item id
 560      item_id: ast::NodeId,
 561  
 562      // true when we are within an item but not within a method.
 563      // see long discussion on region_is_relevant().
 564      anon_implies_rp: bool,
 565  
 566      // encodes the context of the current type; invariant if
 567      // mutable, covariant otherwise
 568      ambient_variance: region_variance,
 569  }
 570  
 571  pub fn join_variance(variance1: region_variance,
 572                       variance2: region_variance)
 573                    -> region_variance {
 574      match (variance1, variance2) {
 575        (rv_invariant, _) => {rv_invariant}
 576        (_, rv_invariant) => {rv_invariant}
 577        (rv_covariant, rv_contravariant) => {rv_invariant}
 578        (rv_contravariant, rv_covariant) => {rv_invariant}
 579        (rv_covariant, rv_covariant) => {rv_covariant}
 580        (rv_contravariant, rv_contravariant) => {rv_contravariant}
 581      }
 582  }
 583  
 584  /// Combines the ambient variance with the variance of a
 585  /// particular site to yield the final variance of the reference.
 586  ///
 587  /// Example: if we are checking function arguments then the ambient
 588  /// variance is contravariant.  If we then find a `&'r T` pointer, `r`
 589  /// appears in a co-variant position.  This implies that this
 590  /// occurrence of `r` is contra-variant with respect to the current
 591  /// item, and hence the function returns `rv_contravariant`.
 592  pub fn add_variance(ambient_variance: region_variance,
 593                      variance: region_variance)
 594                   -> region_variance {
 595      match (ambient_variance, variance) {
 596        (rv_invariant, _) => rv_invariant,
 597        (_, rv_invariant) => rv_invariant,
 598        (rv_covariant, c) => c,
 599        (c, rv_covariant) => c,
 600        (rv_contravariant, rv_contravariant) => rv_covariant
 601      }
 602  }
 603  
 604  impl DetermineRpCtxt {
 605      pub fn add_variance(&self, variance: region_variance) -> region_variance {
 606          add_variance(self.ambient_variance, variance)
 607      }
 608  
 609      /// Records that item `id` is region-parameterized with the
 610      /// variance `variance`.  If `id` was already parameterized, then
 611      /// the new variance is joined with the old variance.
 612      pub fn add_rp(&mut self, id: ast::NodeId, variance: region_variance) {
 613          assert!(id != 0);
 614          let old_variance = self.region_paramd_items.find(&id).map_move(|x| *x);
 615          let joined_variance = match old_variance {
 616            None => variance,
 617            Some(v) => join_variance(v, variance)
 618          };
 619  
 620          debug!("add_rp() variance for %s: %? == %? ^ %?",
 621                 ast_map::node_id_to_str(self.ast_map, id,
 622                                         token::get_ident_interner()),
 623                 joined_variance, old_variance, variance);
 624  
 625          if Some(joined_variance) != old_variance {
 626              let region_paramd_items = self.region_paramd_items;
 627              region_paramd_items.insert(id, joined_variance);
 628              self.worklist.push(id);
 629          }
 630      }
 631  
 632      /// Indicates that the region-parameterization of the current item
 633      /// is dependent on the region-parameterization of the item
 634      /// `from`.  Put another way, it indicates that the current item
 635      /// contains a value of type `from`, so if `from` is
 636      /// region-parameterized, so is the current item.
 637      pub fn add_dep(&mut self, from: ast::NodeId) {
 638          debug!("add dependency from %d -> %d (%s -> %s) with variance %?",
 639                 from, self.item_id,
 640                 ast_map::node_id_to_str(self.ast_map, from,
 641                                         token::get_ident_interner()),
 642                 ast_map::node_id_to_str(self.ast_map, self.item_id,
 643                                         token::get_ident_interner()),
 644                 self.ambient_variance);
 645          let vec = do self.dep_map.find_or_insert_with(from) |_| {
 646              @mut ~[]
 647          };
 648          let dep = region_dep {
 649              ambient_variance: self.ambient_variance,
 650              id: self.item_id
 651          };
 652          if !vec.iter().any(|x| x == &dep) { vec.push(dep); }
 653      }
 654  
 655      // Determines whether a reference to a region that appears in the
 656      // AST implies that the enclosing type is region-parameterized (RP).
 657      // This point is subtle.  Here are some examples to make it more
 658      // concrete.
 659      //
 660      // 1. impl foo for &int { ... }
 661      // 2. impl foo for &'self int { ... }
 662      // 3. impl foo for bar { fn m(@self) -> &'self int { ... } }
 663      // 4. impl foo for bar { fn m(&self) -> &'self int { ... } }
 664      // 5. impl foo for bar { fn m(&self) -> &int { ... } }
 665      //
 666      // In case 1, the anonymous region is being referenced,
 667      // but it appears in a context where the anonymous region
 668      // resolves to self, so the impl foo is RP.
 669      //
 670      // In case 2, the self parameter is written explicitly.
 671      //
 672      // In case 3, the method refers to the region `self`, so that
 673      // implies that the impl must be region parameterized.  (If the
 674      // type bar is not region parameterized, that is an error, because
 675      // the self region is effectively unconstrained, but that is
 676      // detected elsewhere).
 677      //
 678      // In case 4, the method refers to the region `self`, but the
 679      // `self` region is bound by the `&self` receiver, and so this
 680      // does not require that `bar` be RP.
 681      //
 682      // In case 5, the anonymous region is referenced, but it
 683      // bound by the method, so it does not refer to self.  This impl
 684      // need not be region parameterized.
 685      //
 686      // Normally, & or &self implies that the enclosing item is RP.
 687      // However, within a function, & is always bound.  Within a method
 688      // with &self type, &self is also bound.  We detect those last two
 689      // cases via flags (anon_implies_rp and self_implies_rp) that are
 690      // true when the anon or self region implies RP.
 691      pub fn region_is_relevant(&self, r: &Option<ast::Lifetime>) -> bool {
 692          match r {
 693              &None => {
 694                  self.anon_implies_rp
 695              }
 696              &Some(ref l) if l.ident == special_idents::statik => {
 697                  false
 698              }
 699              &Some(ref l) if l.ident == special_idents::self_ => {
 700                  true
 701              }
 702              &Some(_) => {
 703                  false
 704              }
 705          }
 706      }

 708      pub fn with(@mut self,
 709                  item_id: ast::NodeId,
 710                  anon_implies_rp: bool,
 711                  f: &fn()) {
 712          let old_item_id = self.item_id;
 713          let old_anon_implies_rp = self.anon_implies_rp;
 714          self.item_id = item_id;
 715          self.anon_implies_rp = anon_implies_rp;
 716          debug!("with_item_id(%d, %b)",
 717                 item_id,
 718                 anon_implies_rp);
 719          let _i = ::util::common::indenter();
 720          f();
 721          self.item_id = old_item_id;
 722          self.anon_implies_rp = old_anon_implies_rp;
 723      }
 724  
 725      pub fn with_ambient_variance(@mut self,
 726                                   variance: region_variance,
 727                                   f: &fn()) {
 728          let old_ambient_variance = self.ambient_variance;
 729          self.ambient_variance = self.add_variance(variance);
 730          f();
 731          self.ambient_variance = old_ambient_variance;
 732      }
 733  }

 735  fn determine_rp_in_item(visitor: &mut DetermineRpVisitor,
 736                          item: @ast::item,
 737                          cx: @mut DetermineRpCtxt) {
 738      do cx.with(item.id, true) {
 739          visit::walk_item(visitor, item, cx);
 740      }
 741  }
 742  
 743  fn determine_rp_in_fn(visitor: &mut DetermineRpVisitor,
 744                        fk: &visit::fn_kind,
 745                        decl: &ast::fn_decl,
 746                        body: &ast::Block,
 747                        _: Span,
 748                        _: ast::NodeId,
 749                        cx: @mut DetermineRpCtxt) {
 750      do cx.with(cx.item_id, false) {
 751          do cx.with_ambient_variance(rv_contravariant) {
 752              for a in decl.inputs.iter() {
 753                  visitor.visit_ty(&a.ty, cx);
 754              }
 755          }
 756          visitor.visit_ty(&decl.output, cx);
 757          let generics = visit::generics_of_fn(fk);
 758          visitor.visit_generics(&generics, cx);
 759          visitor.visit_block(body, cx);
 760      }
 761  }

 764                               ty_m: &ast::TypeMethod,
 765                               cx: @mut DetermineRpCtxt) {
 766      do cx.with(cx.item_id, false) {
 767          visit::walk_ty_method(visitor, ty_m, cx);
 768      }
 769  }
 770  
 771  fn determine_rp_in_ty(visitor: &mut DetermineRpVisitor,
 772                        ty: &ast::Ty,
 773                        cx: @mut DetermineRpCtxt) {
 774      // we are only interested in types that will require an item to
 775      // be region-parameterized.  if cx.item_id is zero, then this type
 776      // is not a member of a type defn nor is it a constitutent of an
 777      // impl etc.  So we can ignore it and its components.
 778      if cx.item_id == 0 { return; }
 779  
 780      // if this type directly references a region pointer like &'r ty,
 781      // add to the worklist/set.  Note that &'r ty is contravariant with
 782      // respect to &r, because &'r ty can be used whereever a *smaller*
 783      // region is expected (and hence is a supertype of those
 784      // locations)
 785      let sess = cx.sess;
 786      match ty.node {
 787          ast::ty_rptr(ref r, _) => {
 788              debug!("referenced rptr type %s",
 789                     pprust::ty_to_str(ty, sess.intr()));
 790  
 791              if cx.region_is_relevant(r) {
 792                  let rv = cx.add_variance(rv_contravariant);
 793                  cx.add_rp(cx.item_id, rv)
 794              }
 795          }
 796  
 797          ast::ty_closure(ref f) => {
 798              debug!("referenced fn type: %s",
 799                     pprust::ty_to_str(ty, sess.intr()));
 800              match f.region {
 801                  Some(_) => {
 802                      if cx.region_is_relevant(&f.region) {
 803                          let rv = cx.add_variance(rv_contravariant);
 804                          cx.add_rp(cx.item_id, rv)
 805                      }
 806                  }
 807                  None => {
 808                      if f.sigil == ast::BorrowedSigil && cx.anon_implies_rp {
 809                          let rv = cx.add_variance(rv_contravariant);
 810                          cx.add_rp(cx.item_id, rv)
 811                      }
 812                  }
 813              }
 814          }
 815  
 816          _ => {}
 817      }
 818  
 819      // if this references another named type, add the dependency
 820      // to the dep_map.  If the type is not defined in this crate,
 821      // then check whether it is region-parameterized and consider
 822      // that as a direct dependency.
 823      match ty.node {
 824        ast::ty_path(ref path, _, id) => {
 825          match cx.def_map.find(&id) {
 826            Some(&ast::DefTy(did)) |
 827            Some(&ast::DefTrait(did)) |
 828            Some(&ast::DefStruct(did)) => {
 829              if did.crate == ast::LOCAL_CRATE {
 830                  if cx.region_is_relevant(&path.segments.last().lifetime) {
 831                      cx.add_dep(did.node);
 832                  }
 833              } else {
 834                  let cstore = sess.cstore;
 835                  match csearch::get_region_param(cstore, did) {
 836                    None => {}
 837                    Some(variance) => {
 838                      debug!("reference to external, rp'd type %s",
 839                             pprust::ty_to_str(ty, sess.intr()));
 840                      if cx.region_is_relevant(&path.segments.last().lifetime) {
 841                          let rv = cx.add_variance(variance);
 842                          cx.add_rp(cx.item_id, rv)
 843                      }
 844                    }
 845                  }
 846              }
 847            }
 848            _ => {}
 849          }
 850        }
 851        _ => {}
 852      }
 853  
 854      match ty.node {
 855        ast::ty_box(ref mt) | ast::ty_uniq(ref mt) | ast::ty_vec(ref mt) |
 856        ast::ty_rptr(_, ref mt) | ast::ty_ptr(ref mt) => {
 857          visit_mt(visitor, mt, cx);
 858        }
 859  
 860        ast::ty_path(ref path, _, _) => {
 861          // type parameters are---for now, anyway---always invariant
 862          do cx.with_ambient_variance(rv_invariant) {
 863              for tp in path.segments.iter().flat_map(|s| s.types.iter()) {
 864                  visitor.visit_ty(tp, cx);
 865              }
 866          }
 867        }
 868  
 869        ast::ty_closure(@ast::TyClosure {decl: ref decl, _}) |

  27  // Kind analysis pass.
  28  //
  29  // There are several kinds defined by various operations. The most restrictive
  30  // kind is noncopyable. The noncopyable kind can be extended with any number
  31  // of the following attributes.
  32  //
  33  //  Send: Things that can be sent on channels or included in spawned closures. It
  34  //  includes scalar types as well as classes and unique types containing only
  35  //  sendable types.
  36  //  'static: Things that do not contain references.
  37  //
  38  // This pass ensures that type parameters are only instantiated with types
  39  // whose kinds are equal or less general than the way the type parameter was
  40  // annotated (with the `Send` bound).
  41  //
  42  // It also verifies that noncopyable kinds are not copied. Sendability is not
  43  // applied, since none of our language primitives send. Instead, the sending
  44  // primitives in the stdlib are explicitly annotated to only take sendable
  45  // types.
  46

  83  fn check_struct_safe_for_destructor(cx: &mut Context,
  84                                      span: Span,
  85                                      struct_did: DefId) {
  86      let struct_tpt = ty::lookup_item_type(cx.tcx, struct_did);
  87      if !struct_tpt.generics.has_type_params() {
  88          let struct_ty = ty::mk_struct(cx.tcx, struct_did, ty::substs {
  89              regions: ty::NonerasedRegions(OwnedSlice::empty()),
  90              self_ty: None,
  91              tps: Vec::new()
  92          });
  93          if !ty::type_is_sendable(cx.tcx, struct_ty) {
  94              cx.tcx.sess.span_err(span,
  95                                   "cannot implement a destructor on a \
  96                                    structure that does not satisfy Send");
  97              cx.tcx.sess.span_note(span,
  98                                    "use \"#[unsafe_destructor]\" on the \
  99                                     implementation to force the compiler to \
 100                                     allow this");
 101          }
 102      } else {
 103          cx.tcx.sess.span_err(span,
 104                               "cannot implement a destructor on a structure \
 105                                with type parameters");
 106          cx.tcx.sess.span_note(span,
 107                                "use \"#[unsafe_destructor]\" on the \
 108                                 implementation to force the compiler to \
 109                                 allow this");
 110      }
 111  }

 113  fn check_impl_of_trait(cx: &mut Context, it: &Item, trait_ref: &TraitRef, self_type: &Ty) {
 114      let ast_trait_def = *cx.tcx.def_map.borrow()
 115                                .find(&trait_ref.ref_id)
 116                                .expect("trait ref not in def map!");
 117      let trait_def_id = ast_util::def_id_of_def(ast_trait_def);
 118      let trait_def = *cx.tcx.trait_defs.borrow()
 119                             .find(&trait_def_id)
 120                             .expect("trait def not in trait-defs map!");
 121  
 122      // If this trait has builtin-kind supertraits, meet them.
 123      let self_ty: ty::t = ty::node_id_to_type(cx.tcx, it.id);
 124      debug!("checking impl with self type {:?}", ty::get(self_ty).sty);
 125      check_builtin_bounds(cx, self_ty, trait_def.bounds, |missing| {
 126          cx.tcx.sess.span_err(self_type.span,
 127              format!("the type `{}', which does not fulfill `{}`, cannot implement this \
 128                    trait", ty_to_str(cx.tcx, self_ty), missing.user_string(cx.tcx)));
 129          cx.tcx.sess.span_note(self_type.span,
 130              format!("types implementing this trait must fulfill `{}`",
 131                   trait_def.bounds.user_string(cx.tcx)));
 132      });
 133  
 134      // If this is a destructor, check kinds.
 135      if cx.tcx.lang_items.drop_trait() == Some(trait_def_id) {
 136          match self_type.node {
 137              TyPath(_, ref bounds, path_node_id) => {
 138                  assert!(bounds.is_none());
 139                  let struct_def = cx.tcx.def_map.borrow().get_copy(&path_node_id);
 140                  let struct_did = ast_util::def_id_of_def(struct_def);
 141                  check_struct_safe_for_destructor(cx, self_type.span, struct_did);
 142              }
 143              _ => {
 144                  cx.tcx.sess.span_bug(self_type.span,
 145                      "the self type for the Drop trait impl is not a path");
 146              }
 147          }
 148      }
 149  }

 170      fn check_for_uniq(cx: &Context, fv: &freevar_entry, bounds: ty::BuiltinBounds) {

 173          let id = ast_util::def_id_of_def(fv.def).node;
 174          let var_t = ty::node_id_to_type(cx.tcx, id);

 176          check_freevar_bounds(cx, fv.span, var_t, bounds, None);

 179      fn check_for_block(cx: &Context, fv: &freevar_entry,
 180                         bounds: ty::BuiltinBounds, region: ty::Region) {
 181          let id = ast_util::def_id_of_def(fv.def).node;
 182          let var_t = ty::node_id_to_type(cx.tcx, id);
 183          // FIXME(#3569): Figure out whether the implicit borrow is actually
 184          // mutable. Currently we assume all upvars are referenced mutably.
 185          let implicit_borrowed_type = ty::mk_mut_rptr(cx.tcx, region, var_t);
 186          check_freevar_bounds(cx, fv.span, implicit_borrowed_type,
 187                               bounds, Some(var_t));
 188      }

 190      fn check_for_bare(cx: &Context, fv: @freevar_entry) {
 191          cx.tcx.sess.span_err(
 192              fv.span,
 193              "can't capture dynamic environment in a fn item; \
 194              use the || { ... } closure form instead");
 195      } // same check is done in resolve.rs, but shouldn't be done
 196  
 197      let fty = ty::node_id_to_type(cx.tcx, id);
 198      match ty::get(fty).sty {
 199          ty::ty_closure(~ty::ClosureTy {
 200              sigil: OwnedSigil,
 201              bounds: bounds,
 202              ..
 203          }) => {
 204              b(|cx, fv| check_for_uniq(cx, fv, bounds))
 205          }
 206          ty::ty_closure(~ty::ClosureTy {
 207              sigil: ManagedSigil,
 208              ..
 209          }) => {
 210              // can't happen
 211              fail!("internal error: saw closure with managed sigil (@fn)");
 212          }
 213          ty::ty_closure(~ty::ClosureTy {
 214              sigil: BorrowedSigil,
 215              bounds: bounds,
 216              region: region,
 217              ..
 218          }) => {
 219              b(|cx, fv| check_for_block(cx, fv, bounds, region))
 220          }
 221          ty::ty_bare_fn(_) => {
 222              b(check_for_bare)
 223          }
 224          ref s => {
 225              cx.tcx.sess.bug(
 226                  format!("expect fn type in kind checker, not {:?}", s));
 227          }
 228      }
 229  }

 231  // Check that the free variables used in a shared/sendable closure conform
 232  // to the copy/move kind bounds. Then recursively check the function body.
 233  fn check_fn(
 234      cx: &mut Context,
 235      fk: &visit::FnKind,
 236      decl: &FnDecl,
 237      body: &Block,
 238      sp: Span,
 239      fn_id: NodeId) {
 240  
 241      // Check kinds on free variables:
 242      with_appropriate_checker(cx, fn_id, |chk| {
 243          let r = freevars::get_freevars(cx.tcx, fn_id);
 244          for fv in r.iter() {
 245              chk(cx, *fv);
 246          }
 247      });
 248  
 249      visit::walk_fn(cx, fk, decl, body, sp, fn_id, ());
 250  }

 252  pub fn check_expr(cx: &mut Context, e: &Expr) {
 253      debug!("kind::check_expr({})", expr_to_str(e));
 254  
 255      // Handle any kind bounds on type parameters
 256      {
 257          let method_map = cx.method_map.borrow();
 258          let method = method_map.find(&typeck::MethodCall::expr(e.id));
 259          let node_type_substs = cx.tcx.node_type_substs.borrow();
 260          let r = match method {
 261              Some(method) => Some(&method.substs.tps),
 262              None => node_type_substs.find(&e.id)
 263          };
 264          for ts in r.iter() {
 265              let def_map = cx.tcx.def_map.borrow();
 266              let type_param_defs = match e.node {
 267                ExprPath(_) => {
 268                  let did = ast_util::def_id_of_def(def_map.get_copy(&e.id));
 269                  ty::lookup_item_type(cx.tcx, did).generics.type_param_defs.clone()
 270                }
 271                _ => {
 272                  // Type substitutions should only occur on paths and
 273                  // method calls, so this needs to be a method call.
 274  
 275                  // Even though the callee_id may have been the id with
 276                  // node_type_substs, e.id is correct here.
 277                  match method {
 278                      Some(method) => {
 279                          ty::method_call_type_param_defs(cx.tcx, method.origin)
 280                      }
 281                      None => {
 282                          cx.tcx.sess.span_bug(e.span,
 283                              "non path/method call expr has type substs??");
 284                      }
 285                  }
 286                }
 287              };
 288              if ts.len() != type_param_defs.len() {
 289                  // Fail earlier to make debugging easier
 290                  fail!("internal error: in kind::check_expr, length \
 291                        mismatch between actual and declared bounds: actual = \
 292                        {}, declared = {}",
 293                        ts.repr(cx.tcx),
 294                        type_param_defs.repr(cx.tcx));
 295              }
 296              for (&ty, type_param_def) in ts.iter().zip(type_param_defs.iter()) {
 297                  check_typaram_bounds(cx, e.span, ty, type_param_def)
 298              }
 299          }
 300      }
 301  
 302      match e.node {
 303          ExprUnary(UnBox, interior) => {
 304              let interior_type = ty::expr_ty(cx.tcx, interior);
 305              let _ = check_static(cx.tcx, interior_type, interior.span);
 306          }
 307          ExprCast(source, _) => {
 308              let source_ty = ty::expr_ty(cx.tcx, source);
 309              let target_ty = ty::expr_ty(cx.tcx, e);
 310              check_trait_cast(cx, source_ty, target_ty, source.span);
 311          }
 312          ExprRepeat(element, count_expr) => {
 313              let count = ty::eval_repeat_count(cx.tcx, count_expr);
 314              if count > 1 {
 315                  let element_ty = ty::expr_ty(cx.tcx, element);
 316                  check_copy(cx, element_ty, element.span,
 317                             "repeated element will be copied");
 318              }
 319          }
 320          _ => {}
 321      }
 322  
 323      // Search for auto-adjustments to find trait coercions.
 324      match cx.tcx.adjustments.borrow().find(&e.id) {
 325          Some(adjustment) => {
 326              match **adjustment {
 327                  ty::AutoObject(..) => {
 328                      let source_ty = ty::expr_ty(cx.tcx, e);
 329                      let target_ty = ty::expr_ty_adjusted(cx.tcx, e,
 330                                                           &*cx.method_map.borrow());
 331                      check_trait_cast(cx, source_ty, target_ty, e.span);
 332                  }
 333                  ty::AutoAddEnv(..) |
 334                  ty::AutoDerefRef(..) => {}
 335              }
 336          }
 337          None => {}
 338      }
 339  
 340      visit::walk_expr(cx, e, ());
 341  }

 373  // Calls "any_missing" if any bounds were missing.
 374  pub fn check_builtin_bounds(cx: &Context,
 375                              ty: ty::t,
 376                              bounds: ty::BuiltinBounds,
 377                              any_missing: |ty::BuiltinBounds|) {
 378      let kind = ty::type_contents(cx.tcx, ty);
 379      let mut missing = ty::EmptyBuiltinBounds();
 380      for bound in bounds.iter() {
 381          if !kind.meets_bound(cx.tcx, bound) {
 382              missing.add(bound);
 383          }
 384      }
 385      if !missing.is_empty() {
 386          any_missing(missing);
 387      }
 388  }

 407  pub fn check_freevar_bounds(cx: &Context, sp: Span, ty: ty::t,
 408                              bounds: ty::BuiltinBounds, referenced_ty: Option<ty::t>)
 409  {

 471  /// This is rather subtle.  When we are casting a value to an instantiated
 472  /// trait like `a as trait<'r>`, regionck already ensures that any references
 473  /// that appear in the type of `a` are bounded by `'r` (ed.: rem
 474  /// FIXME(#5723)).  However, it is possible that there are *type parameters*
 475  /// in the type of `a`, and those *type parameters* may have references
 476  /// within them.  We have to guarantee that the regions which appear in those
 477  /// type parameters are not obscured.

 479  /// Therefore, we ensure that one of three conditions holds:
 480  ///
 481  /// (1) The trait instance cannot escape the current fn.  This is
 482  /// guaranteed if the region bound `&r` is some scope within the fn
 483  /// itself.  This case is safe because whatever references are
 484  /// found within the type parameter, they must enclose the fn body

 489  /// `deque<T>`, then whatever references may appear in `T` also
 490  /// appear in `deque<T>`.
 491  ///
 492  /// (3) The type parameter is sendable (and therefore does not contain
 493  /// references).

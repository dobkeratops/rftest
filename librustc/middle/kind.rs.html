  27  // Kind analysis pass.
  28  //
  29  // There are several kinds defined by various operations. The most restrictive
  30  // kind is noncopyable. The noncopyable kind can be extended with any number
  31  // of the following attributes.
  32  //
  33  //  Send: Things that can be sent on channels or included in spawned closures. It
  34  //  includes scalar types as well as classes and unique types containing only
  35  //  sendable types.
  36  //  'static: Things that do not contain references.
  37  //
  38  // This pass ensures that type parameters are only instantiated with types
  39  // whose kinds are equal or less general than the way the type parameter was
  40  // annotated (with the `Send` bound).
  41  //
  42  // It also verifies that noncopyable kinds are not copied. Sendability is not
  43  // applied, since none of our language primitives send. Instead, the sending
  44  // primitives in the stdlib are explicitly annotated to only take sendable
  45  // types.
  46  
  47  #[deriving(Clone)]

  83  fn check_struct_safe_for_destructor(cx: &mut Context,
  84                                      span: Span,
  85                                      struct_did: DefId) {
  86      let struct_tpt = ty::lookup_item_type(cx.tcx, struct_did);
  87      if !struct_tpt.generics.has_type_params() {
  88          let struct_ty = ty::mk_struct(cx.tcx, struct_did, ty::substs {
  89              regions: ty::NonerasedRegions(OwnedSlice::empty()),
  90              self_ty: None,
  91              tps: Vec::new()
  92          });
  93          if !ty::type_is_sendable(cx.tcx, struct_ty) {
  94              cx.tcx.sess.span_err(span,
  95                                   "cannot implement a destructor on a \
  96                                    structure that does not satisfy Send");
  97              cx.tcx.sess.span_note(span,
  98                                    "use \"#[unsafe_destructor]\" on the \
  99                                     implementation to force the compiler to \
 100                                     allow this");
 101          }
 102      } else {
 103          cx.tcx.sess.span_err(span,
 104                               "cannot implement a destructor on a structure \
 105                                with type parameters");
 106          cx.tcx.sess.span_note(span,
 107                                "use \"#[unsafe_destructor]\" on the \
 108                                 implementation to force the compiler to \
 109                                 allow this");
 110      }
 111  }

 113  fn check_impl_of_trait(cx: &mut Context, it: &Item, trait_ref: &TraitRef, self_type: &Ty) {
 114      let ast_trait_def = *cx.tcx.def_map.borrow()
 115                                .find(&trait_ref.ref_id)
 116                                .expect("trait ref not in def map!");
 117      let trait_def_id = ast_util::def_id_of_def(ast_trait_def);
 118      let trait_def = *cx.tcx.trait_defs.borrow()
 119                             .find(&trait_def_id)
 120                             .expect("trait def not in trait-defs map!");
 121  
 122      // If this trait has builtin-kind supertraits, meet them.
 123      let self_ty: ty::t = ty::node_id_to_type(cx.tcx, it.id);
 124      debug!("checking impl with self type {:?}", ty::get(self_ty).sty);
 125      check_builtin_bounds(cx, self_ty, trait_def.bounds, |missing| {
 126          cx.tcx.sess.span_err(self_type.span,
 127              format!("the type `{}', which does not fulfill `{}`, cannot implement this \
 128                    trait", ty_to_str(cx.tcx, self_ty), missing.user_string(cx.tcx)));
 129          cx.tcx.sess.span_note(self_type.span,
 130              format!("types implementing this trait must fulfill `{}`",
 131                   trait_def.bounds.user_string(cx.tcx)));
 132      });
 133  
 134      // If this is a destructor, check kinds.
 135      if cx.tcx.lang_items.drop_trait() == Some(trait_def_id) {
 136          match self_type.node {
 137              TyPath(_, ref bounds, path_node_id) => {
 138                  assert!(bounds.is_none());
 139                  let struct_def = cx.tcx.def_map.borrow().get_copy(&path_node_id);
 140                  let struct_did = ast_util::def_id_of_def(struct_def);
 141                  check_struct_safe_for_destructor(cx, self_type.span, struct_did);
 142              }
 143              _ => {
 144                  cx.tcx.sess.span_bug(self_type.span,
 145                      "the self type for the Drop trait impl is not a path");
 146              }
 147          }
 148      }
 149  }

 167  fn with_appropriate_checker(cx: &Context,
 168                              id: NodeId,
 169                              b: |checker: |&Context, @freevar_entry||) {
 170      fn check_for_uniq(cx: &Context, fv: &freevar_entry, bounds: ty::BuiltinBounds) {
 171          // all captured data must be owned, regardless of whether it is
 172          // moved in or copied in.
 173          let id = ast_util::def_id_of_def(fv.def).node;
 174          let var_t = ty::node_id_to_type(cx.tcx, id);
 175  
 176          check_freevar_bounds(cx, fv.span, var_t, bounds, None);
 177      }

 179      fn check_for_block(cx: &Context, fv: &freevar_entry,
 180                         bounds: ty::BuiltinBounds, region: ty::Region) {
 181          let id = ast_util::def_id_of_def(fv.def).node;
 182          let var_t = ty::node_id_to_type(cx.tcx, id);
 183          // FIXME(#3569): Figure out whether the implicit borrow is actually
 184          // mutable. Currently we assume all upvars are referenced mutably.
 185          let implicit_borrowed_type = ty::mk_mut_rptr(cx.tcx, region, var_t);
 186          check_freevar_bounds(cx, fv.span, implicit_borrowed_type,
 187                               bounds, Some(var_t));
 188      }

 190      fn check_for_bare(cx: &Context, fv: @freevar_entry) {
 191          cx.tcx.sess.span_err(
 192              fv.span,
 193              "can't capture dynamic environment in a fn item; \
 194              use the || { ... } closure form instead");
 195      } // same check is done in resolve.rs, but shouldn't be done
 196  
 197      let fty = ty::node_id_to_type(cx.tcx, id);
 198      match ty::get(fty).sty {
 199          ty::ty_closure(~ty::ClosureTy {
 200              store: ty::UniqTraitStore, bounds, ..
 201          }) => b(|cx, fv| check_for_uniq(cx, fv, bounds)),
 202  
 203          ty::ty_closure(~ty::ClosureTy {
 204              store: ty::RegionTraitStore(region, _), bounds, ..
 205          }) => b(|cx, fv| check_for_block(cx, fv, bounds, region)),
 206  
 207          ty::ty_bare_fn(_) => {
 208              b(check_for_bare)
 209          }
 210          ref s => {
 211              cx.tcx.sess.bug(
 212                  format!("expect fn type in kind checker, not {:?}", s));
 213          }
 214      }
 215  }

 217  // Check that the free variables used in a shared/sendable closure conform
 218  // to the copy/move kind bounds. Then recursively check the function body.
 219  fn check_fn(
 220      cx: &mut Context,
 221      fk: &visit::FnKind,
 222      decl: &FnDecl,
 223      body: &Block,
 224      sp: Span,
 225      fn_id: NodeId) {
 226  
 227      // Check kinds on free variables:
 228      with_appropriate_checker(cx, fn_id, |chk| {
 229          let r = freevars::get_freevars(cx.tcx, fn_id);
 230          for fv in r.iter() {
 231              chk(cx, *fv);
 232          }
 233      });
 234  
 235      visit::walk_fn(cx, fk, decl, body, sp, fn_id, ());
 236  }

 238  pub fn check_expr(cx: &mut Context, e: &Expr) {
 239      debug!("kind::check_expr({})", expr_to_str(e));
 240  
 241      // Handle any kind bounds on type parameters
 242      {
 243          let method_map = cx.method_map.borrow();
 244          let method = method_map.find(&typeck::MethodCall::expr(e.id));
 245          let node_type_substs = cx.tcx.node_type_substs.borrow();
 246          let r = match method {
 247              Some(method) => Some(&method.substs.tps),
 248              None => node_type_substs.find(&e.id)
 249          };
 250          for ts in r.iter() {
 251              let def_map = cx.tcx.def_map.borrow();
 252              let type_param_defs = match e.node {
 253                ExprPath(_) => {
 254                  let did = ast_util::def_id_of_def(def_map.get_copy(&e.id));
 255                  ty::lookup_item_type(cx.tcx, did).generics.type_param_defs.clone()
 256                }
 257                _ => {
 258                  // Type substitutions should only occur on paths and
 259                  // method calls, so this needs to be a method call.
 260  
 261                  // Even though the callee_id may have been the id with
 262                  // node_type_substs, e.id is correct here.
 263                  match method {
 264                      Some(method) => {
 265                          ty::method_call_type_param_defs(cx.tcx, method.origin)
 266                      }
 267                      None => {
 268                          cx.tcx.sess.span_bug(e.span,
 269                              "non path/method call expr has type substs??");
 270                      }
 271                  }
 272                }
 273              };
 274              if ts.len() != type_param_defs.len() {
 275                  // Fail earlier to make debugging easier
 276                  fail!("internal error: in kind::check_expr, length \
 277                        mismatch between actual and declared bounds: actual = \
 278                        {}, declared = {}",
 279                        ts.repr(cx.tcx),
 280                        type_param_defs.repr(cx.tcx));
 281              }
 282              for (&ty, type_param_def) in ts.iter().zip(type_param_defs.iter()) {
 283                  check_typaram_bounds(cx, e.span, ty, type_param_def)
 284              }
 285          }
 286      }
 287  
 288      match e.node {
 289          ExprUnary(UnBox, interior) => {
 290              let interior_type = ty::expr_ty(cx.tcx, interior);
 291              let _ = check_static(cx.tcx, interior_type, interior.span);
 292          }
 293          ExprCast(source, _) => {
 294              let source_ty = ty::expr_ty(cx.tcx, source);
 295              let target_ty = ty::expr_ty(cx.tcx, e);
 296              check_trait_cast(cx, source_ty, target_ty, source.span);
 297          }
 298          ExprRepeat(element, count_expr) => {
 299              let count = ty::eval_repeat_count(cx.tcx, count_expr);
 300              if count > 1 {
 301                  let element_ty = ty::expr_ty(cx.tcx, element);
 302                  check_copy(cx, element_ty, element.span,
 303                             "repeated element will be copied");
 304              }
 305          }
 306          _ => {}
 307      }
 308  
 309      // Search for auto-adjustments to find trait coercions.
 310      match cx.tcx.adjustments.borrow().find(&e.id) {
 311          Some(adjustment) => {
 312              match **adjustment {
 313                  ty::AutoObject(..) => {
 314                      let source_ty = ty::expr_ty(cx.tcx, e);
 315                      let target_ty = ty::expr_ty_adjusted(cx.tcx, e,
 316                                                           &*cx.method_map.borrow());
 317                      check_trait_cast(cx, source_ty, target_ty, e.span);
 318                  }
 319                  ty::AutoAddEnv(..) |
 320                  ty::AutoDerefRef(..) => {}
 321              }
 322          }
 323          None => {}
 324      }
 325  
 326      visit::walk_expr(cx, e, ());
 327  }

 359  // Calls "any_missing" if any bounds were missing.
 360  pub fn check_builtin_bounds(cx: &Context,
 361                              ty: ty::t,
 362                              bounds: ty::BuiltinBounds,
 363                              any_missing: |ty::BuiltinBounds|) {
 364      let kind = ty::type_contents(cx.tcx, ty);
 365      let mut missing = ty::EmptyBuiltinBounds();
 366      for bound in bounds.iter() {
 367          if !kind.meets_bound(cx.tcx, bound) {
 368              missing.add(bound);
 369          }
 370      }
 371      if !missing.is_empty() {
 372          any_missing(missing);
 373      }
 374  }

 393  pub fn check_freevar_bounds(cx: &Context, sp: Span, ty: ty::t,
 394                              bounds: ty::BuiltinBounds, referenced_ty: Option<ty::t>)
 395  {
 396      check_builtin_bounds(cx, ty, bounds, |missing| {
 397          // Will be Some if the freevar is implicitly borrowed (stack closure).
 398          // Emit a less mysterious error message in this case.
 399          match referenced_ty {
 400              Some(rty) => cx.tcx.sess.span_err(sp,
 401                  format!("cannot implicitly borrow variable of type `{}` in a bounded \

 457  /// This is rather subtle.  When we are casting a value to an instantiated
 458  /// trait like `a as trait<'r>`, regionck already ensures that any references
 459  /// that appear in the type of `a` are bounded by `'r` (ed.: rem
 460  /// FIXME(#5723)).  However, it is possible that there are *type parameters*
 461  /// in the type of `a`, and those *type parameters* may have references
 462  /// within them.  We have to guarantee that the regions which appear in those
 463  /// type parameters are not obscured.
 464  ///
 465  /// Therefore, we ensure that one of three conditions holds:
 466  ///
 467  /// (1) The trait instance cannot escape the current fn.  This is
 468  /// guaranteed if the region bound `&r` is some scope within the fn
 469  /// itself.  This case is safe because whatever references are
 470  /// found within the type parameter, they must enclose the fn body
 471  /// itself.
 472  ///
 473  /// (2) The type parameter appears in the type of the trait.  For
 474  /// example, if the type parameter is `T` and the trait type is
 475  /// `deque<T>`, then whatever references may appear in `T` also
 476  /// appear in `deque<T>`.
 477  ///
 478  /// (3) The type parameter is sendable (and therefore does not contain
 479  /// references).
 480  ///
 481  /// FIXME(#5723)---This code should probably move into regionck.
 482  pub fn check_cast_for_escaping_regions(
 483      cx: &Context,
 484      source_ty: ty::t,
 485      target_ty: ty::t,

 496      // ensure that these lifetimes are shorter than all lifetimes that are in

 515

 517      // either appears in the trait type or is sendable.
 518      let target_params = ty::param_tys_in_type(target_ty);

 521          source_ty,

 534              // }

 539                  ty::ty_param(source_param) => {

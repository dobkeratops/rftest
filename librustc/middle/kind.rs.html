  27  // Kind analysis pass.
  28  //
  29  // There are several kinds defined by various operations. The most restrictive
  30  // kind is noncopyable. The noncopyable kind can be extended with any number
  31  // of the following attributes.
  32  //
  33  //  send: Things that can be sent on channels or included in spawned closures.
  34  //  freeze: Things thare are deeply immutable. They are guaranteed never to
  35  //    change, and can be safely shared without copying between tasks.
  36  //  'static: Things that do not contain borrowed pointers.
  37  //
  38  // Send includes scalar types as well as classes and unique types containing
  39  // only sendable types.
  40  //
  41  // Freeze include scalar types, things without non-const fields, and pointers
  42  // to freezable things.
  43  //
  44  // This pass ensures that type parameters are only instantiated with types
  45  // whose kinds are equal or less general than the way the type parameter was
  46  // annotated (with the `Send` or `Freeze` bound).
  47  //
  48  // It also verifies that noncopyable kinds are not copied. Sendability is not
  49  // applied, since none of our language primitives send. Instead, the sending
  50  // primitives in the stdlib are explicitly annotated to only take sendable
  51  // types.
  52  
  53  #[deriving(Clone)]
  54  pub struct Context {
  55      tcx: ty::ctxt,
  56      method_map: typeck::method_map,
  57      current_item: NodeId
  58  }

  68      fn visit_fn(&mut self, fk:&visit::fn_kind, fd:&fn_decl, b:&Block, s:Span, n:NodeId, e:Context) {

  70      }

  77      }
  78  }

  80  pub fn check_crate(tcx: ty::ctxt,
  81                     method_map: typeck::method_map,
  82                     crate: &Crate) {
  83      let ctx = Context {
  84          tcx: tcx,
  85          method_map: method_map,
  86          current_item: -1
  87      };
  88      let mut visit = KindAnalysisVisitor;

  93  fn check_struct_safe_for_destructor(cx: Context,
  94                                      span: Span,
  95                                      struct_did: DefId) {
  96      let struct_tpt = ty::lookup_item_type(cx.tcx, struct_did);
  97      if !struct_tpt.generics.has_type_params() {
  98          let struct_ty = ty::mk_struct(cx.tcx, struct_did, ty::substs {
  99              regions: ty::NonerasedRegions(opt_vec::Empty),
 100              self_ty: None,
 101              tps: ~[]
 102          });
 103          if !ty::type_is_sendable(cx.tcx, struct_ty) {
 104              cx.tcx.sess.span_err(span,
 105                                   "cannot implement a destructor on a \
 106                                    structure that does not satisfy Send");
 107              cx.tcx.sess.span_note(span,
 108                                    "use \"#[unsafe_destructor]\" on the \
 109                                     implementation to force the compiler to \
 110                                     allow this");
 111          }
 112      } else {
 113          cx.tcx.sess.span_err(span,
 114                               "cannot implement a destructor on a structure \
 115                                with type parameters");
 116          cx.tcx.sess.span_note(span,
 117                                "use \"#[unsafe_destructor]\" on the \
 118                                 implementation to force the compiler to \
 119                                 allow this");
 120      }
 121  }

 123  fn check_impl_of_trait(cx: Context, it: @item, trait_ref: &trait_ref, self_type: &Ty) {
 124      let ast_trait_def = cx.tcx.def_map.find(&trait_ref.ref_id)
 125                              .expect("trait ref not in def map!");
 126      let trait_def_id = ast_util::def_id_of_def(*ast_trait_def);
 127      let trait_def = cx.tcx.trait_defs.find(&trait_def_id)
 128                          .expect("trait def not in trait-defs map!");
 129  
 130      // If this trait has builtin-kind supertraits, meet them.
 131      let self_ty: ty::t = ty::node_id_to_type(cx.tcx, it.id);
 132      debug!("checking impl with self type %?", ty::get(self_ty).sty);
 133      do check_builtin_bounds(cx, self_ty, trait_def.bounds) |missing| {
 134          cx.tcx.sess.span_err(self_type.span,
 135              fmt!("the type `%s', which does not fulfill `%s`, cannot implement this \
 136                    trait", ty_to_str(cx.tcx, self_ty), missing.user_string(cx.tcx)));
 137          cx.tcx.sess.span_note(self_type.span,
 138              fmt!("types implementing this trait must fulfill `%s`",
 139                   trait_def.bounds.user_string(cx.tcx)));
 140      }
 141  
 142      // If this is a destructor, check kinds.
 143      if cx.tcx.lang_items.drop_trait() == Some(trait_def_id) {
 144          match self_type.node {
 145              ty_path(_, ref bounds, path_node_id) => {
 146                  assert!(bounds.is_none());
 147                  let struct_def = cx.tcx.def_map.get_copy(&path_node_id);
 148                  let struct_did = ast_util::def_id_of_def(struct_def);
 149                  check_struct_safe_for_destructor(cx, self_type.span, struct_did);
 150              }
 151              _ => {
 152                  cx.tcx.sess.span_bug(self_type.span,
 153                      "the self type for the Drop trait impl is not a path");
 154              }
 155          }
 156      }
 157  }

 159  fn check_item(visitor: &mut KindAnalysisVisitor, item: @item, cx: Context) {
 160      if !attr::contains_name(item.attrs, "unsafe_destructor") {
 161          match item.node {
 162              item_impl(_, Some(ref trait_ref), ref self_type, _) => {
 163                  check_impl_of_trait(cx, item, trait_ref, self_type);
 164              }
 165              _ => {}
 166          }
 167      }
 168  
 169      let cx = Context { current_item: item.id, ..cx };
 170      visit::walk_item(visitor, item, cx);
 171  }

 173  // Yields the appropriate function to check the kind of closed over
 174  // variables. `id` is the NodeId for some expression that creates the
 175  // closure.
 176  fn with_appropriate_checker(cx: Context, id: NodeId,
 177                              b: &fn(checker: &fn(Context, @freevar_entry))) {
 178      fn check_for_uniq(cx: Context, fv: &freevar_entry, bounds: ty::BuiltinBounds) {
 179          // all captured data must be owned, regardless of whether it is
 180          // moved in or copied in.
 181          let id = ast_util::def_id_of_def(fv.def).node;
 182          let var_t = ty::node_id_to_type(cx.tcx, id);
 183  
 184          // check that only immutable variables are implicitly copied in
 185          check_imm_free_var(cx, fv.def, fv.span);
 186  
 187          check_freevar_bounds(cx, fv.span, var_t, bounds, None);
 188      }
 189  
 190      fn check_for_box(cx: Context, fv: &freevar_entry, bounds: ty::BuiltinBounds) {
 191          // all captured data must be owned
 192          let id = ast_util::def_id_of_def(fv.def).node;
 193          let var_t = ty::node_id_to_type(cx.tcx, id);
 194  
 195          // check that only immutable variables are implicitly copied in
 196          check_imm_free_var(cx, fv.def, fv.span);
 197  
 198          check_freevar_bounds(cx, fv.span, var_t, bounds, None);
 199      }

 201      fn check_for_block(cx: Context, fv: &freevar_entry,
 202                         bounds: ty::BuiltinBounds, region: ty::Region) {
 203          let id = ast_util::def_id_of_def(fv.def).node;
 204          let var_t = ty::node_id_to_type(cx.tcx, id);
 205          // FIXME(#3569): Figure out whether the implicit borrow is actually
 206          // mutable. Currently we assume all upvars are referenced mutably.
 207          let implicit_borrowed_type = ty::mk_mut_rptr(cx.tcx, region, var_t);
 208          check_freevar_bounds(cx, fv.span, implicit_borrowed_type,
 209                               bounds, Some(var_t));
 210      }

 212      fn check_for_bare(cx: Context, fv: @freevar_entry) {
 213          cx.tcx.sess.span_err(
 214              fv.span,
 215              "can't capture dynamic environment in a fn item; \
 216              use the || { ... } closure form instead");
 217      } // same check is done in resolve.rs, but shouldn't be done
 218  
 219      let fty = ty::node_id_to_type(cx.tcx, id);
 220      match ty::get(fty).sty {
 221          ty::ty_closure(ty::ClosureTy {
 222              sigil: OwnedSigil,
 223              bounds: bounds,
 224              _
 225          }) => {
 226              b(|cx, fv| check_for_uniq(cx, fv, bounds))
 227          }
 228          ty::ty_closure(ty::ClosureTy {
 229              sigil: ManagedSigil,
 230              _
 231          }) => {
 232              // can't happen
 233          }
 234          ty::ty_closure(ty::ClosureTy {
 235              sigil: BorrowedSigil,
 236              bounds: bounds,
 237              region: region,
 238              _
 239          }) => {
 240              b(|cx, fv| check_for_block(cx, fv, bounds, region))
 241          }
 242          ty::ty_bare_fn(_) => {
 243              b(check_for_bare)
 244          }
 245          ref s => {
 246              cx.tcx.sess.bug(
 247                  fmt!("expect fn type in kind checker, not %?", s));
 248          }
 249      }
 250  }

 252  // Check that the free variables used in a shared/sendable closure conform
 253  // to the copy/move kind bounds. Then recursively check the function body.
 254  fn check_fn(
 255      v: &mut KindAnalysisVisitor,
 256      fk: &visit::fn_kind,
 257      decl: &fn_decl,
 258      body: &Block,
 259      sp: Span,
 260      fn_id: NodeId,
 261      cx: Context) {
 262  
 263      // Check kinds on free variables:
 264      do with_appropriate_checker(cx, fn_id) |chk| {
 265          let r = freevars::get_freevars(cx.tcx, fn_id);
 266          for fv in r.iter() {
 267              chk(cx, *fv);
 268          }
 269      }
 270  
 271      visit::walk_fn(v, fk, decl, body, sp, fn_id, cx);
 272  }

 274  pub fn check_expr(v: &mut KindAnalysisVisitor, e: @Expr, cx: Context) {
 275      debug!("kind::check_expr(%s)", expr_to_str(e, cx.tcx.sess.intr()));
 276  
 277      // Handle any kind bounds on type parameters
 278      let type_parameter_id = match e.get_callee_id() {
 279          Some(callee_id) => callee_id,
 280          None => e.id,
 281      };
 282      {
 283          let r = cx.tcx.node_type_substs.find(&type_parameter_id);
 284          for ts in r.iter() {
 285              let type_param_defs = match e.node {
 286                ExprPath(_) => {
 287                  let did = ast_util::def_id_of_def(cx.tcx.def_map.get_copy(&e.id));
 288                  ty::lookup_item_type(cx.tcx, did).generics.type_param_defs
 289                }
 290                _ => {
 291                  // Type substitutions should only occur on paths and
 292                  // method calls, so this needs to be a method call.
 293  
 294                  // Even though the callee_id may have been the id with
 295                  // node_type_substs, e.id is correct here.
 296                  ty::method_call_type_param_defs(cx.tcx, cx.method_map, e.id).expect(
 297                      "non path/method call expr has type substs??")
 298                }
 299              };
 300              if ts.len() != type_param_defs.len() {
 301                  // Fail earlier to make debugging easier
 302                  fail!("internal error: in kind::check_expr, length \
 303                        mismatch between actual and declared bounds: actual = \
 304                        %s, declared = %s",
 305                        ts.repr(cx.tcx),
 306                        type_param_defs.repr(cx.tcx));
 307              }
 308              for (&ty, type_param_def) in ts.iter().zip(type_param_defs.iter()) {
 309                  check_typaram_bounds(cx, type_parameter_id, e.span, ty, type_param_def)
 310              }
 311          }
 312      }
 313  
 314      match e.node {
 315          ExprUnary(_, UnBox(_), interior) => {
 316              let interior_type = ty::expr_ty(cx.tcx, interior);
 317              let _ = check_durable(cx.tcx, interior_type, interior.span);
 318          }
 319          ExprCast(source, _) => {
 320              check_cast_for_escaping_regions(cx, source, e);
 321              match ty::get(ty::expr_ty(cx.tcx, e)).sty {
 322                  ty::ty_trait(_, _, _, _, bounds) => {
 323                      let source_ty = ty::expr_ty(cx.tcx, source);
 324                      check_trait_cast_bounds(cx, e.span, source_ty, bounds)
 325                  }
 326                  _ => { }
 327              }
 328          }
 329          ExprRepeat(element, count_expr, _) => {
 330              let count = ty::eval_repeat_count(&cx.tcx, count_expr);
 331              if count > 1 {
 332                  let element_ty = ty::expr_ty(cx.tcx, element);
 333                  check_copy(cx, element_ty, element.span,
 334                             "repeated element will be copied");
 335              }
 336          }
 337          _ => {}
 338      }
 339      visit::walk_expr(v, e, cx);
 340  }

 342  fn check_ty(v: &mut KindAnalysisVisitor, aty: &Ty, cx: Context) {
 343      match aty.node {
 344        ty_path(_, _, id) => {
 345            let r = cx.tcx.node_type_substs.find(&id);
 346            for ts in r.iter() {
 347                let did = ast_util::def_id_of_def(cx.tcx.def_map.get_copy(&id));
 348                let type_param_defs =
 349                    ty::lookup_item_type(cx.tcx, did).generics.type_param_defs;
 350                for (&ty, type_param_def) in ts.iter().zip(type_param_defs.iter()) {
 351                    check_typaram_bounds(cx, aty.id, aty.span, ty, type_param_def)
 352                }
 353            }
 354        }
 355        _ => {}
 356      }
 357      visit::walk_ty(v, aty, cx);
 358  }

 360  // Calls "any_missing" if any bounds were missing.
 361  pub fn check_builtin_bounds(cx: Context, ty: ty::t, bounds: ty::BuiltinBounds,
 362                              any_missing: &fn(ty::BuiltinBounds))
 363  {
 364      let kind = ty::type_contents(cx.tcx, ty);
 365      let mut missing = ty::EmptyBuiltinBounds();
 366      for bound in bounds.iter() {
 367          if !kind.meets_bound(cx.tcx, bound) {
 368              missing.add(bound);
 369          }
 370      }
 371      if !missing.is_empty() {
 372          any_missing(missing);
 373      }
 374  }

 376  pub fn check_typaram_bounds(cx: Context,
 377                      _type_parameter_id: NodeId,
 378                      sp: Span,
 379                      ty: ty::t,
 380                      type_param_def: &ty::TypeParameterDef)
 381  {
 382      do check_builtin_bounds(cx, ty, type_param_def.bounds.builtin_bounds) |missing| {
 383          cx.tcx.sess.span_err(
 384              sp,
 385              fmt!("instantiating a type parameter with an incompatible type \
 386                    `%s`, which does not fulfill `%s`",
 387                   ty_to_str(cx.tcx, ty),
 388                   missing.user_string(cx.tcx)));
 389      }
 390  }

 392  pub fn check_freevar_bounds(cx: Context, sp: Span, ty: ty::t,
 393                              bounds: ty::BuiltinBounds, referenced_ty: Option<ty::t>)
 394  {
 395      do check_builtin_bounds(cx, ty, bounds) |missing| {
 396          // Will be Some if the freevar is implicitly borrowed (stack closure).
 397          // Emit a less mysterious error message in this case.
 398          match referenced_ty {
 399              Some(rty) => cx.tcx.sess.span_err(sp,
 400                  fmt!("cannot implicitly borrow variable of type `%s` in a bounded \
 401                        stack closure (implicit reference does not fulfill `%s`)",
 402                       ty_to_str(cx.tcx, rty), missing.user_string(cx.tcx))),

 502  /// This is rather subtle.  When we are casting a value to a instantiated
 503  /// trait like `a as trait<'r>`, regionck already ensures that any borrowed
 504  /// pointers that appear in the type of `a` are bounded by `'r` (ed.: rem
 505  /// FIXME(#5723)).  However, it is possible that there are *type parameters*
 506  /// in the type of `a`, and those *type parameters* may have borrowed pointers
 507  /// within them.  We have to guarantee that the regions which appear in those
 508  /// type parameters are not obscured.

 511  ///

 513  /// guaranteed if the region bound `&r` is some scope within the fn
 514  /// itself.  This case is safe because whatever borrowed pointers are
 515  /// found within the type parameter, they must enclose the fn body
 516  /// itself.

 519  /// example, if the type parameter is `T` and the trait type is
 520  /// `deque<T>`, then whatever borrowed ptrs may appear in `T` also
 521  /// appear in `deque<T>`.
 522  ///
 523  /// (3) The type parameter is sendable (and therefore does not contain
 524  /// borrowed ptrs).
 525  ///

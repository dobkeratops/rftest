  105 pub fn decode_inlined_item(cdata: @cstore::crate_metadata,
  106                            tcx: ty::ctxt,
  107                            maps: Maps,
  108                            path: &[ast_map::path_elt],
  109                            par_doc: ebml::Doc)
  110                         -> Option<ast::inlined_item> {
  111     let dcx = @DecodeContext {
  112         cdata: cdata,
  113         tcx: tcx,
  114         maps: maps
  115     };
  116     match par_doc.opt_child(c::tag_ast) {
  117       None => None,
  118       Some(ast_doc) => {
  119         debug!("> Decoding inlined fn: %s::?",
  120                ast_map::path_to_str(path, token::get_ident_interner()));
  121         let mut ast_dsr = reader::Decoder(ast_doc);
  122         let from_id_range = Decodable::decode(&mut ast_dsr);
  123         let to_id_range = reserve_id_range(dcx.tcx.sess, from_id_range);
  124         let xcx = @ExtendedDecodeContext {
  125             dcx: dcx,
  126             from_id_range: from_id_range,
  127             to_id_range: to_id_range
  128         };
  129         let raw_ii = decode_ast(ast_doc);
  130         let ii = renumber_ast(xcx, raw_ii);
  131         debug!("Fn named: %s", tcx.sess.str_of(ii.ident()));
  132         debug!("< Decoded inlined fn: %s::%s",
  133                ast_map::path_to_str(path, token::get_ident_interner()),
  134                tcx.sess.str_of(ii.ident()));
  135         ast_map::map_decoded_item(tcx.sess.diagnostic(),
  136                                   dcx.tcx.items,
  137                                   path.to_owned(),
  138                                   &ii);
  139         decode_side_tables(xcx, ast_doc);
  140         match ii {
  141           ast::ii_item(i) => {
  142             debug!(">>> DECODED ITEM >>>\n%s\n<<< DECODED ITEM <<<",
  143                    syntax::print::pprust::item_to_str(i, tcx.sess.intr()));
  144           }
  145           _ => { }
  146         }
  147         Some(ii)
  148       }
  149     }
  150 }

  155 fn reserve_id_range(sess: Session,
  156                     from_id_range: ast_util::id_range) -> ast_util::id_range {
  157     // Handle the case of an empty range:
  158     if from_id_range.empty() { return from_id_range; }
  159     let cnt = from_id_range.max - from_id_range.min;
  160     let to_id_min = sess.parse_sess.next_id;
  161     let to_id_max = sess.parse_sess.next_id + cnt;
  162     sess.parse_sess.next_id = to_id_max;
  163     ast_util::id_range { min: to_id_min, max: to_id_min }
  164 }

  166 impl ExtendedDecodeContext {
  167     pub fn tr_id(&self, id: ast::NodeId) -> ast::NodeId {
  168         /*!
  169          * Translates an internal id, meaning a node id that is known
  170          * to refer to some part of the item currently being inlined,
  171          * such as a local variable or argument.  All naked node-ids
  172          * that appear in types have this property, since if something
  173          * might refer to an external item we would use a def-id to
  174          * allow for the possibility that the item resides in another
  175          * crate.
  176          */
  177 
  178         // from_id_range should be non-empty
  179         assert!(!self.from_id_range.empty());
  180         (id - self.from_id_range.min + self.to_id_range.min)
  181     }
  182     pub fn tr_def_id(&self, did: ast::def_id) -> ast::def_id {
  183         /*!
  184          * Translates an EXTERNAL def-id, converting the crate number
  185          * from the one used in the encoded data to the current crate
  186          * numbers..  By external, I mean that it be translated to a
  187          * reference to the item in its original crate, as opposed to
  188          * being translated to a reference to the inlined version of
  189          * the item.  This is typically, but not always, what you
  190          * want, because most def-ids refer to external things like
  191          * types or other fns that may or may not be inlined.  Note
  192          * that even when the inlined function is referencing itself
  193          * recursively, we would want `tr_def_id` for that
  194          * reference--- conceptually the function calls the original,
  195          * non-inlined version, and trans deals with linking that
  196          * recursive call to the inlined copy.
  197          *
  198          * However, there are a *few* cases where def-ids are used but
  199          * we know that the thing being referenced is in fact *internal*
  200          * to the item being inlined.  In those cases, you should use
  201          * `tr_intern_def_id()` below.
  202          */
  203 
  204         decoder::translate_def_id(self.dcx.cdata, did)
  205     }
  206     pub fn tr_intern_def_id(&self, did: ast::def_id) -> ast::def_id {
  207         /*!
  208          * Translates an INTERNAL def-id, meaning a def-id that is
  209          * known to refer to some part of the item currently being
  210          * inlined.  In that case, we want to convert the def-id to
  211          * refer to the current crate and to the new, inlined node-id.
  212          */
  213 
  214         assert_eq!(did.crate, ast::LOCAL_CRATE);
  215         ast::def_id { crate: ast::LOCAL_CRATE, node: self.tr_id(did.node) }
  216     }

  219     }
  220 }

  270 // Encoding and decoding the AST itself
  271 //
  272 // The hard work is done by an autogenerated module astencode_gen.  To
  273 // regenerate astencode_gen, run src/etc/gen-astencode.  It will
  274 // replace astencode_gen with a dummy file and regenerate its
  275 // contents.  If you get compile errors, the dummy file
  276 // remains---resolve the errors and then rerun astencode_gen.
  277 // Annoying, I know, but hopefully only temporary.
  278 //
  279 // When decoding, we have to renumber the AST so that the node ids that
  280 // appear within are disjoint from the node ids in our existing ASTs.
  281 // We also have to adjust the spans: for now we just insert a dummy span,
  282 // but eventually we should add entries to the local codemap as required.
  283 
  284 fn encode_ast(ebml_w: &mut writer::Encoder, item: ast::inlined_item) {
  285     ebml_w.start_tag(c::tag_tree as uint);
  286     item.encode(ebml_w);
  287     ebml_w.end_tag();
  288 }

  290 // Produces a simplified copy of the AST which does not include things
  291 // that we do not need to or do not want to export.  For example, we
  292 // do not include any nested items: if these nested items are to be
  293 // inlined, their AST will be exported separately (this only makes
  294 // sense because, in Rust, nested items are independent except for
  295 // their visibility).
  296 //
  297 // As it happens, trans relies on the fact that we do not export
  298 // nested items, as otherwise it would get confused when translating
  299 // inlined items.
  300 fn simplify_ast(ii: &ast::inlined_item) -> ast::inlined_item {
  301     fn drop_nested_items(blk: &ast::Block, fld: @fold::ast_fold) -> ast::Block {
  302         let stmts_sans_items = do blk.stmts.iter().filter_map |stmt| {
  303             match stmt.node {
  304               ast::stmt_expr(_, _) | ast::stmt_semi(_, _) |
  305               ast::stmt_decl(@codemap::spanned { node: ast::decl_local(_), span: _}, _)
  306                 => Some(*stmt),
  307               ast::stmt_decl(@codemap::spanned { node: ast::decl_item(_), span: _}, _)
  308                 => None,
  309               ast::stmt_mac(*) => fail!("unexpanded macro in astencode")
  310             }
  311         }.collect();
  312         let blk_sans_items = ast::Block {
  313             view_items: ~[], // I don't know if we need the view_items here,
  314                              // but it doesn't break tests!
  315             stmts: stmts_sans_items,
  316             expr: blk.expr,
  317             id: blk.id,
  318             rules: blk.rules,
  319             span: blk.span,
  320         };
  321         fold::noop_fold_block(&blk_sans_items, fld)
  322     }
  323 
  324     let fld = fold::make_fold(@fold::AstFoldFns {
  325         fold_block: drop_nested_items,
  326         .. *fold::default_ast_fold()
  327     });
  328 
  329     match *ii {
  330         //hack: we're not dropping items
  331         ast::ii_item(i) => ast::ii_item(fld.fold_item(i).get()),
  332         ast::ii_method(d, is_provided, m) =>
  333           ast::ii_method(d, is_provided, fld.fold_method(m)),
  334         ast::ii_foreign(i) => ast::ii_foreign(fld.fold_foreign_item(i))
  335     }
  336 }

  338 fn decode_ast(par_doc: ebml::Doc) -> ast::inlined_item {
  339     let chi_doc = par_doc.get(c::tag_tree as uint);
  340     let mut d = reader::Decoder(chi_doc);
  341     Decodable::decode(&mut d)
  342 }
  343 
  344 fn renumber_ast(xcx: @ExtendedDecodeContext, ii: ast::inlined_item)
  345     -> ast::inlined_item {
  346     let fld = fold::make_fold(@fold::AstFoldFns{
  347         new_id: |a| xcx.tr_id(a),
  348         new_span: |a| xcx.tr_span(a),
  349         .. *fold::default_ast_fold()
  350     });
  351 
  352     match ii {
  353         ast::ii_item(i) => ast::ii_item(fld.fold_item(i).get()),
  354         ast::ii_method(d, is_provided, m) =>
  355           ast::ii_method(xcx.tr_def_id(d), is_provided, fld.fold_method(m)),
  356         ast::ii_foreign(i) => ast::ii_foreign(fld.fold_foreign_item(i)),
  357     }
  358 }

  373 impl tr for ast::def {
  374     fn tr(&self, xcx: @ExtendedDecodeContext) -> ast::def {
  375         match *self {
  376           ast::def_fn(did, p) => ast::def_fn(did.tr(xcx), p),
  377           ast::def_static_method(did, did2_opt, p) => {
  378             ast::def_static_method(did.tr(xcx),
  379                                    did2_opt.map(|did2| did2.tr(xcx)),
  380                                    p)
  381           }
  382           ast::def_method(did0, did1) => {
  383             ast::def_method(did0.tr(xcx), did1.map(|did1| did1.tr(xcx)))
  384           }
  385           ast::def_self_ty(nid) => { ast::def_self_ty(xcx.tr_id(nid)) }
  386           ast::def_self(nid, i) => { ast::def_self(xcx.tr_id(nid), i) }
  387           ast::def_mod(did) => { ast::def_mod(did.tr(xcx)) }
  388           ast::def_foreign_mod(did) => { ast::def_foreign_mod(did.tr(xcx)) }
  389           ast::def_static(did, m) => { ast::def_static(did.tr(xcx), m) }
  390           ast::def_arg(nid, b) => { ast::def_arg(xcx.tr_id(nid), b) }
  391           ast::def_local(nid, b) => { ast::def_local(xcx.tr_id(nid), b) }
  392           ast::def_variant(e_did, v_did) => {
  393             ast::def_variant(e_did.tr(xcx), v_did.tr(xcx))
  394           },
  395           ast::def_trait(did) => ast::def_trait(did.tr(xcx)),
  396           ast::def_ty(did) => ast::def_ty(did.tr(xcx)),
  397           ast::def_prim_ty(p) => ast::def_prim_ty(p),

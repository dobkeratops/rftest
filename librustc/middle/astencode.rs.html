  106  pub fn decode_inlined_item(cdata: @cstore::crate_metadata,
  107                             tcx: ty::ctxt,
  108                             maps: Maps,
  109                             path: &[ast_map::path_elt],
  110                             par_doc: ebml::Doc)
  111                          -> Option<ast::inlined_item> {
  112      let dcx = @DecodeContext {
  113          cdata: cdata,
  114          tcx: tcx,
  115          maps: maps
  116      };
  117      match par_doc.opt_child(c::tag_ast) {
  118        None => None,
  119        Some(ast_doc) => {
  120          debug!("> Decoding inlined fn: %s::?",
  121                 ast_map::path_to_str(path, token::get_ident_interner()));
  122          let mut ast_dsr = reader::Decoder(ast_doc);
  123          let from_id_range = Decodable::decode(&mut ast_dsr);
  124          let to_id_range = reserve_id_range(dcx.tcx.sess, from_id_range);
  125          let xcx = @ExtendedDecodeContext {
  126              dcx: dcx,
  127              from_id_range: from_id_range,
  128              to_id_range: to_id_range
  129          };
  130          let raw_ii = decode_ast(ast_doc);
  131          let ii = renumber_ast(xcx, raw_ii);
  132          debug!("Fn named: %s", tcx.sess.str_of(ii.ident()));
  133          debug!("< Decoded inlined fn: %s::%s",
  134                 ast_map::path_to_str(path, token::get_ident_interner()),
  135                 tcx.sess.str_of(ii.ident()));
  136          ast_map::map_decoded_item(tcx.sess.diagnostic(),
  137                                    dcx.tcx.items,
  138                                    path.to_owned(),
  139                                    &ii);
  140          decode_side_tables(xcx, ast_doc);
  141          match ii {
  142            ast::ii_item(i) => {
  143              debug!(">>> DECODED ITEM >>>\n%s\n<<< DECODED ITEM <<<",
  144                     syntax::print::pprust::item_to_str(i, tcx.sess.intr()));
  145            }
  146            _ => { }
  147          }
  148          Some(ii)
  149        }
  150      }
  151  }

  156  fn reserve_id_range(sess: Session,
  157                      from_id_range: ast_util::id_range) -> ast_util::id_range {
  158      // Handle the case of an empty range:
  159      if from_id_range.empty() { return from_id_range; }
  160      let cnt = from_id_range.max - from_id_range.min;
  161      assert!(cnt >= 0);
  162      let to_id_min = sess.reserve_node_ids(cnt as uint);
  163      let to_id_max = to_id_min + cnt;
  164      ast_util::id_range { min: to_id_min, max: to_id_max }
  165  }

  167  impl ExtendedDecodeContext {
  168      pub fn tr_id(&self, id: ast::NodeId) -> ast::NodeId {
  169          /*!
  170           * Translates an internal id, meaning a node id that is known
  171           * to refer to some part of the item currently being inlined,
  172           * such as a local variable or argument.  All naked node-ids
  173           * that appear in types have this property, since if something
  174           * might refer to an external item we would use a def-id to
  175           * allow for the possibility that the item resides in another
  176           * crate.
  177           */
  178  
  179          // from_id_range should be non-empty
  180          assert!(!self.from_id_range.empty());
  181          (id - self.from_id_range.min + self.to_id_range.min)
  182      }
  183      pub fn tr_def_id(&self, did: ast::DefId) -> ast::DefId {
  184          /*!
  185           * Translates an EXTERNAL def-id, converting the crate number
  186           * from the one used in the encoded data to the current crate
  187           * numbers..  By external, I mean that it be translated to a
  188           * reference to the item in its original crate, as opposed to
  189           * being translated to a reference to the inlined version of
  190           * the item.  This is typically, but not always, what you
  191           * want, because most def-ids refer to external things like
  192           * types or other fns that may or may not be inlined.  Note
  193           * that even when the inlined function is referencing itself
  194           * recursively, we would want `tr_def_id` for that
  195           * reference--- conceptually the function calls the original,
  196           * non-inlined version, and trans deals with linking that
  197           * recursive call to the inlined copy.
  198           *
  199           * However, there are a *few* cases where def-ids are used but
  200           * we know that the thing being referenced is in fact *internal*
  201           * to the item being inlined.  In those cases, you should use
  202           * `tr_intern_def_id()` below.
  203           */
  204  
  205          decoder::translate_def_id(self.dcx.cdata, did)
  206      }
  207      pub fn tr_intern_def_id(&self, did: ast::DefId) -> ast::DefId {
  208          /*!
  209           * Translates an INTERNAL def-id, meaning a def-id that is
  210           * known to refer to some part of the item currently being
  211           * inlined.  In that case, we want to convert the def-id to
  212           * refer to the current crate and to the new, inlined node-id.
  213           */
  214  
  215          assert_eq!(did.crate, ast::LOCAL_CRATE);
  216          ast::DefId { crate: ast::LOCAL_CRATE, node: self.tr_id(did.node) }
  217      }

  272  //
  273  // The hard work is done by an autogenerated module astencode_gen.  To

  296      fn fold_block(&self, blk: &ast::Block) -> ast::Block {
  297          let stmts_sans_items = do blk.stmts.iter().filter_map |stmt| {
  298              match stmt.node {
  299                  ast::StmtExpr(_, _) | ast::StmtSemi(_, _) |
  300                  ast::StmtDecl(@codemap::Spanned {
  301                      node: ast::DeclLocal(_),
  302                      span: _
  303                  }, _) => Some(*stmt),
  304                  ast::StmtDecl(@codemap::Spanned {
  305                      node: ast::DeclItem(_),
  306                      span: _
  307                  }, _) => None,
  308                  ast::StmtMac(*) => fail!("unexpanded macro in astencode")
  309              }
  310          }.collect();
  311          let blk_sans_items = ast::Block {
  312              view_items: ~[], // I don't know if we need the view_items here,
  313                               // but it doesn't break tests!
  314              stmts: stmts_sans_items,
  315              expr: blk.expr,
  316              id: blk.id,
  317              rules: blk.rules,
  318              span: blk.span,
  319          };
  320          fold::noop_fold_block(&blk_sans_items, self)
  321      }

  324  // Produces a simplified copy of the AST which does not include things
  325  // that we do not need to or do not want to export.  For example, we
  326  // do not include any nested items: if these nested items are to be
  327  // inlined, their AST will be exported separately (this only makes
  328  // sense because, in Rust, nested items are independent except for
  329  // their visibility).
  330  //
  331  // As it happens, trans relies on the fact that we do not export
  332  // nested items, as otherwise it would get confused when translating
  333  // inlined items.
  334  fn simplify_ast(ii: &ast::inlined_item) -> ast::inlined_item {
  335      let fld = NestedItemsDropper {
  336          contents: (),
  337      };
  338  
  339      match *ii {
  340          //hack: we're not dropping items
  341          ast::ii_item(i) => ast::ii_item(fld.fold_item(i).unwrap()),
  342          ast::ii_method(d, is_provided, m) =>
  343            ast::ii_method(d, is_provided, fld.fold_method(m)),
  344          ast::ii_foreign(i) => ast::ii_foreign(fld.fold_foreign_item(i))
  345      }
  346  }

  393  impl tr for ast::Def {
  394      fn tr(&self, xcx: @ExtendedDecodeContext) -> ast::Def {
  395          match *self {
  396            ast::DefFn(did, p) => ast::DefFn(did.tr(xcx), p),
  397            ast::DefStaticMethod(did, wrapped_did2, p) => {
  398              ast::DefStaticMethod(did.tr(xcx),
  399                                     match wrapped_did2 {
  400                                      ast::FromTrait(did2) => {
  401                                          ast::FromTrait(did2.tr(xcx))
  402                                      }
  403                                      ast::FromImpl(did2) => {
  404                                          ast::FromImpl(did2.tr(xcx))
  405                                      }
  406                                     },
  407                                     p)

  114  pub fn decode_inlined_item(cdata: @cstore::crate_metadata,
  115                             tcx: &ty::ctxt,
  116                             maps: &Maps,
  117                             path: Vec<ast_map::PathElem>,
  118                             par_doc: ebml::Doc)
  119                             -> Result<ast::InlinedItem, Vec<ast_map::PathElem>> {
  120      let dcx = &DecodeContext {
  121          cdata: cdata,
  122          tcx: tcx,
  123          maps: maps
  124      };
  125      match par_doc.opt_child(c::tag_ast) {
  126        None => Err(path),
  127        Some(ast_doc) => {
  128          let mut path_as_str = None;
  129          debug!("> Decoding inlined fn: {}::?",
  130          {
  131              // Do an Option dance to use the path after it is moved below.
  132              let s = ast_map::path_to_str(ast_map::Values(path.iter()));
  133              path_as_str = Some(s);
  134              path_as_str.as_ref().map(|x| x.as_slice())
  135          });
  136          let mut ast_dsr = reader::Decoder(ast_doc);
  137          let from_id_range = Decodable::decode(&mut ast_dsr).unwrap();
  138          let to_id_range = reserve_id_range(&dcx.tcx.sess, from_id_range);
  139          let xcx = &ExtendedDecodeContext {
  140              dcx: dcx,
  141              from_id_range: from_id_range,
  142              to_id_range: to_id_range
  143          };
  144          let raw_ii = decode_ast(ast_doc);
  145          let ii = renumber_and_map_ast(xcx, &dcx.tcx.map, path, raw_ii);
  146          let ident = match ii {
  147              ast::IIItem(i) => i.ident,
  148              ast::IIForeign(i) => i.ident,
  149              ast::IIMethod(_, _, m) => m.ident,
  150          };
  151          debug!("Fn named: {}", token::get_ident(ident));
  152          debug!("< Decoded inlined fn: {}::{}",
  153                 path_as_str.unwrap(),
  154                 token::get_ident(ident));
  155          region::resolve_inlined_item(&tcx.sess, &tcx.region_maps, &ii);
  156          decode_side_tables(xcx, ast_doc);
  157          match ii {
  158            ast::IIItem(i) => {
  159              debug!(">>> DECODED ITEM >>>\n{}\n<<< DECODED ITEM <<<",
  160                     syntax::print::pprust::item_to_str(i));
  161            }
  162            _ => { }
  163          }
  164          Ok(ii)
  165        }
  166      }
  167  }

  172  fn reserve_id_range(sess: &Session,
  173                      from_id_range: ast_util::IdRange) -> ast_util::IdRange {
  174      // Handle the case of an empty range:
  175      if from_id_range.empty() { return from_id_range; }
  176      let cnt = from_id_range.max - from_id_range.min;
  177      let to_id_min = sess.reserve_node_ids(cnt);
  178      let to_id_max = to_id_min + cnt;
  179      ast_util::IdRange { min: to_id_min, max: to_id_max }
  180  }

  182  impl<'a> ExtendedDecodeContext<'a> {
  183      pub fn tr_id(&self, id: ast::NodeId) -> ast::NodeId {
  184          /*!
  185           * Translates an internal id, meaning a node id that is known
  186           * to refer to some part of the item currently being inlined,
  187           * such as a local variable or argument.  All naked node-ids
  188           * that appear in types have this property, since if something
  189           * might refer to an external item we would use a def-id to
  190           * allow for the possibility that the item resides in another
  191           * crate.
  192           */
  193  
  194          // from_id_range should be non-empty
  195          assert!(!self.from_id_range.empty());
  196          (id - self.from_id_range.min + self.to_id_range.min)
  197      }
  198      pub fn tr_def_id(&self, did: ast::DefId) -> ast::DefId {
  199          /*!
  200           * Translates an EXTERNAL def-id, converting the crate number
  201           * from the one used in the encoded data to the current crate
  202           * numbers..  By external, I mean that it be translated to a
  203           * reference to the item in its original crate, as opposed to
  204           * being translated to a reference to the inlined version of
  205           * the item.  This is typically, but not always, what you
  206           * want, because most def-ids refer to external things like
  207           * types or other fns that may or may not be inlined.  Note
  208           * that even when the inlined function is referencing itself
  209           * recursively, we would want `tr_def_id` for that
  210           * reference--- conceptually the function calls the original,
  211           * non-inlined version, and trans deals with linking that
  212           * recursive call to the inlined copy.
  213           *
  214           * However, there are a *few* cases where def-ids are used but
  215           * we know that the thing being referenced is in fact *internal*
  216           * to the item being inlined.  In those cases, you should use
  217           * `tr_intern_def_id()` below.
  218           */
  219  
  220          decoder::translate_def_id(self.dcx.cdata, did)
  221      }
  222      pub fn tr_intern_def_id(&self, did: ast::DefId) -> ast::DefId {
  223          /*!
  224           * Translates an INTERNAL def-id, meaning a def-id that is
  225           * known to refer to some part of the item currently being
  226           * inlined.  In that case, we want to convert the def-id to
  227           * refer to the current crate and to the new, inlined node-id.
  228           */
  229  
  230          assert_eq!(did.krate, ast::LOCAL_CRATE);
  231          ast::DefId { krate: ast::LOCAL_CRATE, node: self.tr_id(did.node) }
  232      }

  319                  ast::StmtDecl(decl, _) => {
  320                      match decl.node {

  322                          ast::DeclItem(_) => None,
  323                      }
  324                  }
  325                  ast::StmtMac(..) => fail!("unexpanded macro in astencode")
  326              }

  328          let blk_sans_items = ast::P(ast::Block {
  329              view_items: Vec::new(), // I don't know if we need the view_items
  330                                      // here, but it doesn't break tests!
  331              stmts: stmts_sans_items,

  333              id: blk.id,
  334              rules: blk.rules,

  341  // Produces a simplified copy of the AST which does not include things
  342  // that we do not need to or do not want to export.  For example, we
  343  // do not include any nested items: if these nested items are to be
  344  // inlined, their AST will be exported separately (this only makes
  345  // sense because, in Rust, nested items are independent except for
  346  // their visibility).
  347  //
  348  // As it happens, trans relies on the fact that we do not export
  349  // nested items, as otherwise it would get confused when translating
  350  // inlined items.
  351  fn simplify_ast(ii: e::InlinedItemRef) -> ast::InlinedItem {
  352      let mut fld = NestedItemsDropper;
  353  
  354      match ii {
  355          // HACK we're not dropping items.
  356          e::IIItemRef(i) => ast::IIItem(fold::noop_fold_item(i, &mut fld)
  357                                         .expect_one("expected one item")),
  358          e::IIMethodRef(d, p, m) => ast::IIMethod(d, p, fold::noop_fold_method(m, &mut fld)),
  359          e::IIForeignRef(i) => ast::IIForeign(fold::noop_fold_foreign_item(i, &mut fld))
  360      }
  361  }

  387  fn renumber_and_map_ast(xcx: &ExtendedDecodeContext,
  388                          map: &ast_map::Map,
  389                          path: Vec<ast_map::PathElem> ,
  390                          ii: ast::InlinedItem) -> ast::InlinedItem {
  391      ast_map::map_decoded_item(map,
  392                                path.move_iter().collect(),
  393                                AstRenumberer { xcx: xcx },
  394                                |fld| {
  395          match ii {
  396              ast::IIItem(i) => {
  397                  ast::IIItem(fld.fold_item(i).expect_one("expected one item"))

   51  /// The parent visitor, used to determine what's the parent of what (node-wise)
   52  ////////////////////////////////////////////////////////////////////////////////
   53  
   54  struct ParentVisitor {
   55      parents: NodeMap<ast::NodeId>,
   56      curparent: ast::NodeId,
   57  }
   58  
   59  impl Visitor<()> for ParentVisitor {
   60      fn visit_item(&mut self, item: &ast::Item, _: ()) {
   61          self.parents.insert(item.id, self.curparent);
   62  
   63          let prev = self.curparent;
   64          match item.node {
   65              ast::ItemMod(..) => { self.curparent = item.id; }
   66              // Enum variants are parented to the enum definition itself beacuse
   67              // they inherit privacy
   68              ast::ItemEnum(ref def, _) => {
   69                  for variant in def.variants.iter() {
   70                      // If variants are private, then their logical "parent" is
   71                      // the enclosing module because everyone in the enclosing
   72                      // module can still use the private variant
   73                      if variant.node.vis == ast::Private {
   74                          self.parents.insert(variant.node.id, self.curparent);
   75  
   76                      // Otherwise, if the variant is public, then the parent is
   77                      // considered the enclosing enum because the enum will
   78                      // dictate the privacy visibility of this variant instead.
   79                      } else {
   80                          self.parents.insert(variant.node.id, item.id);
   81                      }
   82                  }
   83              }
   84  
   85              // Trait methods are always considered "public", but if the trait is
   86              // private then we need some private item in the chain from the
   87              // method to the root. In this case, if the trait is private, then
   88              // parent all the methods to the trait to indicate that they're
   89              // private.
   90              ast::ItemTrait(_, _, ref methods) if item.vis != ast::Public => {
   91                  for m in methods.iter() {
   92                      match *m {
   93                          ast::Provided(ref m) => self.parents.insert(m.id, item.id),
   94                          ast::Required(ref m) => self.parents.insert(m.id, item.id),
   95                      };
   96                  }
   97              }
   98  
   99              _ => {}
  100          }
  101          visit::walk_item(self, item, ());
  102          self.curparent = prev;
  103      }
  104  
  105      fn visit_foreign_item(&mut self, a: &ast::ForeignItem, _: ()) {
  106          self.parents.insert(a.id, self.curparent);
  107          visit::walk_foreign_item(self, a, ());
  108      }
  109  
  110      fn visit_fn(&mut self, a: &visit::FnKind, b: &ast::FnDecl,
  111                  c: &ast::Block, d: Span, id: ast::NodeId, _: ()) {
  112          // We already took care of some trait methods above, otherwise things
  113          // like impl methods and pub trait methods are parented to the
  114          // containing module, not the containing trait.
  115          if !self.parents.contains_key(&id) {
  116              self.parents.insert(id, self.curparent);
  117          }
  118          visit::walk_fn(self, a, b, c, d, id, ());
  119      }
  120  
  121      fn visit_struct_def(&mut self, s: &ast::StructDef, i: ast::Ident,
  122                          g: &ast::Generics, n: ast::NodeId, _: ()) {
  123          // Struct constructors are parented to their struct definitions because
  124          // they essentially are the struct definitions.
  125          match s.ctor_id {
  126              Some(id) => { self.parents.insert(id, n); }
  127              None => {}
  128          }
  129  
  130          // While we have the id of the struct definition, go ahead and parent
  131          // all the fields.
  132          for field in s.fields.iter() {
  133              self.parents.insert(field.node.id, self.curparent);
  134          }
  135          visit::walk_struct_def(self, s, i, g, n, ())
  136      }
  137  }

  142  
  143  struct EmbargoVisitor<'a> {
  144      tcx: &'a ty::ctxt,
  145      exp_map2: &'a resolve::ExportMap2,
  146  
  147      // This flag is an indicator of whether the previous item in the
  148      // hierarchical chain was exported or not. This is the indicator of whether
  149      // children should be exported as well. Note that this can flip from false
  150      // to true if a reexported module is entered (or an action similar).
  151      prev_exported: bool,
  152  
  153      // This is a list of all exported items in the AST. An exported item is any
  154      // function/method/item which is usable by external crates. This essentially
  155      // means that the result is "public all the way down", but the "path down"
  156      // may jump across private boundaries through reexport statements.
  157      exported_items: ExportedItems,
  158  
  159      // This sets contains all the destination nodes which are publicly
  160      // re-exported. This is *not* a set of all reexported nodes, only a set of
  161      // all nodes which are reexported *and* reachable from external crates. This
  162      // means that the destination of the reexport is exported, and hence the
  163      // destination must also be exported.
  164      reexports: NodeSet,
  165  
  166      // These two fields are closely related to one another in that they are only
  167      // used for generation of the 'PublicItems' set, not for privacy checking at
  168      // all
  169      public_items: PublicItems,
  170      prev_public: bool,
  171  }

  173  impl<'a> EmbargoVisitor<'a> {
  174      // There are checks inside of privacy which depend on knowing whether a
  175      // trait should be exported or not. The two current consumers of this are:
  176      //
  177      //  1. Should default methods of a trait be exported?
  178      //  2. Should the methods of an implementation of a trait be exported?
  179      //
  180      // The answer to both of these questions partly rely on whether the trait
  181      // itself is exported or not. If the trait is somehow exported, then the
  182      // answers to both questions must be yes. Right now this question involves
  183      // more analysis than is currently done in rustc, so we conservatively
  184      // answer "yes" so that all traits need to be exported.
  185      fn exported_trait(&self, _id: ast::NodeId) -> bool {
  186          true
  187      }
  188  }

  190  impl<'a> Visitor<()> for EmbargoVisitor<'a> {
  191      fn visit_item(&mut self, item: &ast::Item, _: ()) {
  192          let orig_all_pub = self.prev_public;
  193          self.prev_public = orig_all_pub && item.vis == ast::Public;
  194          if self.prev_public {
  195              self.public_items.insert(item.id);
  196          }
  197  
  198          let orig_all_exported = self.prev_exported;
  199          match item.node {
  200              // impls/extern blocks do not break the "public chain" because they
  201              // cannot have visibility qualifiers on them anyway
  202              ast::ItemImpl(..) | ast::ItemForeignMod(..) => {}
  203  
  204              // Traits are a little special in that even if they themselves are
  205              // not public they may still be exported.
  206              ast::ItemTrait(..) => {
  207                  self.prev_exported = self.exported_trait(item.id);
  208              }
  209  
  210              // Private by default, hence we only retain the "public chain" if
  211              // `pub` is explicitly listed.
  212              _ => {
  213                  self.prev_exported =
  214                      (orig_all_exported && item.vis == ast::Public) ||
  215                       self.reexports.contains(&item.id);
  216              }
  217          }
  218  
  219          let public_first = self.prev_exported &&
  220                             self.exported_items.insert(item.id);
  221  
  222          match item.node {
  223              // Enum variants inherit from their parent, so if the enum is
  224              // public all variants are public unless they're explicitly priv
  225              ast::ItemEnum(ref def, _) if public_first => {
  226                  for variant in def.variants.iter() {
  227                      if variant.node.vis != ast::Private {
  228                          self.exported_items.insert(variant.node.id);
  229                      }
  230                  }
  231              }
  232  
  233              // Implementations are a little tricky to determine what's exported
  234              // out of them. Here's a few cases which are currently defined:
  235              //
  236              // * Impls for private types do not need to export their methods
  237              //   (either public or private methods)
  238              //
  239              // * Impls for public types only have public methods exported
  240              //
  241              // * Public trait impls for public types must have all methods
  242              //   exported.
  243              //
  244              // * Private trait impls for public types can be ignored
  245              //
  246              // * Public trait impls for private types have their methods
  247              //   exported. I'm not entirely certain that this is the correct
  248              //   thing to do, but I have seen use cases of where this will cause
  249              //   undefined symbols at linkage time if this case is not handled.
  250              //
  251              // * Private trait impls for private types can be completely ignored
  252              ast::ItemImpl(_, _, ref ty, ref methods) => {
  253                  let public_ty = match ty.node {
  254                      ast::TyPath(_, _, id) => {
  255                          match self.tcx.def_map.borrow().get_copy(&id) {
  256                              ast::DefPrimTy(..) => true,
  257                              def => {
  258                                  let did = def_id_of_def(def);
  259                                  !is_local(did) ||
  260                                   self.exported_items.contains(&did.node)
  261                              }
  262                          }
  263                      }
  264                      _ => true,
  265                  };
  266                  let tr = ty::impl_trait_ref(self.tcx, local_def(item.id));
  267                  let public_trait = tr.map_or(false, |tr| {
  268                      !is_local(tr.def_id) ||
  269                       self.exported_items.contains(&tr.def_id.node)
  270                  });
  271  
  272                  if public_ty || public_trait {
  273                      for method in methods.iter() {
  274                          let meth_public = match method.explicit_self.node {
  275                              ast::SelfStatic => public_ty,
  276                              _ => true,
  277                          } && method.vis == ast::Public;
  278                          if meth_public || tr.is_some() {
  279                              self.exported_items.insert(method.id);
  280                          }
  281                      }
  282                  }
  283              }
  284  
  285              // Default methods on traits are all public so long as the trait
  286              // is public
  287              ast::ItemTrait(_, _, ref methods) if public_first => {
  288                  for method in methods.iter() {
  289                      match *method {
  290                          ast::Provided(ref m) => {
  291                              debug!("provided {}", m.id);
  292                              self.exported_items.insert(m.id);
  293                          }
  294                          ast::Required(ref m) => {
  295                              debug!("required {}", m.id);
  296                              self.exported_items.insert(m.id);
  297                          }
  298                      }
  299                  }
  300              }
  301  
  302              // Struct constructors are public if the struct is all public.
  303              ast::ItemStruct(ref def, _) if public_first => {
  304                  match def.ctor_id {
  305                      Some(id) => { self.exported_items.insert(id); }
  306                      None => {}
  307                  }
  308              }
  309  
  310              _ => {}
  311          }
  312  
  313          visit::walk_item(self, item, ());
  314  
  315          self.prev_exported = orig_all_exported;
  316          self.prev_public = orig_all_pub;
  317      }
  318  
  319      fn visit_foreign_item(&mut self, a: &ast::ForeignItem, _: ()) {
  320          if self.prev_exported && a.vis == ast::Public {
  321              self.exported_items.insert(a.id);
  322          }
  323      }
  324  
  325      fn visit_mod(&mut self, m: &ast::Mod, _sp: Span, id: ast::NodeId, _: ()) {
  326          // This code is here instead of in visit_item so that the
  327          // crate module gets processed as well.
  328          if self.prev_exported {
  329              let exp_map2 = self.exp_map2.borrow();
  330              assert!(exp_map2.contains_key(&id), "wut {:?}", id);
  331              for export in exp_map2.get(&id).iter() {
  332                  if is_local(export.def_id) {
  333                      self.reexports.insert(export.def_id.node);
  334                  }
  335              }
  336          }
  337          visit::walk_mod(self, m, ())
  338      }
  339  }

  345  struct PrivacyVisitor<'a> {
  346      tcx: &'a ty::ctxt,
  347      curitem: ast::NodeId,
  348      in_fn: bool,
  349      in_foreign: bool,
  350      method_map: &'a MethodMap,
  351      parents: NodeMap<ast::NodeId>,
  352      external_exports: resolve::ExternalExports,
  353      last_private_map: resolve::LastPrivateMap,
  354  }
  355  
  356  enum PrivacyResult {
  357      Allowable,
  358      ExternallyDenied,
  359      DisallowedBy(ast::NodeId),
  360  }
  361  
  362  impl<'a> PrivacyVisitor<'a> {
  363      // used when debugging
  364      fn nodestr(&self, id: ast::NodeId) -> ~str {
  365          self.tcx.map.node_to_str(id)
  366      }
  367  
  368      // Determines whether the given definition is public from the point of view
  369      // of the current item.
  370      fn def_privacy(&self, did: ast::DefId) -> PrivacyResult {
  371          if !is_local(did) {
  372              if self.external_exports.contains(&did) {
  373                  debug!("privacy - {:?} was externally exported", did);
  374                  return Allowable;
  375              }
  376              debug!("privacy - is {:?} a public method", did);
  377  
  378              return match self.tcx.methods.borrow().find(&did) {
  379                  Some(meth) => {
  380                      debug!("privacy - well at least it's a method: {:?}", meth);
  381                      match meth.container {
  382                          ty::TraitContainer(id) => {
  383                              debug!("privacy - recursing on trait {:?}", id);
  384                              self.def_privacy(id)
  385                          }
  386                          ty::ImplContainer(id) => {
  387                              match ty::impl_trait_ref(self.tcx, id) {
  388                                  Some(t) => {
  389                                      debug!("privacy - impl of trait {:?}", id);
  390                                      self.def_privacy(t.def_id)
  391                                  }
  392                                  None => {
  393                                      debug!("privacy - found a method {:?}",
  394                                              meth.vis);
  395                                      if meth.vis == ast::Public {
  396                                          Allowable
  397                                      } else {
  398                                          ExternallyDenied
  399                                      }
  400                                  }
  401                              }
  402                          }
  403                      }
  404                  }
  405                  None => {
  406                      debug!("privacy - nope, not even a method");
  407                      ExternallyDenied
  408                  }
  409              };
  410          }
  411  
  412          debug!("privacy - local {:?} not public all the way down", did);
  413          // return quickly for things in the same module
  414          if self.parents.find(&did.node) == self.parents.find(&self.curitem) {
  415              debug!("privacy - same parent, we're done here");
  416              return Allowable;
  417          }
  418  
  419          // We now know that there is at least one private member between the
  420          // destination and the root.
  421          let mut closest_private_id = did.node;
  422          loop {
  423              debug!("privacy - examining {}", self.nodestr(closest_private_id));
  424              let vis = match self.tcx.map.find(closest_private_id) {
  425                  // If this item is a method, then we know for sure that it's an
  426                  // actual method and not a static method. The reason for this is
  427                  // that these cases are only hit in the ExprMethodCall
  428                  // expression, and ExprCall will have its path checked later
  429                  // (the path of the trait/impl) if it's a static method.
  430                  //
  431                  // With this information, then we can completely ignore all
  432                  // trait methods. The privacy violation would be if the trait
  433                  // couldn't get imported, not if the method couldn't be used
  434                  // (all trait methods are public).
  435                  //
  436                  // However, if this is an impl method, then we dictate this
  437                  // decision solely based on the privacy of the method
  438                  // invocation.
  439                  // FIXME(#10573) is this the right behavior? Why not consider
  440                  //               where the method was defined?
  441                  Some(ast_map::NodeMethod(ref m)) => {
  442                      let imp = self.tcx.map.get_parent_did(closest_private_id);
  443                      match ty::impl_trait_ref(self.tcx, imp) {
  444                          Some(..) => return Allowable,
  445                          _ if m.vis == ast::Public => return Allowable,
  446                          _ => m.vis
  447                      }
  448                  }
  449                  Some(ast_map::NodeTraitMethod(_)) => {
  450                      return Allowable;
  451                  }
  452  
  453                  // This is not a method call, extract the visibility as one
  454                  // would normally look at it
  455                  Some(ast_map::NodeItem(it)) => it.vis,
  456                  Some(ast_map::NodeForeignItem(_)) => {
  457                      self.tcx.map.get_foreign_vis(closest_private_id)
  458                  }
  459                  Some(ast_map::NodeVariant(ref v)) => {
  460                      // sadly enum variants still inherit visibility, so only
  461                      // break out of this is explicitly private
  462                      if v.node.vis == ast::Private { break }
  463                      ast::Public // need to move up a level (to the enum)
  464                  }
  465                  _ => ast::Public,
  466              };
  467              if vis != ast::Public { break }
  468              // if we've reached the root, then everything was allowable and this
  469              // access is public.
  470              if closest_private_id == ast::CRATE_NODE_ID { return Allowable }
  471              closest_private_id = *self.parents.get(&closest_private_id);
  472  
  473              // If we reached the top, then we were public all the way down and
  474              // we can allow this access.
  475              if closest_private_id == ast::DUMMY_NODE_ID { return Allowable }
  476          }
  477          debug!("privacy - closest priv {}", self.nodestr(closest_private_id));
  478          if self.private_accessible(closest_private_id) {
  479              Allowable
  480          } else {
  481              DisallowedBy(closest_private_id)
  482          }
  483      }
  484  
  485      /// For a local private node in the AST, this function will determine
  486      /// whether the node is accessible by the current module that iteration is
  487      /// inside.
  488      fn private_accessible(&self, id: ast::NodeId) -> bool {
  489          let parent = *self.parents.get(&id);
  490          debug!("privacy - accessible parent {}", self.nodestr(parent));
  491  
  492          // After finding `did`'s closest private member, we roll ourselves back
  493          // to see if this private member's parent is anywhere in our ancestry.
  494          // By the privacy rules, we can access all of our ancestor's private
  495          // members, so that's why we test the parent, and not the did itself.
  496          let mut cur = self.curitem;
  497          loop {
  498              debug!("privacy - questioning {}", self.nodestr(cur));
  499              match cur {
  500                  // If the relevant parent is in our history, then we're allowed
  501                  // to look inside any of our ancestor's immediate private items,
  502                  // so this access is valid.
  503                  x if x == parent => return true,
  504  
  505                  // If we've reached the root, then we couldn't access this item
  506                  // in the first place
  507                  ast::DUMMY_NODE_ID => return false,
  508  
  509                  // Keep going up
  510                  _ => {}
  511              }
  512  
  513              cur = *self.parents.get(&cur);
  514          }
  515      }

  525                  }
  526                  false
  527              },
  528          }

  531      /// Guarantee that a particular definition is public. Returns a CheckResult
  532      /// which contains any errors found. These can be reported using `report_error`.
  533      /// If the result is `None`, no errors were found.
  534      fn ensure_public(&self, span: Span, to_check: ast::DefId,
  535                       source_did: Option<ast::DefId>, msg: &str) -> CheckResult {
  536          match self.def_privacy(to_check) {
  537              ExternallyDenied => Some((span, format!("{} is private", msg), None)),
  538              DisallowedBy(id) => {
  539                  let (err_span, err_msg) = if id == source_did.unwrap_or(to_check).node {
  540                      return Some((span, format!("{} is private", msg), None));
  541                  } else {
  542                      (span, format!("{} is inaccessible", msg))
  543                  };
  544                  match self.tcx.map.find(id) {
  545                      Some(ast_map::NodeItem(item)) => {
  546                          let desc = match item.node {
  547                              ast::ItemMod(..) => "module",
  548                              ast::ItemTrait(..) => "trait",
  549                              _ => return Some((err_span, err_msg, None)),
  550                          };
  551                          let msg = format!("{} `{}` is private",
  552                                            desc,
  553                                            token::get_ident(item.ident));
  554                          Some((err_span, err_msg, Some((span, msg))))
  555                      },
  556                      _ => Some((err_span, err_msg, None)),
  557                  }
  558              },
  559              Allowable => None,
  560          }
  561      }

  563      // Checks that a field is in scope.
  564      // FIXME #6993: change type (and name) from Ident to Name
  565      fn check_field(&mut self, span: Span, id: ast::DefId, ident: ast::Ident,
  566                     enum_id: Option<ast::DefId>) {
  567          let fields = ty::lookup_struct_fields(self.tcx, id);
  568          let struct_vis = if is_local(id) {
  569              match self.tcx.map.get(id.node) {
  570                  ast_map::NodeItem(ref it) => it.vis,
  571                  ast_map::NodeVariant(ref v) => {
  572                      if v.node.vis == ast::Inherited {
  573                          let parent = self.tcx.map.get_parent(id.node);
  574                          self.tcx.map.expect_item(parent).vis
  575                      } else {
  576                          v.node.vis
  577                      }
  578                  }
  579                  _ => {
  580                      self.tcx.sess.span_bug(span,
  581                                             format!("not an item or variant def"));
  582                  }
  583              }
  584          } else {
  585              let cstore = &self.tcx.sess.cstore;
  586              match enum_id {
  587                  Some(enum_id) => {
  588                      let v = csearch::get_enum_variants(self.tcx, enum_id);
  589                      match v.iter().find(|v| v.id == id) {
  590                          Some(variant) => {
  591                              if variant.vis == ast::Inherited {
  592                                  csearch::get_item_visibility(cstore, enum_id)
  593                              } else {
  594                                  variant.vis
  595                              }
  596                          }
  597                          None => {
  598                              self.tcx.sess.span_bug(span, "no xcrate variant");
  599                          }
  600                      }
  601                  }
  602                  None => csearch::get_item_visibility(cstore, id)
  603              }
  604          };
  605  
  606          for field in fields.iter() {
  607              if field.name != ident.name { continue; }
  608              // public structs have public fields by default, and private structs
  609              // have private fields by default.
  610              if struct_vis == ast::Public && field.vis != ast::Private { break }
  611              if struct_vis != ast::Public && field.vis == ast::Public { break }
  612              if !is_local(field.id) ||
  613                 !self.private_accessible(field.id.node) {
  614                  self.tcx.sess.span_err(span,
  615                                         format!("field `{}` is private",
  616                                                 token::get_ident(ident)))
  617              }
  618              break;
  619          }
  620      }

  622      // Given the ID of a method, checks to ensure it's in scope.
  623      fn check_static_method(&mut self, span: Span, method_id: ast::DefId,
  624                             name: ast::Ident) {
  625          // If the method is a default method, we need to use the def_id of
  626          // the default implementation.
  627          let method_id = ty::method(self.tcx, method_id).provided_source
  628                                                         .unwrap_or(method_id);
  629  
  630          let string = token::get_ident(name);
  631          self.report_error(self.ensure_public(span,
  632                                               method_id,
  633                                               None,
  634                                               format!("method `{}`", string)));
  635      }

  637      // Checks that a path is in scope.
  638      fn check_path(&mut self, span: Span, path_id: ast::NodeId, path: &ast::Path) {
  639          debug!("privacy - path {}", self.nodestr(path_id));
  640          let orig_def = self.tcx.def_map.borrow().get_copy(&path_id);
  641          let ck = |tyname: &str| {
  642              let ck_public = |def: ast::DefId| {
  643                  let name = token::get_ident(path.segments
  644                                                  .last()
  645                                                  .unwrap()
  646                                                  .identifier);
  647                  let origdid = def_id_of_def(orig_def);
  648                  self.ensure_public(span,
  649                                     def,
  650                                     Some(origdid),
  651                                     format!("{} `{}`",
  652                                             tyname,
  653                                             name))
  654              };
  655  
  656              match *self.last_private_map.get(&path_id) {
  657                  resolve::LastMod(resolve::AllPublic) => {},
  658                  resolve::LastMod(resolve::DependsOn(def)) => {
  659                      self.report_error(ck_public(def));
  660                  },
  661                  resolve::LastImport{value_priv: value_priv,
  662                                      value_used: check_value,
  663                                      type_priv: type_priv,
  664                                      type_used: check_type} => {
  665                      // This dance with found_error is because we don't want to report
  666                      // a privacy error twice for the same directive.
  667                      let found_error = match (type_priv, check_type) {
  668                          (Some(resolve::DependsOn(def)), resolve::Used) => {
  669                              !self.report_error(ck_public(def))
  670                          },
  671                          _ => false,
  672                      };
  673                      if !found_error {
  674                          match (value_priv, check_value) {
  675                              (Some(resolve::DependsOn(def)), resolve::Used) => {
  676                                  self.report_error(ck_public(def));
  677                              },
  678                              _ => {},
  679                          }
  680                      }
  681                      // If an import is not used in either namespace, we still want to check
  682                      // that it could be legal. Therefore we check in both namespaces and only
  683                      // report an error if both would be illegal. We only report one error,
  684                      // even if it is illegal to import from both namespaces.
  685                      match (value_priv, check_value, type_priv, check_type) {
  686                          (Some(p), resolve::Unused, None, _) |
  687                          (None, _, Some(p), resolve::Unused) => {
  688                              let p = match p {
  689                                  resolve::AllPublic => None,
  690                                  resolve::DependsOn(def) => ck_public(def),
  691                              };
  692                              if p.is_some() {
  693                                  self.report_error(p);
  694                              }
  695                          },
  696                          (Some(v), resolve::Unused, Some(t), resolve::Unused) => {
  697                              let v = match v {
  698                                  resolve::AllPublic => None,
  699                                  resolve::DependsOn(def) => ck_public(def),
  700                              };
  701                              let t = match t {
  702                                  resolve::AllPublic => None,
  703                                  resolve::DependsOn(def) => ck_public(def),
  704                              };
  705                              match (v, t) {
  706                                  (Some(_), Some(t)) => {
  707                                      self.report_error(Some(t));
  708                                  },
  709                                  _ => {},
  710                              }
  711                          },
  712                          _ => {},
  713                      }
  714                  },
  715              }
  716          };
  717          // FIXME(#12334) Imports can refer to definitions in both the type and
  718          // value namespaces. The privacy information is aware of this, but the
  719          // def map is not. Therefore the names we work out below will not always
  720          // be accurate and we can get slightly wonky error messages (but type
  721          // checking is always correct).
  722          match self.tcx.def_map.borrow().get_copy(&path_id) {
  723              ast::DefStaticMethod(..) => ck("static method"),
  724              ast::DefFn(..) => ck("function"),
  725              ast::DefStatic(..) => ck("static"),
  726              ast::DefVariant(..) => ck("variant"),
  727              ast::DefTy(..) => ck("type"),
  728              ast::DefTrait(..) => ck("trait"),
  729              ast::DefStruct(..) => ck("struct"),
  730              ast::DefMethod(_, Some(..)) => ck("trait method"),
  731              ast::DefMethod(..) => ck("method"),
  732              ast::DefMod(..) => ck("module"),
  733              _ => {}
  734          }
  735      }

  737      // Checks that a method is in scope.
  738      fn check_method(&mut self, span: Span, origin: MethodOrigin,
  739                      ident: ast::Ident) {
  740          match origin {
  741              MethodStatic(method_id) => {
  742                  self.check_static_method(span, method_id, ident)
  743              }
  744              // Trait methods are always all public. The only controlling factor
  745              // is whether the trait itself is accessible or not.
  746              MethodParam(MethodParam { trait_id: trait_id, .. }) |
  747              MethodObject(MethodObject { trait_id: trait_id, .. }) => {
  748                  self.report_error(self.ensure_public(span, trait_id, None, "source trait"));
  749              }
  750          }
  751      }

  767      fn visit_expr(&mut self, expr: &ast::Expr, _: ()) {
  768          match expr.node {
  769              ast::ExprField(base, ident, _) => {
  770                  match ty::get(ty::expr_ty_adjusted(self.tcx, base,
  771                                                     &*self.method_map.borrow())).sty {
  772                      ty::ty_struct(id, _) => {
  773                          self.check_field(expr.span, id, ident, None);
  774                      }
  775                      _ => {}
  776                  }
  777              }
  778              ast::ExprMethodCall(ident, _, _) => {
  779                  let method_call = MethodCall::expr(expr.id);
  780                  match self.method_map.borrow().find(&method_call) {
  781                      None => {
  782                          self.tcx.sess.span_bug(expr.span,
  783                                                  "method call not in \
  784                                                  method map");
  785                      }
  786                      Some(method) => {
  787                          debug!("(privacy checking) checking impl method");
  788                          self.check_method(expr.span, method.origin, ident);
  789                      }
  790                  }

  972  impl<'a> SanePrivacyVisitor<'a> {
  973      /// Validates all of the visibility qualifers placed on the item given. This
  974      /// ensures that there are no extraneous qualifiers that don't actually do
  975      /// anything. In theory these qualifiers wouldn't parse, but that may happen
  976      /// later on down the road...
  977      fn check_sane_privacy(&self, item: &ast::Item) {
  978          let tcx = self.tcx;
  979          let check_inherited = |sp: Span, vis: ast::Visibility, note: &str| {
  980              if vis != ast::Inherited {
  981                  tcx.sess.span_err(sp, "unnecessary visibility qualifier");
  982                  if note.len() > 0 {
  983                      tcx.sess.span_note(sp, note);
  984                  }
  985              }
  986          };
  987          let check_not_priv = |sp: Span, vis: ast::Visibility, note: &str| {
  988              if vis == ast::Private {
  989                  tcx.sess.span_err(sp, "unnecessary `priv` qualifier");
  990                  if note.len() > 0 {
  991                      tcx.sess.span_note(sp, note);
  992                  }
  993              }
  994          };
  995          let check_struct = |def: &@ast::StructDef,
  996                              vis: ast::Visibility,
  997                              parent_vis: Option<ast::Visibility>| {
  998              let public_def = match vis {
  999                  ast::Public => true,
 1000                  ast::Inherited | ast::Private => parent_vis == Some(ast::Public),
 1001              };
 1002              for f in def.fields.iter() {
 1003                 match f.node.kind {
 1004                      ast::NamedField(_, ast::Public) if public_def => {
 1005                          tcx.sess.span_err(f.span, "unnecessary `pub` \
 1006                                                     visibility");
 1007                      }
 1008                      ast::NamedField(_, ast::Private) if !public_def => {
 1009                          tcx.sess.span_err(f.span, "unnecessary `priv` \
 1010                                                     visibility");
 1011                      }
 1012                      ast::NamedField(..) | ast::UnnamedField => {}
 1013                  }
 1014              }
 1015          };
 1016          match item.node {
 1017              // implementations of traits don't need visibility qualifiers because
 1018              // that's controlled by having the trait in scope.
 1019              ast::ItemImpl(_, Some(..), _, ref methods) => {
 1020                  check_inherited(item.span, item.vis,
 1021                                  "visibility qualifiers have no effect on trait \
 1022                                   impls");
 1023                  for m in methods.iter() {
 1024                      check_inherited(m.span, m.vis, "");
 1025                  }
 1026              }
 1027  
 1028              ast::ItemImpl(_, _, _, ref methods) => {
 1029                  check_inherited(item.span, item.vis,
 1030                                  "place qualifiers on individual methods instead");
 1031                  for i in methods.iter() {
 1032                      check_not_priv(i.span, i.vis, "functions are private by \
 1033                                                     default");
 1034                  }
 1035              }
 1036              ast::ItemForeignMod(ref fm) => {
 1037                  check_inherited(item.span, item.vis,
 1038                                  "place qualifiers on individual functions \
 1039                                   instead");
 1040                  for i in fm.items.iter() {
 1041                      check_not_priv(i.span, i.vis, "functions are private by \
 1042                                                     default");
 1043                  }
 1044              }
 1045  
 1046              ast::ItemEnum(ref def, _) => {
 1047                  for v in def.variants.iter() {
 1048                      match v.node.vis {
 1049                          ast::Public => {
 1050                              if item.vis == ast::Public {
 1051                                  tcx.sess.span_err(v.span, "unnecessary `pub` \
 1052                                                             visibility");
 1053                              }
 1054                          }
 1055                          ast::Private => {
 1056                              if item.vis != ast::Public {
 1057                                  tcx.sess.span_err(v.span, "unnecessary `priv` \
 1058                                                             visibility");
 1059                              }
 1060                          }
 1061                          ast::Inherited => {}
 1062                      }
 1063  
 1064                      match v.node.kind {
 1065                          ast::StructVariantKind(ref s) => {
 1066                              check_struct(s, v.node.vis, Some(item.vis));
 1067                          }
 1068                          ast::TupleVariantKind(..) => {}
 1069                      }
 1070                  }
 1071              }
 1072  
 1073              ast::ItemStruct(ref def, _) => check_struct(def, item.vis, None),
 1074  
 1075              ast::ItemTrait(_, _, ref methods) => {

 1077                      match *m {
 1078                          ast::Provided(ref m) => {

 1084                  }
 1085              }

 1089              ast::ItemMac(..) => {

 1092              }
 1093          }
 1094      }

 1120              ast::ItemForeignMod(ref fm) => {

 1122                      check_inherited(i.span, i.vis);

 1124              }
 1125              ast::ItemEnum(ref def, _) => {
 1126                  for v in def.variants.iter() {
 1127                      check_inherited(v.span, v.node.vis);
 1128  
 1129                      match v.node.kind {
 1130                          ast::StructVariantKind(ref s) => check_struct(s),
 1131                          ast::TupleVariantKind(..) => {}
 1132                      }
 1133                  }
 1134              }
 1135  
 1136              ast::ItemStruct(ref def, _) => check_struct(def),

 1138              ast::ItemTrait(_, _, ref methods) => {
 1139                  for m in methods.iter() {
 1140                      match *m {

 1144                  }
 1145              }
 1146  
 1147              ast::ItemStatic(..) |
 1148              ast::ItemFn(..) | ast::ItemMod(..) | ast::ItemTy(..) |

 1151      }

  40  struct PrivacyVisitor {
  41      tcx: ty::ctxt,
  42      privileged_items: @mut ~[NodeId],
  43  }

  45  impl PrivacyVisitor {
  46      // Adds an item to its scope.
  47      fn add_privileged_item(&mut self, item: @ast::item, count: &mut uint) {
  48          match item.node {
  49              item_struct(*) | item_trait(*) | item_enum(*) |
  50              item_fn(*) => {
  51                  self.privileged_items.push(item.id);
  52                  *count += 1;
  53              }
  54              item_impl(_, _, _, ref methods) => {
  55                  for method in methods.iter() {
  56                      self.privileged_items.push(method.id);
  57                      *count += 1;
  58                  }
  59                  self.privileged_items.push(item.id);
  60                  *count += 1;
  61              }
  62              item_foreign_mod(ref foreign_mod) => {
  63                  for foreign_item in foreign_mod.items.iter() {
  64                      self.privileged_items.push(foreign_item.id);
  65                      *count += 1;
  66                  }
  67              }
  68              _ => {}
  69          }
  70      }
  71  
  72      // Adds items that are privileged to this scope.
  73      fn add_privileged_items(&mut self, items: &[@ast::item]) -> uint {
  74          let mut count = 0;
  75          for &item in items.iter() {
  76              self.add_privileged_item(item, &mut count);
  77          }
  78          count
  79      }

  81      // Checks that an enum variant is in scope
  82      fn check_variant(&mut self, span: Span, enum_id: ast::DefId) {
  83          let variant_info = ty::enum_variants(self.tcx, enum_id)[0];
  84          let parental_privacy = if is_local(enum_id) {
  85              let parent_vis = ast_map::node_item_query(self.tcx.items,
  86                                                        enum_id.node,
  87                                     |it| { it.vis },
  88                                     ~"unbound enum parent when checking \
  89                                      dereference of enum type");
  90              visibility_to_privacy(parent_vis)
  91          }
  92          else {
  93              // WRONG
  94              Public
  95          };
  96          debug!("parental_privacy = %?", parental_privacy);
  97          debug!("vis = %?, priv = %?",
  98                 variant_info.vis,
  99                 visibility_to_privacy(variant_info.vis))
 100          // inherited => privacy of the enum item
 101          if variant_visibility_to_privacy(variant_info.vis,
 102                                           parental_privacy == Public)
 103                                           == Private {
 104              self.tcx.sess.span_err(span,
 105                  "can only dereference enums \
 106                   with a single, public variant");
 107          }
 108      }

 110      // Returns true if a crate-local method is private and false otherwise.
 111      fn method_is_private(&mut self, span: Span, method_id: NodeId) -> bool {
 112          let check = |vis: visibility, container_id: DefId| {
 113              let mut is_private = false;
 114              if vis == private {
 115                  is_private = true;
 116              } else if vis == public {
 117                  is_private = false;
 118              } else {
 119                  // Look up the enclosing impl.
 120                  if container_id.crate != LOCAL_CRATE {
 121                      self.tcx.sess.span_bug(span,
 122                                        "local method isn't in local \
 123                                         impl?!");
 124                  }
 125  
 126                  match self.tcx.items.find(&container_id.node) {
 127                      Some(&node_item(item, _)) => {
 128                          match item.node {
 129                              item_impl(_, None, _, _)
 130                                      if item.vis != public => {
 131                                  is_private = true;
 132                              }
 133                              _ => {}
 134                          }
 135                      }
 136                      Some(_) => {
 137                          self.tcx.sess.span_bug(span, "impl wasn't an item?!");
 138                      }
 139                      None => {
 140                          self.tcx.sess.span_bug(span, "impl wasn't in AST map?!");
 141                      }
 142                  }
 143              }
 144  
 145              is_private
 146          };
 147  
 148          match self.tcx.items.find(&method_id) {
 149              Some(&node_method(method, impl_id, _)) => {
 150                  check(method.vis, impl_id)
 151              }
 152              Some(&node_trait_method(trait_method, trait_id, _)) => {
 153                  match *trait_method {
 154                      required(_) => check(public, trait_id),
 155                      provided(method) => check(method.vis, trait_id),
 156                  }
 157              }
 158              Some(_) => {
 159                  self.tcx.sess.span_bug(span,
 160                                    fmt!("method_is_private: method was a %s?!",
 161                                         ast_map::node_id_to_str(
 162                                              self.tcx.items,
 163                                              method_id,
 164                                             token::get_ident_interner())));
 165              }
 166              None => {
 167                  self.tcx.sess.span_bug(span, "method not found in \
 168                                           AST map?!");
 169              }
 170          }
 171      }

 173      // Returns true if the given local item is private and false otherwise.
 174      fn local_item_is_private(&mut self, span: Span, item_id: NodeId) -> bool {
 175          let mut f: &fn(NodeId) -> bool = |_| false;
 176          f = |item_id| {
 177              match self.tcx.items.find(&item_id) {
 178                  Some(&node_item(item, _)) => item.vis != public,
 179                  Some(&node_foreign_item(*)) => false,
 180                  Some(&node_method(method, impl_did, _)) => {
 181                      match method.vis {
 182                          private => true,
 183                          public => false,
 184                          inherited => f(impl_did.node)
 185                      }
 186                  }
 187                  Some(&node_trait_method(_, trait_did, _)) => f(trait_did.node),
 188                  Some(_) => {
 189                      self.tcx.sess.span_bug(span,
 190                                        fmt!("local_item_is_private: item was \
 191                                              a %s?!",
 192                                             ast_map::node_id_to_str(
 193                                                  self.tcx.items,
 194                                                  item_id,
 195                                                 token::get_ident_interner())));
 196                  }
 197                  None => {
 198                      self.tcx.sess.span_bug(span, "item not found in AST map?!");
 199                  }
 200              }
 201          };
 202          f(item_id)
 203      }

 205      // Checks that a private field is in scope.
 206      // FIXME #6993: change type (and name) from Ident to Name
 207      fn check_field(&mut self, span: Span, id: ast::DefId, ident: ast::Ident) {
 208          let fields = ty::lookup_struct_fields(self.tcx, id);
 209          for field in fields.iter() {
 210              if field.name != ident.name { loop; }
 211              if field.vis == private {
 212                  self.tcx.sess.span_err(span, fmt!("field `%s` is private",
 213                                               token::ident_to_str(&ident)));
 214              }
 215              break;
 216          }
 217      }

 219      // Given the ID of a method, checks to ensure it's in scope.
 220      fn check_method_common(&mut self, span: Span, method_id: DefId, name: &Ident) {
 221          // If the method is a default method, we need to use the def_id of
 222          // the default implementation.
 223          // Having to do this this is really unfortunate.
 224          let method_id = ty::method(self.tcx, method_id).provided_source.unwrap_or(method_id);
 225  
 226          if method_id.crate == LOCAL_CRATE {
 227              let is_private = self.method_is_private(span, method_id.node);
 228              let container_id = ty::method(self.tcx, method_id).container_id();
 229              if is_private &&
 230                      (container_id.crate != LOCAL_CRATE ||
 231                       !self.privileged_items.iter().any(|x| x == &(container_id.node))) {
 232                  self.tcx.sess.span_err(span,
 233                                    fmt!("method `%s` is private",
 234                                         token::ident_to_str(name)));
 235              }
 236          } else {
 237              let visibility =
 238                  csearch::get_item_visibility(self.tcx.sess.cstore, method_id);
 239              if visibility != public {
 240                  self.tcx.sess.span_err(span,
 241                                    fmt!("method `%s` is private",
 242                                         token::ident_to_str(name)));
 243              }
 244          }
 245      }

 247      // Checks that a private path is in scope.
 248      fn check_path(&mut self, span: Span, def: Def, path: &Path) {
 249          debug!("checking path");
 250          match def {
 251              DefStaticMethod(method_id, _, _) => {
 252                  debug!("found static method def, checking it");
 253                  self.check_method_common(span,
 254                                           method_id,
 255                                           &path.segments.last().identifier)
 256              }
 257              DefFn(def_id, _) => {
 258                  if def_id.crate == LOCAL_CRATE {
 259                      if self.local_item_is_private(span, def_id.node) &&
 260                              !self.privileged_items.iter().any(|x| x == &def_id.node) {
 261                          self.tcx.sess.span_err(span,
 262                                            fmt!("function `%s` is private",
 263                                                 token::ident_to_str(
 264                                                  &path.segments
 265                                                       .last()
 266                                                       .identifier)));
 267                      }
 268                  } else if csearch::get_item_visibility(self.tcx.sess.cstore,
 269                                                         def_id) != public {
 270                      self.tcx.sess.span_err(span,
 271                                        fmt!("function `%s` is private",
 272                                             token::ident_to_str(
 273                                                  &path.segments
 274                                                       .last()
 275                                                       .identifier)));
 276                  }
 277              }
 278              _ => {}
 279          }
 280      }

 282      // Checks that a private method is in scope.
 283      fn check_method(&mut self, span: Span, origin: &method_origin, ident: ast::Ident) {
 284          match *origin {
 285              method_static(method_id) => {
 286                  self.check_method_common(span, method_id, &ident)
 287              }
 288              method_param(method_param {
 289                  trait_id: trait_id,
 290                  method_num: method_num,
 291                   _
 292              }) |
 293              method_object(method_object {
 294                  trait_id: trait_id,
 295                  method_num: method_num,
 296                   _
 297              }) => {
 298                  if trait_id.crate == LOCAL_CRATE {
 299                      match self.tcx.items.find(&trait_id.node) {
 300                          Some(&node_item(item, _)) => {
 301                              match item.node {
 302                                  item_trait(_, _, ref methods) => {
 303                                      if method_num >= (*methods).len() {
 304                                          self.tcx.sess.span_bug(span,
 305                                                                 "method number out of range?!");
 306                                      }
 307                                      match (*methods)[method_num] {
 308                                          provided(method)
 309                                               if method.vis == private &&
 310                                               !self.privileged_items.iter()
 311                                               .any(|x| x == &(trait_id.node)) => {
 312                                              self.tcx.sess.span_err(span,
 313                                                                fmt!("method `%s` is private",
 314                                                                     token::ident_to_str(&method
 315                                                                                          .ident)));
 316                                          }
 317                                          provided(_) | required(_) => {
 318                                              // Required methods can't be
 319                                              // private.
 320                                          }
 321                                      }
 322                                  }
 323                                  _ => {
 324                                      self.tcx.sess.span_bug(span, "trait wasn't actually a trait?!");
 325                                  }
 326                              }
 327                          }
 328                          Some(_) => {
 329                              self.tcx.sess.span_bug(span, "trait wasn't an item?!");
 330                          }
 331                          None => {
 332                              self.tcx.sess.span_bug(span,
 333                                                     "trait item wasn't found in the AST map?!");
 334                          }
 335                      }
 336                  } else {
 337                      // FIXME #4732: External crates.
 338                  }
 339              }
 340          }
 341      }
 342  }

 344  impl<'self> Visitor<&'self method_map> for PrivacyVisitor {
 345  
 346      fn visit_mod<'mm>(&mut self, the_module:&_mod, _:Span, _:NodeId,
 347                        method_map:&'mm method_map) {
 348  
 349              let n_added = self.add_privileged_items(the_module.items);
 350  
 351              visit::walk_mod(self, the_module, method_map);
 352  
 353              do n_added.times {
 354                  ignore(self.privileged_items.pop());
 355              }
 356      }
 357  
 358      fn visit_item<'mm>(&mut self, item:@item, method_map:&'mm method_map) {
 359  
 360              // Do not check privacy inside items with the resolve_unexported
 361              // attribute. This is used for the test runner.
 362              if !attr::contains_name(item.attrs, "!resolve_unexported") {
 363                  check_sane_privacy(self.tcx, item);
 364                  visit::walk_item(self, item, method_map);
 365              }
 366      }
 367  
 368      fn visit_block<'mm>(&mut self, block:&Block, method_map:&'mm method_map) {
 369  
 370              // Gather up all the privileged items.
 371              let mut n_added = 0;
 372              for stmt in block.stmts.iter() {
 373                  match stmt.node {
 374                      StmtDecl(decl, _) => {
 375                          match decl.node {
 376                              DeclItem(item) => {
 377                                  self.add_privileged_item(item, &mut n_added);
 378                              }
 379                              _ => {}
 380                          }
 381                      }
 382                      _ => {}
 383                  }
 384              }
 385  
 386              visit::walk_block(self, block, method_map);
 387  
 388              do n_added.times {
 389                  ignore(self.privileged_items.pop());
 390              }
 391  
 392      }
 393  
 394      fn visit_expr<'mm>(&mut self, expr:@Expr, method_map:&'mm method_map) {
 395  
 396              match expr.node {
 397                  ExprField(base, ident, _) => {
 398                      // Method calls are now a special syntactic form,
 399                      // so `a.b` should always be a field.
 400                      assert!(!method_map.contains_key(&expr.id));
 401  
 402                      // With type_autoderef, make sure we don't
 403                      // allow pointers to violate privacy
 404                      match ty::get(ty::type_autoderef(self.tcx, ty::expr_ty(self.tcx,
 405                                                            base))).sty {
 406                          ty_struct(id, _)
 407                          if id.crate != LOCAL_CRATE || !self.privileged_items.iter()
 408                                  .any(|x| x == &(id.node)) => {
 409                              debug!("(privacy checking) checking field access");
 410                              self.check_field(expr.span, id, ident);
 411                          }
 412                          _ => {}
 413                      }
 414                  }
 415                  ExprMethodCall(_, base, ident, _, _, _) => {
 416                      // Ditto
 417                      match ty::get(ty::type_autoderef(self.tcx, ty::expr_ty(self.tcx,
 418                                                            base))).sty {
 419                          ty_enum(id, _) |
 420                          ty_struct(id, _)
 421                          if id.crate != LOCAL_CRATE ||
 422                             !self.privileged_items.iter().any(|x| x == &(id.node)) => {
 423                              match method_map.find(&expr.id) {
 424                                  None => {
 425                                      self.tcx.sess.span_bug(expr.span,
 426                                                        "method call not in \
 427                                                         method map");
 428                                  }
 429                                  Some(ref entry) => {
 430                                      debug!("(privacy checking) checking \
 431                                              impl method");
 432                                      self.check_method(expr.span, &entry.origin, ident);
 433                                  }
 434                              }
 435                          }
 436                          _ => {}
 437                      }
 438                  }
 439                  ExprPath(ref path) => {
 440                      self.check_path(expr.span, self.tcx.def_map.get_copy(&expr.id), path);
 441                  }
 442                  ExprStruct(_, ref fields, _) => {
 443                      match ty::get(ty::expr_ty(self.tcx, expr)).sty {
 444                          ty_struct(id, _) => {

  39  pub fn check_crate<'mm>(tcx: ty::ctxt,
  40                     method_map: &'mm method_map,
  41                     crate: &ast::Crate) {
  42      let privileged_items = @mut ~[];
  43  
  44      // Adds an item to its scope.
  45      let add_privileged_item: @fn(@ast::item, &mut uint) = |item, count| {
  46          match item.node {
  47              item_struct(*) | item_trait(*) | item_enum(*) |
  48              item_fn(*) => {
  49                  privileged_items.push(item.id);
  50                  *count += 1;
  51              }
  52              item_impl(_, _, _, ref methods) => {
  53                  for method in methods.iter() {
  54                      privileged_items.push(method.id);
  55                      *count += 1;
  56                  }
  57                  privileged_items.push(item.id);
  58                  *count += 1;
  59              }
  60              item_foreign_mod(ref foreign_mod) => {
  61                  for foreign_item in foreign_mod.items.iter() {
  62                      privileged_items.push(foreign_item.id);
  63                      *count += 1;
  64                  }
  65              }
  66              _ => {}
  67          }
  68      };
  69  
  70      // Adds items that are privileged to this scope.
  71      let add_privileged_items: @fn(&[@ast::item]) -> uint = |items| {
  72          let mut count = 0;
  73          for &item in items.iter() {
  74              add_privileged_item(item, &mut count);
  75          }
  76          count
  77      };
  78  
  79      // Checks that an enum variant is in scope
  80      let check_variant: @fn(span: span, enum_id: ast::def_id) =
  81              |span, enum_id| {
  82          let variant_info = ty::enum_variants(tcx, enum_id)[0];
  83          let parental_privacy = if is_local(enum_id) {
  84              let parent_vis = ast_map::node_item_query(tcx.items, enum_id.node,
  85                                     |it| { it.vis },
  86                                     ~"unbound enum parent when checking \
  87                                      dereference of enum type");
  88              visibility_to_privacy(parent_vis)
  89          }
  90          else {
  91              // WRONG
  92              Public
  93          };
  94          debug!("parental_privacy = %?", parental_privacy);
  95          debug!("vis = %?, priv = %?",
  96                 variant_info.vis,
  97                 visibility_to_privacy(variant_info.vis))
  98          // inherited => privacy of the enum item
  99          if variant_visibility_to_privacy(variant_info.vis,
 100                                           parental_privacy == Public)
 101                                           == Private {
 102              tcx.sess.span_err(span,
 103                  "can only dereference enums \
 104                   with a single, public variant");
 105          }
 106      };
 107  
 108      // Returns true if a crate-local method is private and false otherwise.
 109      let method_is_private: @fn(span: span, method_id: NodeId) -> bool =
 110              |span, method_id| {
 111          let check = |vis: visibility, container_id: def_id| {
 112              let mut is_private = false;
 113              if vis == private {
 114                  is_private = true;
 115              } else if vis == public {
 116                  is_private = false;
 117              } else {
 118                  // Look up the enclosing impl.
 119                  if container_id.crate != LOCAL_CRATE {
 120                      tcx.sess.span_bug(span,
 121                                        "local method isn't in local \
 122                                         impl?!");
 123                  }
 124  
 125                  match tcx.items.find(&container_id.node) {
 126                      Some(&node_item(item, _)) => {
 127                          match item.node {
 128                              item_impl(_, None, _, _)
 129                                      if item.vis != public => {
 130                                  is_private = true;
 131                              }
 132                              _ => {}
 133                          }
 134                      }
 135                      Some(_) => {
 136                          tcx.sess.span_bug(span, "impl wasn't an item?!");
 137                      }
 138                      None => {
 139                          tcx.sess.span_bug(span, "impl wasn't in AST map?!");
 140                      }
 141                  }
 142              }
 143  
 144              is_private
 145          };
 146  
 147          match tcx.items.find(&method_id) {
 148              Some(&node_method(method, impl_id, _)) => {
 149                  check(method.vis, impl_id)
 150              }
 151              Some(&node_trait_method(trait_method, trait_id, _)) => {
 152                  match *trait_method {
 153                      required(_) => check(public, trait_id),
 154                      provided(method) => check(method.vis, trait_id),
 155                  }
 156              }
 157              Some(_) => {
 158                  tcx.sess.span_bug(span,
 159                                    fmt!("method_is_private: method was a %s?!",
 160                                         ast_map::node_id_to_str(
 161                                              tcx.items,
 162                                              method_id,
 163                                             token::get_ident_interner())));
 164              }
 165              None => {
 166                  tcx.sess.span_bug(span, "method not found in \
 167                                           AST map?!");
 168              }
 169          }
 170      };
 171  
 172      // Returns true if the given local item is private and false otherwise.
 173      let local_item_is_private: @fn(span: span, item_id: NodeId) -> bool =
 174              |span, item_id| {
 175          let mut f: &fn(NodeId) -> bool = |_| false;
 176          f = |item_id| {
 177              match tcx.items.find(&item_id) {
 178                  Some(&node_item(item, _)) => item.vis != public,
 179                  Some(&node_foreign_item(*)) => false,
 180                  Some(&node_method(method, impl_did, _)) => {
 181                      match method.vis {
 182                          private => true,
 183                          public => false,
 184                          inherited => f(impl_did.node)
 185                      }
 186                  }
 187                  Some(&node_trait_method(_, trait_did, _)) => f(trait_did.node),
 188                  Some(_) => {
 189                      tcx.sess.span_bug(span,
 190                                        fmt!("local_item_is_private: item was \
 191                                              a %s?!",
 192                                             ast_map::node_id_to_str(
 193                                                  tcx.items,
 194                                                  item_id,
 195                                                 token::get_ident_interner())));
 196                  }
 197                  None => {
 198                      tcx.sess.span_bug(span, "item not found in AST map?!");
 199                  }
 200              }
 201          };
 202          f(item_id)
 203      };
 204  
 205      // Checks that a private field is in scope.
 206      let check_field: @fn(span: span, id: ast::def_id, ident: ast::ident) =
 207              |span, id, ident| {
 208          let fields = ty::lookup_struct_fields(tcx, id);
 209          for field in fields.iter() {
 210              if field.ident != ident { loop; }
 211              if field.vis == private {
 212                  tcx.sess.span_err(span, fmt!("field `%s` is private",
 213                                               token::ident_to_str(&ident)));
 214              }
 215              break;
 216          }
 217      };
 218  
 219      // Given the ID of a method, checks to ensure it's in scope.
 220      let check_method_common: @fn(span: span,
 221                                   method_id: def_id,
 222                                   name: &ident) =
 223              |span, method_id, name| {
 224          // If the method is a default method, we need to use the def_id of
 225          // the default implementation.
 226          // Having to do this this is really unfortunate.
 227          let method_id = ty::method(tcx, method_id).provided_source
 228              .unwrap_or_default(method_id);
 229  
 230          if method_id.crate == LOCAL_CRATE {
 231              let is_private = method_is_private(span, method_id.node);
 232              let container_id = ty::method(tcx, method_id).container_id;
 233              if is_private &&
 234                      (container_id.crate != LOCAL_CRATE ||
 235                       !privileged_items.iter().any(|x| x == &(container_id.node))) {
 236                  tcx.sess.span_err(span,
 237                                    fmt!("method `%s` is private",
 238                                         token::ident_to_str(name)));
 239              }
 240          } else {
 241              let visibility =
 242                  csearch::get_item_visibility(tcx.sess.cstore, method_id);
 243              if visibility != public {
 244                  tcx.sess.span_err(span,
 245                                    fmt!("method `%s` is private",
 246                                         token::ident_to_str(name)));
 247              }
 248          }
 249      };
 250  
 251      // Checks that a private path is in scope.
 252      let check_path: @fn(span: span, def: def, path: &Path) =
 253              |span, def, path| {
 254          debug!("checking path");
 255          match def {
 256              def_static_method(method_id, _, _) => {
 257                  debug!("found static method def, checking it");
 258                  check_method_common(span, method_id, path.idents.last())
 259              }
 260              def_fn(def_id, _) => {
 261                  if def_id.crate == LOCAL_CRATE {
 262                      if local_item_is_private(span, def_id.node) &&
 263                              !privileged_items.iter().any(|x| x == &def_id.node) {
 264                          tcx.sess.span_err(span,
 265                                            fmt!("function `%s` is private",
 266                                                 token::ident_to_str(path.idents.last())));
 267                      }
 268                  } else if csearch::get_item_visibility(tcx.sess.cstore,
 269                                                         def_id) != public {
 270                      tcx.sess.span_err(span,
 271                                        fmt!("function `%s` is private",
 272                                             token::ident_to_str(path.idents.last())));
 273                  }
 274              }
 275              _ => {}
 276          }
 277      };
 278  
 279      // Checks that a private method is in scope.
 280      let check_method: @fn(span: span,
 281                            origin: &method_origin,
 282                            ident: ast::ident) =
 283              |span, origin, ident| {
 284          match *origin {
 285              method_static(method_id) => {
 286                  check_method_common(span, method_id, &ident)
 287              }
 288              method_param(method_param {
 289                  trait_id: trait_id,
 290                   method_num: method_num,
 291                   _
 292              }) |
 293              method_trait(trait_id, method_num) => {
 294                  if trait_id.crate == LOCAL_CRATE {
 295                      match tcx.items.find(&trait_id.node) {
 296                          Some(&node_item(item, _)) => {
 297                              match item.node {
 298                                  item_trait(_, _, ref methods) => {
 299                                      if method_num >= (*methods).len() {
 300                                          tcx.sess.span_bug(span, "method number out of range?!");
 301                                      }
 302                                      match (*methods)[method_num] {
 303                                          provided(method)
 304                                               if method.vis == private &&
 305                                               !privileged_items.iter()
 306                                               .any(|x| x == &(trait_id.node)) => {
 307                                              tcx.sess.span_err(span,
 308                                                                fmt!("method `%s` is private",
 309                                                                     token::ident_to_str(&method
 310                                                                                          .ident)));
 311                                          }
 312                                          provided(_) | required(_) => {
 313                                              // Required methods can't be
 314                                              // private.
 315                                          }
 316                                      }
 317                                  }
 318                                  _ => {
 319                                      tcx.sess.span_bug(span, "trait wasn't actually a trait?!");
 320                                  }
 321                              }
 322                          }
 323                          Some(_) => {
 324                              tcx.sess.span_bug(span, "trait wasn't an item?!");
 325                          }
 326                          None => {
 327                              tcx.sess.span_bug(span, "trait item wasn't found in the AST map?!");
 328                          }
 329                      }
 330                  } else {
 331                      // FIXME #4732: External crates.
 332                  }
 333              }
 334          }
 335      };
 336  
 337      let visitor = oldvisit::mk_vt(@oldvisit::Visitor {
 338          visit_mod: |the_module, span, node_id, (method_map, visitor)| {
 339              let n_added = add_privileged_items(the_module.items);
 340  
 341              oldvisit::visit_mod(the_module,
 342                                  span,
 343                                  node_id,
 344                                  (method_map, visitor));
 345  
 346              do n_added.times {
 347                  ignore(privileged_items.pop());
 348              }
 349          },
 350          visit_item: |item, (method_map, visitor)| {
 351              // Do not check privacy inside items with the resolve_unexported
 352              // attribute. This is used for the test runner.
 353              if !attr::contains_name(item.attrs, "!resolve_unexported") {
 354                  check_sane_privacy(tcx, item);
 355                  oldvisit::visit_item(item, (method_map, visitor));
 356              }
 357          },
 358          visit_block: |block, (method_map, visitor)| {
 359              // Gather up all the privileged items.
 360              let mut n_added = 0;
 361              for stmt in block.stmts.iter() {
 362                  match stmt.node {
 363                      stmt_decl(decl, _) => {
 364                          match decl.node {
 365                              decl_item(item) => {
 366                                  add_privileged_item(item, &mut n_added);
 367                              }
 368                              _ => {}
 369                          }
 370                      }
 371                      _ => {}
 372                  }
 373              }
 374  
 375              oldvisit::visit_block(block, (method_map, visitor));
 376  
 377              do n_added.times {
 378                  ignore(privileged_items.pop());
 379              }
 380          },
 381          visit_expr: |expr,
 382                       (method_map, visitor):
 383                          (&'mm method_map, oldvisit::vt<&'mm method_map>)| {
 384              match expr.node {
 385                  expr_field(base, ident, _) => {
 386                      // Method calls are now a special syntactic form,
 387                      // so `a.b` should always be a field.
 388                      assert!(!method_map.contains_key(&expr.id));
 389  
 390                      // With type_autoderef, make sure we don't
 391                      // allow pointers to violate privacy
 392                      match ty::get(ty::type_autoderef(tcx, ty::expr_ty(tcx,
 393                                                            base))).sty {
 394                          ty_struct(id, _)
 395                          if id.crate != LOCAL_CRATE || !privileged_items.iter()
 396                                  .any(|x| x == &(id.node)) => {
 397                              debug!("(privacy checking) checking field access");
 398                              check_field(expr.span, id, ident);
 399                          }
 400                          _ => {}
 401                      }
 402                  }
 403                  expr_method_call(_, base, ident, _, _, _) => {
 404                      // Ditto
 405                      match ty::get(ty::type_autoderef(tcx, ty::expr_ty(tcx,
 406                                                            base))).sty {
 407                          ty_enum(id, _) |
 408                          ty_struct(id, _)
 409                          if id.crate != LOCAL_CRATE ||
 410                             !privileged_items.iter().any(|x| x == &(id.node)) => {
 411                              match method_map.find(&expr.id) {
 412                                  None => {
 413                                      tcx.sess.span_bug(expr.span,
 414                                                        "method call not in \
 415                                                         method map");
 416                                  }
 417                                  Some(ref entry) => {
 418                                      debug!("(privacy checking) checking \
 419                                              impl method");
 420                                      check_method(expr.span, &entry.origin, ident);
 421                                  }
 422                              }
 423                          }
 424                          _ => {}
 425                      }
 426                  }
 427                  expr_path(ref path) => {
 428                      check_path(expr.span, tcx.def_map.get_copy(&expr.id), path);
 429                  }
 430                  expr_struct(_, ref fields, _) => {
 431                      match ty::get(ty::expr_ty(tcx, expr)).sty {
 432                          ty_struct(id, _) => {
 433                              if id.crate != LOCAL_CRATE ||
 434                                      !privileged_items.iter().any(|x| x == &(id.node)) {
 435                                  for field in (*fields).iter() {
 436                                          debug!("(privacy checking) checking \
 437                                                  field in struct literal");
 438                                      check_field(expr.span, id, field.ident);
 439                                  }
 440                              }

   51  /// The parent visitor, used to determine what's the parent of what (node-wise)
   52  ////////////////////////////////////////////////////////////////////////////////
   53  
   54  struct ParentVisitor {
   55      parents: NodeMap<ast::NodeId>,
   56      curparent: ast::NodeId,
   57  }
   58  
   59  impl Visitor<()> for ParentVisitor {
   60      fn visit_item(&mut self, item: &ast::Item, _: ()) {
   61          self.parents.insert(item.id, self.curparent);
   62  
   63          let prev = self.curparent;
   64          match item.node {
   65              ast::ItemMod(..) => { self.curparent = item.id; }
   66              // Enum variants are parented to the enum definition itself beacuse
   67              // they inherit privacy
   68              ast::ItemEnum(ref def, _) => {
   69                  for variant in def.variants.iter() {
   70                      // If variants are private, then their logical "parent" is
   71                      // the enclosing module because everyone in the enclosing
   72                      // module can still use the private variant
   73                      if variant.node.vis == ast::Private {
   74                          self.parents.insert(variant.node.id, self.curparent);
   75  
   76                      // Otherwise, if the variant is public, then the parent is
   77                      // considered the enclosing enum because the enum will
   78                      // dictate the privacy visibility of this variant instead.
   79                      } else {
   80                          self.parents.insert(variant.node.id, item.id);
   81                      }
   82                  }
   83              }
   84  
   85              // Trait methods are always considered "public", but if the trait is
   86              // private then we need some private item in the chain from the
   87              // method to the root. In this case, if the trait is private, then
   88              // parent all the methods to the trait to indicate that they're
   89              // private.
   90              ast::ItemTrait(_, _, ref methods) if item.vis != ast::Public => {
   91                  for m in methods.iter() {
   92                      match *m {
   93                          ast::Provided(ref m) => self.parents.insert(m.id, item.id),
   94                          ast::Required(ref m) => self.parents.insert(m.id, item.id),
   95                      };
   96                  }
   97              }
   98  
   99              _ => {}
  100          }
  101          visit::walk_item(self, item, ());
  102          self.curparent = prev;
  103      }
  104  
  105      fn visit_foreign_item(&mut self, a: &ast::ForeignItem, _: ()) {
  106          self.parents.insert(a.id, self.curparent);
  107          visit::walk_foreign_item(self, a, ());
  108      }
  109  
  110      fn visit_fn(&mut self, a: &visit::FnKind, b: &ast::FnDecl,
  111                  c: &ast::Block, d: Span, id: ast::NodeId, _: ()) {
  112          // We already took care of some trait methods above, otherwise things
  113          // like impl methods and pub trait methods are parented to the
  114          // containing module, not the containing trait.
  115          if !self.parents.contains_key(&id) {
  116              self.parents.insert(id, self.curparent);
  117          }
  118          visit::walk_fn(self, a, b, c, d, id, ());
  119      }
  120  
  121      fn visit_struct_def(&mut self, s: &ast::StructDef, i: ast::Ident,
  122                          g: &ast::Generics, n: ast::NodeId, _: ()) {
  123          // Struct constructors are parented to their struct definitions because
  124          // they essentially are the struct definitions.
  125          match s.ctor_id {
  126              Some(id) => { self.parents.insert(id, n); }
  127              None => {}
  128          }
  129  
  130          // While we have the id of the struct definition, go ahead and parent
  131          // all the fields.
  132          for field in s.fields.iter() {
  133              self.parents.insert(field.node.id, self.curparent);
  134          }
  135          visit::walk_struct_def(self, s, i, g, n, ())
  136      }
  137  }

  142  
  143  struct EmbargoVisitor<'a> {
  144      tcx: &'a ty::ctxt,
  145      exp_map2: &'a resolve::ExportMap2,
  146  
  147      // This flag is an indicator of whether the previous item in the
  148      // hierarchical chain was exported or not. This is the indicator of whether
  149      // children should be exported as well. Note that this can flip from false
  150      // to true if a reexported module is entered (or an action similar).
  151      prev_exported: bool,
  152  
  153      // This is a list of all exported items in the AST. An exported item is any
  154      // function/method/item which is usable by external crates. This essentially
  155      // means that the result is "public all the way down", but the "path down"
  156      // may jump across private boundaries through reexport statements.
  157      exported_items: ExportedItems,
  158  
  159      // This sets contains all the destination nodes which are publicly
  160      // re-exported. This is *not* a set of all reexported nodes, only a set of
  161      // all nodes which are reexported *and* reachable from external crates. This
  162      // means that the destination of the reexport is exported, and hence the
  163      // destination must also be exported.
  164      reexports: NodeSet,
  165  
  166      // These two fields are closely related to one another in that they are only
  167      // used for generation of the 'PublicItems' set, not for privacy checking at
  168      // all
  169      public_items: PublicItems,
  170      prev_public: bool,
  171  }

  173  impl<'a> EmbargoVisitor<'a> {
  174      // There are checks inside of privacy which depend on knowing whether a
  175      // trait should be exported or not. The two current consumers of this are:
  176      //
  177      //  1. Should default methods of a trait be exported?
  178      //  2. Should the methods of an implementation of a trait be exported?
  179      //
  180      // The answer to both of these questions partly rely on whether the trait
  181      // itself is exported or not. If the trait is somehow exported, then the
  182      // answers to both questions must be yes. Right now this question involves
  183      // more analysis than is currently done in rustc, so we conservatively
  184      // answer "yes" so that all traits need to be exported.
  185      fn exported_trait(&self, _id: ast::NodeId) -> bool {
  186          true
  187      }
  188  }

  190  impl<'a> Visitor<()> for EmbargoVisitor<'a> {
  191      fn visit_item(&mut self, item: &ast::Item, _: ()) {
  192          let orig_all_pub = self.prev_public;
  193          self.prev_public = orig_all_pub && item.vis == ast::Public;
  194          if self.prev_public {
  195              self.public_items.insert(item.id);
  196          }
  197  
  198          let orig_all_exported = self.prev_exported;
  199          match item.node {
  200              // impls/extern blocks do not break the "public chain" because they
  201              // cannot have visibility qualifiers on them anyway
  202              ast::ItemImpl(..) | ast::ItemForeignMod(..) => {}
  203  
  204              // Traits are a little special in that even if they themselves are
  205              // not public they may still be exported.
  206              ast::ItemTrait(..) => {
  207                  self.prev_exported = self.exported_trait(item.id);
  208              }
  209  
  210              // Private by default, hence we only retain the "public chain" if
  211              // `pub` is explicitly listed.
  212              _ => {
  213                  self.prev_exported =
  214                      (orig_all_exported && item.vis == ast::Public) ||
  215                       self.reexports.contains(&item.id);
  216              }
  217          }
  218  
  219          let public_first = self.prev_exported &&
  220                             self.exported_items.insert(item.id);
  221  
  222          match item.node {
  223              // Enum variants inherit from their parent, so if the enum is
  224              // public all variants are public unless they're explicitly priv
  225              ast::ItemEnum(ref def, _) if public_first => {
  226                  for variant in def.variants.iter() {
  227                      if variant.node.vis != ast::Private {
  228                          self.exported_items.insert(variant.node.id);
  229                      }
  230                  }
  231              }
  232  
  233              // Implementations are a little tricky to determine what's exported
  234              // out of them. Here's a few cases which are currently defined:
  235              //
  236              // * Impls for private types do not need to export their methods
  237              //   (either public or private methods)
  238              //
  239              // * Impls for public types only have public methods exported
  240              //
  241              // * Public trait impls for public types must have all methods
  242              //   exported.
  243              //
  244              // * Private trait impls for public types can be ignored
  245              //
  246              // * Public trait impls for private types have their methods
  247              //   exported. I'm not entirely certain that this is the correct
  248              //   thing to do, but I have seen use cases of where this will cause
  249              //   undefined symbols at linkage time if this case is not handled.
  250              //
  251              // * Private trait impls for private types can be completely ignored
  252              ast::ItemImpl(_, _, ref ty, ref methods) => {
  253                  let public_ty = match ty.node {
  254                      ast::TyPath(_, _, id) => {
  255                          match self.tcx.def_map.borrow().get_copy(&id) {
  256                              ast::DefPrimTy(..) => true,
  257                              def => {
  258                                  let did = def_id_of_def(def);
  259                                  !is_local(did) ||
  260                                   self.exported_items.contains(&did.node)
  261                              }
  262                          }
  263                      }
  264                      _ => true,
  265                  };
  266                  let tr = ty::impl_trait_ref(self.tcx, local_def(item.id));
  267                  let public_trait = tr.map_or(false, |tr| {
  268                      !is_local(tr.def_id) ||
  269                       self.exported_items.contains(&tr.def_id.node)
  270                  });
  271  
  272                  if public_ty || public_trait {
  273                      for method in methods.iter() {
  274                          let meth_public = match method.explicit_self.node {
  275                              ast::SelfStatic => public_ty,
  276                              _ => true,
  277                          } && method.vis == ast::Public;
  278                          if meth_public || tr.is_some() {
  279                              self.exported_items.insert(method.id);
  280                          }
  281                      }
  282                  }
  283              }
  284  
  285              // Default methods on traits are all public so long as the trait
  286              // is public
  287              ast::ItemTrait(_, _, ref methods) if public_first => {
  288                  for method in methods.iter() {
  289                      match *method {
  290                          ast::Provided(ref m) => {
  291                              debug!("provided {}", m.id);
  292                              self.exported_items.insert(m.id);
  293                          }
  294                          ast::Required(ref m) => {
  295                              debug!("required {}", m.id);
  296                              self.exported_items.insert(m.id);
  297                          }
  298                      }
  299                  }
  300              }
  301  
  302              // Struct constructors are public if the struct is all public.
  303              ast::ItemStruct(ref def, _) if public_first => {
  304                  match def.ctor_id {
  305                      Some(id) => { self.exported_items.insert(id); }
  306                      None => {}
  307                  }
  308              }
  309  
  310              _ => {}
  311          }
  312  
  313          visit::walk_item(self, item, ());
  314  
  315          self.prev_exported = orig_all_exported;
  316          self.prev_public = orig_all_pub;
  317      }

  319      fn visit_foreign_item(&mut self, a: &ast::ForeignItem, _: ()) {
  320          if self.prev_exported && a.vis == ast::Public {
  321              self.exported_items.insert(a.id);
  322          }
  323      }

  325      fn visit_mod(&mut self, m: &ast::Mod, _sp: Span, id: ast::NodeId, _: ()) {
  326          // This code is here instead of in visit_item so that the
  327          // crate module gets processed as well.
  328          if self.prev_exported {
  329              let exp_map2 = self.exp_map2.borrow();
  330              assert!(exp_map2.contains_key(&id), "wut {:?}", id);
  331              for export in exp_map2.get(&id).iter() {
  332                  if is_local(export.def_id) {
  333                      self.reexports.insert(export.def_id.node);
  334                  }
  335              }
  336          }
  337          visit::walk_mod(self, m, ())
  338      }
  339  }

  368  impl<'a> PrivacyVisitor<'a> {
  369      // used when debugging
  370      fn nodestr(&self, id: ast::NodeId) -> ~str {
  371          self.tcx.map.node_to_str(id)
  372      }
  373  
  374      // Determines whether the given definition is public from the point of view
  375      // of the current item.
  376      fn def_privacy(&self, did: ast::DefId) -> PrivacyResult {
  377          if !is_local(did) {
  378              if self.external_exports.contains(&did) {
  379                  debug!("privacy - {:?} was externally exported", did);
  380                  return Allowable;
  381              }
  382              debug!("privacy - is {:?} a public method", did);
  383  
  384              return match self.tcx.methods.borrow().find(&did) {
  385                  Some(meth) => {
  386                      debug!("privacy - well at least it's a method: {:?}", meth);
  387                      match meth.container {
  388                          ty::TraitContainer(id) => {
  389                              debug!("privacy - recursing on trait {:?}", id);
  390                              self.def_privacy(id)
  391                          }
  392                          ty::ImplContainer(id) => {
  393                              match ty::impl_trait_ref(self.tcx, id) {
  394                                  Some(t) => {
  395                                      debug!("privacy - impl of trait {:?}", id);
  396                                      self.def_privacy(t.def_id)
  397                                  }
  398                                  None => {
  399                                      debug!("privacy - found a method {:?}",
  400                                              meth.vis);
  401                                      if meth.vis == ast::Public {
  402                                          Allowable
  403                                      } else {
  404                                          ExternallyDenied
  405                                      }
  406                                  }
  407                              }
  408                          }
  409                      }
  410                  }
  411                  None => {
  412                      debug!("privacy - nope, not even a method");
  413                      ExternallyDenied
  414                  }
  415              };
  416          }
  417  
  418          debug!("privacy - local {:?} not public all the way down", did);
  419          // return quickly for things in the same module
  420          if self.parents.find(&did.node) == self.parents.find(&self.curitem) {
  421              debug!("privacy - same parent, we're done here");
  422              return Allowable;
  423          }
  424  
  425          // We now know that there is at least one private member between the
  426          // destination and the root.
  427          let mut closest_private_id = did.node;
  428          loop {
  429              debug!("privacy - examining {}", self.nodestr(closest_private_id));
  430              let vis = match self.tcx.map.find(closest_private_id) {
  431                  // If this item is a method, then we know for sure that it's an
  432                  // actual method and not a static method. The reason for this is
  433                  // that these cases are only hit in the ExprMethodCall
  434                  // expression, and ExprCall will have its path checked later
  435                  // (the path of the trait/impl) if it's a static method.
  436                  //
  437                  // With this information, then we can completely ignore all
  438                  // trait methods. The privacy violation would be if the trait
  439                  // couldn't get imported, not if the method couldn't be used
  440                  // (all trait methods are public).
  441                  //
  442                  // However, if this is an impl method, then we dictate this
  443                  // decision solely based on the privacy of the method
  444                  // invocation.
  445                  // FIXME(#10573) is this the right behavior? Why not consider
  446                  //               where the method was defined?
  447                  Some(ast_map::NodeMethod(ref m)) => {
  448                      let imp = self.tcx.map.get_parent_did(closest_private_id);
  449                      match ty::impl_trait_ref(self.tcx, imp) {
  450                          Some(..) => return Allowable,
  451                          _ if m.vis == ast::Public => return Allowable,
  452                          _ => m.vis
  453                      }
  454                  }
  455                  Some(ast_map::NodeTraitMethod(_)) => {
  456                      return Allowable;
  457                  }
  458  
  459                  // This is not a method call, extract the visibility as one
  460                  // would normally look at it
  461                  Some(ast_map::NodeItem(it)) => it.vis,
  462                  Some(ast_map::NodeForeignItem(_)) => {
  463                      self.tcx.map.get_foreign_vis(closest_private_id)
  464                  }
  465                  Some(ast_map::NodeVariant(ref v)) => {
  466                      // sadly enum variants still inherit visibility, so only
  467                      // break out of this is explicitly private
  468                      if v.node.vis == ast::Private { break }
  469                      ast::Public // need to move up a level (to the enum)
  470                  }
  471                  _ => ast::Public,
  472              };
  473              if vis != ast::Public { break }
  474              // if we've reached the root, then everything was allowable and this
  475              // access is public.
  476              if closest_private_id == ast::CRATE_NODE_ID { return Allowable }
  477              closest_private_id = *self.parents.get(&closest_private_id);
  478  
  479              // If we reached the top, then we were public all the way down and
  480              // we can allow this access.
  481              if closest_private_id == ast::DUMMY_NODE_ID { return Allowable }
  482          }
  483          debug!("privacy - closest priv {}", self.nodestr(closest_private_id));
  484          if self.private_accessible(closest_private_id) {
  485              Allowable
  486          } else {
  487              DisallowedBy(closest_private_id)
  488          }
  489      }
  490  
  491      /// For a local private node in the AST, this function will determine
  492      /// whether the node is accessible by the current module that iteration is
  493      /// inside.
  494      fn private_accessible(&self, id: ast::NodeId) -> bool {
  495          let parent = *self.parents.get(&id);
  496          debug!("privacy - accessible parent {}", self.nodestr(parent));
  497  
  498          // After finding `did`'s closest private member, we roll ourselves back
  499          // to see if this private member's parent is anywhere in our ancestry.
  500          // By the privacy rules, we can access all of our ancestor's private
  501          // members, so that's why we test the parent, and not the did itself.
  502          let mut cur = self.curitem;
  503          loop {
  504              debug!("privacy - questioning {}", self.nodestr(cur));
  505              match cur {
  506                  // If the relevant parent is in our history, then we're allowed
  507                  // to look inside any of our ancestor's immediate private items,
  508                  // so this access is valid.
  509                  x if x == parent => return true,
  510  
  511                  // If we've reached the root, then we couldn't access this item
  512                  // in the first place
  513                  ast::DUMMY_NODE_ID => return false,
  514  
  515                  // Keep going up
  516                  _ => {}
  517              }
  518  
  519              cur = *self.parents.get(&cur);
  520          }
  521      }

  531                  }
  532                  false
  533              },
  534          }
  535      }

  537      /// Guarantee that a particular definition is public. Returns a CheckResult
  538      /// which contains any errors found. These can be reported using `report_error`.
  539      /// If the result is `None`, no errors were found.
  540      fn ensure_public(&self, span: Span, to_check: ast::DefId,
  541                       source_did: Option<ast::DefId>, msg: &str) -> CheckResult {
  542          match self.def_privacy(to_check) {
  543              ExternallyDenied => Some((span, format!("{} is private", msg), None)),
  544              DisallowedBy(id) => {
  545                  let (err_span, err_msg) = if id == source_did.unwrap_or(to_check).node {
  546                      return Some((span, format!("{} is private", msg), None));
  547                  } else {
  548                      (span, format!("{} is inaccessible", msg))
  549                  };
  550                  match self.tcx.map.find(id) {
  551                      Some(ast_map::NodeItem(item)) => {
  552                          let desc = match item.node {
  553                              ast::ItemMod(..) => "module",
  554                              ast::ItemTrait(..) => "trait",
  555                              _ => return Some((err_span, err_msg, None)),
  556                          };
  557                          let msg = format!("{} `{}` is private",
  558                                            desc,
  559                                            token::get_ident(item.ident));
  560                          Some((err_span, err_msg, Some((span, msg))))
  561                      },
  562                      _ => Some((err_span, err_msg, None)),
  563                  }
  564              },
  565              Allowable => None,
  566          }
  567      }

  569      // Checks that a field is in scope.
  570      fn check_field(&mut self, span: Span, id: ast::DefId,
  571                     name: FieldName) {
  572          let fields = ty::lookup_struct_fields(self.tcx, id);
  573          let field = match name {
  574              NamedField(ident) => {
  575                  fields.iter().find(|f| f.name == ident.name).unwrap()
  576              }
  577              UnnamedField(idx) => fields.get(idx)
  578          };
  579          if field.vis == ast::Public ||
  580              (is_local(field.id) && self.private_accessible(field.id.node)) {
  581              return
  582          }
  583  
  584          let struct_type = ty::lookup_item_type(self.tcx, id).ty;
  585          let struct_desc = match ty::get(struct_type).sty {
  586              ty::ty_struct(_, _) => format!("struct `{}`", ty::item_path_str(self.tcx, id)),
  587              ty::ty_bare_fn(ty::BareFnTy { sig: ty::FnSig { output, .. }, .. }) => {
  588                  // Struct `id` is really a struct variant of an enum,
  589                  // and we're really looking at the variant's constructor
  590                  // function. So get the return type for a detailed error
  591                  // message.
  592                  let enum_id = match ty::get(output).sty {
  593                      ty::ty_enum(id, _) => id,
  594                      _ => self.tcx.sess.span_bug(span, "enum variant doesn't \
  595                                                         belong to an enum")
  596                  };
  597                  format!("variant `{}` of enum `{}`",
  598                          ty::with_path(self.tcx, id, |mut p| p.last().unwrap()),
  599                          ty::item_path_str(self.tcx, enum_id))
  600              }
  601              _ => self.tcx.sess.span_bug(span, "can't find struct for field")
  602          };
  603          let msg = match name {
  604              NamedField(name) => format!("field `{}` of {} is private",
  605                                          token::get_ident(name), struct_desc),
  606              UnnamedField(idx) => format!("field \\#{} of {} is private",
  607                                           idx + 1, struct_desc),
  608          };
  609          self.tcx.sess.span_err(span, msg);
  610      }

  612      // Given the ID of a method, checks to ensure it's in scope.
  613      fn check_static_method(&mut self, span: Span, method_id: ast::DefId,
  614                             name: ast::Ident) {
  615          // If the method is a default method, we need to use the def_id of
  616          // the default implementation.
  617          let method_id = ty::method(self.tcx, method_id).provided_source
  618                                                         .unwrap_or(method_id);
  619  
  620          let string = token::get_ident(name);
  621          self.report_error(self.ensure_public(span,
  622                                               method_id,
  623                                               None,
  624                                               format!("method `{}`", string)));
  625      }

  627      // Checks that a path is in scope.
  628      fn check_path(&mut self, span: Span, path_id: ast::NodeId, path: &ast::Path) {
  629          debug!("privacy - path {}", self.nodestr(path_id));
  630          let orig_def = self.tcx.def_map.borrow().get_copy(&path_id);
  631          let ck = |tyname: &str| {
  632              let ck_public = |def: ast::DefId| {
  633                  let name = token::get_ident(path.segments
  634                                                  .last()
  635                                                  .unwrap()
  636                                                  .identifier);
  637                  let origdid = def_id_of_def(orig_def);
  638                  self.ensure_public(span,
  639                                     def,
  640                                     Some(origdid),
  641                                     format!("{} `{}`",
  642                                             tyname,
  643                                             name))
  644              };
  645  
  646              match *self.last_private_map.get(&path_id) {
  647                  resolve::LastMod(resolve::AllPublic) => {},
  648                  resolve::LastMod(resolve::DependsOn(def)) => {
  649                      self.report_error(ck_public(def));
  650                  },
  651                  resolve::LastImport{value_priv: value_priv,
  652                                      value_used: check_value,
  653                                      type_priv: type_priv,
  654                                      type_used: check_type} => {
  655                      // This dance with found_error is because we don't want to report
  656                      // a privacy error twice for the same directive.
  657                      let found_error = match (type_priv, check_type) {
  658                          (Some(resolve::DependsOn(def)), resolve::Used) => {
  659                              !self.report_error(ck_public(def))
  660                          },
  661                          _ => false,
  662                      };
  663                      if !found_error {
  664                          match (value_priv, check_value) {
  665                              (Some(resolve::DependsOn(def)), resolve::Used) => {
  666                                  self.report_error(ck_public(def));
  667                              },
  668                              _ => {},
  669                          }
  670                      }
  671                      // If an import is not used in either namespace, we still
  672                      // want to check that it could be legal. Therefore we check
  673                      // in both namespaces and only report an error if both would
  674                      // be illegal. We only report one error, even if it is
  675                      // illegal to import from both namespaces.
  676                      match (value_priv, check_value, type_priv, check_type) {
  677                          (Some(p), resolve::Unused, None, _) |
  678                          (None, _, Some(p), resolve::Unused) => {
  679                              let p = match p {
  680                                  resolve::AllPublic => None,
  681                                  resolve::DependsOn(def) => ck_public(def),
  682                              };
  683                              if p.is_some() {
  684                                  self.report_error(p);
  685                              }
  686                          },
  687                          (Some(v), resolve::Unused, Some(t), resolve::Unused) => {
  688                              let v = match v {
  689                                  resolve::AllPublic => None,
  690                                  resolve::DependsOn(def) => ck_public(def),
  691                              };
  692                              let t = match t {
  693                                  resolve::AllPublic => None,
  694                                  resolve::DependsOn(def) => ck_public(def),
  695                              };
  696                              match (v, t) {
  697                                  (Some(_), Some(t)) => {
  698                                      self.report_error(Some(t));
  699                                  },
  700                                  _ => {},
  701                              }
  702                          },
  703                          _ => {},
  704                      }
  705                  },
  706              }
  707          };
  708          // FIXME(#12334) Imports can refer to definitions in both the type and
  709          // value namespaces. The privacy information is aware of this, but the
  710          // def map is not. Therefore the names we work out below will not always
  711          // be accurate and we can get slightly wonky error messages (but type
  712          // checking is always correct).
  713          match self.tcx.def_map.borrow().get_copy(&path_id) {
  714              ast::DefStaticMethod(..) => ck("static method"),
  715              ast::DefFn(..) => ck("function"),
  716              ast::DefStatic(..) => ck("static"),
  717              ast::DefVariant(..) => ck("variant"),
  718              ast::DefTy(..) => ck("type"),
  719              ast::DefTrait(..) => ck("trait"),
  720              ast::DefStruct(..) => ck("struct"),
  721              ast::DefMethod(_, Some(..)) => ck("trait method"),
  722              ast::DefMethod(..) => ck("method"),
  723              ast::DefMod(..) => ck("module"),
  724              _ => {}
  725          }
  726      }

  728      // Checks that a method is in scope.
  729      fn check_method(&mut self, span: Span, origin: MethodOrigin,
  730                      ident: ast::Ident) {
  731          match origin {
  732              MethodStatic(method_id) => {

  737              MethodParam(MethodParam { trait_id: trait_id, .. }) |
  738              MethodObject(MethodObject { trait_id: trait_id, .. }) => {
  739                  self.report_error(self.ensure_public(span, trait_id, None,

  741              }
  742          }
  743      }

  759      fn visit_expr(&mut self, expr: &ast::Expr, _: ()) {
  760          match expr.node {
  761              ast::ExprField(base, ident, _) => {
  762                  match ty::get(ty::expr_ty_adjusted(self.tcx, base,
  763                                                     &*self.method_map.borrow())).sty {
  764                      ty::ty_struct(id, _) => {
  765                          self.check_field(expr.span, id, NamedField(ident));
  766                      }
  767                      _ => {}
  768                  }
  769              }
  770              ast::ExprMethodCall(ident, _, _) => {
  771                  let method_call = MethodCall::expr(expr.id);
  772                  match self.method_map.borrow().find(&method_call) {
  773                      None => {
  774                          self.tcx.sess.span_bug(expr.span,
  775                                                  "method call not in \
  776                                                  method map");
  777                      }
  778                      Some(method) => {
  779                          debug!("(privacy checking) checking impl method");
  780                          self.check_method(expr.span, method.origin, ident);
  781                      }
  782                  }
  783              }

 1024  impl<'a> SanePrivacyVisitor<'a> {
 1025      /// Validates all of the visibility qualifers placed on the item given. This
 1026      /// ensures that there are no extraneous qualifiers that don't actually do
 1027      /// anything. In theory these qualifiers wouldn't parse, but that may happen
 1028      /// later on down the road...
 1029      fn check_sane_privacy(&self, item: &ast::Item) {
 1030          let tcx = self.tcx;
 1031          let check_inherited = |sp: Span, vis: ast::Visibility, note: &str| {
 1032              if vis != ast::Inherited {
 1033                  tcx.sess.span_err(sp, "unnecessary visibility qualifier");
 1034                  if note.len() > 0 {
 1035                      tcx.sess.span_note(sp, note);
 1036                  }
 1037              }
 1038          };
 1039          let check_not_priv = |sp: Span, vis: ast::Visibility, note: &str| {
 1040              if vis == ast::Private {
 1041                  tcx.sess.span_err(sp, "unnecessary `priv` qualifier");
 1042                  if note.len() > 0 {
 1043                      tcx.sess.span_note(sp, note);
 1044                  }
 1045              }
 1046          };
 1047          let check_struct = |def: &@ast::StructDef| {
 1048              for f in def.fields.iter() {
 1049                  match f.node.kind {
 1050                      ast::NamedField(_, ast::Private) => {
 1051                          tcx.sess.span_err(f.span, "unnecessary `priv` \
 1052                                                     visibility");
 1053                      }
 1054                      ast::NamedField(..) | ast::UnnamedField(..) => {}
 1055                  }
 1056              }
 1057          };
 1058          match item.node {
 1059              // implementations of traits don't need visibility qualifiers because
 1060              // that's controlled by having the trait in scope.
 1061              ast::ItemImpl(_, Some(..), _, ref methods) => {
 1062                  check_inherited(item.span, item.vis,
 1063                                  "visibility qualifiers have no effect on trait \
 1064                                   impls");
 1065                  for m in methods.iter() {
 1066                      check_inherited(m.span, m.vis, "");
 1067                  }
 1068              }
 1069  
 1070              ast::ItemImpl(_, _, _, ref methods) => {
 1071                  check_inherited(item.span, item.vis,
 1072                                  "place qualifiers on individual methods instead");
 1073                  for i in methods.iter() {
 1074                      check_not_priv(i.span, i.vis, "functions are private by \
 1075                                                     default");
 1076                  }
 1077              }
 1078              ast::ItemForeignMod(ref fm) => {
 1079                  check_inherited(item.span, item.vis,
 1080                                  "place qualifiers on individual functions \
 1081                                   instead");
 1082                  for i in fm.items.iter() {
 1083                      check_not_priv(i.span, i.vis, "functions are private by \
 1084                                                     default");
 1085                  }
 1086              }
 1087  
 1088              ast::ItemEnum(ref def, _) => {
 1089                  for v in def.variants.iter() {
 1090                      match v.node.vis {

 1171                          ast::TupleVariantKind(..) => {}
 1172                      }
 1173                  }
 1174              }
 1175  
 1176              ast::ItemStruct(ref def, _) => check_struct(def),
 1177

 1179                  for m in methods.iter() {

  24  //
  25  // This pass classifies expressions by their constant-ness.
  26  //
  27  // Constant-ness comes in 3 flavours:
  28  //
  29  //   - Integer-constants: can be evaluated by the frontend all the way down
  30  //     to their actual value. They are used in a few places (enum
  31  //     discriminants, switch arms) and are a subset of
  32  //     general-constants. They cover all the integer and integer-ish
  33  //     literals (nil, bool, int, uint, char, iNN, uNN) and all integer
  34  //     operators and copies applied to them.
  35  //
  36  //   - General-constants: can be evaluated by LLVM but not necessarily by
  37  //     the frontend; usually due to reliance on target-specific stuff such
  38  //     as "where in memory the value goes" or "what floating point mode the
  39  //     target uses". This _includes_ integer-constants, plus the following
  40  //     constructors:
  41  //
  42  //        fixed-size vectors and strings: [] and ""/_
  43  //        vector and string slices: &[] and &""
  44  //        tuples: (,)
  45  //        records: {...}
  46  //        enums: foo(...)
  47  //        floating point literals and operators
  48  //        & and * pointers
  49  //        copies of general constants
  50  //
  51  //        (in theory, probably not at first: if/match on integer-const
  52  //         conditions / descriminants)
  53  //
  54  //   - Non-constants: everything else.
  55  //
  56  
  57  pub enum constness {
  58      integral_const,
  59      general_const,
  60      non_const
  61  }
  62  
  63  pub fn join(a: constness, b: constness) -> constness {
  64      match (a, b) {
  65        (integral_const, integral_const) => integral_const,
  66        (integral_const, general_const)
  67        | (general_const, integral_const)
  68        | (general_const, general_const) => general_const,
  69        _ => non_const
  70      }
  71  }
  72  
  73  pub fn join_all<It: Iterator<constness>>(mut cs: It) -> constness {
  74      cs.fold(integral_const, |a, b| join(a, b))
  75  }

  77  pub fn classify(e: &Expr,
  78                  tcx: ty::ctxt)
  79               -> constness {
  80      let did = ast_util::local_def(e.id);
  81      match tcx.ccache.find(&did) {
  82        Some(&x) => x,
  83        None => {
  84          let cn =
  85              match e.node {
  86                ast::ExprLit(lit) => {
  87                  match lit.node {
  88                    ast::lit_str(*) |
  89                    ast::lit_float(*) => general_const,
  90                    _ => integral_const
  91                  }
  92                }
  93  
  94                ast::ExprUnary(_, _, inner) |
  95                ast::ExprParen(inner) => {
  96                  classify(inner, tcx)
  97                }
  98  
  99                ast::ExprBinary(_, _, a, b) => {
 100                  join(classify(a, tcx),
 101                       classify(b, tcx))
 102                }
 103  
 104                ast::ExprTup(ref es) |
 105                ast::ExprVec(ref es, ast::MutImmutable) => {
 106                  join_all(es.iter().map(|e| classify(*e, tcx)))
 107                }
 108  
 109                ast::ExprVstore(e, vstore) => {
 110                    match vstore {
 111                        ast::ExprVstoreSlice => classify(e, tcx),
 112                        ast::ExprVstoreUniq |
 113                        ast::ExprVstoreBox |
 114                        ast::ExprVstoreMutBox |
 115                        ast::ExprVstoreMutSlice => non_const
 116                    }
 117                }
 118  
 119                ast::ExprStruct(_, ref fs, None) => {
 120                  let cs = do fs.iter().map |f| {
 121                      classify(f.expr, tcx)
 122                  };
 123                  join_all(cs)
 124                }
 125  
 126                ast::ExprCast(base, _) => {
 127                  let ty = ty::expr_ty(tcx, e);
 128                  let base = classify(base, tcx);
 129                  if ty::type_is_integral(ty) {
 130                      join(integral_const, base)
 131                  } else if ty::type_is_fp(ty) {
 132                      join(general_const, base)
 133                  } else {
 134                      non_const
 135                  }
 136                }
 137  
 138                ast::ExprField(base, _, _) => {
 139                  classify(base, tcx)
 140                }
 141  
 142                ast::ExprIndex(_, base, idx) => {
 143                  join(classify(base, tcx),
 144                       classify(idx, tcx))
 145                }
 146  
 147                ast::ExprAddrOf(ast::MutImmutable, base) => {
 148                  classify(base, tcx)
 149                }
 150  
 151                // FIXME: (#3728) we can probably do something CCI-ish
 152                // surrounding nonlocal constants. But we don't yet.
 153                ast::ExprPath(_) => {
 154                  lookup_constness(tcx, e)
 155                }
 156  
 157                ast::ExprRepeat(*) => general_const,
 158  
 159                _ => non_const
 160              };
 161          tcx.ccache.insert(did, cn);
 162          cn
 163        }
 164      }
 165  }

 167  pub fn lookup_const(tcx: ty::ctxt, e: &Expr) -> Option<@Expr> {
 168      match tcx.def_map.find(&e.id) {
 169          Some(&ast::DefStatic(def_id, false)) => lookup_const_by_id(tcx, def_id),
 170          Some(&ast::DefVariant(enum_def, variant_def, _)) => lookup_variant_by_id(tcx,
 171                                                                                 enum_def,
 172                                                                                 variant_def),
 173          _ => None
 174      }
 175  }
 176  
 177  pub fn lookup_variant_by_id(tcx: ty::ctxt,
 178                              enum_def: ast::DefId,
 179                              variant_def: ast::DefId)
 180                         -> Option<@Expr> {
 181      fn variant_expr(variants: &[ast::variant], id: ast::NodeId) -> Option<@Expr> {
 182          for variant in variants.iter() {
 183              if variant.node.id == id {
 184                  return variant.node.disr_expr;
 185              }
 186          }
 187          None
 188      }
 189  
 190      if ast_util::is_local(enum_def) {
 191          match tcx.items.find(&enum_def.node) {
 192              None => None,
 193              Some(&ast_map::node_item(it, _)) => match it.node {
 194                  item_enum(ast::enum_def { variants: ref variants }, _) => {
 195                      variant_expr(*variants, variant_def.node)
 196                  }
 197                  _ => None
 198              },
 199              Some(_) => None
 200          }
 201      } else {
 202          let maps = astencode::Maps {
 203              root_map: @mut HashMap::new(),
 204              method_map: @mut HashMap::new(),
 205              vtable_map: @mut HashMap::new(),
 206              write_guard_map: @mut HashSet::new(),
 207              capture_map: @mut HashMap::new()
 208          };
 209          match csearch::maybe_get_item_ast(tcx, enum_def,
 210              |a, b, c, d| astencode::decode_inlined_item(a,
 211                                                          b,
 212                                                          maps,
 213                                                          /*bad*/ c.clone(),
 214                                                          d)) {
 215              csearch::found(ast::ii_item(item)) => match item.node {
 216                  item_enum(ast::enum_def { variants: ref variants }, _) => {
 217                      variant_expr(*variants, variant_def.node)
 218                  }
 219                  _ => None
 220              },
 221              _ => None
 222          }
 223      }
 224  }

 226  pub fn lookup_const_by_id(tcx: ty::ctxt,
 227                            def_id: ast::DefId)
 228                         -> Option<@Expr> {
 229      if ast_util::is_local(def_id) {
 230          match tcx.items.find(&def_id.node) {
 231              None => None,
 232              Some(&ast_map::node_item(it, _)) => match it.node {
 233                  item_static(_, ast::MutImmutable, const_expr) => Some(const_expr),
 234                  _ => None
 235              },
 236              Some(_) => None
 237          }
 238      } else {
 239          let maps = astencode::Maps {
 240              root_map: @mut HashMap::new(),
 241              method_map: @mut HashMap::new(),
 242              vtable_map: @mut HashMap::new(),
 243              write_guard_map: @mut HashSet::new(),
 244              capture_map: @mut HashMap::new()
 245          };
 246          match csearch::maybe_get_item_ast(tcx, def_id,
 247              |a, b, c, d| astencode::decode_inlined_item(a, b, maps, c, d)) {
 248              csearch::found(ast::ii_item(item)) => match item.node {
 249                  item_static(_, ast::MutImmutable, const_expr) => Some(const_expr),
 250                  _ => None
 251              },
 252              _ => None
 253          }
 254      }
 255  }

 257  pub fn lookup_constness(tcx: ty::ctxt, e: &Expr) -> constness {
 258      match lookup_const(tcx, e) {
 259          Some(rhs) => {
 260              let ty = ty::expr_ty(tcx, rhs);
 261              if ty::type_is_integral(ty) {
 262                  integral_const
 263              } else {
 264                  general_const
 265              }
 266          }
 267          None => non_const
 268      }
 269  }

 271  struct ConstEvalVisitor { tcx: ty::ctxt }
 272  
 273  impl Visitor<()> for ConstEvalVisitor {
 274      fn visit_expr_post(&mut self, e:@Expr, _:()) {
 275          classify(e, self.tcx);
 276      }
 277  }

 279  pub fn process_crate(crate: &ast::Crate,
 280                       tcx: ty::ctxt) {
 281      let mut v = ConstEvalVisitor { tcx: tcx };
 282      visit::walk_crate(&mut v, crate, ());
 283      tcx.sess.abort_if_errors();
 284  }

 289  #[deriving(Clone, Eq)]
 290  pub enum const_val {
 291      const_float(f64),
 292      const_int(i64),
 293      const_uint(u64),
 294      const_str(@str),
 295      const_bool(bool)
 296  }

 298  pub fn eval_const_expr(tcx: middle::ty::ctxt, e: &Expr) -> const_val {
 299      match eval_const_expr_partial(&tcx, e) {
 300          Ok(r) => r,
 301          Err(s) => tcx.sess.span_fatal(e.span, s)
 302      }
 303  }
 304  
 305  pub fn eval_const_expr_partial<T: ty::ExprTyProvider>(tcx: &T, e: &Expr)
 306                              -> Result<const_val, ~str> {
 307      use middle::ty;
 308      fn fromb(b: bool) -> Result<const_val, ~str> { Ok(const_int(b as i64)) }
 309      match e.node {
 310        ExprUnary(_, UnNeg, inner) => {
 311          match eval_const_expr_partial(tcx, inner) {
 312            Ok(const_float(f)) => Ok(const_float(-f)),
 313            Ok(const_int(i)) => Ok(const_int(-i)),
 314            Ok(const_uint(i)) => Ok(const_uint(-i)),
 315            Ok(const_str(_)) => Err(~"Negate on string"),
 316            Ok(const_bool(_)) => Err(~"Negate on boolean"),
 317            ref err => ((*err).clone())
 318          }
 319        }
 320        ExprUnary(_, UnNot, inner) => {
 321          match eval_const_expr_partial(tcx, inner) {
 322            Ok(const_int(i)) => Ok(const_int(!i)),
 323            Ok(const_uint(i)) => Ok(const_uint(!i)),
 324            Ok(const_bool(b)) => Ok(const_bool(!b)),
 325            _ => Err(~"Not on float or string")
 326          }
 327        }
 328        ExprBinary(_, op, a, b) => {
 329          match (eval_const_expr_partial(tcx, a),
 330                 eval_const_expr_partial(tcx, b)) {
 331            (Ok(const_float(a)), Ok(const_float(b))) => {
 332              match op {
 333                BiAdd => Ok(const_float(a + b)),
 334                BiSub => Ok(const_float(a - b)),
 335                BiMul => Ok(const_float(a * b)),
 336                BiDiv => Ok(const_float(a / b)),
 337                BiRem => Ok(const_float(a % b)),
 338                BiEq => fromb(a == b),
 339                BiLt => fromb(a < b),
 340                BiLe => fromb(a <= b),
 341                BiNe => fromb(a != b),
 342                BiGe => fromb(a >= b),
 343                BiGt => fromb(a > b),
 344                _ => Err(~"Can't do this op on floats")
 345              }
 346            }
 347            (Ok(const_int(a)), Ok(const_int(b))) => {
 348              match op {
 349                BiAdd => Ok(const_int(a + b)),
 350                BiSub => Ok(const_int(a - b)),
 351                BiMul => Ok(const_int(a * b)),
 352                BiDiv if b == 0 => Err(~"attempted to divide by zero"),
 353                BiDiv => Ok(const_int(a / b)),
 354                BiRem if b == 0 => Err(~"attempted remainder with a divisor of zero"),
 355                BiRem => Ok(const_int(a % b)),
 356                BiAnd | BiBitAnd => Ok(const_int(a & b)),
 357                BiOr | BiBitOr => Ok(const_int(a | b)),
 358                BiBitXor => Ok(const_int(a ^ b)),
 359                BiShl => Ok(const_int(a << b)),
 360                BiShr => Ok(const_int(a >> b)),
 361                BiEq => fromb(a == b),
 362                BiLt => fromb(a < b),
 363                BiLe => fromb(a <= b),
 364                BiNe => fromb(a != b),
 365                BiGe => fromb(a >= b),
 366                BiGt => fromb(a > b)
 367              }
 368            }
 369            (Ok(const_uint(a)), Ok(const_uint(b))) => {
 370              match op {
 371                BiAdd => Ok(const_uint(a + b)),
 372                BiSub => Ok(const_uint(a - b)),
 373                BiMul => Ok(const_uint(a * b)),
 374                BiDiv if b == 0 => Err(~"attempted to divide by zero"),
 375                BiDiv => Ok(const_uint(a / b)),
 376                BiRem if b == 0 => Err(~"attempted remainder with a divisor of zero"),
 377                BiRem => Ok(const_uint(a % b)),
 378                BiAnd | BiBitAnd => Ok(const_uint(a & b)),
 379                BiOr | BiBitOr => Ok(const_uint(a | b)),
 380                BiBitXor => Ok(const_uint(a ^ b)),
 381                BiShl => Ok(const_uint(a << b)),
 382                BiShr => Ok(const_uint(a >> b)),
 383                BiEq => fromb(a == b),
 384                BiLt => fromb(a < b),
 385                BiLe => fromb(a <= b),
 386                BiNe => fromb(a != b),
 387                BiGe => fromb(a >= b),
 388                BiGt => fromb(a > b),
 389              }
 390            }
 391            // shifts can have any integral type as their rhs
 392            (Ok(const_int(a)), Ok(const_uint(b))) => {
 393              match op {
 394                BiShl => Ok(const_int(a << b)),
 395                BiShr => Ok(const_int(a >> b)),
 396                _ => Err(~"Can't do this op on an int and uint")
 397              }
 398            }
 399            (Ok(const_uint(a)), Ok(const_int(b))) => {
 400              match op {
 401                BiShl => Ok(const_uint(a << b)),
 402                BiShr => Ok(const_uint(a >> b)),
 403                _ => Err(~"Can't do this op on a uint and int")
 404              }
 405            }
 406            (Ok(const_bool(a)), Ok(const_bool(b))) => {
 407              Ok(const_bool(match op {
 408                BiAnd => a && b,
 409                BiOr => a || b,
 410                BiBitXor => a ^ b,
 411                BiBitAnd => a & b,
 412                BiBitOr => a | b,
 413                BiEq => a == b,
 414                BiNe => a != b,
 415                _ => return Err(~"Can't do this op on bools")
 416               }))
 417            }
 418            _ => Err(~"Bad operands for binary")
 419          }
 420        }
 421        ExprCast(base, _) => {
 422          let ety = tcx.expr_ty(e);
 423          let base = eval_const_expr_partial(tcx, base);
 424          match base {
 425              Err(_) => base,
 426              Ok(val) => {
 427                  match ty::get(ety).sty {
 428                      ty::ty_float(_) => {
 429                          match val {
 430                              const_uint(u) => Ok(const_float(u as f64)),
 431                              const_int(i) => Ok(const_float(i as f64)),
 432                              const_float(f) => Ok(const_float(f)),
 433                              _ => Err(~"Can't cast float to str"),
 434                          }
 435                      }

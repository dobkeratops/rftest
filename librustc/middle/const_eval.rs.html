  23 //
  24 // This pass classifies expressions by their constant-ness.
  25 //
  26 // Constant-ness comes in 3 flavours:
  27 //
  28 //   - Integer-constants: can be evaluated by the frontend all the way down
  29 //     to their actual value. They are used in a few places (enum
  30 //     discriminants, switch arms) and are a subset of
  31 //     general-constants. They cover all the integer and integer-ish
  32 //     literals (nil, bool, int, uint, char, iNN, uNN) and all integer
  33 //     operators and copies applied to them.
  34 //
  35 //   - General-constants: can be evaluated by LLVM but not necessarily by
  36 //     the frontend; usually due to reliance on target-specific stuff such
  37 //     as "where in memory the value goes" or "what floating point mode the
  38 //     target uses". This _includes_ integer-constants, plus the following
  39 //     constructors:
  40 //
  41 //        fixed-size vectors and strings: [] and ""/_
  42 //        vector and string slices: &[] and &""
  43 //        tuples: (,)
  44 //        records: {...}
  45 //        enums: foo(...)
  46 //        floating point literals and operators
  47 //        & and * pointers
  48 //        copies of general constants
  49 //
  50 //        (in theory, probably not at first: if/match on integer-const
  51 //         conditions / descriminants)
  52 //
  53 //   - Non-constants: everything else.
  54 //
  55 
  56 pub enum constness {
  57     integral_const,
  58     general_const,
  59     non_const
  60 }
  61 
  62 pub fn join(a: constness, b: constness) -> constness {
  63     match (a, b) {
  64       (integral_const, integral_const) => integral_const,
  65       (integral_const, general_const)
  66       | (general_const, integral_const)
  67       | (general_const, general_const) => general_const,
  68       _ => non_const
  69     }
  70 }
  71 
  72 pub fn join_all<It: Iterator<constness>>(mut cs: It) -> constness {
  73     cs.fold(integral_const, |a, b| join(a, b))
  74 }

  76 pub fn classify(e: &expr,
  77                 tcx: ty::ctxt)
  78              -> constness {
  79     let did = ast_util::local_def(e.id);
  80     match tcx.ccache.find(&did) {
  81       Some(&x) => x,
  82       None => {
  83         let cn =
  84             match e.node {
  85               ast::expr_lit(lit) => {
  86                 match lit.node {
  87                   ast::lit_str(*) |
  88                   ast::lit_float(*) => general_const,
  89                   _ => integral_const
  90                 }
  91               }
  92 
  93               ast::expr_unary(_, _, inner) |
  94               ast::expr_paren(inner) => {
  95                 classify(inner, tcx)
  96               }
  97 
  98               ast::expr_binary(_, _, a, b) => {
  99                 join(classify(a, tcx),
 100                      classify(b, tcx))
 101               }
 102 
 103               ast::expr_tup(ref es) |
 104               ast::expr_vec(ref es, ast::m_imm) => {
 105                 join_all(es.iter().transform(|e| classify(*e, tcx)))
 106               }
 107 
 108               ast::expr_vstore(e, vstore) => {
 109                   match vstore {
 110                       ast::expr_vstore_slice => classify(e, tcx),
 111                       ast::expr_vstore_uniq |
 112                       ast::expr_vstore_box |
 113                       ast::expr_vstore_mut_box |
 114                       ast::expr_vstore_mut_slice => non_const
 115                   }
 116               }
 117 
 118               ast::expr_struct(_, ref fs, None) => {
 119                 let cs = do fs.iter().transform |f| {
 120                     classify(f.expr, tcx)
 121                 };
 122                 join_all(cs)
 123               }
 124 
 125               ast::expr_cast(base, _) => {
 126                 let ty = ty::expr_ty(tcx, e);
 127                 let base = classify(base, tcx);
 128                 if ty::type_is_integral(ty) {
 129                     join(integral_const, base)
 130                 } else if ty::type_is_fp(ty) {
 131                     join(general_const, base)
 132                 } else {
 133                     non_const
 134                 }
 135               }
 136 
 137               ast::expr_field(base, _, _) => {
 138                 classify(base, tcx)
 139               }
 140 
 141               ast::expr_index(_, base, idx) => {
 142                 join(classify(base, tcx),
 143                      classify(idx, tcx))
 144               }
 145 
 146               ast::expr_addr_of(ast::m_imm, base) => {
 147                 classify(base, tcx)
 148               }
 149 
 150               // FIXME: (#3728) we can probably do something CCI-ish
 151               // surrounding nonlocal constants. But we don't yet.
 152               ast::expr_path(_) => {
 153                 lookup_constness(tcx, e)
 154               }
 155 
 156               _ => non_const
 157             };
 158         tcx.ccache.insert(did, cn);
 159         cn
 160       }
 161     }
 162 }

 164 pub fn lookup_const(tcx: ty::ctxt, e: &expr) -> Option<@expr> {
 165     match tcx.def_map.find(&e.id) {
 166         Some(&ast::def_static(def_id, false)) => lookup_const_by_id(tcx, def_id),
 167         Some(&ast::def_variant(enum_def, variant_def)) => lookup_variant_by_id(tcx,
 168                                                                                enum_def,
 169                                                                                variant_def),
 170         _ => None
 171     }
 172 }
 173 
 174 pub fn lookup_variant_by_id(tcx: ty::ctxt,
 175                             enum_def: ast::def_id,
 176                             variant_def: ast::def_id)
 177                        -> Option<@expr> {
 178     fn variant_expr(variants: &[ast::variant], id: ast::NodeId) -> Option<@expr> {
 179         for variant in variants.iter() {
 180             if variant.node.id == id {
 181                 return variant.node.disr_expr;
 182             }
 183         }
 184         None
 185     }
 186 
 187     if ast_util::is_local(enum_def) {
 188         match tcx.items.find(&enum_def.node) {
 189             None => None,
 190             Some(&ast_map::node_item(it, _)) => match it.node {
 191                 item_enum(ast::enum_def { variants: ref variants }, _) => {
 192                     variant_expr(*variants, variant_def.node)
 193                 }
 194                 _ => None
 195             },
 196             Some(_) => None
 197         }
 198     } else {
 199         let maps = astencode::Maps {
 200             root_map: @mut HashMap::new(),
 201             method_map: @mut HashMap::new(),
 202             vtable_map: @mut HashMap::new(),
 203             write_guard_map: @mut HashSet::new(),
 204             capture_map: @mut HashMap::new()
 205         };
 206         match csearch::maybe_get_item_ast(tcx, enum_def,
 207             |a, b, c, d| astencode::decode_inlined_item(a,
 208                                                         b,
 209                                                         maps,
 210                                                         /*bad*/ c.clone(),
 211                                                         d)) {
 212             csearch::found(ast::ii_item(item)) => match item.node {
 213                 item_enum(ast::enum_def { variants: ref variants }, _) => {
 214                     variant_expr(*variants, variant_def.node)
 215                 }
 216                 _ => None
 217             },
 218             _ => None
 219         }
 220     }
 221 }

 223 pub fn lookup_const_by_id(tcx: ty::ctxt,
 224                           def_id: ast::def_id)
 225                        -> Option<@expr> {
 226     if ast_util::is_local(def_id) {
 227         match tcx.items.find(&def_id.node) {
 228             None => None,
 229             Some(&ast_map::node_item(it, _)) => match it.node {
 230                 item_static(_, ast::m_imm, const_expr) => Some(const_expr),
 231                 _ => None
 232             },
 233             Some(_) => None
 234         }
 235     } else {
 236         let maps = astencode::Maps {
 237             root_map: @mut HashMap::new(),
 238             method_map: @mut HashMap::new(),
 239             vtable_map: @mut HashMap::new(),
 240             write_guard_map: @mut HashSet::new(),
 241             capture_map: @mut HashMap::new()
 242         };
 243         match csearch::maybe_get_item_ast(tcx, def_id,
 244             |a, b, c, d| astencode::decode_inlined_item(a, b, maps, c, d)) {
 245             csearch::found(ast::ii_item(item)) => match item.node {
 246                 item_static(_, ast::m_imm, const_expr) => Some(const_expr),
 247                 _ => None
 248             },
 249             _ => None
 250         }
 251     }
 252 }

 254 pub fn lookup_constness(tcx: ty::ctxt, e: &expr) -> constness {
 255     match lookup_const(tcx, e) {
 256         Some(rhs) => {
 257             let ty = ty::expr_ty(tcx, rhs);
 258             if ty::type_is_integral(ty) {
 259                 integral_const
 260             } else {
 261                 general_const
 262             }
 263         }
 264         None => non_const
 265     }
 266 }

 268 pub fn process_crate(crate: &ast::Crate,
 269                      tcx: ty::ctxt) {
 270     let v = oldvisit::mk_simple_visitor(@oldvisit::SimpleVisitor {
 271         visit_expr_post: |e| { classify(e, tcx); },
 272         .. *oldvisit::default_simple_visitor()
 273     });
 274     oldvisit::visit_crate(crate, ((), v));
 275     tcx.sess.abort_if_errors();
 276 }

 290 pub fn eval_const_expr(tcx: middle::ty::ctxt, e: &expr) -> const_val {
 291     match eval_const_expr_partial(&tcx, e) {
 292         Ok(r) => r,
 293         Err(s) => tcx.sess.span_fatal(e.span, s)
 294     }
 295 }
 296 
 297 pub fn eval_const_expr_partial<T: ty::ExprTyProvider>(tcx: &T, e: &expr)
 298                             -> Result<const_val, ~str> {
 299     use middle::ty;
 300     fn fromb(b: bool) -> Result<const_val, ~str> { Ok(const_int(b as i64)) }
 301     match e.node {
 302       expr_unary(_, neg, inner) => {
 303         match eval_const_expr_partial(tcx, inner) {
 304           Ok(const_float(f)) => Ok(const_float(-f)),
 305           Ok(const_int(i)) => Ok(const_int(-i)),
 306           Ok(const_uint(i)) => Ok(const_uint(-i)),
 307           Ok(const_str(_)) => Err(~"Negate on string"),
 308           Ok(const_bool(_)) => Err(~"Negate on boolean"),
 309           ref err => ((*err).clone())
 310         }
 311       }
 312       expr_unary(_, not, inner) => {
 313         match eval_const_expr_partial(tcx, inner) {
 314           Ok(const_int(i)) => Ok(const_int(!i)),
 315           Ok(const_uint(i)) => Ok(const_uint(!i)),
 316           Ok(const_bool(b)) => Ok(const_bool(!b)),
 317           _ => Err(~"Not on float or string")
 318         }
 319       }
 320       expr_binary(_, op, a, b) => {
 321         match (eval_const_expr_partial(tcx, a),
 322                eval_const_expr_partial(tcx, b)) {
 323           (Ok(const_float(a)), Ok(const_float(b))) => {
 324             match op {
 325               add => Ok(const_float(a + b)),
 326               subtract => Ok(const_float(a - b)),
 327               mul => Ok(const_float(a * b)),
 328               div => Ok(const_float(a / b)),
 329               rem => Ok(const_float(a % b)),
 330               eq => fromb(a == b),
 331               lt => fromb(a < b),
 332               le => fromb(a <= b),
 333               ne => fromb(a != b),
 334               ge => fromb(a >= b),
 335               gt => fromb(a > b),
 336               _ => Err(~"Can't do this op on floats")
 337             }
 338           }
 339           (Ok(const_int(a)), Ok(const_int(b))) => {
 340             match op {
 341               add => Ok(const_int(a + b)),
 342               subtract => Ok(const_int(a - b)),
 343               mul => Ok(const_int(a * b)),
 344               div if b == 0 => Err(~"attempted to divide by zero"),
 345               div => Ok(const_int(a / b)),
 346               rem if b == 0 => Err(~"attempted remainder with a divisor of zero"),
 347               rem => Ok(const_int(a % b)),
 348               and | bitand => Ok(const_int(a & b)),
 349               or | bitor => Ok(const_int(a | b)),
 350               bitxor => Ok(const_int(a ^ b)),
 351               shl => Ok(const_int(a << b)),
 352               shr => Ok(const_int(a >> b)),
 353               eq => fromb(a == b),
 354               lt => fromb(a < b),
 355               le => fromb(a <= b),
 356               ne => fromb(a != b),
 357               ge => fromb(a >= b),
 358               gt => fromb(a > b)
 359             }
 360           }
 361           (Ok(const_uint(a)), Ok(const_uint(b))) => {
 362             match op {
 363               add => Ok(const_uint(a + b)),
 364               subtract => Ok(const_uint(a - b)),
 365               mul => Ok(const_uint(a * b)),
 366               div if b == 0 => Err(~"attempted to divide by zero"),
 367               div => Ok(const_uint(a / b)),
 368               rem if b == 0 => Err(~"attempted remainder with a divisor of zero"),
 369               rem => Ok(const_uint(a % b)),
 370               and | bitand => Ok(const_uint(a & b)),
 371               or | bitor => Ok(const_uint(a | b)),
 372               bitxor => Ok(const_uint(a ^ b)),
 373               shl => Ok(const_uint(a << b)),
 374               shr => Ok(const_uint(a >> b)),
 375               eq => fromb(a == b),
 376               lt => fromb(a < b),
 377               le => fromb(a <= b),
 378               ne => fromb(a != b),
 379               ge => fromb(a >= b),
 380               gt => fromb(a > b),
 381             }
 382           }
 383           // shifts can have any integral type as their rhs
 384           (Ok(const_int(a)), Ok(const_uint(b))) => {
 385             match op {
 386               shl => Ok(const_int(a << b)),
 387               shr => Ok(const_int(a >> b)),
 388               _ => Err(~"Can't do this op on an int and uint")
 389             }
 390           }
 391           (Ok(const_uint(a)), Ok(const_int(b))) => {
 392             match op {
 393               shl => Ok(const_uint(a << b)),
 394               shr => Ok(const_uint(a >> b)),
 395               _ => Err(~"Can't do this op on a uint and int")
 396             }
 397           }
 398           (Ok(const_bool(a)), Ok(const_bool(b))) => {
 399             Ok(const_bool(match op {
 400               and => a && b,
 401               or => a || b,
 402               bitxor => a ^ b,
 403               bitand => a & b,
 404               bitor => a | b,
 405               eq => a == b,
 406               ne => a != b,
 407               _ => return Err(~"Can't do this op on bools")
 408              }))
 409           }
 410           _ => Err(~"Bad operands for binary")
 411         }
 412       }
 413       expr_cast(base, _) => {
 414         let ety = tcx.expr_ty(e);
 415         let base = eval_const_expr_partial(tcx, base);
 416         match base {
 417             Err(_) => base,
 418             Ok(val) => {
 419                 match ty::get(ety).sty {
 420                     ty::ty_float(_) => {
 421                         match val {
 422                             const_uint(u) => Ok(const_float(u as f64)),
 423                             const_int(i) => Ok(const_float(i as f64)),
 424                             const_float(f) => Ok(const_float(f)),
 425                             _ => Err(~"Can't cast float to str"),
 426                         }
 427                     }
 428                     ty::ty_uint(_) => {
 429                         match val {
 430                             const_uint(u) => Ok(const_uint(u)),
 431                             const_int(i) => Ok(const_uint(i as u64)),
 432                             const_float(f) => Ok(const_uint(f as u64)),
 433                             _ => Err(~"Can't cast str to uint"),
 434                         }
 435                     }
 436                     ty::ty_int(_) | ty::ty_bool => {
 437                         match val {
 438                             const_uint(u) => Ok(const_int(u as i64)),
 439                             const_int(i) => Ok(const_int(i)),
 440                             const_float(f) => Ok(const_int(f as i64)),
 441                             _ => Err(~"Can't cast str to int"),
 442                         }
 443                     }
 444                     _ => Err(~"Can't cast this type")
 445                 }
 446             }
 447         }
 448       }

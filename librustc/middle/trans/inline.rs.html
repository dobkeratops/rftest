  25 pub fn maybe_instantiate_inline(ccx: @mut CrateContext, fn_id: ast::def_id)
  26     -> ast::def_id {
  27     let _icx = push_ctxt("maybe_instantiate_inline");
  28     match ccx.external.find(&fn_id) {
  29         Some(&Some(node_id)) => {
  30             // Already inline
  31             debug!("maybe_instantiate_inline(%s): already inline as node id %d",
  32                    ty::item_path_str(ccx.tcx, fn_id), node_id);
  33             return local_def(node_id);
  34         }
  35         Some(&None) => {
  36             return fn_id; // Not inlinable
  37         }
  38         None => {
  39             // Not seen yet
  40         }
  41     }
  42 
  43     let csearch_result =
  44         csearch::maybe_get_item_ast(
  45             ccx.tcx, fn_id,
  46             |a,b,c,d| {
  47                 astencode::decode_inlined_item(a, b, ccx.maps, c.clone(), d)
  48             });
  49     return match csearch_result {
  50         csearch::not_found => {
  51             ccx.external.insert(fn_id, None);
  52             fn_id
  53         }
  54         csearch::found(ast::ii_item(item)) => {
  55             ccx.external.insert(fn_id, Some(item.id));
  56             ccx.stats.n_inlines += 1;
  57             trans_item(ccx, item);
  58             local_def(item.id)
  59         }
  60         csearch::found(ast::ii_foreign(item)) => {
  61           ccx.external.insert(fn_id, Some(item.id));
  62           local_def(item.id)
  63         }
  64         csearch::found_parent(parent_id, ast::ii_item(item)) => {
  65           ccx.external.insert(parent_id, Some(item.id));
  66           let mut my_id = 0;
  67           match item.node {
  68             ast::item_enum(_, _) => {
  69               let vs_here = ty::enum_variants(ccx.tcx, local_def(item.id));
  70               let vs_there = ty::enum_variants(ccx.tcx, parent_id);
  71               for (here, there) in vs_here.iter().zip(vs_there.iter()) {
  72                   if there.id == fn_id { my_id = here.id.node; }
  73                   ccx.external.insert(there.id, Some(here.id.node));
  74               }
  75             }
  76             _ => ccx.sess.bug("maybe_instantiate_inline: item has a \
  77                                non-enum parent")
  78           }
  79           trans_item(ccx, item);
  80           local_def(my_id)
  81         }
  82         csearch::found_parent(_, _) => {
  83             ccx.sess.bug("maybe_get_item_ast returned a found_parent \
  84              with a non-item parent");
  85         }
  86         csearch::found(ast::ii_method(impl_did, is_provided, mth)) => {
  87           ccx.stats.n_inlines += 1;
  88           ccx.external.insert(fn_id, Some(mth.id));
  89           // If this is a default method, we can't look up the
  90           // impl type. But we aren't going to translate anyways, so don't.
  91           if is_provided { return local_def(mth.id); }
  92 
  93             let impl_tpt = ty::lookup_item_type(ccx.tcx, impl_did);
  94             let num_type_params =
  95                 impl_tpt.generics.type_param_defs.len() +
  96                 mth.generics.ty_params.len();
  97 
  98           if num_type_params == 0 {
  99               let llfn = get_item_val(ccx, mth.id);
 100               let path = vec::append(
 101                   ty::item_path(ccx.tcx, impl_did),
 102                   [path_name(mth.ident)]);
 103               let self_kind = match mth.explicit_self.node {
 104                   ast::sty_static => no_self,
 105                   _ => {
 106                       let self_ty = ty::node_id_to_type(ccx.tcx,
 107                                                         mth.self_id);
 108                       debug!("calling inline trans_fn with self_ty %s",
 109                              ty_to_str(ccx.tcx, self_ty));
 110                       match mth.explicit_self.node {
 111                           ast::sty_value => impl_self(self_ty, ty::ByRef),
 112                           _ => impl_self(self_ty, ty::ByCopy),
 113                       }
 114                   }
 115               };
 116               trans_fn(ccx,
 117                        path,
 118                        &mth.decl,
 119                        &mth.body,
 120                        llfn,
 121                        self_kind,
 122                        None,
 123                        mth.id,
 124                        []);
 125           }
 126           local_def(mth.id)
 127         }
 128     };
 129 }

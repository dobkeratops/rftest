  47              |a,b,c,d| {

  54          }

 123                  mth.generics.ty_params.len();

 126                let llfn = get_item_val(ccx, mth.id);
 127                let path = vec::append(
 128                    ty::item_path(ccx.tcx, impl_did),
 129                    [path_name(mth.ident)]);
 130                let self_kind = match mth.explicit_self.node {
 131                    ast::sty_static => no_self,
 132                    _ => {

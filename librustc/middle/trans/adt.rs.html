  11  /*!
  12   * # Representation of Algebraic Data Types
  13   *
  14   * This module determines how to represent enums, structs, and tuples
  15   * based on their monomorphized types; it is responsible both for
  16   * choosing a representation and translating basic operations on
  17   * values of those types.  (Note: exporting the representations for
  18   * debuggers is handled in debuginfo.rs, not here.)
  19   *
  20   * Note that the interface treats everything as a general case of an
  21   * enum, so structs/tuples/etc. have one pseudo-variant with
  22   * discriminant 0; i.e., as if they were a univariant enum.
  23   *
  24   * Having everything in one place will enable improvements to data
  25   * structure representation; possibilities include:
  26   *
  27   * - User-specified alignment (e.g., cacheline-aligning parts of
  28   *   concurrently accessed data structures); LLVM can't represent this
  29   *   directly, so we'd have to insert padding fields in any structure
  30   *   that might contain one and adjust GEP indices accordingly.  See
  31   *   issue #4578.
  32   *
  33   * - Store nested enums' discriminants in the same word.  Rather, if
  34   *   some variants start with enums, and those enums representations
  35   *   have unused alignment padding between discriminant and body, the
  36   *   outer enum's discriminant can be stored there and those variants
  37   *   can start at offset 0.  Kind of fancy, and might need work to
  38   *   make copies of the inner enum type cooperate, but it could help
  39   *   with `Option` or `Result` wrapped around another enum.
  40   *
  41   * - Tagged pointers would be neat, but given that any type can be
  42   *   used unboxed and any field can have pointers (including mutable)
  43   *   taken to it, implementing them for Rust seems difficult.
  44   */

  69  /// Representations.
  70  pub enum Repr {
  71      /// C-like enums; basically an int.
  72      CEnum(IntType, Disr, Disr), // discriminant range (signedness based on the IntType)
  73      /**
  74       * Single-case variants, and structs/tuples/records.
  75       *
  76       * Structs with destructors need a dynamic destroyedness flag to
  77       * avoid running the destructor too many times; this is included
  78       * in the `Struct` if present.
  79       */
  80      Univariant(Struct, bool),
  81      /**
  82       * General-case enums: for each case there is a struct, and they
  83       * all start with a field for the discriminant.
  84       */
  85      General(IntType, Vec<Struct>),
  86      /**
  87       * Two cases distinguished by a nullable pointer: the case with discriminant
  88       * `nndiscr` is represented by the struct `nonnull`, where the `ptrfield`th
  89       * field is known to be nonnull due to its type; if that field is null, then
  90       * it represents the other case, which is inhabited by at most one value
  91       * (and all other fields are undefined/unused).
  92       *
  93       * For example, `std::option::Option` instantiated at a safe pointer type
  94       * is represented such that `None` is a null pointer and `Some` is the
  95       * identity function.
  96       */
  97      NullablePointer {
  98          pub nonnull: Struct,
  99          pub nndiscr: Disr,
 100          pub ptrfield: uint,
 101          pub nullfields: Vec<ty::t>,
 102      }
 103  }

 122  /// Decides how to represent a given type.
 123  pub fn represent_type(cx: &CrateContext, t: ty::t) -> @Repr {
 124      debug!("Representing: {}", ty_to_str(cx.tcx(), t));
 125      match cx.adt_reprs.borrow().find(&t) {
 126          Some(repr) => return *repr,
 127          None => {}
 128      }
 129  
 130      let repr = @represent_type_uncached(cx, t);
 131      debug!("Represented as: {:?}", repr)
 132      cx.adt_reprs.borrow_mut().insert(t, repr);
 133      return repr;
 134  }

 136  fn represent_type_uncached(cx: &CrateContext, t: ty::t) -> Repr {
 137      match ty::get(t).sty {
 138          ty::ty_tup(ref elems) => {
 139              return Univariant(mk_struct(cx, elems.as_slice(), false), false)
 140          }
 141          ty::ty_struct(def_id, ref substs) => {
 142              let fields = ty::lookup_struct_fields(cx.tcx(), def_id);
 143              let mut ftys = fields.iter().map(|field| {
 144                  ty::lookup_field_type(cx.tcx(), def_id, field.id, substs)
 145              }).collect::<Vec<_>>();
 146              let packed = ty::lookup_packed(cx.tcx(), def_id);
 147              let dtor = ty::ty_dtor(cx.tcx(), def_id).has_drop_flag();
 148              if dtor { ftys.push(ty::mk_bool()); }
 149  
 150              return Univariant(mk_struct(cx, ftys.as_slice(), packed), dtor)
 151          }
 152          ty::ty_enum(def_id, ref substs) => {
 153              let cases = get_cases(cx.tcx(), def_id, substs);
 154              let hint = ty::lookup_repr_hint(cx.tcx(), def_id);
 155  
 156              if cases.len() == 0 {
 157                  // Uninhabitable; represent as unit
 158                  // (Typechecking will reject discriminant-sizing attrs.)
 159                  assert_eq!(hint, attr::ReprAny);
 160                  return Univariant(mk_struct(cx, [], false), false);
 161              }
 162  
 163              if cases.iter().all(|c| c.tys.len() == 0) {
 164                  // All bodies empty -> intlike
 165                  let discrs: Vec<u64> = cases.iter().map(|c| c.discr).collect();
 166                  let bounds = IntBounds {
 167                      ulo: *discrs.iter().min().unwrap(),
 168                      uhi: *discrs.iter().max().unwrap(),
 169                      slo: discrs.iter().map(|n| *n as i64).min().unwrap(),
 170                      shi: discrs.iter().map(|n| *n as i64).max().unwrap()
 171                  };
 172                  return mk_cenum(cx, hint, &bounds);
 173              }
 174  
 175              // Since there's at least one
 176              // non-empty body, explicit discriminants should have
 177              // been rejected by a checker before this point.
 178              if !cases.iter().enumerate().all(|(i,c)| c.discr == (i as Disr)) {
 179                  cx.sess().bug(format!("non-C-like enum {} with specified \
 180                                        discriminants",
 181                                        ty::item_path_str(cx.tcx(), def_id)))
 182              }
 183  
 184              if cases.len() == 1 {
 185                  // Equivalent to a struct/tuple/newtype.
 186                  // (Typechecking will reject discriminant-sizing attrs.)
 187                  assert_eq!(hint, attr::ReprAny);
 188                  return Univariant(mk_struct(cx,
 189                                              cases.get(0).tys.as_slice(),
 190                                              false),
 191                                    false)
 192              }
 193  
 194              if cases.len() == 2 && hint == attr::ReprAny {
 195                  // Nullable pointer optimization
 196                  let mut discr = 0;
 197                  while discr < 2 {
 198                      if cases.get(1 - discr).is_zerolen(cx) {
 199                          match cases.get(discr).find_ptr() {
 200                              Some(ptrfield) => {
 201                                  return NullablePointer {
 202                                      nndiscr: discr as u64,
 203                                      nonnull: mk_struct(cx,
 204                                                         cases.get(discr)
 205                                                              .tys
 206                                                              .as_slice(),
 207                                                         false),
 208                                      ptrfield: ptrfield,
 209                                      nullfields: cases.get(1 - discr).tys
 210                                                                      .clone()
 211                                  }
 212                              }
 213                              None => { }
 214                          }
 215                      }
 216                      discr += 1;
 217                  }
 218              }
 219  
 220              // The general case.
 221              assert!((cases.len() - 1) as i64 >= 0);
 222              let bounds = IntBounds { ulo: 0, uhi: (cases.len() - 1) as u64,
 223                                       slo: 0, shi: (cases.len() - 1) as i64 };
 224              let ity = range_to_inttype(cx, hint, &bounds);
 225              return General(ity, cases.iter().map(|c| {
 226                  let discr = vec!(ty_of_inttype(ity));
 227                  mk_struct(cx, discr.append(c.tys.as_slice()).as_slice(), false)
 228              }).collect())
 229          }
 230          _ => cx.sess().bug("adt::represent_type called on non-ADT type")
 231      }
 232  }

 234  /// Determine, without doing translation, whether an ADT must be FFI-safe.
 235  /// For use in lint or similar, where being sound but slightly incomplete is acceptable.
 236  pub fn is_ffi_safe(tcx: &ty::ctxt, def_id: ast::DefId) -> bool {
 237      match ty::get(ty::lookup_item_type(tcx, def_id).ty).sty {
 238          ty::ty_enum(def_id, _) => {
 239              let variants = ty::enum_variants(tcx, def_id);
 240              // Univariant => like struct/tuple.
 241              if variants.len() <= 1 {
 242                  return true;
 243              }
 244              let hint = ty::lookup_repr_hint(tcx, def_id);
 245              // Appropriate representation explicitly selected?
 246              if hint.is_ffi_safe() {
 247                  return true;
 248              }
 249              // Option<~T> and similar are used in FFI.  Rather than try to resolve type parameters
 250              // and recognize this case exactly, this overapproximates -- assuming that if a
 251              // non-C-like enum is being used in FFI then the user knows what they're doing.
 252              if variants.iter().any(|vi| !vi.args.is_empty()) {
 253                  return true;
 254              }
 255              false
 256          }
 257          // struct, tuple, etc.
 258          // (is this right in the present of typedefs?)
 259          _ => true
 260      }
 261  }

 263  // this should probably all be in ty
 264  struct Case { discr: Disr, tys: Vec<ty::t> }
 265  impl Case {
 266      fn is_zerolen(&self, cx: &CrateContext) -> bool {
 267          mk_struct(cx, self.tys.as_slice(), false).size == 0
 268      }
 269      fn find_ptr(&self) -> Option<uint> {
 270          self.tys.iter().position(|&ty| mono_data_classify(ty) == MonoNonNull)
 271      }
 272  }

 274  fn get_cases(tcx: &ty::ctxt, def_id: ast::DefId, substs: &ty::substs) -> Vec<Case> {
 275      ty::enum_variants(tcx, def_id).iter().map(|vi| {
 276          let arg_tys = vi.args.iter().map(|&raw_ty| {
 277              ty::subst(tcx, substs, raw_ty)
 278          }).collect();
 279          Case { discr: vi.disr_val, tys: arg_tys }
 280      }).collect()
 281  }

 284  fn mk_struct(cx: &CrateContext, tys: &[ty::t], packed: bool) -> Struct {
 285      let lltys = tys.iter().map(|&ty| type_of::sizing_type_of(cx, ty)).collect::<Vec<_>>();
 286      let llty_rec = Type::struct_(cx, lltys.as_slice(), packed);
 287      Struct {
 288          size: machine::llsize_of_alloc(cx, llty_rec) /*bad*/as u64,
 289          align: machine::llalign_of_min(cx, llty_rec) /*bad*/as u64,
 290          packed: packed,
 291          fields: Vec::from_slice(tys),
 292      }
 293  }

 302  fn mk_cenum(cx: &CrateContext, hint: Hint, bounds: &IntBounds) -> Repr {
 303      let it = range_to_inttype(cx, hint, bounds);
 304      match it {
 305          attr::SignedInt(_) => CEnum(it, bounds.slo as Disr, bounds.shi as Disr),
 306          attr::UnsignedInt(_) => CEnum(it, bounds.ulo, bounds.uhi)
 307      }
 308  }

 310  fn range_to_inttype(cx: &CrateContext, hint: Hint, bounds: &IntBounds) -> IntType {
 311      debug!("range_to_inttype: {:?} {:?}", hint, bounds);
 312      // Lists of sizes to try.  u64 is always allowed as a fallback.
 313      static choose_shortest: &'static[IntType] = &[
 314          attr::UnsignedInt(ast::TyU8), attr::SignedInt(ast::TyI8),
 315          attr::UnsignedInt(ast::TyU16), attr::SignedInt(ast::TyI16),
 316          attr::UnsignedInt(ast::TyU32), attr::SignedInt(ast::TyI32)];
 317      static at_least_32: &'static[IntType] = &[
 318          attr::UnsignedInt(ast::TyU32), attr::SignedInt(ast::TyI32)];
 319  
 320      let attempts;
 321      match hint {
 322          attr::ReprInt(span, ity) => {
 323              if !bounds_usable(cx, ity, bounds) {
 324                  cx.sess().span_bug(span, "representation hint insufficient for discriminant range")
 325              }
 326              return ity;
 327          }
 328          attr::ReprExtern => {
 329              attempts = match cx.sess().targ_cfg.arch {
 330                  X86 | X86_64 => at_least_32,
 331                  // WARNING: the ARM EABI has two variants; the one corresponding to `at_least_32`
 332                  // appears to be used on Linux and NetBSD, but some systems may use the variant
 333                  // corresponding to `choose_shortest`.  However, we don't run on those yet...?
 334                  Arm => at_least_32,
 335                  Mips => at_least_32,
 336              }
 337          }
 338          attr::ReprAny => {
 339              attempts = choose_shortest;
 340          }
 341      }
 342      for &ity in attempts.iter() {
 343          if bounds_usable(cx, ity, bounds) {
 344              return ity;
 345          }
 346      }
 347      return attr::UnsignedInt(ast::TyU64);
 348  }

 350  pub fn ll_inttype(cx: &CrateContext, ity: IntType) -> Type {
 351      match ity {
 352          attr::SignedInt(t) => Type::int_from_ty(cx, t),
 353          attr::UnsignedInt(t) => Type::uint_from_ty(cx, t)
 354      }
 355  }

 357  fn bounds_usable(cx: &CrateContext, ity: IntType, bounds: &IntBounds) -> bool {
 358      debug!("bounds_usable: {:?} {:?}", ity, bounds);
 359      match ity {
 360          attr::SignedInt(_) => {
 361              let lllo = C_integral(ll_inttype(cx, ity), bounds.slo as u64, true);
 362              let llhi = C_integral(ll_inttype(cx, ity), bounds.shi as u64, true);
 363              bounds.slo == const_to_int(lllo) as i64 && bounds.shi == const_to_int(llhi) as i64
 364          }
 365          attr::UnsignedInt(_) => {
 366              let lllo = C_integral(ll_inttype(cx, ity), bounds.ulo, false);
 367              let llhi = C_integral(ll_inttype(cx, ity), bounds.uhi, false);
 368              bounds.ulo == const_to_uint(lllo) as u64 && bounds.uhi == const_to_uint(llhi) as u64
 369          }
 370      }
 371  }

 409  fn generic_type_of(cx: &CrateContext, r: &Repr, name: Option<&str>, sizing: bool) -> Type {
 410      match *r {
 411          CEnum(ity, _, _) => ll_inttype(cx, ity),
 412          Univariant(ref st, _) | NullablePointer{ nonnull: ref st, .. } => {
 413              match name {
 414                  None => {
 415                      Type::struct_(cx, struct_llfields(cx, st, sizing).as_slice(),
 416                                    st.packed)
 417                  }
 418                  Some(name) => { assert_eq!(sizing, false); Type::named_struct(cx, name) }
 419              }
 420          }
 421          General(ity, ref sts) => {
 422              // We need a representation that has:
 423              // * The alignment of the most-aligned field
 424              // * The size of the largest variant (rounded up to that alignment)
 425              // * No alignment padding anywhere any variant has actual data
 426              //   (currently matters only for enums small enough to be immediate)
 427              // * The discriminant in an obvious place.
 428              //
 429              // So we start with the discriminant, pad it up to the alignment with
 430              // more of its own type, then use alignment-sized ints to get the rest
 431              // of the size.
 432              //
 433              // FIXME #10604: this breaks when vector types are present.
 434              let size = sts.iter().map(|st| st.size).max().unwrap();
 435              let most_aligned = sts.iter().max_by(|st| st.align).unwrap();
 436              let align = most_aligned.align;
 437              let discr_ty = ll_inttype(cx, ity);
 438              let discr_size = machine::llsize_of_alloc(cx, discr_ty) as u64;
 439              let align_units = (size + align - 1) / align - 1;
 440              let pad_ty = match align {
 441                  1 => Type::array(&Type::i8(cx), align_units),
 442                  2 => Type::array(&Type::i16(cx), align_units),
 443                  4 => Type::array(&Type::i32(cx), align_units),
 444                  8 if machine::llalign_of_min(cx, Type::i64(cx)) == 8 =>
 445                                   Type::array(&Type::i64(cx), align_units),
 446                  a if a.count_ones() == 1 => Type::array(&Type::vector(&Type::i32(cx), a / 4),
 447                                                                align_units),
 448                  _ => fail!("unsupported enum alignment: {:?}", align)
 449              };
 450              assert_eq!(machine::llalign_of_min(cx, pad_ty) as u64, align);
 451              assert_eq!(align % discr_size, 0);
 452              let fields = vec!(discr_ty,
 453                             Type::array(&discr_ty, align / discr_size - 1),
 454                             pad_ty);
 455              match name {
 456                  None => Type::struct_(cx, fields.as_slice(), false),
 457                  Some(name) => {
 458                      let mut llty = Type::named_struct(cx, name);
 459                      llty.set_struct_body(fields.as_slice(), false);
 460                      llty
 461                  }
 462              }
 463          }
 464      }
 465  }

 481  pub fn trans_switch(bcx: &Block, r: &Repr, scrutinee: ValueRef)

 498  /// Obtain the actual discriminant of a value.
 499  pub fn trans_get_discr(bcx: &Block, r: &Repr, scrutinee: ValueRef, cast_to: Option<Type>)
 500      -> ValueRef {
 501      let signed;
 502      let val;
 503      match *r {
 504          CEnum(ity, min, max) => {
 505              val = load_discr(bcx, ity, scrutinee, min, max);
 506              signed = ity.is_signed();
 507          }
 508          General(ity, ref cases) => {
 509              let ptr = GEPi(bcx, scrutinee, [0, 0]);
 510              val = load_discr(bcx, ity, ptr, 0, (cases.len() - 1) as Disr);
 511              signed = ity.is_signed();
 512          }
 513          Univariant(..) => {
 514              val = C_u8(bcx.ccx(), 0);
 515              signed = false;
 516          }
 517          NullablePointer{ nonnull: ref nonnull, nndiscr, ptrfield, .. } => {
 518              val = nullable_bitdiscr(bcx, nonnull, nndiscr, ptrfield, scrutinee);
 519              signed = false;
 520          }
 521      }
 522      match cast_to {
 523          None => val,
 524          Some(llty) => if signed { SExt(bcx, val, llty) } else { ZExt(bcx, val, llty) }
 525      }
 526  }

 528  fn nullable_bitdiscr(bcx: &Block, nonnull: &Struct, nndiscr: Disr, ptrfield: uint,
 529                       scrutinee: ValueRef) -> ValueRef {
 530      let cmp = if nndiscr == 0 { IntEQ } else { IntNE };
 531      let llptr = Load(bcx, GEPi(bcx, scrutinee, [0, ptrfield]));
 532      let llptrty = type_of::type_of(bcx.ccx(), *nonnull.fields.get(ptrfield));
 533      ICmp(bcx, cmp, llptr, C_null(llptrty))
 534  }

 536  /// Helper for cases where the discriminant is simply loaded.
 537  fn load_discr(bcx: &Block, ity: IntType, ptr: ValueRef, min: Disr, max: Disr)
 538      -> ValueRef {
 539      let llty = ll_inttype(bcx.ccx(), ity);
 540      assert_eq!(val_ty(ptr), llty.ptr_to());
 541      let bits = machine::llbitsize_of_real(bcx.ccx(), llty);
 542      assert!(bits <= 64);
 543      let mask = (-1u64 >> (64 - bits)) as Disr;
 544      if (max + 1) & mask == min & mask {
 545          // i.e., if the range is everything.  The lo==hi case would be
 546          // rejected by the LLVM verifier (it would mean either an
 547          // empty set, which is impossible, or the entire range of the
 548          // type, which is pointless).
 549          Load(bcx, ptr)
 550      } else {
 551          // llvm::ConstantRange can deal with ranges that wrap around,
 552          // so an overflow on (max + 1) is fine.
 553          LoadRangeAssert(bcx, ptr, min as c_ulonglong,
 554                          (max + 1) as c_ulonglong,
 555                          /* signed: */ True)
 556      }
 557  }

 561   * discriminant-like value returned by `trans_switch`.
 562   *
 563   * This should ideally be less tightly tied to `_match`.
 564   */
 565  pub fn trans_case<'a>(bcx: &'a Block<'a>, r: &Repr, discr: Disr)
 566                    -> _match::opt_result<'a> {
 567      match *r {
 568          CEnum(ity, _, _) => {
 569              _match::single_result(rslt(bcx, C_integral(ll_inttype(bcx.ccx(), ity),
 570                                                         discr as u64, true)))
 571          }
 572          General(ity, _) => {
 573              _match::single_result(rslt(bcx, C_integral(ll_inttype(bcx.ccx(), ity),
 574                                                         discr as u64, true)))
 575          }
 576          Univariant(..) => {
 577              bcx.ccx().sess().bug("no cases for univariants or structs")
 578          }
 579          NullablePointer{ .. } => {
 580              assert!(discr == 0 || discr == 1);
 581              _match::single_result(rslt(bcx, C_i1(bcx.ccx(), discr != 0)))
 582          }
 583      }
 584  }

 587   * Begin initializing a new value of the given case of the given
 588   * representation.  The fields, if any, should then be initialized via
 589   * `trans_field_ptr`.
 590   */
 591  pub fn trans_start_init(bcx: &Block, r: &Repr, val: ValueRef, discr: Disr) {
 592      match *r {
 593          CEnum(ity, min, max) => {
 594              assert_discr_in_range(ity, min, max, discr);
 595              Store(bcx, C_integral(ll_inttype(bcx.ccx(), ity), discr as u64, true),
 596                    val)
 597          }
 598          General(ity, _) => {
 599              Store(bcx, C_integral(ll_inttype(bcx.ccx(), ity), discr as u64, true),
 600                    GEPi(bcx, val, [0, 0]))
 601          }
 602          Univariant(ref st, true) => {
 603              assert_eq!(discr, 0);
 604              Store(bcx, C_bool(bcx.ccx(), true),
 605                    GEPi(bcx, val, [0, st.fields.len() - 1]))
 606          }
 607          Univariant(..) => {
 608              assert_eq!(discr, 0);
 609          }
 610          NullablePointer{ nonnull: ref nonnull, nndiscr, ptrfield, .. } => {
 611              if discr != nndiscr {
 612                  let llptrptr = GEPi(bcx, val, [0, ptrfield]);
 613                  let llptrty = type_of::type_of(bcx.ccx(),
 614                                                 *nonnull.fields.get(ptrfield));
 615                  Store(bcx, C_null(llptrty), llptrptr)
 616              }
 617          }
 618      }
 619  }

 647  /// Access a field, at a point when the value's case is known.
 648  pub fn trans_field_ptr(bcx: &Block, r: &Repr, val: ValueRef, discr: Disr,
 649                         ix: uint) -> ValueRef {
 650      // Note: if this ever needs to generate conditionals (e.g., if we
 651      // decide to do some kind of cdr-coding-like non-unique repr
 652      // someday), it will need to return a possibly-new bcx as well.
 653      match *r {
 654          CEnum(..) => {
 655              bcx.ccx().sess().bug("element access in C-like enum")
 656          }
 657          Univariant(ref st, _dtor) => {
 658              assert_eq!(discr, 0);
 659              struct_field_ptr(bcx, st, val, ix, false)
 660          }
 661          General(_, ref cases) => {
 662              struct_field_ptr(bcx, cases.get(discr as uint), val, ix + 1, true)
 663          }
 664          NullablePointer{ nonnull: ref nonnull, nullfields: ref nullfields,
 665                           nndiscr, .. } => {
 666              if discr == nndiscr {
 667                  struct_field_ptr(bcx, nonnull, val, ix, false)
 668              } else {
 669                  // The unit-like case might have a nonzero number of unit-like fields.
 670                  // (e.g., Result or Either with () as one side.)
 671                  let ty = type_of::type_of(bcx.ccx(), *nullfields.get(ix));
 672                  assert_eq!(machine::llsize_of_alloc(bcx.ccx(), ty), 0);
 673                  // The contents of memory at this pointer can't matter, but use
 674                  // the value that's "reasonable" in case of pointer comparison.
 675                  PointerCast(bcx, val, ty.ptr_to())
 676              }
 677          }
 678      }
 679  }

 688          PointerCast(bcx, val, real_ty.ptr_to())
 689      } else {
 690          val
 691      };
 692  
 693      GEPi(bcx, val, [0, ix])
 694  }

 704  /**
 705   * Construct a constant value, suitable for initializing a
 706   * GlobalVariable, given a case and constant values for its fields.
 707   * Note that this may have a different LLVM type (and different
 708   * alignment!) from the representation's `type_of`, so it needs a
 709   * pointer cast before use.
 710   *
 711   * The LLVM type system does not directly support unions, and only
 712   * pointers can be bitcast, so a constant (and, by extension, the
 713   * GlobalVariable initialized by it) will have a type that can vary
 714   * depending on which case of an enum it is.
 715   *
 716   * To understand the alignment situation, consider `enum E { V64(u64),
 717   * V32(u32, u32) }` on win32.  The type has 8-byte alignment to
 718   * accommodate the u64, but `V32(x, y)` would have LLVM type `{i32,
 719   * i32, i32}`, which is 4-byte aligned.
 720   *
 721   * Currently the returned value has the same size as the type, but
 722   * this could be changed in the future to avoid allocating unnecessary
 723   * space after values of shorter-than-maximum cases.
 724   */
 725  pub fn trans_const(ccx: &CrateContext, r: &Repr, discr: Disr,
 726                     vals: &[ValueRef]) -> ValueRef {
 727      match *r {
 728          CEnum(ity, min, max) => {
 729              assert_eq!(vals.len(), 0);
 730              assert_discr_in_range(ity, min, max, discr);
 731              C_integral(ll_inttype(ccx, ity), discr as u64, true)
 732          }
 733          General(ity, ref cases) => {
 734              let case = cases.get(discr as uint);
 735              let max_sz = cases.iter().map(|x| x.size).max().unwrap();
 736              let lldiscr = C_integral(ll_inttype(ccx, ity), discr as u64, true);
 737              let contents = build_const_struct(ccx,
 738                                                case,
 739                                                (vec!(lldiscr)).append(vals).as_slice());
 740              C_struct(ccx, contents.append([padding(ccx, max_sz - case.size)]).as_slice(),
 741                       false)
 742          }
 743          Univariant(ref st, _dro) => {
 744              assert!(discr == 0);
 745              let contents = build_const_struct(ccx, st, vals);
 746              C_struct(ccx, contents.as_slice(), st.packed)
 747          }
 748          NullablePointer{ nonnull: ref nonnull, nndiscr, .. } => {
 749              if discr == nndiscr {
 750                  C_struct(ccx, build_const_struct(ccx,
 751                                                   nonnull,
 752                                                   vals).as_slice(),
 753                           false)
 754              } else {
 755                  let vals = nonnull.fields.iter().map(|&ty| {
 756                      // Always use null even if it's not the `ptrfield`th
 757                      // field; see #8506.
 758                      C_null(type_of::sizing_type_of(ccx, ty))
 759                  }).collect::<Vec<ValueRef>>();
 760                  C_struct(ccx, build_const_struct(ccx,
 761                                                   nonnull,
 762                                                   vals.as_slice()).as_slice(),
 763                           false)
 764              }
 765          }
 766      }

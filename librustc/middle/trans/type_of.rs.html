  40 pub fn type_of_fn(cx: &mut CrateContext, inputs: &[ty::t], output: ty::t) -> Type {
  41     let mut atys: ~[Type] = ~[];
  42 
  43     // Arg 0: Output pointer.
  44     // (if the output type is non-immediate)
  45     let output_is_immediate = ty::type_is_immediate(cx.tcx, output);
  46     let lloutputtype = type_of(cx, output);
  47     if !output_is_immediate {
  48         atys.push(lloutputtype.ptr_to());
  49     }
  50 
  51     // Arg 1: Environment
  52     atys.push(Type::opaque_box(cx).ptr_to());
  53 
  54     // ... then explicit args.
  55     atys.push_all(type_of_explicit_args(cx, inputs));
  56 
  57     // Use the output as the actual return value if it's immediate.
  58     if output_is_immediate && !ty::type_is_nil(output) {
  59         Type::func(atys, &lloutputtype)
  60     } else {
  61         Type::func(atys, &Type::void())
  62     }
  63 }

  76 // A "sizing type" is an LLVM type, the size and alignment of which are
  77 // guaranteed to be equivalent to what you would get out of `type_of()`. It's
  78 // useful because:
  79 //
  80 // (1) It may be cheaper to compute the sizing type than the full type if all
  81 //     you're interested in is the size and/or alignment;
  82 //
  83 // (2) It won't make any recursive calls to determine the structure of the
  84 //     type behind pointers. This can help prevent infinite loops for
  85 //     recursive types. For example, enum types rely on this behavior.
  86

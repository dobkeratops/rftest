  45  pub fn type_of_rust_fn(cx: &mut CrateContext,
  46                         inputs: &[ty::t],
  47                         output: ty::t) -> Type {
  48      let mut atys: ~[Type] = ~[];
  49  
  50      // Arg 0: Output pointer.
  51      // (if the output type is non-immediate)
  52      let use_out_pointer = return_uses_outptr(cx.tcx, output);
  53      let lloutputtype = type_of(cx, output);
  54      if use_out_pointer {
  55          atys.push(lloutputtype.ptr_to());
  56      }
  57  
  58      // Arg 1: Environment
  59      atys.push(Type::opaque_box(cx).ptr_to());
  60  
  61      // ... then explicit args.
  62      atys.push_all(type_of_explicit_args(cx, inputs));
  63  
  64      // Use the output as the actual return value if it's immediate.
  65      if !use_out_pointer && !ty::type_is_voidish(output) {
  66          Type::func(atys, &lloutputtype)
  67      } else {
  68          Type::func(atys, &Type::void())
  69      }
  70  }

  91  // A "sizing type" is an LLVM type, the size and alignment of which are
  92  // guaranteed to be equivalent to what you would get out of `type_of()`. It's
  93  // useful because:
  94  //
  95  // (1) It may be cheaper to compute the sizing type than the full type if all
  96  //     you're interested in is the size and/or alignment;
  97  //
  98  // (2) It won't make any recursive calls to determine the structure of the
  99  //     type behind pointers. This can help prevent infinite loops for
 100  //     recursive types. For example, enum types rely on this behavior.

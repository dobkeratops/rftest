  35  // ___Good to know (tm)__________________________________________________
  36  //
  37  // The layout of a closure environment in memory is
  38  // roughly as follows:
  39  //
  40  // struct rust_opaque_box {         // see rust_internal.h
  41  //   unsigned ref_count;            // only used for @fn()
  42  //   type_desc *tydesc;             // describes closure_data struct
  43  //   rust_opaque_box *prev;         // (used internally by memory alloc)
  44  //   rust_opaque_box *next;         // (used internally by memory alloc)
  45  //   struct closure_data {
  46  //       type_desc *bound_tdescs[]; // bound descriptors
  47  //       struct {
  48  //         upvar1_t upvar1;
  49  //         ...
  50  //         upvarN_t upvarN;
  51  //       } bound_data;
  52  //    }
  53  // };
  54  //
  55  // Note that the closure is itself a rust_opaque_box.  This is true
  56  // even for ~fn and ||, because we wish to keep binary compatibility
  57  // between all kinds of closures.  The allocation strategy for this
  58  // closure depends on the closure type.  For a sendfn, the closure
  59  // (and the referenced type descriptors) will be allocated in the
  60  // exchange heap.  For a fn, the closure is allocated in the task heap
  61  // and is reference counted.  For a block, the closure is allocated on
  62  // the stack.
  63  //
  64  // ## Opaque closures and the embedded type descriptor ##
  65  //
  66  // One interesting part of closures is that they encapsulate the data
  67  // that they close over.  So when I have a ptr to a closure, I do not
  68  // know how many type descriptors it contains nor what upvars are
  69  // captured within.  That means I do not know precisely how big it is
  70  // nor where its fields are located.  This is called an "opaque
  71  // closure".
  72  //
  73  // Typically an opaque closure suffices because we only manipulate it
  74  // by ptr.  The routine Type::at_box().ptr_to() returns an appropriate
  75  // type for such an opaque closure; it allows access to the box fields,
  76  // but not the closure_data itself.
  77  //
  78  // But sometimes, such as when cloning or freeing a closure, we need
  79  // to know the full information.  That is where the type descriptor
  80  // that defines the closure comes in handy.  We can use its take and
  81  // drop glue functions to allocate/free data as needed.
  82  //
  83  // ## Subtleties concerning alignment ##
  84  //
  85  // It is important that we be able to locate the closure data *without
  86  // knowing the kind of data that is being bound*.  This can be tricky
  87  // because the alignment requirements of the bound data affects the
  88  // alignment requires of the closure_data struct as a whole.  However,
  89  // right now this is a non-issue in any case, because the size of the
  90  // rust_opaque_box header is always a mutiple of 16-bytes, which is
  91  // the maximum alignment requirement we ever have to worry about.
  92  //
  93  // The only reason alignment matters is that, in order to learn what data
  94  // is bound, we would normally first load the type descriptors: but their
  95  // location is ultimately depend on their content!  There is, however, a
  96  // workaround.  We can load the tydesc from the rust_opaque_box, which
  97  // describes the closure_data struct and has self-contained derived type
  98  // descriptors, and read the alignment from there.   It's just annoying to
  99  // do.  Hopefully should this ever become an issue we'll have monomorphized
 100  // and type descriptors will all be a bad dream.
 101  //
 102  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

 136  // Given a closure ty, emits a corresponding tuple ty
 137  pub fn mk_closure_tys(tcx: &ty::ctxt,
 138                        bound_values: &[EnvValue])
 139                     -> ty::t {
 140      // determine the types of the values in the env.  Note that this
 141      // is the actual types that will be stored in the map, not the
 142      // logical types as the user sees them, so by-ref upvars must be
 143      // converted to ptrs.
 144      let bound_tys = bound_values.iter().map(|bv| {
 145          match bv.action {
 146              EnvCopy | EnvMove => bv.datum.ty,
 147              EnvRef => ty::mk_mut_ptr(tcx, bv.datum.ty)
 148          }
 149      }).collect();
 150      let cdata_ty = ty::mk_tup(tcx, bound_tys);
 151      debug!("cdata_ty={}", ty_to_str(tcx, cdata_ty));
 152      return cdata_ty;
 153  }

 155  fn tuplify_box_ty(tcx: &ty::ctxt, t: ty::t) -> ty::t {
 156      let ptr = ty::mk_imm_ptr(tcx, ty::mk_i8());
 157      ty::mk_tup(tcx, vec!(ty::mk_uint(), ty::mk_nil_ptr(tcx), ptr, ptr, t))
 158  }

 160  fn allocate_cbox<'a>(bcx: &'a Block<'a>,
 161                       sigil: ast::Sigil,
 162                       cdata_ty: ty::t)
 163                       -> Result<'a> {
 164      let _icx = push_ctxt("closure::allocate_cbox");
 165      let tcx = bcx.tcx();
 166  
 167      // Allocate and initialize the box:
 168      match sigil {
 169          ast::ManagedSigil => {
 170              tcx.sess.bug("trying to trans allocation of @fn")
 171          }
 172          ast::OwnedSigil => {
 173              let ty = type_of(bcx.ccx(), cdata_ty);
 174              let size = llsize_of(bcx.ccx(), ty);
 175              // we treat proc as @ here, which isn't ideal
 176              malloc_raw_dyn_managed(bcx, cdata_ty, ClosureExchangeMallocFnLangItem, size)
 177          }
 178          ast::BorrowedSigil => {
 179              let cbox_ty = tuplify_box_ty(tcx, cdata_ty);
 180              let llbox = alloc_ty(bcx, cbox_ty, "__closure");
 181              rslt(bcx, llbox)
 182          }
 183      }
 184  }

 186  pub struct ClosureResult<'a> {
 187      llbox: ValueRef,    // llvalue of ptr to closure
 188      cdata_ty: ty::t,    // type of the closure data
 189      bcx: &'a Block<'a>  // final bcx
 190  }
 191  
 192  // Given a block context and a list of tydescs and values to bind
 193  // construct a closure out of them. If copying is true, it is a
 194  // heap allocated closure that copies the upvars into environment.
 195  // Otherwise, it is stack allocated and copies pointers to the upvars.
 196  pub fn store_environment<'a>(
 197                           bcx: &'a Block<'a>,
 198                           bound_values: Vec<EnvValue> ,
 199                           sigil: ast::Sigil)
 200                           -> ClosureResult<'a> {
 201      let _icx = push_ctxt("closure::store_environment");
 202      let ccx = bcx.ccx();
 203      let tcx = ccx.tcx();
 204  
 205      // compute the type of the closure
 206      let cdata_ty = mk_closure_tys(tcx, bound_values.as_slice());
 207  
 208      // cbox_ty has the form of a tuple: (a, b, c) we want a ptr to a
 209      // tuple.  This could be a ptr in uniq or a box or on stack,
 210      // whatever.
 211      let cbox_ty = tuplify_box_ty(tcx, cdata_ty);
 212      let cboxptr_ty = ty::mk_ptr(tcx, ty::mt {ty:cbox_ty, mutbl:ast::MutImmutable});
 213      let llboxptr_ty = type_of(ccx, cboxptr_ty);
 214  
 215      // If there are no bound values, no point in allocating anything.
 216      if bound_values.is_empty() {
 217          return ClosureResult {llbox: C_null(llboxptr_ty),
 218                                cdata_ty: cdata_ty,
 219                                bcx: bcx};
 220      }
 221  
 222      // allocate closure in the heap
 223      let Result {bcx: bcx, val: llbox} = allocate_cbox(bcx, sigil, cdata_ty);
 224  
 225      let llbox = PointerCast(bcx, llbox, llboxptr_ty);
 226      debug!("tuplify_box_ty = {}", ty_to_str(tcx, cbox_ty));
 227  
 228      // Copy expr values into boxed bindings.
 229      let mut bcx = bcx;
 230      for (i, bv) in bound_values.move_iter().enumerate() {
 231          debug!("Copy {} into closure", bv.to_str(ccx));
 232  
 233          if ccx.sess().asm_comments() {
 234              add_comment(bcx, format!("Copy {} into closure",
 235                                    bv.to_str(ccx)));
 236          }
 237  
 238          let bound_data = GEPi(bcx, llbox, [0u, abi::box_field_body, i]);
 239  
 240          match bv.action {
 241              EnvCopy | EnvMove => {
 242                  bcx = bv.datum.store_to(bcx, bound_data);
 243              }
 244              EnvRef => {
 245                  Store(bcx, bv.datum.to_llref(), bound_data);
 246              }
 247          }
 248      }
 249  
 250      ClosureResult { llbox: llbox, cdata_ty: cdata_ty, bcx: bcx }
 251  }
 252  
 253  // Given a context and a list of upvars, build a closure. This just
 254  // collects the upvars and packages them up for store_environment.
 255  fn build_closure<'a>(bcx0: &'a Block<'a>,
 256                       cap_vars: &[moves::CaptureVar],
 257                       sigil: ast::Sigil)
 258                       -> ClosureResult<'a> {
 259      let _icx = push_ctxt("closure::build_closure");
 260  
 261      // If we need to, package up the iterator body to call
 262      let bcx = bcx0;
 263  
 264      // Package up the captured upvars
 265      let mut env_vals = Vec::new();
 266      for cap_var in cap_vars.iter() {
 267          debug!("Building closure: captured variable {:?}", *cap_var);
 268          let datum = expr::trans_local_var(bcx, cap_var.def);
 269          match cap_var.mode {
 270              moves::CapRef => {
 271                  assert_eq!(sigil, ast::BorrowedSigil);
 272                  env_vals.push(EnvValue {action: EnvRef,
 273                                          datum: datum});
 274              }
 275              moves::CapCopy => {
 276                  env_vals.push(EnvValue {action: EnvCopy,
 277                                          datum: datum});
 278              }
 279              moves::CapMove => {
 280                  env_vals.push(EnvValue {action: EnvMove,
 281                                          datum: datum});
 282              }
 283          }
 284      }
 285  
 286      return store_environment(bcx, env_vals, sigil);
 287  }

 289  // Given an enclosing block context, a new function context, a closure type,
 290  // and a list of upvars, generate code to load and populate the environment
 291  // with the upvars and type descriptors.
 292  fn load_environment<'a>(bcx: &'a Block<'a>, cdata_ty: ty::t,
 293                          cap_vars: &[moves::CaptureVar],
 294                          sigil: ast::Sigil) -> &'a Block<'a> {
 295      let _icx = push_ctxt("closure::load_environment");
 296  
 297      // Don't bother to create the block if there's nothing to load
 298      if cap_vars.len() == 0 {
 299          return bcx;
 300      }
 301  
 302      // Load a pointer to the closure data, skipping over the box header:
 303      let llcdata = at_box_body(bcx, cdata_ty, bcx.fcx.llenv.unwrap());
 304  
 305      // Store the pointer to closure data in an alloca for debug info because that's what the
 306      // llvm.dbg.declare intrinsic expects
 307      let env_pointer_alloca = if bcx.sess().opts.debuginfo == FullDebugInfo {
 308          let alloc = alloc_ty(bcx, ty::mk_mut_ptr(bcx.tcx(), cdata_ty), "__debuginfo_env_ptr");
 309          Store(bcx, llcdata, alloc);
 310          Some(alloc)
 311      } else {
 312          None
 313      };
 314  
 315      // Populate the upvars from the environment
 316      let mut i = 0u;
 317      for cap_var in cap_vars.iter() {
 318          let mut upvarptr = GEPi(bcx, llcdata, [0u, i]);
 319          match sigil {
 320              ast::BorrowedSigil => { upvarptr = Load(bcx, upvarptr); }
 321              ast::ManagedSigil | ast::OwnedSigil => {}
 322          }
 323          let def_id = ast_util::def_id_of_def(cap_var.def);
 324  
 325          bcx.fcx.llupvars.borrow_mut().insert(def_id.node, upvarptr);
 326  
 327          for &env_pointer_alloca in env_pointer_alloca.iter() {
 328              debuginfo::create_captured_var_metadata(
 329                  bcx,
 330                  def_id.node,
 331                  cdata_ty,
 332                  env_pointer_alloca,
 333                  i,
 334                  sigil,
 335                  cap_var.span);
 336          }
 337  
 338          i += 1u;
 339      }
 340  
 341      bcx
 342  }

 344  fn fill_fn_pair(bcx: &Block, pair: ValueRef, llfn: ValueRef, llenvptr: ValueRef) {
 345      Store(bcx, llfn, GEPi(bcx, pair, [0u, abi::fn_field_code]));
 346      let llenvptr = PointerCast(bcx, llenvptr, Type::i8p(bcx.ccx()));
 347      Store(bcx, llenvptr, GEPi(bcx, pair, [0u, abi::fn_field_box]));
 348  }

 350  pub fn trans_expr_fn<'a>(
 351                       bcx: &'a Block<'a>,
 352                       sigil: ast::Sigil,
 353                       decl: &ast::FnDecl,
 354                       body: &ast::Block,
 355                       id: ast::NodeId,
 356                       dest: expr::Dest)
 357                       -> &'a Block<'a> {
 358      /*!
 359       *
 360       * Translates the body of a closure expression.
 361       *
 362       * - `sigil`
 363       * - `decl`
 364       * - `body`
 365       * - `id`: The id of the closure expression.
 366       * - `cap_clause`: information about captured variables, if any.
 367       * - `dest`: where to write the closure value, which must be a
 368           (fn ptr, env) pair
 369       */
 370  
 371      let _icx = push_ctxt("closure::trans_expr_fn");
 372  
 373      let dest_addr = match dest {
 374          expr::SaveIn(p) => p,
 375          expr::Ignore => {
 376              return bcx; // closure construction is non-side-effecting
 377          }
 378      };
 379  
 380      let ccx = bcx.ccx();
 381      let fty = node_id_type(bcx, id);
 382      let f = match ty::get(fty).sty {
 383          ty::ty_closure(ref f) => f,
 384          _ => fail!("expected closure")
 385      };
 386  
 387      let tcx = bcx.tcx();
 388      let s = tcx.map.with_path(id, |path| {
 389          mangle_internal_name_by_path_and_seq(path, "closure")
 390      });
 391      let llfn = decl_internal_rust_fn(ccx,
 392                                       true,
 393                                       f.sig.inputs.as_slice(),
 394                                       f.sig.output,
 395                                       s);
 396  
 397      // set an inline hint for all closures
 398      set_inline_hint(llfn);
 399  
 400      let cap_vars = ccx.maps.capture_map.borrow().get_copy(&id);
 401      let ClosureResult {llbox, cdata_ty, bcx} =
 402          build_closure(bcx, cap_vars.as_slice(), sigil);

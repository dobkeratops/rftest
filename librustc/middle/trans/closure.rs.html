  33  // ___Good to know (tm)__________________________________________________
  34  //
  35  // The layout of a closure environment in memory is
  36  // roughly as follows:
  37  //
  38  // struct rust_opaque_box {         // see rust_internal.h
  39  //   unsigned ref_count;            // only used for @fn()
  40  //   type_desc *tydesc;             // describes closure_data struct
  41  //   rust_opaque_box *prev;         // (used internally by memory alloc)
  42  //   rust_opaque_box *next;         // (used internally by memory alloc)
  43  //   struct closure_data {
  44  //       type_desc *bound_tdescs[]; // bound descriptors
  45  //       struct {
  46  //         upvar1_t upvar1;
  47  //         ...
  48  //         upvarN_t upvarN;
  49  //       } bound_data;
  50  //    }
  51  // };
  52  //
  53  // Note that the closure is itself a rust_opaque_box.  This is true
  54  // even for ~fn and &fn, because we wish to keep binary compatibility
  55  // between all kinds of closures.  The allocation strategy for this
  56  // closure depends on the closure type.  For a sendfn, the closure
  57  // (and the referenced type descriptors) will be allocated in the
  58  // exchange heap.  For a fn, the closure is allocated in the task heap
  59  // and is reference counted.  For a block, the closure is allocated on
  60  // the stack.
  61  //
  62  // ## Opaque closures and the embedded type descriptor ##
  63  //
  64  // One interesting part of closures is that they encapsulate the data
  65  // that they close over.  So when I have a ptr to a closure, I do not
  66  // know how many type descriptors it contains nor what upvars are
  67  // captured within.  That means I do not know precisely how big it is
  68  // nor where its fields are located.  This is called an "opaque
  69  // closure".
  70  //
  71  // Typically an opaque closure suffices because we only manipulate it
  72  // by ptr.  The routine Type::opaque_box().ptr_to() returns an
  73  // appropriate type for such an opaque closure; it allows access to
  74  // the box fields, but not the closure_data itself.
  75  //
  76  // But sometimes, such as when cloning or freeing a closure, we need
  77  // to know the full information.  That is where the type descriptor
  78  // that defines the closure comes in handy.  We can use its take and
  79  // drop glue functions to allocate/free data as needed.
  80  //
  81  // ## Subtleties concerning alignment ##
  82  //
  83  // It is important that we be able to locate the closure data *without
  84  // knowing the kind of data that is being bound*.  This can be tricky
  85  // because the alignment requirements of the bound data affects the
  86  // alignment requires of the closure_data struct as a whole.  However,
  87  // right now this is a non-issue in any case, because the size of the
  88  // rust_opaque_box header is always a mutiple of 16-bytes, which is
  89  // the maximum alignment requirement we ever have to worry about.
  90  //
  91  // The only reason alignment matters is that, in order to learn what data
  92  // is bound, we would normally first load the type descriptors: but their
  93  // location is ultimately depend on their content!  There is, however, a
  94  // workaround.  We can load the tydesc from the rust_opaque_box, which
  95  // describes the closure_data struct and has self-contained derived type
  96  // descriptors, and read the alignment from there.   It's just annoying to
  97  // do.  Hopefully should this ever become an issue we'll have monomorphized
  98  // and type descriptors will all be a bad dream.
  99  //

 139  // Given a closure ty, emits a corresponding tuple ty
 140  pub fn mk_closure_tys(tcx: ty::ctxt,
 141                        bound_values: &[EnvValue])
 142                     -> ty::t {
 143      // determine the types of the values in the env.  Note that this
 144      // is the actual types that will be stored in the map, not the
 145      // logical types as the user sees them, so by-ref upvars must be
 146      // converted to ptrs.
 147      let bound_tys = bound_values.map(|bv| {
 148          match bv.action {
 149              EnvCopy | EnvMove => bv.datum.ty,
 150              EnvRef => ty::mk_mut_ptr(tcx, bv.datum.ty)
 151          }
 152      });
 153      let cdata_ty = ty::mk_tup(tcx, bound_tys);
 154      debug!("cdata_ty=%s", ty_to_str(tcx, cdata_ty));
 155      return cdata_ty;
 156  }

 166  pub fn allocate_cbox(bcx: @mut Block, sigil: ast::Sigil, cdata_ty: ty::t)
 167                    -> Result {
 168      let _icx = push_ctxt("closure::allocate_cbox");
 169      let ccx = bcx.ccx();
 170      let tcx = ccx.tcx;
 171  
 172      // Allocate and initialize the box:
 173      match sigil {
 174          ast::ManagedSigil => {
 175              tcx.sess.bug("trying to trans allocation of @fn")
 176          }
 177          ast::OwnedSigil => {
 178              malloc_raw(bcx, cdata_ty, heap_for_unique_closure(bcx, cdata_ty))
 179          }
 180          ast::BorrowedSigil => {
 181              let cbox_ty = tuplify_box_ty(tcx, cdata_ty);
 182              let llbox = alloc_ty(bcx, cbox_ty, "__closure");
 183              rslt(bcx, llbox)
 184          }
 185      }
 186  }

 188  pub struct ClosureResult {
 189      llbox: ValueRef, // llvalue of ptr to closure
 190      cdata_ty: ty::t, // type of the closure data
 191      bcx: @mut Block       // final bcx
 192  }
 193  
 194  // Given a block context and a list of tydescs and values to bind
 195  // construct a closure out of them. If copying is true, it is a
 196  // heap allocated closure that copies the upvars into environment.
 197  // Otherwise, it is stack allocated and copies pointers to the upvars.
 198  pub fn store_environment(bcx: @mut Block,
 199                           bound_values: ~[EnvValue],
 200                           sigil: ast::Sigil)
 201                           -> ClosureResult {
 202      let _icx = push_ctxt("closure::store_environment");
 203      let ccx = bcx.ccx();
 204      let tcx = ccx.tcx;
 205  
 206      // compute the type of the closure
 207      let cdata_ty = mk_closure_tys(tcx, bound_values);
 208  
 209      // cbox_ty has the form of a tuple: (a, b, c) we want a ptr to a
 210      // tuple.  This could be a ptr in uniq or a box or on stack,
 211      // whatever.
 212      let cbox_ty = tuplify_box_ty(tcx, cdata_ty);
 213      let cboxptr_ty = ty::mk_ptr(tcx, ty::mt {ty:cbox_ty, mutbl:ast::MutImmutable});
 214      let llboxptr_ty = type_of(ccx, cboxptr_ty);
 215  
 216      // If there are no bound values, no point in allocating anything.
 217      if bound_values.is_empty() {
 218          return ClosureResult {llbox: C_null(llboxptr_ty),
 219                                cdata_ty: cdata_ty,
 220                                bcx: bcx};
 221      }
 222  
 223      // allocate closure in the heap
 224      let Result {bcx: bcx, val: llbox} = allocate_cbox(bcx, sigil, cdata_ty);
 225  
 226      let llbox = PointerCast(bcx, llbox, llboxptr_ty);
 227      debug!("tuplify_box_ty = %s", ty_to_str(tcx, cbox_ty));
 228  
 229      // Copy expr values into boxed bindings.
 230      let mut bcx = bcx;
 231      for (i, bv) in bound_values.iter().enumerate() {
 232          debug!("Copy %s into closure", bv.to_str(ccx));
 233  
 234          if ccx.sess.asm_comments() {
 235              add_comment(bcx, fmt!("Copy %s into closure",
 236                                    bv.to_str(ccx)));
 237          }
 238  
 239          let bound_data = GEPi(bcx, llbox, [0u, abi::box_field_body, i]);
 240  
 241          match bv.action {
 242              EnvCopy => {
 243                  bcx = bv.datum.copy_to(bcx, INIT, bound_data);
 244              }
 245              EnvMove => {
 246                  bcx = bv.datum.move_to(bcx, INIT, bound_data);
 247              }
 248              EnvRef => {
 249                  Store(bcx, bv.datum.to_ref_llval(bcx), bound_data);
 250              }
 251          }
 252  
 253      }
 254  
 255      ClosureResult { llbox: llbox, cdata_ty: cdata_ty, bcx: bcx }
 256  }

 258  // Given a context and a list of upvars, build a closure. This just
 259  // collects the upvars and packages them up for store_environment.
 260  pub fn build_closure(bcx0: @mut Block,
 261                       cap_vars: &[moves::CaptureVar],
 262                       sigil: ast::Sigil) -> ClosureResult {
 263      let _icx = push_ctxt("closure::build_closure");
 264  
 265      // If we need to, package up the iterator body to call
 266      let bcx = bcx0;
 267  
 268      // Package up the captured upvars
 269      let mut env_vals = ~[];
 270      for cap_var in cap_vars.iter() {
 271          debug!("Building closure: captured variable %?", *cap_var);
 272          let datum = expr::trans_local_var(bcx, cap_var.def);
 273          match cap_var.mode {
 274              moves::CapRef => {
 275                  assert_eq!(sigil, ast::BorrowedSigil);
 276                  env_vals.push(EnvValue {action: EnvRef,
 277                                          datum: datum});
 278              }
 279              moves::CapCopy => {
 280                  env_vals.push(EnvValue {action: EnvCopy,
 281                                          datum: datum});
 282              }
 283              moves::CapMove => {
 284                  env_vals.push(EnvValue {action: EnvMove,
 285                                          datum: datum});
 286              }
 287          }
 288      }
 289  
 290      return store_environment(bcx, env_vals, sigil);
 291  }

 293  // Given an enclosing block context, a new function context, a closure type,
 294  // and a list of upvars, generate code to load and populate the environment
 295  // with the upvars and type descriptors.
 296  pub fn load_environment(fcx: @mut FunctionContext,
 297                          cdata_ty: ty::t,
 298                          cap_vars: &[moves::CaptureVar],
 299                          sigil: ast::Sigil) {
 300      let _icx = push_ctxt("closure::load_environment");
 301  
 302      // Don't bother to create the block if there's nothing to load
 303      if cap_vars.len() == 0 {
 304          return;
 305      }
 306  
 307      let bcx = fcx.entry_bcx.unwrap();
 308  
 309      // Load a pointer to the closure data, skipping over the box header:
 310      let llcdata = opaque_box_body(bcx, cdata_ty, fcx.llenv);
 311  
 312      // Store the pointer to closure data in an alloca for debug info because that's what the
 313      // llvm.dbg.declare intrinsic expects
 314      let env_pointer_alloca = if fcx.ccx.sess.opts.extra_debuginfo {
 315          let alloc = alloc_ty(bcx, ty::mk_mut_ptr(bcx.tcx(), cdata_ty), "__debuginfo_env_ptr");
 316          Store(bcx, llcdata, alloc);
 317          Some(alloc)
 318      } else {
 319          None
 320      };
 321  
 322      // Populate the upvars from the environment
 323      let mut i = 0u;
 324      for cap_var in cap_vars.iter() {
 325          let mut upvarptr = GEPi(bcx, llcdata, [0u, i]);
 326          match sigil {
 327              ast::BorrowedSigil => { upvarptr = Load(bcx, upvarptr); }
 328              ast::ManagedSigil | ast::OwnedSigil => {}
 329          }
 330          let def_id = ast_util::def_id_of_def(cap_var.def);
 331          fcx.llupvars.insert(def_id.node, upvarptr);
 332  
 333          for &env_pointer_alloca in env_pointer_alloca.iter() {
 334              debuginfo::create_captured_var_metadata(
 335                  bcx,
 336                  def_id.node,
 337                  cdata_ty,
 338                  env_pointer_alloca,
 339                  i,
 340                  sigil,
 341                  cap_var.span);
 342          }
 343  
 344          i += 1u;
 345      }
 346  }

 348  pub fn trans_expr_fn(bcx: @mut Block,
 349                       sigil: ast::Sigil,
 350                       decl: &ast::fn_decl,
 351                       body: &ast::Block,
 352                       outer_id: ast::NodeId,
 353                       user_id: ast::NodeId,
 354                       dest: expr::Dest) -> @mut Block {
 355      /*!
 356       *
 357       * Translates the body of a closure expression.
 358       *
 359       * - `sigil`
 360       * - `decl`
 361       * - `body`
 362       * - `outer_id`: The id of the closure expression with the correct
 363       *   type.  This is usually the same as `user_id`, but in the
 364       *   case of a `for` loop, the `outer_id` will have the return
 365       *   type of boolean, and the `user_id` will have the return type
 366       *   of `nil`.
 367       * - `user_id`: The id of the closure as the user expressed it.
 368           Generally the same as `outer_id`
 369       * - `cap_clause`: information about captured variables, if any.
 370       * - `dest`: where to write the closure value, which must be a
 371           (fn ptr, env) pair
 372       */
 373  
 374      let _icx = push_ctxt("closure::trans_expr_fn");
 375  
 376      let dest_addr = match dest {
 377          expr::SaveIn(p) => p,
 378          expr::Ignore => {
 379              return bcx; // closure construction is non-side-effecting
 380          }
 381      };
 382  
 383      let ccx = bcx.ccx();
 384      let fty = node_id_type(bcx, outer_id);
 385      let f = match ty::get(fty).sty {
 386          ty::ty_closure(ref f) => f,
 387          _ => fail!("expected closure")
 388      };
 389  
 390      let sub_path = vec::append_one(bcx.fcx.path.clone(),
 391                                     path_name(special_idents::anon));
 392      // XXX: Bad copy.
 393      let s = mangle_internal_name_by_path_and_seq(ccx,
 394                                                   sub_path.clone(),
 395                                                   "expr_fn");
 396      let llfn = decl_internal_rust_fn(ccx, f.sig.inputs, f.sig.output, s);
 397  
 398      // set an inline hint for all closures
 399      set_inline_hint(llfn);
 400  
 401      let Result {bcx: bcx, val: closure} = match sigil {
 402          ast::BorrowedSigil | ast::ManagedSigil | ast::OwnedSigil => {
 403              let cap_vars = ccx.maps.capture_map.get_copy(&user_id);
 404              let ClosureResult {llbox, cdata_ty, bcx}
 405                  = build_closure(bcx, cap_vars, sigil);

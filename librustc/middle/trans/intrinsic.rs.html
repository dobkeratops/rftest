  92      fn with_overflow_instrinsic(bcx: &Block, name: &'static str, t: ty::t) {
  93          let first_real_arg = bcx.fcx.arg_pos(0u);
  94          let a = get_param(bcx.fcx.llfn, first_real_arg);
  95          let b = get_param(bcx.fcx.llfn, first_real_arg + 1);
  96          let llfn = bcx.ccx().intrinsics.get_copy(&name);
  97  
  98          // convert `i1` to a `bool`, and write to the out parameter
  99          let val = Call(bcx, llfn, [a, b], []);
 100          let result = ExtractValue(bcx, val, 0);
 101          let overflow = ZExt(bcx, ExtractValue(bcx, val, 1), Type::bool(bcx.ccx()));
 102          let ret = C_undef(type_of::type_of(bcx.ccx(), t));
 103          let ret = InsertValue(bcx, ret, result, 0);
 104          let ret = InsertValue(bcx, ret, overflow, 1);
 105  
 106          if type_is_immediate(bcx.ccx(), t) {
 107              Ret(bcx, ret);
 108          } else {
 109              let retptr = get_param(bcx.fcx.llfn, bcx.fcx.out_arg_pos());
 110              Store(bcx, ret, retptr);
 111              RetVoid(bcx);
 112          }
 113      }

 132      fn copy_intrinsic(bcx: &Block, allow_overlap: bool, tp_ty: ty::t) {
 133          let ccx = bcx.ccx();
 134          let lltp_ty = type_of::type_of(ccx, tp_ty);
 135          let align = C_i32(ccx, machine::llalign_of_min(ccx, lltp_ty) as i32);
 136          let size = machine::llsize_of(ccx, lltp_ty);
 137          let int_size = machine::llbitsize_of_real(ccx, ccx.int_type);
 138          let name = if allow_overlap {
 139              if int_size == 32 {
 140                  "llvm.memmove.p0i8.p0i8.i32"
 141              } else {
 142                  "llvm.memmove.p0i8.p0i8.i64"
 143              }
 144          } else {
 145              if int_size == 32 {
 146                  "llvm.memcpy.p0i8.p0i8.i32"
 147              } else {
 148                  "llvm.memcpy.p0i8.p0i8.i64"
 149              }
 150          };
 151  
 152          let decl = bcx.fcx.llfn;
 153          let first_real_arg = bcx.fcx.arg_pos(0u);
 154          let dst_ptr = PointerCast(bcx, get_param(decl, first_real_arg), Type::i8p(ccx));
 155          let src_ptr = PointerCast(bcx, get_param(decl, first_real_arg + 1), Type::i8p(ccx));
 156          let count = get_param(decl, first_real_arg + 2);
 157          let volatile = C_i1(ccx, false);
 158          let llfn = ccx.intrinsics.get_copy(&name);
 159          Call(bcx, llfn, [dst_ptr, src_ptr, Mul(bcx, size, count), align, volatile], []);
 160          RetVoid(bcx);
 161      }

 185      fn count_zeros_intrinsic(bcx: &Block, name: &'static str) {
 186          let x = get_param(bcx.fcx.llfn, bcx.fcx.arg_pos(0u));
 187          let y = C_i1(bcx.ccx(), false);
 188          let llfn = bcx.ccx().intrinsics.get_copy(&name);
 189          let llcall = Call(bcx, llfn, [x, y], []);
 190          Ret(bcx, llcall);
 191      }
 192  
 193      let output_type = ty::ty_fn_ret(ty::node_id_to_type(ccx.tcx(), item.id));
 194  
 195      let arena = TypedArena::new();
 196      let fcx = new_fn_ctxt(ccx, decl, item.id, false, output_type,
 197                            Some(substs), Some(item.span), &arena);
 198      init_function(&fcx, true, output_type, Some(substs));
 199  
 200      set_always_inline(fcx.llfn);
 201  
 202      let mut bcx = fcx.entry_bcx.borrow().clone().unwrap();
 203      let first_real_arg = fcx.arg_pos(0u);
 204  
 205      let name = token::get_ident(item.ident);
 206  
 207      // This requires that atomic intrinsics follow a specific naming pattern:
 208      // "atomic_<operation>[_<ordering>], and no ordering means SeqCst
 209      if name.get().starts_with("atomic_") {
 210          let split: Vec<&str> = name.get().split('_').collect();
 211          assert!(split.len() >= 2, "Atomic intrinsic not correct format");
 212          let order = if split.len() == 2 {
 213              lib::llvm::SequentiallyConsistent
 214          } else {
 215              match *split.get(2) {
 216                  "relaxed" => lib::llvm::Monotonic,
 217                  "acq"     => lib::llvm::Acquire,
 218                  "rel"     => lib::llvm::Release,
 219                  "acqrel"  => lib::llvm::AcquireRelease,
 220                  _ => ccx.sess().fatal("unknown ordering in atomic intrinsic")
 221              }
 222          };
 223  
 224          match *split.get(1) {
 225              "cxchg" => {
 226                  let old = AtomicCmpXchg(bcx, get_param(decl, first_real_arg),
 227                                          get_param(decl, first_real_arg + 1u),
 228                                          get_param(decl, first_real_arg + 2u),
 229                                          order);
 230                  Ret(bcx, old);
 231              }
 232              "load" => {
 233                  let old = AtomicLoad(bcx, get_param(decl, first_real_arg),
 234                                       order);
 235                  Ret(bcx, old);
 236              }
 237              "store" => {
 238                  AtomicStore(bcx, get_param(decl, first_real_arg + 1u),
 239                              get_param(decl, first_real_arg),
 240                              order);
 241                  RetVoid(bcx);
 242              }
 243              "fence" => {
 244                  AtomicFence(bcx, order);
 245                  RetVoid(bcx);
 246              }
 247              op => {
 248                  // These are all AtomicRMW ops
 249                  let atom_op = match op {
 250                      "xchg"  => lib::llvm::Xchg,
 251                      "xadd"  => lib::llvm::Add,
 252                      "xsub"  => lib::llvm::Sub,
 253                      "and"   => lib::llvm::And,
 254                      "nand"  => lib::llvm::Nand,
 255                      "or"    => lib::llvm::Or,
 256                      "xor"   => lib::llvm::Xor,
 257                      "max"   => lib::llvm::Max,
 258                      "min"   => lib::llvm::Min,
 259                      "umax"  => lib::llvm::UMax,
 260                      "umin"  => lib::llvm::UMin,
 261                      _ => ccx.sess().fatal("unknown atomic operation")
 262                  };
 263  
 264                  let old = AtomicRMW(bcx, atom_op, get_param(decl, first_real_arg),
 265                                      get_param(decl, first_real_arg + 1u),
 266                                      order);
 267                  Ret(bcx, old);
 268              }
 269          }
 270  
 271          fcx.cleanup();
 272          return;
 273      }
 274  
 275      match name.get() {
 276          "abort" => {
 277              let llfn = bcx.ccx().intrinsics.get_copy(&("llvm.trap"));
 278              Call(bcx, llfn, [], []);
 279              Unreachable(bcx);
 280          }
 281          "breakpoint" => {
 282              let llfn = bcx.ccx().intrinsics.get_copy(&("llvm.debugtrap"));
 283              Call(bcx, llfn, [], []);
 284              RetVoid(bcx);
 285          }
 286          "size_of" => {
 287              let tp_ty = *substs.tys.get(0);
 288              let lltp_ty = type_of::type_of(ccx, tp_ty);
 289              Ret(bcx, C_uint(ccx, machine::llsize_of_real(ccx, lltp_ty) as uint));
 290          }
 291          "move_val_init" => {
 292              // Create a datum reflecting the value being moved.
 293              // Use `appropriate_mode` so that the datum is by ref
 294              // if the value is non-immediate. Note that, with
 295              // intrinsics, there are no argument cleanups to
 296              // concern ourselves with, so we can use an rvalue datum.
 297              let tp_ty = *substs.tys.get(0);
 298              let mode = appropriate_rvalue_mode(ccx, tp_ty);
 299              let src = Datum {val: get_param(decl, first_real_arg + 1u),
 300                               ty: tp_ty,
 301                               kind: Rvalue(mode)};
 302              bcx = src.store_to(bcx, get_param(decl, first_real_arg));
 303              RetVoid(bcx);
 304          }
 305          "min_align_of" => {
 306              let tp_ty = *substs.tys.get(0);
 307              let lltp_ty = type_of::type_of(ccx, tp_ty);
 308              Ret(bcx, C_uint(ccx, machine::llalign_of_min(ccx, lltp_ty) as uint));
 309          }
 310          "pref_align_of"=> {
 311              let tp_ty = *substs.tys.get(0);
 312              let lltp_ty = type_of::type_of(ccx, tp_ty);
 313              Ret(bcx, C_uint(ccx, machine::llalign_of_pref(ccx, lltp_ty) as uint));
 314          }
 315          "get_tydesc" => {
 316              let tp_ty = *substs.tys.get(0);
 317              let static_ti = get_tydesc(ccx, tp_ty);
 318              glue::lazily_emit_visit_glue(ccx, static_ti);
 319  
 320              // FIXME (#3730): ideally this shouldn't need a cast,
 321              // but there's a circularity between translating rust types to llvm
 322              // types and having a tydesc type available. So I can't directly access
 323              // the llvm type of intrinsic::TyDesc struct.
 324              let userland_tydesc_ty = type_of::type_of(ccx, output_type);
 325              let td = PointerCast(bcx, static_ti.tydesc, userland_tydesc_ty);
 326              Ret(bcx, td);
 327          }
 328          "type_id" => {
 329              let hash = ty::hash_crate_independent(
 330                  ccx.tcx(),
 331                  *substs.tys.get(0),
 332                  &ccx.link_meta.crate_hash);
 333              // NB: This needs to be kept in lockstep with the TypeId struct in
 334              //     libstd/unstable/intrinsics.rs
 335              let val = C_named_struct(type_of::type_of(ccx, output_type),
 336                                       [C_u64(ccx, hash)]);
 337              match bcx.fcx.llretptr.get() {
 338                  Some(ptr) => {
 339                      Store(bcx, val, ptr);
 340                      RetVoid(bcx);
 341                  },
 342                  None => Ret(bcx, val)
 343              }
 344          }
 345          "init" => {
 346              let tp_ty = *substs.tys.get(0);
 347              let lltp_ty = type_of::type_of(ccx, tp_ty);
 348              match bcx.fcx.llretptr.get() {
 349                  Some(ptr) => { Store(bcx, C_null(lltp_ty), ptr); RetVoid(bcx); }
 350                  None if ty::type_is_nil(tp_ty) => RetVoid(bcx),
 351                  None => Ret(bcx, C_null(lltp_ty)),
 352              }
 353          }
 354          "uninit" => {
 355              // Do nothing, this is effectively a no-op
 356              let retty = *substs.tys.get(0);
 357              if type_is_immediate(ccx, retty) && !return_type_is_void(ccx, retty) {
 358                  unsafe {
 359                      Ret(bcx, lib::llvm::llvm::LLVMGetUndef(type_of(ccx, retty).to_ref()));
 360                  }
 361              } else {
 362                  RetVoid(bcx)
 363              }
 364          }
 365          "forget" => {
 366              RetVoid(bcx);
 367          }
 368          "transmute" => {
 369              let (in_type, out_type) = (*substs.tys.get(0), *substs.tys.get(1));
 370              let llintype = type_of::type_of(ccx, in_type);

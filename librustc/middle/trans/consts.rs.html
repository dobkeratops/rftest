  37  pub fn const_lit(cx: &mut CrateContext, e: &ast::expr, lit: ast::lit)
  38      -> ValueRef {
  39      let _icx = push_ctxt("trans_lit");
  40      match lit.node {
  41        ast::lit_int(i, t) => C_integral(Type::int_from_ty(cx, t), i as u64, true),
  42        ast::lit_uint(u, t) => C_integral(Type::uint_from_ty(cx, t), u, false),
  43        ast::lit_int_unsuffixed(i) => {
  44          let lit_int_ty = ty::node_id_to_type(cx.tcx, e.id);
  45          match ty::get(lit_int_ty).sty {
  46            ty::ty_int(t) => {
  47              C_integral(Type::int_from_ty(cx, t), i as u64, true)
  48            }
  49            ty::ty_uint(t) => {
  50              C_integral(Type::uint_from_ty(cx, t), i as u64, false)
  51            }
  52            _ => cx.sess.span_bug(lit.span,
  53                     fmt!("integer literal has type %s (expected int or uint)",
  54                          ty_to_str(cx.tcx, lit_int_ty)))
  55          }
  56        }
  57        ast::lit_float(fs, t) => C_floating(fs, Type::float_from_ty(cx, t)),
  58        ast::lit_float_unsuffixed(fs) => {
  59          let lit_float_ty = ty::node_id_to_type(cx.tcx, e.id);
  60          match ty::get(lit_float_ty).sty {
  61            ty::ty_float(t) => {
  62              C_floating(fs, Type::float_from_ty(cx, t))
  63            }
  64            _ => {
  65              cx.sess.span_bug(lit.span,
  66                               "floating point literal doesn't have the right type");
  67            }
  68          }
  69        }
  70        ast::lit_bool(b) => C_bool(b),
  71        ast::lit_nil => C_nil(),
  72        ast::lit_str(s) => C_estr_slice(cx, s)
  73      }
  74  }

  81      }

  84  pub fn const_vec(cx: @mut CrateContext, e: &ast::expr, es: &[@ast::expr])
  85      -> (ValueRef, ValueRef, Type) {
  86      unsafe {
  87          let vec_ty = ty::expr_ty(cx.tcx, e);
  88          let unit_ty = ty::sequence_element_type(cx.tcx, vec_ty);
  89          let llunitty = type_of::type_of(cx, unit_ty);
  90          let unit_sz = machine::llsize_of(cx, llunitty);
  91          let sz = llvm::LLVMConstMul(C_uint(cx, es.len()), unit_sz);
  92          let vs = es.map(|e| const_expr(cx, *e));
  93          // If the vector contains enums, an LLVM array won't work.
  94          let v = if vs.iter().any(|vi| val_ty(*vi) != llunitty) {
  95              C_struct(vs)
  96          } else {
  97              C_array(llunitty, vs)
  98          };
  99          return (v, sz, llunitty);
 100      }
 101  }

 103  fn const_addr_of(cx: &mut CrateContext, cv: ValueRef) -> ValueRef {
 104      unsafe {
 105          let gv = do "const".to_c_str().with_ref |name| {
 106              llvm::LLVMAddGlobal(cx.llmod, val_ty(cv).to_ref(), name)
 107          };
 108          llvm::LLVMSetInitializer(gv, cv);
 109          llvm::LLVMSetGlobalConstant(gv, True);
 110          SetLinkage(gv, PrivateLinkage);
 111          gv
 112      }
 113  }

 115  fn const_deref_ptr(cx: &mut CrateContext, v: ValueRef) -> ValueRef {
 116      let v = match cx.const_globals.find(&(v as int)) {
 117          Some(&v) => v,
 118          None => v
 119      };
 120      unsafe {
 121          assert_eq!(llvm::LLVMIsGlobalConstant(v), True);
 122          llvm::LLVMGetInitializer(v)
 123      }
 124  }

 130  }

 132  fn const_deref(cx: &mut CrateContext, v: ValueRef, t: ty::t, explicit: bool)
 133      -> (ValueRef, ty::t) {
 134      match ty::deref(cx.tcx, t, explicit) {
 135          Some(ref mt) => {
 136              assert!(mt.mutbl != ast::m_mutbl);
 137              let dv = match ty::get(t).sty {
 138                  ty::ty_ptr(*) | ty::ty_rptr(*) => {
 139                       const_deref_ptr(cx, v)
 140                  }
 141                  ty::ty_enum(*) | ty::ty_struct(*) => {
 142                      const_deref_newtype(cx, v, t)
 143                  }
 144                  _ => {
 145                      cx.sess.bug(fmt!("Unexpected dereferenceable type %s",
 146                                       ty_to_str(cx.tcx, t)))
 147                  }
 148              };
 149              (dv, mt.ty)
 150          }
 151          None => {
 152              cx.sess.bug(fmt!("Can't dereference const of type %s",
 153                               ty_to_str(cx.tcx, t)))
 154          }
 155      }
 156  }

 158  pub fn get_const_val(cx: @mut CrateContext, mut def_id: ast::def_id) -> ValueRef {
 159      let contains_key = cx.const_values.contains_key(&def_id.node);
 160      if !ast_util::is_local(def_id) || !contains_key {
 161          if !ast_util::is_local(def_id) {
 162              def_id = inline::maybe_instantiate_inline(cx, def_id);
 163          }
 164          match cx.tcx.items.get_copy(&def_id.node) {
 165              ast_map::node_item(@ast::item {
 166                  node: ast::item_static(_, ast::m_imm, _), _
 167              }, _) => {
 168                  trans_const(cx, ast::m_imm, def_id.node);
 169              }
 170              _ => cx.tcx.sess.bug("expected a const to be an item")
 171          }
 172      }
 173      cx.const_values.get_copy(&def_id.node)
 174  }

 176  pub fn const_expr(cx: @mut CrateContext, e: @ast::expr) -> ValueRef {
 177      let mut llconst = const_expr_unadjusted(cx, e);
 178      let ety = ty::expr_ty(cx.tcx, e);
 179      let adjustment = cx.tcx.adjustments.find_copy(&e.id);
 180      match adjustment {
 181          None => { }
 182          Some(@ty::AutoAddEnv(ty::re_static, ast::BorrowedSigil)) => {
 183              llconst = C_struct([llconst, C_null(Type::opaque_box(cx).ptr_to())])
 184          }
 185          Some(@ty::AutoAddEnv(ref r, ref s)) => {
 186              cx.sess.span_bug(e.span, fmt!("unexpected static function: \
 187                                             region %? sigil %?", *r, *s))
 188          }
 189          Some(@ty::AutoDerefRef(ref adj)) => {
 190              let mut ty = ety;
 191              let mut maybe_ptr = None;
 192              do adj.autoderefs.times {
 193                  let (dv, dt) = const_deref(cx, llconst, ty, false);
 194                  maybe_ptr = Some(llconst);
 195                  llconst = dv;
 196                  ty = dt;
 197              }
 198  
 199              match adj.autoref {
 200                  None => { }
 201                  Some(ref autoref) => {
 202                      // Don't copy data to do a deref+ref.
 203                      let llptr = match maybe_ptr {
 204                          Some(ptr) => ptr,
 205                          None => const_addr_of(cx, llconst)
 206                      };
 207                      match *autoref {
 208                          ty::AutoUnsafe(m) |
 209                          ty::AutoPtr(ty::re_static, m) => {
 210                              assert!(m != ast::m_mutbl);
 211                              llconst = llptr;
 212                          }
 213                          ty::AutoBorrowVec(ty::re_static, m) => {
 214                              assert!(m != ast::m_mutbl);
 215                              assert_eq!(abi::slice_elt_base, 0);
 216                              assert_eq!(abi::slice_elt_len, 1);
 217  
 218                              match ty::get(ty).sty {
 219                                  ty::ty_evec(_, ty::vstore_fixed(*)) => {
 220                                      let size = machine::llsize_of(cx, val_ty(llconst));
 221                                      llconst = C_struct([llptr, size]);
 222                                  }
 223                                  _ => {}
 224                              }
 225                          }
 226                          _ => {
 227                              cx.sess.span_bug(e.span,
 228                                               fmt!("unimplemented const \
 229                                                     autoref %?", autoref))
 230                          }
 231                      }
 232                  }
 233              }
 234          }
 235      }
 236  
 237      let ety_adjusted = ty::expr_ty_adjusted(cx.tcx, e);
 238      let llty = type_of::sizing_type_of(cx, ety_adjusted);
 239      let csize = machine::llsize_of_alloc(cx, val_ty(llconst));
 240      let tsize = machine::llsize_of_alloc(cx, llty);
 241      if csize != tsize {
 242          unsafe {
 243              // XXX these values could use some context
 244              llvm::LLVMDumpValue(llconst);
 245              llvm::LLVMDumpValue(C_undef(llty));
 246          }
 247          cx.sess.bug(fmt!("const %s of type %s has size %u instead of %u",
 248                           e.repr(cx.tcx), ty_to_str(cx.tcx, ety),
 249                           csize, tsize));
 250      }
 251      llconst
 252  }

 254  fn const_expr_unadjusted(cx: @mut CrateContext, e: &ast::expr) -> ValueRef {
 255      unsafe {
 256          let _icx = push_ctxt("const_expr");
 257          return match e.node {
 258            ast::expr_lit(lit) => consts::const_lit(cx, e, *lit),
 259            ast::expr_binary(_, b, e1, e2) => {
 260              let te1 = const_expr(cx, e1);
 261              let te2 = const_expr(cx, e2);
 262  
 263              let te2 = base::cast_shift_const_rhs(b, te1, te2);
 264  
 265              /* Neither type is bottom, and we expect them to be unified
 266               * already, so the following is safe. */
 267              let ty = ty::expr_ty(cx.tcx, e1);
 268              let is_float = ty::type_is_fp(ty);
 269              let signed = ty::type_is_signed(ty);
 270              return match b {
 271                ast::add   => {
 272                  if is_float { llvm::LLVMConstFAdd(te1, te2) }
 273                  else        { llvm::LLVMConstAdd(te1, te2) }
 274                }
 275                ast::subtract => {
 276                  if is_float { llvm::LLVMConstFSub(te1, te2) }
 277                  else        { llvm::LLVMConstSub(te1, te2) }
 278                }
 279                ast::mul    => {
 280                  if is_float { llvm::LLVMConstFMul(te1, te2) }
 281                  else        { llvm::LLVMConstMul(te1, te2) }
 282                }
 283                ast::div    => {
 284                  if is_float    { llvm::LLVMConstFDiv(te1, te2) }
 285                  else if signed { llvm::LLVMConstSDiv(te1, te2) }
 286                  else           { llvm::LLVMConstUDiv(te1, te2) }
 287                }
 288                ast::rem    => {
 289                  if is_float    { llvm::LLVMConstFRem(te1, te2) }
 290                  else if signed { llvm::LLVMConstSRem(te1, te2) }
 291                  else           { llvm::LLVMConstURem(te1, te2) }
 292                }
 293                ast::and    => llvm::LLVMConstAnd(te1, te2),
 294                ast::or     => llvm::LLVMConstOr(te1, te2),
 295                ast::bitxor => llvm::LLVMConstXor(te1, te2),
 296                ast::bitand => llvm::LLVMConstAnd(te1, te2),
 297                ast::bitor  => llvm::LLVMConstOr(te1, te2),
 298                ast::shl    => llvm::LLVMConstShl(te1, te2),
 299                ast::shr    => {
 300                  if signed { llvm::LLVMConstAShr(te1, te2) }
 301                  else      { llvm::LLVMConstLShr(te1, te2) }
 302                }
 303                ast::eq     => {
 304                    if is_float { ConstFCmp(RealOEQ, te1, te2) }
 305                    else        { ConstICmp(IntEQ, te1, te2)   }
 306                },
 307                ast::lt     => {
 308                    if is_float { ConstFCmp(RealOLT, te1, te2) }
 309                    else        {
 310                        if signed { ConstICmp(IntSLT, te1, te2) }
 311                        else      { ConstICmp(IntULT, te1, te2) }
 312                    }
 313                },
 314                ast::le     => {
 315                    if is_float { ConstFCmp(RealOLE, te1, te2) }
 316                    else        {
 317                        if signed { ConstICmp(IntSLE, te1, te2) }
 318                        else      { ConstICmp(IntULE, te1, te2) }
 319                    }
 320                },
 321                ast::ne     => {
 322                    if is_float { ConstFCmp(RealONE, te1, te2) }
 323                    else        { ConstICmp(IntNE, te1, te2) }
 324                },
 325                ast::ge     => {
 326                    if is_float { ConstFCmp(RealOGE, te1, te2) }
 327                    else        {
 328                        if signed { ConstICmp(IntSGE, te1, te2) }
 329                        else      { ConstICmp(IntUGE, te1, te2) }
 330                    }
 331                },
 332                ast::gt     => {
 333                    if is_float { ConstFCmp(RealOGT, te1, te2) }
 334                    else        {
 335                        if signed { ConstICmp(IntSGT, te1, te2) }
 336                        else      { ConstICmp(IntUGT, te1, te2) }
 337                    }
 338                },
 339              };
 340            },
 341            ast::expr_unary(_, u, e) => {
 342              let te = const_expr(cx, e);
 343              let ty = ty::expr_ty(cx.tcx, e);
 344              let is_float = ty::type_is_fp(ty);
 345              return match u {
 346                ast::box(_)  |
 347                ast::uniq |
 348                ast::deref  => {
 349                  let (dv, _dt) = const_deref(cx, te, ty, true);
 350                  dv
 351                }
 352                ast::not    => {
 353                  match ty::get(ty).sty {
 354                      ty::ty_bool => {
 355                          // Somewhat questionable, but I believe this is
 356                          // correct.
 357                          let te = llvm::LLVMConstTrunc(te, Type::i1().to_ref());
 358                          let te = llvm::LLVMConstNot(te);
 359                          llvm::LLVMConstZExt(te, Type::bool().to_ref())
 360                      }
 361                      _ => llvm::LLVMConstNot(te),
 362                  }
 363                }
 364                ast::neg    => {
 365                  if is_float { llvm::LLVMConstFNeg(te) }
 366                  else        { llvm::LLVMConstNeg(te) }
 367                }
 368              }
 369            }
 370            ast::expr_field(base, field, _) => {
 371                let bt = ty::expr_ty_adjusted(cx.tcx, base);
 372                let brepr = adt::represent_type(cx, bt);
 373                let bv = const_expr(cx, base);
 374                do expr::with_field_tys(cx.tcx, bt, None) |discr, field_tys| {
 375                    let ix = ty::field_idx_strict(cx.tcx, field, field_tys);
 376                    adt::const_get_field(cx, brepr, bv, discr, ix)
 377                }
 378            }
 379  
 380            ast::expr_index(_, base, index) => {
 381                let bt = ty::expr_ty_adjusted(cx.tcx, base);
 382                let bv = const_expr(cx, base);
 383                let iv = match const_eval::eval_const_expr(cx.tcx, index) {
 384                    const_eval::const_int(i) => i as u64,
 385                    const_eval::const_uint(u) => u,
 386                    _ => cx.sess.span_bug(index.span,
 387                                          "index is not an integer-constant expression")
 388                };
 389                let (arr, len) = match ty::get(bt).sty {
 390                    ty::ty_evec(_, vstore) | ty::ty_estr(vstore) =>
 391                        match vstore {
 392                        ty::vstore_fixed(u) =>
 393                            (bv, C_uint(cx, u)),
 394

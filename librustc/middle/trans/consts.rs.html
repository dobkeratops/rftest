  41  pub fn const_lit(cx: &CrateContext, e: &ast::Expr, lit: ast::Lit)
  42      -> ValueRef {
  43      let _icx = push_ctxt("trans_lit");
  44      match lit.node {
  45          ast::LitChar(i) => C_integral(Type::char(cx), i as u64, false),
  46          ast::LitInt(i, t) => C_integral(Type::int_from_ty(cx, t), i as u64, true),
  47          ast::LitUint(u, t) => C_integral(Type::uint_from_ty(cx, t), u, false),
  48          ast::LitIntUnsuffixed(i) => {
  49              let lit_int_ty = ty::node_id_to_type(cx.tcx(), e.id);
  50              match ty::get(lit_int_ty).sty {
  51                  ty::ty_int(t) => {
  52                      C_integral(Type::int_from_ty(cx, t), i as u64, true)
  53                  }
  54                  ty::ty_uint(t) => {
  55                      C_integral(Type::uint_from_ty(cx, t), i as u64, false)
  56                  }
  57                  _ => cx.sess().span_bug(lit.span,
  58                          format!("integer literal has type {} (expected int or uint)",
  59                                  ty_to_str(cx.tcx(), lit_int_ty)))
  60              }
  61          }
  62          ast::LitFloat(ref fs, t) => {
  63              C_floating(fs.get(), Type::float_from_ty(cx, t))
  64          }
  65          ast::LitFloatUnsuffixed(ref fs) => {
  66              let lit_float_ty = ty::node_id_to_type(cx.tcx(), e.id);
  67              match ty::get(lit_float_ty).sty {
  68                  ty::ty_float(t) => {
  69                      C_floating(fs.get(), Type::float_from_ty(cx, t))
  70                  }
  71                  _ => {
  72                      cx.sess().span_bug(lit.span,
  73                          "floating point literal doesn't have the right type");
  74                  }
  75              }
  76          }
  77          ast::LitBool(b) => C_bool(cx, b),
  78          ast::LitNil => C_nil(cx),
  79          ast::LitStr(ref s, _) => C_str_slice(cx, (*s).clone()),
  80          ast::LitBinary(ref data) => C_binary_slice(cx, data.as_slice()),
  81      }
  82  }

  92  fn const_vec(cx: &CrateContext, e: &ast::Expr,
  93               es: &[@ast::Expr], is_local: bool) -> (ValueRef, Type, bool) {
  94      let vec_ty = ty::expr_ty(cx.tcx(), e);
  95      let unit_ty = ty::sequence_element_type(cx.tcx(), vec_ty);
  96      let llunitty = type_of::type_of(cx, unit_ty);
  97      let (vs, inlineable) = slice::unzip(es.iter().map(|e| const_expr(cx, *e, is_local)));
  98      // If the vector contains enums, an LLVM array won't work.
  99      let v = if vs.iter().any(|vi| val_ty(*vi) != llunitty) {
 100          C_struct(cx, vs, false)
 101      } else {
 102          C_array(llunitty, vs)
 103      };
 104      (v, llunitty, inlineable.iter().fold(true, |a, &b| a && b))
 105  }

 136  fn const_deref(cx: &CrateContext, v: ValueRef, t: ty::t, explicit: bool)
 137      -> (ValueRef, ty::t) {
 138      match ty::deref(t, explicit) {
 139          Some(ref mt) => {
 140              assert!(mt.mutbl != ast::MutMutable);
 141              let dv = match ty::get(t).sty {
 142                  ty::ty_ptr(..) | ty::ty_rptr(..) => {
 143                       const_deref_ptr(cx, v)
 144                  }
 145                  ty::ty_enum(..) | ty::ty_struct(..) => {
 146                      const_deref_newtype(cx, v, t)
 147                  }
 148                  _ => {
 149                      cx.sess().bug(format!("unexpected dereferenceable type {}",
 150                                            ty_to_str(cx.tcx(), t)))
 151                  }
 152              };
 153              (dv, mt.ty)
 154          }
 155          None => {
 156              cx.sess().bug(format!("can't dereference const of type {}",
 157                                    ty_to_str(cx.tcx(), t)))
 158          }
 159      }
 160  }

 162  pub fn get_const_val(cx: &CrateContext,
 163                       mut def_id: ast::DefId) -> (ValueRef, bool) {
 164      let contains_key = cx.const_values.borrow().contains_key(&def_id.node);
 165      if !ast_util::is_local(def_id) || !contains_key {
 166          if !ast_util::is_local(def_id) {
 167              def_id = inline::maybe_instantiate_inline(cx, def_id);
 168          }
 169  
 170          match cx.tcx.map.expect_item(def_id.node).node {
 171              ast::ItemStatic(_, ast::MutImmutable, _) => {
 172                  trans_const(cx, ast::MutImmutable, def_id.node);
 173              }
 174              _ => {}
 175          }
 176      }
 177  
 178      (cx.const_values.borrow().get_copy(&def_id.node),
 179       !cx.non_inlineable_statics.borrow().contains(&def_id.node))
 180  }

 182  pub fn const_expr(cx: &CrateContext, e: &ast::Expr, is_local: bool) -> (ValueRef, bool) {
 183      let (llconst, inlineable) = const_expr_unadjusted(cx, e, is_local);
 184      let mut llconst = llconst;
 185      let mut inlineable = inlineable;
 186      let ety = ty::expr_ty(cx.tcx(), e);
 187      let ety_adjusted = ty::expr_ty_adjusted(cx.tcx(), e,
 188                                              &*cx.maps.method_map.borrow());
 189      let opt_adj = cx.tcx.adjustments.borrow().find_copy(&e.id);
 190      match opt_adj {
 191          None => { }
 192          Some(adj) => {
 193              match *adj {
 194                  ty::AutoAddEnv(ty::RegionTraitStore(ty::ReStatic, _)) => {
 195                      let def = ty::resolve_expr(cx.tcx(), e);
 196                      let wrapper = closure::get_wrapper_for_bare_fn(cx,
 197                                                                     ety_adjusted,
 198                                                                     def,
 199                                                                     llconst,
 200                                                                     is_local);
 201                      llconst = C_struct(cx, [wrapper, C_null(Type::i8p(cx))], false)
 202                  }
 203                  ty::AutoAddEnv(store) => {
 204                      cx.sess()
 205                        .span_bug(e.span,
 206                                  format!("unexpected static function: {:?}",
 207                                          store))
 208                  }
 209                  ty::AutoObject(..) => {
 210                      cx.sess()
 211                        .span_unimpl(e.span,
 212                                     "unimplemented const coercion to trait \
 213                                      object");
 214                  }
 215                  ty::AutoDerefRef(ref adj) => {
 216                      let mut ty = ety;
 217                      let mut maybe_ptr = None;
 218                      for _ in range(0, adj.autoderefs) {
 219                          let (dv, dt) = const_deref(cx, llconst, ty, false);
 220                          maybe_ptr = Some(llconst);
 221                          llconst = dv;
 222                          ty = dt;
 223                      }
 224  
 225                      match adj.autoref {
 226                          None => { }
 227                          Some(ref autoref) => {
 228                              // Don't copy data to do a deref+ref.
 229                              let llptr = match maybe_ptr {
 230                                  Some(ptr) => ptr,
 231                                  None => {
 232                                      inlineable = false;
 233                                      const_addr_of(cx, llconst)
 234                                  }
 235                              };
 236                              match *autoref {
 237                                  ty::AutoUnsafe(m) |
 238                                  ty::AutoPtr(ty::ReStatic, m) => {
 239                                      assert!(m != ast::MutMutable);
 240                                      llconst = llptr;
 241                                  }
 242                                  ty::AutoBorrowVec(ty::ReStatic, m) => {
 243                                      assert!(m != ast::MutMutable);
 244                                      assert_eq!(abi::slice_elt_base, 0);
 245                                      assert_eq!(abi::slice_elt_len, 1);
 246                                      match ty::get(ty).sty {
 247                                          ty::ty_vec(_, ty::VstoreFixed(len)) => {
 248                                              llconst = C_struct(cx, [
 249                                                  llptr,
 250                                                  C_uint(cx, len)
 251                                              ], false);
 252                                          }
 253                                          _ => {}
 254                                      }
 255                                  }
 256                                  _ => {
 257                                      cx.sess().span_bug(e.span,
 258                                                         format!("unimplemented \
 259                                                                  const autoref \
 260                                                                  {:?}",
 261                                                                 autoref))
 262                                  }
 263                              }
 264                          }
 265                      }
 266                  }
 267              }
 268          }
 269      }
 270  
 271      let llty = type_of::sizing_type_of(cx, ety_adjusted);
 272      let csize = machine::llsize_of_alloc(cx, val_ty(llconst));
 273      let tsize = machine::llsize_of_alloc(cx, llty);
 274      if csize != tsize {
 275          unsafe {
 276              // FIXME these values could use some context
 277              llvm::LLVMDumpValue(llconst);
 278              llvm::LLVMDumpValue(C_undef(llty));
 279          }
 280          cx.sess().bug(format!("const {} of type {} has size {} instead of {}",
 281                           e.repr(cx.tcx()), ty_to_str(cx.tcx(), ety),
 282                           csize, tsize));
 283      }
 284      (llconst, inlineable)
 285  }

 287  // the bool returned is whether this expression can be inlined into other crates
 288  // if it's assigned to a static.
 289  fn const_expr_unadjusted(cx: &CrateContext, e: &ast::Expr,
 290                           is_local: bool) -> (ValueRef, bool) {
 291      let map_list = |exprs: &[@ast::Expr]| {
 292          exprs.iter().map(|&e| const_expr(cx, e, is_local))
 293               .fold((Vec::new(), true),
 294                     |(l, all_inlineable), (val, inlineable)| {
 295                  (l.append_one(val), all_inlineable && inlineable)
 296               })
 297      };
 298      unsafe {
 299          let _icx = push_ctxt("const_expr");
 300          return match e.node {
 301            ast::ExprLit(lit) => {
 302                (consts::const_lit(cx, e, (*lit).clone()), true)
 303            }
 304            ast::ExprBinary(b, e1, e2) => {
 305              let (te1, _) = const_expr(cx, e1, is_local);
 306              let (te2, _) = const_expr(cx, e2, is_local);
 307  
 308              let te2 = base::cast_shift_const_rhs(b, te1, te2);
 309  
 310              /* Neither type is bottom, and we expect them to be unified
 311               * already, so the following is safe. */
 312              let ty = ty::expr_ty(cx.tcx(), e1);
 313              let is_float = ty::type_is_fp(ty);
 314              let signed = ty::type_is_signed(ty);
 315              return (match b {
 316                ast::BiAdd   => {
 317                  if is_float { llvm::LLVMConstFAdd(te1, te2) }
 318                  else        { llvm::LLVMConstAdd(te1, te2) }
 319                }
 320                ast::BiSub => {
 321                  if is_float { llvm::LLVMConstFSub(te1, te2) }
 322                  else        { llvm::LLVMConstSub(te1, te2) }
 323                }
 324                ast::BiMul    => {
 325                  if is_float { llvm::LLVMConstFMul(te1, te2) }
 326                  else        { llvm::LLVMConstMul(te1, te2) }
 327                }
 328                ast::BiDiv    => {
 329                  if is_float    { llvm::LLVMConstFDiv(te1, te2) }
 330                  else if signed { llvm::LLVMConstSDiv(te1, te2) }
 331                  else           { llvm::LLVMConstUDiv(te1, te2) }
 332                }
 333                ast::BiRem    => {
 334                  if is_float    { llvm::LLVMConstFRem(te1, te2) }
 335                  else if signed { llvm::LLVMConstSRem(te1, te2) }
 336                  else           { llvm::LLVMConstURem(te1, te2) }
 337                }
 338                ast::BiAnd    => llvm::LLVMConstAnd(te1, te2),
 339                ast::BiOr     => llvm::LLVMConstOr(te1, te2),
 340                ast::BiBitXor => llvm::LLVMConstXor(te1, te2),
 341                ast::BiBitAnd => llvm::LLVMConstAnd(te1, te2),
 342                ast::BiBitOr  => llvm::LLVMConstOr(te1, te2),
 343                ast::BiShl    => llvm::LLVMConstShl(te1, te2),
 344                ast::BiShr    => {
 345                  if signed { llvm::LLVMConstAShr(te1, te2) }
 346                  else      { llvm::LLVMConstLShr(te1, te2) }
 347                }
 348                ast::BiEq     => {
 349                    if is_float { ConstFCmp(RealOEQ, te1, te2) }
 350                    else        { ConstICmp(IntEQ, te1, te2)   }
 351                },
 352                ast::BiLt     => {
 353                    if is_float { ConstFCmp(RealOLT, te1, te2) }
 354                    else        {
 355                        if signed { ConstICmp(IntSLT, te1, te2) }
 356                        else      { ConstICmp(IntULT, te1, te2) }
 357                    }
 358                },
 359                ast::BiLe     => {
 360                    if is_float { ConstFCmp(RealOLE, te1, te2) }
 361                    else        {
 362                        if signed { ConstICmp(IntSLE, te1, te2) }
 363                        else      { ConstICmp(IntULE, te1, te2) }
 364                    }
 365                },
 366                ast::BiNe     => {
 367                    if is_float { ConstFCmp(RealONE, te1, te2) }
 368                    else        { ConstICmp(IntNE, te1, te2) }
 369                },
 370                ast::BiGe     => {
 371                    if is_float { ConstFCmp(RealOGE, te1, te2) }
 372                    else        {
 373                        if signed { ConstICmp(IntSGE, te1, te2) }
 374                        else      { ConstICmp(IntUGE, te1, te2) }
 375                    }
 376                },
 377                ast::BiGt     => {
 378                    if is_float { ConstFCmp(RealOGT, te1, te2) }
 379                    else        {
 380                        if signed { ConstICmp(IntSGT, te1, te2) }
 381                        else      { ConstICmp(IntUGT, te1, te2) }
 382                    }
 383                },
 384              }, true)
 385            },
 386            ast::ExprUnary(u, e) => {
 387              let (te, _) = const_expr(cx, e, is_local);
 388              let ty = ty::expr_ty(cx.tcx(), e);
 389              let is_float = ty::type_is_fp(ty);
 390              return (match u {
 391                ast::UnBox | ast::UnUniq | ast::UnDeref => {
 392                  let (dv, _dt) = const_deref(cx, te, ty, true);
 393                  dv
 394                }
 395                ast::UnNot    => {
 396                  match ty::get(ty).sty {
 397                      ty::ty_bool => {
 398                          // Somewhat questionable, but I believe this is

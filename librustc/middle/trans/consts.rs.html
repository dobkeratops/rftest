  36 pub fn const_lit(cx: &mut CrateContext, e: &ast::expr, lit: ast::lit)
  37     -> ValueRef {
  38     let _icx = push_ctxt("trans_lit");
  39     match lit.node {
  40       ast::lit_int(i, t) => C_integral(Type::int_from_ty(cx, t), i as u64, true),
  41       ast::lit_uint(u, t) => C_integral(Type::uint_from_ty(cx, t), u, false),
  42       ast::lit_int_unsuffixed(i) => {
  43         let lit_int_ty = ty::node_id_to_type(cx.tcx, e.id);
  44         match ty::get(lit_int_ty).sty {
  45           ty::ty_int(t) => {
  46             C_integral(Type::int_from_ty(cx, t), i as u64, true)
  47           }
  48           ty::ty_uint(t) => {
  49             C_integral(Type::uint_from_ty(cx, t), i as u64, false)
  50           }
  51           _ => cx.sess.span_bug(lit.span,
  52                    fmt!("integer literal has type %s (expected int or uint)",
  53                         ty_to_str(cx.tcx, lit_int_ty)))
  54         }
  55       }
  56       ast::lit_float(fs, t) => C_floating(fs, Type::float_from_ty(cx, t)),
  57       ast::lit_float_unsuffixed(fs) => {
  58         let lit_float_ty = ty::node_id_to_type(cx.tcx, e.id);
  59         match ty::get(lit_float_ty).sty {
  60           ty::ty_float(t) => {
  61             C_floating(fs, Type::float_from_ty(cx, t))
  62           }
  63           _ => {
  64             cx.sess.span_bug(lit.span,
  65                              "floating point literal doesn't have the right type");
  66           }
  67         }
  68       }
  69       ast::lit_bool(b) => C_bool(b),
  70       ast::lit_nil => C_nil(),
  71       ast::lit_str(s) => C_estr_slice(cx, s)
  72     }
  73 }

  83 pub fn const_vec(cx: @mut CrateContext, e: &ast::expr, es: &[@ast::expr])
  84     -> (ValueRef, ValueRef, Type) {
  85     unsafe {
  86         let vec_ty = ty::expr_ty(cx.tcx, e);
  87         let unit_ty = ty::sequence_element_type(cx.tcx, vec_ty);
  88         let llunitty = type_of::type_of(cx, unit_ty);
  89         let unit_sz = machine::llsize_of(cx, llunitty);
  90         let sz = llvm::LLVMConstMul(C_uint(cx, es.len()), unit_sz);
  91         let vs = es.map(|e| const_expr(cx, *e));
  92         // If the vector contains enums, an LLVM array won't work.
  93         let v = if vs.iter().any(|vi| val_ty(*vi) != llunitty) {
  94             C_struct(vs)
  95         } else {
  96             C_array(llunitty, vs)
  97         };
  98         return (v, sz, llunitty);
  99     }
 100 }

 102 fn const_addr_of(cx: &mut CrateContext, cv: ValueRef) -> ValueRef {
 103     unsafe {
 104         let gv = do "const".as_c_str |name| {
 105             llvm::LLVMAddGlobal(cx.llmod, val_ty(cv).to_ref(), name)
 106         };
 107         llvm::LLVMSetInitializer(gv, cv);
 108         llvm::LLVMSetGlobalConstant(gv, True);
 109         SetLinkage(gv, PrivateLinkage);
 110         gv
 111     }
 112 }

 114 fn const_deref_ptr(cx: &mut CrateContext, v: ValueRef) -> ValueRef {
 115     let v = match cx.const_globals.find(&(v as int)) {
 116         Some(&v) => v,
 117         None => v
 118     };
 119     unsafe {
 120         assert_eq!(llvm::LLVMIsGlobalConstant(v), True);
 121         llvm::LLVMGetInitializer(v)
 122     }
 123 }

 131 fn const_deref(cx: &mut CrateContext, v: ValueRef, t: ty::t, explicit: bool)
 132     -> (ValueRef, ty::t) {
 133     match ty::deref(cx.tcx, t, explicit) {
 134         Some(ref mt) => {
 135             assert!(mt.mutbl != ast::m_mutbl);
 136             let dv = match ty::get(t).sty {
 137                 ty::ty_ptr(*) | ty::ty_rptr(*) => {
 138                      const_deref_ptr(cx, v)
 139                 }
 140                 ty::ty_enum(*) | ty::ty_struct(*) => {
 141                     const_deref_newtype(cx, v, t)
 142                 }
 143                 _ => {
 144                     cx.sess.bug(fmt!("Unexpected dereferenceable type %s",
 145                                      ty_to_str(cx.tcx, t)))
 146                 }
 147             };
 148             (dv, mt.ty)
 149         }
 150         None => {
 151             cx.sess.bug(fmt!("Can't dereference const of type %s",
 152                              ty_to_str(cx.tcx, t)))
 153         }
 154     }
 155 }

 157 pub fn get_const_val(cx: @mut CrateContext, mut def_id: ast::def_id) -> ValueRef {
 158     let contains_key = cx.const_values.contains_key(&def_id.node);
 159     if !ast_util::is_local(def_id) || !contains_key {
 160         if !ast_util::is_local(def_id) {
 161             def_id = inline::maybe_instantiate_inline(cx, def_id);
 162         }
 163         match cx.tcx.items.get_copy(&def_id.node) {
 164             ast_map::node_item(@ast::item {
 165                 node: ast::item_static(_, ast::m_imm, _), _
 166             }, _) => {
 167                 trans_const(cx, ast::m_imm, def_id.node);
 168             }
 169             _ => cx.tcx.sess.bug("expected a const to be an item")
 170         }
 171     }
 172     cx.const_values.get_copy(&def_id.node)
 173 }

 175 pub fn const_expr(cx: @mut CrateContext, e: @ast::expr) -> ValueRef {
 176     let mut llconst = const_expr_unadjusted(cx, e);
 177     let ety = ty::expr_ty(cx.tcx, e);
 178     let adjustment = cx.tcx.adjustments.find_copy(&e.id);
 179     match adjustment {
 180         None => { }
 181         Some(@ty::AutoAddEnv(ty::re_static, ast::BorrowedSigil)) => {
 182             llconst = C_struct([llconst, C_null(Type::opaque_box(cx).ptr_to())])
 183         }
 184         Some(@ty::AutoAddEnv(ref r, ref s)) => {
 185             cx.sess.span_bug(e.span, fmt!("unexpected static function: \
 186                                            region %? sigil %?", *r, *s))
 187         }
 188         Some(@ty::AutoDerefRef(ref adj)) => {
 189             let mut ty = ety;
 190             let mut maybe_ptr = None;
 191             do adj.autoderefs.times {
 192                 let (dv, dt) = const_deref(cx, llconst, ty, false);
 193                 maybe_ptr = Some(llconst);
 194                 llconst = dv;
 195                 ty = dt;
 196             }
 197 
 198             match adj.autoref {
 199                 None => { }
 200                 Some(ref autoref) => {
 201                     // Don't copy data to do a deref+ref.
 202                     let llptr = match maybe_ptr {
 203                         Some(ptr) => ptr,
 204                         None => const_addr_of(cx, llconst)
 205                     };
 206                     match *autoref {
 207                         ty::AutoUnsafe(m) |
 208                         ty::AutoPtr(ty::re_static, m) => {
 209                             assert!(m != ast::m_mutbl);
 210                             llconst = llptr;
 211                         }
 212                         ty::AutoBorrowVec(ty::re_static, m) => {
 213                             assert!(m != ast::m_mutbl);
 214                             assert_eq!(abi::slice_elt_base, 0);
 215                             assert_eq!(abi::slice_elt_len, 1);
 216 
 217                             match ty::get(ty).sty {
 218                                 ty::ty_evec(_, ty::vstore_fixed(*)) => {
 219                                     let size = machine::llsize_of(cx, val_ty(llconst));
 220                                     llconst = C_struct([llptr, size]);
 221                                 }
 222                                 _ => {}
 223                             }
 224                         }
 225                         _ => {
 226                             cx.sess.span_bug(e.span,
 227                                              fmt!("unimplemented const \
 228                                                    autoref %?", autoref))
 229                         }
 230                     }
 231                 }
 232             }
 233         }
 234     }
 235 
 236     let ety_adjusted = ty::expr_ty_adjusted(cx.tcx, e);
 237     let llty = type_of::sizing_type_of(cx, ety_adjusted);
 238     let csize = machine::llsize_of_alloc(cx, val_ty(llconst));
 239     let tsize = machine::llsize_of_alloc(cx, llty);
 240     if csize != tsize {
 241         unsafe {
 242             // XXX these values could use some context
 243             llvm::LLVMDumpValue(llconst);
 244             llvm::LLVMDumpValue(C_undef(llty));
 245         }
 246         cx.sess.bug(fmt!("const %s of type %s has size %u instead of %u",
 247                          e.repr(cx.tcx), ty_to_str(cx.tcx, ety),
 248                          csize, tsize));
 249     }
 250     llconst
 251 }

 253 fn const_expr_unadjusted(cx: @mut CrateContext, e: &ast::expr) -> ValueRef {
 254     unsafe {
 255         let _icx = push_ctxt("const_expr");
 256         return match e.node {
 257           ast::expr_lit(lit) => consts::const_lit(cx, e, *lit),
 258           ast::expr_binary(_, b, e1, e2) => {
 259             let te1 = const_expr(cx, e1);
 260             let te2 = const_expr(cx, e2);
 261 
 262             let te2 = base::cast_shift_const_rhs(b, te1, te2);
 263 
 264             /* Neither type is bottom, and we expect them to be unified
 265              * already, so the following is safe. */
 266             let ty = ty::expr_ty(cx.tcx, e1);
 267             let is_float = ty::type_is_fp(ty);
 268             let signed = ty::type_is_signed(ty);
 269             return match b {
 270               ast::add   => {
 271                 if is_float { llvm::LLVMConstFAdd(te1, te2) }
 272                 else        { llvm::LLVMConstAdd(te1, te2) }
 273               }
 274               ast::subtract => {
 275                 if is_float { llvm::LLVMConstFSub(te1, te2) }
 276                 else        { llvm::LLVMConstSub(te1, te2) }
 277               }
 278               ast::mul    => {
 279                 if is_float { llvm::LLVMConstFMul(te1, te2) }
 280                 else        { llvm::LLVMConstMul(te1, te2) }
 281               }
 282               ast::div    => {
 283                 if is_float    { llvm::LLVMConstFDiv(te1, te2) }
 284                 else if signed { llvm::LLVMConstSDiv(te1, te2) }
 285                 else           { llvm::LLVMConstUDiv(te1, te2) }
 286               }
 287               ast::rem    => {
 288                 if is_float    { llvm::LLVMConstFRem(te1, te2) }
 289                 else if signed { llvm::LLVMConstSRem(te1, te2) }
 290                 else           { llvm::LLVMConstURem(te1, te2) }
 291               }
 292               ast::and    => llvm::LLVMConstAnd(te1, te2),
 293               ast::or     => llvm::LLVMConstOr(te1, te2),
 294               ast::bitxor => llvm::LLVMConstXor(te1, te2),
 295               ast::bitand => llvm::LLVMConstAnd(te1, te2),
 296               ast::bitor  => llvm::LLVMConstOr(te1, te2),
 297               ast::shl    => llvm::LLVMConstShl(te1, te2),
 298               ast::shr    => {
 299                 if signed { llvm::LLVMConstAShr(te1, te2) }
 300                 else      { llvm::LLVMConstLShr(te1, te2) }
 301               }
 302               ast::eq     => {
 303                   if is_float { ConstFCmp(RealOEQ, te1, te2) }
 304                   else        { ConstICmp(IntEQ, te1, te2)   }
 305               },
 306               ast::lt     => {
 307                   if is_float { ConstFCmp(RealOLT, te1, te2) }
 308                   else        {
 309                       if signed { ConstICmp(IntSLT, te1, te2) }
 310                       else      { ConstICmp(IntULT, te1, te2) }
 311                   }
 312               },
 313               ast::le     => {
 314                   if is_float { ConstFCmp(RealOLE, te1, te2) }
 315                   else        {
 316                       if signed { ConstICmp(IntSLE, te1, te2) }
 317                       else      { ConstICmp(IntULE, te1, te2) }
 318                   }
 319               },
 320               ast::ne     => {
 321                   if is_float { ConstFCmp(RealONE, te1, te2) }
 322                   else        { ConstICmp(IntNE, te1, te2) }
 323               },
 324               ast::ge     => {
 325                   if is_float { ConstFCmp(RealOGE, te1, te2) }
 326                   else        {
 327                       if signed { ConstICmp(IntSGE, te1, te2) }
 328                       else      { ConstICmp(IntUGE, te1, te2) }
 329                   }
 330               },
 331               ast::gt     => {
 332                   if is_float { ConstFCmp(RealOGT, te1, te2) }
 333                   else        {
 334                       if signed { ConstICmp(IntSGT, te1, te2) }
 335                       else      { ConstICmp(IntUGT, te1, te2) }
 336                   }
 337               },
 338             };
 339           },
 340           ast::expr_unary(_, u, e) => {
 341             let te = const_expr(cx, e);
 342             let ty = ty::expr_ty(cx.tcx, e);
 343             let is_float = ty::type_is_fp(ty);
 344             return match u {
 345               ast::box(_)  |
 346               ast::uniq |
 347               ast::deref  => {
 348                 let (dv, _dt) = const_deref(cx, te, ty, true);
 349                 dv
 350               }
 351               ast::not    => {
 352                 match ty::get(ty).sty {
 353                     ty::ty_bool => {
 354                         // Somewhat questionable, but I believe this is
 355                         // correct.
 356                         let te = llvm::LLVMConstTrunc(te, Type::i1().to_ref());
 357                         let te = llvm::LLVMConstNot(te);
 358                         llvm::LLVMConstZExt(te, Type::bool().to_ref())
 359                     }
 360                     _ => llvm::LLVMConstNot(te),
 361                 }
 362               }
 363               ast::neg    => {
 364                 if is_float { llvm::LLVMConstFNeg(te) }
 365                 else        { llvm::LLVMConstNeg(te) }
 366               }
 367             }
 368           }
 369           ast::expr_field(base, field, _) => {
 370               let bt = ty::expr_ty_adjusted(cx.tcx, base);
 371               let brepr = adt::represent_type(cx, bt);
 372               let bv = const_expr(cx, base);
 373               do expr::with_field_tys(cx.tcx, bt, None) |discr, field_tys| {
 374                   let ix = ty::field_idx_strict(cx.tcx, field, field_tys);
 375                   adt::const_get_field(cx, brepr, bv, discr, ix)
 376               }
 377           }
 378 
 379           ast::expr_index(_, base, index) => {
 380               let bt = ty::expr_ty_adjusted(cx.tcx, base);
 381               let bv = const_expr(cx, base);
 382               let iv = match const_eval::eval_const_expr(cx.tcx, index) {
 383                   const_eval::const_int(i) => i as u64,
 384                   const_eval::const_uint(u) => u,
 385                   _ => cx.sess.span_bug(index.span,
 386                                         "index is not an integer-constant expression")
 387               };
 388               let (arr, len) = match ty::get(bt).sty {
 389                   ty::ty_evec(_, vstore) | ty::ty_estr(vstore) =>
 390                       match vstore {
 391                       ty::vstore_fixed(u) =>
 392                           (bv, C_uint(cx, u)),
 393 
 394                       ty::vstore_slice(_) => {
 395                           let unit_ty = ty::sequence_element_type(cx.tcx, bt);
 396                           let llunitty = type_of::type_of(cx, unit_ty);
 397                           let unit_sz = machine::llsize_of(cx, llunitty);
 398 
 399                           let e1 = const_get_elt(cx, bv, [0]);
 400                           (const_deref_ptr(cx, e1),
 401                            llvm::LLVMConstUDiv(const_get_elt(cx, bv, [1]),
 402                                                unit_sz))
 403                       },
 404                       _ => cx.sess.span_bug(base.span,
 405                                             "index-expr base must be fixed-size or slice")
 406                   },
 407                   _ =>  cx.sess.span_bug(base.span,
 408                                          "index-expr base must be a vector or string type")
 409               };

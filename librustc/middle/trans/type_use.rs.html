  11  // Determines the ways in which a generic function body depends
  12  // on its type parameters. Used to aggressively reuse compiled
  13  // function bodies for different types.
  14  
  15  // This unfortunately depends on quite a bit of knowledge about the
  16  // details of the language semantics, and is likely to accidentally go
  17  // out of sync when something is changed. It could be made more
  18  // powerful by distinguishing between functions that only need to know
  19  // the size and alignment of a type, and those that also use its
  20  // drop/take glue. But this would increase the fragility of the code
  21  // to a ridiculous level, and probably not catch all that many extra
  22  // opportunities for reuse.
  23  
  24  // (An other approach to doing what this code does is to instrument
  25  // the translation code to set flags whenever it does something like
  26  // alloca a type or get a tydesc. This would not duplicate quite as
  27  // much information, but have the disadvantage of being very
  28  // invasive.)

  47  pub type type_uses = uint; // Bitmask
  48  pub static use_repr: uint = 1;   /* Dependency on size/alignment/mode and
  49                                       take/drop glue */
  50  pub static use_tydesc: uint = 2; /* Takes the tydesc, or compares */
  51  pub static use_all: uint = use_repr|use_tydesc;
  52  
  53  
  54  pub struct Context {
  55      ccx: @mut CrateContext,
  56      uses: @mut ~[type_uses]
  57  }

  59  pub fn type_uses_for(ccx: @mut CrateContext, fn_id: def_id, n_tps: uint)
  60      -> @~[type_uses] {
  61  
  62      fn store_type_uses(cx: Context, fn_id: def_id) -> @~[type_uses] {
  63          let Context { uses, ccx } = cx;
  64          let uses = @(*uses).clone(); // freeze
  65          ccx.type_use_cache.insert(fn_id, uses);
  66          uses
  67      }
  68  
  69      match ccx.type_use_cache.find(&fn_id) {
  70        Some(uses) => return *uses,
  71        None => ()
  72      }
  73  
  74      let fn_id_loc = if fn_id.crate == LOCAL_CRATE {
  75          fn_id
  76      } else {
  77          inline::maybe_instantiate_inline(ccx, fn_id)
  78      };
  79  
  80      // Conservatively assume full use for recursive loops
  81      ccx.type_use_cache.insert(fn_id, @vec::from_elem(n_tps, use_all));
  82  
  83      let cx = Context {
  84          ccx: ccx,
  85          uses: @mut vec::from_elem(n_tps, 0u)
  86      };
  87  
  88      // If the method is a default method, we mark all of the types as
  89      // used.  This is imprecise, but simple. Getting it right is
  90      // tricky because the substs on the call and the substs on the
  91      // default method differ, because of substs on the trait/impl.
  92      let is_default = ty::provided_source(ccx.tcx, fn_id_loc).is_some();
  93      // We also mark all of the params as used if it is an extern thing
  94      // that we haven't been able to inline yet.
  95      if is_default || fn_id_loc.crate != LOCAL_CRATE {
  96          for n in range(0u, n_tps) { cx.uses[n] |= use_all; }
  97          return store_type_uses(cx, fn_id);
  98      }
  99  
 100      let map_node = match ccx.tcx.items.find(&fn_id_loc.node) {
 101          Some(x) => {
 102              (*x).clone()
 103          }
 104          None => {
 105              ccx.sess.bug(fmt!("type_uses_for: unbound item ID %?",
 106                                fn_id_loc))
 107          }
 108      };
 109  
 110      match map_node {
 111        ast_map::node_item(@ast::item { node: item_fn(_, _, _, _, ref body),
 112                                        _ }, _) |
 113        ast_map::node_method(@ast::method {body: ref body, _}, _, _) => {
 114          handle_body(&cx, body);
 115        }
 116        ast_map::node_trait_method(*) => {
 117          // This will be a static trait method. For now, we just assume
 118          // it fully depends on all of the type information. (Doing
 119          // otherwise would require finding the actual implementation).
 120          for n in range(0u, n_tps) { cx.uses[n] |= use_all;}
 121          // We need to return early, before the arguments are processed,
 122          // because of difficulties in the handling of Self.
 123          return store_type_uses(cx, fn_id);
 124        }
 125        ast_map::node_variant(_, _, _) => {
 126          for n in range(0u, n_tps) { cx.uses[n] |= use_repr;}
 127        }
 128        ast_map::node_foreign_item(i@@foreign_item {
 129              node: foreign_item_fn(*),
 130              _
 131          },
 132          abi,
 133          _,
 134          _) => {
 135          if abi.is_intrinsic() {
 136              let nm = cx.ccx.sess.str_of(i.ident);
 137              let name = nm.as_slice();
 138              let flags = if name.starts_with("atomic_") {
 139                  0
 140              } else {
 141                  match name {
 142                      "size_of"  | "pref_align_of" | "min_align_of" |
 143                      "uninit"   | "init" | "transmute" | "move_val" |
 144                      "move_val_init" => use_repr,
 145  
 146                      "get_tydesc" | "needs_drop" | "contains_managed" => use_tydesc,
 147  
 148                      "visit_tydesc"  | "forget" | "frame_address" |
 149                      "morestack_addr" => 0,
 150  
 151                      "offset" | "offset_inbounds" |
 152                      "memcpy32" | "memcpy64" | "memmove32" | "memmove64" |
 153                      "memset32" | "memset64" => use_repr,
 154  
 155                      "sqrtf32" | "sqrtf64" | "powif32" | "powif64" |
 156                      "sinf32"  | "sinf64"  | "cosf32"  | "cosf64"  |
 157                      "powf32"  | "powf64"  | "expf32"  | "expf64"  |
 158                      "exp2f32" | "exp2f64" | "logf32"  | "logf64"  |
 159                      "log10f32"| "log10f64"| "log2f32" | "log2f64" |
 160                      "fmaf32"  | "fmaf64"  | "fabsf32" | "fabsf64" |
 161                      "floorf32"| "floorf64"| "ceilf32" | "ceilf64" |
 162                      "truncf32"| "truncf64" => 0,
 163  
 164                      "ctpop8" | "ctpop16" | "ctpop32" | "ctpop64" => 0,
 165  
 166                      "ctlz8" | "ctlz16" | "ctlz32" | "ctlz64" => 0,
 167                      "cttz8" | "cttz16" | "cttz32" | "cttz64" => 0,
 168  
 169                      "bswap16" | "bswap32" | "bswap64" => 0,
 170  
 171  
 172                      "i8_add_with_overflow"  | "u8_add_with_overflow" |
 173                      "i16_add_with_overflow" | "u16_add_with_overflow" |
 174                      "i32_add_with_overflow" | "u32_add_with_overflow" |
 175                      "i64_add_with_overflow" | "u64_add_with_overflow" => 0,
 176  
 177                      "i8_sub_with_overflow"  | "u8_sub_with_overflow" |
 178                      "i16_sub_with_overflow" | "u16_sub_with_overflow" |
 179                      "i32_sub_with_overflow" | "u32_sub_with_overflow" |
 180                      "i64_sub_with_overflow" | "u64_sub_with_overflow" => 0,
 181  
 182                      "i8_mul_with_overflow"  | "u8_mul_with_overflow" |
 183                      "i16_mul_with_overflow" | "u16_mul_with_overflow" |
 184                      "i32_mul_with_overflow" | "u32_mul_with_overflow" |
 185                      "i64_mul_with_overflow" | "u64_mul_with_overflow" => 0,
 186  
 187                      // would be cool to make these an enum instead of
 188                      // strings!
 189                      _ => fail!("unknown intrinsic in type_use")
 190                  }
 191              };
 192              for n in range(0u, n_tps) { cx.uses[n] |= flags;}
 193          }
 194        }
 195        ast_map::node_struct_ctor(*) => {
 196          // Similarly to node_variant, this monomorphized function just
 197          // uses the representations of all of its type parameters.
 198          for n in range(0u, n_tps) { cx.uses[n] |= use_repr; }
 199        }
 200        _ => {
 201          ccx.tcx.sess.bug(fmt!("unknown node type in type_use: %s",
 202                                ast_map::node_id_to_str(
 203                                  ccx.tcx.items,
 204                                  fn_id_loc.node,
 205                                  token::get_ident_interner())));
 206        }
 207      }
 208  
 209      // Now handle arguments
 210      match ty::get(ty::lookup_item_type(cx.ccx.tcx, fn_id).ty).sty {
 211          ty::ty_bare_fn(ty::BareFnTy {sig: ref sig, _}) |
 212          ty::ty_closure(ty::ClosureTy {sig: ref sig, _}) => {
 213              for arg in sig.inputs.iter() {
 214                  type_needs(&cx, use_repr, *arg);
 215              }
 216          }
 217          _ => ()
 218      }
 219  
 220      store_type_uses(cx, fn_id)
 221  }

 223  pub fn type_needs(cx: &Context, use_: uint, ty: ty::t) {
 224      // Optimization -- don't descend type if all params already have this use
 225      if cx.uses.iter().any(|&elt| elt & use_ != use_) {
 226          type_needs_inner(cx, use_, ty, @Nil);
 227      }
 228  }
 229  
 230  pub fn type_needs_inner(cx: &Context,
 231                          use_: uint,
 232                          ty: ty::t,
 233                          enums_seen: @List<def_id>) {
 234      do ty::maybe_walk_ty(ty) |ty| {
 235          if ty::type_has_params(ty) {
 236              match ty::get(ty).sty {
 237                  /*
 238                   This previously included ty_box -- that was wrong
 239                   because if we cast an @T to an trait (for example) and return
 240                   it, we depend on the drop glue for T (we have to write the
 241                   right tydesc into the result)
 242                   */
 243                  ty::ty_closure(*) |
 244                  ty::ty_bare_fn(*) |
 245                  ty::ty_ptr(_) |
 246                  ty::ty_rptr(_, _) |
 247                  ty::ty_trait(*) => false,
 248  
 249                ty::ty_enum(did, ref substs) => {
 250                  if list::find(enums_seen, |id| *id == did).is_none() {
 251                      let seen = @Cons(did, enums_seen);
 252                      let r = ty::enum_variants(cx.ccx.tcx, did);
 253                      for v in r.iter() {
 254                          for aty in v.args.iter() {
 255                              let t = ty::subst(cx.ccx.tcx, &(*substs), *aty);
 256                              type_needs_inner(cx, use_, t, seen);
 257                          }
 258                      }
 259                  }
 260                  false
 261                }
 262                ty::ty_param(p) => {
 263                  cx.uses[p.idx] |= use_;
 264                  false
 265                }
 266                _ => true
 267              }
 268          } else { false }
 269      }
 270  }

 272  pub fn node_type_needs(cx: &Context, use_: uint, id: NodeId) {
 273      type_needs(cx, use_, ty::node_id_to_type(cx.ccx.tcx, id));
 274  }

 276  pub fn mark_for_method_call(cx: &Context, e_id: NodeId, callee_id: NodeId) {
 277      let mut opt_static_did = None;
 278      {
 279          let r = cx.ccx.maps.method_map.find(&e_id);
 280          for mth in r.iter() {
 281              match mth.origin {
 282                typeck::method_static(did) => {
 283                    opt_static_did = Some(did);
 284                }
 285                typeck::method_param(typeck::method_param {
 286                    param_num: typeck::param_numbered(param),
 287                    _
 288                }) => {
 289                  cx.uses[param] |= use_tydesc;
 290                }
 291                _ => (),
 292              }
 293          }
 294      }
 295  
 296      // Note: we do not execute this code from within the each() call
 297      // above because the recursive call to `type_needs` can trigger
 298      // inlining and hence can cause `method_map` and
 299      // `node_type_substs` to be modified.
 300      for &did in opt_static_did.iter() {
 301          {
 302              let r = cx.ccx.tcx.node_type_substs.find_copy(&callee_id);
 303              for ts in r.iter() {
 304                  let type_uses = type_uses_for(cx.ccx, did, ts.len());
 305                  for (uses, subst) in type_uses.iter().zip(ts.iter()) {
 306                      type_needs(cx, *uses, *subst)
 307                  }
 308              }
 309          }
 310      }
 311  }

 313  pub fn mark_for_expr(cx: &Context, e: &expr) {
 314      match e.node {
 315        expr_vstore(_, _) | expr_vec(_, _) | expr_struct(*) | expr_tup(_) |
 316        expr_unary(_, box(_), _) | expr_unary(_, uniq, _) |
 317        expr_binary(_, add, _, _) | expr_repeat(*) => {
 318          node_type_needs(cx, use_repr, e.id);
 319        }
 320        expr_cast(base, _) => {
 321          let result_t = ty::node_id_to_type(cx.ccx.tcx, e.id);
 322          match ty::get(result_t).sty {
 323              ty::ty_trait(*) => {
 324                // When we're casting to an trait, we need the
 325                // tydesc for the expr that's being cast.
 326                node_type_needs(cx, use_tydesc, base.id);
 327              }
 328              _ => ()
 329          }
 330        }
 331        expr_binary(_, op, lhs, _) => {
 332          match op {
 333            eq | lt | le | ne | ge | gt => {
 334              node_type_needs(cx, use_tydesc, lhs.id)
 335            }
 336            _ => ()
 337          }
 338        }
 339        expr_path(_) | expr_self => {
 340          let opt_ts = cx.ccx.tcx.node_type_substs.find_copy(&e.id);
 341          for ts in opt_ts.iter() {
 342              let id = ast_util::def_id_of_def(cx.ccx.tcx.def_map.get_copy(&e.id));
 343              let uses_for_ts = type_uses_for(cx.ccx, id, ts.len());
 344              for (uses, subst) in uses_for_ts.iter().zip(ts.iter()) {
 345                  type_needs(cx, *uses, *subst)
 346              }
 347          }
 348        }
 349        expr_fn_block(*) => {
 350            match ty::ty_closure_sigil(ty::expr_ty(cx.ccx.tcx, e)) {
 351                ast::OwnedSigil => {}
 352                ast::BorrowedSigil | ast::ManagedSigil => {
 353                    for fv in freevars::get_freevars(cx.ccx.tcx, e.id).iter() {
 354                        let node_id = ast_util::def_id_of_def(fv.def).node;
 355                        node_type_needs(cx, use_repr, node_id);
 356                    }
 357                }
 358            }
 359        }
 360        expr_assign(val, _) | expr_assign_op(_, _, val, _) |
 361        expr_ret(Some(val)) => {
 362          node_type_needs(cx, use_repr, val.id);
 363        }
 364        expr_index(callee_id, base, _) => {
 365          // FIXME (#2537): could be more careful and not count fields after
 366          // the chosen field.
 367          let base_ty = ty::node_id_to_type(cx.ccx.tcx, base.id);
 368          type_needs(cx, use_repr, ty::type_autoderef(cx.ccx.tcx, base_ty));
 369          mark_for_method_call(cx, e.id, callee_id);
 370        }
 371        expr_field(base, _, _) => {
 372          // Method calls are now a special syntactic form,
 373          // so `a.b` should always be a field.
 374          assert!(!cx.ccx.maps.method_map.contains_key(&e.id));
 375  
 376          let base_ty = ty::node_id_to_type(cx.ccx.tcx, base.id);
 377          type_needs(cx, use_repr, ty::type_autoderef(cx.ccx.tcx, base_ty));
 378        }
 379        expr_log(_, val) => {
 380          node_type_needs(cx, use_tydesc, val.id);
 381        }
 382        expr_call(f, _, _) => {
 383            let r = ty::ty_fn_args(ty::node_id_to_type(cx.ccx.tcx, f.id));
 384            for a in r.iter() {
 385                type_needs(cx, use_repr, *a);
 386            }

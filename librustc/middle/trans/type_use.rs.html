  11  // Determines the ways in which a generic function body depends
  12  // on its type parameters. Used to aggressively reuse compiled
  13  // function bodies for different types.
  14  
  15  // This unfortunately depends on quite a bit of knowledge about the
  16  // details of the language semantics, and is likely to accidentally go
  17  // out of sync when something is changed. It could be made more
  18  // powerful by distinguishing between functions that only need to know
  19  // the size and alignment of a type, and those that also use its
  20  // drop/take glue. But this would increase the fragility of the code
  21  // to a ridiculous level, and probably not catch all that many extra
  22  // opportunities for reuse.
  23  
  24  // (An other approach to doing what this code does is to instrument
  25  // the translation code to set flags whenever it does something like
  26  // alloca a type or get a tydesc. This would not duplicate quite as
  27  // much information, but have the disadvantage of being very
  28  // invasive.)

  48  pub type type_uses = uint; // Bitmask
  49  pub static use_repr: uint = 1;   /* Dependency on size/alignment/mode and
  50                                       take/drop glue */
  51  pub static use_tydesc: uint = 2; /* Takes the tydesc, or compares */
  52  pub static use_all: uint = use_repr|use_tydesc;
  53  
  54  #[deriving(Clone)]
  55  pub struct Context {
  56      ccx: @mut CrateContext,
  57      uses: @mut ~[type_uses]
  58  }

  60  pub fn type_uses_for(ccx: @mut CrateContext, fn_id: DefId, n_tps: uint)
  61      -> @~[type_uses] {
  62  
  63      fn store_type_uses(cx: Context, fn_id: DefId) -> @~[type_uses] {
  64          let Context { uses, ccx } = cx;
  65          let uses = @(*uses).clone(); // freeze
  66          ccx.type_use_cache.insert(fn_id, uses);
  67          uses
  68      }
  69  
  70      match ccx.type_use_cache.find(&fn_id) {
  71        Some(uses) => return *uses,
  72        None => ()
  73      }
  74  
  75      let fn_id_loc = if fn_id.crate == LOCAL_CRATE {
  76          fn_id
  77      } else {
  78          inline::maybe_instantiate_inline(ccx, fn_id)
  79      };
  80  
  81      // Conservatively assume full use for recursive loops
  82      ccx.type_use_cache.insert(fn_id, @vec::from_elem(n_tps, use_all));
  83  
  84      let cx = Context {
  85          ccx: ccx,
  86          uses: @mut vec::from_elem(n_tps, 0u)
  87      };
  88  
  89      // If the method is a default method, we mark all of the types as
  90      // used.  This is imprecise, but simple. Getting it right is
  91      // tricky because the substs on the call and the substs on the
  92      // default method differ, because of substs on the trait/impl.
  93      let is_default = ty::provided_source(ccx.tcx, fn_id_loc).is_some();
  94      // We also mark all of the params as used if it is an extern thing
  95      // that we haven't been able to inline yet.
  96      if is_default || fn_id_loc.crate != LOCAL_CRATE {
  97          for n in range(0u, n_tps) { cx.uses[n] |= use_all; }
  98          return store_type_uses(cx, fn_id);
  99      }
 100  
 101      let map_node = match ccx.tcx.items.find(&fn_id_loc.node) {
 102          Some(x) => {
 103              (*x).clone()
 104          }
 105          None => {
 106              ccx.sess.bug(fmt!("type_uses_for: unbound item ID %?",
 107                                fn_id_loc))
 108          }
 109      };
 110  
 111      match map_node {
 112        ast_map::node_item(@ast::item { node: item_fn(_, _, _, _, ref body),
 113                                        _ }, _) |
 114        ast_map::node_method(@ast::method {body: ref body, _}, _, _) => {
 115          handle_body(&cx, body);
 116        }
 117        ast_map::node_trait_method(*) => {
 118          // This will be a static trait method. For now, we just assume
 119          // it fully depends on all of the type information. (Doing
 120          // otherwise would require finding the actual implementation).
 121          for n in range(0u, n_tps) { cx.uses[n] |= use_all;}
 122          // We need to return early, before the arguments are processed,
 123          // because of difficulties in the handling of Self.
 124          return store_type_uses(cx, fn_id);
 125        }
 126        ast_map::node_variant(_, _, _) => {
 127          for n in range(0u, n_tps) { cx.uses[n] |= use_repr;}
 128        }
 129        ast_map::node_foreign_item(i@@foreign_item {
 130              node: foreign_item_fn(*),
 131              _
 132          },
 133          abi,
 134          _,
 135          _) => {
 136          if abi.is_intrinsic() {
 137              let nm = cx.ccx.sess.str_of(i.ident);
 138              let name = nm.as_slice();
 139              let flags = if name.starts_with("atomic_") {
 140                  0
 141              } else {
 142                  match name {
 143                      "size_of"  | "pref_align_of" | "min_align_of" |
 144                      "uninit"   | "init" | "transmute" | "move_val" |
 145                      "move_val_init" => use_repr,
 146  
 147                      "get_tydesc" | "needs_drop" | "contains_managed" => use_tydesc,
 148  
 149                      "visit_tydesc"  | "forget" | "frame_address" |
 150                      "morestack_addr" => 0,
 151  
 152                      "offset" |
 153                      "memcpy32" | "memcpy64" | "memmove32" | "memmove64" |
 154                      "memset32" | "memset64" => use_repr,
 155  
 156                      "sqrtf32" | "sqrtf64" | "powif32" | "powif64" |
 157                      "sinf32"  | "sinf64"  | "cosf32"  | "cosf64"  |
 158                      "powf32"  | "powf64"  | "expf32"  | "expf64"  |
 159                      "exp2f32" | "exp2f64" | "logf32"  | "logf64"  |
 160                      "log10f32"| "log10f64"| "log2f32" | "log2f64" |
 161                      "fmaf32"  | "fmaf64"  | "fabsf32" | "fabsf64" |
 162                      "floorf32"| "floorf64"| "ceilf32" | "ceilf64" |
 163                      "truncf32"| "truncf64" => 0,
 164  
 165                      "ctpop8" | "ctpop16" | "ctpop32" | "ctpop64" => 0,
 166  
 167                      "ctlz8" | "ctlz16" | "ctlz32" | "ctlz64" => 0,
 168                      "cttz8" | "cttz16" | "cttz32" | "cttz64" => 0,
 169  
 170                      "bswap16" | "bswap32" | "bswap64" => 0,
 171  
 172  
 173                      "i8_add_with_overflow"  | "u8_add_with_overflow" |
 174                      "i16_add_with_overflow" | "u16_add_with_overflow" |
 175                      "i32_add_with_overflow" | "u32_add_with_overflow" |
 176                      "i64_add_with_overflow" | "u64_add_with_overflow" => 0,
 177  
 178                      "i8_sub_with_overflow"  | "u8_sub_with_overflow" |
 179                      "i16_sub_with_overflow" | "u16_sub_with_overflow" |
 180                      "i32_sub_with_overflow" | "u32_sub_with_overflow" |
 181                      "i64_sub_with_overflow" | "u64_sub_with_overflow" => 0,
 182  
 183                      "i8_mul_with_overflow"  | "u8_mul_with_overflow" |
 184                      "i16_mul_with_overflow" | "u16_mul_with_overflow" |
 185                      "i32_mul_with_overflow" | "u32_mul_with_overflow" |
 186                      "i64_mul_with_overflow" | "u64_mul_with_overflow" => 0,
 187  
 188                      // would be cool to make these an enum instead of
 189                      // strings!
 190                      _ => fail!("unknown intrinsic in type_use")
 191                  }
 192              };
 193              for n in range(0u, n_tps) { cx.uses[n] |= flags;}
 194          }
 195        }
 196        ast_map::node_struct_ctor(*) => {
 197          // Similarly to node_variant, this monomorphized function just
 198          // uses the representations of all of its type parameters.
 199          for n in range(0u, n_tps) { cx.uses[n] |= use_repr; }
 200        }
 201        _ => {
 202          ccx.tcx.sess.bug(fmt!("unknown node type in type_use: %s",
 203                                ast_map::node_id_to_str(
 204                                  ccx.tcx.items,
 205                                  fn_id_loc.node,
 206                                  token::get_ident_interner())));
 207        }
 208      }
 209  
 210      // Now handle arguments
 211      match ty::get(ty::lookup_item_type(cx.ccx.tcx, fn_id).ty).sty {
 212          ty::ty_bare_fn(ty::BareFnTy {sig: ref sig, _}) |
 213          ty::ty_closure(ty::ClosureTy {sig: ref sig, _}) => {
 214              for arg in sig.inputs.iter() {
 215                  type_needs(&cx, use_repr, *arg);
 216              }
 217          }
 218          _ => ()
 219      }
 220  
 221      store_type_uses(cx, fn_id)
 222  }

 224  pub fn type_needs(cx: &Context, use_: uint, ty: ty::t) {
 225      // Optimization -- don't descend type if all params already have this use
 226      if cx.uses.iter().any(|&elt| elt & use_ != use_) {
 227          type_needs_inner(cx, use_, ty, @Nil);
 228      }
 229  }
 230  
 231  pub fn type_needs_inner(cx: &Context,
 232                          use_: uint,
 233                          ty: ty::t,
 234                          enums_seen: @List<DefId>) {
 235      do ty::maybe_walk_ty(ty) |ty| {
 236          if ty::type_has_params(ty) {
 237              match ty::get(ty).sty {
 238                  /*
 239                   This previously included ty_box -- that was wrong
 240                   because if we cast an @T to an trait (for example) and return
 241                   it, we depend on the drop glue for T (we have to write the
 242                   right tydesc into the result)
 243                   */
 244                  ty::ty_closure(*) |
 245                  ty::ty_bare_fn(*) |
 246                  ty::ty_ptr(_) |
 247                  ty::ty_rptr(_, _) |
 248                  ty::ty_trait(*) => false,
 249  
 250                ty::ty_enum(did, ref substs) => {
 251                  if list::find(enums_seen, |id| *id == did).is_none() {
 252                      let seen = @Cons(did, enums_seen);
 253                      let r = ty::enum_variants(cx.ccx.tcx, did);
 254                      for v in r.iter() {
 255                          for aty in v.args.iter() {
 256                              let t = ty::subst(cx.ccx.tcx, &(*substs), *aty);
 257                              type_needs_inner(cx, use_, t, seen);
 258                          }
 259                      }
 260                  }
 261                  false
 262                }
 263                ty::ty_param(p) => {
 264                  cx.uses[p.idx] |= use_;
 265                  false
 266                }
 267                _ => true
 268              }
 269          } else { false }
 270      }
 271  }

 273  pub fn node_type_needs(cx: &Context, use_: uint, id: NodeId) {
 274      type_needs(cx, use_, ty::node_id_to_type(cx.ccx.tcx, id));
 275  }

 277  pub fn mark_for_method_call(cx: &Context, e_id: NodeId, callee_id: NodeId) {
 278      let mut opt_static_did = None;
 279      {
 280          let r = cx.ccx.maps.method_map.find(&e_id);
 281          for mth in r.iter() {
 282              match mth.origin {
 283                typeck::method_static(did) => {
 284                    opt_static_did = Some(did);
 285                }
 286                typeck::method_param(typeck::method_param {
 287                    param_num: typeck::param_numbered(param),
 288                    _
 289                }) => {
 290                  cx.uses[param] |= use_tydesc;
 291                }
 292                _ => (),
 293              }
 294          }
 295      }
 296  
 297      // Note: we do not execute this code from within the each() call
 298      // above because the recursive call to `type_needs` can trigger
 299      // inlining and hence can cause `method_map` and
 300      // `node_type_substs` to be modified.
 301      for &did in opt_static_did.iter() {
 302          {
 303              let r = cx.ccx.tcx.node_type_substs.find_copy(&callee_id);
 304              for ts in r.iter() {
 305                  let type_uses = type_uses_for(cx.ccx, did, ts.len());
 306                  for (uses, subst) in type_uses.iter().zip(ts.iter()) {
 307                      type_needs(cx, *uses, *subst)
 308                  }
 309              }
 310          }
 311      }
 312  }

 314  pub fn mark_for_expr(cx: &Context, e: &Expr) {
 315      match e.node {
 316        ExprVstore(_, _) | ExprVec(_, _) | ExprStruct(*) | ExprTup(_) |
 317        ExprUnary(_, UnBox(_), _) | ExprUnary(_, UnUniq, _) |
 318        ExprBinary(_, BiAdd, _, _) | ExprRepeat(*) => {
 319          node_type_needs(cx, use_repr, e.id);
 320        }
 321        ExprCast(base, _) => {
 322          let result_t = ty::node_id_to_type(cx.ccx.tcx, e.id);
 323          match ty::get(result_t).sty {
 324              ty::ty_trait(*) => {
 325                // When we're casting to an trait, we need the
 326                // tydesc for the expr that's being cast.
 327                node_type_needs(cx, use_tydesc, base.id);
 328              }
 329              _ => ()
 330          }
 331        }
 332        ExprBinary(_, op, lhs, _) => {
 333          match op {
 334            BiEq | BiLt | BiLe | BiNe | BiGe | BiGt => {
 335              node_type_needs(cx, use_tydesc, lhs.id)
 336            }
 337            _ => ()
 338          }
 339        }
 340        ExprPath(_) | ExprSelf => {
 341          let opt_ts = cx.ccx.tcx.node_type_substs.find_copy(&e.id);
 342          for ts in opt_ts.iter() {
 343              let id = ast_util::def_id_of_def(cx.ccx.tcx.def_map.get_copy(&e.id));
 344              let uses_for_ts = type_uses_for(cx.ccx, id, ts.len());
 345              for (uses, subst) in uses_for_ts.iter().zip(ts.iter()) {
 346                  type_needs(cx, *uses, *subst)
 347              }
 348          }
 349        }
 350        ExprFnBlock(*) => {
 351            match ty::ty_closure_sigil(ty::expr_ty(cx.ccx.tcx, e)) {
 352                ast::OwnedSigil => {}
 353                ast::BorrowedSigil | ast::ManagedSigil => {
 354                    for fv in freevars::get_freevars(cx.ccx.tcx, e.id).iter() {
 355                        let node_id = ast_util::def_id_of_def(fv.def).node;
 356                        node_type_needs(cx, use_repr, node_id);
 357                    }
 358                }
 359            }
 360        }
 361        ExprAssign(val, _) | ExprAssignOp(_, _, val, _) |
 362        ExprRet(Some(val)) => {
 363          node_type_needs(cx, use_repr, val.id);
 364        }
 365        ExprIndex(callee_id, base, _) => {
 366          // FIXME (#2537): could be more careful and not count fields after
 367          // the chosen field.
 368          let base_ty = ty::node_id_to_type(cx.ccx.tcx, base.id);
 369          type_needs(cx, use_repr, ty::type_autoderef(cx.ccx.tcx, base_ty));
 370          mark_for_method_call(cx, e.id, callee_id);
 371        }
 372        ExprField(base, _, _) => {
 373          // Method calls are now a special syntactic form,
 374          // so `a.b` should always be a field.
 375          assert!(!cx.ccx.maps.method_map.contains_key(&e.id));
 376  
 377          let base_ty = ty::node_id_to_type(cx.ccx.tcx, base.id);
 378          type_needs(cx, use_repr, ty::type_autoderef(cx.ccx.tcx, base_ty));
 379        }
 380        ExprCall(f, _, _) => {
 381            let r = ty::ty_fn_args(ty::node_id_to_type(cx.ccx.tcx, f.id));
 382            for a in r.iter() {
 383                type_needs(cx, use_repr, *a);
 384            }
 385        }

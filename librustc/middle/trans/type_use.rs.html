  11 // Determines the ways in which a generic function body depends
  12 // on its type parameters. Used to aggressively reuse compiled
  13 // function bodies for different types.
  14 
  15 // This unfortunately depends on quite a bit of knowledge about the
  16 // details of the language semantics, and is likely to accidentally go
  17 // out of sync when something is changed. It could be made more
  18 // powerful by distinguishing between functions that only need to know
  19 // the size and alignment of a type, and those that also use its
  20 // drop/take glue. But this would increase the fragility of the code
  21 // to a ridiculous level, and probably not catch all that many extra
  22 // opportunities for reuse.
  23 
  24 // (An other approach to doing what this code does is to instrument
  25 // the translation code to set flags whenever it does something like
  26 // alloca a type or get a tydesc. This would not duplicate quite as
  27 // much information, but have the disadvantage of being very
  28 // invasive.)

  47 pub type type_uses = uint; // Bitmask
  48 pub static use_repr: uint = 1;   /* Dependency on size/alignment/mode and
  49                                      take/drop glue */
  50 pub static use_tydesc: uint = 2; /* Takes the tydesc, or compares */
  51 pub static use_all: uint = use_repr|use_tydesc;
  52 
  53 
  54 pub struct Context {
  55     ccx: @mut CrateContext,
  56     uses: @mut ~[type_uses]
  57 }

  59 pub fn type_uses_for(ccx: @mut CrateContext, fn_id: def_id, n_tps: uint)
  60     -> @~[type_uses] {
  61 
  62     fn store_type_uses(cx: Context, fn_id: def_id) -> @~[type_uses] {
  63         let Context { uses, ccx } = cx;
  64         let uses = @(*uses).clone(); // freeze
  65         ccx.type_use_cache.insert(fn_id, uses);
  66         uses
  67     }
  68 
  69     match ccx.type_use_cache.find(&fn_id) {
  70       Some(uses) => return *uses,
  71       None => ()
  72     }
  73 
  74     let fn_id_loc = if fn_id.crate == LOCAL_CRATE {
  75         fn_id
  76     } else {
  77         inline::maybe_instantiate_inline(ccx, fn_id)
  78     };
  79 
  80     // Conservatively assume full use for recursive loops
  81     ccx.type_use_cache.insert(fn_id, @vec::from_elem(n_tps, use_all));
  82 
  83     let cx = Context {
  84         ccx: ccx,
  85         uses: @mut vec::from_elem(n_tps, 0u)
  86     };
  87 
  88     // If the method is a default method, we mark all of the types as
  89     // used.  This is imprecise, but simple. Getting it right is
  90     // tricky because the substs on the call and the substs on the
  91     // default method differ, because of substs on the trait/impl.
  92     let is_default = ty::provided_source(ccx.tcx, fn_id_loc).is_some();
  93     // We also mark all of the params as used if it is an extern thing
  94     // that we haven't been able to inline yet.
  95     if is_default || fn_id_loc.crate != LOCAL_CRATE {
  96         for n in range(0u, n_tps) { cx.uses[n] |= use_all; }
  97         return store_type_uses(cx, fn_id);
  98     }
  99 
 100     let map_node = match ccx.tcx.items.find(&fn_id_loc.node) {
 101         Some(x) => {
 102             (*x).clone()
 103         }
 104         None => {
 105             ccx.sess.bug(fmt!("type_uses_for: unbound item ID %?",
 106                               fn_id_loc))
 107         }
 108     };
 109 
 110     match map_node {
 111       ast_map::node_item(@ast::item { node: item_fn(_, _, _, _, ref body),
 112                                       _ }, _) |
 113       ast_map::node_method(@ast::method {body: ref body, _}, _, _) => {
 114         handle_body(&cx, body);
 115       }
 116       ast_map::node_trait_method(*) => {
 117         // This will be a static trait method. For now, we just assume
 118         // it fully depends on all of the type information. (Doing
 119         // otherwise would require finding the actual implementation).
 120         for n in range(0u, n_tps) { cx.uses[n] |= use_all;}
 121         // We need to return early, before the arguments are processed,
 122         // because of difficulties in the handling of Self.
 123         return store_type_uses(cx, fn_id);
 124       }
 125       ast_map::node_variant(_, _, _) => {
 126         for n in range(0u, n_tps) { cx.uses[n] |= use_repr;}
 127       }
 128       ast_map::node_foreign_item(i@@foreign_item {
 129             node: foreign_item_fn(*),
 130             _
 131         },
 132         abi,
 133         _,
 134         _) => {
 135         if abi.is_intrinsic() {
 136             let nm = cx.ccx.sess.str_of(i.ident);
 137             let name = nm.as_slice();
 138             let flags = if name.starts_with("atomic_") {
 139                 0
 140             } else {
 141                 match name {
 142                     "size_of"  | "pref_align_of" | "min_align_of" |
 143                     "uninit"   | "init" | "transmute" | "move_val" |
 144                     "move_val_init" => use_repr,
 145 
 146                     "get_tydesc" | "needs_drop" | "contains_managed" => use_tydesc,
 147 
 148                     "visit_tydesc"  | "forget" | "frame_address" |
 149                     "morestack_addr" => 0,
 150 
 151                     "offset" | "memcpy32" | "memcpy64" | "memmove32" | "memmove64" |
 152                     "memset32" | "memset64" => use_repr,
 153 
 154                     "sqrtf32" | "sqrtf64" | "powif32" | "powif64" |
 155                     "sinf32"  | "sinf64"  | "cosf32"  | "cosf64"  |
 156                     "powf32"  | "powf64"  | "expf32"  | "expf64"  |
 157                     "exp2f32" | "exp2f64" | "logf32"  | "logf64"  |
 158                     "log10f32"| "log10f64"| "log2f32" | "log2f64" |
 159                     "fmaf32"  | "fmaf64"  | "fabsf32" | "fabsf64" |
 160                     "floorf32"| "floorf64"| "ceilf32" | "ceilf64" |
 161                     "truncf32"| "truncf64" => 0,
 162 
 163                     "ctpop8" | "ctpop16" | "ctpop32" | "ctpop64" => 0,
 164 
 165                     "ctlz8" | "ctlz16" | "ctlz32" | "ctlz64" => 0,
 166                     "cttz8" | "cttz16" | "cttz32" | "cttz64" => 0,
 167 
 168                     "bswap16" | "bswap32" | "bswap64" => 0,
 169 
 170                     // would be cool to make these an enum instead of
 171                     // strings!
 172                     _ => fail!("unknown intrinsic in type_use")
 173                 }
 174             };
 175             for n in range(0u, n_tps) { cx.uses[n] |= flags;}
 176         }
 177       }
 178       ast_map::node_struct_ctor(*) => {
 179         // Similarly to node_variant, this monomorphized function just
 180         // uses the representations of all of its type parameters.
 181         for n in range(0u, n_tps) { cx.uses[n] |= use_repr; }
 182       }
 183       _ => {
 184         ccx.tcx.sess.bug(fmt!("unknown node type in type_use: %s",
 185                               ast_map::node_id_to_str(
 186                                 ccx.tcx.items,
 187                                 fn_id_loc.node,
 188                                 token::get_ident_interner())));
 189       }
 190     }
 191 
 192     // Now handle arguments
 193     match ty::get(ty::lookup_item_type(cx.ccx.tcx, fn_id).ty).sty {
 194         ty::ty_bare_fn(ty::BareFnTy {sig: ref sig, _}) |
 195         ty::ty_closure(ty::ClosureTy {sig: ref sig, _}) => {
 196             for arg in sig.inputs.iter() {
 197                 type_needs(&cx, use_repr, *arg);
 198             }
 199         }
 200         _ => ()
 201     }
 202 
 203     store_type_uses(cx, fn_id)
 204 }

 206 pub fn type_needs(cx: &Context, use_: uint, ty: ty::t) {
 207     // Optimization -- don't descend type if all params already have this use
 208     let len = {
 209         let uses = &*cx.uses;
 210         uses.len()
 211     };
 212     for i in range(0u, len) {
 213         if cx.uses[i] & use_ != use_ {
 214             type_needs_inner(cx, use_, ty, @Nil);
 215             return;
 216         }
 217     }
 218 }
 219 
 220 pub fn type_needs_inner(cx: &Context,
 221                         use_: uint,
 222                         ty: ty::t,
 223                         enums_seen: @List<def_id>) {
 224     do ty::maybe_walk_ty(ty) |ty| {
 225         if ty::type_has_params(ty) {
 226             match ty::get(ty).sty {
 227                 /*
 228                  This previously included ty_box -- that was wrong
 229                  because if we cast an @T to an trait (for example) and return
 230                  it, we depend on the drop glue for T (we have to write the
 231                  right tydesc into the result)
 232                  */
 233                 ty::ty_closure(*) |
 234                 ty::ty_bare_fn(*) |
 235                 ty::ty_ptr(_) |
 236                 ty::ty_rptr(_, _) |
 237                 ty::ty_trait(*) => false,
 238 
 239               ty::ty_enum(did, ref substs) => {
 240                 if list::find(enums_seen, |id| *id == did).is_none() {
 241                     let seen = @Cons(did, enums_seen);
 242                     let r = ty::enum_variants(cx.ccx.tcx, did);
 243                     for v in r.iter() {
 244                         for aty in v.args.iter() {
 245                             let t = ty::subst(cx.ccx.tcx, &(*substs), *aty);
 246                             type_needs_inner(cx, use_, t, seen);
 247                         }
 248                     }
 249                 }
 250                 false
 251               }
 252               ty::ty_param(p) => {
 253                 cx.uses[p.idx] |= use_;
 254                 false
 255               }
 256               _ => true
 257             }
 258         } else { false }
 259     }
 260 }

 262 pub fn node_type_needs(cx: &Context, use_: uint, id: NodeId) {
 263     type_needs(cx, use_, ty::node_id_to_type(cx.ccx.tcx, id));
 264 }

 266 pub fn mark_for_method_call(cx: &Context, e_id: NodeId, callee_id: NodeId) {
 267     let mut opt_static_did = None;
 268     {
 269         let r = cx.ccx.maps.method_map.find(&e_id);
 270         for mth in r.iter() {
 271             match mth.origin {
 272               typeck::method_static(did) => {
 273                   opt_static_did = Some(did);
 274               }
 275               typeck::method_param(typeck::method_param {
 276                   param_num: typeck::param_numbered(param),
 277                   _
 278               }) => {
 279                 cx.uses[param] |= use_tydesc;
 280               }
 281               _ => (),
 282             }
 283         }
 284     }
 285 
 286     // Note: we do not execute this code from within the each() call
 287     // above because the recursive call to `type_needs` can trigger
 288     // inlining and hence can cause `method_map` and
 289     // `node_type_substs` to be modified.
 290     for &did in opt_static_did.iter() {
 291         {
 292             let r = cx.ccx.tcx.node_type_substs.find_copy(&callee_id);
 293             for ts in r.iter() {
 294                 let type_uses = type_uses_for(cx.ccx, did, ts.len());
 295                 for (uses, subst) in type_uses.iter().zip(ts.iter()) {
 296                     type_needs(cx, *uses, *subst)
 297                 }
 298             }
 299         }
 300     }
 301 }

 303 pub fn mark_for_expr(cx: &Context, e: &expr) {
 304     match e.node {
 305       expr_vstore(_, _) | expr_vec(_, _) | expr_struct(*) | expr_tup(_) |
 306       expr_unary(_, box(_), _) | expr_unary(_, uniq, _) |
 307       expr_binary(_, add, _, _) | expr_repeat(*) => {
 308         node_type_needs(cx, use_repr, e.id);
 309       }
 310       expr_cast(base, _) => {
 311         let result_t = ty::node_id_to_type(cx.ccx.tcx, e.id);
 312         match ty::get(result_t).sty {
 313             ty::ty_trait(*) => {
 314               // When we're casting to an trait, we need the
 315               // tydesc for the expr that's being cast.
 316               node_type_needs(cx, use_tydesc, base.id);
 317             }
 318             _ => ()
 319         }
 320       }
 321       expr_binary(_, op, lhs, _) => {
 322         match op {
 323           eq | lt | le | ne | ge | gt => {
 324             node_type_needs(cx, use_tydesc, lhs.id)
 325           }
 326           _ => ()
 327         }
 328       }
 329       expr_path(_) | expr_self => {
 330         let opt_ts = cx.ccx.tcx.node_type_substs.find_copy(&e.id);
 331         for ts in opt_ts.iter() {
 332             let id = ast_util::def_id_of_def(cx.ccx.tcx.def_map.get_copy(&e.id));
 333             let uses_for_ts = type_uses_for(cx.ccx, id, ts.len());
 334             for (uses, subst) in uses_for_ts.iter().zip(ts.iter()) {
 335                 type_needs(cx, *uses, *subst)
 336             }
 337         }
 338       }
 339       expr_fn_block(*) => {
 340           match ty::ty_closure_sigil(ty::expr_ty(cx.ccx.tcx, e)) {
 341               ast::OwnedSigil => {}
 342               ast::BorrowedSigil | ast::ManagedSigil => {
 343                   for fv in freevars::get_freevars(cx.ccx.tcx, e.id).iter() {
 344                       let node_id = ast_util::def_id_of_def(fv.def).node;
 345                       node_type_needs(cx, use_repr, node_id);
 346                   }
 347               }
 348           }
 349       }
 350       expr_assign(val, _) | expr_assign_op(_, _, val, _) |
 351       expr_ret(Some(val)) => {
 352         node_type_needs(cx, use_repr, val.id);
 353       }
 354       expr_index(callee_id, base, _) => {
 355         // FIXME (#2537): could be more careful and not count fields after
 356         // the chosen field.
 357         let base_ty = ty::node_id_to_type(cx.ccx.tcx, base.id);
 358         type_needs(cx, use_repr, ty::type_autoderef(cx.ccx.tcx, base_ty));
 359         mark_for_method_call(cx, e.id, callee_id);
 360       }
 361       expr_field(base, _, _) => {
 362         // Method calls are now a special syntactic form,
 363         // so `a.b` should always be a field.
 364         assert!(!cx.ccx.maps.method_map.contains_key(&e.id));
 365 
 366         let base_ty = ty::node_id_to_type(cx.ccx.tcx, base.id);
 367         type_needs(cx, use_repr, ty::type_autoderef(cx.ccx.tcx, base_ty));
 368       }
 369       expr_log(_, val) => {
 370         node_type_needs(cx, use_tydesc, val.id);
 371       }
 372       expr_call(f, _, _) => {
 373           let r = ty::ty_fn_args(ty::node_id_to_type(cx.ccx.tcx, f.id));
 374           for a in r.iter() {
 375               type_needs(cx, use_repr, *a);
 376           }
 377       }
 378       expr_method_call(callee_id, rcvr, _, _, _, _) => {
 379         let base_ty = ty::node_id_to_type(cx.ccx.tcx, rcvr.id);
 380         type_needs(cx, use_repr, ty::type_autoderef(cx.ccx.tcx, base_ty));
 381 
 382         let r = ty::ty_fn_args(ty::node_id_to_type(cx.ccx.tcx, callee_id));
 383         for a in r.iter() {
 384             type_needs(cx, use_repr, *a);
 385         }
 386         mark_for_method_call(cx, e.id, callee_id);
 387       }
 388 
 389       expr_inline_asm(ref ia) => {
 390         for &(_, input) in ia.inputs.iter() {
 391           node_type_needs(cx, use_repr, input.id);
 392         }
 393         for &(_, out) in ia.outputs.iter() {
 394           node_type_needs(cx, use_repr, out.id);
 395         }
 396       }
 397 
 398       expr_paren(e) => mark_for_expr(cx, e),
 399 
 400       expr_match(*) | expr_block(_) | expr_if(*) | expr_while(*) |
 401       expr_break(_) | expr_again(_) | expr_unary(*) | expr_lit(_) |
 402       expr_mac(_) | expr_addr_of(*) | expr_ret(_) | expr_loop(*) |
 403       expr_do_body(_) => (),
 404 
 405       expr_for_loop(*) => fail!("non-desugared expr_for_loop")
 406     }
 407 }

 409 pub fn handle_body(cx: &Context, body: &Block) {

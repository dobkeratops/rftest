   11  /*!
   12   * # Translation of Expressions
   13   *
   14   * Public entry points:
   15   *
   16   * - `trans_into(bcx, expr, dest) -> bcx`: evaluates an expression,
   17   *   storing the result into `dest`. This is the preferred form, if you
   18   *   can manage it.
   19   *
   20   * - `trans(bcx, expr) -> DatumBlock`: evaluates an expression, yielding
   21   *   `Datum` with the result. You can then store the datum, inspect
   22   *   the value, etc. This may introduce temporaries if the datum is a
   23   *   structural type.
   24   *
   25   * - `trans_to_lvalue(bcx, expr, "...") -> DatumBlock`: evaluates an
   26   *   expression and ensures that the result has a cleanup associated with it,
   27   *   creating a temporary stack slot if necessary.
   28   *
   29   * - `trans_local_var -> Datum`: looks up a local variable or upvar.
   30   *
   31   * See doc.rs for more comments.
   32   */

   98  pub fn trans_into<'a>(bcx: &'a Block<'a>,
   99                        expr: &ast::Expr,
  100                        dest: Dest)
  101                        -> &'a Block<'a> {
  102      /*!
  103       * This function is equivalent to `trans(bcx, expr).store_to_dest(dest)`
  104       * but it may generate better optimized LLVM code.
  105       */
  106  
  107      let mut bcx = bcx;
  108  
  109      if bcx.tcx().adjustments.borrow().contains_key(&expr.id) {
  110          // use trans, which may be less efficient but
  111          // which will perform the adjustments:
  112          let datum = unpack_datum!(bcx, trans(bcx, expr));
  113          return datum.store_to_dest(bcx, dest, expr.id)
  114      }
  115  
  116      debug!("trans_into() expr={}", expr.repr(bcx.tcx()));
  117      debuginfo::set_source_location(bcx.fcx, expr.id, expr.span);
  118  
  119      bcx.fcx.push_ast_cleanup_scope(expr.id);
  120  
  121      let kind = ty::expr_kind(bcx.tcx(), bcx.ccx().maps.method_map, expr);
  122      bcx = match kind {
  123          ty::LvalueExpr | ty::RvalueDatumExpr => {
  124              trans_unadjusted(bcx, expr).store_to_dest(dest, expr.id)
  125          }
  126          ty::RvalueDpsExpr => {
  127              trans_rvalue_dps_unadjusted(bcx, expr, dest)
  128          }
  129          ty::RvalueStmtExpr => {
  130              trans_rvalue_stmt_unadjusted(bcx, expr)
  131          }
  132      };
  133  
  134      bcx.fcx.pop_and_trans_ast_cleanup_scope(bcx, expr.id)
  135  }

  137  pub fn trans<'a>(bcx: &'a Block<'a>,
  138                   expr: &ast::Expr)
  139                   -> DatumBlock<'a, Expr> {
  140      /*!
  141       * Translates an expression, returning a datum (and new block)
  142       * encapsulating the result. When possible, it is preferred to
  143       * use `trans_into`, as that may avoid creating a temporary on
  144       * the stack.
  145       */
  146  
  147      debug!("trans(expr={})", bcx.expr_to_str(expr));
  148  
  149      let mut bcx = bcx;
  150      let fcx = bcx.fcx;
  151  
  152      fcx.push_ast_cleanup_scope(expr.id);
  153      let datum = unpack_datum!(bcx, trans_unadjusted(bcx, expr));
  154      let datum = unpack_datum!(bcx, apply_adjustments(bcx, expr, datum));
  155      bcx = fcx.pop_and_trans_ast_cleanup_scope(bcx, expr.id);
  156      return DatumBlock(bcx, datum);
  157  }

  159  fn apply_adjustments<'a>(bcx: &'a Block<'a>,
  160                           expr: &ast::Expr,
  161                           datum: Datum<Expr>)
  162                           -> DatumBlock<'a, Expr> {
  163      /*!
  164       * Helper for trans that apply adjustments from `expr` to `datum`,
  165       * which should be the unadjusted translation of `expr`.
  166       */
  167  
  168      let mut bcx = bcx;
  169      let mut datum = datum;
  170      let adjustment = match bcx.tcx().adjustments.borrow().find_copy(&expr.id) {
  171          None => {
  172              return DatumBlock(bcx, datum);
  173          }
  174          Some(adj) => { adj }
  175      };
  176      debug!("unadjusted datum for expr {}: {}",
  177             expr.id, datum.to_str(bcx.ccx()));
  178      match *adjustment {
  179          AutoAddEnv(..) => {
  180              datum = unpack_datum!(bcx, add_env(bcx, expr, datum));
  181          }
  182          AutoDerefRef(ref adj) => {
  183              if adj.autoderefs > 0 {
  184                  datum = unpack_datum!(
  185                      bcx, deref_multiple(bcx, expr, datum, adj.autoderefs));
  186              }
  187  
  188              datum = match adj.autoref {
  189                  None => {
  190                      datum
  191                  }
  192                  Some(AutoUnsafe(..)) | // region + unsafe ptrs have same repr
  193                  Some(AutoPtr(..)) => {
  194                      unpack_datum!(bcx, auto_ref(bcx, datum, expr))
  195                  }
  196                  Some(AutoBorrowVec(..)) => {
  197                      unpack_datum!(bcx, auto_slice(bcx, expr, datum))
  198                  }
  199                  Some(AutoBorrowVecRef(..)) => {
  200                      unpack_datum!(bcx, auto_slice_and_ref(bcx, expr, datum))
  201                  }
  202                  Some(AutoBorrowFn(..)) => {
  203                      let adjusted_ty = ty::adjust_ty(bcx.tcx(), expr.span, expr.id, datum.ty,
  204                                                      Some(adjustment), |method_call| {
  205                          bcx.ccx().maps.method_map.borrow()
  206                             .find(&method_call).map(|method| method.ty)
  207                      });
  208                      unpack_datum!(bcx, auto_borrow_fn(bcx, adjusted_ty, datum))
  209                  }
  210                  Some(AutoBorrowObj(..)) => {
  211                      unpack_datum!(bcx, auto_borrow_obj(bcx, expr, datum))
  212                  }
  213              };
  214          }
  215          AutoObject(..) => {
  216              let adjusted_ty = ty::expr_ty_adjusted(bcx.tcx(), expr,
  217                                                     &*bcx.ccx().maps.method_map.borrow());
  218              let scratch = rvalue_scratch_datum(bcx, adjusted_ty, "__adjust");
  219              bcx = meth::trans_trait_cast(
  220                  bcx, datum, expr.id, SaveIn(scratch.val));
  221              datum = scratch.to_expr_datum();
  222          }
  223      }
  224      debug!("after adjustments, datum={}", datum.to_str(bcx.ccx()));
  225      return DatumBlock {bcx: bcx, datum: datum};

  232          // Currently, all closure types are represented precisely the

  241      fn auto_slice<'a>(
  242                    bcx: &'a Block<'a>,
  243                    expr: &ast::Expr,
  244                    datum: Datum<Expr>)
  245                    -> DatumBlock<'a, Expr> {
  246          // This is not the most efficient thing possible; since slices
  247          // are two words it'd be better if this were compiled in
  248          // 'dest' mode, but I can't find a nice way to structure the
  249          // code and keep it DRY that accommodates that use case at the
  250          // moment.
  251  
  252          let mut bcx = bcx;
  253          let tcx = bcx.tcx();
  254          let unit_ty = ty::sequence_element_type(tcx, datum.ty);
  255  
  256          // Arrange cleanup, if not already done. This is needed in
  257          // case we are auto-slicing an owned vector or some such.
  258          let datum = unpack_datum!(
  259              bcx, datum.to_lvalue_datum(bcx, "auto_slice", expr.id));
  260  
  261          let (base, len) = datum.get_vec_base_and_len(bcx);
  262  
  263          // this type may have a different region/mutability than the
  264          // real one, but it will have the same runtime representation
  265          let slice_ty = ty::mk_vec(tcx,
  266                                    ty::mt { ty: unit_ty, mutbl: ast::MutImmutable },
  267                                    ty::vstore_slice(ty::ReStatic));
  268  
  269          let scratch = rvalue_scratch_datum(bcx, slice_ty, "__adjust");
  270          Store(bcx, base, GEPi(bcx, scratch.val, [0u, abi::slice_elt_base]));
  271          Store(bcx, len, GEPi(bcx, scratch.val, [0u, abi::slice_elt_len]));
  272          DatumBlock(bcx, scratch.to_expr_datum())
  273      }

  314  pub fn trans_to_lvalue<'a>(bcx: &'a Block<'a>,
  315                             expr: &ast::Expr,
  316                             name: &str)
  317                             -> DatumBlock<'a, Lvalue> {
  318      /*!
  319       * Translates an expression in "lvalue" mode -- meaning that it
  320       * returns a reference to the memory that the expr represents.
  321       *
  322       * If this expression is an rvalue, this implies introducing a
  323       * temporary.  In other words, something like `x().f` is
  324       * translated into roughly the equivalent of
  325       *
  326       *   { tmp = x(); tmp.f }
  327       */
  328  
  329      let mut bcx = bcx;
  330      let datum = unpack_datum!(bcx, trans(bcx, expr));
  331      return datum.to_lvalue_datum(bcx, name, expr.id);
  332  }

  334  fn trans_unadjusted<'a>(bcx: &'a Block<'a>,
  335                          expr: &ast::Expr)
  336                          -> DatumBlock<'a, Expr> {
  337      /*!
  338       * A version of `trans` that ignores adjustments. You almost
  339       * certainly do not want to call this directly.
  340       */
  341  
  342      let mut bcx = bcx;
  343  
  344      debug!("trans_unadjusted(expr={})", bcx.expr_to_str(expr));
  345      let _indenter = indenter();
  346  
  347      debuginfo::set_source_location(bcx.fcx, expr.id, expr.span);
  348  
  349      return match ty::expr_kind(bcx.tcx(), bcx.ccx().maps.method_map, expr) {
  350          ty::LvalueExpr | ty::RvalueDatumExpr => {
  351              let datum = unpack_datum!(bcx, {
  352                  trans_datum_unadjusted(bcx, expr)
  353              });
  354  
  355              DatumBlock {bcx: bcx, datum: datum}
  356          }
  357  
  358          ty::RvalueStmtExpr => {
  359              bcx = trans_rvalue_stmt_unadjusted(bcx, expr);
  360              nil(bcx, expr_ty(bcx, expr))
  361          }
  362  
  363          ty::RvalueDpsExpr => {
  364              let ty = expr_ty(bcx, expr);
  365              if type_is_zero_size(bcx.ccx(), ty) {
  366                  bcx = trans_rvalue_dps_unadjusted(bcx, expr, Ignore);
  367                  nil(bcx, ty)
  368              } else {
  369                  let scratch = rvalue_scratch_datum(bcx, ty, "");
  370                  bcx = trans_rvalue_dps_unadjusted(
  371                      bcx, expr, SaveIn(scratch.val));
  372  
  373                  // Note: this is not obviously a good idea.  It causes
  374                  // immediate values to be loaded immediately after a
  375                  // return from a call or other similar expression,
  376                  // which in turn leads to alloca's having shorter
  377                  // lifetimes and hence larger stack frames.  However,
  378                  // in turn it can lead to more register pressure.
  379                  // Still, in practice it seems to increase
  380                  // performance, since we have fewer problems with
  381                  // morestack churn.
  382                  let scratch = unpack_datum!(
  383                      bcx, scratch.to_appropriate_datum(bcx));
  384  
  385                  DatumBlock(bcx, scratch.to_expr_datum())
  386              }
  387          }
  388      };

  390      fn nil<'a>(bcx: &'a Block<'a>, ty: ty::t) -> DatumBlock<'a, Expr> {
  391          let llval = C_undef(type_of::type_of(bcx.ccx(), ty));
  392          let datum = immediate_rvalue(llval, ty);
  393          DatumBlock(bcx, datum.to_expr_datum())
  394      }
  395  }

  397  fn trans_datum_unadjusted<'a>(bcx: &'a Block<'a>,
  398                                expr: &ast::Expr)
  399                                -> DatumBlock<'a, Expr> {
  400      let mut bcx = bcx;
  401      let fcx = bcx.fcx;
  402      let _icx = push_ctxt("trans_datum_unadjusted");
  403  
  404      match expr.node {
  405          ast::ExprParen(e) => {
  406              trans(bcx, e)
  407          }
  408          ast::ExprPath(_) => {
  409              trans_def(bcx, expr, bcx.def(expr.id))
  410          }

  412              trans_rec_field(bcx, base, ident)
  413          }
  414          ast::ExprIndex(base, idx) => {
  415              trans_index(bcx, expr, base, idx)
  416          }
  417          ast::ExprVstore(contents, ast::ExprVstoreUniq) => {
  418              fcx.push_ast_cleanup_scope(contents.id);
  419              let datum = unpack_datum!(
  420                  bcx, tvec::trans_uniq_vstore(bcx, expr, contents));
  421              bcx = fcx.pop_and_trans_ast_cleanup_scope(bcx, contents.id);

  423          }

  428              let contents_ty = expr_ty(bcx, contents);

  433          ast::ExprBinary(op, lhs, rhs) => {
  434              trans_binary(bcx, expr, op, lhs, rhs)
  435          }
  436          ast::ExprUnary(op, x) => {
  437              trans_unary(bcx, expr, op, x)
  438          }
  439          ast::ExprAddrOf(_, x) => {
  440              trans_addr_of(bcx, expr, x)
  441          }
  442          ast::ExprCast(val, _) => {

  445          }
  446          _ => {

  448                  expr.span,
  449                  format!("trans_rvalue_datum_unadjusted reached \
  450                        fall-through case: {:?}",
  451                       expr.node));
  452          }
  453      }
  454  }

  476  fn trans_index<'a>(bcx: &'a Block<'a>,
  477                     index_expr: &ast::Expr,
  478                     base: &ast::Expr,
  479                     idx: &ast::Expr)
  480                     -> DatumBlock<'a, Expr> {
  481      //! Translates `base[idx]`.
  482  
  483      let _icx = push_ctxt("trans_index");
  484      let ccx = bcx.ccx();
  485      let mut bcx = bcx;
  486  
  487      let base_datum = unpack_datum!(bcx, trans_to_lvalue(bcx, base, "index"));
  488  
  489      // Translate index expression and cast to a suitable LLVM integer.
  490      // Rust is less strict than LLVM in this regard.
  491      let ix_datum = unpack_datum!(bcx, trans(bcx, idx));
  492      let ix_val = ix_datum.to_llscalarish(bcx);
  493      let ix_size = machine::llbitsize_of_real(bcx.ccx(), val_ty(ix_val));
  494      let int_size = machine::llbitsize_of_real(bcx.ccx(), ccx.int_type);
  495      let ix_val = {
  496          if ix_size < int_size {
  497              if ty::type_is_signed(expr_ty(bcx, idx)) {
  498                  SExt(bcx, ix_val, ccx.int_type)
  499              } else { ZExt(bcx, ix_val, ccx.int_type) }
  500          } else if ix_size > int_size {
  501              Trunc(bcx, ix_val, ccx.int_type)
  502          } else {
  503              ix_val
  504          }
  505      };
  506  
  507      let vt = tvec::vec_types(bcx, base_datum.ty);
  508      base::maybe_name_value(bcx.ccx(), vt.llunit_size, "unit_sz");
  509  
  510      let (base, len) = base_datum.get_vec_base_and_len(bcx);
  511  
  512      debug!("trans_index: base {}", bcx.val_to_str(base));
  513      debug!("trans_index: len {}", bcx.val_to_str(len));
  514  
  515      let bounds_check = ICmp(bcx, lib::llvm::IntUGE, ix_val, len);
  516      let expect = ccx.intrinsics.get_copy(&("llvm.expect.i1"));
  517      let expected = Call(bcx, expect, [bounds_check, C_i1(ccx, false)], []);
  518      let bcx = with_cond(bcx, expected, |bcx| {
  519              controlflow::trans_fail_bounds_check(bcx, index_expr.span, ix_val, len)
  520          });
  521      let elt = InBoundsGEP(bcx, base, [ix_val]);
  522      let elt = PointerCast(bcx, elt, vt.llunit_ty.ptr_to());
  523      DatumBlock(bcx, Datum(elt, vt.unit_ty, LvalueExpr))
  524  }

  526  fn trans_def<'a>(bcx: &'a Block<'a>,
  527                   ref_expr: &ast::Expr,
  528                   def: ast::Def)
  529                   -> DatumBlock<'a, Expr>
  530  {
  531      //! Translates a reference to a path.
  532  
  533      let _icx = push_ctxt("trans_def_lvalue");
  534      match def {
  535          ast::DefFn(..) | ast::DefStaticMethod(..) |
  536          ast::DefStruct(_) | ast::DefVariant(..) => {
  537              trans_def_fn_unadjusted(bcx, ref_expr, def)
  538          }
  539          ast::DefStatic(did, _) => {
  540              let const_ty = expr_ty(bcx, ref_expr);
  541  
  542              fn get_did(ccx: &CrateContext, did: ast::DefId)
  543                         -> ast::DefId {
  544                  if did.krate != ast::LOCAL_CRATE {
  545                      inline::maybe_instantiate_inline(ccx, did)
  546                  } else {
  547                      did
  548                  }
  549              }
  550  
  551              fn get_val<'a>(bcx: &'a Block<'a>, did: ast::DefId, const_ty: ty::t)
  552                         -> ValueRef {
  553                  // For external constants, we don't inline.
  554                  if did.krate == ast::LOCAL_CRATE {
  555                      // The LLVM global has the type of its initializer,
  556                      // which may not be equal to the enum's type for
  557                      // non-C-like enums.
  558                      let val = base::get_item_val(bcx.ccx(), did.node);
  559                      let pty = type_of::type_of(bcx.ccx(), const_ty).ptr_to();
  560                      PointerCast(bcx, val, pty)
  561                  } else {
  562                      match bcx.ccx().extern_const_values.borrow().find(&did) {
  563                          None => {}  // Continue.
  564                          Some(llval) => {
  565                              return *llval;
  566                          }
  567                      }
  568  
  569                      unsafe {
  570                          let llty = type_of::type_of(bcx.ccx(), const_ty);
  571                          let symbol = csearch::get_symbol(
  572                              &bcx.ccx().sess().cstore,
  573                              did);
  574                          let llval = symbol.with_c_str(|buf| {
  575                                  llvm::LLVMAddGlobal(bcx.ccx().llmod,
  576                                                      llty.to_ref(),
  577                                                      buf)
  578                              });
  579                          bcx.ccx().extern_const_values.borrow_mut()
  580                             .insert(did, llval);
  581                          llval
  582                      }
  583                  }
  584              }
  585  
  586              let did = get_did(bcx.ccx(), did);
  587              let val = get_val(bcx, did, const_ty);
  588              DatumBlock(bcx, Datum(val, const_ty, LvalueExpr))
  589          }
  590          _ => {
  591              DatumBlock(bcx, trans_local_var(bcx, def).to_expr_datum())
  592          }
  593      }
  594  }

  596  fn trans_rvalue_stmt_unadjusted<'a>(bcx: &'a Block<'a>,
  597                                      expr: &ast::Expr)
  598                                      -> &'a Block<'a> {
  599      let mut bcx = bcx;
  600      let _icx = push_ctxt("trans_rvalue_stmt");
  601  
  602      if bcx.unreachable.get() {
  603          return bcx;
  604      }
  605  
  606      match expr.node {
  607          ast::ExprParen(e) => {
  608              trans_into(bcx, e, Ignore)
  609          }
  610          ast::ExprBreak(label_opt) => {
  611              controlflow::trans_break(bcx, expr.id, label_opt)
  612          }
  613          ast::ExprAgain(label_opt) => {
  614              controlflow::trans_cont(bcx, expr.id, label_opt)
  615          }
  616          ast::ExprRet(ex) => {
  617              controlflow::trans_ret(bcx, ex)
  618          }
  619          ast::ExprWhile(cond, body) => {
  620              controlflow::trans_while(bcx, expr.id, cond, body)
  621          }
  622          ast::ExprLoop(body, _) => {
  623              controlflow::trans_loop(bcx, expr.id, body)
  624          }
  625          ast::ExprAssign(dst, src) => {
  626              let src_datum = unpack_datum!(bcx, trans(bcx, src));
  627              let dst_datum = unpack_datum!(bcx, trans_to_lvalue(bcx, dst, "assign"));
  628  
  629              if ty::type_needs_drop(bcx.tcx(), dst_datum.ty) {
  630                  // If there are destructors involved, make sure we
  631                  // are copying from an rvalue, since that cannot possible
  632                  // alias an lvalue. We are concerned about code like:
  633                  //
  634                  //   a = a
  635                  //
  636                  // but also
  637                  //
  638                  //   a = a.b
  639                  //
  640                  // where e.g. a : Option<Foo> and a.b :
  641                  // Option<Foo>. In that case, freeing `a` before the
  642                  // assignment may also free `a.b`!
  643                  //
  644                  // We could avoid this intermediary with some analysis
  645                  // to determine whether `dst` may possibly own `src`.
  646                  let src_datum = unpack_datum!(
  647                      bcx, src_datum.to_rvalue_datum(bcx, "ExprAssign"));
  648                  bcx = glue::drop_ty(bcx, dst_datum.val, dst_datum.ty);
  649                  src_datum.store_to(bcx, dst_datum.val)
  650              } else {
  651                  src_datum.store_to(bcx, dst_datum.val)
  652              }
  653          }
  654          ast::ExprAssignOp(op, dst, src) => {
  655              trans_assign_op(bcx, expr, op, dst, src)
  656          }
  657          ast::ExprInlineAsm(ref a) => {
  658              asm::trans_inline_asm(bcx, a)
  659          }
  660          _ => {
  661              bcx.tcx().sess.span_bug(
  662                  expr.span,
  663                  format!("trans_rvalue_stmt_unadjusted reached \
  664                        fall-through case: {:?}",
  665                       expr.node));
  666          }
  667      }
  668  }

  670  fn trans_rvalue_dps_unadjusted<'a>(bcx: &'a Block<'a>,
  671                                     expr: &ast::Expr,
  672                                     dest: Dest)
  673                                     -> &'a Block<'a> {
  674      let _icx = push_ctxt("trans_rvalue_dps_unadjusted");
  675      let mut bcx = bcx;
  676      let tcx = bcx.tcx();
  677      let fcx = bcx.fcx;
  678  
  679      match expr.node {
  680          ast::ExprParen(e) => {
  681              trans_into(bcx, e, dest)
  682          }
  683          ast::ExprPath(_) => {
  684              trans_def_dps_unadjusted(bcx, expr, bcx.def(expr.id), dest)
  685          }
  686          ast::ExprIf(cond, thn, els) => {
  687              controlflow::trans_if(bcx, expr.id, cond, thn, els, dest)
  688          }
  689          ast::ExprMatch(discr, ref arms) => {
  690              _match::trans_match(bcx, expr, discr, arms.as_slice(), dest)
  691          }
  692          ast::ExprBlock(blk) => {
  693              controlflow::trans_block(bcx, blk, dest)
  694          }
  695          ast::ExprStruct(_, ref fields, base) => {
  696              trans_rec_or_struct(bcx,
  697                                  fields.as_slice(),
  698                                  base,
  699                                  expr.span,
  700                                  expr.id,
  701                                  dest)
  702          }
  703          ast::ExprTup(ref args) => {
  704              let repr = adt::represent_type(bcx.ccx(), expr_ty(bcx, expr));
  705              let numbered_fields: Vec<(uint, @ast::Expr)> =
  706                  args.iter().enumerate().map(|(i, arg)| (i, *arg)).collect();
  707              trans_adt(bcx, repr, 0, numbered_fields.as_slice(), None, dest)
  708          }
  709          ast::ExprLit(lit) => {
  710              match lit.node {
  711                  ast::LitStr(ref s, _) => {
  712                      tvec::trans_lit_str(bcx, expr, (*s).clone(), dest)
  713                  }
  714                  _ => {
  715                      bcx.tcx()
  716                         .sess
  717                         .span_bug(expr.span,
  718                                   "trans_rvalue_dps_unadjusted shouldn't be \
  719                                    translating this type of literal")
  720                  }
  721              }
  722          }
  723          ast::ExprVstore(contents, ast::ExprVstoreSlice) |
  724          ast::ExprVstore(contents, ast::ExprVstoreMutSlice) => {
  725              fcx.push_ast_cleanup_scope(contents.id);
  726              bcx = tvec::trans_slice_vstore(bcx, expr, contents, dest);
  727              fcx.pop_and_trans_ast_cleanup_scope(bcx, contents.id)
  728          }
  729          ast::ExprVec(..) | ast::ExprRepeat(..) => {
  730              tvec::trans_fixed_vstore(bcx, expr, expr, dest)
  731          }
  732          ast::ExprFnBlock(decl, body) |
  733          ast::ExprProc(decl, body) => {
  734              let expr_ty = expr_ty(bcx, expr);
  735              let sigil = ty::ty_closure_sigil(expr_ty);
  736              debug!("translating block function {} with type {}",
  737                     expr_to_str(expr), expr_ty.repr(tcx));
  738              closure::trans_expr_fn(bcx, sigil, decl, body, expr.id, dest)
  739          }
  740          ast::ExprCall(f, ref args) => {
  741              callee::trans_call(bcx, expr, f, callee::ArgExprs(args.as_slice()), dest)
  742          }
  743          ast::ExprMethodCall(_, _, ref args) => {
  744              callee::trans_method_call(bcx,
  745                                        expr,
  746                                        *args.get(0),
  747                                        callee::ArgExprs(args.as_slice()),
  748                                        dest)
  749          }
  750          ast::ExprBinary(_, lhs, rhs) => {
  751              // if not overloaded, would be RvalueDatumExpr
  752              let lhs = unpack_datum!(bcx, trans(bcx, lhs));
  753              let rhs_datum = unpack_datum!(bcx, trans(bcx, rhs));
  754              trans_overloaded_op(bcx, expr, MethodCall::expr(expr.id), lhs,
  755                                  Some((rhs_datum, rhs.id)), Some(dest)).bcx
  756          }
  757          ast::ExprUnary(_, subexpr) => {
  758              // if not overloaded, would be RvalueDatumExpr
  759              let arg = unpack_datum!(bcx, trans(bcx, subexpr));
  760              trans_overloaded_op(bcx, expr, MethodCall::expr(expr.id),
  761                                  arg, None, Some(dest)).bcx
  762          }
  763          ast::ExprIndex(base, idx) => {
  764              // if not overloaded, would be RvalueDatumExpr
  765              let base = unpack_datum!(bcx, trans(bcx, base));
  766              let idx_datum = unpack_datum!(bcx, trans(bcx, idx));
  767              trans_overloaded_op(bcx, expr, MethodCall::expr(expr.id), base,
  768                                  Some((idx_datum, idx.id)), Some(dest)).bcx
  769          }
  770          ast::ExprCast(val, _) => {

  777                  _ => {

  779                                              "expr_cast of non-trait");
  780                  }
  781              }
  782          }
  783          ast::ExprAssignOp(op, dst, src) => {
  784              trans_assign_op(bcx, expr, op, dst, src)
  785          }
  786          ast::ExprBox(_, contents) => {

  49      pub fn count_insn(&self, category: &str) {
  50          if self.ccx.sess.trans_stats() {
  51              self.ccx.stats.n_llvm_insns += 1;
  52          }
  53          if self.ccx.sess.count_llvm_insns() {
  54              do base::with_insn_ctxt |v| {
  55                  let h = &mut self.ccx.stats.llvm_insns;
  56  
  57                  // Build version of path with cycles removed.
  58  
  59                  // Pass 1: scan table mapping str -> rightmost pos.
  60                  let mut mm = HashMap::new();
  61                  let len = v.len();
  62                  let mut i = 0u;
  63                  while i < len {
  64                      mm.insert(v[i], i);
  65                      i += 1u;
  66                  }
  67  
  68                  // Pass 2: concat strings for each elt, skipping
  69                  // forwards over any cycles by advancing to rightmost
  70                  // occurrence of each element in path.
  71                  let mut s = ~".";
  72                  i = 0u;
  73                  while i < len {
  74                      i = *mm.get(&v[i]);
  75                      s.push_char('/');
  76                      s.push_str(v[i]);
  77                      i += 1u;
  78                  }
  79  
  80                  s.push_char('/');
  81                  s.push_str(category);
  82  
  83                  let n = match h.find(&s) {
  84                      Some(&n) => n,
  85                      _ => 0u
  86                  };
  87                  h.insert(s, n+1u);
  88              }
  89          }
  90      }

 459      pub fn load_range_assert(&self, ptr: ValueRef, lo: c_ulonglong,
 460                             hi: c_ulonglong, signed: lib::llvm::Bool) -> ValueRef {
 461          let value = self.load(ptr);
 462  
 463          unsafe {
 464              let t = llvm::LLVMGetElementType(llvm::LLVMTypeOf(ptr));
 465              let min = llvm::LLVMConstInt(t, lo, signed);
 466              let max = llvm::LLVMConstInt(t, hi, signed);
 467  
 468              do [min, max].as_imm_buf |ptr, len| {
 469                  llvm::LLVMSetMetadata(value, lib::llvm::MD_range as c_uint,
 470                                        llvm::LLVMMDNodeInContext(self.ccx.llcx,
 471                                                                  ptr, len as c_uint));
 472              }
 473          }
 474  
 475          value
 476      }

 478      pub fn store(&self, val: ValueRef, ptr: ValueRef) {
 479          debug!("Store %s -> %s",
 480                 self.ccx.tn.val_to_str(val),

 508      // Simple wrapper around GEP that takes an array of ints and wraps them
 509      // in C_i32()
 510      #[inline]
 511      pub fn gepi(&self, base: ValueRef, ixs: &[uint]) -> ValueRef {
 512          // Small vector optimization. This should catch 100% of the cases that
 513          // we care about.
 514          if ixs.len() < 16 {
 515              let mut small_vec = [ C_i32(0), ..16 ];
 516              for (small_vec_e, &ix) in small_vec.mut_iter().zip(ixs.iter()) {
 517                  *small_vec_e = C_i32(ix as i32);
 518              }
 519              self.inbounds_gep(base, small_vec.slice(0, ixs.len()))
 520          } else {
 521              let v = do ixs.iter().map |i| { C_i32(*i as i32) }.collect::<~[ValueRef]>();
 522              self.count_insn("gepi");
 523              self.inbounds_gep(base, v)
 524          }
 525      }

  76  pub fn llvm_calling_convention(ccx: &mut CrateContext,
  77                                 abis: AbiSet) -> Option<CallConv> {
  78      let arch = ccx.sess.targ_cfg.arch;
  79      abis.for_arch(arch).map(|abi| {
  80          match *abi {
  81              RustIntrinsic => {
  82                  // Intrinsics are emitted by monomorphic fn
  83                  ccx.sess.bug(fmt!("Asked to register intrinsic fn"));
  84              }
  85  
  86              Rust => {
  87                  // FIXME(#3678) Implement linking to foreign fns with Rust ABI
  88                  ccx.sess.unimpl(
  89                      fmt!("Foreign functions with Rust ABI"));
  90              }
  91  
  92              Stdcall => lib::llvm::X86StdcallCallConv,
  93              Fastcall => lib::llvm::X86FastcallCallConv,
  94              C => lib::llvm::CCallConv,
  95  
  96              // NOTE These API constants ought to be more specific
  97              Cdecl => lib::llvm::CCallConv,
  98              Aapcs => lib::llvm::CCallConv,
  99          }
 100      })
 101  }

 104  pub fn register_foreign_item_fn(ccx: @mut CrateContext,
 105                                  abis: AbiSet,
 106                                  path: &ast_map::path,
 107                                  foreign_item: @ast::foreign_item) -> ValueRef {
 108      /*!
 109       * Registers a foreign function found in a library.
 110       * Just adds a LLVM global.
 111       */
 112  
 113      debug!("register_foreign_item_fn(abis=%s, \
 114              path=%s, \
 115              foreign_item.id=%?)",
 116             abis.repr(ccx.tcx),
 117             path.repr(ccx.tcx),
 118             foreign_item.id);
 119  
 120      let cc = match llvm_calling_convention(ccx, abis) {
 121          Some(cc) => cc,
 122          None => {
 123              // FIXME(#8357) We really ought to report a span here
 124              ccx.sess.fatal(
 125                  fmt!("ABI `%s` has no suitable ABI \
 126                        for target architecture \
 127                        in module %s",
 128                       abis.user_string(ccx.tcx),
 129                       ast_map::path_to_str(*path,
 130                                            ccx.sess.intr())));
 131          }
 132      };
 133  
 134      // Register the function as a C extern fn
 135      let lname = link_name(ccx, foreign_item);
 136      let tys = foreign_types_for_id(ccx, foreign_item.id);
 137  
 138      // Create the LLVM value for the C extern fn
 139      let llfn_ty = lltype_for_fn_from_foreign_types(&tys);
 140      let llfn = base::get_extern_fn(&mut ccx.externs, ccx.llmod,
 141                                     lname, cc, llfn_ty);
 142      add_argument_attributes(&tys, llfn);
 143  
 144      return llfn;
 145  }

 147  pub fn trans_native_call(bcx: @mut Block,
 148                           callee_ty: ty::t,
 149                           llfn: ValueRef,
 150                           llretptr: ValueRef,
 151                           llargs_rust: &[ValueRef]) -> @mut Block {
 152      /*!
 153       * Prepares a call to a native function. This requires adapting
 154       * from the Rust argument passing rules to the native rules.
 155       *
 156       * # Parameters
 157       *
 158       * - `callee_ty`: Rust type for the function we are calling
 159       * - `llfn`: the function pointer we are calling
 160       * - `llretptr`: where to store the return value of the function
 161       * - `llargs_rust`: a list of the argument values, prepared
 162       *   as they would be if calling a Rust function
 163       */
 164  
 165      let ccx = bcx.ccx();
 166      let tcx = bcx.tcx();
 167  
 168      debug!("trans_native_call(callee_ty=%s, \
 169              llfn=%s, \
 170              llretptr=%s)",
 171             callee_ty.repr(tcx),
 172             ccx.tn.val_to_str(llfn),
 173             ccx.tn.val_to_str(llretptr));
 174  
 175      let (fn_abis, fn_sig) = match ty::get(callee_ty).sty {
 176          ty::ty_bare_fn(ref fn_ty) => (fn_ty.abis, fn_ty.sig.clone()),
 177          _ => ccx.sess.bug("trans_native_call called on non-function type")
 178      };
 179      let llsig = foreign_signature(ccx, &fn_sig);
 180      let ret_def = !ty::type_is_voidish(fn_sig.output);
 181      let fn_type = cabi::compute_abi_info(ccx,
 182                                           llsig.llarg_tys,
 183                                           llsig.llret_ty,
 184                                           ret_def);
 185  
 186      let all_arg_tys: &[cabi::LLVMType] = fn_type.arg_tys;
 187      let all_attributes: &[Option<Attribute>] = fn_type.attrs;
 188  
 189      let mut llargs_foreign = ~[];
 190  
 191      // If the foreign ABI expects return value by pointer, supply the
 192      // pointer that Rust gave us. Sometimes we have to bitcast
 193      // because foreign fns return slightly different (but equivalent)
 194      // views on the same type (e.g., i64 in place of {i32,i32}).
 195      let (arg_tys, attributes) = {
 196          if fn_type.sret {
 197              if all_arg_tys[0].cast {
 198                  let llcastedretptr =
 199                      BitCast(bcx, llretptr, all_arg_tys[0].ty.ptr_to());
 200                  llargs_foreign.push(llcastedretptr);
 201              } else {
 202                  llargs_foreign.push(llretptr);
 203              }
 204              (all_arg_tys.tail(), all_attributes.tail())
 205          } else {
 206              (all_arg_tys, all_attributes)
 207          }
 208      };
 209  
 210      for (i, &llarg_rust) in llargs_rust.iter().enumerate() {
 211          let mut llarg_rust = llarg_rust;
 212  
 213          // Does Rust pass this argument by pointer?
 214          let rust_indirect = type_of::arg_is_indirect(ccx, fn_sig.inputs[i]);
 215  
 216          debug!("argument %u, llarg_rust=%s, rust_indirect=%b, arg_ty=%s",
 217                 i,
 218                 ccx.tn.val_to_str(llarg_rust),
 219                 rust_indirect,
 220                 ccx.tn.type_to_str(arg_tys[i].ty));
 221  
 222          // Ensure that we always have the Rust value indirectly,
 223          // because it makes bitcasting easier.
 224          if !rust_indirect {
 225              let scratch = base::alloca(bcx, arg_tys[i].ty, "__arg");
 226              Store(bcx, llarg_rust, scratch);
 227              llarg_rust = scratch;
 228          }
 229  
 230          debug!("llarg_rust=%s (after indirection)",
 231                 ccx.tn.val_to_str(llarg_rust));
 232  
 233          // Check whether we need to do any casting
 234          let foreignarg_ty = arg_tys[i].ty;
 235          if arg_tys[i].cast {
 236              llarg_rust = BitCast(bcx, llarg_rust, foreignarg_ty.ptr_to());
 237          }
 238  
 239          debug!("llarg_rust=%s (after casting)",
 240                 ccx.tn.val_to_str(llarg_rust));
 241  
 242          // Finally, load the value if needed for the foreign ABI
 243          let foreign_indirect = attributes[i].is_some();
 244          let llarg_foreign = if foreign_indirect {
 245              llarg_rust
 246          } else {
 247              Load(bcx, llarg_rust)
 248          };
 249  
 250          debug!("argument %u, llarg_foreign=%s",
 251                 i, ccx.tn.val_to_str(llarg_foreign));
 252  
 253          llargs_foreign.push(llarg_foreign);
 254      }
 255  
 256      let cc = match llvm_calling_convention(ccx, fn_abis) {
 257          Some(cc) => cc,
 258          None => {
 259              // FIXME(#8357) We really ought to report a span here
 260              ccx.sess.fatal(
 261                  fmt!("ABI string `%s` has no suitable ABI \
 262                        for target architecture",
 263                       fn_abis.user_string(ccx.tcx)));
 264          }
 265      };
 266  
 267      // A function pointer is called without the declaration available, so we have to apply
 268      // any attributes with ABI implications directly to the call instruction. Right now, the
 269      // only attribute we need to worry about is `sret`.
 270      let attrs;
 271      if fn_type.sret {
 272          attrs = &[(1, StructRetAttribute)];
 273      } else {
 274          attrs = &[];
 275      }
 276      let llforeign_retval = CallWithConv(bcx, llfn, llargs_foreign, cc, attrs);
 277  
 278      // If the function we just called does not use an outpointer,
 279      // store the result into the rust outpointer. Cast the outpointer
 280      // type to match because some ABIs will use a different type than
 281      // the Rust type. e.g., a {u32,u32} struct could be returned as
 282      // u64.
 283      if ret_def && !fn_type.sret {
 284          let llrust_ret_ty = llsig.llret_ty;
 285          let llforeign_ret_ty = fn_type.ret_ty.ty;
 286  
 287          debug!("llretptr=%s", ccx.tn.val_to_str(llretptr));
 288          debug!("llforeign_retval=%s", ccx.tn.val_to_str(llforeign_retval));
 289          debug!("llrust_ret_ty=%s", ccx.tn.type_to_str(llrust_ret_ty));
 290          debug!("llforeign_ret_ty=%s", ccx.tn.type_to_str(llforeign_ret_ty));
 291  
 292          if llrust_ret_ty == llforeign_ret_ty {
 293              Store(bcx, llforeign_retval, llretptr);
 294          } else {
 295              // The actual return type is a struct, but the ABI
 296              // adaptation code has cast it into some scalar type.  The
 297              // code that follows is the only reliable way I have
 298              // found to do a transform like i64 -> {i32,i32}.
 299              // Basically we dump the data onto the stack then memcpy it.
 300              //
 301              // Other approaches I tried:
 302              // - Casting rust ret pointer to the foreign type and using Store
 303              //   is (a) unsafe if size of foreign type > size of rust type and
 304              //   (b) runs afoul of strict aliasing rules, yielding invalid
 305              //   assembly under -O (specifically, the store gets removed).
 306              // - Truncating foreign type to correct integral type and then
 307              //   bitcasting to the struct type yields invalid cast errors.
 308              let llscratch = base::alloca(bcx, llforeign_ret_ty, "__cast");
 309              Store(bcx, llforeign_retval, llscratch);
 310              let llscratch_i8 = BitCast(bcx, llscratch, Type::i8().ptr_to());
 311              let llretptr_i8 = BitCast(bcx, llretptr, Type::i8().ptr_to());
 312              let llrust_size = machine::llsize_of_store(ccx, llrust_ret_ty);
 313              let llforeign_align = machine::llalign_of_min(ccx, llforeign_ret_ty);
 314              let llrust_align = machine::llalign_of_min(ccx, llrust_ret_ty);
 315              let llalign = uint::min(llforeign_align, llrust_align);
 316              debug!("llrust_size=%?", llrust_size);
 317              base::call_memcpy(bcx, llretptr_i8, llscratch_i8,
 318                                C_uint(ccx, llrust_size), llalign as u32);
 319          }
 320      }
 321  
 322      return bcx;
 323  }

 334  ///////////////////////////////////////////////////////////////////////////
 335  // Rust functions with foreign ABIs
 336  //
 337  // These are normal Rust functions defined with foreign ABIs.  For
 338  // now, and perhaps forever, we translate these using a "layer of
 339  // indirection". That is, given a Rust declaration like:
 340  //
 341  //     extern "C" fn foo(i: u32) -> u32 { ... }
 342  //
 343  // we will generate a function like:
 344  //
 345  //     S foo(T i) {
 346  //         S r;
 347  //         foo0(&r, NULL, i);
 348  //         return r;
 349  //     }
 350  //
 351  //     #[inline_always]
 352  //     void foo0(uint32_t *r, void *env, uint32_t i) { ... }
 353  //
 354  // Here the (internal) `foo0` function follows the Rust ABI as normal,
 355  // where the `foo` function follows the C ABI. We rely on LLVM to
 356  // inline the one into the other. Of course we could just generate the
 357  // correct code in the first place, but this is much simpler.
 358  
 359  pub fn register_rust_fn_with_foreign_abi(ccx: @mut CrateContext,
 360                                           sp: Span,
 361                                           sym: ~str,
 362                                           node_id: ast::NodeId)
 363                                           -> ValueRef {
 364      let _icx = push_ctxt("foreign::register_foreign_fn");
 365  
 366      let tys = foreign_types_for_id(ccx, node_id);
 367      let llfn_ty = lltype_for_fn_from_foreign_types(&tys);
 368      let llfn = base::register_fn_llvmty(ccx,
 369                                          sp,
 370                                          sym,
 371                                          node_id,
 372                                          lib::llvm::CCallConv,
 373                                          llfn_ty);
 374      add_argument_attributes(&tys, llfn);
 375      debug!("register_rust_fn_with_foreign_abi(node_id=%?, llfn_ty=%s, llfn=%s)",
 376             node_id, ccx.tn.type_to_str(llfn_ty), ccx.tn.val_to_str(llfn));
 377      llfn
 378  }

 380  pub fn trans_rust_fn_with_foreign_abi(ccx: @mut CrateContext,
 381                                        path: &ast_map::path,
 382                                        decl: &ast::fn_decl,
 383                                        body: &ast::Block,
 384                                        llwrapfn: ValueRef,
 385                                        id: ast::NodeId) {
 386      let _icx = push_ctxt("foreign::build_foreign_fn");
 387      let tys = foreign_types_for_id(ccx, id);
 388  
 389      unsafe { // unsafe because we call LLVM operations
 390          // Build up the Rust function (`foo0` above).
 391          let llrustfn = build_rust_fn(ccx, path, decl, body, id);
 392  
 393          // Build up the foreign wrapper (`foo` above).
 394          return build_wrap_fn(ccx, llrustfn, llwrapfn, &tys);
 395      }

 397      fn build_rust_fn(ccx: @mut CrateContext,
 398                       path: &ast_map::path,
 399                       decl: &ast::fn_decl,
 400                       body: &ast::Block,
 401                       id: ast::NodeId)
 402                       -> ValueRef {
 403          let _icx = push_ctxt("foreign::foreign::build_rust_fn");
 404          let tcx = ccx.tcx;
 405          let t = ty::node_id_to_type(tcx, id);
 406          let ps = link::mangle_internal_name_by_path(
 407              ccx, vec::append_one((*path).clone(), ast_map::path_name(
 408                  special_idents::clownshoe_abi
 409              )));
 410  
 411          // Compute the type that the function would have if it were just a
 412          // normal Rust function. This will be the type of the wrappee fn.
 413          let f = match ty::get(t).sty {
 414              ty::ty_bare_fn(ref f) => {

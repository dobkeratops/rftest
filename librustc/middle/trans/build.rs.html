  42 // The difference between a block being unreachable and being terminated is
  43 // somewhat obscure, and has to do with error checking. When a block is
  44 // terminated, we're saying that trying to add any further statements in the
  45 // block is an error. On the other hand, if something is unreachable, that
  46 // means that the block was terminated in some way that we don't want to check
  47 // for (fail/break/return statements, call to diverging functions, etc), and
  48 // further instructions to the block should simply be ignored.
  49 
  50 pub fn RetVoid(cx: @mut Block) {

  52     check_not_terminated(cx);

  54     B(cx).ret_void();
  55 }

 108 pub fn Invoke(cx: @mut Block,
 109               Fn: ValueRef,
 110               Args: &[ValueRef],
 111               Then: BasicBlockRef,
 112               Catch: BasicBlockRef)
 113            -> ValueRef {
 114     if cx.unreachable {
 115         return C_null(Type::i8());
 116     }
 117     check_not_terminated(cx);
 118     terminate(cx, "Invoke");
 119     debug!("Invoke(%s with arguments (%s))",
 120            cx.val_to_str(Fn),
 121            Args.map(|a| cx.val_to_str(*a)).connect(", "));
 122     B(cx).invoke(Fn, Args, Then, Catch)
 123 }

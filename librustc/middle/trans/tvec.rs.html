  86  pub fn trans_fixed_vstore<'a>(
  87                            bcx: &'a Block<'a>,
  88                            vstore_expr: &ast::Expr,
  89                            content_expr: &ast::Expr,
  90                            dest: expr::Dest)
  91                            -> &'a Block<'a> {
  92      //!
  93      //
  94      // [...] allocates a fixed-size array and moves it around "by value".
  95      // In this case, it means that the caller has already given us a location
  96      // to store the array of the suitable size, so all we have to do is
  97      // generate the content.
  98  
  99      debug!("trans_fixed_vstore(vstore_expr={}, dest={:?})",
 100             bcx.expr_to_str(vstore_expr), dest.to_str(bcx.ccx()));
 101  
 102      let vt = vec_types_from_expr(bcx, vstore_expr);
 103  
 104      return match dest {
 105          Ignore => write_content(bcx, &vt, vstore_expr, content_expr, dest),
 106          SaveIn(lldest) => {
 107              // lldest will have type *[T x N], but we want the type *T,
 108              // so use GEP to convert:
 109              let lldest = GEPi(bcx, lldest, [0, 0]);
 110              write_content(bcx, &vt, vstore_expr, content_expr, SaveIn(lldest))
 111          }
 112      };
 113  }

 115  pub fn trans_slice_vstore<'a>(
 116                            bcx: &'a Block<'a>,
 117                            vstore_expr: &ast::Expr,
 118                            content_expr: &ast::Expr,
 119                            dest: expr::Dest)
 120                            -> &'a Block<'a> {
 121      /*!
 122       * &[...] allocates memory on the stack and writes the values into it,
 123       * returning a slice (pair of ptr, len).  &"..." is similar except that
 124       * the memory can be statically allocated.
 125       */
 126  
 127      let fcx = bcx.fcx;
 128      let ccx = fcx.ccx;
 129      let mut bcx = bcx;
 130  
 131      debug!("trans_slice_vstore(vstore_expr={}, dest={})",
 132             bcx.expr_to_str(vstore_expr), dest.to_str(ccx));
 133  
 134      // Handle the &"..." case:
 135      match content_expr.node {
 136          ast::ExprLit(lit) => {
 137              match lit.node {
 138                  ast::LitStr(ref s, _) => {
 139                      return trans_lit_str(bcx,
 140                                           content_expr,
 141                                           s.clone(),
 142                                           dest)
 143                  }
 144                  _ => {}
 145              }
 146          }
 147          _ => {}
 148      }
 149  
 150      // Handle the &[...] case:
 151      let vt = vec_types_from_expr(bcx, vstore_expr);
 152      let count = elements_required(bcx, content_expr);
 153      debug!("vt={}, count={:?}", vt.to_str(ccx), count);
 154  
 155      let llcount = C_uint(ccx, count);
 156      let llfixed;
 157      if count == 0 {
 158          // Zero-length array: just use NULL as the data pointer
 159          llfixed = C_null(vt.llunit_ty.ptr_to());
 160      } else {
 161          // Make a fixed-length backing array and allocate it on the stack.
 162          llfixed = base::arrayalloca(bcx, vt.llunit_ty, llcount);
 163  
 164          // Arrange for the backing array to be cleaned up.
 165          let fixed_ty = ty::mk_vec(bcx.tcx(), vt.unit_ty,
 166                                    ty::VstoreFixed(count));
 167          let llfixed_ty = type_of::type_of(bcx.ccx(), fixed_ty).ptr_to();
 168          let llfixed_casted = BitCast(bcx, llfixed, llfixed_ty);
 169          let cleanup_scope = cleanup::temporary_scope(bcx.tcx(), content_expr.id);
 170          fcx.schedule_drop_mem(cleanup_scope, llfixed_casted, fixed_ty);
 171  
 172          // Generate the content into the backing array.
 173          bcx = write_content(bcx, &vt, vstore_expr,
 174                              content_expr, SaveIn(llfixed));
 175      }
 176  
 177      // Finally, create the slice pair itself.
 178      match dest {
 179          Ignore => {}
 180          SaveIn(lldest) => {
 181              Store(bcx, llfixed, GEPi(bcx, lldest, [0u, abi::slice_elt_base]));
 182              Store(bcx, llcount, GEPi(bcx, lldest, [0u, abi::slice_elt_len]));
 183          }
 184      }
 185  
 186      return bcx;
 187  }

 189  pub fn trans_lit_str<'a>(
 190                       bcx: &'a Block<'a>,
 191                       lit_expr: &ast::Expr,
 192                       str_lit: InternedString,
 193                       dest: Dest)
 194                       -> &'a Block<'a> {
 195      /*!
 196       * Literal strings translate to slices into static memory.  This is
 197       * different from trans_slice_vstore() above because it does need to copy
 198       * the content anywhere.
 199       */
 200  
 201      debug!("trans_lit_str(lit_expr={}, dest={})",
 202             bcx.expr_to_str(lit_expr),
 203             dest.to_str(bcx.ccx()));
 204  
 205      match dest {
 206          Ignore => bcx,
 207          SaveIn(lldest) => {
 208              unsafe {
 209                  let bytes = str_lit.get().len();
 210                  let llbytes = C_uint(bcx.ccx(), bytes);
 211                  let llcstr = C_cstr(bcx.ccx(), str_lit, false);
 212                  let llcstr = llvm::LLVMConstPointerCast(llcstr, Type::i8p(bcx.ccx()).to_ref());
 213                  Store(bcx, llcstr,
 214                        GEPi(bcx, lldest, [0u, abi::slice_elt_base]));
 215                  Store(bcx, llbytes,
 216                        GEPi(bcx, lldest, [0u, abi::slice_elt_len]));
 217                  bcx
 218              }
 219          }
 220      }
 221  }

 224  pub fn trans_uniq_vstore<'a>(bcx: &'a Block<'a>,
 225                               vstore_expr: &ast::Expr,
 226                               content_expr: &ast::Expr)
 227                               -> DatumBlock<'a, Expr> {
 228      /*!
 229       * ~[...] and "...".to_owned() allocate boxes in the exchange heap and write
 230       * the array elements into them.
 231       */
 232  
 233      debug!("trans_uniq_vstore(vstore_expr={})", bcx.expr_to_str(vstore_expr));
 234      let fcx = bcx.fcx;
 235      let ccx = fcx.ccx;
 236  
 237      // Handle "".to_owned().
 238      match content_expr.node {
 239          ast::ExprLit(lit) => {
 240              match lit.node {
 241                  ast::LitStr(ref s, _) => {
 242                      let llptrval = C_cstr(ccx, (*s).clone(), false);
 243                      let llptrval = PointerCast(bcx, llptrval, Type::i8p(ccx));
 244                      let llsizeval = C_uint(ccx, s.get().len());
 245                      let typ = ty::mk_str(bcx.tcx(), ty::VstoreUniq);
 246                      let lldestval = rvalue_scratch_datum(bcx,
 247                                                           typ,
 248                                                           "");
 249                      let alloc_fn = langcall(bcx,
 250                                              Some(lit.span),
 251                                              "",
 252                                              StrDupUniqFnLangItem);
 253                      let bcx = callee::trans_lang_call(
 254                          bcx,
 255                          alloc_fn,
 256                          [ llptrval, llsizeval ],
 257                          Some(expr::SaveIn(lldestval.val))).bcx;
 258                      return DatumBlock(bcx, lldestval).to_expr_datumblock();
 259                  }
 260                  _ => {}
 261              }
 262          }
 263          _ => {}
 264      }
 265  
 266      let vec_ty = node_id_type(bcx, vstore_expr.id);
 267      let vt = vec_types(bcx, ty::sequence_element_type(bcx.tcx(), vec_ty));
 268      let count = elements_required(bcx, content_expr);
 269  
 270      let llunitty = type_of::type_of(ccx, vt.unit_ty);
 271      let unit_sz = nonzero_llsize_of(ccx, llunitty);
 272  
 273      let fill = Mul(bcx, C_uint(ccx, count), unit_sz);
 274      let alloc = if count < 4u { Mul(bcx, C_int(ccx, 4), unit_sz) }
 275      else { fill };
 276  
 277      let vecsize = Add(bcx, alloc, llsize_of(ccx, ccx.opaque_vec_type));
 278  
 279      let Result { bcx: bcx, val: val } = malloc_raw_dyn(bcx, vec_ty, vecsize);
 280      Store(bcx, fill, GEPi(bcx, val, [0u, abi::vec_elt_fill]));
 281      Store(bcx, alloc, GEPi(bcx, val, [0u, abi::vec_elt_alloc]));
 282  
 283      // Create a temporary scope lest execution should fail while
 284      // constructing the vector.
 285      let temp_scope = fcx.push_custom_cleanup_scope();
 286      fcx.schedule_free_value(cleanup::CustomScope(temp_scope),
 287                              val, cleanup::HeapExchange);
 288  
 289      let dataptr = get_dataptr(bcx, val);
 290  
 291      debug!("alloc_uniq_vec() returned val={}, dataptr={}",
 292             bcx.val_to_str(val), bcx.val_to_str(dataptr));
 293  
 294      let bcx = write_content(bcx, &vt, vstore_expr,
 295                              content_expr, SaveIn(dataptr));
 296  
 297      fcx.pop_custom_cleanup_scope(temp_scope);
 298  
 299      immediate_rvalue_bcx(bcx, val, vec_ty).to_expr_datumblock()
 300  }

 302  pub fn write_content<'a>(
 303                       bcx: &'a Block<'a>,
 304                       vt: &VecTypes,
 305                       vstore_expr: &ast::Expr,
 306                       content_expr: &ast::Expr,
 307                       dest: Dest)
 308                       -> &'a Block<'a> {
 309      let _icx = push_ctxt("tvec::write_content");
 310      let fcx = bcx.fcx;
 311      let mut bcx = bcx;
 312  
 313      debug!("write_content(vt={}, dest={}, vstore_expr={:?})",
 314             vt.to_str(bcx.ccx()),
 315             dest.to_str(bcx.ccx()),
 316             bcx.expr_to_str(vstore_expr));
 317  
 318      match content_expr.node {
 319          ast::ExprLit(lit) => {
 320              match lit.node {
 321                  ast::LitStr(ref s, _) => {
 322                      match dest {
 323                          Ignore => return bcx,
 324                          SaveIn(lldest) => {
 325                              let bytes = s.get().len();
 326                              let llbytes = C_uint(bcx.ccx(), bytes);
 327                              let llcstr = C_cstr(bcx.ccx(), (*s).clone(), false);
 328                              base::call_memcpy(bcx,
 329                                                lldest,
 330                                                llcstr,
 331                                                llbytes,
 332                                                1);
 333                              return bcx;
 334                          }
 335                      }
 336                  }
 337                  _ => {
 338                      bcx.tcx().sess.span_bug(content_expr.span,
 339                                              "unexpected evec content");
 340                  }
 341              }
 342          }
 343          ast::ExprVec(ref elements) => {
 344              match dest {
 345                  Ignore => {
 346                      for element in elements.iter() {
 347                          bcx = expr::trans_into(bcx, *element, Ignore);
 348                      }
 349                  }
 350  
 351                  SaveIn(lldest) => {
 352                      let temp_scope = fcx.push_custom_cleanup_scope();
 353                      for (i, element) in elements.iter().enumerate() {
 354                          let lleltptr = GEPi(bcx, lldest, [i]);
 355                          debug!("writing index {:?} with lleltptr={:?}",
 356                                 i, bcx.val_to_str(lleltptr));
 357                          bcx = expr::trans_into(bcx, *element,
 358                                                 SaveIn(lleltptr));
 359                          fcx.schedule_drop_mem(
 360                              cleanup::CustomScope(temp_scope),
 361                              lleltptr,
 362                              vt.unit_ty);
 363                      }
 364                      fcx.pop_custom_cleanup_scope(temp_scope);
 365                  }
 366              }
 367              return bcx;
 368          }
 369          ast::ExprRepeat(element, count_expr) => {
 370              match dest {
 371                  Ignore => {
 372                      return expr::trans_into(bcx, element, Ignore);
 373                  }
 374                  SaveIn(lldest) => {
 375                      let count = ty::eval_repeat_count(bcx.tcx(), count_expr);
 376                      if count == 0 {
 377                          return bcx;
 378                      }
 379  
 380                      // Some cleanup would be required in the case in which failure happens
 381                      // during a copy. But given that copy constructors are not overridable,
 382                      // this can only happen as a result of OOM. So we just skip out on the
 383                      // cleanup since things would *probably* be broken at that point anyways.
 384  
 385                      let elem = unpack_datum!(bcx, expr::trans(bcx, element));
 386                      assert!(!ty::type_moves_by_default(bcx.tcx(), elem.ty));
 387  
 388                      let bcx = iter_vec_loop(bcx, lldest, vt,
 389                                    C_uint(bcx.ccx(), count), |set_bcx, lleltptr, _| {
 390                          elem.shallow_copy_and_take(set_bcx, lleltptr)
 391                      });
 392  
 393                      elem.add_clean_if_rvalue(bcx, element.id);
 394                      bcx
 395                  }
 396              }
 397          }
 398          _ => {
 399              bcx.tcx().sess.span_bug(content_expr.span,
 400                                      "unexpected vec content");
 401          }
 402      }
 403  }

  37  // Boxed vector types are in some sense currently a "shorthand" for a box
  38  // containing an unboxed vector. This expands a boxed vector type into such an
  39  // expanded type. It doesn't respect mutability, but that doesn't matter at
  40  // this point.
  41  pub fn expand_boxed_vec_ty(tcx: &ty::ctxt, t: ty::t) -> ty::t {
  42      let unit_ty = ty::sequence_element_type(tcx, t);
  43      let unboxed_vec_ty = ty::mk_mut_unboxed_vec(tcx, unit_ty);
  44      match ty::get(t).sty {
  45          ty::ty_str(ty::vstore_uniq) | ty::ty_vec(_, ty::vstore_uniq) => {
  46              ty::mk_uniq(tcx, unboxed_vec_ty)
  47          }
  48          _ => tcx.sess.bug("non boxed-vec type \
  49                             in tvec::expand_boxed_vec_ty")
  50      }
  51  }

  78  pub fn alloc_raw<'a>(
  79                   bcx: &'a Block<'a>,
  80                   unit_ty: ty::t,
  81                   fill: ValueRef,
  82                   alloc: ValueRef,
  83                   heap: heap)
  84                   -> Result<'a> {
  85      let _icx = push_ctxt("tvec::alloc_uniq");
  86      let ccx = bcx.ccx();
  87  
  88      let vecbodyty = ty::mk_mut_unboxed_vec(bcx.tcx(), unit_ty);
  89      let vecsize = Add(bcx, alloc, llsize_of(ccx, ccx.opaque_vec_type));
  90  
  91      if heap == heap_exchange {
  92          let Result { bcx: bcx, val: val } = malloc_raw_dyn(bcx, vecbodyty, heap_exchange, vecsize);
  93          Store(bcx, fill, GEPi(bcx, val, [0u, abi::vec_elt_fill]));
  94          Store(bcx, alloc, GEPi(bcx, val, [0u, abi::vec_elt_alloc]));
  95          return rslt(bcx, val);
  96      } else {
  97          let base::MallocResult {bcx, smart_ptr: bx, body} =
  98              base::malloc_general_dyn(bcx, vecbodyty, heap, vecsize);
  99          Store(bcx, fill, GEPi(bcx, body, [0u, abi::vec_elt_fill]));
 100          Store(bcx, alloc, GEPi(bcx, body, [0u, abi::vec_elt_alloc]));
 101          return rslt(bcx, bx);
 102      }
 103  }

 114  pub fn alloc_uniq_vec<'a>(
 115                   bcx: &'a Block<'a>,
 116                   unit_ty: ty::t,
 117                   elts: uint)
 118                   -> Result<'a> {
 119      let _icx = push_ctxt("tvec::alloc_uniq");
 120      let ccx = bcx.ccx();
 121      let llunitty = type_of::type_of(ccx, unit_ty);
 122      let unit_sz = nonzero_llsize_of(ccx, llunitty);
 123  
 124      let fill = Mul(bcx, C_uint(ccx, elts), unit_sz);
 125      let alloc = if elts < 4u { Mul(bcx, C_int(ccx, 4), unit_sz) }
 126                  else { fill };
 127      let Result {bcx: bcx, val: vptr} =
 128          alloc_raw(bcx, unit_ty, fill, alloc, heap_exchange);
 129      return rslt(bcx, vptr);
 130  }

 165  pub fn trans_fixed_vstore<'a>(
 166                            bcx: &'a Block<'a>,
 167                            vstore_expr: &ast::Expr,
 168                            content_expr: &ast::Expr,
 169                            dest: expr::Dest)
 170                            -> &'a Block<'a> {
 171      //!
 172      //
 173      // [...] allocates a fixed-size array and moves it around "by value".
 174      // In this case, it means that the caller has already given us a location
 175      // to store the array of the suitable size, so all we have to do is
 176      // generate the content.
 177  
 178      debug!("trans_fixed_vstore(vstore_expr={}, dest={:?})",
 179             bcx.expr_to_str(vstore_expr), dest.to_str(bcx.ccx()));
 180  
 181      let vt = vec_types_from_expr(bcx, vstore_expr);
 182  
 183      return match dest {
 184          Ignore => write_content(bcx, &vt, vstore_expr, content_expr, dest),
 185          SaveIn(lldest) => {
 186              // lldest will have type *[T x N], but we want the type *T,
 187              // so use GEP to convert:
 188              let lldest = GEPi(bcx, lldest, [0, 0]);
 189              write_content(bcx, &vt, vstore_expr, content_expr, SaveIn(lldest))
 190          }
 191      };
 192  }

 194  pub fn trans_slice_vstore<'a>(
 195                            bcx: &'a Block<'a>,
 196                            vstore_expr: &ast::Expr,
 197                            content_expr: &ast::Expr,
 198                            dest: expr::Dest)
 199                            -> &'a Block<'a> {
 200      /*!
 201       * &[...] allocates memory on the stack and writes the values into it,
 202       * returning a slice (pair of ptr, len).  &"..." is similar except that
 203       * the memory can be statically allocated.
 204       */
 205  
 206      let fcx = bcx.fcx;
 207      let ccx = fcx.ccx;
 208      let mut bcx = bcx;
 209  
 210      debug!("trans_slice_vstore(vstore_expr={}, dest={})",
 211             bcx.expr_to_str(vstore_expr), dest.to_str(ccx));
 212  
 213      // Handle the &"..." case:
 214      match content_expr.node {
 215          ast::ExprLit(lit) => {
 216              match lit.node {
 217                  ast::LitStr(ref s, _) => {
 218                      return trans_lit_str(bcx,
 219                                           content_expr,
 220                                           s.clone(),
 221                                           dest)
 222                  }
 223                  _ => {}
 224              }
 225          }
 226          _ => {}
 227      }
 228  
 229      // Handle the &[...] case:
 230      let vt = vec_types_from_expr(bcx, vstore_expr);
 231      let count = elements_required(bcx, content_expr);
 232      debug!("vt={}, count={:?}", vt.to_str(ccx), count);
 233  
 234      let llcount = C_uint(ccx, count);
 235      let llfixed;
 236      if count == 0 {
 237          // Zero-length array: just use NULL as the data pointer
 238          llfixed = C_null(vt.llunit_ty.ptr_to());
 239      } else {
 240          // Make a fixed-length backing array and allocate it on the stack.
 241          llfixed = base::arrayalloca(bcx, vt.llunit_ty, llcount);
 242  
 243          // Arrange for the backing array to be cleaned up.
 244          let fixed_ty = ty::mk_vec(bcx.tcx(),
 245                                    ty::mt {ty: vt.unit_ty,
 246                                            mutbl: ast::MutMutable},
 247                                    ty::vstore_fixed(count));
 248          let llfixed_ty = type_of::type_of(bcx.ccx(), fixed_ty).ptr_to();
 249          let llfixed_casted = BitCast(bcx, llfixed, llfixed_ty);
 250          let cleanup_scope = cleanup::temporary_scope(bcx.tcx(), content_expr.id);
 251          fcx.schedule_drop_mem(cleanup_scope, llfixed_casted, fixed_ty);
 252  
 253          // Generate the content into the backing array.
 254          bcx = write_content(bcx, &vt, vstore_expr,
 255                              content_expr, SaveIn(llfixed));
 256      }
 257  
 258      // Finally, create the slice pair itself.
 259      match dest {
 260          Ignore => {}
 261          SaveIn(lldest) => {
 262              Store(bcx, llfixed, GEPi(bcx, lldest, [0u, abi::slice_elt_base]));
 263              Store(bcx, llcount, GEPi(bcx, lldest, [0u, abi::slice_elt_len]));
 264          }
 265      }
 266  
 267      return bcx;
 268  }

 270  pub fn trans_lit_str<'a>(
 271                       bcx: &'a Block<'a>,
 272                       lit_expr: &ast::Expr,
 273                       str_lit: InternedString,
 274                       dest: Dest)
 275                       -> &'a Block<'a> {
 276      /*!
 277       * Literal strings translate to slices into static memory.  This is
 278       * different from trans_slice_vstore() above because it does need to copy
 279       * the content anywhere.
 280       */
 281  
 282      debug!("trans_lit_str(lit_expr={}, dest={})",
 283             bcx.expr_to_str(lit_expr),
 284             dest.to_str(bcx.ccx()));
 285  
 286      match dest {
 287          Ignore => bcx,
 288          SaveIn(lldest) => {
 289              unsafe {
 290                  let bytes = str_lit.get().len();
 291                  let llbytes = C_uint(bcx.ccx(), bytes);
 292                  let llcstr = C_cstr(bcx.ccx(), str_lit);
 293                  let llcstr = llvm::LLVMConstPointerCast(llcstr, Type::i8p(bcx.ccx()).to_ref());
 294                  Store(bcx, llcstr,
 295                        GEPi(bcx, lldest, [0u, abi::slice_elt_base]));
 296                  Store(bcx, llbytes,
 297                        GEPi(bcx, lldest, [0u, abi::slice_elt_len]));
 298                  bcx
 299              }
 300          }
 301      }
 302  }

 305  pub fn trans_uniq_vstore<'a>(bcx: &'a Block<'a>,
 306                               vstore_expr: &ast::Expr,
 307                               content_expr: &ast::Expr)
 308                               -> DatumBlock<'a, Expr> {
 309      /*!
 310       * ~[...] and ~"..." allocate boxes in the exchange heap and write
 311       * the array elements into them.
 312       */
 313  
 314      debug!("trans_uniq_vstore(vstore_expr={})", bcx.expr_to_str(vstore_expr));
 315      let fcx = bcx.fcx;
 316  
 317      // Handle ~"".
 318      match content_expr.node {
 319          ast::ExprLit(lit) => {
 320              match lit.node {
 321                  ast::LitStr(ref s, _) => {
 322                      let llptrval = C_cstr(bcx.ccx(), (*s).clone());
 323                      let llptrval = PointerCast(bcx,
 324                                                 llptrval,
 325                                                 Type::i8p(bcx.ccx()));
 326                      let llsizeval = C_uint(bcx.ccx(), s.get().len());
 327                      let typ = ty::mk_str(bcx.tcx(), ty::vstore_uniq);
 328                      let lldestval = rvalue_scratch_datum(bcx,
 329                                                           typ,
 330                                                           "");
 331                      let alloc_fn = langcall(bcx,
 332                                              Some(lit.span),
 333                                              "",
 334                                              StrDupUniqFnLangItem);
 335                      let bcx = callee::trans_lang_call(
 336                          bcx,
 337                          alloc_fn,
 338                          [ llptrval, llsizeval ],
 339                          Some(expr::SaveIn(lldestval.val))).bcx;
 340                      return DatumBlock(bcx, lldestval).to_expr_datumblock();
 341                  }
 342                  _ => {}
 343              }
 344          }
 345          _ => {}
 346      }
 347  
 348      let vt = vec_types_from_expr(bcx, vstore_expr);
 349      let count = elements_required(bcx, content_expr);
 350  
 351      let Result {bcx, val} = alloc_uniq_vec(bcx, vt.unit_ty, count);
 352  
 353      // Create a temporary scope lest execution should fail while
 354      // constructing the vector.
 355      let temp_scope = fcx.push_custom_cleanup_scope();
 356      fcx.schedule_free_value(cleanup::CustomScope(temp_scope), val, heap_exchange);
 357  
 358      let dataptr = get_dataptr(bcx, val);
 359  
 360      debug!("alloc_uniq_vec() returned val={}, dataptr={}",
 361             bcx.val_to_str(val), bcx.val_to_str(dataptr));
 362  
 363      let bcx = write_content(bcx, &vt, vstore_expr,
 364                              content_expr, SaveIn(dataptr));
 365  
 366      fcx.pop_custom_cleanup_scope(temp_scope);
 367  
 368      return immediate_rvalue_bcx(bcx, val, vt.vec_ty).to_expr_datumblock();
 369  }

 371  pub fn write_content<'a>(
 372                       bcx: &'a Block<'a>,
 373                       vt: &VecTypes,
 374                       vstore_expr: &ast::Expr,
 375                       content_expr: &ast::Expr,
 376                       dest: Dest)
 377                       -> &'a Block<'a> {
 378      let _icx = push_ctxt("tvec::write_content");
 379      let fcx = bcx.fcx;
 380      let mut bcx = bcx;
 381  
 382      debug!("write_content(vt={}, dest={}, vstore_expr={:?})",
 383             vt.to_str(bcx.ccx()),
 384             dest.to_str(bcx.ccx()),
 385             bcx.expr_to_str(vstore_expr));
 386  
 387      match content_expr.node {
 388          ast::ExprLit(lit) => {
 389              match lit.node {
 390                  ast::LitStr(ref s, _) => {
 391                      match dest {
 392                          Ignore => return bcx,
 393                          SaveIn(lldest) => {
 394                              let bytes = s.get().len();
 395                              let llbytes = C_uint(bcx.ccx(), bytes);
 396                              let llcstr = C_cstr(bcx.ccx(), (*s).clone());
 397                              base::call_memcpy(bcx,
 398                                                lldest,
 399                                                llcstr,
 400                                                llbytes,
 401                                                1);
 402                              return bcx;
 403                          }
 404                      }
 405                  }
 406                  _ => {
 407                      bcx.tcx().sess.span_bug(content_expr.span,
 408                                              "unexpected evec content");
 409                  }
 410              }
 411          }
 412          ast::ExprVec(ref elements, _) => {
 413              match dest {
 414                  Ignore => {
 415                      for element in elements.iter() {
 416                          bcx = expr::trans_into(bcx, *element, Ignore);
 417                      }
 418                  }
 419  
 420                  SaveIn(lldest) => {

 594  pub type iter_vec_block<'r,'b> =
 595      'r |&'b Block<'b>, ValueRef, ty::t| -> &'b Block<'b>;
 596  
 597  pub fn iter_vec_loop<'r,
 598                       'b>(
 599                       bcx: &'b Block<'b>,
 600                       data_ptr: ValueRef,
 601                       vt: &VecTypes,
 602                       count: ValueRef,
 603                       f: iter_vec_block<'r,'b>)
 604                       -> &'b Block<'b> {
 605      let _icx = push_ctxt("tvec::iter_vec_loop");
 606      let fcx = bcx.fcx;
 607  
 608      let next_bcx = fcx.new_temp_block("expr_repeat: while next");
 609      let loop_bcx = fcx.new_temp_block("expr_repeat");
 610      let cond_bcx = fcx.new_temp_block("expr_repeat: loop cond");
 611      let body_bcx = fcx.new_temp_block("expr_repeat: body: set");
 612      let inc_bcx = fcx.new_temp_block("expr_repeat: body: inc");
 613      Br(bcx, loop_bcx.llbb);
 614  
 615      let loop_counter = {
 616          // i = 0
 617          let i = alloca(loop_bcx, bcx.ccx().int_type, "__i");
 618          Store(loop_bcx, C_uint(bcx.ccx(), 0), i);
 619  
 620          Br(loop_bcx, cond_bcx.llbb);
 621          i
 622      };
 623  
 624      { // i < count
 625          let lhs = Load(cond_bcx, loop_counter);
 626          let rhs = count;
 627          let cond_val = ICmp(cond_bcx, lib::llvm::IntULT, lhs, rhs);
 628  
 629          CondBr(cond_bcx, cond_val, body_bcx.llbb, next_bcx.llbb);
 630      }
 631  
 632      { // loop body
 633          let i = Load(body_bcx, loop_counter);
 634          let lleltptr = if vt.llunit_alloc_size == 0 {
 635              data_ptr
 636          } else {
 637              InBoundsGEP(body_bcx, data_ptr, [i])
 638          };
 639          let body_bcx = f(body_bcx, lleltptr, vt.unit_ty);
 640  
 641          Br(body_bcx, inc_bcx.llbb);
 642      }
 643  
 644      { // i += 1
 645          let i = Load(inc_bcx, loop_counter);
 646          let plusone = Add(inc_bcx, i, C_uint(bcx.ccx(), 1));
 647          Store(inc_bcx, plusone, loop_counter);
 648  
 649          Br(inc_bcx, cond_bcx.llbb);
 650      }
 651  
 652      next_bcx
 653  }

 655  pub fn iter_vec_raw<'r,
 656                      'b>(
 657                      bcx: &'b Block<'b>,
 658                      data_ptr: ValueRef,
 659                      vec_ty: ty::t,
 660                      fill: ValueRef,
 661                      f: iter_vec_block<'r,'b>)
 662                      -> &'b Block<'b> {
 663      let _icx = push_ctxt("tvec::iter_vec_raw");

 665  
 666      let vt = vec_types(bcx, vec_ty);
 667      if vt.llunit_alloc_size == 0 {
 668          // Special-case vectors with elements of size 0  so they don't go out of bounds (#9890)
 669          iter_vec_loop(bcx, data_ptr, &vt, fill, f)
 670      } else {
 671          // Calculate the last pointer address we want to handle.
 672          // FIXME (#3729): Optimize this when the size of the unit type is
 673          // statically known to not use pointer casts, which tend to confuse

 679          Br(bcx, header_bcx.llbb);
 680          let data_ptr =
 681              Phi(header_bcx, val_ty(data_ptr), [data_ptr], [bcx.llbb]);
 682          let not_yet_at_end =
 683              ICmp(header_bcx, lib::llvm::IntULT, data_ptr, data_end_ptr);
 684          let body_bcx = fcx.new_temp_block("iter_vec_loop_body");
 685          let next_bcx = fcx.new_temp_block("iter_vec_next");
 686          CondBr(header_bcx, not_yet_at_end, body_bcx.llbb, next_bcx.llbb);
 687          let body_bcx = f(body_bcx, data_ptr, vt.unit_ty);
 688          AddIncomingToPhi(data_ptr, InBoundsGEP(body_bcx, data_ptr,
 689                                                 [C_int(bcx.ccx(), 1)]),
 690                           body_bcx.llbb);
 691          Br(body_bcx, header_bcx.llbb);
 692          next_bcx
 693      }
 694  }

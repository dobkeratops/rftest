  37  // Boxed vector types are in some sense currently a "shorthand" for a box
  38  // containing an unboxed vector. This expands a boxed vector type into such an
  39  // expanded type. It doesn't respect mutability, but that doesn't matter at
  40  // this point.
  41  pub fn expand_boxed_vec_ty(tcx: ty::ctxt, t: ty::t) -> ty::t {
  42      let unit_ty = ty::sequence_element_type(tcx, t);
  43      let unboxed_vec_ty = ty::mk_mut_unboxed_vec(tcx, unit_ty);
  44      match ty::get(t).sty {
  45        ty::ty_estr(ty::vstore_uniq) | ty::ty_evec(_, ty::vstore_uniq) => {
  46          ty::mk_imm_uniq(tcx, unboxed_vec_ty)
  47        }
  48        ty::ty_estr(ty::vstore_box) | ty::ty_evec(_, ty::vstore_box) => {
  49          ty::mk_imm_box(tcx, unboxed_vec_ty)
  50        }
  51        _ => tcx.sess.bug("non boxed-vec type \
  52                           in tvec::expand_boxed_vec_ty")
  53      }
  54  }

  87  pub fn alloc_raw(bcx: @mut Block, unit_ty: ty::t,
  88                   fill: ValueRef, alloc: ValueRef, heap: heap) -> Result {
  89      let _icx = push_ctxt("tvec::alloc_uniq");
  90      let ccx = bcx.ccx();
  91  
  92      let vecbodyty = ty::mk_mut_unboxed_vec(bcx.tcx(), unit_ty);
  93      let vecsize = Add(bcx, alloc, llsize_of(ccx, ccx.opaque_vec_type));
  94  
  95      if heap == heap_exchange {
  96          let Result { bcx: bcx, val: val } = malloc_raw_dyn(bcx, vecbodyty, heap_exchange, vecsize);
  97          Store(bcx, fill, GEPi(bcx, val, [0u, abi::vec_elt_fill]));
  98          Store(bcx, alloc, GEPi(bcx, val, [0u, abi::vec_elt_alloc]));
  99          return rslt(bcx, val);
 100      } else {
 101          let base::MallocResult {bcx, box: bx, body} =
 102              base::malloc_general_dyn(bcx, vecbodyty, heap, vecsize);
 103          Store(bcx, fill, GEPi(bcx, body, [0u, abi::vec_elt_fill]));
 104          Store(bcx, alloc, GEPi(bcx, body, [0u, abi::vec_elt_alloc]));
 105          base::maybe_set_managed_unique_rc(bcx, bx, heap);
 106          return rslt(bcx, bx);
 107      }
 108  }

 115  pub fn alloc_vec(bcx: @mut Block,
 116                   unit_ty: ty::t,
 117                   elts: uint,
 118                   heap: heap)
 119                -> Result {
 120      let _icx = push_ctxt("tvec::alloc_uniq");
 121      let ccx = bcx.ccx();
 122      let llunitty = type_of::type_of(ccx, unit_ty);
 123      let unit_sz = nonzero_llsize_of(ccx, llunitty);
 124  
 125      let fill = Mul(bcx, C_uint(ccx, elts), unit_sz);
 126      let alloc = if elts < 4u { Mul(bcx, C_int(ccx, 4), unit_sz) }
 127                  else { fill };
 128      let Result {bcx: bcx, val: vptr} =
 129          alloc_raw(bcx, unit_ty, fill, alloc, heap);
 130      return rslt(bcx, vptr);
 131  }

 160  pub fn trans_fixed_vstore(bcx: @mut Block,
 161                            vstore_expr: @ast::expr,
 162                            content_expr: &ast::expr,
 163                            dest: expr::Dest)
 164                         -> @mut Block {
 165      //!
 166      //
 167      // [...] allocates a fixed-size array and moves it around "by value".
 168      // In this case, it means that the caller has already given us a location
 169      // to store the array of the suitable size, so all we have to do is
 170      // generate the content.
 171  
 172      debug!("trans_fixed_vstore(vstore_expr=%s, dest=%?)",
 173             bcx.expr_to_str(vstore_expr), dest.to_str(bcx.ccx()));
 174      let _indenter = indenter();
 175  
 176      let vt = vec_types_from_expr(bcx, vstore_expr);
 177  
 178      return match dest {
 179          Ignore => write_content(bcx, &vt, vstore_expr, content_expr, dest),
 180          SaveIn(lldest) => {
 181              // lldest will have type *[T x N], but we want the type *T,
 182              // so use GEP to convert:
 183              let lldest = GEPi(bcx, lldest, [0, 0]);
 184              write_content(bcx, &vt, vstore_expr, content_expr, SaveIn(lldest))
 185          }
 186      };
 187  }

 189  pub fn trans_slice_vstore(bcx: @mut Block,
 190                            vstore_expr: @ast::expr,
 191                            content_expr: @ast::expr,
 192                            dest: expr::Dest)
 193                         -> @mut Block {
 194      //!
 195      //
 196      // &[...] allocates memory on the stack and writes the values into it,
 197      // returning a slice (pair of ptr, len).  &"..." is similar except that
 198      // the memory can be statically allocated.
 199  
 200      let ccx = bcx.ccx();
 201  
 202      debug!("trans_slice_vstore(vstore_expr=%s, dest=%s)",
 203             bcx.expr_to_str(vstore_expr), dest.to_str(ccx));
 204      let _indenter = indenter();
 205  
 206      // Handle the &"..." case:
 207      match content_expr.node {
 208          ast::expr_lit(@codemap::spanned {node: ast::lit_str(s), span: _}) => {
 209              return trans_lit_str(bcx, content_expr, s, dest);
 210          }
 211          _ => {}
 212      }
 213  
 214      // Handle the &[...] case:
 215      let vt = vec_types_from_expr(bcx, vstore_expr);
 216      let count = elements_required(bcx, content_expr);
 217      debug!("vt=%s, count=%?", vt.to_str(ccx), count);
 218  
 219      // Make a fixed-length backing array and allocate it on the stack.
 220      let llcount = C_uint(ccx, count);
 221      let llfixed = base::arrayalloca(bcx, vt.llunit_ty, llcount);
 222  
 223      // Arrange for the backing array to be cleaned up.
 224      let fixed_ty = ty::mk_evec(bcx.tcx(),
 225                                 ty::mt {ty: vt.unit_ty, mutbl: ast::m_mutbl},
 226                                 ty::vstore_fixed(count));
 227      let llfixed_ty = type_of::type_of(bcx.ccx(), fixed_ty).ptr_to();
 228      let llfixed_casted = BitCast(bcx, llfixed, llfixed_ty);
 229      add_clean(bcx, llfixed_casted, fixed_ty);
 230  
 231      // Generate the content into the backing array.
 232      let bcx = write_content(bcx, &vt, vstore_expr,
 233                              content_expr, SaveIn(llfixed));
 234  
 235      // Finally, create the slice pair itself.
 236      match dest {
 237          Ignore => {}
 238          SaveIn(lldest) => {
 239              Store(bcx, llfixed, GEPi(bcx, lldest, [0u, abi::slice_elt_base]));
 240              let lllen = Mul(bcx, llcount, vt.llunit_size);
 241              Store(bcx, lllen, GEPi(bcx, lldest, [0u, abi::slice_elt_len]));
 242          }
 243      }
 244  
 245      return bcx;
 246  }

 248  pub fn trans_lit_str(bcx: @mut Block,
 249                       lit_expr: @ast::expr,
 250                       str_lit: @str,
 251                       dest: Dest)
 252                    -> @mut Block {
 253      //!
 254      //
 255      // Literal strings translate to slices into static memory.  This is
 256      // different from trans_slice_vstore() above because it does need to copy
 257      // the content anywhere.
 258  
 259      debug!("trans_lit_str(lit_expr=%s, dest=%s)",
 260             bcx.expr_to_str(lit_expr),
 261             dest.to_str(bcx.ccx()));
 262      let _indenter = indenter();
 263  
 264      match dest {
 265          Ignore => bcx,
 266          SaveIn(lldest) => {
 267              unsafe {
 268                  let bytes = str_lit.len(); // count null-terminator too
 269                  let llbytes = C_uint(bcx.ccx(), bytes);
 270                  let llcstr = C_cstr(bcx.ccx(), str_lit);
 271                  let llcstr = llvm::LLVMConstPointerCast(llcstr, Type::i8p().to_ref());
 272                  Store(bcx, llcstr,
 273                        GEPi(bcx, lldest, [0u, abi::slice_elt_base]));
 274                  Store(bcx, llbytes,
 275                        GEPi(bcx, lldest, [0u, abi::slice_elt_len]));
 276                  bcx
 277              }
 278          }
 279      }
 280  }

 283  pub fn trans_uniq_or_managed_vstore(bcx: @mut Block, heap: heap, vstore_expr: @ast::expr,
 284                                      content_expr: &ast::expr) -> DatumBlock {
 285      //!
 286      //
 287      // @[...] or ~[...] (also @"..." or ~"...") allocate boxes in the
 288      // appropriate heap and write the array elements into them.
 289  
 290      debug!("trans_uniq_or_managed_vstore(vstore_expr=%s, heap=%?)",
 291             bcx.expr_to_str(vstore_expr), heap);
 292      let _indenter = indenter();
 293  
 294      // Handle ~"".
 295      match heap {
 296          heap_exchange => {
 297              match content_expr.node {
 298                  ast::expr_lit(@codemap::spanned {
 299                      node: ast::lit_str(s), span
 300                  }) => {
 301                      let llptrval = C_cstr(bcx.ccx(), s);
 302                      let llptrval = PointerCast(bcx, llptrval, Type::i8p());
 303                      let llsizeval = C_uint(bcx.ccx(), s.len());
 304                      let typ = ty::mk_estr(bcx.tcx(), ty::vstore_uniq);
 305                      let lldestval = scratch_datum(bcx, typ, "", false);
 306                      let alloc_fn = langcall(bcx, Some(span), "",
 307                                              StrDupUniqFnLangItem);
 308                      let bcx = callee::trans_lang_call(
 309                          bcx,
 310                          alloc_fn,
 311                          [ llptrval, llsizeval ],
 312                          Some(expr::SaveIn(lldestval.to_ref_llval(bcx)))).bcx;
 313                      return DatumBlock {
 314                          bcx: bcx,
 315                          datum: lldestval
 316                      };
 317                  }
 318                  _ => {}
 319              }
 320          }
 321          heap_exchange_closure => fail!("vectors use exchange_alloc"),
 322          heap_managed | heap_managed_unique => {}
 323      }
 324  
 325      let vt = vec_types_from_expr(bcx, vstore_expr);
 326      let count = elements_required(bcx, content_expr);
 327  
 328      let Result {bcx, val} = alloc_vec(bcx, vt.unit_ty, count, heap);
 329  
 330      add_clean_free(bcx, val, heap);
 331      let dataptr = get_dataptr(bcx, get_bodyptr(bcx, val, vt.vec_ty));
 332  
 333      debug!("alloc_vec() returned val=%s, dataptr=%s",
 334             bcx.val_to_str(val), bcx.val_to_str(dataptr));
 335  
 336      let bcx = write_content(bcx, &vt, vstore_expr,
 337                              content_expr, SaveIn(dataptr));
 338  
 339      revoke_clean(bcx, val);
 340  
 341      return immediate_rvalue_bcx(bcx, val, vt.vec_ty);
 342  }

 344  pub fn write_content(bcx: @mut Block,
 345                       vt: &VecTypes,
 346                       vstore_expr: @ast::expr,
 347                       content_expr: &ast::expr,
 348                       dest: Dest)
 349                    -> @mut Block {
 350      let _icx = push_ctxt("tvec::write_content");
 351      let mut bcx = bcx;
 352  
 353      debug!("write_content(vt=%s, dest=%s, vstore_expr=%?)",
 354             vt.to_str(bcx.ccx()),
 355             dest.to_str(bcx.ccx()),
 356             bcx.expr_to_str(vstore_expr));
 357      let _indenter = indenter();
 358  
 359      match content_expr.node {
 360          ast::expr_lit(@codemap::spanned { node: ast::lit_str(s), _ }) => {
 361              match dest {
 362                  Ignore => {
 363                      return bcx;
 364                  }
 365                  SaveIn(lldest) => {
 366                      let bytes = s.len();
 367                      let llbytes = C_uint(bcx.ccx(), bytes);
 368                      let llcstr = C_cstr(bcx.ccx(), s);
 369                      base::call_memcpy(bcx, lldest, llcstr, llbytes, 1);
 370                      return bcx;
 371                  }
 372              }
 373          }
 374          ast::expr_vec(ref elements, _) => {
 375              match dest {
 376                  Ignore => {
 377                      for element in elements.iter() {
 378                          bcx = expr::trans_into(bcx, *element, Ignore);
 379                      }
 380                  }
 381  
 382                  SaveIn(lldest) => {
 383                      let mut temp_cleanups = ~[];
 384                      for (i, element) in elements.iter().enumerate() {
 385                          let lleltptr = GEPi(bcx, lldest, [i]);
 386                          debug!("writing index %? with lleltptr=%?",
 387                                 i, bcx.val_to_str(lleltptr));
 388                          bcx = expr::trans_into(bcx, *element,
 389                                                 SaveIn(lleltptr));
 390                          add_clean_temp_mem(bcx, lleltptr, vt.unit_ty);
 391                          temp_cleanups.push(lleltptr);
 392                      }
 393                      for cleanup in temp_cleanups.iter() {
 394                          revoke_clean(bcx, *cleanup);
 395                      }
 396                  }
 397              }
 398              return bcx;
 399          }
 400          ast::expr_repeat(element, count_expr, _) => {
 401              match dest {
 402                  Ignore => {

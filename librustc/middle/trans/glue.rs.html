  77 pub fn drop_ty_immediate(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
  78     let _icx = push_ctxt("drop_ty_immediate");
  79     match ty::get(t).sty {
  80         ty::ty_uniq(_)
  81       | ty::ty_evec(_, ty::vstore_uniq)
  82       | ty::ty_estr(ty::vstore_uniq) => {
  83         free_ty_immediate(bcx, v, t)
  84       }
  85         ty::ty_box(_) | ty::ty_opaque_box
  86       | ty::ty_evec(_, ty::vstore_box)
  87       | ty::ty_estr(ty::vstore_box) => {
  88         decr_refcnt_maybe_free(bcx, v, None, t)
  89       }
  90       _ => bcx.tcx().sess.bug("drop_ty_immediate: non-box ty")
  91     }
  92 }

 103 pub fn free_ty_immediate(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
 104     let _icx = push_ctxt("free_ty_immediate");
 105     match ty::get(t).sty {
 106       ty::ty_uniq(_) |
 107       ty::ty_evec(_, ty::vstore_uniq) |
 108       ty::ty_estr(ty::vstore_uniq) |
 109       ty::ty_box(_) | ty::ty_opaque_box |
 110       ty::ty_evec(_, ty::vstore_box) |
 111       ty::ty_estr(ty::vstore_box) |
 112       ty::ty_opaque_closure_ptr(_) => {
 113         let vp = alloca(bcx, type_of(bcx.ccx(), t), "");
 114         Store(bcx, v, vp);
 115         free_ty(bcx, vp, t)
 116       }
 117       _ => bcx.tcx().sess.bug("free_ty_immediate: non-box ty")
 118     }
 119 }

 129 pub fn simplified_glue_type(tcx: ty::ctxt, field: uint, t: ty::t) -> ty::t {
 130     if (field == abi::tydesc_field_take_glue ||
 131         field == abi::tydesc_field_drop_glue ||
 132         field == abi::tydesc_field_free_glue) &&
 133         ! ty::type_needs_drop(tcx, t) {
 134           return ty::mk_u32();
 135     }
 136 
 137     if field == abi::tydesc_field_take_glue {
 138         match ty::get(t).sty {
 139           ty::ty_unboxed_vec(*) |
 140               ty::ty_uniq(*) |
 141               ty::ty_estr(ty::vstore_uniq) |
 142               ty::ty_evec(_, ty::vstore_uniq) => { return ty::mk_u32(); }
 143           _ => ()
 144         }
 145     }
 146 
 147     if field == abi::tydesc_field_take_glue &&
 148         ty::type_is_boxed(t) {
 149           return ty::mk_imm_box(tcx, ty::mk_u32());
 150     }
 151 
 152     if field == abi::tydesc_field_free_glue {
 153         match ty::get(t).sty {
 154           ty::ty_bare_fn(*) |
 155           ty::ty_closure(*) |
 156           ty::ty_box(*) |
 157           ty::ty_opaque_box |
 158           ty::ty_uniq(*) |
 159           ty::ty_evec(_, ty::vstore_uniq) | ty::ty_estr(ty::vstore_uniq) |
 160           ty::ty_evec(_, ty::vstore_box) | ty::ty_estr(ty::vstore_box) |
 161           ty::ty_opaque_closure_ptr(*) => (),
 162           _ => { return ty::mk_u32(); }
 163         }
 164     }
 165 
 166     if (field == abi::tydesc_field_free_glue ||
 167         field == abi::tydesc_field_drop_glue) {
 168         match ty::get(t).sty {
 169           ty::ty_box(mt) |
 170           ty::ty_evec(mt, ty::vstore_box)
 171           if ! ty::type_needs_drop(tcx, mt.ty) =>
 172           return ty::mk_imm_box(tcx, ty::mk_u32()),
 173 
 174           ty::ty_uniq(mt) |
 175           ty::ty_evec(mt, ty::vstore_uniq)
 176           if ! ty::type_needs_drop(tcx, mt.ty) =>
 177           return ty::mk_imm_uniq(tcx, ty::mk_u32()),
 178 
 179           _ => ()
 180         }
 181     }
 182 
 183     return t;
 184 }

 186 pub fn lazily_emit_simplified_tydesc_glue(ccx: @mut CrateContext,
 187                                           field: uint,
 188                                           ti: &mut tydesc_info) -> bool {
 189     let _icx = push_ctxt("lazily_emit_simplified_tydesc_glue");
 190     let simpl = simplified_glue_type(ccx.tcx, field, ti.ty);
 191     if simpl != ti.ty {
 192         let simpl_ti = get_tydesc(ccx, simpl);
 193         lazily_emit_tydesc_glue(ccx, field, simpl_ti);
 194         {
 195             if field == abi::tydesc_field_take_glue {
 196                 ti.take_glue = simpl_ti.take_glue;
 197             } else if field == abi::tydesc_field_drop_glue {
 198                 ti.drop_glue = simpl_ti.drop_glue;
 199             } else if field == abi::tydesc_field_free_glue {
 200                 ti.free_glue = simpl_ti.free_glue;
 201             } else if field == abi::tydesc_field_visit_glue {
 202                 ti.visit_glue = simpl_ti.visit_glue;
 203             }
 204         }
 205         return true;
 206     }
 207     return false;
 208 }

 211 pub fn lazily_emit_tydesc_glue(ccx: @mut CrateContext,
 212                                field: uint,
 213                                ti: @mut tydesc_info) {
 214     let _icx = push_ctxt("lazily_emit_tydesc_glue");
 215     let llfnty = Type::glue_fn(type_of::type_of(ccx, ti.ty).ptr_to());
 216 
 217     if lazily_emit_simplified_tydesc_glue(ccx, field, ti) {
 218         return;
 219     }
 220 
 221     if field == abi::tydesc_field_take_glue {
 222         match ti.take_glue {
 223           Some(_) => (),
 224           None => {
 225             debug!("+++ lazily_emit_tydesc_glue TAKE %s",
 226                    ppaux::ty_to_str(ccx.tcx, ti.ty));
 227             let glue_fn = declare_generic_glue(ccx, ti.ty, llfnty, "take");
 228             ti.take_glue = Some(glue_fn);
 229             make_generic_glue(ccx, ti.ty, glue_fn, make_take_glue, "take");
 230             debug!("--- lazily_emit_tydesc_glue TAKE %s",
 231                    ppaux::ty_to_str(ccx.tcx, ti.ty));
 232           }
 233         }
 234     } else if field == abi::tydesc_field_drop_glue {
 235         match ti.drop_glue {
 236           Some(_) => (),
 237           None => {
 238             debug!("+++ lazily_emit_tydesc_glue DROP %s",
 239                    ppaux::ty_to_str(ccx.tcx, ti.ty));
 240             let glue_fn = declare_generic_glue(ccx, ti.ty, llfnty, "drop");
 241             ti.drop_glue = Some(glue_fn);
 242             make_generic_glue(ccx, ti.ty, glue_fn, make_drop_glue, "drop");
 243             debug!("--- lazily_emit_tydesc_glue DROP %s",
 244                    ppaux::ty_to_str(ccx.tcx, ti.ty));
 245           }
 246         }
 247     } else if field == abi::tydesc_field_free_glue {
 248         match ti.free_glue {
 249           Some(_) => (),
 250           None => {
 251             debug!("+++ lazily_emit_tydesc_glue FREE %s",
 252                    ppaux::ty_to_str(ccx.tcx, ti.ty));
 253             let glue_fn = declare_generic_glue(ccx, ti.ty, llfnty, "free");
 254             ti.free_glue = Some(glue_fn);
 255             make_generic_glue(ccx, ti.ty, glue_fn, make_free_glue, "free");
 256             debug!("--- lazily_emit_tydesc_glue FREE %s",
 257                    ppaux::ty_to_str(ccx.tcx, ti.ty));
 258           }
 259         }
 260     } else if field == abi::tydesc_field_visit_glue {
 261         match ti.visit_glue {
 262           Some(_) => (),
 263           None => {
 264             debug!("+++ lazily_emit_tydesc_glue VISIT %s",
 265                    ppaux::ty_to_str(ccx.tcx, ti.ty));
 266             let glue_fn = declare_generic_glue(ccx, ti.ty, llfnty, "visit");
 267             ti.visit_glue = Some(glue_fn);
 268             make_generic_glue(ccx, ti.ty, glue_fn, make_visit_glue, "visit");
 269             debug!("--- lazily_emit_tydesc_glue VISIT %s",
 270                    ppaux::ty_to_str(ccx.tcx, ti.ty));
 271           }
 272         }
 273     }
 274 }

 276 // See [Note-arg-mode]
 277 pub fn call_tydesc_glue_full(bcx: @mut Block,
 278                              v: ValueRef,
 279                              tydesc: ValueRef,
 280                              field: uint,
 281                              static_ti: Option<@mut tydesc_info>) {
 282     let _icx = push_ctxt("call_tydesc_glue_full");
 283     let ccx = bcx.ccx();
 284     // NB: Don't short-circuit even if this block is unreachable because
 285     // GC-based cleanup needs to the see that the roots are live.
 286     let no_lpads =
 287         ccx.sess.opts.debugging_opts & session::no_landing_pads != 0;
 288     if bcx.unreachable && !no_lpads { return; }
 289 
 290     let static_glue_fn = match static_ti {
 291       None => None,
 292       Some(sti) => {
 293         lazily_emit_tydesc_glue(ccx, field, sti);
 294         if field == abi::tydesc_field_take_glue {
 295             sti.take_glue
 296         } else if field == abi::tydesc_field_drop_glue {
 297             sti.drop_glue
 298         } else if field == abi::tydesc_field_free_glue {
 299             sti.free_glue
 300         } else if field == abi::tydesc_field_visit_glue {
 301             sti.visit_glue
 302         } else {
 303             None
 304         }
 305       }
 306     };
 307 
 308     // When static type info is available, avoid casting parameter unless the
 309     // glue is using a simplified type, because the function already has the
 310     // right type. Otherwise cast to generic pointer.
 311     let llrawptr = if static_ti.is_none() || static_glue_fn.is_none() {
 312         PointerCast(bcx, v, Type::i8p())
 313     } else {
 314         let ty = static_ti.get().ty;
 315         let simpl = simplified_glue_type(ccx.tcx, field, ty);
 316         if simpl != ty {
 317             PointerCast(bcx, v, type_of(ccx, simpl).ptr_to())
 318         } else {
 319             v
 320         }
 321     };
 322 
 323     let llfn = {
 324         match static_glue_fn {
 325           None => {
 326             // Select out the glue function to call from the tydesc
 327             let llfnptr = GEPi(bcx, tydesc, [0u, field]);
 328             Load(bcx, llfnptr)
 329           }
 330           Some(sgf) => sgf
 331         }
 332     };
 333 
 334     Call(bcx, llfn, [C_null(Type::nil().ptr_to()), llrawptr]);
 335 }

 346 pub fn make_visit_glue(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
 347     let _icx = push_ctxt("make_visit_glue");
 348     do with_scope(bcx, None, "visitor cleanup") |bcx| {
 349         let mut bcx = bcx;
 350         let (visitor_trait, object_ty) = match ty::visitor_object_ty(bcx.tcx()){
 351             Ok(pair) => pair,
 352             Err(s) => {
 353                 bcx.tcx().sess.fatal(s);
 354             }
 355         };
 356         let v = PointerCast(bcx, v, type_of::type_of(bcx.ccx(), object_ty).ptr_to());
 357         bcx = reflect::emit_calls_to_trait_visit_ty(bcx, t, v, visitor_trait.def_id);
 358         // The visitor is a boxed object and needs to be dropped
 359         add_clean(bcx, v, object_ty);
 360         bcx
 361     }
 362 }

 364 pub fn make_free_glue(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
 365     // NB: v0 is an *alias* of type t here, not a direct value.
 366     let _icx = push_ctxt("make_free_glue");
 367     match ty::get(t).sty {
 368       ty::ty_box(body_mt) => {
 369         let v = Load(bcx, v);
 370         let body = GEPi(bcx, v, [0u, abi::box_field_body]);
 371         let bcx = drop_ty(bcx, body, body_mt.ty);
 372         trans_free(bcx, v)
 373       }
 374       ty::ty_opaque_box => {
 375         let v = Load(bcx, v);
 376         let td = Load(bcx, GEPi(bcx, v, [0u, abi::box_field_tydesc]));
 377         let valptr = GEPi(bcx, v, [0u, abi::box_field_body]);
 378         // Generate code that, dynamically, indexes into the
 379         // tydesc and calls the drop glue that got set dynamically
 380         call_tydesc_glue_full(bcx, valptr, td, abi::tydesc_field_drop_glue,
 381                               None);
 382         trans_free(bcx, v)
 383       }
 384       ty::ty_uniq(*) => {
 385         uniq::make_free_glue(bcx, v, t)
 386       }
 387       ty::ty_evec(_, ty::vstore_uniq) | ty::ty_estr(ty::vstore_uniq) |
 388       ty::ty_evec(_, ty::vstore_box) | ty::ty_estr(ty::vstore_box) => {
 389         make_free_glue(bcx, v, tvec::expand_boxed_vec_ty(bcx.tcx(), t))
 390       }
 391       ty::ty_closure(_) => {
 392         closure::make_closure_glue(bcx, v, t, free_ty)
 393       }
 394       ty::ty_opaque_closure_ptr(ck) => {
 395         closure::make_opaque_cbox_free_glue(bcx, ck, v)
 396       }
 397       _ => bcx
 398     }
 399 }

 401 pub fn trans_struct_drop_flag(bcx: @mut Block, t: ty::t, v0: ValueRef, dtor_did: ast::def_id,
 402                               class_did: ast::def_id, substs: &ty::substs) -> @mut Block {
 403     let repr = adt::represent_type(bcx.ccx(), t);
 404     let drop_flag = adt::trans_drop_flag_ptr(bcx, repr, v0);
 405     do with_cond(bcx, IsNotNull(bcx, Load(bcx, drop_flag))) |cx| {
 406         let mut bcx = cx;
 407 
 408         // Find and call the actual destructor
 409         let dtor_addr = get_res_dtor(bcx.ccx(), dtor_did,
 410                                      class_did, substs.tps.clone());
 411 
 412         // The second argument is the "self" argument for drop
 413         let params = unsafe {
 414             let ty = Type::from_ref(llvm::LLVMTypeOf(dtor_addr));
 415             ty.element_type().func_params()
 416         };
 417 
 418         // Class dtors have no explicit args, so the params should
 419         // just consist of the environment (self)
 420         assert_eq!(params.len(), 1);
 421 
 422         let self_arg = PointerCast(bcx, v0, params[0]);
 423         let args = ~[self_arg];
 424 
 425         Call(bcx, dtor_addr, args);
 426 
 427         // Drop the fields
 428         let field_tys = ty::struct_fields(bcx.tcx(), class_did, substs);
 429         for (i, fld) in field_tys.iter().enumerate() {
 430             let llfld_a = adt::trans_field_ptr(bcx, repr, v0, 0, i);
 431             bcx = drop_ty(bcx, llfld_a, fld.mt.ty);
 432         }
 433 
 434         Store(bcx, C_u8(0), drop_flag);
 435         bcx
 436     }
 437 }

 472 pub fn make_drop_glue(bcx: @mut Block, v0: ValueRef, t: ty::t) -> @mut Block {
 473     // NB: v0 is an *alias* of type t here, not a direct value.
 474     let _icx = push_ctxt("make_drop_glue");
 475     let ccx = bcx.ccx();
 476     match ty::get(t).sty {
 477       ty::ty_box(_) | ty::ty_opaque_box |
 478       ty::ty_estr(ty::vstore_box) | ty::ty_evec(_, ty::vstore_box) => {
 479         decr_refcnt_maybe_free(bcx, Load(bcx, v0), Some(v0), t)
 480       }
 481       ty::ty_uniq(_) |
 482       ty::ty_evec(_, ty::vstore_uniq) | ty::ty_estr(ty::vstore_uniq) => {
 483         free_ty(bcx, v0, t)
 484       }
 485       ty::ty_unboxed_vec(_) => {
 486         tvec::make_drop_glue_unboxed(bcx, v0, t)
 487       }
 488       ty::ty_struct(did, ref substs) => {
 489         let tcx = bcx.tcx();
 490         match ty::ty_dtor(tcx, did) {
 491           ty::TraitDtor(dtor, true) => {
 492             trans_struct_drop_flag(bcx, t, v0, dtor, did, substs)
 493           }
 494           ty::TraitDtor(dtor, false) => {
 495             trans_struct_drop(bcx, t, v0, dtor, did, substs)
 496           }
 497           ty::NoDtor => {
 498             // No dtor? Just the default case
 499             iter_structural_ty(bcx, v0, t, drop_ty)
 500           }
 501         }
 502       }
 503       ty::ty_closure(_) => {
 504         closure::make_closure_glue(bcx, v0, t, drop_ty)
 505       }
 506       ty::ty_trait(_, _, ty::BoxTraitStore, _, _) => {
 507           let llbox_ptr = GEPi(bcx, v0, [0u, abi::trt_field_box]);
 508           let llbox = Load(bcx, llbox_ptr);
 509           decr_refcnt_maybe_free(bcx, llbox, Some(llbox_ptr),
 510                                  ty::mk_opaque_box(ccx.tcx))
 511       }
 512       ty::ty_trait(_, _, ty::UniqTraitStore, _, _) => {
 513           let lluniquevalue = GEPi(bcx, v0, [0, abi::trt_field_box]);
 514           // Only drop the value when it is non-null
 515           do with_cond(bcx, IsNotNull(bcx, Load(bcx, lluniquevalue))) |bcx| {
 516               let llvtable = Load(bcx, GEPi(bcx, v0, [0, abi::trt_field_vtable]));
 517 
 518               // Cast the vtable to a pointer to a pointer to a tydesc.
 519               let llvtable = PointerCast(bcx, llvtable,
 520                                          ccx.tydesc_type.ptr_to().ptr_to());
 521               let lltydesc = Load(bcx, llvtable);
 522               call_tydesc_glue_full(bcx,
 523                                     lluniquevalue,
 524                                     lltydesc,
 525                                     abi::tydesc_field_free_glue,
 526                                     None);
 527               bcx
 528           }
 529       }
 530       ty::ty_opaque_closure_ptr(ck) => {
 531         closure::make_opaque_cbox_drop_glue(bcx, ck, v0)
 532       }
 533       _ => {
 534         if ty::type_needs_drop(ccx.tcx, t) &&
 535             ty::type_is_structural(t) {
 536             iter_structural_ty(bcx, v0, t, drop_ty)
 537         } else { bcx }
 538       }
 539     }
 540 }

 542 // box_ptr_ptr is optional, it is constructed if not supplied.
 543 pub fn decr_refcnt_maybe_free(bcx: @mut Block, box_ptr: ValueRef,
 544                               box_ptr_ptr: Option<ValueRef>,
 545                               t: ty::t)
 546                            -> @mut Block {
 547     let _icx = push_ctxt("decr_refcnt_maybe_free");
 548     let ccx = bcx.ccx();
 549 
 550     let decr_bcx = sub_block(bcx, "decr");
 551     let free_bcx = sub_block(decr_bcx, "free");
 552     let next_bcx = sub_block(bcx, "next");
 553     CondBr(bcx, IsNotNull(bcx, box_ptr), decr_bcx.llbb, next_bcx.llbb);
 554 
 555     let rc_ptr = GEPi(decr_bcx, box_ptr, [0u, abi::box_field_refcnt]);
 556     let rc = Sub(decr_bcx, Load(decr_bcx, rc_ptr), C_int(ccx, 1));
 557     Store(decr_bcx, rc, rc_ptr);
 558     CondBr(decr_bcx, IsNull(decr_bcx, rc), free_bcx.llbb, next_bcx.llbb);
 559 
 560     let free_bcx = match box_ptr_ptr {
 561         Some(p) => free_ty(free_bcx, p, t),
 562         None => free_ty_immediate(free_bcx, box_ptr, t)
 563     };
 564     Br(free_bcx, next_bcx.llbb);
 565 
 566     next_bcx
 567 }

 570 pub fn make_take_glue(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
 571     let _icx = push_ctxt("make_take_glue");
 572     // NB: v is a *pointer* to type t here, not a direct value.
 573     match ty::get(t).sty {
 574       ty::ty_box(_) | ty::ty_opaque_box |
 575       ty::ty_evec(_, ty::vstore_box) | ty::ty_estr(ty::vstore_box) => {
 576         incr_refcnt_of_boxed(bcx, Load(bcx, v)); bcx
 577       }
 578       ty::ty_evec(_, ty::vstore_slice(_))
 579       | ty::ty_estr(ty::vstore_slice(_)) => {
 580         bcx
 581       }
 582       ty::ty_closure(ty::ClosureTy { sigil: ast::BorrowedSigil, _ }) |
 583       ty::ty_closure(ty::ClosureTy { sigil: ast::ManagedSigil, _ }) => {
 584         closure::make_closure_glue(bcx, v, t, take_ty)
 585       }
 586       ty::ty_closure(ty::ClosureTy { sigil: ast::OwnedSigil, _ }) => bcx,
 587       ty::ty_trait(_, _, ty::BoxTraitStore, _, _) => {
 588         let llbox = Load(bcx, GEPi(bcx, v, [0u, abi::trt_field_box]));
 589         incr_refcnt_of_boxed(bcx, llbox);
 590         bcx
 591       }
 592       ty::ty_trait(_, _, ty::UniqTraitStore, _, _) => {
 593           let lluniquevalue = GEPi(bcx, v, [0, abi::trt_field_box]);
 594           let llvtable = Load(bcx, GEPi(bcx, v, [0, abi::trt_field_vtable]));
 595 
 596           // Cast the vtable to a pointer to a pointer to a tydesc.
 597           let llvtable = PointerCast(bcx, llvtable,
 598                                      bcx.ccx().tydesc_type.ptr_to().ptr_to());
 599           let lltydesc = Load(bcx, llvtable);
 600           call_tydesc_glue_full(bcx,
 601                                 lluniquevalue,
 602                                 lltydesc,
 603                                 abi::tydesc_field_take_glue,
 604                                 None);
 605           bcx
 606       }
 607       ty::ty_opaque_closure_ptr(ck) => {
 608         closure::make_opaque_cbox_take_glue(bcx, ck, v)
 609       }
 610       ty::ty_struct(did, _) => {
 611         let tcx = bcx.tcx();
 612         let bcx = iter_structural_ty(bcx, v, t, take_ty);
 613 
 614         match ty::ty_dtor(tcx, did) {
 615           ty::TraitDtor(_, false) => {
 616             // Zero out the struct
 617             unsafe {
 618                 let ty = Type::from_ref(llvm::LLVMTypeOf(v));
 619                 memzero(&B(bcx), v, ty);
 620             }
 621 
 622           }
 623           _ => { }
 624         }
 625         bcx
 626       }
 627       _ if ty::type_is_structural(t) => {
 628         iter_structural_ty(bcx, v, t, take_ty)
 629       }
 630       _ => bcx
 631     }
 632 }

 694 pub fn make_generic_glue_inner(ccx: @mut CrateContext,

 733 pub fn emit_tydescs(ccx: &mut CrateContext) {
 734     let _icx = push_ctxt("emit_tydescs");
 735     // As of this point, allow no more tydescs to be created.
 736     ccx.finished_tydescs = true;
 737     let glue_fn_ty = Type::generic_glue_fn(ccx).ptr_to();
 738     let tyds = &mut ccx.tydescs;
 739     for (_, &val) in tyds.iter() {
 740         let ti = val;
 741 
 742         // Each of the glue functions needs to be cast to a generic type
 743         // before being put into the tydesc because we only have a singleton
 744         // tydesc type. Then we'll recast each function to its real type when
 745         // calling it.
 746         let take_glue =
 747             match ti.take_glue {
 748               None => { ccx.stats.n_null_glues += 1u; C_null(glue_fn_ty) }
 749               Some(v) => {
 750                 unsafe {
 751                     ccx.stats.n_real_glues += 1u;
 752                     llvm::LLVMConstPointerCast(v, glue_fn_ty.to_ref())
 753                 }
 754               }
 755             };
 756         let drop_glue =
 757             match ti.drop_glue {
 758               None => { ccx.stats.n_null_glues += 1u; C_null(glue_fn_ty) }
 759               Some(v) => {
 760                 unsafe {
 761                     ccx.stats.n_real_glues += 1u;
 762                     llvm::LLVMConstPointerCast(v, glue_fn_ty.to_ref())
 763                 }
 764               }
 765             };
 766         let free_glue =
 767             match ti.free_glue {
 768               None => { ccx.stats.n_null_glues += 1u; C_null(glue_fn_ty) }
 769               Some(v) => {
 770                 unsafe {
 771                     ccx.stats.n_real_glues += 1u;
 772                     llvm::LLVMConstPointerCast(v, glue_fn_ty.to_ref())
 773                 }
 774               }
 775             };
 776         let visit_glue =
 777             match ti.visit_glue {
 778               None => { ccx.stats.n_null_glues += 1u; C_null(glue_fn_ty) }
 779               Some(v) => {
 780                 unsafe {
 781                     ccx.stats.n_real_glues += 1u;
 782                     llvm::LLVMConstPointerCast(v, glue_fn_ty.to_ref())
 783                 }
 784               }
 785             };
 786 
 787         let tydesc = C_named_struct(ccx.tydesc_type,
 788                                     [ti.size, // size
 789                                     ti.align, // align
 790                                     take_glue, // take_glue
 791                                     drop_glue, // drop_glue
 792                                     free_glue, // free_glue
 793                                     visit_glue]); // visit_glue
 794 
 795         unsafe {
 796             let gvar = ti.tydesc;

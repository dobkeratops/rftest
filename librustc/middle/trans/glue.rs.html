  78  pub fn drop_ty_immediate(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
  79      let _icx = push_ctxt("drop_ty_immediate");
  80      match ty::get(t).sty {
  81          ty::ty_uniq(_)
  82        | ty::ty_evec(_, ty::vstore_uniq)
  83        | ty::ty_estr(ty::vstore_uniq) => {
  84          free_ty_immediate(bcx, v, t)
  85        }
  86          ty::ty_box(_) | ty::ty_opaque_box
  87        | ty::ty_evec(_, ty::vstore_box)
  88        | ty::ty_estr(ty::vstore_box) => {
  89          decr_refcnt_maybe_free(bcx, v, None, t)
  90        }
  91        _ => bcx.tcx().sess.bug("drop_ty_immediate: non-box ty")
  92      }
  93  }

 104  pub fn free_ty_immediate(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
 105      let _icx = push_ctxt("free_ty_immediate");
 106      match ty::get(t).sty {
 107        ty::ty_uniq(_) |
 108        ty::ty_evec(_, ty::vstore_uniq) |
 109        ty::ty_estr(ty::vstore_uniq) |
 110        ty::ty_box(_) | ty::ty_opaque_box |
 111        ty::ty_evec(_, ty::vstore_box) |
 112        ty::ty_estr(ty::vstore_box) |
 113        ty::ty_opaque_closure_ptr(_) => {
 114          let vp = alloca(bcx, type_of(bcx.ccx(), t), "");
 115          Store(bcx, v, vp);
 116          free_ty(bcx, vp, t)
 117        }
 118        _ => bcx.tcx().sess.bug("free_ty_immediate: non-box ty")
 119      }
 120  }

 130  pub fn simplified_glue_type(tcx: ty::ctxt, field: uint, t: ty::t) -> ty::t {
 131      if (field == abi::tydesc_field_take_glue ||
 132          field == abi::tydesc_field_drop_glue ||
 133          field == abi::tydesc_field_free_glue) &&
 134          ! ty::type_needs_drop(tcx, t) {
 135            return ty::mk_u32();
 136      }
 137  
 138      if field == abi::tydesc_field_take_glue {
 139          match ty::get(t).sty {
 140            ty::ty_unboxed_vec(*) |
 141                ty::ty_uniq(*) |
 142                ty::ty_estr(ty::vstore_uniq) |
 143                ty::ty_evec(_, ty::vstore_uniq) => { return ty::mk_u32(); }
 144            _ => ()
 145          }
 146      }
 147  
 148      if field == abi::tydesc_field_take_glue &&
 149          ty::type_is_boxed(t) {
 150            return ty::mk_imm_box(tcx, ty::mk_u32());
 151      }
 152  
 153      if field == abi::tydesc_field_free_glue {
 154          match ty::get(t).sty {
 155            ty::ty_bare_fn(*) |
 156            ty::ty_closure(*) |
 157            ty::ty_box(*) |
 158            ty::ty_opaque_box |
 159            ty::ty_uniq(*) |
 160            ty::ty_evec(_, ty::vstore_uniq) | ty::ty_estr(ty::vstore_uniq) |
 161            ty::ty_evec(_, ty::vstore_box) | ty::ty_estr(ty::vstore_box) |
 162            ty::ty_opaque_closure_ptr(*) => (),
 163            _ => { return ty::mk_u32(); }
 164          }
 165      }
 166  
 167      if (field == abi::tydesc_field_free_glue ||
 168          field == abi::tydesc_field_drop_glue) {
 169          match ty::get(t).sty {
 170            ty::ty_box(mt) |
 171            ty::ty_evec(mt, ty::vstore_box)
 172            if ! ty::type_needs_drop(tcx, mt.ty) =>
 173            return ty::mk_imm_box(tcx, ty::mk_u32()),
 174  
 175            ty::ty_uniq(mt) |
 176            ty::ty_evec(mt, ty::vstore_uniq)
 177            if ! ty::type_needs_drop(tcx, mt.ty) =>
 178            return ty::mk_imm_uniq(tcx, ty::mk_u32()),
 179  
 180            _ => ()
 181          }
 182      }
 183  
 184      return t;
 185  }

 187  pub fn lazily_emit_simplified_tydesc_glue(ccx: @mut CrateContext,
 188                                            field: uint,
 189                                            ti: &mut tydesc_info) -> bool {
 190      let _icx = push_ctxt("lazily_emit_simplified_tydesc_glue");
 191      let simpl = simplified_glue_type(ccx.tcx, field, ti.ty);
 192      if simpl != ti.ty {
 193          let simpl_ti = get_tydesc(ccx, simpl);
 194          lazily_emit_tydesc_glue(ccx, field, simpl_ti);
 195          {
 196              if field == abi::tydesc_field_take_glue {
 197                  ti.take_glue = simpl_ti.take_glue;
 198              } else if field == abi::tydesc_field_drop_glue {
 199                  ti.drop_glue = simpl_ti.drop_glue;
 200              } else if field == abi::tydesc_field_free_glue {
 201                  ti.free_glue = simpl_ti.free_glue;
 202              } else if field == abi::tydesc_field_visit_glue {
 203                  ti.visit_glue = simpl_ti.visit_glue;
 204              }
 205          }
 206          return true;
 207      }
 208      return false;
 209  }

 212  pub fn lazily_emit_tydesc_glue(ccx: @mut CrateContext,
 213                                 field: uint,
 214                                 ti: @mut tydesc_info) {
 215      let _icx = push_ctxt("lazily_emit_tydesc_glue");
 216      let llfnty = Type::glue_fn(type_of::type_of(ccx, ti.ty).ptr_to());
 217  
 218      if lazily_emit_simplified_tydesc_glue(ccx, field, ti) {
 219          return;
 220      }
 221  
 222      if field == abi::tydesc_field_take_glue {
 223          match ti.take_glue {
 224            Some(_) => (),
 225            None => {
 226              debug!("+++ lazily_emit_tydesc_glue TAKE %s",
 227                     ppaux::ty_to_str(ccx.tcx, ti.ty));
 228              let glue_fn = declare_generic_glue(ccx, ti.ty, llfnty, "take");
 229              ti.take_glue = Some(glue_fn);
 230              make_generic_glue(ccx, ti.ty, glue_fn, make_take_glue, "take");
 231              debug!("--- lazily_emit_tydesc_glue TAKE %s",
 232                     ppaux::ty_to_str(ccx.tcx, ti.ty));
 233            }
 234          }
 235      } else if field == abi::tydesc_field_drop_glue {
 236          match ti.drop_glue {
 237            Some(_) => (),
 238            None => {
 239              debug!("+++ lazily_emit_tydesc_glue DROP %s",
 240                     ppaux::ty_to_str(ccx.tcx, ti.ty));
 241              let glue_fn = declare_generic_glue(ccx, ti.ty, llfnty, "drop");
 242              ti.drop_glue = Some(glue_fn);
 243              make_generic_glue(ccx, ti.ty, glue_fn, make_drop_glue, "drop");
 244              debug!("--- lazily_emit_tydesc_glue DROP %s",
 245                     ppaux::ty_to_str(ccx.tcx, ti.ty));
 246            }
 247          }
 248      } else if field == abi::tydesc_field_free_glue {
 249          match ti.free_glue {
 250            Some(_) => (),
 251            None => {
 252              debug!("+++ lazily_emit_tydesc_glue FREE %s",
 253                     ppaux::ty_to_str(ccx.tcx, ti.ty));
 254              let glue_fn = declare_generic_glue(ccx, ti.ty, llfnty, "free");
 255              ti.free_glue = Some(glue_fn);
 256              make_generic_glue(ccx, ti.ty, glue_fn, make_free_glue, "free");
 257              debug!("--- lazily_emit_tydesc_glue FREE %s",
 258                     ppaux::ty_to_str(ccx.tcx, ti.ty));
 259            }
 260          }
 261      } else if field == abi::tydesc_field_visit_glue {
 262          match ti.visit_glue {
 263            Some(_) => (),
 264            None => {
 265              debug!("+++ lazily_emit_tydesc_glue VISIT %s",
 266                     ppaux::ty_to_str(ccx.tcx, ti.ty));
 267              let glue_fn = declare_generic_glue(ccx, ti.ty, llfnty, "visit");
 268              ti.visit_glue = Some(glue_fn);
 269              make_generic_glue(ccx, ti.ty, glue_fn, make_visit_glue, "visit");
 270              debug!("--- lazily_emit_tydesc_glue VISIT %s",
 271                     ppaux::ty_to_str(ccx.tcx, ti.ty));
 272            }
 273          }
 274      }
 275  }

 277  // See [Note-arg-mode]
 278  pub fn call_tydesc_glue_full(bcx: @mut Block,
 279                               v: ValueRef,
 280                               tydesc: ValueRef,
 281                               field: uint,
 282                               static_ti: Option<@mut tydesc_info>) {
 283      let _icx = push_ctxt("call_tydesc_glue_full");
 284      let ccx = bcx.ccx();
 285      // NB: Don't short-circuit even if this block is unreachable because
 286      // GC-based cleanup needs to the see that the roots are live.
 287      let no_lpads =
 288          ccx.sess.opts.debugging_opts & session::no_landing_pads != 0;
 289      if bcx.unreachable && !no_lpads { return; }
 290  
 291      let static_glue_fn = match static_ti {
 292        None => None,
 293        Some(sti) => {
 294          lazily_emit_tydesc_glue(ccx, field, sti);
 295          if field == abi::tydesc_field_take_glue {
 296              sti.take_glue
 297          } else if field == abi::tydesc_field_drop_glue {
 298              sti.drop_glue
 299          } else if field == abi::tydesc_field_free_glue {
 300              sti.free_glue
 301          } else if field == abi::tydesc_field_visit_glue {
 302              sti.visit_glue
 303          } else {
 304              None
 305          }
 306        }
 307      };
 308  
 309      // When static type info is available, avoid casting parameter unless the
 310      // glue is using a simplified type, because the function already has the
 311      // right type. Otherwise cast to generic pointer.
 312      let llrawptr = if static_ti.is_none() || static_glue_fn.is_none() {
 313          PointerCast(bcx, v, Type::i8p())
 314      } else {
 315          let ty = static_ti.unwrap().ty;
 316          let simpl = simplified_glue_type(ccx.tcx, field, ty);
 317          if simpl != ty {
 318              PointerCast(bcx, v, type_of(ccx, simpl).ptr_to())
 319          } else {
 320              v
 321          }
 322      };
 323  
 324      let llfn = {
 325          match static_glue_fn {
 326            None => {
 327              // Select out the glue function to call from the tydesc
 328              let llfnptr = GEPi(bcx, tydesc, [0u, field]);
 329              Load(bcx, llfnptr)
 330            }
 331            Some(sgf) => sgf
 332          }
 333      };
 334  
 335      Call(bcx, llfn, [C_null(Type::nil().ptr_to()), llrawptr]);
 336  }

 347  pub fn make_visit_glue(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
 348      let _icx = push_ctxt("make_visit_glue");
 349      do with_scope(bcx, None, "visitor cleanup") |bcx| {
 350          let mut bcx = bcx;
 351          let (visitor_trait, object_ty) = match ty::visitor_object_ty(bcx.tcx()){
 352              Ok(pair) => pair,
 353              Err(s) => {
 354                  bcx.tcx().sess.fatal(s);
 355              }
 356          };
 357          let v = PointerCast(bcx, v, type_of::type_of(bcx.ccx(), object_ty).ptr_to());
 358          bcx = reflect::emit_calls_to_trait_visit_ty(bcx, t, v, visitor_trait.def_id);
 359          // The visitor is a boxed object and needs to be dropped
 360          add_clean(bcx, v, object_ty);
 361          bcx
 362      }
 363  }

 365  pub fn make_free_glue(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
 366      // NB: v0 is an *alias* of type t here, not a direct value.
 367      let _icx = push_ctxt("make_free_glue");
 368      match ty::get(t).sty {
 369        ty::ty_box(body_mt) => {
 370          let v = Load(bcx, v);
 371          let body = GEPi(bcx, v, [0u, abi::box_field_body]);
 372          let bcx = drop_ty(bcx, body, body_mt.ty);
 373          trans_free(bcx, v)
 374        }
 375        ty::ty_opaque_box => {
 376          let v = Load(bcx, v);
 377          let td = Load(bcx, GEPi(bcx, v, [0u, abi::box_field_tydesc]));
 378          let valptr = GEPi(bcx, v, [0u, abi::box_field_body]);
 379          // Generate code that, dynamically, indexes into the
 380          // tydesc and calls the drop glue that got set dynamically
 381          call_tydesc_glue_full(bcx, valptr, td, abi::tydesc_field_drop_glue,
 382                                None);
 383          trans_free(bcx, v)
 384        }
 385        ty::ty_uniq(*) => {
 386          uniq::make_free_glue(bcx, v, t)
 387        }
 388        ty::ty_evec(_, ty::vstore_uniq) | ty::ty_estr(ty::vstore_uniq) |
 389        ty::ty_evec(_, ty::vstore_box) | ty::ty_estr(ty::vstore_box) => {
 390          make_free_glue(bcx, v, tvec::expand_boxed_vec_ty(bcx.tcx(), t))
 391        }
 392        ty::ty_closure(_) => {
 393          closure::make_closure_glue(bcx, v, t, free_ty)
 394        }
 395        ty::ty_opaque_closure_ptr(ck) => {
 396          closure::make_opaque_cbox_free_glue(bcx, ck, v)
 397        }
 398        _ => bcx
 399      }
 400  }

 402  pub fn trans_struct_drop_flag(bcx: @mut Block, t: ty::t, v0: ValueRef, dtor_did: ast::def_id,
 403                                class_did: ast::def_id, substs: &ty::substs) -> @mut Block {
 404      let repr = adt::represent_type(bcx.ccx(), t);
 405      let drop_flag = adt::trans_drop_flag_ptr(bcx, repr, v0);
 406      do with_cond(bcx, IsNotNull(bcx, Load(bcx, drop_flag))) |cx| {
 407          let mut bcx = cx;
 408  
 409          // Find and call the actual destructor
 410          let dtor_addr = get_res_dtor(bcx.ccx(), dtor_did,
 411                                       class_did, substs.tps.clone());
 412  
 413          // The second argument is the "self" argument for drop
 414          let params = unsafe {
 415              let ty = Type::from_ref(llvm::LLVMTypeOf(dtor_addr));
 416              ty.element_type().func_params()
 417          };
 418  
 419          // Class dtors have no explicit args, so the params should
 420          // just consist of the environment (self)
 421          assert_eq!(params.len(), 1);
 422  
 423          let self_arg = PointerCast(bcx, v0, params[0]);
 424          let args = ~[self_arg];
 425  
 426          Call(bcx, dtor_addr, args);
 427  
 428          // Drop the fields
 429          let field_tys = ty::struct_fields(bcx.tcx(), class_did, substs);
 430          for (i, fld) in field_tys.iter().enumerate() {
 431              let llfld_a = adt::trans_field_ptr(bcx, repr, v0, 0, i);
 432              bcx = drop_ty(bcx, llfld_a, fld.mt.ty);
 433          }
 434  
 435          Store(bcx, C_u8(0), drop_flag);
 436          bcx
 437      }
 438  }

 473  pub fn make_drop_glue(bcx: @mut Block, v0: ValueRef, t: ty::t) -> @mut Block {
 474      // NB: v0 is an *alias* of type t here, not a direct value.
 475      let _icx = push_ctxt("make_drop_glue");
 476      let ccx = bcx.ccx();
 477      match ty::get(t).sty {
 478        ty::ty_box(_) | ty::ty_opaque_box |
 479        ty::ty_estr(ty::vstore_box) | ty::ty_evec(_, ty::vstore_box) => {
 480          decr_refcnt_maybe_free(bcx, Load(bcx, v0), Some(v0), t)
 481        }
 482        ty::ty_uniq(_) |
 483        ty::ty_evec(_, ty::vstore_uniq) | ty::ty_estr(ty::vstore_uniq) => {
 484          free_ty(bcx, v0, t)
 485        }
 486        ty::ty_unboxed_vec(_) => {
 487          tvec::make_drop_glue_unboxed(bcx, v0, t)
 488        }
 489        ty::ty_struct(did, ref substs) => {
 490          let tcx = bcx.tcx();
 491          match ty::ty_dtor(tcx, did) {
 492            ty::TraitDtor(dtor, true) => {
 493              trans_struct_drop_flag(bcx, t, v0, dtor, did, substs)
 494            }
 495            ty::TraitDtor(dtor, false) => {
 496              trans_struct_drop(bcx, t, v0, dtor, did, substs)
 497            }
 498            ty::NoDtor => {
 499              // No dtor? Just the default case
 500              iter_structural_ty(bcx, v0, t, drop_ty)
 501            }
 502          }
 503        }
 504        ty::ty_closure(_) => {
 505          closure::make_closure_glue(bcx, v0, t, drop_ty)
 506        }
 507        ty::ty_trait(_, _, ty::BoxTraitStore, _, _) => {
 508            let llbox_ptr = GEPi(bcx, v0, [0u, abi::trt_field_box]);
 509            let llbox = Load(bcx, llbox_ptr);
 510            decr_refcnt_maybe_free(bcx, llbox, Some(llbox_ptr),
 511                                   ty::mk_opaque_box(ccx.tcx))
 512        }
 513        ty::ty_trait(_, _, ty::UniqTraitStore, _, _) => {
 514            let lluniquevalue = GEPi(bcx, v0, [0, abi::trt_field_box]);
 515            // Only drop the value when it is non-null
 516            do with_cond(bcx, IsNotNull(bcx, Load(bcx, lluniquevalue))) |bcx| {
 517                let llvtable = Load(bcx, GEPi(bcx, v0, [0, abi::trt_field_vtable]));
 518  
 519                // Cast the vtable to a pointer to a pointer to a tydesc.
 520                let llvtable = PointerCast(bcx, llvtable,
 521                                           ccx.tydesc_type.ptr_to().ptr_to());
 522                let lltydesc = Load(bcx, llvtable);
 523                call_tydesc_glue_full(bcx,
 524                                      lluniquevalue,
 525                                      lltydesc,
 526                                      abi::tydesc_field_free_glue,
 527                                      None);
 528                bcx
 529            }
 530        }
 531        ty::ty_opaque_closure_ptr(ck) => {
 532          closure::make_opaque_cbox_drop_glue(bcx, ck, v0)
 533        }
 534        _ => {
 535          if ty::type_needs_drop(ccx.tcx, t) &&
 536              ty::type_is_structural(t) {
 537              iter_structural_ty(bcx, v0, t, drop_ty)
 538          } else { bcx }
 539        }
 540      }
 541  }

 543  // box_ptr_ptr is optional, it is constructed if not supplied.
 544  pub fn decr_refcnt_maybe_free(bcx: @mut Block, box_ptr: ValueRef,
 545                                box_ptr_ptr: Option<ValueRef>,
 546                                t: ty::t)
 547                             -> @mut Block {
 548      let _icx = push_ctxt("decr_refcnt_maybe_free");
 549      let ccx = bcx.ccx();
 550  
 551      let decr_bcx = sub_block(bcx, "decr");
 552      let free_bcx = sub_block(decr_bcx, "free");
 553      let next_bcx = sub_block(bcx, "next");
 554      CondBr(bcx, IsNotNull(bcx, box_ptr), decr_bcx.llbb, next_bcx.llbb);
 555  
 556      let rc_ptr = GEPi(decr_bcx, box_ptr, [0u, abi::box_field_refcnt]);
 557      let rc = Sub(decr_bcx, Load(decr_bcx, rc_ptr), C_int(ccx, 1));
 558      Store(decr_bcx, rc, rc_ptr);
 559      CondBr(decr_bcx, IsNull(decr_bcx, rc), free_bcx.llbb, next_bcx.llbb);
 560  
 561      let free_bcx = match box_ptr_ptr {
 562          Some(p) => free_ty(free_bcx, p, t),
 563          None => free_ty_immediate(free_bcx, box_ptr, t)
 564      };
 565      Br(free_bcx, next_bcx.llbb);
 566  
 567      next_bcx
 568  }

 571  pub fn make_take_glue(bcx: @mut Block, v: ValueRef, t: ty::t) -> @mut Block {
 572      let _icx = push_ctxt("make_take_glue");
 573      // NB: v is a *pointer* to type t here, not a direct value.
 574      match ty::get(t).sty {
 575        ty::ty_box(_) | ty::ty_opaque_box |
 576        ty::ty_evec(_, ty::vstore_box) | ty::ty_estr(ty::vstore_box) => {
 577          incr_refcnt_of_boxed(bcx, Load(bcx, v)); bcx
 578        }
 579        ty::ty_evec(_, ty::vstore_slice(_))
 580        | ty::ty_estr(ty::vstore_slice(_)) => {
 581          bcx
 582        }
 583        ty::ty_closure(ty::ClosureTy { sigil: ast::BorrowedSigil, _ }) |
 584        ty::ty_closure(ty::ClosureTy { sigil: ast::ManagedSigil, _ }) => {
 585          closure::make_closure_glue(bcx, v, t, take_ty)
 586        }
 587        ty::ty_closure(ty::ClosureTy { sigil: ast::OwnedSigil, _ }) => bcx,
 588        ty::ty_trait(_, _, ty::BoxTraitStore, _, _) => {
 589          let llbox = Load(bcx, GEPi(bcx, v, [0u, abi::trt_field_box]));
 590          incr_refcnt_of_boxed(bcx, llbox);
 591          bcx
 592        }
 593        ty::ty_trait(_, _, ty::UniqTraitStore, _, _) => {
 594            let lluniquevalue = GEPi(bcx, v, [0, abi::trt_field_box]);
 595            let llvtable = Load(bcx, GEPi(bcx, v, [0, abi::trt_field_vtable]));
 596  
 597            // Cast the vtable to a pointer to a pointer to a tydesc.
 598            let llvtable = PointerCast(bcx, llvtable,
 599                                       bcx.ccx().tydesc_type.ptr_to().ptr_to());
 600            let lltydesc = Load(bcx, llvtable);
 601            call_tydesc_glue_full(bcx,
 602                                  lluniquevalue,
 603                                  lltydesc,
 604                                  abi::tydesc_field_take_glue,
 605                                  None);
 606            bcx
 607        }
 608        ty::ty_opaque_closure_ptr(ck) => {
 609          closure::make_opaque_cbox_take_glue(bcx, ck, v)
 610        }
 611        ty::ty_struct(did, _) => {
 612          let tcx = bcx.tcx();
 613          let bcx = iter_structural_ty(bcx, v, t, take_ty);
 614  
 615          match ty::ty_dtor(tcx, did) {
 616            ty::TraitDtor(_, false) => {
 617              // Zero out the struct
 618              unsafe {
 619                  let ty = Type::from_ref(llvm::LLVMTypeOf(v));
 620                  memzero(&B(bcx), v, ty);
 621              }
 622  
 623            }
 624            _ => { }
 625          }
 626          bcx
 627        }
 628        _ if ty::type_is_structural(t) => {
 629          iter_structural_ty(bcx, v, t, take_ty)
 630        }
 631        _ => bcx
 632      }
 633  }

 695  pub fn make_generic_glue_inner(ccx: @mut CrateContext,
 696                                 t: ty::t,
 697                                 llfn: ValueRef,
 698                                 helper: glue_helper)
 699                              -> ValueRef {
 700      let _icx = push_ctxt("make_generic_glue_inner");
 701      let fcx = new_fn_ctxt(ccx, ~[], llfn, ty::mk_nil(), None);
 702      lib::llvm::SetLinkage(llfn, lib::llvm::InternalLinkage);
 703      ccx.stats.n_glues_created += 1u;
 704      // All glue functions take values passed *by alias*; this is a
 705      // requirement since in many contexts glue is invoked indirectly and
 706      // the caller has no idea if it's dealing with something that can be
 707      // passed by value.
 708      //
 709      // llfn is expected be declared to take a parameter of the appropriate
 710      // type, so we don't need to explicitly cast the function parameter.
 711  
 712      let bcx = fcx.entry_bcx.unwrap();
 713      let rawptr0_arg = fcx.arg_pos(0u);
 714      let llrawptr0 = unsafe { llvm::LLVMGetParam(llfn, rawptr0_arg as c_uint) };
 715      let bcx = helper(bcx, llrawptr0, t);
 716  
 717      finish_fn(fcx, bcx);
 718  
 719      return llfn;
 720  }

 734  pub fn emit_tydescs(ccx: &mut CrateContext) {
 735      let _icx = push_ctxt("emit_tydescs");
 736      // As of this point, allow no more tydescs to be created.
 737      ccx.finished_tydescs = true;
 738      let glue_fn_ty = Type::generic_glue_fn(ccx).ptr_to();
 739      let tyds = &mut ccx.tydescs;
 740      for (_, &val) in tyds.iter() {
 741          let ti = val;
 742  
 743          // Each of the glue functions needs to be cast to a generic type
 744          // before being put into the tydesc because we only have a singleton
 745          // tydesc type. Then we'll recast each function to its real type when
 746          // calling it.
 747          let take_glue =
 748              match ti.take_glue {
 749                None => { ccx.stats.n_null_glues += 1u; C_null(glue_fn_ty) }
 750                Some(v) => {
 751                  unsafe {
 752                      ccx.stats.n_real_glues += 1u;
 753                      llvm::LLVMConstPointerCast(v, glue_fn_ty.to_ref())
 754                  }
 755                }
 756              };
 757          let drop_glue =
 758              match ti.drop_glue {
 759                None => { ccx.stats.n_null_glues += 1u; C_null(glue_fn_ty) }
 760                Some(v) => {
 761                  unsafe {
 762                      ccx.stats.n_real_glues += 1u;
 763                      llvm::LLVMConstPointerCast(v, glue_fn_ty.to_ref())
 764                  }
 765                }
 766              };

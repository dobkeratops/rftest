  37  pub struct Reflector {
  38      visitor_val: ValueRef,
  39      visitor_methods: @~[@ty::Method],
  40      final_bcx: @mut Block,
  41      tydesc_ty: Type,
  42      bcx: @mut Block
  43  }

  45  impl Reflector {
  46      pub fn c_uint(&mut self, u: uint) -> ValueRef {
  47          C_uint(self.bcx.ccx(), u)
  48      }
  49  
  50      pub fn c_int(&mut self, i: int) -> ValueRef {
  51          C_int(self.bcx.ccx(), i)
  52      }

  54      pub fn c_slice(&mut self, s: @str) -> ValueRef {
  55          // We're careful to not use first class aggregates here because that
  56          // will kick us off fast isel. (Issue #4352.)
  57          let bcx = self.bcx;
  58          let str_vstore = ty::vstore_slice(ty::re_static);
  59          let str_ty = ty::mk_estr(bcx.tcx(), str_vstore);
  60          let scratch = scratch_datum(bcx, str_ty, "", false);
  61          let len = C_uint(bcx.ccx(), s.len());
  62          let c_str = PointerCast(bcx, C_cstr(bcx.ccx(), s), Type::i8p());
  63          Store(bcx, c_str, GEPi(bcx, scratch.val, [ 0, 0 ]));
  64          Store(bcx, len, GEPi(bcx, scratch.val, [ 0, 1 ]));
  65          scratch.val
  66      }
  67  
  68      pub fn c_size_and_align(&mut self, t: ty::t) -> ~[ValueRef] {
  69          let tr = type_of(self.bcx.ccx(), t);
  70          let s = machine::llsize_of_real(self.bcx.ccx(), tr);
  71          let a = machine::llalign_of_min(self.bcx.ccx(), tr);
  72          return ~[self.c_uint(s),
  73               self.c_uint(a)];
  74      }

  76      pub fn c_tydesc(&mut self, t: ty::t) -> ValueRef {
  77          let bcx = self.bcx;
  78          let static_ti = get_tydesc(bcx.ccx(), t);
  79          glue::lazily_emit_all_tydesc_glue(bcx.ccx(), static_ti);
  80          PointerCast(bcx, static_ti.tydesc, self.tydesc_ty.ptr_to())
  81      }

  83      pub fn c_mt(&mut self, mt: &ty::mt) -> ~[ValueRef] {
  84          ~[self.c_uint(mt.mutbl as uint),
  85            self.c_tydesc(mt.ty)]
  86      }

  88      pub fn visit(&mut self, ty_name: &str, args: &[ValueRef]) {
  89          let tcx = self.bcx.tcx();
  90          let mth_idx = ty::method_idx(
  91              tcx.sess.ident_of(~"visit_" + ty_name),
  92              *self.visitor_methods).expect(fmt!("Couldn't find visit method \
  93                                                  for %s", ty_name));
  94          let mth_ty =
  95              ty::mk_bare_fn(tcx, self.visitor_methods[mth_idx].fty.clone());
  96          let v = self.visitor_val;
  97          debug!("passing %u args:", args.len());
  98          let mut bcx = self.bcx;
  99          for (i, a) in args.iter().enumerate() {
 100              debug!("arg %u: %s", i, bcx.val_to_str(*a));
 101          }
 102          let bool_ty = ty::mk_bool();
 103          let result = unpack_result!(bcx, callee::trans_call_inner(
 104              self.bcx, None, mth_ty, bool_ty,
 105              |bcx| meth::trans_trait_callee_from_llval(bcx,
 106                                                        mth_ty,
 107                                                        mth_idx,
 108                                                        v,
 109                                                        None),
 110              ArgVals(args), None, DontAutorefArg));
 111          let result = bool_to_i1(bcx, result);
 112          let next_bcx = sub_block(bcx, "next");
 113          CondBr(bcx, result, next_bcx.llbb, self.final_bcx.llbb);
 114          self.bcx = next_bcx
 115      }

 117      pub fn bracketed(&mut self,
 118                       bracket_name: &str,
 119                       extra: &[ValueRef],
 120                       inner: &fn(&mut Reflector)) {
 121          self.visit("enter_" + bracket_name, extra);
 122          inner(self);
 123          self.visit("leave_" + bracket_name, extra);
 124      }

 126      pub fn vstore_name_and_extra(&mut self,
 127                                   t: ty::t,
 128                                   vstore: ty::vstore)
 129                                   -> (~str, ~[ValueRef]) {
 130          match vstore {
 131              ty::vstore_fixed(n) => {
 132                  let extra = vec::append(~[self.c_uint(n)],
 133                                          self.c_size_and_align(t));
 134                  (~"fixed", extra)
 135              }
 136              ty::vstore_slice(_) => (~"slice", ~[]),
 137              ty::vstore_uniq => (~"uniq", ~[]),
 138              ty::vstore_box => (~"box", ~[])
 139          }
 140      }

 142      pub fn leaf(&mut self, name: &str) {
 143          self.visit(name, []);
 144      }

 146      // Entrypoint
 147      pub fn visit_ty(&mut self, t: ty::t) {
 148          let bcx = self.bcx;
 149          debug!("reflect::visit_ty %s", ty_to_str(bcx.ccx().tcx, t));
 150  
 151          match ty::get(t).sty {
 152            ty::ty_bot => self.leaf("bot"),
 153            ty::ty_nil => self.leaf("nil"),
 154            ty::ty_bool => self.leaf("bool"),
 155            ty::ty_int(ast::ty_i) => self.leaf("int"),
 156            ty::ty_int(ast::ty_char) => self.leaf("char"),
 157            ty::ty_int(ast::ty_i8) => self.leaf("i8"),
 158            ty::ty_int(ast::ty_i16) => self.leaf("i16"),
 159            ty::ty_int(ast::ty_i32) => self.leaf("i32"),
 160            ty::ty_int(ast::ty_i64) => self.leaf("i64"),
 161            ty::ty_uint(ast::ty_u) => self.leaf("uint"),
 162            ty::ty_uint(ast::ty_u8) => self.leaf("u8"),
 163            ty::ty_uint(ast::ty_u16) => self.leaf("u16"),
 164            ty::ty_uint(ast::ty_u32) => self.leaf("u32"),
 165            ty::ty_uint(ast::ty_u64) => self.leaf("u64"),
 166            ty::ty_float(ast::ty_f) => self.leaf("float"),
 167            ty::ty_float(ast::ty_f32) => self.leaf("f32"),
 168            ty::ty_float(ast::ty_f64) => self.leaf("f64"),
 169  
 170            ty::ty_unboxed_vec(ref mt) => {
 171                let values = self.c_mt(mt);
 172                self.visit("vec", values)
 173            }
 174  
 175            ty::ty_estr(vst) => {
 176                let (name, extra) = self.vstore_name_and_extra(t, vst);
 177                self.visit(~"estr_" + name, extra)
 178            }
 179            ty::ty_evec(ref mt, vst) => {
 180                let (name, extra) = self.vstore_name_and_extra(t, vst);
 181                let extra = extra + self.c_mt(mt);
 182                if "uniq" == name && ty::type_contents(bcx.tcx(), t).contains_managed() {
 183                    self.visit("evec_uniq_managed", extra)
 184                } else {
 185                    self.visit(~"evec_" + name, extra)
 186                }
 187            }
 188            ty::ty_box(ref mt) => {
 189                let extra = self.c_mt(mt);
 190                self.visit("box", extra)
 191            }
 192            ty::ty_uniq(ref mt) => {
 193                let extra = self.c_mt(mt);
 194                if ty::type_contents(bcx.tcx(), t).contains_managed() {
 195                    self.visit("uniq_managed", extra)
 196                } else {
 197                    self.visit("uniq", extra)
 198                }
 199            }
 200            ty::ty_ptr(ref mt) => {
 201                let extra = self.c_mt(mt);
 202                self.visit("ptr", extra)
 203            }
 204            ty::ty_rptr(_, ref mt) => {
 205                let extra = self.c_mt(mt);
 206                self.visit("rptr", extra)
 207            }
 208  
 209            ty::ty_tup(ref tys) => {
 210                let extra = ~[self.c_uint(tys.len())]
 211                    + self.c_size_and_align(t);
 212                do self.bracketed("tup", extra) |this| {
 213                    for (i, t) in tys.iter().enumerate() {
 214                        let extra = ~[this.c_uint(i), this.c_tydesc(*t)];
 215                        this.visit("tup_field", extra);
 216                    }
 217                }
 218            }
 219  
 220            // FIXME (#2594): fetch constants out of intrinsic
 221            // FIXME (#4809): visitor should break out bare fns from other fns
 222            ty::ty_closure(ref fty) => {
 223              let pureval = ast_purity_constant(fty.purity);
 224              let sigilval = ast_sigil_constant(fty.sigil);
 225              let retval = if ty::type_is_bot(fty.sig.output) {0u} else {1u};
 226              let extra = ~[self.c_uint(pureval),
 227                            self.c_uint(sigilval),
 228                            self.c_uint(fty.sig.inputs.len()),
 229                            self.c_uint(retval)];
 230              self.visit("enter_fn", extra);
 231              self.visit_sig(retval, &fty.sig);
 232              self.visit("leave_fn", extra);
 233            }
 234  
 235            // FIXME (#2594): fetch constants out of intrinsic:: for the
 236            // numbers.
 237            ty::ty_bare_fn(ref fty) => {
 238              let pureval = ast_purity_constant(fty.purity);
 239              let sigilval = 0u;
 240              let retval = if ty::type_is_bot(fty.sig.output) {0u} else {1u};
 241              let extra = ~[self.c_uint(pureval),
 242                            self.c_uint(sigilval),
 243                            self.c_uint(fty.sig.inputs.len()),
 244                            self.c_uint(retval)];
 245              self.visit("enter_fn", extra);
 246              self.visit_sig(retval, &fty.sig);
 247              self.visit("leave_fn", extra);
 248            }
 249  
 250            ty::ty_struct(did, ref substs) => {
 251                let bcx = self.bcx;
 252                let tcx = bcx.ccx().tcx;
 253                let fields = ty::struct_fields(tcx, did, substs);
 254  
 255                let extra = ~[self.c_uint(fields.len())]
 256                    + self.c_size_and_align(t);
 257                do self.bracketed("class", extra) |this| {
 258                    for (i, field) in fields.iter().enumerate() {
 259                        let extra = ~[this.c_uint(i),
 260                                      this.c_slice(
 261                                          bcx.ccx().sess.str_of(field.ident))]
 262                            + this.c_mt(&field.mt);
 263                        this.visit("class_field", extra);
 264                    }
 265                }
 266            }
 267  
 268            // FIXME (#2595): visiting all the variants in turn is probably
 269            // not ideal. It'll work but will get costly on big enums. Maybe
 270            // let the visitor tell us if it wants to visit only a particular
 271            // variant?
 272            ty::ty_enum(did, ref substs) => {
 273              let bcx = self.bcx;
 274              let ccx = bcx.ccx();
 275              let repr = adt::represent_type(bcx.ccx(), t);
 276              let variants = ty::substd_enum_variants(ccx.tcx, did, substs);
 277              let llptrty = type_of(ccx, t).ptr_to();
 278              let opaquety = ty::get_opaque_ty(ccx.tcx).unwrap();
 279              let opaqueptrty = ty::mk_ptr(ccx.tcx, ty::mt { ty: opaquety, mutbl: ast::m_imm });
 280  
 281              let make_get_disr = || {
 282                  let sub_path = bcx.fcx.path + &[path_name(special_idents::anon)];
 283                  let sym = mangle_internal_name_by_path_and_seq(ccx,
 284                                                                 sub_path,
 285                                                                 "get_disr");
 286  
 287                  let llfty = type_of_fn(ccx, [opaqueptrty], ty::mk_int());
 288                  let llfdecl = decl_internal_cdecl_fn(ccx.llmod, sym, llfty);
 289                  let fcx = new_fn_ctxt(ccx,
 290                                        ~[],
 291                                        llfdecl,
 292                                        ty::mk_uint(),
 293                                        None);
 294                  let arg = unsafe {
 295                      //
 296                      // we know the return type of llfdecl is an int here, so
 297                      // no need for a special check to see if the return type
 298                      // is immediate.
 299                      //
 300                      llvm::LLVMGetParam(llfdecl, fcx.arg_pos(0u) as c_uint)
 301                  };
 302                  let mut bcx = fcx.entry_bcx.unwrap();
 303                  let arg = BitCast(bcx, arg, llptrty);
 304                  let ret = adt::trans_get_discr(bcx, repr, arg);
 305                  Store(bcx, ret, fcx.llretptr.unwrap());
 306                  match fcx.llreturn {
 307                      Some(llreturn) => cleanup_and_Br(bcx, bcx, llreturn),
 308                      None => bcx = cleanup_block(bcx, Some(bcx.llbb))
 309                  };
 310                  finish_fn(fcx, bcx);
 311                  llfdecl
 312              };
 313  
 314              let enum_args = ~[self.c_uint(variants.len()), make_get_disr()]
 315                  + self.c_size_and_align(t);
 316              do self.bracketed("enum", enum_args) |this| {
 317                  for (i, v) in variants.iter().enumerate() {
 318                      let name = ccx.sess.str_of(v.name);
 319                      let variant_args = ~[this.c_uint(i),
 320                                           this.c_uint(v.disr_val),
 321                                           this.c_uint(v.args.len()),
 322                                           this.c_slice(name)];
 323                      do this.bracketed("enum_variant", variant_args) |this| {
 324                          for (j, a) in v.args.iter().enumerate() {
 325                              let bcx = this.bcx;
 326                              let null = C_null(llptrty);
 327                              let ptr = adt::trans_field_ptr(bcx, repr, null, v.disr_val, j);
 328                              let offset = p2i(ccx, ptr);
 329                              let field_args = ~[this.c_uint(j),
 330                                                 offset,
 331                                                 this.c_tydesc(*a)];
 332                              this.visit("enum_variant_field", field_args);
 333                          }
 334                      }
 335                  }
 336              }
 337            }
 338  
 339            // Miscallaneous extra types
 340            ty::ty_trait(_, _, _, _, _) => self.leaf("trait"),
 341            ty::ty_infer(_) => self.leaf("infer"),
 342            ty::ty_err => self.leaf("err"),
 343            ty::ty_param(ref p) => {
 344                let extra = ~[self.c_uint(p.idx)];
 345                self.visit("param", extra)
 346            }
 347            ty::ty_self(*) => self.leaf("self"),
 348            ty::ty_type => self.leaf("type"),
 349            ty::ty_opaque_box => self.leaf("opaque_box"),
 350            ty::ty_opaque_closure_ptr(ck) => {
 351                let ckval = ast_sigil_constant(ck);
 352                let extra = ~[self.c_uint(ckval)];
 353                self.visit("closure_ptr", extra)
 354            }
 355          }
 356      }

 358      pub fn visit_sig(&mut self, retval: uint, sig: &ty::FnSig) {
 359          for (i, arg) in sig.inputs.iter().enumerate() {
 360              let modeval = 5u;   // "by copy"
 361              let extra = ~[self.c_uint(i),
 362                           self.c_uint(modeval),
 363                           self.c_tydesc(*arg)];
 364              self.visit("fn_input", extra);
 365          }
 366          let extra = ~[self.c_uint(retval),
 367                        self.c_tydesc(sig.output)];
 368          self.visit("fn_output", extra);
 369      }
 370  }

 372  // Emit a sequence of calls to visit_ty::visit_foo
 373  pub fn emit_calls_to_trait_visit_ty(bcx: @mut Block,
 374                                      t: ty::t,
 375                                      visitor_val: ValueRef,
 376                                      visitor_trait_id: def_id)
 377                                   -> @mut Block {
 378      let final = sub_block(bcx, "final");
 379      let tydesc_ty = ty::get_tydesc_ty(bcx.ccx().tcx).unwrap();
 380      let tydesc_ty = type_of(bcx.ccx(), tydesc_ty);
 381      let mut r = Reflector {
 382          visitor_val: visitor_val,
 383          visitor_methods: ty::trait_methods(bcx.tcx(), visitor_trait_id),
 384          final_bcx: final,
 385          tydesc_ty: tydesc_ty,
 386          bcx: bcx
 387      };
 388      r.visit_ty(t);
 389      Br(r.bcx, final.llbb);
 390      return final;
 391  }

  53      pub fn c_slice(&mut self, s: InternedString) -> ValueRef {
  54          // We're careful to not use first class aggregates here because that
  55          // will kick us off fast isel. (Issue #4352.)
  56          let bcx = self.bcx;
  57          let str_vstore = ty::VstoreSlice(ty::ReStatic, ());
  58          let str_ty = ty::mk_str(bcx.tcx(), str_vstore);
  59          let scratch = rvalue_scratch_datum(bcx, str_ty, "");
  60          let len = C_uint(bcx.ccx(), s.get().len());
  61          let c_str = PointerCast(bcx, C_cstr(bcx.ccx(), s, false), Type::i8p(bcx.ccx()));
  62          Store(bcx, c_str, GEPi(bcx, scratch.val, [ 0, 0 ]));
  63          Store(bcx, len, GEPi(bcx, scratch.val, [ 0, 1 ]));
  64          scratch.val
  65      }

  67      pub fn c_size_and_align(&mut self, t: ty::t) -> Vec<ValueRef> {
  68          let tr = type_of(self.bcx.ccx(), t);
  69          let s = machine::llsize_of_real(self.bcx.ccx(), tr);
  70          let a = machine::llalign_of_min(self.bcx.ccx(), tr);

  72               self.c_uint(a as uint));
  73      }

  87      pub fn visit(&mut self, ty_name: &str, args: &[ValueRef]) {
  88          let fcx = self.bcx.fcx;
  89          let tcx = self.bcx.tcx();
  90          let mth_idx = ty::method_idx(token::str_to_ident("visit_".to_owned() + ty_name),
  91                                       self.visitor_methods.as_slice()).expect(
  92                  format!("couldn't find visit method for {}", ty_name));
  93          let mth_ty =
  94              ty::mk_bare_fn(tcx,
  95                             self.visitor_methods.get(mth_idx).fty.clone());
  96          let v = self.visitor_val;
  97          debug!("passing {} args:", args.len());
  98          let mut bcx = self.bcx;
  99          for (i, a) in args.iter().enumerate() {
 100              debug!("arg {}: {}", i, bcx.val_to_str(*a));
 101          }
 102          let result = unpack_result!(bcx, callee::trans_call_inner(
 103              self.bcx, None, mth_ty,
 104              |bcx, _| meth::trans_trait_callee_from_llval(bcx,
 105                                                           mth_ty,
 106                                                           mth_idx,
 107                                                           v),
 108              ArgVals(args), None));
 109          let result = bool_to_i1(bcx, result);
 110          let next_bcx = fcx.new_temp_block("next");
 111          CondBr(bcx, result, next_bcx.llbb, self.final_bcx.llbb);
 112          self.bcx = next_bcx
 113      }

 115      pub fn bracketed(&mut self,
 116                       bracket_name: &str,
 117                       extra: &[ValueRef],
 118                       inner: |&mut Reflector|) {
 119          self.visit("enter_" + bracket_name, extra);

 121          self.visit("leave_" + bracket_name, extra);

 125                                      t: ty::t,

 156            ty::ty_int(ast::TyI32) => self.leaf("i32"),

 159            ty::ty_uint(ast::TyU8) => self.leaf("u8"),
 160            ty::ty_uint(ast::TyU16) => self.leaf("u16"),
 161            ty::ty_uint(ast::TyU32) => self.leaf("u32"),

 171            ty::ty_vec(ty, vst) => {

 173                let extra = extra.append(self.c_mt(&ty::mt {
 174                    ty: ty,
 175                    mutbl: match vst {
 176                        ty::VstoreSlice(_, m) => m,
 177                        _ => ast::MutImmutable
 178                    }
 179                }).as_slice());
 180                self.visit("evec_".to_owned() + name, extra.as_slice())
 181            }
 182            // Should remove mt from box and uniq.
 183            ty::ty_box(typ) => {
 184                let extra = self.c_mt(&ty::mt {
 185                    ty: typ,
 186                    mutbl: ast::MutImmutable,
 187                });
 188                self.visit("box", extra.as_slice())
 189            }
 190            ty::ty_uniq(typ) => {
 191                let extra = self.c_mt(&ty::mt {
 192                    ty: typ,
 193                    mutbl: ast::MutImmutable,
 194                });
 195                self.visit("uniq", extra.as_slice())
 196            }
 197            ty::ty_ptr(ref mt) => {

 199                self.visit("ptr", extra.as_slice())
 200            }

 380  // Emit a sequence of calls to visit_ty::visit_foo
 381  pub fn emit_calls_to_trait_visit_ty<'a>(
 382                                      bcx: &'a Block<'a>,
 383                                      t: ty::t,
 384                                      visitor_val: ValueRef,
 385                                      visitor_trait_id: DefId)
 386                                      -> &'a Block<'a> {
 387      let fcx = bcx.fcx;
 388      let final = fcx.new_temp_block("final");
 389      let tydesc_ty = ty::get_tydesc_ty(bcx.tcx()).unwrap();
 390      let tydesc_ty = type_of(bcx.ccx(), tydesc_ty);

  36  pub struct Reflector<'a> {
  37      visitor_val: ValueRef,
  38      visitor_methods: @Vec<@ty::Method> ,
  39      final_bcx: &'a Block<'a>,
  40      tydesc_ty: Type,
  41      bcx: &'a Block<'a>
  42  }

  45      pub fn c_uint(&mut self, u: uint) -> ValueRef {
  46          C_uint(self.bcx.ccx(), u)
  47      }

  49      pub fn c_bool(&mut self, b: bool) -> ValueRef {
  50          C_bool(self.bcx.ccx(), b)
  51      }

  53      pub fn c_slice(&mut self, s: InternedString) -> ValueRef {
  54          // We're careful to not use first class aggregates here because that
  55          // will kick us off fast isel. (Issue #4352.)
  56          let bcx = self.bcx;
  57          let str_vstore = ty::vstore_slice(ty::ReStatic);
  58          let str_ty = ty::mk_str(bcx.tcx(), str_vstore);
  59          let scratch = rvalue_scratch_datum(bcx, str_ty, "");
  60          let len = C_uint(bcx.ccx(), s.get().len());
  61          let c_str = PointerCast(bcx, C_cstr(bcx.ccx(), s, false), Type::i8p(bcx.ccx()));
  62          Store(bcx, c_str, GEPi(bcx, scratch.val, [ 0, 0 ]));
  63          Store(bcx, len, GEPi(bcx, scratch.val, [ 0, 1 ]));
  64          scratch.val
  65      }

  67      pub fn c_size_and_align(&mut self, t: ty::t) -> Vec<ValueRef> {
  68          let tr = type_of(self.bcx.ccx(), t);
  69          let s = machine::llsize_of_real(self.bcx.ccx(), tr);
  70          let a = machine::llalign_of_min(self.bcx.ccx(), tr);
  71          return vec!(self.c_uint(s as uint),
  72               self.c_uint(a as uint));
  73      }

  75      pub fn c_tydesc(&mut self, t: ty::t) -> ValueRef {
  76          let bcx = self.bcx;
  77          let static_ti = get_tydesc(bcx.ccx(), t);
  78          glue::lazily_emit_visit_glue(bcx.ccx(), static_ti);
  79          PointerCast(bcx, static_ti.tydesc, self.tydesc_ty.ptr_to())
  80      }

  82      pub fn c_mt(&mut self, mt: &ty::mt) -> Vec<ValueRef> {
  83          vec!(self.c_uint(mt.mutbl as uint),
  84            self.c_tydesc(mt.ty))
  85      }

  87      pub fn visit(&mut self, ty_name: &str, args: &[ValueRef]) {
  88          let fcx = self.bcx.fcx;
  89          let tcx = self.bcx.tcx();
  90          let mth_idx = ty::method_idx(token::str_to_ident(~"visit_" + ty_name),
  91                                       self.visitor_methods.as_slice()).expect(
  92                  format!("couldn't find visit method for {}", ty_name));
  93          let mth_ty =
  94              ty::mk_bare_fn(tcx,
  95                             self.visitor_methods.get(mth_idx).fty.clone());
  96          let v = self.visitor_val;
  97          debug!("passing {} args:", args.len());
  98          let mut bcx = self.bcx;
  99          for (i, a) in args.iter().enumerate() {
 100              debug!("arg {}: {}", i, bcx.val_to_str(*a));
 101          }
 102          let result = unpack_result!(bcx, callee::trans_call_inner(
 103              self.bcx, None, mth_ty,
 104              |bcx, _| meth::trans_trait_callee_from_llval(bcx,
 105                                                           mth_ty,
 106                                                           mth_idx,
 107                                                           v),
 108              ArgVals(args), None));
 109          let result = bool_to_i1(bcx, result);
 110          let next_bcx = fcx.new_temp_block("next");
 111          CondBr(bcx, result, next_bcx.llbb, self.final_bcx.llbb);
 112          self.bcx = next_bcx
 113      }

 115      pub fn bracketed(&mut self,
 116                       bracket_name: &str,
 117                       extra: &[ValueRef],
 118                       inner: |&mut Reflector|) {
 119          self.visit("enter_" + bracket_name, extra);
 120          inner(self);
 121          self.visit("leave_" + bracket_name, extra);
 122      }

 124      pub fn vstore_name_and_extra(&mut self,
 125                                   t: ty::t,
 126                                   vstore: ty::vstore)
 127                                   -> (~str, Vec<ValueRef> ) {
 128          match vstore {
 129              ty::vstore_fixed(n) => {
 130                  let extra = (vec!(self.c_uint(n))).append(self.c_size_and_align(t).as_slice());
 131                  (~"fixed", extra)
 132              }
 133              ty::vstore_slice(_) => (~"slice", Vec::new()),
 134              ty::vstore_uniq => (~"uniq", Vec::new()),
 135          }
 136      }

 142      // Entrypoint
 143      pub fn visit_ty(&mut self, t: ty::t) {
 144          let bcx = self.bcx;
 145          let tcx = bcx.tcx();
 146          debug!("reflect::visit_ty {}", ty_to_str(bcx.tcx(), t));
 147  
 148          match ty::get(t).sty {
 149            ty::ty_bot => self.leaf("bot"),
 150            ty::ty_nil => self.leaf("nil"),
 151            ty::ty_bool => self.leaf("bool"),
 152            ty::ty_char => self.leaf("char"),
 153            ty::ty_int(ast::TyI) => self.leaf("int"),
 154            ty::ty_int(ast::TyI8) => self.leaf("i8"),
 155            ty::ty_int(ast::TyI16) => self.leaf("i16"),
 156            ty::ty_int(ast::TyI32) => self.leaf("i32"),
 157            ty::ty_int(ast::TyI64) => self.leaf("i64"),
 158            ty::ty_uint(ast::TyU) => self.leaf("uint"),
 159            ty::ty_uint(ast::TyU8) => self.leaf("u8"),
 160            ty::ty_uint(ast::TyU16) => self.leaf("u16"),
 161            ty::ty_uint(ast::TyU32) => self.leaf("u32"),
 162            ty::ty_uint(ast::TyU64) => self.leaf("u64"),
 163            ty::ty_float(ast::TyF32) => self.leaf("f32"),
 164            ty::ty_float(ast::TyF64) => self.leaf("f64"),
 165  
 166            // Should rename to str_*/vec_*.
 167            ty::ty_str(vst) => {
 168                let (name, extra) = self.vstore_name_and_extra(t, vst);
 169                self.visit(~"estr_" + name, extra.as_slice())
 170            }
 171            ty::ty_vec(ref mt, vst) => {
 172                let (name, extra) = self.vstore_name_and_extra(t, vst);
 173                let extra = extra.append(self.c_mt(mt).as_slice());
 174                self.visit(~"evec_" + name, extra.as_slice())
 175            }
 176            // Should remove mt from box and uniq.
 177            ty::ty_box(typ) => {
 178                let extra = self.c_mt(&ty::mt {
 179                    ty: typ,
 180                    mutbl: ast::MutImmutable,
 181                });
 182                self.visit("box", extra.as_slice())
 183            }
 184            ty::ty_uniq(typ) => {
 185                let extra = self.c_mt(&ty::mt {
 186                    ty: typ,
 187                    mutbl: ast::MutImmutable,
 188                });
 189                self.visit("uniq", extra.as_slice())
 190            }
 191            ty::ty_ptr(ref mt) => {
 192                let extra = self.c_mt(mt);
 193                self.visit("ptr", extra.as_slice())
 194            }
 195            ty::ty_rptr(_, ref mt) => {
 196                let extra = self.c_mt(mt);
 197                self.visit("rptr", extra.as_slice())
 198            }
 199  
 200            ty::ty_tup(ref tys) => {
 201                let extra = (vec!(self.c_uint(tys.len())))
 202                            .append(self.c_size_and_align(t).as_slice());
 203                self.bracketed("tup", extra.as_slice(), |this| {
 204                    for (i, t) in tys.iter().enumerate() {
 205                        let extra = vec!(this.c_uint(i), this.c_tydesc(*t));
 206                        this.visit("tup_field", extra.as_slice());
 207                    }
 208                })
 209            }
 210  
 211            // FIXME (#2594): fetch constants out of intrinsic
 212            // FIXME (#4809): visitor should break out bare fns from other fns
 213            ty::ty_closure(ref fty) => {
 214              let pureval = ast_purity_constant(fty.purity);
 215              let sigilval = ast_sigil_constant(fty.sigil);
 216              let retval = if ty::type_is_bot(fty.sig.output) {0u} else {1u};
 217              let extra = vec!(self.c_uint(pureval),
 218                            self.c_uint(sigilval),
 219                            self.c_uint(fty.sig.inputs.len()),
 220                            self.c_uint(retval));
 221              self.visit("enter_fn", extra.as_slice());
 222              self.visit_sig(retval, &fty.sig);
 223              self.visit("leave_fn", extra.as_slice());
 224            }
 225  
 226            // FIXME (#2594): fetch constants out of intrinsic:: for the
 227            // numbers.
 228            ty::ty_bare_fn(ref fty) => {
 229              let pureval = ast_purity_constant(fty.purity);
 230              let sigilval = 0u;
 231              let retval = if ty::type_is_bot(fty.sig.output) {0u} else {1u};
 232              let extra = vec!(self.c_uint(pureval),
 233                            self.c_uint(sigilval),
 234                            self.c_uint(fty.sig.inputs.len()),
 235                            self.c_uint(retval));
 236              self.visit("enter_fn", extra.as_slice());
 237              self.visit_sig(retval, &fty.sig);
 238              self.visit("leave_fn", extra.as_slice());
 239            }
 240  
 241            ty::ty_struct(did, ref substs) => {
 242                let fields = ty::struct_fields(tcx, did, substs);
 243                let mut named_fields = false;
 244                if !fields.is_empty() {
 245                    named_fields = fields.get(0).ident.name !=
 246                        special_idents::unnamed_field.name;
 247                }
 248  
 249                let extra = (vec!(
 250                    self.c_slice(token::intern_and_get_ident(ty_to_str(tcx,
 251                                                                       t))),
 252                    self.c_bool(named_fields),
 253                    self.c_uint(fields.len())
 254                )).append(self.c_size_and_align(t).as_slice());
 255                self.bracketed("class", extra.as_slice(), |this| {
 256                    for (i, field) in fields.iter().enumerate() {
 257                        let extra = (vec!(
 258                          this.c_uint(i),
 259                          this.c_slice(token::get_ident(field.ident)),
 260                          this.c_bool(named_fields)
 261                        )).append(this.c_mt(&field.mt).as_slice());
 262                        this.visit("class_field", extra.as_slice());
 263                    }
 264                })
 265            }
 266  
 267            // FIXME (#2595): visiting all the variants in turn is probably
 268            // not ideal. It'll work but will get costly on big enums. Maybe
 269            // let the visitor tell us if it wants to visit only a particular
 270            // variant?
 271            ty::ty_enum(did, ref substs) => {
 272              let ccx = bcx.ccx();
 273              let repr = adt::represent_type(bcx.ccx(), t);
 274              let variants = ty::substd_enum_variants(ccx.tcx(), did, substs);
 275              let llptrty = type_of(ccx, t).ptr_to();
 276              let opaquety = ty::get_opaque_ty(ccx.tcx()).unwrap();
 277              let opaqueptrty = ty::mk_ptr(ccx.tcx(), ty::mt { ty: opaquety,
 278                                                             mutbl: ast::MutImmutable });
 279  
 280              let make_get_disr = || {
 281                  let sym = mangle_internal_name_by_path_and_seq(
 282                      ast_map::Values([].iter()).chain(None), "get_disr");
 283  
 284                  let llfdecl = decl_internal_rust_fn(ccx, false, [opaqueptrty], ty::mk_u64(), sym);
 285                  let arena = TypedArena::new();
 286                  let fcx = new_fn_ctxt(ccx, llfdecl, -1, false,
 287                                        ty::mk_u64(), None, None, &arena);
 288                  init_function(&fcx, false, ty::mk_u64(), None);
 289  
 290                  let arg = unsafe {
 291                      //
 292                      // we know the return type of llfdecl is an int here, so
 293                      // no need for a special check to see if the return type
 294                      // is immediate.
 295                      //
 296                      llvm::LLVMGetParam(llfdecl, fcx.arg_pos(0u) as c_uint)
 297                  };
 298                  let bcx = fcx.entry_bcx.borrow().clone().unwrap();
 299                  let arg = BitCast(bcx, arg, llptrty);
 300                  let ret = adt::trans_get_discr(bcx, repr, arg, Some(Type::i64(ccx)));
 301                  Store(bcx, ret, fcx.llretptr.get().unwrap());
 302                  match fcx.llreturn.get() {
 303                      Some(llreturn) => Br(bcx, llreturn),
 304                      None => {}
 305                  };
 306                  finish_fn(&fcx, bcx);
 307                  llfdecl
 308              };
 309  
 310              let enum_args = (vec!(self.c_uint(variants.len()), make_get_disr()))
 311                              .append(self.c_size_and_align(t).as_slice());
 312              self.bracketed("enum", enum_args.as_slice(), |this| {
 313                  for (i, v) in variants.iter().enumerate() {
 314                      let name = token::get_ident(v.name);
 315                      let variant_args = [this.c_uint(i),
 316                                           C_u64(ccx, v.disr_val),
 317                                           this.c_uint(v.args.len()),
 318                                           this.c_slice(name)];
 319                      this.bracketed("enum_variant",
 320                                     variant_args,
 321                                     |this| {
 322                          for (j, a) in v.args.iter().enumerate() {
 323                              let bcx = this.bcx;
 324                              let null = C_null(llptrty);
 325                              let ptr = adt::trans_field_ptr(bcx, repr, null, v.disr_val, j);
 326                              let offset = p2i(ccx, ptr);
 327                              let field_args = [this.c_uint(j),
 328                                                 offset,
 329                                                 this.c_tydesc(*a)];
 330                              this.visit("enum_variant_field",
 331                                         field_args);
 332                          }
 333                      })
 334                  }
 335              })
 336            }
 337  
 338            ty::ty_trait(..) => {
 339                let extra = [
 340                    self.c_slice(token::intern_and_get_ident(ty_to_str(tcx, t)))
 341                ];

 343            }
 344

 346            ty::ty_infer(_) => self.leaf("infer"),
 347            ty::ty_err => self.leaf("err"),

 350                self.visit("param", extra.as_slice())
 351            }

 353          }

 356      pub fn visit_sig(&mut self, retval: uint, sig: &ty::FnSig) {
 357          for (i, arg) in sig.inputs.iter().enumerate() {
 358              let modeval = 5u;   // "by copy"
 359              let extra = vec!(self.c_uint(i),
 360                           self.c_uint(modeval),
 361                           self.c_tydesc(*arg));
 362              self.visit("fn_input", extra.as_slice());
 363          }
 364          let extra = vec!(self.c_uint(retval),
 365                        self.c_bool(sig.variadic),
 366                        self.c_tydesc(sig.output));
 367          self.visit("fn_output", extra.as_slice());
 368      }

 371  // Emit a sequence of calls to visit_ty::visit_foo
 372  pub fn emit_calls_to_trait_visit_ty<'a>(
 373                                      bcx: &'a Block<'a>,
 374                                      t: ty::t,
 375                                      visitor_val: ValueRef,
 376                                      visitor_trait_id: DefId)
 377                                      -> &'a Block<'a> {
 378      let fcx = bcx.fcx;
 379      let final = fcx.new_temp_block("final");
 380      let tydesc_ty = ty::get_tydesc_ty(bcx.tcx()).unwrap();
 381      let tydesc_ty = type_of(bcx.ccx(), tydesc_ty);
 382      let mut r = Reflector {
 383          visitor_val: visitor_val,
 384          visitor_methods: ty::trait_methods(bcx.tcx(), visitor_trait_id),
 385          final_bcx: final,
 386          tydesc_ty: tydesc_ty,
 387          bcx: bcx
 388      };
 389      r.visit_ty(t);
 390      Br(r.bcx, final.llbb);
 391      return final;
 392  }

 394  pub fn ast_sigil_constant(sigil: ast::Sigil) -> uint {
 395      match sigil {

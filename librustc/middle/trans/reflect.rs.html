  58      pub fn c_slice(&mut self, s: @str) -> ValueRef {
  59          // We're careful to not use first class aggregates here because that
  60          // will kick us off fast isel. (Issue #4352.)
  61          let bcx = self.bcx;
  62          let str_vstore = ty::vstore_slice(ty::re_static);
  63          let str_ty = ty::mk_estr(bcx.tcx(), str_vstore);
  64          let scratch = scratch_datum(bcx, str_ty, "", false);
  65          let len = C_uint(bcx.ccx(), s.len());
  66          let c_str = PointerCast(bcx, C_cstr(bcx.ccx(), s), Type::i8p());
  67          Store(bcx, c_str, GEPi(bcx, scratch.val, [ 0, 0 ]));
  68          Store(bcx, len, GEPi(bcx, scratch.val, [ 0, 1 ]));
  69          scratch.val
  70      }

  72      pub fn c_size_and_align(&mut self, t: ty::t) -> ~[ValueRef] {
  73          let tr = type_of(self.bcx.ccx(), t);
  74          let s = machine::llsize_of_real(self.bcx.ccx(), tr);
  75          let a = machine::llalign_of_min(self.bcx.ccx(), tr);
  76          return ~[self.c_uint(s),
  77               self.c_uint(a)];
  78      }

  85      }

  92      pub fn visit(&mut self, ty_name: &str, args: &[ValueRef]) {
  93          let tcx = self.bcx.tcx();
  94          let mth_idx = ty::method_idx(
  95              tcx.sess.ident_of(~"visit_" + ty_name),
  96              *self.visitor_methods).expect(fmt!("Couldn't find visit method \
  97                                                  for %s", ty_name));
  98          let mth_ty =
  99              ty::mk_bare_fn(tcx, self.visitor_methods[mth_idx].fty.clone());
 100          let v = self.visitor_val;
 101          debug!("passing %u args:", args.len());
 102          let mut bcx = self.bcx;
 103          for (i, a) in args.iter().enumerate() {
 104              debug!("arg %u: %s", i, bcx.val_to_str(*a));
 105          }
 106          let bool_ty = ty::mk_bool();
 107          let result = unpack_result!(bcx, callee::trans_call_inner(
 108              self.bcx, None, mth_ty, bool_ty,
 109              |bcx| meth::trans_trait_callee_from_llval(bcx,
 110                                                        mth_ty,
 111                                                        mth_idx,
 112                                                        v,
 113                                                        None),
 114              ArgVals(args), None, DontAutorefArg));
 115          let result = bool_to_i1(bcx, result);
 116          let next_bcx = sub_block(bcx, "next");
 117          CondBr(bcx, result, next_bcx.llbb, self.final_bcx.llbb);
 118          self.bcx = next_bcx
 119      }

 121      pub fn bracketed(&mut self,
 122                       bracket_name: &str,
 123                       extra: &[ValueRef],
 124                       inner: &fn(&mut Reflector)) {
 125          self.visit("enter_" + bracket_name, extra);
 126          inner(self);
 127          self.visit("leave_" + bracket_name, extra);
 128      }

 130      pub fn vstore_name_and_extra(&mut self,
 131                                   t: ty::t,
 132                                   vstore: ty::vstore)
 133                                   -> (~str, ~[ValueRef]) {
 134          match vstore {
 135              ty::vstore_fixed(n) => {
 136                  let extra = vec::append(~[self.c_uint(n)],
 137                                          self.c_size_and_align(t));
 138                  (~"fixed", extra)
 139              }
 140              ty::vstore_slice(_) => (~"slice", ~[]),
 141              ty::vstore_uniq => (~"uniq", ~[]),
 142              ty::vstore_box => (~"box", ~[])
 143          }
 144      }

 151      pub fn visit_ty(&mut self, t: ty::t) {
 152          let bcx = self.bcx;
 153          let tcx = bcx.ccx().tcx;
 154          debug!("reflect::visit_ty %s", ty_to_str(bcx.ccx().tcx, t));
 155  
 156          match ty::get(t).sty {
 157            ty::ty_bot => self.leaf("bot"),
 158            ty::ty_nil => self.leaf("nil"),
 159            ty::ty_bool => self.leaf("bool"),
 160            ty::ty_char => self.leaf("char"),
 161            ty::ty_int(ast::ty_i) => self.leaf("int"),
 162            ty::ty_int(ast::ty_i8) => self.leaf("i8"),
 163            ty::ty_int(ast::ty_i16) => self.leaf("i16"),
 164            ty::ty_int(ast::ty_i32) => self.leaf("i32"),
 165            ty::ty_int(ast::ty_i64) => self.leaf("i64"),
 166            ty::ty_uint(ast::ty_u) => self.leaf("uint"),
 167            ty::ty_uint(ast::ty_u8) => self.leaf("u8"),
 168            ty::ty_uint(ast::ty_u16) => self.leaf("u16"),
 169            ty::ty_uint(ast::ty_u32) => self.leaf("u32"),
 170            ty::ty_uint(ast::ty_u64) => self.leaf("u64"),
 171            ty::ty_float(ast::ty_f) => self.leaf("float"),
 172            ty::ty_float(ast::ty_f32) => self.leaf("f32"),
 173            ty::ty_float(ast::ty_f64) => self.leaf("f64"),
 174  
 175            ty::ty_unboxed_vec(ref mt) => {
 176                let values = self.c_mt(mt);
 177                self.visit("vec", values)
 178            }
 179  
 180            ty::ty_estr(vst) => {
 181                let (name, extra) = self.vstore_name_and_extra(t, vst);
 182                self.visit(~"estr_" + name, extra)
 183            }
 184            ty::ty_evec(ref mt, vst) => {
 185                let (name, extra) = self.vstore_name_and_extra(t, vst);
 186                let extra = extra + self.c_mt(mt);
 187                if "uniq" == name && ty::type_contents(bcx.tcx(), t).contains_managed() {
 188                    self.visit("evec_uniq_managed", extra)
 189                } else {
 190                    self.visit(~"evec_" + name, extra)
 191                }
 192            }
 193            ty::ty_box(ref mt) => {
 194                let extra = self.c_mt(mt);
 195                self.visit("box", extra)
 196            }
 197            ty::ty_uniq(ref mt) => {
 198                let extra = self.c_mt(mt);
 199                if ty::type_contents(bcx.tcx(), t).contains_managed() {
 200                    self.visit("uniq_managed", extra)
 201                } else {
 202                    self.visit("uniq", extra)
 203                }
 204            }
 205            ty::ty_ptr(ref mt) => {
 206                let extra = self.c_mt(mt);
 207                self.visit("ptr", extra)
 208            }
 209            ty::ty_rptr(_, ref mt) => {
 210                let extra = self.c_mt(mt);
 211                self.visit("rptr", extra)
 212            }
 213  
 214            ty::ty_tup(ref tys) => {
 215                let extra = ~[self.c_uint(tys.len())]
 216                    + self.c_size_and_align(t);
 217                do self.bracketed("tup", extra) |this| {
 218                    for (i, t) in tys.iter().enumerate() {
 219                        let extra = ~[this.c_uint(i), this.c_tydesc(*t)];
 220                        this.visit("tup_field", extra);
 221                    }
 222                }
 223            }
 224  
 225            // FIXME (#2594): fetch constants out of intrinsic
 226            // FIXME (#4809): visitor should break out bare fns from other fns
 227            ty::ty_closure(ref fty) => {
 228              let pureval = ast_purity_constant(fty.purity);
 229              let sigilval = ast_sigil_constant(fty.sigil);
 230              let retval = if ty::type_is_bot(fty.sig.output) {0u} else {1u};
 231              let extra = ~[self.c_uint(pureval),
 232                            self.c_uint(sigilval),
 233                            self.c_uint(fty.sig.inputs.len()),
 234                            self.c_uint(retval)];
 235              self.visit("enter_fn", extra);
 236              self.visit_sig(retval, &fty.sig);
 237              self.visit("leave_fn", extra);
 238            }
 239  
 240            // FIXME (#2594): fetch constants out of intrinsic:: for the
 241            // numbers.
 242            ty::ty_bare_fn(ref fty) => {
 243              let pureval = ast_purity_constant(fty.purity);
 244              let sigilval = 0u;
 245              let retval = if ty::type_is_bot(fty.sig.output) {0u} else {1u};
 246              let extra = ~[self.c_uint(pureval),
 247                            self.c_uint(sigilval),
 248                            self.c_uint(fty.sig.inputs.len()),
 249                            self.c_uint(retval)];
 250              self.visit("enter_fn", extra);
 251              self.visit_sig(retval, &fty.sig);
 252              self.visit("leave_fn", extra);
 253            }

 255            ty::ty_struct(did, ref substs) => {
 256                let fields = ty::struct_fields(tcx, did, substs);
 257                let mut named_fields = false;
 258                if !fields.is_empty() {
 259                    named_fields =
 260                          fields[0].ident.name != special_idents::unnamed_field.name;
 261                }
 262  
 263                let extra = ~[self.c_slice(ty_to_str(tcx, t).to_managed()),
 264                              self.c_bool(named_fields),
 265                              self.c_uint(fields.len())] + self.c_size_and_align(t);
 266                do self.bracketed("class", extra) |this| {
 267                    for (i, field) in fields.iter().enumerate() {
 268                        let extra = ~[this.c_uint(i),
 269                                      this.c_slice(bcx.ccx().sess.str_of(field.ident)),
 270                                      this.c_bool(named_fields)]
 271                            + this.c_mt(&field.mt);
 272                        this.visit("class_field", extra);
 273                    }
 274                }
 275            }
 276  
 277            // FIXME (#2595): visiting all the variants in turn is probably
 278            // not ideal. It'll work but will get costly on big enums. Maybe
 279            // let the visitor tell us if it wants to visit only a particular
 280            // variant?
 281            ty::ty_enum(did, ref substs) => {
 282              let ccx = bcx.ccx();
 283              let repr = adt::represent_type(bcx.ccx(), t);
 284              let variants = ty::substd_enum_variants(ccx.tcx, did, substs);
 285              let llptrty = type_of(ccx, t).ptr_to();
 286              let opaquety = ty::get_opaque_ty(ccx.tcx).unwrap();
 287              let opaqueptrty = ty::mk_ptr(ccx.tcx, ty::mt { ty: opaquety,
 288                                                             mutbl: ast::MutImmutable });
 289  
 290              let make_get_disr = || {
 291                  let sub_path = bcx.fcx.path + &[path_name(special_idents::anon)];
 292                  let sym = mangle_internal_name_by_path_and_seq(ccx,
 293                                                                 sub_path,
 294                                                                 "get_disr");
 295  
 296                  let llfdecl = decl_internal_rust_fn(ccx, [opaqueptrty], ty::mk_int(), sym);
 297                  let fcx = new_fn_ctxt(ccx,
 298                                        ~[],
 299                                        llfdecl,
 300                                        ty::mk_uint(),
 301                                        None);
 302                  let arg = unsafe {
 303                      //
 304                      // we know the return type of llfdecl is an int here, so
 305                      // no need for a special check to see if the return type
 306                      // is immediate.
 307                      //
 308                      llvm::LLVMGetParam(llfdecl, fcx.arg_pos(0u) as c_uint)
 309                  };
 310                  let mut bcx = fcx.entry_bcx.unwrap();
 311                  let arg = BitCast(bcx, arg, llptrty);
 312                  let ret = adt::trans_get_discr(bcx, repr, arg);
 313                  Store(bcx, ret, fcx.llretptr.unwrap());
 314                  match fcx.llreturn {
 315                      Some(llreturn) => cleanup_and_Br(bcx, bcx, llreturn),
 316                      None => bcx = cleanup_block(bcx, Some(bcx.llbb))
 317                  };
 318                  finish_fn(fcx, bcx);
 319                  llfdecl
 320              };
 321  
 322              let enum_args = ~[self.c_uint(variants.len()), make_get_disr()]
 323                  + self.c_size_and_align(t);
 324              do self.bracketed("enum", enum_args) |this| {
 325                  for (i, v) in variants.iter().enumerate() {
 326                      let name = ccx.sess.str_of(v.name);
 327                      let variant_args = ~[this.c_uint(i),
 328                                           C_integral(self.bcx.ccx().int_type, v.disr_val, false),
 329                                           this.c_uint(v.args.len()),
 330                                           this.c_slice(name)];
 331                      do this.bracketed("enum_variant", variant_args) |this| {
 332                          for (j, a) in v.args.iter().enumerate() {
 333                              let bcx = this.bcx;
 334                              let null = C_null(llptrty);
 335                              let ptr = adt::trans_field_ptr(bcx, repr, null, v.disr_val, j);
 336                              let offset = p2i(ccx, ptr);
 337                              let field_args = ~[this.c_uint(j),
 338                                                 offset,
 339                                                 this.c_tydesc(*a)];
 340                              this.visit("enum_variant_field", field_args);
 341                          }
 342                      }
 343                  }
 344              }
 345            }

 370      pub fn visit_sig(&mut self, retval: uint, sig: &ty::FnSig) {
 371          for (i, arg) in sig.inputs.iter().enumerate() {
 372              let modeval = 5u;   // "by copy"
 373              let extra = ~[self.c_uint(i),
 374                           self.c_uint(modeval),
 375                           self.c_tydesc(*arg)];
 376              self.visit("fn_input", extra);
 377          }
 378          let extra = ~[self.c_uint(retval),
 379                        self.c_tydesc(sig.output)];
 380          self.visit("fn_output", extra);
 381      }
 382  }

 384  // Emit a sequence of calls to visit_ty::visit_foo
 385  pub fn emit_calls_to_trait_visit_ty(bcx: @mut Block,
 386                                      t: ty::t,
 387                                      visitor_val: ValueRef,
 388                                      visitor_trait_id: DefId)
 389                                   -> @mut Block {
 390      let final = sub_block(bcx, "final");
 391      let tydesc_ty = ty::get_tydesc_ty(bcx.ccx().tcx).unwrap();
 392      let tydesc_ty = type_of(bcx.ccx(), tydesc_ty);
 393      let mut r = Reflector {
 394          visitor_val: visitor_val,

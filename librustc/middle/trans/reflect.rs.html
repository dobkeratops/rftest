  37  pub struct Reflector<'a> {
  38      visitor_val: ValueRef,
  39      visitor_methods: @Vec<@ty::Method> ,
  40      final_bcx: &'a Block<'a>,

  43  }

  58      pub fn c_slice(&mut self, s: InternedString) -> ValueRef {
  59          // We're careful to not use first class aggregates here because that
  60          // will kick us off fast isel. (Issue #4352.)
  61          let bcx = self.bcx;
  62          let str_vstore = ty::vstore_slice(ty::ReStatic);
  63          let str_ty = ty::mk_str(bcx.tcx(), str_vstore);
  64          let scratch = rvalue_scratch_datum(bcx, str_ty, "");
  65          let len = C_uint(bcx.ccx(), s.get().len());
  66          let c_str = PointerCast(bcx, C_cstr(bcx.ccx(), s), Type::i8p(bcx.ccx()));
  67          Store(bcx, c_str, GEPi(bcx, scratch.val, [ 0, 0 ]));
  68          Store(bcx, len, GEPi(bcx, scratch.val, [ 0, 1 ]));
  69          scratch.val
  70      }

  72      pub fn c_size_and_align(&mut self, t: ty::t) -> Vec<ValueRef> {
  73          let tr = type_of(self.bcx.ccx(), t);
  74          let s = machine::llsize_of_real(self.bcx.ccx(), tr);
  75          let a = machine::llalign_of_min(self.bcx.ccx(), tr);
  76          return vec!(self.c_uint(s as uint),
  77               self.c_uint(a as uint));
  78      }

  80      pub fn c_tydesc(&mut self, t: ty::t) -> ValueRef {
  81          let bcx = self.bcx;
  82          let static_ti = get_tydesc(bcx.ccx(), t);
  83          glue::lazily_emit_visit_glue(bcx.ccx(), static_ti);
  84          PointerCast(bcx, static_ti.tydesc, self.tydesc_ty.ptr_to())
  85      }

  89            self.c_tydesc(mt.ty))
  90      }

  92      pub fn visit(&mut self, ty_name: &str, args: &[ValueRef]) {
  93          let fcx = self.bcx.fcx;
  94          let tcx = self.bcx.tcx();
  95          let mth_idx = ty::method_idx(token::str_to_ident(~"visit_" + ty_name),
  96                                       self.visitor_methods.as_slice()).expect(
  97                  format!("couldn't find visit method for {}", ty_name));
  98          let mth_ty =
  99              ty::mk_bare_fn(tcx,
 100                             self.visitor_methods.get(mth_idx).fty.clone());
 101          let v = self.visitor_val;
 102          debug!("passing {} args:", args.len());
 103          let mut bcx = self.bcx;
 104          for (i, a) in args.iter().enumerate() {
 105              debug!("arg {}: {}", i, bcx.val_to_str(*a));
 106          }
 107          let result = unpack_result!(bcx, callee::trans_call_inner(
 108              self.bcx, None, mth_ty,
 109              |bcx, _| meth::trans_trait_callee_from_llval(bcx,
 110                                                           mth_ty,
 111                                                           mth_idx,
 112                                                           v),
 113              ArgVals(args), None));
 114          let result = bool_to_i1(bcx, result);
 115          let next_bcx = fcx.new_temp_block("next");
 116          CondBr(bcx, result, next_bcx.llbb, self.final_bcx.llbb);
 117          self.bcx = next_bcx
 118      }

 120      pub fn bracketed(&mut self,
 121                       bracket_name: &str,
 122                       extra: &[ValueRef],
 123                       inner: |&mut Reflector|) {
 124          self.visit("enter_" + bracket_name, extra);
 125          inner(self);
 126          self.visit("leave_" + bracket_name, extra);
 127      }

 129      pub fn vstore_name_and_extra(&mut self,
 130                                   t: ty::t,
 131                                   vstore: ty::vstore)
 132                                   -> (~str, Vec<ValueRef> ) {
 133          match vstore {
 134              ty::vstore_fixed(n) => {
 135                  let extra = vec::append(vec!(self.c_uint(n)),
 136                                             self.c_size_and_align(t)
 137                                                 .as_slice());
 138                  (~"fixed", extra)
 139              }
 140              ty::vstore_slice(_) => (~"slice", Vec::new()),
 141              ty::vstore_uniq => (~"uniq", Vec::new()),
 142          }
 143      }

 150      pub fn visit_ty(&mut self, t: ty::t) {
 151          let bcx = self.bcx;
 152          let tcx = bcx.tcx();
 153          debug!("reflect::visit_ty {}", ty_to_str(bcx.tcx(), t));
 154  
 155          match ty::get(t).sty {
 156            ty::ty_bot => self.leaf("bot"),
 157            ty::ty_nil => self.leaf("nil"),
 158            ty::ty_bool => self.leaf("bool"),
 159            ty::ty_char => self.leaf("char"),
 160            ty::ty_int(ast::TyI) => self.leaf("int"),
 161            ty::ty_int(ast::TyI8) => self.leaf("i8"),
 162            ty::ty_int(ast::TyI16) => self.leaf("i16"),
 163            ty::ty_int(ast::TyI32) => self.leaf("i32"),
 164            ty::ty_int(ast::TyI64) => self.leaf("i64"),
 165            ty::ty_uint(ast::TyU) => self.leaf("uint"),
 166            ty::ty_uint(ast::TyU8) => self.leaf("u8"),
 167            ty::ty_uint(ast::TyU16) => self.leaf("u16"),
 168            ty::ty_uint(ast::TyU32) => self.leaf("u32"),
 169            ty::ty_uint(ast::TyU64) => self.leaf("u64"),
 170            ty::ty_float(ast::TyF32) => self.leaf("f32"),
 171            ty::ty_float(ast::TyF64) => self.leaf("f64"),
 172  
 173            ty::ty_unboxed_vec(ref mt) => {
 174                let values = self.c_mt(mt);
 175                self.visit("vec", values.as_slice())
 176            }
 177  
 178            // Should rename to str_*/vec_*.
 179            ty::ty_str(vst) => {
 180                let (name, extra) = self.vstore_name_and_extra(t, vst);
 181                self.visit(~"estr_" + name, extra.as_slice())
 182            }
 183            ty::ty_vec(ref mt, vst) => {
 184                let (name, extra) = self.vstore_name_and_extra(t, vst);
 185                let extra = vec::append(extra, self.c_mt(mt).as_slice());
 186                self.visit(~"evec_" + name, extra.as_slice())
 187            }
 188            // Should remove mt from box and uniq.
 189            ty::ty_box(typ) => {
 190                let extra = self.c_mt(&ty::mt {
 191                    ty: typ,
 192                    mutbl: ast::MutImmutable,
 193                });
 194                self.visit("box", extra.as_slice())
 195            }
 196            ty::ty_uniq(typ) => {
 197                let extra = self.c_mt(&ty::mt {
 198                    ty: typ,
 199                    mutbl: ast::MutImmutable,
 200                });
 201                self.visit("uniq", extra.as_slice())
 202            }
 203            ty::ty_ptr(ref mt) => {
 204                let extra = self.c_mt(mt);
 205                self.visit("ptr", extra.as_slice())
 206            }
 207            ty::ty_rptr(_, ref mt) => {
 208                let extra = self.c_mt(mt);
 209                self.visit("rptr", extra.as_slice())
 210            }
 211  
 212            ty::ty_tup(ref tys) => {
 213                let extra = vec::append(vec!(self.c_uint(tys.len())),
 214                                           self.c_size_and_align(t).as_slice());
 215                self.bracketed("tup", extra.as_slice(), |this| {
 216                    for (i, t) in tys.iter().enumerate() {
 217                        let extra = vec!(this.c_uint(i), this.c_tydesc(*t));
 218                        this.visit("tup_field", extra.as_slice());
 219                    }
 220                })
 221            }
 222  
 223            // FIXME (#2594): fetch constants out of intrinsic
 224            // FIXME (#4809): visitor should break out bare fns from other fns
 225            ty::ty_closure(ref fty) => {
 226              let pureval = ast_purity_constant(fty.purity);
 227              let sigilval = ast_sigil_constant(fty.sigil);
 228              let retval = if ty::type_is_bot(fty.sig.output) {0u} else {1u};
 229              let extra = vec!(self.c_uint(pureval),
 230                            self.c_uint(sigilval),
 231                            self.c_uint(fty.sig.inputs.len()),
 232                            self.c_uint(retval));
 233              self.visit("enter_fn", extra.as_slice());
 234              self.visit_sig(retval, &fty.sig);
 235              self.visit("leave_fn", extra.as_slice());
 236            }
 237  
 238            // FIXME (#2594): fetch constants out of intrinsic:: for the
 239            // numbers.
 240            ty::ty_bare_fn(ref fty) => {
 241              let pureval = ast_purity_constant(fty.purity);
 242              let sigilval = 0u;
 243              let retval = if ty::type_is_bot(fty.sig.output) {0u} else {1u};
 244              let extra = vec!(self.c_uint(pureval),
 245                            self.c_uint(sigilval),
 246                            self.c_uint(fty.sig.inputs.len()),
 247                            self.c_uint(retval));
 248              self.visit("enter_fn", extra.as_slice());
 249              self.visit_sig(retval, &fty.sig);
 250              self.visit("leave_fn", extra.as_slice());
 251            }
 252  
 253            ty::ty_struct(did, ref substs) => {
 254                let fields = ty::struct_fields(tcx, did, substs);
 255                let mut named_fields = false;
 256                if !fields.is_empty() {
 257                    named_fields = fields.get(0).ident.name !=
 258                        special_idents::unnamed_field.name;
 259                }
 260  
 261                let extra = vec::append(vec!(
 262                    self.c_slice(token::intern_and_get_ident(ty_to_str(tcx,
 263                                                                       t))),
 264                    self.c_bool(named_fields),
 265                    self.c_uint(fields.len())
 266                ), self.c_size_and_align(t).as_slice());
 267                self.bracketed("class", extra.as_slice(), |this| {
 268                    for (i, field) in fields.iter().enumerate() {
 269                        let extra = vec::append(vec!(
 270                          this.c_uint(i),
 271                          this.c_slice(token::get_ident(field.ident)),
 272                          this.c_bool(named_fields)
 273                        ), this.c_mt(&field.mt).as_slice());
 274                        this.visit("class_field", extra.as_slice());
 275                    }
 276                })
 277            }
 278  
 279            // FIXME (#2595): visiting all the variants in turn is probably
 280            // not ideal. It'll work but will get costly on big enums. Maybe
 281            // let the visitor tell us if it wants to visit only a particular
 282            // variant?
 283            ty::ty_enum(did, ref substs) => {
 284              let ccx = bcx.ccx();
 285              let repr = adt::represent_type(bcx.ccx(), t);
 286              let variants = ty::substd_enum_variants(ccx.tcx(), did, substs);
 287              let llptrty = type_of(ccx, t).ptr_to();
 288              let opaquety = ty::get_opaque_ty(ccx.tcx()).unwrap();
 289              let opaqueptrty = ty::mk_ptr(ccx.tcx(), ty::mt { ty: opaquety,
 290                                                             mutbl: ast::MutImmutable });
 291  
 292              let make_get_disr = || {
 293                  let sym = mangle_internal_name_by_path_and_seq(
 294                      ast_map::Values([].iter()).chain(None), "get_disr");
 295  
 296                  let llfdecl = decl_internal_rust_fn(ccx, false, [opaqueptrty], ty::mk_u64(), sym);
 297                  let arena = TypedArena::new();
 298                  let fcx = new_fn_ctxt(ccx, llfdecl, -1, false,
 299                                        ty::mk_u64(), None, None, &arena);
 300                  init_function(&fcx, false, ty::mk_u64(), None);
 301  
 302                  let arg = unsafe {
 303                      //
 304                      // we know the return type of llfdecl is an int here, so
 305                      // no need for a special check to see if the return type
 306                      // is immediate.
 307                      //
 308                      llvm::LLVMGetParam(llfdecl, fcx.arg_pos(0u) as c_uint)
 309                  };
 310                  let bcx = fcx.entry_bcx.get().unwrap();
 311                  let arg = BitCast(bcx, arg, llptrty);
 312                  let ret = adt::trans_get_discr(bcx, repr, arg, Some(Type::i64(ccx)));
 313                  Store(bcx, ret, fcx.llretptr.get().unwrap());
 314                  match fcx.llreturn.get() {
 315                      Some(llreturn) => Br(bcx, llreturn),
 316                      None => {}
 317                  };
 318                  finish_fn(&fcx, bcx);
 319                  llfdecl
 320              };
 321  
 322              let enum_args = vec::append(vec!(self.c_uint(variants.len()),
 323                                                  make_get_disr()),
 324                                             self.c_size_and_align(t)
 325                                                 .as_slice());
 326              self.bracketed("enum", enum_args.as_slice(), |this| {
 327                  for (i, v) in variants.iter().enumerate() {
 328                      let name = token::get_ident(v.name);
 329                      let variant_args = [this.c_uint(i),
 330                                           C_u64(ccx, v.disr_val),
 331                                           this.c_uint(v.args.len()),
 332                                           this.c_slice(name)];
 333                      this.bracketed("enum_variant",
 334                                     variant_args,
 335                                     |this| {
 336                          for (j, a) in v.args.iter().enumerate() {
 337                              let bcx = this.bcx;
 338                              let null = C_null(llptrty);
 339                              let ptr = adt::trans_field_ptr(bcx, repr, null, v.disr_val, j);
 340                              let offset = p2i(ccx, ptr);
 341                              let field_args = [this.c_uint(j),
 342                                                 offset,
 343                                                 this.c_tydesc(*a)];
 344                              this.visit("enum_variant_field",
 345                                         field_args);
 346                          }
 347                      })
 348                  }
 349              })
 350            }

 352            ty::ty_trait(..) => {
 353                let extra = [
 354                    self.c_slice(token::intern_and_get_ident(ty_to_str(tcx, t)))
 355                ];

 357            }
 358

 360            ty::ty_infer(_) => self.leaf("infer"),
 361            ty::ty_err => self.leaf("err"),

 363                let extra = vec!(self.c_uint(p.idx));
 364                self.visit("param", extra.as_slice())
 365            }
 366            ty::ty_self(..) => self.leaf("self")
 367          }

 370      pub fn visit_sig(&mut self, retval: uint, sig: &ty::FnSig) {
 371          for (i, arg) in sig.inputs.iter().enumerate() {
 372              let modeval = 5u;   // "by copy"
 373              let extra = vec!(self.c_uint(i),
 374                           self.c_uint(modeval),
 375                           self.c_tydesc(*arg));
 376              self.visit("fn_input", extra.as_slice());
 377          }
 378          let extra = vec!(self.c_uint(retval),
 379                        self.c_bool(sig.variadic),
 380                        self.c_tydesc(sig.output));
 381          self.visit("fn_output", extra.as_slice());
 382      }

 385  // Emit a sequence of calls to visit_ty::visit_foo
 386  pub fn emit_calls_to_trait_visit_ty<'a>(
 387                                      bcx: &'a Block<'a>,
 388                                      t: ty::t,
 389                                      visitor_val: ValueRef,
 390                                      visitor_trait_id: DefId)
 391                                      -> &'a Block<'a> {
 392      let fcx = bcx.fcx;
 393      let final = fcx.new_temp_block("final");
 394      let tydesc_ty = ty::get_tydesc_ty(bcx.tcx()).unwrap();
 395      let tydesc_ty = type_of(bcx.ccx(), tydesc_ty);
 396      let mut r = Reflector {

  75  fn trans<'a>(bcx: &'a Block<'a>, expr: &ast::Expr) -> Callee<'a> {
  76      let _icx = push_ctxt("trans_callee");
  77      debug!("callee::trans(expr={})", expr.repr(bcx.tcx()));
  78  
  79      // pick out special kinds of expressions that can be called:
  80      match expr.node {
  81          ast::ExprPath(_) => {
  82              return trans_def(bcx, bcx.def(expr.id), expr);
  83          }
  84          _ => {}
  85      }
  86  
  87      // any other expressions are closures:
  88      return datum_callee(bcx, expr);
  89  
  90      fn datum_callee<'a>(bcx: &'a Block<'a>, expr: &ast::Expr) -> Callee<'a> {
  91          let DatumBlock {bcx: mut bcx, datum} = expr::trans(bcx, expr);
  92          match ty::get(datum.ty).sty {
  93              ty::ty_bare_fn(..) => {
  94                  let llval = datum.to_llscalarish(bcx);
  95                  return Callee {bcx: bcx, data: Fn(llval)};
  96              }
  97              ty::ty_closure(..) => {
  98                  let datum = unpack_datum!(
  99                      bcx, datum.to_lvalue_datum(bcx, "callee", expr.id));
 100                  return Callee {bcx: bcx, data: Closure(datum)};
 101              }
 102              _ => {
 103                  bcx.tcx().sess.span_bug(
 104                      expr.span,
 105                      format!("type of callee is neither bare-fn nor closure: {}",
 106                           bcx.ty_to_str(datum.ty)));
 107              }
 108          }
 109      }
 110  
 111      fn fn_callee<'a>(bcx: &'a Block<'a>, llfn: ValueRef) -> Callee<'a> {
 112          return Callee {bcx: bcx, data: Fn(llfn)};
 113      }

 115      fn trans_def<'a>(bcx: &'a Block<'a>, def: ast::Def, ref_expr: &ast::Expr)
 116                   -> Callee<'a> {
 117          match def {
 118              ast::DefFn(did, _) |
 119              ast::DefStaticMethod(did, ast::FromImpl(_), _) => {
 120                  fn_callee(bcx, trans_fn_ref(bcx, did, ExprId(ref_expr.id)))
 121              }
 122              ast::DefStaticMethod(impl_did,
 123                                     ast::FromTrait(trait_did),
 124                                     _) => {
 125                  fn_callee(bcx, meth::trans_static_method_callee(bcx, impl_did,
 126                                                                  trait_did,
 127                                                                  ref_expr.id))
 128              }
 129              ast::DefVariant(tid, vid, _) => {
 130                  // nullary variants are not callable
 131                  assert!(ty::enum_variant_with_id(bcx.tcx(),
 132                                                        tid,
 133                                                        vid).args.len() > 0u);
 134                  fn_callee(bcx, trans_fn_ref(bcx, vid, ExprId(ref_expr.id)))
 135              }
 136              ast::DefStruct(def_id) => {
 137                  fn_callee(bcx, trans_fn_ref(bcx, def_id, ExprId(ref_expr.id)))
 138              }
 139              ast::DefStatic(..) |
 140              ast::DefArg(..) |
 141              ast::DefLocal(..) |
 142              ast::DefBinding(..) |
 143              ast::DefUpvar(..) => {
 144                  datum_callee(bcx, ref_expr)
 145              }
 146              ast::DefMod(..) | ast::DefForeignMod(..) | ast::DefTrait(..) |
 147              ast::DefTy(..) | ast::DefPrimTy(..) |
 148              ast::DefUse(..) | ast::DefTyParamBinder(..) |
 149              ast::DefRegion(..) | ast::DefLabel(..) | ast::DefTyParam(..) |
 150              ast::DefSelfTy(..) | ast::DefMethod(..) => {
 151                  bcx.tcx().sess.span_bug(
 152                      ref_expr.span,
 153                      format!("cannot translate def {:?} \
 154                            to a callable thing!", def));
 155              }
 156          }
 157      }
 158  }

 160  pub fn trans_fn_ref(bcx: &Block, def_id: ast::DefId, node: ExprOrMethodCall) -> ValueRef {
 161      /*!
 162       *
 163       * Translates a reference (with id `ref_id`) to the fn/method
 164       * with id `def_id` into a function pointer.  This may require
 165       * monomorphization or inlining. */
 166  
 167      let _icx = push_ctxt("trans_fn_ref");
 168  
 169      let type_params = node_id_type_params(bcx, node);
 170      let vtable_key = match node {
 171          ExprId(id) => MethodCall::expr(id),
 172          MethodCall(method_call) => method_call
 173      };
 174      let vtables = node_vtables(bcx, vtable_key);
 175      debug!("trans_fn_ref(def_id={}, node={:?}, type_params={}, vtables={})",
 176             def_id.repr(bcx.tcx()), node, type_params.repr(bcx.tcx()),
 177             vtables.repr(bcx.tcx()));
 178      trans_fn_ref_with_vtables(bcx, def_id, node,
 179                                type_params.as_slice(),
 180                                vtables)
 181  }

 183  fn trans_fn_ref_with_vtables_to_callee<'a>(bcx: &'a Block<'a>,
 184                                             def_id: ast::DefId,
 185                                             ref_id: ast::NodeId,
 186                                             type_params: &[ty::t],
 187                                             vtables: Option<typeck::vtable_res>)
 188                                             -> Callee<'a> {
 189      Callee {bcx: bcx,
 190              data: Fn(trans_fn_ref_with_vtables(bcx, def_id, ExprId(ref_id),
 191                                                 type_params, vtables))}
 192  }

 194  fn resolve_default_method_vtables(bcx: &Block,
 195                                    impl_id: ast::DefId,
 196                                    method: &ty::Method,
 197                                    substs: &ty::substs,
 198                                    impl_vtables: Option<typeck::vtable_res>)
 199                            -> (typeck::vtable_res, typeck::vtable_param_res) {
 200  
 201      // Get the vtables that the impl implements the trait at
 202      let impl_res = ty::lookup_impl_vtables(bcx.tcx(), impl_id);
 203  
 204      // Build up a param_substs that we are going to resolve the
 205      // trait_vtables under.
 206      let param_substs = Some(@param_substs {
 207          tys: substs.tps.clone(),
 208          self_ty: substs.self_ty,
 209          vtables: impl_vtables,
 210          self_vtables: None
 211      });
 212  
 213      let trait_vtables_fixed = resolve_vtables_under_param_substs(
 214          bcx.tcx(), param_substs, impl_res.trait_vtables);
 215  
 216      // Now we pull any vtables for parameters on the actual method.
 217      let num_method_vtables = method.generics.type_param_defs().len();
 218      let method_vtables = match impl_vtables {
 219          Some(vtables) => {
 220              let num_impl_type_parameters =
 221                  vtables.len() - num_method_vtables;
 222              Vec::from_slice(vtables.tailn(num_impl_type_parameters))
 223          },
 224          None => Vec::from_elem(num_method_vtables, @Vec::new())
 225      };
 226  
 227      let method_vtables = method_vtables.as_slice();
 228      let param_vtables = @((*trait_vtables_fixed).clone().append(method_vtables));
 229  
 230      let self_vtables = resolve_param_vtables_under_param_substs(
 231          bcx.tcx(), param_substs, impl_res.self_vtables);
 232  
 233      (param_vtables, self_vtables)
 234  }

 237  pub fn trans_fn_ref_with_vtables(
 238          bcx: &Block,       //
 239          def_id: ast::DefId,   // def id of fn
 240          node: ExprOrMethodCall,  // node id of use of fn; may be zero if N/A
 241          type_params: &[ty::t], // values for fn's ty params
 242          vtables: Option<typeck::vtable_res>) // vtables for the call
 243       -> ValueRef {
 244      /*!
 245       * Translates a reference to a fn/method item, monomorphizing and
 246       * inlining as it goes.
 247       *
 248       * # Parameters
 249       *
 250       * - `bcx`: the current block where the reference to the fn occurs
 251       * - `def_id`: def id of the fn or method item being referenced
 252       * - `node`: node id of the reference to the fn/method, if applicable.
 253       *   This parameter may be zero; but, if so, the resulting value may not
 254       *   have the right type, so it must be cast before being used.
 255       * - `type_params`: values for each of the fn/method's type parameters
 256       * - `vtables`: values for each bound on each of the type parameters
 257       */
 258  
 259      let _icx = push_ctxt("trans_fn_ref_with_vtables");
 260      let ccx = bcx.ccx();
 261      let tcx = bcx.tcx();
 262  
 263      debug!("trans_fn_ref_with_vtables(bcx={}, def_id={}, node={:?}, \
 264              type_params={}, vtables={})",
 265             bcx.to_str(),
 266             def_id.repr(tcx),
 267             node,
 268             type_params.repr(tcx),
 269             vtables.repr(tcx));
 270  
 271      assert!(type_params.iter().all(|t| !ty::type_needs_infer(*t)));
 272  
 273      // Polytype of the function item (may have type params)
 274      let fn_tpt = ty::lookup_item_type(tcx, def_id);
 275  
 276      let substs = ty::substs { regions: ty::ErasedRegions,
 277                                self_ty: None,
 278                                tps: /*bad*/ Vec::from_slice(type_params) };
 279  
 280      // Load the info for the appropriate trait if necessary.
 281      match ty::trait_of_method(tcx, def_id) {
 282          None => {}
 283          Some(trait_id) => {
 284              ty::populate_implementations_for_trait_if_necessary(tcx, trait_id)
 285          }
 286      }
 287  
 288      // We need to do a bunch of special handling for default methods.
 289      // We need to modify the def_id and our substs in order to monomorphize
 290      // the function.
 291      let (is_default, def_id, substs, self_vtables, vtables) =
 292          match ty::provided_source(tcx, def_id) {
 293          None => (false, def_id, substs, None, vtables),
 294          Some(source_id) => {
 295              // There are two relevant substitutions when compiling
 296              // default methods. First, there is the substitution for
 297              // the type parameters of the impl we are using and the
 298              // method we are calling. This substitution is the substs
 299              // argument we already have.
 300              // In order to compile a default method, though, we need
 301              // to consider another substitution: the substitution for
 302              // the type parameters on trait; the impl we are using
 303              // implements the trait at some particular type
 304              // parameters, and we need to substitute for those first.
 305              // So, what we need to do is find this substitution and
 306              // compose it with the one we already have.
 307  
 308              let impl_id = ty::method(tcx, def_id).container_id();
 309              let method = ty::method(tcx, source_id);
 310              let trait_ref = ty::impl_trait_ref(tcx, impl_id)
 311                  .expect("could not find trait_ref for impl with \
 312                           default methods");
 313  
 314              // Compute the first substitution
 315              let first_subst = make_substs_for_receiver_types(
 316                  tcx, impl_id, trait_ref, method);
 317  
 318              // And compose them
 319              let new_substs = first_subst.subst(tcx, &substs);
 320  
 321  
 322              let (param_vtables, self_vtables) =
 323                  resolve_default_method_vtables(bcx, impl_id,
 324                                                 method, &substs, vtables);
 325  
 326              debug!("trans_fn_with_vtables - default method: \
 327                      substs = {}, trait_subst = {}, \
 328                      first_subst = {}, new_subst = {}, \
 329                      vtables = {}, \
 330                      self_vtable = {}, param_vtables = {}",
 331                     substs.repr(tcx), trait_ref.substs.repr(tcx),
 332                     first_subst.repr(tcx), new_substs.repr(tcx),
 333                     vtables.repr(tcx),
 334                     self_vtables.repr(tcx), param_vtables.repr(tcx));
 335  
 336              (true, source_id,
 337               new_substs, Some(self_vtables), Some(param_vtables))
 338          }
 339      };
 340  
 341      // Check whether this fn has an inlined copy and, if so, redirect
 342      // def_id to the local id of the inlined copy.
 343      let def_id = {
 344          if def_id.krate != ast::LOCAL_CRATE {
 345              inline::maybe_instantiate_inline(ccx, def_id)
 346          } else {
 347              def_id
 348          }
 349      };
 350  
 351      // We must monomorphise if the fn has type parameters, is a rust
 352      // intrinsic, or is a default method.  In particular, if we see an
 353      // intrinsic that is inlined from a different crate, we want to reemit the
 354      // intrinsic instead of trying to call it in the other crate.
 355      let must_monomorphise = if type_params.len() > 0 || is_default {
 356          true
 357      } else if def_id.krate == ast::LOCAL_CRATE {
 358          let map_node = session::expect(
 359              ccx.sess(),
 360              tcx.map.find(def_id.node),
 361              || format!("local item should be in ast map"));
 362  
 363          match map_node {
 364              ast_map::NodeForeignItem(_) => {
 365                  tcx.map.get_foreign_abi(def_id.node) == synabi::RustIntrinsic
 366              }
 367              _ => false
 368          }
 369      } else {
 370          false
 371      };
 372  
 373      // Create a monomorphic verison of generic functions
 374      if must_monomorphise {
 375          // Should be either intra-crate or inlined.
 376          assert_eq!(def_id.krate, ast::LOCAL_CRATE);
 377  
 378          let opt_ref_id = match node {
 379              ExprId(id) => if id != 0 { Some(id) } else { None },
 380              MethodCall(_) => None,
 381          };
 382  
 383          let (val, must_cast) =
 384              monomorphize::monomorphic_fn(ccx, def_id, &substs,
 385                                           vtables, self_vtables,
 386                                           opt_ref_id);
 387          let mut val = val;
 388          if must_cast && node != ExprId(0) {
 389              // Monotype of the REFERENCE to the function (type params
 390              // are subst'd)
 391              let ref_ty = match node {
 392                  ExprId(id) => node_id_type(bcx, id),
 393                  MethodCall(method_call) => {
 394                      let t = bcx.ccx().maps.method_map.borrow().get(&method_call).ty;
 395                      monomorphize_type(bcx, t)
 396                  }
 397              };
 398  
 399              val = PointerCast(
 400                  bcx, val, type_of::type_of_fn_from_ty(ccx, ref_ty).ptr_to());
 401          }
 402          return val;
 403      }
 404  
 405      // Find the actual function pointer.
 406      let mut val = {
 407          if def_id.krate == ast::LOCAL_CRATE {
 408              // Internal reference.
 409              get_item_val(ccx, def_id.node)
 410          } else {
 411              // External reference.
 412              trans_external_path(ccx, def_id, fn_tpt.ty)
 413          }
 414      };
 415  
 416      // This is subtle and surprising, but sometimes we have to bitcast
 417      // the resulting fn pointer.  The reason has to do with external
 418      // functions.  If you have two crates that both bind the same C
 419      // library, they may not use precisely the same types: for
 420      // example, they will probably each declare their own structs,
 421      // which are distinct types from LLVM's point of view (nominal
 422      // types).
 423      //
 424      // Now, if those two crates are linked into an application, and
 425      // they contain inlined code, you can wind up with a situation
 426      // where both of those functions wind up being loaded into this
 427      // application simultaneously. In that case, the same function
 428      // (from LLVM's point of view) requires two types. But of course
 429      // LLVM won't allow one function to have two types.
 430      //
 431      // What we currently do, therefore, is declare the function with
 432      // one of the two types (whichever happens to come first) and then
 433      // bitcast as needed when the function is referenced to make sure
 434      // it has the type we expect.
 435      //
 436      // This can occur on either a crate-local or crate-external
 437      // reference. It also occurs when testing libcore and in some
 438      // other weird situations. Annoying.
 439      let llty = type_of::type_of_fn_from_ty(ccx, fn_tpt.ty);
 440      let llptrty = llty.ptr_to();
 441      if val_ty(val) != llptrty {
 442          val = BitCast(bcx, val, llptrty);
 443      }
 444  
 445      val
 446  }

 514  pub fn trans_call_inner<'a>(
 515                          bcx: &'a Block<'a>,
 516                          call_info: Option<NodeInfo>,
 517                          callee_ty: ty::t,
 518                          get_callee: |bcx: &'a Block<'a>,
 519                                       arg_cleanup_scope: cleanup::ScopeId|
 520                                       -> Callee<'a>,
 521                          args: CallArgs,
 522                          dest: Option<expr::Dest>)
 523                          -> Result<'a> {
 524      /*!
 525       * This behemoth of a function translates function calls.
 526       * Unfortunately, in order to generate more efficient LLVM
 527       * output at -O0, it has quite a complex signature (refactoring
 528       * this into two functions seems like a good idea).
 529       *
 530       * In particular, for lang items, it is invoked with a dest of
 531       * None, and in that case the return value contains the result of
 532       * the fn. The lang item must not return a structural type or else
 533       * all heck breaks loose.
 534       *
 535       * For non-lang items, `dest` is always Some, and hence the result
 536       * is written into memory somewhere. Nonetheless we return the
 537       * actual return value of the function.
 538       */
 539  
 540      // Introduce a temporary cleanup scope that will contain cleanups
 541      // for the arguments while they are being evaluated. The purpose
 542      // this cleanup is to ensure that, should a failure occur while
 543      // evaluating argument N, the values for arguments 0...N-1 are all
 544      // cleaned up. If no failure occurs, the values are handed off to
 545      // the callee, and hence none of the cleanups in this temporary
 546      // scope will ever execute.
 547      let fcx = bcx.fcx;
 548      let ccx = fcx.ccx;
 549      let arg_cleanup_scope = fcx.push_custom_cleanup_scope();
 550  
 551      let callee = get_callee(bcx, cleanup::CustomScope(arg_cleanup_scope));
 552      let mut bcx = callee.bcx;
 553  
 554      let (llfn, llenv, llself) = match callee.data {
 555          Fn(llfn) => {
 556              (llfn, None, None)
 557          }
 558          TraitMethod(d) => {
 559              (d.llfn, None, Some(d.llself))
 560          }
 561          Closure(d) => {
 562              // Closures are represented as (llfn, llclosure) pair:
 563              // load the requisite values out.
 564              let pair = d.to_llref();
 565              let llfn = GEPi(bcx, pair, [0u, abi::fn_field_code]);
 566              let llfn = Load(bcx, llfn);
 567              let llenv = GEPi(bcx, pair, [0u, abi::fn_field_box]);
 568              let llenv = Load(bcx, llenv);
 569              (llfn, Some(llenv), None)
 570          }
 571      };
 572  
 573      let (abi, ret_ty) = match ty::get(callee_ty).sty {
 574          ty::ty_bare_fn(ref f) => (f.abi, f.sig.output),
 575          ty::ty_closure(ref f) => (synabi::Rust, f.sig.output),
 576          _ => fail!("expected bare rust fn or closure in trans_call_inner")
 577      };
 578      let is_rust_fn = abi == synabi::Rust || abi == synabi::RustIntrinsic;
 579  
 580      // Generate a location to store the result. If the user does
 581      // not care about the result, just make a stack slot.
 582      let opt_llretslot = match dest {
 583          None => {
 584              assert!(!type_of::return_uses_outptr(ccx, ret_ty));
 585              None
 586          }
 587          Some(expr::SaveIn(dst)) => Some(dst),
 588          Some(expr::Ignore) => {
 589              if !type_is_zero_size(ccx, ret_ty) {
 590                  Some(alloc_ty(bcx, ret_ty, "__llret"))
 591              } else {
 592                  let llty = type_of::type_of(ccx, ret_ty);
 593                  Some(C_undef(llty.ptr_to()))
 594              }
 595          }
 596      };
 597  
 598      let mut llresult = unsafe {
 599          llvm::LLVMGetUndef(Type::nil(ccx).ptr_to().to_ref())
 600      };
 601  
 602      // The code below invokes the function, using either the Rust
 603      // conventions (if it is a rust fn) or the native conventions
 604      // (otherwise).  The important part is that, when all is sad
 605      // and done, either the return value of the function will have been
 606      // written in opt_llretslot (if it is Some) or `llresult` will be
 607      // set appropriately (otherwise).
 608      if is_rust_fn {
 609          let mut llargs = Vec::new();
 610  
 611          // Push the out-pointer if we use an out-pointer for this
 612          // return type, otherwise push "undef".
 613          if type_of::return_uses_outptr(ccx, ret_ty) {
 614              llargs.push(opt_llretslot.unwrap());
 615          }
 616  
 617          // start at 1, because index 0 is the return value of the llvm func
 618          let mut first_arg_offset = 1;
 619  
 620          // Push the environment (or a trait object's self).
 621          match (llenv, llself) {
 622              (Some(llenv), None) => {
 623                  first_arg_offset += 1;
 624                  llargs.push(llenv)
 625              },
 626              (None, Some(llself)) => llargs.push(llself),
 627              _ => {}
 628          }
 629  
 630          // Push the arguments.
 631          bcx = trans_args(bcx, args, callee_ty, &mut llargs,
 632                           cleanup::CustomScope(arg_cleanup_scope),
 633                           llself.is_some());
 634  
 635          fcx.pop_custom_cleanup_scope(arg_cleanup_scope);
 636  
 637          // A function pointer is called without the declaration
 638          // available, so we have to apply any attributes with ABI
 639          // implications directly to the call instruction. Right now,
 640          // the only attribute we need to worry about is `sret`.
 641          let mut attrs = Vec::new();
 642          if type_of::return_uses_outptr(ccx, ret_ty) {
 643              attrs.push((1, StructRetAttribute));
 644              // The outptr can be noalias and nocapture because it's entirely
 645              // invisible to the program.
 646              attrs.push((1, NoAliasAttribute));
 647              attrs.push((1, NoCaptureAttribute));
 648              first_arg_offset += 1;
 649          }
 650  
 651          // The `noalias` attribute on the return value is useful to a
 652          // function ptr caller.
 653          match ty::get(ret_ty).sty {
 654              // `~` pointer return values never alias because ownership
 655              // is transferred
 656              ty::ty_uniq(..) | ty::ty_vec(_, ty::VstoreUniq) => {
 657                  attrs.push((0, NoAliasAttribute));
 658              }
 659              _ => {}
 660          }
 661  
 662          debug!("trans_callee_inner: first_arg_offset={}", first_arg_offset);
 663  
 664          for (idx, &t) in ty::ty_fn_args(callee_ty).iter().enumerate()
 665                                                    .map(|(i, v)| (i+first_arg_offset, v)) {
 666              use middle::ty::{BrAnon, ReLateBound};
 667              if !type_is_immediate(ccx, t) {
 668                  // if it's not immediate, we have a program-invisible pointer,
 669                  // which it can't possibly capture
 670                  attrs.push((idx, NoCaptureAttribute));
 671                  debug!("trans_callee_inner: argument {} nocapture because it's non-immediate", idx);
 672                  continue;
 673              }
 674  
 675              let t_ = ty::get(t);
 676              match t_.sty {
 677                  ty::ty_rptr(ReLateBound(_, BrAnon(_)), _) => {
 678                      debug!("trans_callee_inner: argument {} nocapture because \
 679                             of anonymous lifetime", idx);
 680                      attrs.push((idx, NoCaptureAttribute));
 681                  },
 682                  _ => { }
 683              }
 684          }
 685  
 686          // Invoke the actual rust fn and update bcx/llresult.
 687          let (llret, b) = base::invoke(bcx,
 688                                        llfn,
 689                                        llargs,
 690                                        attrs.as_slice(),
 691                                        call_info);
 692          bcx = b;
 693          llresult = llret;
 694  
 695          // If the Rust convention for this type is return via
 696          // the return value, copy it into llretslot.
 697          match opt_llretslot {
 698              Some(llretslot) => {
 699                  if !type_of::return_uses_outptr(bcx.ccx(), ret_ty) &&
 700                      !type_is_zero_size(bcx.ccx(), ret_ty)
 701                  {
 702                      Store(bcx, llret, llretslot);
 703                  }
 704              }
 705              None => {}
 706          }
 707      } else {
 708          // Lang items are the only case where dest is None, and
 709          // they are always Rust fns.
 710          assert!(dest.is_some());
 711  
 712          let mut llargs = Vec::new();
 713          let arg_tys = match args {
 714              ArgExprs(a) => a.iter().map(|x| expr_ty(bcx, *x)).collect(),
 715              _ => fail!("expected arg exprs.")
 716          };
 717          bcx = trans_args(bcx, args, callee_ty, &mut llargs,
 718                           cleanup::CustomScope(arg_cleanup_scope), false);
 719          fcx.pop_custom_cleanup_scope(arg_cleanup_scope);
 720          bcx = foreign::trans_native_call(bcx, callee_ty,
 721                                           llfn, opt_llretslot.unwrap(),
 722                                           llargs.as_slice(), arg_tys);
 723      }
 724  
 725      // If the caller doesn't care about the result of this fn call,
 726      // drop the temporary slot we made.
 727      match dest {
 728          None => {
 729              assert!(!type_of::return_uses_outptr(bcx.ccx(), ret_ty));
 730          }
 731          Some(expr::Ignore) => {
 732              // drop the value if it is not being saved.
 733              bcx = glue::drop_ty(bcx, opt_llretslot.unwrap(), ret_ty);
 734          }
 735          Some(expr::SaveIn(_)) => { }
 736      }
 737  
 738      if ty::type_is_bot(ret_ty) {
 739          Unreachable(bcx);
 740      }
 741  
 742      rslt(bcx, llresult)
 743  }

 745  pub enum CallArgs<'a> {
 746      // Supply value of arguments as a list of expressions that must be
 747      // translated. This is used in the common case of `foo(bar, qux)`.
 748      ArgExprs(&'a [@ast::Expr]),
 749  
 750      // Supply value of arguments as a list of LLVM value refs; frequently
 751      // used with lang items and so forth, when the argument is an internal
 752      // value.
 753      ArgVals(&'a [ValueRef]),
 754  
 755      // For overloaded operators: `(lhs, Option(rhs, rhs_id))`. `lhs`
 756      // is the left-hand-side and `rhs/rhs_id` is the datum/expr-id of
 757      // the right-hand-side (if any).
 758      ArgOverloadedOp(Datum<Expr>, Option<(Datum<Expr>, ast::NodeId)>),
 759  }
 760  
 761  fn trans_args<'a>(cx: &'a Block<'a>,
 762                    args: CallArgs,
 763                    fn_ty: ty::t,
 764                    llargs: &mut Vec<ValueRef> ,
 765                    arg_cleanup_scope: cleanup::ScopeId,
 766                    ignore_self: bool)
 767                    -> &'a Block<'a> {
 768      let _icx = push_ctxt("trans_args");
 769      let arg_tys = ty::ty_fn_args(fn_ty);
 770      let variadic = ty::fn_is_variadic(fn_ty);
 771  
 772      let mut bcx = cx;
 773  
 774      // First we figure out the caller's view of the types of the arguments.
 775      // This will be needed if this is a generic call, because the callee has
 776      // to cast her view of the arguments to the caller's view.
 777      match args {
 778          ArgExprs(arg_exprs) => {
 779              let num_formal_args = arg_tys.len();
 780              for (i, &arg_expr) in arg_exprs.iter().enumerate() {
 781                  if i == 0 && ignore_self {
 782                      continue;
 783                  }
 784                  let arg_ty = if i >= num_formal_args {
 785                      assert!(variadic);
 786                      expr_ty_adjusted(cx, arg_expr)
 787                  } else {
 788                      *arg_tys.get(i)
 789                  };
 790  
 791                  let arg_datum = unpack_datum!(bcx, expr::trans(bcx, arg_expr));
 792                  llargs.push(unpack_result!(bcx, {
 793                      trans_arg_datum(bcx, arg_ty, arg_datum,
 794                                      arg_cleanup_scope,
 795                                      DontAutorefArg)
 796                  }));
 797              }
 798          }
 799          ArgOverloadedOp(lhs, rhs) => {
 800              assert!(!variadic);
 801  
 802              llargs.push(unpack_result!(bcx, {
 803                  trans_arg_datum(bcx, *arg_tys.get(0), lhs,
 804                                  arg_cleanup_scope,
 805                                  DontAutorefArg)
 806              }));
 807  
 808              match rhs {
 809                  Some((rhs, rhs_id)) => {
 810                      assert_eq!(arg_tys.len(), 2);
 811  
 812                      llargs.push(unpack_result!(bcx, {
 813                          trans_arg_datum(bcx, *arg_tys.get(1), rhs,
 814                                          arg_cleanup_scope,
 815                                          DoAutorefArg(rhs_id))
 816                      }));
 817                  }
 818                  None => assert_eq!(arg_tys.len(), 1)
 819              }
 820          }
 821          ArgVals(vs) => {
 822              llargs.push_all(vs);
 823          }
 824      }
 825  
 826      bcx
 827  }

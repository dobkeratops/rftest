  61  pub enum CalleeData {
  62      Closure(Datum<Lvalue>),
  63  
  64      // Represents a (possibly monomorphized) top-level fn item or method
  65      // item. Note that this is just the fn-ptr and is not a Rust closure
  66      // value (which is a pair).

  69      TraitMethod(MethodData)
  70  }

  72  pub struct Callee<'a> {

  77  fn trans<'a>(bcx: &'a Block<'a>, expr: &ast::Expr) -> Callee<'a> {
  78      let _icx = push_ctxt("trans_callee");
  79      debug!("callee::trans(expr={})", expr.repr(bcx.tcx()));
  80  
  81      // pick out special kinds of expressions that can be called:
  82      match expr.node {
  83          ast::ExprPath(_) => {
  84              return trans_def(bcx, bcx.def(expr.id), expr);
  85          }
  86          _ => {}
  87      }
  88  
  89      // any other expressions are closures:
  90      return datum_callee(bcx, expr);
  91  
  92      fn datum_callee<'a>(bcx: &'a Block<'a>, expr: &ast::Expr) -> Callee<'a> {
  93          let DatumBlock {bcx: mut bcx, datum} = expr::trans(bcx, expr);
  94          match ty::get(datum.ty).sty {
  95              ty::ty_bare_fn(..) => {
  96                  let llval = datum.to_llscalarish(bcx);
  97                  return Callee {bcx: bcx, data: Fn(llval)};
  98              }
  99              ty::ty_closure(..) => {
 100                  let datum = unpack_datum!(
 101                      bcx, datum.to_lvalue_datum(bcx, "callee", expr.id));
 102                  return Callee {bcx: bcx, data: Closure(datum)};
 103              }
 104              _ => {
 105                  bcx.tcx().sess.span_bug(
 106                      expr.span,
 107                      format!("type of callee is neither bare-fn nor closure: {}",
 108                           bcx.ty_to_str(datum.ty)));
 109              }
 110          }
 111      }
 112  
 113      fn fn_callee<'a>(bcx: &'a Block<'a>, llfn: ValueRef) -> Callee<'a> {
 114          return Callee {bcx: bcx, data: Fn(llfn)};
 115      }
 116  
 117      fn trans_def<'a>(bcx: &'a Block<'a>, def: ast::Def, ref_expr: &ast::Expr)
 118                   -> Callee<'a> {
 119          match def {
 120              ast::DefFn(did, _) |
 121              ast::DefStaticMethod(did, ast::FromImpl(_), _) => {
 122                  fn_callee(bcx, trans_fn_ref(bcx, did, ExprId(ref_expr.id)))
 123              }
 124              ast::DefStaticMethod(impl_did,
 125                                     ast::FromTrait(trait_did),
 126                                     _) => {
 127                  fn_callee(bcx, meth::trans_static_method_callee(bcx, impl_did,
 128                                                                  trait_did,
 129                                                                  ref_expr.id))
 130              }
 131              ast::DefVariant(tid, vid, _) => {
 132                  // nullary variants are not callable
 133                  assert!(ty::enum_variant_with_id(bcx.tcx(),
 134                                                        tid,
 135                                                        vid).args.len() > 0u);
 136                  fn_callee(bcx, trans_fn_ref(bcx, vid, ExprId(ref_expr.id)))
 137              }
 138              ast::DefStruct(def_id) => {
 139                  fn_callee(bcx, trans_fn_ref(bcx, def_id, ExprId(ref_expr.id)))
 140              }
 141              ast::DefStatic(..) |
 142              ast::DefArg(..) |
 143              ast::DefLocal(..) |
 144              ast::DefBinding(..) |
 145              ast::DefUpvar(..) => {
 146                  datum_callee(bcx, ref_expr)
 147              }
 148              ast::DefMod(..) | ast::DefForeignMod(..) | ast::DefTrait(..) |
 149              ast::DefTy(..) | ast::DefPrimTy(..) |
 150              ast::DefUse(..) | ast::DefTyParamBinder(..) |
 151              ast::DefRegion(..) | ast::DefLabel(..) | ast::DefTyParam(..) |
 152              ast::DefSelfTy(..) | ast::DefMethod(..) => {
 153                  bcx.tcx().sess.span_bug(
 154                      ref_expr.span,
 155                      format!("cannot translate def {:?} \
 156                            to a callable thing!", def));
 157              }
 158          }
 159      }
 160  }

 162  pub fn trans_fn_ref(bcx: &Block, def_id: ast::DefId, node: ExprOrMethodCall) -> ValueRef {
 163      /*!
 164       *
 165       * Translates a reference (with id `ref_id`) to the fn/method
 166       * with id `def_id` into a function pointer.  This may require
 167       * monomorphization or inlining. */
 168  
 169      let _icx = push_ctxt("trans_fn_ref");
 170  
 171      let type_params = node_id_type_params(bcx, node);
 172      let vtable_key = match node {
 173          ExprId(id) => MethodCall::expr(id),
 174          MethodCall(method_call) => method_call
 175      };
 176      let vtables = node_vtables(bcx, vtable_key);
 177      debug!("trans_fn_ref(def_id={}, node={:?}, type_params={}, vtables={})",
 178             def_id.repr(bcx.tcx()), node, type_params.repr(bcx.tcx()),
 179             vtables.repr(bcx.tcx()));
 180      trans_fn_ref_with_vtables(bcx, def_id, node,
 181                                type_params.as_slice(),
 182                                vtables)
 183  }

 185  fn trans_fn_ref_with_vtables_to_callee<'a>(bcx: &'a Block<'a>,
 186                                             def_id: ast::DefId,
 187                                             ref_id: ast::NodeId,
 188                                             type_params: &[ty::t],
 189                                             vtables: Option<typeck::vtable_res>)
 190                                             -> Callee<'a> {
 191      Callee {bcx: bcx,
 192              data: Fn(trans_fn_ref_with_vtables(bcx, def_id, ExprId(ref_id),
 193                                                 type_params, vtables))}
 194  }

 196  fn resolve_default_method_vtables(bcx: &Block,
 197                                    impl_id: ast::DefId,
 198                                    method: &ty::Method,
 199                                    substs: &ty::substs,
 200                                    impl_vtables: Option<typeck::vtable_res>)
 201                            -> (typeck::vtable_res, typeck::vtable_param_res) {
 202  
 203      // Get the vtables that the impl implements the trait at
 204      let impl_res = ty::lookup_impl_vtables(bcx.tcx(), impl_id);
 205  
 206      // Build up a param_substs that we are going to resolve the
 207      // trait_vtables under.
 208      let param_substs = Some(@param_substs {
 209          tys: substs.tps.clone(),
 210          self_ty: substs.self_ty,
 211          vtables: impl_vtables,
 212          self_vtables: None
 213      });
 214  
 215      let trait_vtables_fixed = resolve_vtables_under_param_substs(
 216          bcx.tcx(), param_substs, impl_res.trait_vtables);
 217  
 218      // Now we pull any vtables for parameters on the actual method.
 219      let num_method_vtables = method.generics.type_param_defs().len();
 220      let method_vtables = match impl_vtables {
 221          Some(vtables) => {
 222              let num_impl_type_parameters =
 223                  vtables.len() - num_method_vtables;
 224              vtables.tailn(num_impl_type_parameters).to_owned()
 225          },
 226          None => slice::from_elem(num_method_vtables, @Vec::new())
 227      };
 228  
 229      let param_vtables = @((*trait_vtables_fixed).clone().append(method_vtables));
 230  
 231      let self_vtables = resolve_param_vtables_under_param_substs(
 232          bcx.tcx(), param_substs, impl_res.self_vtables);
 233  
 234      (param_vtables, self_vtables)
 235  }

 238  pub fn trans_fn_ref_with_vtables(
 239          bcx: &Block,       //
 240          def_id: ast::DefId,   // def id of fn
 241          node: ExprOrMethodCall,  // node id of use of fn; may be zero if N/A
 242          type_params: &[ty::t], // values for fn's ty params
 243          vtables: Option<typeck::vtable_res>) // vtables for the call
 244       -> ValueRef {
 245      /*!
 246       * Translates a reference to a fn/method item, monomorphizing and
 247       * inlining as it goes.
 248       *
 249       * # Parameters
 250       *
 251       * - `bcx`: the current block where the reference to the fn occurs
 252       * - `def_id`: def id of the fn or method item being referenced
 253       * - `node`: node id of the reference to the fn/method, if applicable.
 254       *   This parameter may be zero; but, if so, the resulting value may not
 255       *   have the right type, so it must be cast before being used.
 256       * - `type_params`: values for each of the fn/method's type parameters
 257       * - `vtables`: values for each bound on each of the type parameters
 258       */
 259  
 260      let _icx = push_ctxt("trans_fn_ref_with_vtables");
 261      let ccx = bcx.ccx();
 262      let tcx = bcx.tcx();
 263  
 264      debug!("trans_fn_ref_with_vtables(bcx={}, def_id={}, node={:?}, \
 265              type_params={}, vtables={})",
 266             bcx.to_str(),
 267             def_id.repr(tcx),
 268             node,
 269             type_params.repr(tcx),
 270             vtables.repr(tcx));
 271  
 272      assert!(type_params.iter().all(|t| !ty::type_needs_infer(*t)));
 273  
 274      // Polytype of the function item (may have type params)
 275      let fn_tpt = ty::lookup_item_type(tcx, def_id);
 276  
 277      let substs = ty::substs { regions: ty::ErasedRegions,
 278                                self_ty: None,
 279                                tps: /*bad*/ Vec::from_slice(type_params) };
 280  
 281      // Load the info for the appropriate trait if necessary.
 282      match ty::trait_of_method(tcx, def_id) {
 283          None => {}
 284          Some(trait_id) => {
 285              ty::populate_implementations_for_trait_if_necessary(tcx, trait_id)
 286          }
 287      }
 288  
 289      // We need to do a bunch of special handling for default methods.
 290      // We need to modify the def_id and our substs in order to monomorphize
 291      // the function.
 292      let (is_default, def_id, substs, self_vtables, vtables) =
 293          match ty::provided_source(tcx, def_id) {
 294          None => (false, def_id, substs, None, vtables),
 295          Some(source_id) => {
 296              // There are two relevant substitutions when compiling
 297              // default methods. First, there is the substitution for
 298              // the type parameters of the impl we are using and the
 299              // method we are calling. This substitution is the substs
 300              // argument we already have.
 301              // In order to compile a default method, though, we need
 302              // to consider another substitution: the substitution for
 303              // the type parameters on trait; the impl we are using
 304              // implements the trait at some particular type
 305              // parameters, and we need to substitute for those first.
 306              // So, what we need to do is find this substitution and
 307              // compose it with the one we already have.
 308  
 309              let impl_id = ty::method(tcx, def_id).container_id();
 310              let method = ty::method(tcx, source_id);
 311              let trait_ref = ty::impl_trait_ref(tcx, impl_id)
 312                  .expect("could not find trait_ref for impl with \
 313                           default methods");
 314  
 315              // Compute the first substitution
 316              let first_subst = make_substs_for_receiver_types(
 317                  tcx, impl_id, trait_ref, method);
 318  
 319              // And compose them
 320              let new_substs = first_subst.subst(tcx, &substs);
 321  
 322  
 323              let (param_vtables, self_vtables) =
 324                  resolve_default_method_vtables(bcx, impl_id,
 325                                                 method, &substs, vtables);
 326  
 327              debug!("trans_fn_with_vtables - default method: \
 328                      substs = {}, trait_subst = {}, \
 329                      first_subst = {}, new_subst = {}, \
 330                      vtables = {}, \
 331                      self_vtable = {}, param_vtables = {}",
 332                     substs.repr(tcx), trait_ref.substs.repr(tcx),
 333                     first_subst.repr(tcx), new_substs.repr(tcx),
 334                     vtables.repr(tcx),
 335                     self_vtables.repr(tcx), param_vtables.repr(tcx));
 336  
 337              (true, source_id,
 338               new_substs, Some(self_vtables), Some(param_vtables))
 339          }
 340      };
 341  
 342      // Check whether this fn has an inlined copy and, if so, redirect
 343      // def_id to the local id of the inlined copy.
 344      let def_id = {
 345          if def_id.krate != ast::LOCAL_CRATE {
 346              inline::maybe_instantiate_inline(ccx, def_id)
 347          } else {
 348              def_id
 349          }
 350      };
 351  
 352      // We must monomorphise if the fn has type parameters, is a rust
 353      // intrinsic, or is a default method.  In particular, if we see an
 354      // intrinsic that is inlined from a different crate, we want to reemit the
 355      // intrinsic instead of trying to call it in the other crate.
 356      let must_monomorphise = if type_params.len() > 0 || is_default {
 357          true
 358      } else if def_id.krate == ast::LOCAL_CRATE {
 359          let map_node = session::expect(
 360              ccx.sess(),
 361              tcx.map.find(def_id.node),
 362              || format!("local item should be in ast map"));
 363  
 364          match map_node {
 365              ast_map::NodeForeignItem(_) => {
 366                  tcx.map.get_foreign_abi(def_id.node) == synabi::RustIntrinsic
 367              }
 368              _ => false
 369          }
 370      } else {
 371          false
 372      };
 373  
 374      // Create a monomorphic verison of generic functions
 375      if must_monomorphise {
 376          // Should be either intra-crate or inlined.
 377          assert_eq!(def_id.krate, ast::LOCAL_CRATE);
 378  
 379          let opt_ref_id = match node {
 380              ExprId(id) => if id != 0 { Some(id) } else { None },
 381              MethodCall(_) => None,
 382          };
 383  
 384          let (val, must_cast) =
 385              monomorphize::monomorphic_fn(ccx, def_id, &substs,
 386                                           vtables, self_vtables,
 387                                           opt_ref_id);
 388          let mut val = val;
 389          if must_cast && node != ExprId(0) {
 390              // Monotype of the REFERENCE to the function (type params
 391              // are subst'd)
 392              let ref_ty = match node {
 393                  ExprId(id) => node_id_type(bcx, id),
 394                  MethodCall(method_call) => {
 395                      let t = bcx.ccx().maps.method_map.borrow().get(&method_call).ty;
 396                      monomorphize_type(bcx, t)
 397                  }
 398              };
 399  
 400              val = PointerCast(
 401                  bcx, val, type_of::type_of_fn_from_ty(ccx, ref_ty).ptr_to());
 402          }
 403          return val;
 404      }
 405  
 406      // Find the actual function pointer.
 407      let mut val = {
 408          if def_id.krate == ast::LOCAL_CRATE {
 409              // Internal reference.
 410              get_item_val(ccx, def_id.node)
 411          } else {
 412              // External reference.
 413              trans_external_path(ccx, def_id, fn_tpt.ty)
 414          }
 415      };
 416  
 417      // This is subtle and surprising, but sometimes we have to bitcast
 418      // the resulting fn pointer.  The reason has to do with external
 419      // functions.  If you have two crates that both bind the same C
 420      // library, they may not use precisely the same types: for
 421      // example, they will probably each declare their own structs,
 422      // which are distinct types from LLVM's point of view (nominal
 423      // types).
 424      //
 425      // Now, if those two crates are linked into an application, and
 426      // they contain inlined code, you can wind up with a situation
 427      // where both of those functions wind up being loaded into this
 428      // application simultaneously. In that case, the same function
 429      // (from LLVM's point of view) requires two types. But of course
 430      // LLVM won't allow one function to have two types.
 431      //
 432      // What we currently do, therefore, is declare the function with
 433      // one of the two types (whichever happens to come first) and then
 434      // bitcast as needed when the function is referenced to make sure
 435      // it has the type we expect.
 436      //
 437      // This can occur on either a crate-local or crate-external
 438      // reference. It also occurs when testing libcore and in some
 439      // other weird situations. Annoying.
 440      let llty = type_of::type_of_fn_from_ty(ccx, fn_tpt.ty);
 441      let llptrty = llty.ptr_to();
 442      if val_ty(val) != llptrty {
 443          val = BitCast(bcx, val, llptrty);
 444      }
 445  
 446      val
 447  }

 515  pub fn trans_call_inner<'a>(
 516                          bcx: &'a Block<'a>,
 517                          call_info: Option<NodeInfo>,
 518                          callee_ty: ty::t,
 519                          get_callee: |bcx: &'a Block<'a>,
 520                                       arg_cleanup_scope: cleanup::ScopeId|
 521                                       -> Callee<'a>,
 522                          args: CallArgs,
 523                          dest: Option<expr::Dest>)
 524                          -> Result<'a> {
 525      /*!
 526       * This behemoth of a function translates function calls.
 527       * Unfortunately, in order to generate more efficient LLVM
 528       * output at -O0, it has quite a complex signature (refactoring
 529       * this into two functions seems like a good idea).
 530       *
 531       * In particular, for lang items, it is invoked with a dest of
 532       * None, and in that case the return value contains the result of
 533       * the fn. The lang item must not return a structural type or else
 534       * all heck breaks loose.
 535       *
 536       * For non-lang items, `dest` is always Some, and hence the result
 537       * is written into memory somewhere. Nonetheless we return the
 538       * actual return value of the function.
 539       */
 540  
 541      // Introduce a temporary cleanup scope that will contain cleanups
 542      // for the arguments while they are being evaluated. The purpose
 543      // this cleanup is to ensure that, should a failure occur while
 544      // evaluating argument N, the values for arguments 0...N-1 are all
 545      // cleaned up. If no failure occurs, the values are handed off to
 546      // the callee, and hence none of the cleanups in this temporary
 547      // scope will ever execute.
 548      let fcx = bcx.fcx;
 549      let ccx = fcx.ccx;
 550      let arg_cleanup_scope = fcx.push_custom_cleanup_scope();
 551  
 552      let callee = get_callee(bcx, cleanup::CustomScope(arg_cleanup_scope));
 553      let mut bcx = callee.bcx;
 554  
 555      let (llfn, llenv, llself) = match callee.data {
 556          Fn(llfn) => {
 557              (llfn, None, None)
 558          }
 559          TraitMethod(d) => {
 560              (d.llfn, None, Some(d.llself))
 561          }
 562          Closure(d) => {
 563              // Closures are represented as (llfn, llclosure) pair:
 564              // load the requisite values out.
 565              let pair = d.to_llref();
 566              let llfn = GEPi(bcx, pair, [0u, abi::fn_field_code]);
 567              let llfn = Load(bcx, llfn);
 568              let llenv = GEPi(bcx, pair, [0u, abi::fn_field_box]);
 569              let llenv = Load(bcx, llenv);
 570              (llfn, Some(llenv), None)
 571          }
 572      };
 573  
 574      let (abi, ret_ty) = match ty::get(callee_ty).sty {
 575          ty::ty_bare_fn(ref f) => (f.abi, f.sig.output),
 576          ty::ty_closure(ref f) => (synabi::Rust, f.sig.output),
 577          _ => fail!("expected bare rust fn or closure in trans_call_inner")
 578      };
 579      let is_rust_fn = abi == synabi::Rust || abi == synabi::RustIntrinsic;
 580  
 581      // Generate a location to store the result. If the user does
 582      // not care about the result, just make a stack slot.
 583      let opt_llretslot = match dest {
 584          None => {
 585              assert!(!type_of::return_uses_outptr(ccx, ret_ty));
 586              None
 587          }
 588          Some(expr::SaveIn(dst)) => Some(dst),
 589          Some(expr::Ignore) => {
 590              if !type_is_zero_size(ccx, ret_ty) {
 591                  Some(alloc_ty(bcx, ret_ty, "__llret"))
 592              } else {
 593                  let llty = type_of::type_of(ccx, ret_ty);
 594                  Some(C_undef(llty.ptr_to()))
 595              }
 596          }
 597      };
 598  
 599      let mut llresult = unsafe {
 600          llvm::LLVMGetUndef(Type::nil(ccx).ptr_to().to_ref())
 601      };
 602  
 603      // The code below invokes the function, using either the Rust
 604      // conventions (if it is a rust fn) or the native conventions
 605      // (otherwise).  The important part is that, when all is sad
 606      // and done, either the return value of the function will have been
 607      // written in opt_llretslot (if it is Some) or `llresult` will be
 608      // set appropriately (otherwise).
 609      if is_rust_fn {
 610          let mut llargs = Vec::new();
 611  
 612          // Push the out-pointer if we use an out-pointer for this
 613          // return type, otherwise push "undef".
 614          if type_of::return_uses_outptr(ccx, ret_ty) {
 615              llargs.push(opt_llretslot.unwrap());
 616          }
 617  
 618          // start at 1, because index 0 is the return value of the llvm func
 619          let mut first_arg_offset = 1;
 620  
 621          // Push the environment (or a trait object's self).
 622          match (llenv, llself) {
 623              (Some(llenv), None) => {
 624                  first_arg_offset += 1;
 625                  llargs.push(llenv)
 626              },
 627              (None, Some(llself)) => llargs.push(llself),
 628              _ => {}
 629          }
 630  
 631          // Push the arguments.
 632          bcx = trans_args(bcx, args, callee_ty, &mut llargs,
 633                           cleanup::CustomScope(arg_cleanup_scope),
 634                           llself.is_some());
 635  
 636          fcx.pop_custom_cleanup_scope(arg_cleanup_scope);
 637  
 638          // A function pointer is called without the declaration
 639          // available, so we have to apply any attributes with ABI
 640          // implications directly to the call instruction. Right now,
 641          // the only attribute we need to worry about is `sret`.
 642          let mut attrs = Vec::new();
 643          if type_of::return_uses_outptr(ccx, ret_ty) {
 644              attrs.push((1, StructRetAttribute));
 645              // The outptr can be noalias and nocapture because it's entirely
 646              // invisible to the program.
 647              attrs.push((1, NoAliasAttribute));
 648              attrs.push((1, NoCaptureAttribute));
 649              first_arg_offset += 1;
 650          }
 651  
 652          // The `noalias` attribute on the return value is useful to a
 653          // function ptr caller.
 654          match ty::get(ret_ty).sty {
 655              // `~` pointer return values never alias because ownership
 656              // is transferred
 657              ty::ty_uniq(..) | ty::ty_vec(_, ty::vstore_uniq) => {
 658                  attrs.push((0, NoAliasAttribute));
 659              }
 660              _ => {}
 661          }
 662  
 663          debug!("trans_callee_inner: first_arg_offset={}", first_arg_offset);
 664  
 665          for (idx, &t) in ty::ty_fn_args(callee_ty).iter().enumerate()
 666                                                    .map(|(i, v)| (i+first_arg_offset, v)) {
 667              use middle::ty::{BrAnon, ReLateBound};
 668              if !type_is_immediate(ccx, t) {
 669                  // if it's not immediate, we have a program-invisible pointer,
 670                  // which it can't possibly capture
 671                  attrs.push((idx, NoCaptureAttribute));
 672                  debug!("trans_callee_inner: argument {} nocapture because it's non-immediate", idx);
 673                  continue;
 674              }
 675  
 676              let t_ = ty::get(t);
 677              match t_.sty {
 678                  ty::ty_rptr(ReLateBound(_, BrAnon(_)), _) => {
 679                      debug!("trans_callee_inner: argument {} nocapture because \
 680                             of anonymous lifetime", idx);
 681                      attrs.push((idx, NoCaptureAttribute));
 682                  },
 683                  _ => { }
 684              }
 685          }
 686  
 687          // Invoke the actual rust fn and update bcx/llresult.
 688          let (llret, b) = base::invoke(bcx,
 689                                        llfn,
 690                                        llargs,
 691                                        attrs.as_slice(),
 692                                        call_info);
 693          bcx = b;
 694          llresult = llret;
 695  
 696          // If the Rust convention for this type is return via
 697          // the return value, copy it into llretslot.
 698          match opt_llretslot {
 699              Some(llretslot) => {
 700                  if !type_of::return_uses_outptr(bcx.ccx(), ret_ty) &&
 701                      !type_is_zero_size(bcx.ccx(), ret_ty)
 702                  {
 703                      Store(bcx, llret, llretslot);
 704                  }
 705              }
 706              None => {}
 707          }
 708      } else {
 709          // Lang items are the only case where dest is None, and
 710          // they are always Rust fns.
 711          assert!(dest.is_some());
 712  
 713          let mut llargs = Vec::new();
 714          let arg_tys = match args {
 715              ArgExprs(a) => a.iter().map(|x| expr_ty(bcx, *x)).collect(),
 716              _ => fail!("expected arg exprs.")
 717          };
 718          bcx = trans_args(bcx, args, callee_ty, &mut llargs,
 719                           cleanup::CustomScope(arg_cleanup_scope), false);
 720          fcx.pop_custom_cleanup_scope(arg_cleanup_scope);
 721          bcx = foreign::trans_native_call(bcx, callee_ty,
 722                                           llfn, opt_llretslot.unwrap(),
 723                                           llargs.as_slice(), arg_tys);
 724      }
 725  
 726      // If the caller doesn't care about the result of this fn call,
 727      // drop the temporary slot we made.
 728      match dest {
 729          None => {
 730              assert!(!type_of::return_uses_outptr(bcx.ccx(), ret_ty));
 731          }
 732          Some(expr::Ignore) => {
 733              // drop the value if it is not being saved.
 734              bcx = glue::drop_ty(bcx, opt_llretslot.unwrap(), ret_ty);
 735          }
 736          Some(expr::SaveIn(_)) => { }
 737      }
 738  
 739      if ty::type_is_bot(ret_ty) {
 740          Unreachable(bcx);
 741      }
 742  
 743      rslt(bcx, llresult)
 744  }

 746  pub enum CallArgs<'a> {
 747      // Supply value of arguments as a list of expressions that must be
 748      // translated. This is used in the common case of `foo(bar, qux)`.
 749      ArgExprs(&'a [@ast::Expr]),
 750  
 751      // Supply value of arguments as a list of LLVM value refs; frequently
 752      // used with lang items and so forth, when the argument is an internal
 753      // value.
 754      ArgVals(&'a [ValueRef]),
 755  
 756      // For overloaded operators: `(lhs, Option(rhs, rhs_id))`. `lhs`
 757      // is the left-hand-side and `rhs/rhs_id` is the datum/expr-id of
 758      // the right-hand-side (if any).
 759      ArgOverloadedOp(Datum<Expr>, Option<(Datum<Expr>, ast::NodeId)>),
 760  }
 761  
 762  fn trans_args<'a>(cx: &'a Block<'a>,
 763                    args: CallArgs,
 764                    fn_ty: ty::t,
 765                    llargs: &mut Vec<ValueRef> ,
 766                    arg_cleanup_scope: cleanup::ScopeId,
 767                    ignore_self: bool)
 768                    -> &'a Block<'a> {
 769      let _icx = push_ctxt("trans_args");
 770      let arg_tys = ty::ty_fn_args(fn_ty);
 771      let variadic = ty::fn_is_variadic(fn_ty);
 772  
 773      let mut bcx = cx;
 774  
 775      // First we figure out the caller's view of the types of the arguments.
 776      // This will be needed if this is a generic call, because the callee has
 777      // to cast her view of the arguments to the caller's view.
 778      match args {
 779          ArgExprs(arg_exprs) => {
 780              let num_formal_args = arg_tys.len();
 781              for (i, &arg_expr) in arg_exprs.iter().enumerate() {
 782                  if i == 0 && ignore_self {
 783                      continue;
 784                  }
 785                  let arg_ty = if i >= num_formal_args {
 786                      assert!(variadic);
 787                      expr_ty_adjusted(cx, arg_expr)
 788                  } else {
 789                      *arg_tys.get(i)
 790                  };
 791  
 792                  let arg_datum = unpack_datum!(bcx, expr::trans(bcx, arg_expr));
 793                  llargs.push(unpack_result!(bcx, {
 794                      trans_arg_datum(bcx, arg_ty, arg_datum,
 795                                      arg_cleanup_scope,
 796                                      DontAutorefArg)
 797                  }));
 798              }
 799          }
 800          ArgOverloadedOp(lhs, rhs) => {
 801              assert!(!variadic);
 802  
 803              llargs.push(unpack_result!(bcx, {
 804                  trans_arg_datum(bcx, *arg_tys.get(0), lhs,
 805                                  arg_cleanup_scope,
 806                                  DontAutorefArg)
 807              }));
 808  
 809              match rhs {
 810                  Some((rhs, rhs_id)) => {
 811                      assert_eq!(arg_tys.len(), 2);
 812  
 813                      llargs.push(unpack_result!(bcx, {
 814                          trans_arg_datum(bcx, *arg_tys.get(1), rhs,
 815                                          arg_cleanup_scope,
 816                                          DoAutorefArg(rhs_id))
 817                      }));
 818                  }
 819                  None => assert_eq!(arg_tys.len(), 1)
 820              }
 821          }
 822          ArgVals(vs) => {
 823              llargs.push_all(vs);
 824          }
 825      }
 826  
 827      bcx
 828  }

 830  pub enum AutorefArg {
 831      DontAutorefArg,
 832      DoAutorefArg(ast::NodeId)
 833  }
 834  
 835  pub fn trans_arg_datum<'a>(

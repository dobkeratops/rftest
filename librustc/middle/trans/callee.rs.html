  62  pub enum CalleeData {

  64  
  65      // Represents a (possibly monomorphized) top-level fn item or method
  66      // item. Note that this is just the fn-ptr and is not a Rust closure

  78  fn trans<'a>(bcx: &'a Block<'a>, expr: &ast::Expr) -> Callee<'a> {
  79      let _icx = push_ctxt("trans_callee");
  80      debug!("callee::trans(expr={})", expr.repr(bcx.tcx()));
  81  
  82      // pick out special kinds of expressions that can be called:
  83      match expr.node {
  84          ast::ExprPath(_) => {
  85              return trans_def(bcx, bcx.def(expr.id), expr);
  86          }
  87          _ => {}
  88      }
  89  
  90      // any other expressions are closures:
  91      return datum_callee(bcx, expr);
  92  
  93      fn datum_callee<'a>(bcx: &'a Block<'a>, expr: &ast::Expr) -> Callee<'a> {
  94          let DatumBlock {bcx: mut bcx, datum} = expr::trans(bcx, expr);
  95          match ty::get(datum.ty).sty {
  96              ty::ty_bare_fn(..) => {
  97                  let llval = datum.to_llscalarish(bcx);
  98                  return Callee {bcx: bcx, data: Fn(llval)};
  99              }
 100              ty::ty_closure(..) => {
 101                  let datum = unpack_datum!(
 102                      bcx, datum.to_lvalue_datum(bcx, "callee", expr.id));
 103                  return Callee {bcx: bcx, data: Closure(datum)};
 104              }
 105              _ => {
 106                  bcx.tcx().sess.span_bug(
 107                      expr.span,
 108                      format!("type of callee is neither bare-fn nor closure: {}",
 109                           bcx.ty_to_str(datum.ty)));
 110              }
 111          }
 112      }
 113  
 114      fn fn_callee<'a>(bcx: &'a Block<'a>, llfn: ValueRef) -> Callee<'a> {
 115          return Callee {bcx: bcx, data: Fn(llfn)};
 116      }
 117  
 118      fn trans_def<'a>(bcx: &'a Block<'a>, def: ast::Def, ref_expr: &ast::Expr)
 119                   -> Callee<'a> {
 120          match def {
 121              ast::DefFn(did, _) |
 122              ast::DefStaticMethod(did, ast::FromImpl(_), _) => {
 123                  fn_callee(bcx, trans_fn_ref(bcx, did, ExprId(ref_expr.id)))
 124              }
 125              ast::DefStaticMethod(impl_did,
 126                                     ast::FromTrait(trait_did),
 127                                     _) => {
 128                  fn_callee(bcx, meth::trans_static_method_callee(bcx, impl_did,
 129                                                                  trait_did,
 130                                                                  ref_expr.id))
 131              }
 132              ast::DefVariant(tid, vid, _) => {
 133                  // nullary variants are not callable
 134                  assert!(ty::enum_variant_with_id(bcx.tcx(),
 135                                                        tid,
 136                                                        vid).args.len() > 0u);
 137                  fn_callee(bcx, trans_fn_ref(bcx, vid, ExprId(ref_expr.id)))
 138              }
 139              ast::DefStruct(def_id) => {
 140                  fn_callee(bcx, trans_fn_ref(bcx, def_id, ExprId(ref_expr.id)))
 141              }
 142              ast::DefStatic(..) |
 143              ast::DefArg(..) |
 144              ast::DefLocal(..) |
 145              ast::DefBinding(..) |
 146              ast::DefUpvar(..) => {
 147                  datum_callee(bcx, ref_expr)
 148              }
 149              ast::DefMod(..) | ast::DefForeignMod(..) | ast::DefTrait(..) |
 150              ast::DefTy(..) | ast::DefPrimTy(..) |
 151              ast::DefUse(..) | ast::DefTyParamBinder(..) |
 152              ast::DefRegion(..) | ast::DefLabel(..) | ast::DefTyParam(..) |
 153              ast::DefSelfTy(..) | ast::DefMethod(..) => {
 154                  bcx.tcx().sess.span_bug(
 155                      ref_expr.span,
 156                      format!("cannot translate def {:?} \
 157                            to a callable thing!", def));
 158              }
 159          }
 160      }
 161  }

 163  pub fn trans_fn_ref(bcx: &Block, def_id: ast::DefId, node: ExprOrMethodCall) -> ValueRef {
 164      /*!
 165       *
 166       * Translates a reference (with id `ref_id`) to the fn/method
 167       * with id `def_id` into a function pointer.  This may require
 168       * monomorphization or inlining. */
 169  
 170      let _icx = push_ctxt("trans_fn_ref");
 171  
 172      let type_params = node_id_type_params(bcx, node);
 173      let vtable_key = match node {
 174          ExprId(id) => MethodCall::expr(id),
 175          MethodCall(method_call) => method_call
 176      };
 177      let vtables = node_vtables(bcx, vtable_key);
 178      debug!("trans_fn_ref(def_id={}, node={:?}, type_params={}, vtables={})",
 179             def_id.repr(bcx.tcx()), node, type_params.repr(bcx.tcx()),
 180             vtables.repr(bcx.tcx()));
 181      trans_fn_ref_with_vtables(bcx, def_id, node,
 182                                type_params.as_slice(),
 183                                vtables)
 184  }

 186  fn trans_fn_ref_with_vtables_to_callee<'a>(bcx: &'a Block<'a>,
 187                                             def_id: ast::DefId,
 188                                             ref_id: ast::NodeId,
 189                                             type_params: &[ty::t],
 190                                             vtables: Option<typeck::vtable_res>)
 191                                             -> Callee<'a> {
 192      Callee {bcx: bcx,
 193              data: Fn(trans_fn_ref_with_vtables(bcx, def_id, ExprId(ref_id),
 194                                                 type_params, vtables))}
 195  }

 197  fn resolve_default_method_vtables(bcx: &Block,
 198                                    impl_id: ast::DefId,
 199                                    method: &ty::Method,
 200                                    substs: &ty::substs,
 201                                    impl_vtables: Option<typeck::vtable_res>)
 202                            -> (typeck::vtable_res, typeck::vtable_param_res) {
 203  
 204      // Get the vtables that the impl implements the trait at
 205      let impl_res = ty::lookup_impl_vtables(bcx.tcx(), impl_id);
 206  
 207      // Build up a param_substs that we are going to resolve the
 208      // trait_vtables under.
 209      let param_substs = Some(@param_substs {
 210          tys: substs.tps.clone(),
 211          self_ty: substs.self_ty,
 212          vtables: impl_vtables,
 213          self_vtables: None
 214      });
 215  
 216      let trait_vtables_fixed = resolve_vtables_under_param_substs(
 217          bcx.tcx(), param_substs, impl_res.trait_vtables);
 218  
 219      // Now we pull any vtables for parameters on the actual method.
 220      let num_method_vtables = method.generics.type_param_defs().len();
 221      let method_vtables = match impl_vtables {
 222          Some(vtables) => {
 223              let num_impl_type_parameters =
 224                  vtables.len() - num_method_vtables;
 225              vtables.tailn(num_impl_type_parameters).to_owned()
 226          },
 227          None => slice::from_elem(num_method_vtables, @Vec::new())
 228      };
 229  
 230      let param_vtables = @(vec::append((*trait_vtables_fixed).clone(),
 231                                            method_vtables));
 232  
 233      let self_vtables = resolve_param_vtables_under_param_substs(
 234          bcx.tcx(), param_substs, impl_res.self_vtables);
 235  
 236      (param_vtables, self_vtables)
 237  }

 240  pub fn trans_fn_ref_with_vtables(
 241          bcx: &Block,       //
 242          def_id: ast::DefId,   // def id of fn
 243          node: ExprOrMethodCall,  // node id of use of fn; may be zero if N/A
 244          type_params: &[ty::t], // values for fn's ty params
 245          vtables: Option<typeck::vtable_res>) // vtables for the call
 246       -> ValueRef {
 247      /*!
 248       * Translates a reference to a fn/method item, monomorphizing and
 249       * inlining as it goes.
 250       *
 251       * # Parameters
 252       *
 253       * - `bcx`: the current block where the reference to the fn occurs
 254       * - `def_id`: def id of the fn or method item being referenced
 255       * - `node`: node id of the reference to the fn/method, if applicable.
 256       *   This parameter may be zero; but, if so, the resulting value may not
 257       *   have the right type, so it must be cast before being used.
 258       * - `type_params`: values for each of the fn/method's type parameters
 259       * - `vtables`: values for each bound on each of the type parameters
 260       */
 261  
 262      let _icx = push_ctxt("trans_fn_ref_with_vtables");
 263      let ccx = bcx.ccx();
 264      let tcx = bcx.tcx();
 265  
 266      debug!("trans_fn_ref_with_vtables(bcx={}, def_id={}, node={:?}, \
 267              type_params={}, vtables={})",
 268             bcx.to_str(),
 269             def_id.repr(tcx),
 270             node,
 271             type_params.repr(tcx),
 272             vtables.repr(tcx));
 273  
 274      assert!(type_params.iter().all(|t| !ty::type_needs_infer(*t)));
 275  
 276      // Polytype of the function item (may have type params)
 277      let fn_tpt = ty::lookup_item_type(tcx, def_id);
 278  
 279      let substs = ty::substs { regions: ty::ErasedRegions,
 280                                self_ty: None,
 281                                tps: /*bad*/ Vec::from_slice(type_params) };
 282  
 283      // Load the info for the appropriate trait if necessary.
 284      match ty::trait_of_method(tcx, def_id) {
 285          None => {}
 286          Some(trait_id) => {
 287              ty::populate_implementations_for_trait_if_necessary(tcx, trait_id)
 288          }
 289      }
 290  
 291      // We need to do a bunch of special handling for default methods.
 292      // We need to modify the def_id and our substs in order to monomorphize
 293      // the function.
 294      let (is_default, def_id, substs, self_vtables, vtables) =
 295          match ty::provided_source(tcx, def_id) {
 296          None => (false, def_id, substs, None, vtables),
 297          Some(source_id) => {
 298              // There are two relevant substitutions when compiling
 299              // default methods. First, there is the substitution for
 300              // the type parameters of the impl we are using and the
 301              // method we are calling. This substitution is the substs
 302              // argument we already have.
 303              // In order to compile a default method, though, we need
 304              // to consider another substitution: the substitution for
 305              // the type parameters on trait; the impl we are using
 306              // implements the trait at some particular type
 307              // parameters, and we need to substitute for those first.
 308              // So, what we need to do is find this substitution and
 309              // compose it with the one we already have.
 310  
 311              let impl_id = ty::method(tcx, def_id).container_id();
 312              let method = ty::method(tcx, source_id);
 313              let trait_ref = ty::impl_trait_ref(tcx, impl_id)
 314                  .expect("could not find trait_ref for impl with \
 315                           default methods");
 316  
 317              // Compute the first substitution
 318              let first_subst = make_substs_for_receiver_types(
 319                  tcx, impl_id, trait_ref, method);
 320  
 321              // And compose them
 322              let new_substs = first_subst.subst(tcx, &substs);
 323  
 324  
 325              let (param_vtables, self_vtables) =
 326                  resolve_default_method_vtables(bcx, impl_id,
 327                                                 method, &substs, vtables);
 328  
 329              debug!("trans_fn_with_vtables - default method: \
 330                      substs = {}, trait_subst = {}, \
 331                      first_subst = {}, new_subst = {}, \
 332                      vtables = {}, \
 333                      self_vtable = {}, param_vtables = {}",
 334                     substs.repr(tcx), trait_ref.substs.repr(tcx),
 335                     first_subst.repr(tcx), new_substs.repr(tcx),
 336                     vtables.repr(tcx),
 337                     self_vtables.repr(tcx), param_vtables.repr(tcx));
 338  
 339              (true, source_id,
 340               new_substs, Some(self_vtables), Some(param_vtables))
 341          }
 342      };
 343  
 344      // Check whether this fn has an inlined copy and, if so, redirect
 345      // def_id to the local id of the inlined copy.
 346      let def_id = {
 347          if def_id.krate != ast::LOCAL_CRATE {
 348              inline::maybe_instantiate_inline(ccx, def_id)
 349          } else {
 350              def_id
 351          }
 352      };
 353  
 354      // We must monomorphise if the fn has type parameters, is a rust
 355      // intrinsic, or is a default method.  In particular, if we see an
 356      // intrinsic that is inlined from a different crate, we want to reemit the
 357      // intrinsic instead of trying to call it in the other crate.
 358      let must_monomorphise = if type_params.len() > 0 || is_default {
 359          true
 360      } else if def_id.krate == ast::LOCAL_CRATE {
 361          let map_node = session::expect(
 362              ccx.sess(),
 363              tcx.map.find(def_id.node),
 364              || format!("local item should be in ast map"));
 365  
 366          match map_node {
 367              ast_map::NodeForeignItem(_) => {
 368                  tcx.map.get_foreign_abis(def_id.node).is_intrinsic()
 369              }
 370              _ => false
 371          }
 372      } else {
 373          false
 374      };
 375  
 376      // Create a monomorphic verison of generic functions
 377      if must_monomorphise {
 378          // Should be either intra-crate or inlined.
 379          assert_eq!(def_id.krate, ast::LOCAL_CRATE);
 380  
 381          let opt_ref_id = match node {
 382              ExprId(id) => if id != 0 { Some(id) } else { None },
 383              MethodCall(_) => None,
 384          };
 385  
 386          let (val, must_cast) =
 387              monomorphize::monomorphic_fn(ccx, def_id, &substs,
 388                                           vtables, self_vtables,
 389                                           opt_ref_id);
 390          let mut val = val;
 391          if must_cast && node != ExprId(0) {
 392              // Monotype of the REFERENCE to the function (type params
 393              // are subst'd)
 394              let ref_ty = match node {
 395                  ExprId(id) => node_id_type(bcx, id),
 396                  MethodCall(method_call) => {
 397                      let t = bcx.ccx().maps.method_map.borrow().get(&method_call).ty;
 398                      monomorphize_type(bcx, t)
 399                  }
 400              };
 401  
 402              val = PointerCast(
 403                  bcx, val, type_of::type_of_fn_from_ty(ccx, ref_ty).ptr_to());
 404          }
 405          return val;
 406      }
 407  
 408      // Find the actual function pointer.
 409      let mut val = {
 410          if def_id.krate == ast::LOCAL_CRATE {
 411              // Internal reference.
 412              get_item_val(ccx, def_id.node)
 413          } else {
 414              // External reference.
 415              trans_external_path(ccx, def_id, fn_tpt.ty)
 416          }
 417      };
 418  
 419      // This is subtle and surprising, but sometimes we have to bitcast
 420      // the resulting fn pointer.  The reason has to do with external
 421      // functions.  If you have two crates that both bind the same C
 422      // library, they may not use precisely the same types: for
 423      // example, they will probably each declare their own structs,
 424      // which are distinct types from LLVM's point of view (nominal
 425      // types).
 426      //
 427      // Now, if those two crates are linked into an application, and
 428      // they contain inlined code, you can wind up with a situation
 429      // where both of those functions wind up being loaded into this
 430      // application simultaneously. In that case, the same function
 431      // (from LLVM's point of view) requires two types. But of course
 432      // LLVM won't allow one function to have two types.
 433      //
 434      // What we currently do, therefore, is declare the function with
 435      // one of the two types (whichever happens to come first) and then
 436      // bitcast as needed when the function is referenced to make sure
 437      // it has the type we expect.
 438      //
 439      // This can occur on either a crate-local or crate-external
 440      // reference. It also occurs when testing libcore and in some
 441      // other weird situations. Annoying.
 442      let llty = type_of::type_of_fn_from_ty(ccx, fn_tpt.ty);
 443      let llptrty = llty.ptr_to();
 444      if val_ty(val) != llptrty {
 445          val = BitCast(bcx, val, llptrty);
 446      }
 447  
 448      val
 449  }

 559  pub fn trans_call_inner<'a>(
 560                          bcx: &'a Block<'a>,
 561                          call_info: Option<NodeInfo>,
 562                          callee_ty: ty::t,
 563                          get_callee: |bcx: &'a Block<'a>,
 564                                       arg_cleanup_scope: cleanup::ScopeId|
 565                                       -> Callee<'a>,
 566                          args: CallArgs,
 567                          dest: Option<expr::Dest>)
 568                          -> Result<'a> {
 569      /*!
 570       * This behemoth of a function translates function calls.
 571       * Unfortunately, in order to generate more efficient LLVM
 572       * output at -O0, it has quite a complex signature (refactoring
 573       * this into two functions seems like a good idea).
 574       *
 575       * In particular, for lang items, it is invoked with a dest of
 576       * None, and in that case the return value contains the result of
 577       * the fn. The lang item must not return a structural type or else
 578       * all heck breaks loose.
 579       *
 580       * For non-lang items, `dest` is always Some, and hence the result
 581       * is written into memory somewhere. Nonetheless we return the
 582       * actual return value of the function.
 583       */
 584  
 585      // Introduce a temporary cleanup scope that will contain cleanups
 586      // for the arguments while they are being evaluated. The purpose
 587      // this cleanup is to ensure that, should a failure occur while
 588      // evaluating argument N, the values for arguments 0...N-1 are all
 589      // cleaned up. If no failure occurs, the values are handed off to
 590      // the callee, and hence none of the cleanups in this temporary
 591      // scope will ever execute.
 592      let fcx = bcx.fcx;
 593      let ccx = fcx.ccx;
 594      let arg_cleanup_scope = fcx.push_custom_cleanup_scope();
 595  
 596      let callee = get_callee(bcx, cleanup::CustomScope(arg_cleanup_scope));
 597      let mut bcx = callee.bcx;
 598  
 599      let (llfn, llenv, llself) = match callee.data {
 600          Fn(llfn) => {
 601              (llfn, None, None)
 602          }
 603          TraitMethod(d) => {
 604              (d.llfn, None, Some(d.llself))
 605          }
 606          Closure(d) => {
 607              // Closures are represented as (llfn, llclosure) pair:
 608              // load the requisite values out.
 609              let pair = d.to_llref();
 610              let llfn = GEPi(bcx, pair, [0u, abi::fn_field_code]);
 611              let llfn = Load(bcx, llfn);
 612              let llenv = GEPi(bcx, pair, [0u, abi::fn_field_box]);
 613              let llenv = Load(bcx, llenv);
 614              (llfn, Some(llenv), None)
 615          }
 616      };
 617  
 618      let (abi, ret_ty) = match ty::get(callee_ty).sty {
 619          ty::ty_bare_fn(ref f) => (f.abis, f.sig.output),
 620          ty::ty_closure(ref f) => (AbiSet::Rust(), f.sig.output),
 621          _ => fail!("expected bare rust fn or closure in trans_call_inner")
 622      };
 623      let is_rust_fn =
 624          abi.is_rust() ||
 625          abi.is_intrinsic();
 626  
 627      // Generate a location to store the result. If the user does
 628      // not care about the result, just make a stack slot.
 629      let opt_llretslot = match dest {
 630          None => {
 631              assert!(!type_of::return_uses_outptr(ccx, ret_ty));
 632              None
 633          }
 634          Some(expr::SaveIn(dst)) => Some(dst),
 635          Some(expr::Ignore) => {
 636              if !type_is_zero_size(ccx, ret_ty) {
 637                  Some(alloc_ty(bcx, ret_ty, "__llret"))
 638              } else {
 639                  let llty = type_of::type_of(ccx, ret_ty);
 640                  Some(C_undef(llty.ptr_to()))
 641              }
 642          }
 643      };
 644  
 645      let mut llresult = unsafe {
 646          llvm::LLVMGetUndef(Type::nil(ccx).ptr_to().to_ref())
 647      };
 648  
 649      // The code below invokes the function, using either the Rust
 650      // conventions (if it is a rust fn) or the native conventions
 651      // (otherwise).  The important part is that, when all is sad
 652      // and done, either the return value of the function will have been
 653      // written in opt_llretslot (if it is Some) or `llresult` will be
 654      // set appropriately (otherwise).
 655      if is_rust_fn {
 656          let mut llargs = Vec::new();
 657  
 658          // Push the out-pointer if we use an out-pointer for this
 659          // return type, otherwise push "undef".
 660          if type_of::return_uses_outptr(ccx, ret_ty) {
 661              llargs.push(opt_llretslot.unwrap());
 662          }
 663  
 664          // Push the environment (or a trait object's self).
 665          match (llenv, llself) {
 666              (Some(llenv), None) => llargs.push(llenv),
 667              (None, Some(llself)) => llargs.push(llself),
 668              _ => {}
 669          }
 670  
 671          // Push the arguments.
 672          bcx = trans_args(bcx, args, callee_ty, &mut llargs,
 673                           cleanup::CustomScope(arg_cleanup_scope),
 674                           llself.is_some());
 675  
 676          fcx.pop_custom_cleanup_scope(arg_cleanup_scope);
 677  
 678          // A function pointer is called without the declaration
 679          // available, so we have to apply any attributes with ABI
 680          // implications directly to the call instruction. Right now,
 681          // the only attribute we need to worry about is `sret`.
 682          let mut attrs = Vec::new();
 683          if type_of::return_uses_outptr(ccx, ret_ty) {
 684              attrs.push((1, StructRetAttribute));
 685          }
 686  
 687          // The `noalias` attribute on the return value is useful to a
 688          // function ptr caller.
 689          match ty::get(ret_ty).sty {
 690              // `~` pointer return values never alias because ownership
 691              // is transferred
 692              ty::ty_uniq(..) | ty::ty_vec(_, ty::vstore_uniq) => {
 693                  attrs.push((0, NoAliasAttribute));
 694              }
 695              _ => {}
 696          }
 697  
 698          // Invoke the actual rust fn and update bcx/llresult.
 699          let (llret, b) = base::invoke(bcx,
 700                                        llfn,
 701                                        llargs,
 702                                        attrs.as_slice(),
 703                                        call_info);
 704          bcx = b;
 705          llresult = llret;
 706  
 707          // If the Rust convention for this type is return via
 708          // the return value, copy it into llretslot.
 709          match opt_llretslot {
 710              Some(llretslot) => {
 711                  if !type_of::return_uses_outptr(bcx.ccx(), ret_ty) &&
 712                      !type_is_zero_size(bcx.ccx(), ret_ty)
 713                  {
 714                      Store(bcx, llret, llretslot);
 715                  }
 716              }
 717              None => {}
 718          }
 719      } else {
 720          // Lang items are the only case where dest is None, and
 721          // they are always Rust fns.
 722          assert!(dest.is_some());
 723  
 724          let mut llargs = Vec::new();
 725          let arg_tys = match args {
 726              ArgExprs(a) => a.iter().map(|x| expr_ty(bcx, *x)).collect(),
 727              _ => fail!("expected arg exprs.")
 728          };
 729          bcx = trans_args(bcx, args, callee_ty, &mut llargs,
 730                           cleanup::CustomScope(arg_cleanup_scope), false);
 731          fcx.pop_custom_cleanup_scope(arg_cleanup_scope);
 732          bcx = foreign::trans_native_call(bcx, callee_ty,
 733                                           llfn, opt_llretslot.unwrap(),
 734                                           llargs.as_slice(), arg_tys);
 735      }
 736  
 737      // If the caller doesn't care about the result of this fn call,
 738      // drop the temporary slot we made.
 739      match dest {
 740          None => {
 741              assert!(!type_of::return_uses_outptr(bcx.ccx(), ret_ty));
 742          }
 743          Some(expr::Ignore) => {
 744              // drop the value if it is not being saved.
 745              bcx = glue::drop_ty(bcx, opt_llretslot.unwrap(), ret_ty);
 746          }
 747          Some(expr::SaveIn(_)) => { }
 748      }
 749  
 750      if ty::type_is_bot(ret_ty) {
 751          Unreachable(bcx);
 752      }
 753  
 754      rslt(bcx, llresult)
 755  }

 757  pub enum CallArgs<'a> {
 758      // Supply value of arguments as a list of expressions that must be
 759      // translated. This is used in the common case of `foo(bar, qux)`.
 760      ArgExprs(&'a [@ast::Expr]),
 761  
 762      // Supply value of arguments as a list of LLVM value refs; frequently
 763      // used with lang items and so forth, when the argument is an internal
 764      // value.
 765      ArgVals(&'a [ValueRef]),
 766  
 767      // For overloaded operators: `(lhs, Option(rhs, rhs_id))`. `lhs`
 768      // is the left-hand-side and `rhs/rhs_id` is the datum/expr-id of
 769      // the right-hand-side (if any).
 770      ArgOverloadedOp(Datum<Expr>, Option<(Datum<Expr>, ast::NodeId)>),
 771  }
 772  
 773  fn trans_args<'a>(cx: &'a Block<'a>,
 774                    args: CallArgs,
 775                    fn_ty: ty::t,
 776                    llargs: &mut Vec<ValueRef> ,
 777                    arg_cleanup_scope: cleanup::ScopeId,
 778                    ignore_self: bool)
 779                    -> &'a Block<'a> {
 780      let _icx = push_ctxt("trans_args");
 781      let arg_tys = ty::ty_fn_args(fn_ty);
 782      let variadic = ty::fn_is_variadic(fn_ty);
 783  
 784      let mut bcx = cx;
 785  
 786      // First we figure out the caller's view of the types of the arguments.
 787      // This will be needed if this is a generic call, because the callee has
 788      // to cast her view of the arguments to the caller's view.
 789      match args {
 790          ArgExprs(arg_exprs) => {
 791              let num_formal_args = arg_tys.len();
 792              for (i, &arg_expr) in arg_exprs.iter().enumerate() {
 793                  if i == 0 && ignore_self {
 794                      continue;
 795                  }
 796                  let arg_ty = if i >= num_formal_args {
 797                      assert!(variadic);
 798                      expr_ty_adjusted(cx, arg_expr)
 799                  } else {
 800                      *arg_tys.get(i)
 801                  };
 802  
 803                  let arg_datum = unpack_datum!(bcx, expr::trans(bcx, arg_expr));
 804                  llargs.push(unpack_result!(bcx, {
 805                      trans_arg_datum(bcx, arg_ty, arg_datum,
 806                                      arg_cleanup_scope,
 807                                      DontAutorefArg)
 808                  }));
 809              }
 810          }
 811          ArgOverloadedOp(lhs, rhs) => {
 812              assert!(!variadic);
 813  
 814              llargs.push(unpack_result!(bcx, {
 815                  trans_arg_datum(bcx, *arg_tys.get(0), lhs,
 816                                  arg_cleanup_scope,
 817                                  DontAutorefArg)
 818              }));
 819  
 820              match rhs {
 821                  Some((rhs, rhs_id)) => {
 822                      assert_eq!(arg_tys.len(), 2);
 823  
 824                      llargs.push(unpack_result!(bcx, {
 825                          trans_arg_datum(bcx, *arg_tys.get(1), rhs,
 826                                          arg_cleanup_scope,
 827                                          DoAutorefArg(rhs_id))
 828                      }));
 829                  }
 830                  None => assert_eq!(arg_tys.len(), 1)
 831              }
 832          }
 833          ArgVals(vs) => {
 834              llargs.push_all(vs);
 835          }
 836      }
 837  
 838      bcx
 839  }

 842      DontAutorefArg,
 843      DoAutorefArg(ast::NodeId)
 844  }

 846  pub fn trans_arg_datum<'a>(

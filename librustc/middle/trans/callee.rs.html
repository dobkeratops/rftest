  52  // Represents a (possibly monomorphized) top-level fn item or method
  53  // item.  Note that this is just the fn-ptr and is not a Rust closure
  54  // value (which is a pair).
  55  pub struct FnData {
  56      llfn: ValueRef,
  57  }
  58  
  59  pub struct MethodData {
  60      llfn: ValueRef,
  61      llself: ValueRef,
  62      temp_cleanup: Option<ValueRef>,
  63      self_mode: ty::SelfMode,
  64  }

  66  pub enum CalleeData {
  67      Closure(Datum),
  68      Fn(FnData),
  69      Method(MethodData)
  70  }

  73      bcx: @mut Block,
  74      data: CalleeData

  77  pub fn trans(bcx: @mut Block, expr: @ast::expr) -> Callee {
  78      let _icx = push_ctxt("trans_callee");
  79      debug!("callee::trans(expr=%s)", expr.repr(bcx.tcx()));
  80  
  81      // pick out special kinds of expressions that can be called:
  82      match expr.node {
  83          ast::expr_path(_) => {
  84              return trans_def(bcx, bcx.def(expr.id), expr);
  85          }
  86          _ => {}
  87      }
  88  
  89      // any other expressions are closures:
  90      return datum_callee(bcx, expr);
  91  
  92      fn datum_callee(bcx: @mut Block, expr: @ast::expr) -> Callee {
  93          let DatumBlock {bcx, datum} = expr::trans_to_datum(bcx, expr);
  94          match ty::get(datum.ty).sty {
  95              ty::ty_bare_fn(*) => {
  96                  let llval = datum.to_appropriate_llval(bcx);
  97                  return Callee {bcx: bcx, data: Fn(FnData {llfn: llval})};
  98              }
  99              ty::ty_closure(*) => {
 100                  return Callee {bcx: bcx, data: Closure(datum)};
 101              }
 102              _ => {
 103                  bcx.tcx().sess.span_bug(
 104                      expr.span,
 105                      fmt!("Type of callee is neither bare-fn nor closure: %s",
 106                           bcx.ty_to_str(datum.ty)));
 107              }
 108          }
 109      }
 110  
 111      fn fn_callee(bcx: @mut Block, fd: FnData) -> Callee {
 112          return Callee {bcx: bcx, data: Fn(fd)};
 113      }
 114  
 115      fn trans_def(bcx: @mut Block, def: ast::def, ref_expr: @ast::expr) -> Callee {
 116          match def {
 117              ast::def_fn(did, _) | ast::def_static_method(did, None, _) => {
 118                  fn_callee(bcx, trans_fn_ref(bcx, did, ref_expr.id))
 119              }
 120              ast::def_static_method(impl_did, Some(trait_did), _) => {
 121                  fn_callee(bcx, meth::trans_static_method_callee(bcx, impl_did,
 122                                                                  trait_did,
 123                                                                  ref_expr.id))
 124              }
 125              ast::def_variant(tid, vid) => {
 126                  // nullary variants are not callable
 127                  assert!(ty::enum_variant_with_id(bcx.tcx(),
 128                                                        tid,
 129                                                        vid).args.len() > 0u);
 130                  fn_callee(bcx, trans_fn_ref(bcx, vid, ref_expr.id))
 131              }
 132              ast::def_struct(def_id) => {
 133                  fn_callee(bcx, trans_fn_ref(bcx, def_id, ref_expr.id))
 134              }
 135              ast::def_arg(*) |
 136              ast::def_local(*) |
 137              ast::def_binding(*) |
 138              ast::def_upvar(*) |
 139              ast::def_self(*) => {
 140                  datum_callee(bcx, ref_expr)
 141              }
 142              ast::def_mod(*) | ast::def_foreign_mod(*) | ast::def_trait(*) |
 143              ast::def_static(*) | ast::def_ty(*) | ast::def_prim_ty(*) |
 144              ast::def_use(*) | ast::def_typaram_binder(*) |
 145              ast::def_region(*) | ast::def_label(*) | ast::def_ty_param(*) |
 146              ast::def_self_ty(*) | ast::def_method(*) => {
 147                  bcx.tcx().sess.span_bug(
 148                      ref_expr.span,
 149                      fmt!("Cannot translate def %? \
 150                            to a callable thing!", def));
 151              }
 152          }
 153      }
 154  }

 156  pub fn trans_fn_ref_to_callee(bcx: @mut Block,

 163  pub fn trans_fn_ref(bcx: @mut Block,
 164                      def_id: ast::def_id,
 165                      ref_id: ast::NodeId) -> FnData {
 166      /*!
 167       *
 168       * Translates a reference (with id `ref_id`) to the fn/method
 169       * with id `def_id` into a function pointer.  This may require
 170       * monomorphization or inlining. */
 171  
 172      let _icx = push_ctxt("trans_fn_ref");
 173  
 174      let type_params = node_id_type_params(bcx, ref_id);
 175      let vtables = node_vtables(bcx, ref_id);
 176      debug!("trans_fn_ref(def_id=%s, ref_id=%?, type_params=%s, vtables=%s)",
 177             def_id.repr(bcx.tcx()), ref_id, type_params.repr(bcx.tcx()),
 178             vtables.repr(bcx.tcx()));
 179      trans_fn_ref_with_vtables(bcx, def_id, ref_id, type_params, vtables)
 180  }

 182  pub fn trans_fn_ref_with_vtables_to_callee(
 183          bcx: @mut Block,
 184          def_id: ast::def_id,
 185          ref_id: ast::NodeId,
 186          type_params: &[ty::t],
 187          vtables: Option<typeck::vtable_res>)
 188       -> Callee {
 189      Callee {bcx: bcx,
 190              data: Fn(trans_fn_ref_with_vtables(bcx, def_id, ref_id,
 191                                                 type_params, vtables))}
 192  }

 194  fn resolve_default_method_vtables(bcx: @mut Block,
 195                                    impl_id: ast::def_id,
 196                                    method: &ty::Method,
 197                                    substs: &ty::substs,
 198                                    impl_vtables: Option<typeck::vtable_res>)
 199                            -> (typeck::vtable_res, typeck::vtable_param_res) {
 200  
 201      // Get the vtables that the impl implements the trait at
 202      let impl_res = ty::lookup_impl_vtables(bcx.tcx(), impl_id);
 203  
 204      // Build up a param_substs that we are going to resolve the
 205      // trait_vtables under.
 206      let param_substs = Some(@param_substs {
 207          tys: substs.tps.clone(),
 208          self_ty: substs.self_ty,
 209          vtables: impl_vtables,
 210          self_vtables: None
 211      });
 212  
 213      let trait_vtables_fixed = resolve_vtables_under_param_substs(
 214          bcx.tcx(), param_substs, impl_res.trait_vtables);
 215  
 216      // Now we pull any vtables for parameters on the actual method.
 217      let num_method_vtables = method.generics.type_param_defs.len();
 218      let method_vtables = match impl_vtables {
 219          Some(vtables) => {
 220              let num_impl_type_parameters =
 221                  vtables.len() - num_method_vtables;
 222              vtables.tailn(num_impl_type_parameters).to_owned()
 223          },
 224          None => vec::from_elem(num_method_vtables, @~[])
 225      };
 226  
 227      let param_vtables = @(*trait_vtables_fixed + method_vtables);
 228  
 229      let self_vtables = resolve_param_vtables_under_param_substs(
 230          bcx.tcx(), param_substs, impl_res.self_vtables);
 231  
 232      (param_vtables, self_vtables)
 233  }

 236  pub fn trans_fn_ref_with_vtables(
 237          bcx: @mut Block,       //
 238          def_id: ast::def_id,   // def id of fn
 239          ref_id: ast::NodeId,  // node id of use of fn; may be zero if N/A
 240          type_params: &[ty::t], // values for fn's ty params
 241          vtables: Option<typeck::vtable_res>) // vtables for the call
 242       -> FnData {
 243      //!
 244      //
 245      // Translates a reference to a fn/method item, monomorphizing and
 246      // inlining as it goes.
 247      //
 248      // # Parameters
 249      //
 250      // - `bcx`: the current block where the reference to the fn occurs
 251      // - `def_id`: def id of the fn or method item being referenced
 252      // - `ref_id`: node id of the reference to the fn/method, if applicable.
 253      //   This parameter may be zero; but, if so, the resulting value may not
 254      //   have the right type, so it must be cast before being used.
 255      // - `type_params`: values for each of the fn/method's type parameters
 256      // - `vtables`: values for each bound on each of the type parameters
 257  
 258      let _icx = push_ctxt("trans_fn_ref_with_vtables");
 259      let ccx = bcx.ccx();
 260      let tcx = ccx.tcx;
 261  
 262      debug!("trans_fn_ref_with_vtables(bcx=%s, def_id=%s, ref_id=%?, \
 263              type_params=%s, vtables=%s)",
 264             bcx.to_str(),
 265             def_id.repr(bcx.tcx()),
 266             ref_id,
 267             type_params.repr(bcx.tcx()),
 268             vtables.repr(bcx.tcx()));
 269  
 270      assert!(type_params.iter().all(|t| !ty::type_needs_infer(*t)));
 271  
 272      // Polytype of the function item (may have type params)
 273      let fn_tpt = ty::lookup_item_type(tcx, def_id);
 274  
 275      let substs = ty::substs { regions: ty::ErasedRegions,
 276                                self_ty: None,
 277                                tps: /*bad*/ type_params.to_owned() };
 278  
 279      // We need to do a bunch of special handling for default methods.
 280      // We need to modify the def_id and our substs in order to monomorphize
 281      // the function.
 282      let (is_default, def_id, substs, self_vtables, vtables) =
 283          match ty::provided_source(tcx, def_id) {
 284          None => (false, def_id, substs, None, vtables),
 285          Some(source_id) => {
 286              // There are two relevant substitutions when compiling
 287              // default methods. First, there is the substitution for
 288              // the type parameters of the impl we are using and the
 289              // method we are calling. This substitution is the substs
 290              // argument we already have.
 291              // In order to compile a default method, though, we need
 292              // to consider another substitution: the substitution for
 293              // the type parameters on trait; the impl we are using
 294              // implements the trait at some particular type
 295              // parameters, and we need to substitute for those first.
 296              // So, what we need to do is find this substitution and
 297              // compose it with the one we already have.
 298  
 299              let impl_id = ty::method(tcx, def_id).container_id;
 300              let method = ty::method(tcx, source_id);
 301              let trait_ref = ty::impl_trait_ref(tcx, impl_id)
 302                  .expect("could not find trait_ref for impl with \
 303                           default methods");
 304  
 305              // Compute the first substitution
 306              let first_subst = make_substs_for_receiver_types(
 307                  tcx, impl_id, trait_ref, method);
 308  
 309              // And compose them
 310              let new_substs = first_subst.subst(tcx, &substs);
 311  
 312  
 313              let (param_vtables, self_vtables) =
 314                  resolve_default_method_vtables(bcx, impl_id,
 315                                                 method, &substs, vtables);
 316  
 317              debug!("trans_fn_with_vtables - default method: \
 318                      substs = %s, trait_subst = %s, \
 319                      first_subst = %s, new_subst = %s, \
 320                      vtables = %s, \
 321                      self_vtable = %s, param_vtables = %s",
 322                     substs.repr(tcx), trait_ref.substs.repr(tcx),
 323                     first_subst.repr(tcx), new_substs.repr(tcx),
 324                     vtables.repr(tcx),
 325                     self_vtables.repr(tcx), param_vtables.repr(tcx));
 326  
 327              (true, source_id,
 328               new_substs, Some(self_vtables), Some(param_vtables))
 329          }
 330      };
 331  
 332      // Check whether this fn has an inlined copy and, if so, redirect
 333      // def_id to the local id of the inlined copy.
 334      let def_id = {
 335          if def_id.crate != ast::LOCAL_CRATE {
 336              inline::maybe_instantiate_inline(ccx, def_id)
 337          } else {
 338              def_id
 339          }
 340      };
 341  
 342      // We must monomorphise if the fn has type parameters, is a rust
 343      // intrinsic, or is a default method.  In particular, if we see an
 344      // intrinsic that is inlined from a different crate, we want to reemit the
 345      // intrinsic instead of trying to call it in the other crate.
 346      let must_monomorphise;
 347      if type_params.len() > 0 || is_default {
 348          must_monomorphise = true;
 349      } else if def_id.crate == ast::LOCAL_CRATE {
 350          let map_node = session::expect(
 351              ccx.sess,
 352              ccx.tcx.items.find(&def_id.node),
 353              || fmt!("local item should be in ast map"));
 354  
 355          match *map_node {
 356              ast_map::node_foreign_item(_, abis, _, _) => {
 357                  must_monomorphise = abis.is_intrinsic()
 358              }
 359              _ => {
 360                  must_monomorphise = false;
 361              }
 362          }
 363      } else {
 364          must_monomorphise = false;
 365      }
 366  
 367      // Create a monomorphic verison of generic functions
 368      if must_monomorphise {
 369          // Should be either intra-crate or inlined.
 370          assert_eq!(def_id.crate, ast::LOCAL_CRATE);
 371  
 372          let (val, must_cast) =
 373              monomorphize::monomorphic_fn(ccx, def_id, &substs,
 374                                           vtables, self_vtables,
 375                                           Some(ref_id));
 376          let mut val = val;
 377          if must_cast && ref_id != 0 {
 378              // Monotype of the REFERENCE to the function (type params
 379              // are subst'd)
 380              let ref_ty = common::node_id_type(bcx, ref_id);
 381  
 382              val = PointerCast(
 383                  bcx, val, type_of::type_of_fn_from_ty(ccx, ref_ty).ptr_to());
 384          }
 385          return FnData {llfn: val};
 386      }
 387  
 388      // Find the actual function pointer.
 389      let val = {
 390          if def_id.crate == ast::LOCAL_CRATE {
 391              // Internal reference.
 392              get_item_val(ccx, def_id.node)
 393          } else {
 394              // External reference.
 395              trans_external_path(ccx, def_id, fn_tpt.ty)
 396          }
 397      };
 398  
 399      return FnData {llfn: val};
 400  }

 403  // Translating calls
 404  
 405  pub fn trans_call(in_cx: @mut Block,
 406                    call_ex: @ast::expr,
 407                    f: @ast::expr,
 408                    args: CallArgs,
 409                    id: ast::NodeId,
 410                    dest: expr::Dest)
 411                    -> @mut Block {
 412      let _icx = push_ctxt("trans_call");
 413      trans_call_inner(in_cx,
 414                       call_ex.info(),
 415                       expr_ty(in_cx, f),
 416                       node_id_type(in_cx, id),
 417                       |cx| trans(cx, f),
 418                       args,
 419                       Some(dest),

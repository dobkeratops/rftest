 290  pub fn make_mono_id(ccx: &CrateContext,
 291                      item: ast::DefId,
 292                      substs: &param_substs) -> mono_id {

 295      let substs_iter = substs.self_ty.iter().chain(substs.tys.iter());

 297        Some(vts) => {

 303              (*subst, if !v.is_empty() { Some(@v) } else { None })
 304          }).collect()
 305        }

 307      };

 311          let (a, b) = *x;

 315  }

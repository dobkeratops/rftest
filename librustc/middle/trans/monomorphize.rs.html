  40  pub fn monomorphic_fn(ccx: @mut CrateContext,
  41                        fn_id: ast::def_id,
  42                        real_substs: &ty::substs,
  43                        vtables: Option<typeck::vtable_res>,
  44                        self_vtables: Option<typeck::vtable_param_res>,
  45                        ref_id: Option<ast::NodeId>)
  46      -> (ValueRef, bool)
  47  {
  48      debug!("monomorphic_fn(\
  49              fn_id=%s, \
  50              real_substs=%s, \
  51              vtables=%s, \
  52              self_vtable=%s, \
  53              ref_id=%?)",
  54             fn_id.repr(ccx.tcx),
  55             real_substs.repr(ccx.tcx),
  56             vtables.repr(ccx.tcx),
  57             self_vtables.repr(ccx.tcx),
  58             ref_id);
  59  
  60      assert!(real_substs.tps.iter().all(|t| !ty::type_needs_infer(*t)));
  61      let _icx = push_ctxt("monomorphic_fn");
  62      let mut must_cast = false;
  63  
  64      let do_normalize = |t: &ty::t| {
  65          match normalize_for_monomorphization(ccx.tcx, *t) {
  66            Some(t) => { must_cast = true; t }
  67            None => *t
  68          }
  69      };
  70  
  71      let psubsts = @param_substs {
  72          tys: real_substs.tps.map(|x| do_normalize(x)),
  73          vtables: vtables,
  74          self_ty: real_substs.self_ty.map(|x| do_normalize(x)),
  75          self_vtables: self_vtables
  76      };
  77  
  78      for s in real_substs.tps.iter() { assert!(!ty::type_has_params(*s)); }
  79      for s in psubsts.tys.iter() { assert!(!ty::type_has_params(*s)); }
  80      let param_uses = type_use::type_uses_for(ccx, fn_id, psubsts.tys.len());
  81  
  82  
  83      let hash_id = make_mono_id(ccx, fn_id, &*psubsts, Some(param_uses));
  84      if hash_id.params.iter().any(
  85                  |p| match *p { mono_precise(_, _) => false, _ => true }) {
  86          must_cast = true;
  87      }
  88  
  89      debug!("monomorphic_fn(\
  90              fn_id=%s, \
  91              psubsts=%s, \
  92              hash_id=%?)",
  93             fn_id.repr(ccx.tcx),
  94             psubsts.repr(ccx.tcx),
  95             hash_id);
  96  
  97      match ccx.monomorphized.find(&hash_id) {
  98        Some(&val) => {
  99          debug!("leaving monomorphic fn %s",
 100                 ty::item_path_str(ccx.tcx, fn_id));
 101          return (val, must_cast);
 102        }
 103        None => ()
 104      }
 105  
 106      let tpt = ty::lookup_item_type(ccx.tcx, fn_id);
 107      let llitem_ty = tpt.ty;
 108  
 109      // We need to do special handling of the substitutions if we are
 110      // calling a static provided method. This is sort of unfortunate.
 111      let mut is_static_provided = None;
 112  
 113      let map_node = session::expect(
 114          ccx.sess,
 115          ccx.tcx.items.find_copy(&fn_id.node),
 116          || fmt!("While monomorphizing %?, couldn't find it in the item map \
 117                   (may have attempted to monomorphize an item \
 118                   defined in a different crate?)", fn_id));
 119      // Get the path so that we can create a symbol
 120      let (pt, name, span) = match map_node {
 121        ast_map::node_item(i, pt) => (pt, i.ident, i.span),
 122        ast_map::node_variant(ref v, enm, pt) => (pt, (*v).node.name, enm.span),
 123        ast_map::node_method(m, _, pt) => (pt, m.ident, m.span),
 124        ast_map::node_foreign_item(i, abis, _, pt) if abis.is_intrinsic()
 125        => (pt, i.ident, i.span),
 126        ast_map::node_foreign_item(*) => {
 127          // Foreign externs don't have to be monomorphized.
 128          return (get_item_val(ccx, fn_id.node), true);
 129        }
 130        ast_map::node_trait_method(@ast::provided(m), _, pt) => {
 131          // If this is a static provided method, indicate that
 132          // and stash the number of params on the method.
 133          if m.explicit_self.node == ast::sty_static {
 134              is_static_provided = Some(m.generics.ty_params.len());
 135          }
 136  
 137          (pt, m.ident, m.span)
 138        }
 139        ast_map::node_trait_method(@ast::required(_), _, _) => {
 140          ccx.tcx.sess.bug("Can't monomorphize a required trait method")
 141        }
 142        ast_map::node_expr(*) => {
 143          ccx.tcx.sess.bug("Can't monomorphize an expr")
 144        }
 145        ast_map::node_stmt(*) => {
 146          ccx.tcx.sess.bug("Can't monomorphize a stmt")
 147        }
 148        ast_map::node_arg(*) => ccx.tcx.sess.bug("Can't monomorphize an arg"),
 149        ast_map::node_block(*) => {
 150            ccx.tcx.sess.bug("Can't monomorphize a block")
 151        }
 152        ast_map::node_local(*) => {
 153            ccx.tcx.sess.bug("Can't monomorphize a local")
 154        }
 155        ast_map::node_callee_scope(*) => {
 156            ccx.tcx.sess.bug("Can't monomorphize a callee-scope")
 157        }
 158        ast_map::node_struct_ctor(_, i, pt) => (pt, i.ident, i.span)
 159      };
 160  
 161      debug!("monomorphic_fn about to subst into %s", llitem_ty.repr(ccx.tcx));
 162      let mono_ty = match is_static_provided {
 163          None => ty::subst_tps(ccx.tcx, psubsts.tys,
 164                                psubsts.self_ty, llitem_ty),
 165          Some(num_method_ty_params) => {
 166              // Static default methods are a little unfortunate, in
 167              // that the "internal" and "external" type of them differ.
 168              // Internally, the method body can refer to Self, but the
 169              // externally visable type of the method has a type param
 170              // inserted in between the trait type params and the
 171              // method type params. The substs that we are given are
 172              // the proper substs *internally* to the method body, so
 173              // we have to use those when compiling it.
 174              //
 175              // In order to get the proper substitution to use on the
 176              // type of the method, we pull apart the substitution and
 177              // stick a substitution for the self type in.
 178              // This is a bit unfortunate.
 179  
 180              let idx = psubsts.tys.len() - num_method_ty_params;
 181              let substs =
 182                  (psubsts.tys.slice(0, idx) +
 183                   &[psubsts.self_ty.unwrap()] +
 184                   psubsts.tys.tailn(idx));
 185              debug!("static default: changed substitution to %s",
 186                     substs.repr(ccx.tcx));
 187  
 188              ty::subst_tps(ccx.tcx, substs, None, llitem_ty)
 189          }
 190      };
 191      let llfty = type_of_fn_from_ty(ccx, mono_ty);
 192  
 193      ccx.stats.n_monos += 1;
 194  
 195      let depth = match ccx.monomorphizing.find(&fn_id) {
 196          Some(&d) => d, None => 0
 197      };
 198      // Random cut-off -- code that needs to instantiate the same function
 199      // recursively more than thirty times can probably safely be assumed to be
 200      // causing an infinite expansion.
 201      if depth > 30 {
 202          ccx.sess.span_fatal(
 203              span, "overly deep expansion of inlined function");
 204      }
 205      ccx.monomorphizing.insert(fn_id, depth + 1);
 206  
 207      let elt = path_name(gensym_name(ccx.sess.str_of(name)));
 208      let mut pt = (*pt).clone();
 209      pt.push(elt);
 210      let s = mangle_exported_name(ccx, pt.clone(), mono_ty);
 211      debug!("monomorphize_fn mangled to %s", s);
 212  
 213      let mk_lldecl = || {
 214          let lldecl = decl_internal_cdecl_fn(ccx.llmod, s, llfty);
 215          ccx.monomorphized.insert(hash_id, lldecl);
 216          lldecl
 217      };
 218  
 219      let lldecl = match map_node {
 220        ast_map::node_item(i@@ast::item {
 221                  node: ast::item_fn(ref decl, _, _, _, ref body),
 222                  _
 223              }, _) => {
 224          let d = mk_lldecl();
 225          set_inline_hint_if_appr(i.attrs, d);
 226          trans_fn(ccx,
 227                   pt,
 228                   decl,
 229                   body,
 230                   d,
 231                   no_self,
 232                   Some(psubsts),
 233                   fn_id.node,
 234                   []);
 235          d
 236        }
 237        ast_map::node_item(*) => {
 238            ccx.tcx.sess.bug("Can't monomorphize this kind of item")
 239        }
 240        ast_map::node_foreign_item(i, _, _, _) => {
 241            let d = mk_lldecl();
 242            foreign::trans_intrinsic(ccx, d, i, pt, psubsts, i.attrs,
 243                                  ref_id);
 244            d
 245        }
 246        ast_map::node_variant(ref v, enum_item, _) => {
 247          let tvs = ty::enum_variants(ccx.tcx, local_def(enum_item.id));
 248          let this_tv = *tvs.iter().find(|tv| { tv.id.node == fn_id.node}).unwrap();
 249          let d = mk_lldecl();
 250          set_inline_hint(d);
 251          match v.node.kind {
 252              ast::tuple_variant_kind(ref args) => {
 253                  trans_enum_variant(ccx,
 254                                     enum_item.id,
 255                                     v,
 256                                     (*args).clone(),
 257                                     this_tv.disr_val,
 258                                     Some(psubsts),
 259                                     d);
 260              }
 261              ast::struct_variant_kind(_) =>
 262                  ccx.tcx.sess.bug("can't monomorphize struct variants"),
 263          }
 264          d
 265        }
 266        ast_map::node_method(mth, _, _) => {
 267          // XXX: What should the self type be here?
 268          let d = mk_lldecl();
 269          set_inline_hint_if_appr(mth.attrs.clone(), d);
 270          meth::trans_method(ccx, pt, mth, Some(psubsts), d);
 271          d
 272        }
 273        ast_map::node_trait_method(@ast::provided(mth), _, pt) => {
 274          let d = mk_lldecl();
 275          set_inline_hint_if_appr(mth.attrs.clone(), d);
 276          meth::trans_method(ccx, (*pt).clone(), mth, Some(psubsts), d);
 277          d
 278        }
 279        ast_map::node_struct_ctor(struct_def, _, _) => {
 280          let d = mk_lldecl();
 281          set_inline_hint(d);
 282          base::trans_tuple_struct(ccx,
 283                                   struct_def.fields,
 284                                   struct_def.ctor_id.expect("ast-mapped tuple struct \
 285                                                              didn't have a ctor id"),
 286                                   Some(psubsts),
 287                                   d);
 288          d
 289        }
 290  
 291        // Ugh -- but this ensures any new variants won't be forgotten
 292        ast_map::node_expr(*) |
 293        ast_map::node_stmt(*) |
 294        ast_map::node_trait_method(*) |
 295        ast_map::node_arg(*) |
 296        ast_map::node_block(*) |
 297        ast_map::node_callee_scope(*) |
 298        ast_map::node_local(*) => {
 299          ccx.tcx.sess.bug(fmt!("Can't monomorphize a %?", map_node))
 300        }
 301      };
 302      ccx.monomorphizing.insert(fn_id, depth);
 303  
 304      debug!("leaving monomorphic fn %s", ty::item_path_str(ccx.tcx, fn_id));
 305      (lldecl, must_cast)
 306  }

 308  pub fn normalize_for_monomorphization(tcx: ty::ctxt,
 309                                        ty: ty::t) -> Option<ty::t> {
 310      // FIXME[mono] could do this recursively. is that worthwhile? (#2529)
 311      return match ty::get(ty).sty {
 312          ty::ty_box(*) => {
 313              Some(ty::mk_opaque_box(tcx))
 314          }
 315          ty::ty_bare_fn(_) => {
 316              Some(ty::mk_bare_fn(
 317                  tcx,
 318                  ty::BareFnTy {
 319                      purity: ast::impure_fn,
 320                      abis: AbiSet::Rust(),
 321                      sig: FnSig {bound_lifetime_names: opt_vec::Empty,
 322                                  inputs: ~[],
 323                                  output: ty::mk_nil()}}))
 324          }
 325          ty::ty_closure(ref fty) => {
 326              Some(normalized_closure_ty(tcx, fty.sigil))
 327          }
 328          ty::ty_trait(_, _, ref store, _, _) => {
 329              let sigil = match *store {
 330                  ty::UniqTraitStore => ast::OwnedSigil,
 331                  ty::BoxTraitStore => ast::ManagedSigil,
 332                  ty::RegionTraitStore(_) => ast::BorrowedSigil,
 333              };
 334  
 335              // Traits have the same runtime representation as closures.
 336              Some(normalized_closure_ty(tcx, sigil))
 337          }
 338          ty::ty_ptr(_) => {
 339              Some(ty::mk_uint())
 340          }
 341          _ => {
 342              None
 343          }
 344      };
 345  
 346      fn normalized_closure_ty(tcx: ty::ctxt,
 347                               sigil: ast::Sigil) -> ty::t
 348      {
 349          ty::mk_closure(
 350              tcx,
 351              ty::ClosureTy {
 352                  purity: ast::impure_fn,
 353                  sigil: sigil,
 354                  onceness: ast::Many,
 355                  region: ty::re_static,
 356                  bounds: ty::EmptyBuiltinBounds(),
 357                  sig: ty::FnSig {bound_lifetime_names: opt_vec::Empty,
 358                                  inputs: ~[],
 359                                  output: ty::mk_nil()}})
 360      }
 361  }

 363  pub fn make_mono_id(ccx: @mut CrateContext,
 364                      item: ast::def_id,
 365                      substs: &param_substs,
 366                      param_uses: Option<@~[type_use::type_uses]>) -> mono_id {
 367      // FIXME (possibly #5801): Need a lot of type hints to get
 368      // .collect() to work.
 369      let substs_iter = substs.self_ty.iter().chain(substs.tys.iter());
 370      let precise_param_ids: ~[(ty::t, Option<@~[mono_id]>)] = match substs.vtables {
 371        Some(vts) => {
 372          debug!("make_mono_id vtables=%s substs=%s",
 373                 vts.repr(ccx.tcx), substs.tys.repr(ccx.tcx));
 374          let vts_iter = substs.self_vtables.iter().chain(vts.iter());
 375          vts_iter.zip(substs_iter).map(|(vtable, subst)| {
 376              let v = vtable.map(|vt| meth::vtable_id(ccx, vt));
 377              (*subst, if !v.is_empty() { Some(@v) } else { None })
 378          }).collect()
 379        }
 380        None => substs_iter.map(|subst| (*subst, None::<@~[mono_id]>)).collect()
 381      };
 382  
 383  
 384      let param_ids = match param_uses {
 385        Some(ref uses) => {
 386          // param_uses doesn't include a use for the self type.
 387          // We just say it is fully used.
 388          let self_use =
 389              substs.self_ty.map(|_| type_use::use_repr|type_use::use_tydesc);
 390          let uses_iter = self_use.iter().chain(uses.iter());
 391  
 392          precise_param_ids.iter().zip(uses_iter).map(|(id, uses)| {
 393              if ccx.sess.no_monomorphic_collapse() {
 394                  match *id {
 395                      (a, b) => mono_precise(a, b)
 396                  }
 397              } else {
 398                  match *id {
 399                      (a, b@Some(_)) => mono_precise(a, b),
 400                      (subst, None) => {
 401                          if *uses == 0 {
 402                              mono_any
 403                          } else if *uses == type_use::use_repr &&
 404                              !ty::type_needs_drop(ccx.tcx, subst)
 405                          {
 406                              let llty = type_of::type_of(ccx, subst);
 407                              let size = machine::llbitsize_of_real(ccx, llty);
 408                              let align = machine::llalign_of_min(ccx, llty);

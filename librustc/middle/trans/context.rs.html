 236      pub fn get_intrinsic(&self, key: & &'static str) -> ValueRef {
 237          match self.intrinsics.borrow().find_copy(key) {
 238              Some(v) => return v,
 239              _ => {}
 240          }
 241          match declare_intrinsic(self, key) {
 242              Some(v) => return v,
 243              None => fail!()
 244          }
 245      }
 246  }

 248  fn declare_intrinsic(ccx: &CrateContext, key: & &'static str) -> Option<ValueRef> {
 249      macro_rules! ifn (
 250          ($name:expr fn() -> $ret:expr) => (
 251              if *key == $name {
 252                  let f = base::decl_cdecl_fn(ccx.llmod, $name, Type::func([], &$ret), ty::mk_nil());
 253                  ccx.intrinsics.borrow_mut().insert($name, f.clone());
 254                  return Some(f);
 255              }
 256          );
 257          ($name:expr fn($($arg:expr),*) -> $ret:expr) => (
 258              if *key == $name {
 259                  let f = base::decl_cdecl_fn(ccx.llmod, $name,
 260                                    Type::func([$($arg),*], &$ret), ty::mk_nil());
 261                  ccx.intrinsics.borrow_mut().insert($name, f.clone());
 262                  return Some(f);
 263              }
 264          )
 265      )
 266      macro_rules! mk_struct (
 267          ($($field_ty:expr),*) => (Type::struct_(ccx, [$($field_ty),*], false))
 268      )

  35         }

  37         Float => 4,
  38         Double => 8,

  55 fn ty_size(ty: Type) -> uint {
  56     match ty.kind() {
  57         Integer => {
  58             unsafe {
  59                 ((llvm::LLVMGetIntTypeWidth(ty.to_ref()) as uint) + 7) / 8
  60             }
  61         }
  62         Pointer => 4,
  63         Float => 4,
  64         Double => 8,
  65         Struct => {
  66             if ty.is_packed() {

 103 fn classify_arg_ty(ty: Type) -> (LLVMType, Option<Attribute>) {
 104     if is_reg_ty(ty) {
 105         return (LLVMType { cast: false, ty: ty }, None);
 106     }

 110         Type::array(&Type::i32(), ((size + 3) / 4) as u64)

 112         Type::array(&Type::i64(), ((size + 7) / 8) as u64)
 113     };

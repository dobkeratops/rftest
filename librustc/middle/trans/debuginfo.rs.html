   11  /*!
   12  # Debug Info Module
   13  
   14  This module serves the purpose of generating debug symbols. We use LLVM's
   15  [source level debugging](http://llvm.org/docs/SourceLevelDebugging.html) features for generating
   16  the debug information. The general principle is this:
   17  
   18  Given the right metadata in the LLVM IR, the LLVM code generator is able to create DWARF debug
   19  symbols for the given code. The [metadata](http://llvm.org/docs/LangRef.html#metadata-type) is
   20  structured much like DWARF *debugging information entries* (DIE), representing type information
   21  such as datatype layout, function signatures, block layout, variable location and scope information,
   22  etc. It is the purpose of this module to generate correct metadata and insert it into the LLVM IR.
   23  
   24  As the exact format of metadata trees may change between different LLVM versions, we now use LLVM
   25  [DIBuilder](http://llvm.org/docs/doxygen/html/classllvm_1_1DIBuilder.html) to create metadata
   26  where possible. This will hopefully ease the adaption of this module to future LLVM versions.
   27  
   28  The public API of the module is a set of functions that will insert the correct metadata into the
   29  LLVM IR when called with the right parameters. The module is thus driven from an outside client with
   30  functions like `debuginfo::local_var_metadata(bcx: block, local: &ast::local)`.
   31  
   32  Internally the module will try to reuse already created metadata by utilizing a cache. The way to
   33  get a shared metadata node when needed is thus to just call the corresponding function in this
   34  module:
   35  
   36      let file_metadata = file_metadata(crate_context, path);
   37  
   38  The function will take care of probing the cache for an existing node for that exact file path.
   39  
   40  All private state used by the module is stored within a DebugContext struct, which in turn is
   41  contained in the CrateContext.
   42  
   43  
   44  This file consists of three conceptual sections:
   45  1. The public interface of the module
   46  2. Module-internal metadata creation functions
   47  3. Minor utility functions
   48  
   49  */

  173  /// Creates debug information for the given function argument.
  174  ///
  175  /// Adds the created metadata nodes directly to the crate's IR.
  176  pub fn create_argument_metadata(bcx: @mut Block,
  177                                  arg: &ast::arg) {
  178      let fcx = bcx.fcx;
  179      let cx = fcx.ccx;
  180  
  181      let pattern = arg.pat;
  182      let filename = span_start(cx, pattern.span).file.name;
  183  
  184      if fcx.id == -1 ||
  185         fcx.span.is_none() ||
  186         "<intrinsic>" == filename {
  187          return;
  188      }
  189  
  190      let def_map = cx.tcx.def_map;
  191      let file_metadata = file_metadata(cx, filename);
  192      let scope = create_function_metadata(fcx);
  193  
  194      do pat_util::pat_bindings(def_map, pattern) |_, node_id, span, path_ref| {
  195  
  196          let ty = node_id_type(bcx, node_id);
  197          let type_metadata = type_metadata(cx, ty, codemap::dummy_sp());
  198          let loc = span_start(cx, span);
  199          let ident = ast_util::path_to_ident(path_ref);
  200          let name: &str = cx.sess.str_of(ident);
  201          debug!("create_argument_metadata: %s", name);
  202  
  203          let argument_index = {
  204              let counter = &mut fcx.debug_context.get_mut_ref().argument_counter;
  205              let argument_index = *counter;
  206              *counter += 1;
  207              argument_index as c_uint
  208          };
  209  
  210          let arg_metadata = do name.to_c_str().with_ref |name| {
  211              unsafe {
  212                  llvm::LLVMDIBuilderCreateLocalVariable(
  213                      DIB(cx),
  214                      DW_TAG_arg_variable,
  215                      scope,
  216                      name,
  217                      file_metadata,
  218                      loc.line as c_uint,
  219                      type_metadata,
  220                      false,
  221                      0,
  222                      argument_index)
  223              }
  224          };
  225  
  226          let llptr = match bcx.fcx.llargs.find_copy(&node_id) {
  227              Some(v) => v,
  228              None => {
  229                  bcx.tcx().sess.span_bug(span, fmt!("No entry in llargs table for %?", node_id));
  230              }
  231          };
  232  
  233          set_debug_location(cx, scope, loc.line, loc.col.to_uint());
  234          unsafe {
  235              let instr = llvm::LLVMDIBuilderInsertDeclareAtEnd(
  236                  DIB(cx),
  237                  llptr,
  238                  arg_metadata,
  239                  bcx.llbb);
  240  
  241              llvm::LLVMSetInstDebugLocation(trans::build::B(bcx).llbuilder, instr);
  242          }
  243      }
  244  }

  246  /// Sets the current debug location at the beginning of the span

  248  /// Maps to a call to llvm::LLVMSetCurrentDebugLocation(...)
  249  pub fn update_source_pos(fcx: &FunctionContext,

  256      }
  257  
  258      debug!("update_source_pos: %s", cx.sess.codemap.span_to_str(span));

  264  }

  266  /// Creates debug information for the given function.
  267  ///
  268  /// Adds the created metadata nodes directly to the crate's IR.
  269  /// The return value should be ignored if called from outside of the debuginfo module.
  270  pub fn create_function_metadata(fcx: &mut FunctionContext) -> DISubprogram {
  271      let cx = fcx.ccx;
  272  
  273      let fnitem = cx.tcx.items.get_copy(&fcx.id);
  274      let (ident, fn_decl, id) = match fnitem {
  275          ast_map::node_item(ref item, _) => {
  276              match item.node {
  277                  ast::item_fn(ref fn_decl, _, _, _, _) => {
  278                      (item.ident, fn_decl, item.id)
  279                  }
  280                  _ => fcx.ccx.sess.span_bug(item.span,
  281                                             "create_function_metadata: item bound to non-function")
  282              }
  283          }
  284          ast_map::node_method(
  285              @ast::method {
  286                  decl: ref fn_decl,
  287                  id: id,
  288                  ident: ident,
  289                  _
  290              },
  291              _,
  292              _) => {
  293              (ident, fn_decl, id)
  294          }
  295          ast_map::node_expr(ref expr) => {
  296              match expr.node {
  297                  ast::expr_fn_block(ref fn_decl, _) => {
  298                      let name = gensym_name("fn");
  299                      (name, fn_decl, expr.id)
  300                  }
  301                  _ => fcx.ccx.sess.span_bug(expr.span,
  302                          "create_function_metadata: expected an expr_fn_block here")
  303              }
  304          }
  305          ast_map::node_trait_method(
  306              @ast::provided(
  307                  @ast::method {
  308                      decl: ref fn_decl,
  309                      id: id,
  310                      ident: ident,
  311                      _
  312                  }),
  313              _,
  314              _) => {
  315              (ident, fn_decl, id)
  316          }
  317          _ => fcx.ccx.sess.bug(fmt!("create_function_metadata: unexpected sort of node: %?", fnitem))
  318      };
  319  
  320      match dbg_cx(cx).created_functions.find_copy(&id) {
  321          Some(fn_metadata) => return fn_metadata,
  322          None => ()
  323      }
  324  
  325      let span = match fcx.span {
  326          Some(value) => value,
  327          None => codemap::dummy_sp()
  328      };
  329  
  330      debug!("create_function_metadata: %s, %s",
  331             cx.sess.str_of(ident),
  332             cx.sess.codemap.span_to_str(span));
  333  
  334      let loc = span_start(cx, span);
  335      let file_metadata = file_metadata(cx, loc.file.name);
  336  
  337      let return_type_metadata = if cx.sess.opts.extra_debuginfo {
  338          match fn_decl.output.node {
  339            ast::ty_nil => ptr::null(),
  340            _ => type_metadata(cx, ty::node_id_to_type(cx.tcx, id), fn_decl.output.span)
  341          }
  342      } else {
  343          ptr::null()
  344      };
  345  
  346      let fn_ty = unsafe {
  347          llvm::LLVMDIBuilderCreateSubroutineType(
  348              DIB(cx),
  349              file_metadata,
  350              create_DIArray(DIB(cx), [return_type_metadata]))
  351      };
  352  
  353      let fn_metadata =
  354          do cx.sess.str_of(ident).to_c_str().with_ref |name| {
  355          do cx.sess.str_of(ident).to_c_str().with_ref |linkage| {
  356              unsafe {
  357                  llvm::LLVMDIBuilderCreateFunction(
  358                      DIB(cx),
  359                      file_metadata,
  360                      name,
  361                      linkage,
  362                      file_metadata,
  363                      loc.line as c_uint,
  364                      fn_ty,
  365                      false,
  366                      true,
  367                      loc.line as c_uint,
  368                      FlagPrototyped as c_uint,
  369                      cx.sess.opts.optimize != session::No,
  370                      fcx.llfn,
  371                      ptr::null(),
  372                      ptr::null())
  373              }
  374          }};
  375  
  376      assert!(fcx.debug_context.is_none());
  377  
  378      let mut fn_debug_context = ~FunctionDebugContext::new();
  379      let entry_block_id = fcx.entry_bcx.get_ref().node_info.get_ref().id;
  380      let entry_block = cx.tcx.items.get(&entry_block_id);
  381  
  382      match *entry_block {
  383          ast_map::node_block(ref block) => {
  384              let scope_map = &mut fn_debug_context.scope_map;
  385              let arg_pats = do fn_decl.inputs.map |arg_ref| { arg_ref.pat };
  386  
  387              populate_scope_map(cx, arg_pats, block, fn_metadata, scope_map);
  388          }
  389          _ => cx.sess.span_bug(span,
  390                  fmt!("debuginfo::create_function_metadata() - \
  391                       FunctionContext::entry_bcx::node_info points to wrong type of ast_map entry. \
  392                       Expected: ast_map::node_block, actual: %?", *entry_block))
  393      }
  394  
  395      fcx.debug_context = Some(fn_debug_context);
  396  
  397      dbg_cx(cx).created_functions.insert(id, fn_metadata);
  398      return fn_metadata;
  399  }

  419  
  420      let work_dir = cx.sess.working_dir.to_str();

  422

  424      do work_dir.to_c_str().with_ref |work_dir| {
  425      do producer.to_c_str().with_ref |producer| {
  426      do "".to_c_str().with_ref |flags| {
  427      do "".to_c_str().with_ref |split_name| {
  428          unsafe {
  429              llvm::LLVMDIBuilderCreateCompileUnit(dcx.builder,
  430                  DW_LANG_RUST as c_uint, crate_name, work_dir, producer,
  431                  cx.sess.opts.optimize != session::No,
  432                  flags, 0, split_name);
  433          }
  434      }}}}};
  435  }

  654  fn enum_metadata(cx: &mut CrateContext,
  655                   enum_type: ty::t,
  656                   enum_def_id: ast::def_id,
  657                   span: span)
  658                -> DIType {
  659  
  660      let enum_name = ty_to_str(cx.tcx, enum_type);
  661  
  662      // For empty enums there is an early exit. Just describe it as an empty struct with the
  663      // appropriate type name
  664      if ty::type_is_empty(cx.tcx, enum_type) {
  665          return composite_type_metadata(cx, Type::nil(), enum_name, [], [], [], span);
  666      }
  667  
  668      // Prepare some data (llvm type, size, align, ...) about the discriminant. This data will be
  669      // needed in all of the following cases.
  670      let discriminant_llvm_type = Type::enum_discrim(cx);
  671      let (discriminant_size, discriminant_align) = size_and_align_of(cx, discriminant_llvm_type);
  672  
  673      assert!(Type::enum_discrim(cx) == cx.int_type);
  674      let discriminant_type_metadata = type_metadata(cx, ty::mk_int(), span);
  675  
  676      let variants: &[@ty::VariantInfo] = *ty::enum_variants(cx.tcx, enum_def_id);
  677  
  678      let enumerators_metadata: ~[DIDescriptor] = variants
  679          .iter()
  680          .map(|v| {
  681              let name: &str = cx.sess.str_of(v.name);
  682              let discriminant_value = v.disr_val as c_ulonglong;

  685                  unsafe {
  686                      llvm::LLVMDIBuilderCreateEnumerator(
  687                          DIB(cx),
  688                          name,
  689                          discriminant_value)
  690                  }
  691              }
  692          })
  693          .collect();
  694  
  695      let loc = span_start(cx, span);
  696      let file_metadata = file_metadata(cx, loc.file.name);
  697  
  698      let discriminant_type_metadata = do enum_name.to_c_str().with_ref |enum_name| {
  699          unsafe {
  700              llvm::LLVMDIBuilderCreateEnumerationType(
  701                  DIB(cx),
  702                  file_metadata,
  703                  enum_name,
  704                  file_metadata,
  705                  loc.line as c_uint,
  706                  bytes_to_bits(discriminant_size),
  707                  bytes_to_bits(discriminant_align),
  708                  create_DIArray(DIB(cx), enumerators_metadata),
  709                  discriminant_type_metadata)
  710          }
  711      };
  712  
  713      let type_rep = adt::represent_type(cx, enum_type);
  714  
  715      match *type_rep {
  716          adt::CEnum(*) => {
  717              return discriminant_type_metadata;
  718          }
  719          adt::Univariant(ref struct_def, _) => {
  720              assert!(variants.len() == 1);
  721              return adt_struct_metadata(cx, struct_def, variants[0], None, span);
  722          }
  723          adt::General(ref struct_defs) => {
  724              let variants_member_metadata: ~[DIDescriptor] = do struct_defs
  725                  .iter()
  726                  .enumerate()
  727                  .map |(i, struct_def)| {
  728                      let variant_type_metadata = adt_struct_metadata(
  729                          cx,
  730                          struct_def,
  731                          variants[i],
  732                          Some(discriminant_type_metadata),
  733                          span);
  734  
  735                      do "".to_c_str().with_ref |name| {
  736                          unsafe {
  737                              llvm::LLVMDIBuilderCreateMemberType(
  738                                  DIB(cx),
  739                                  file_metadata,
  740                                  name,
  741                                  file_metadata,
  742                                  loc.line as c_uint,
  743                                  bytes_to_bits(struct_def.size as uint),
  744                                  bytes_to_bits(struct_def.align as uint),
  745                                  bytes_to_bits(0),
  746                                  0,
  747                                  variant_type_metadata)
  748                          }
  749                      }
  750              }.collect();
  751  
  752              let enum_llvm_type = type_of::type_of(cx, enum_type);
  753              let (enum_type_size, enum_type_align) = size_and_align_of(cx, enum_llvm_type);
  754  
  755              return do enum_name.to_c_str().with_ref |enum_name| {

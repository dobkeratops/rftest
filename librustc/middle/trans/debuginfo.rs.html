   11  /*!
   12  # Debug Info Module
   13  
   14  This module serves the purpose of generating debug symbols. We use LLVM's
   15  [source level debugging](http://llvm.org/docs/SourceLevelDebugging.html) features for generating
   16  the debug information. The general principle is this:
   17  
   18  Given the right metadata in the LLVM IR, the LLVM code generator is able to create DWARF debug
   19  symbols for the given code. The [metadata](http://llvm.org/docs/LangRef.html#metadata-type) is
   20  structured much like DWARF *debugging information entries* (DIE), representing type information
   21  such as datatype layout, function signatures, block layout, variable location and scope information,
   22  etc. It is the purpose of this module to generate correct metadata and insert it into the LLVM IR.
   23  
   24  As the exact format of metadata trees may change between different LLVM versions, we now use LLVM
   25  [DIBuilder](http://llvm.org/docs/doxygen/html/classllvm_1_1DIBuilder.html) to create metadata
   26  where possible. This will hopefully ease the adaption of this module to future LLVM versions.
   27  
   28  The public API of the module is a set of functions that will insert the correct metadata into the
   29  LLVM IR when called with the right parameters. The module is thus driven from an outside client with
   30  functions like `debuginfo::local_var_metadata(bcx: block, local: &ast::local)`.
   31  
   32  Internally the module will try to reuse already created metadata by utilizing a cache. The way to
   33  get a shared metadata node when needed is thus to just call the corresponding function in this
   34  module:
   35  
   36      let file_metadata = file_metadata(crate_context, path);
   37  
   38  The function will take care of probing the cache for an existing node for that exact file path.
   39  
   40  All private state used by the module is stored within a DebugContext struct, which in turn is
   41  contained in the CrateContext.
   42  
   43  
   44  This file consists of three conceptual sections:
   45  1. The public interface of the module
   46  2. Module-internal metadata creation functions
   47  3. Minor utility functions
   48  
   49  */

  138  /// Creates debug information for the given local variable.
  139  ///
  140  /// Adds the created metadata nodes directly to the crate's IR.
  141  pub fn create_local_var_metadata(bcx: @mut Block, local: &ast::Local) {
  142      let cx = bcx.ccx();
  143      let def_map = cx.tcx.def_map;
  144      let pattern = local.pat;
  145  
  146      let scope = match bcx.parent {
  147          None => create_function_metadata(bcx.fcx),
  148          Some(_) => lexical_block_metadata(bcx)
  149      };
  150  
  151      let filename = span_start(cx, local.span).file.name;
  152      let file_metadata = file_metadata(cx, filename);
  153  
  154      do pat_util::pat_bindings(def_map, pattern) |_, node_id, span, path_ref| {
  155  
  156          let ident = ast_util::path_to_ident(path_ref);
  157          let name: &str = cx.sess.str_of(ident);
  158          debug!("create_local_var_metadata: %s", name);
  159          let loc = span_start(cx, span);
  160          let ty = node_id_type(bcx, node_id);
  161          let type_metadata = type_metadata(cx, ty, span);
  162  
  163          let var_metadata = do name.to_c_str().with_ref |name| {
  164              unsafe {
  165                  llvm::LLVMDIBuilderCreateLocalVariable(
  166                      DIB(cx),
  167                      DW_TAG_auto_variable,
  168                      scope,
  169                      name,
  170                      file_metadata,
  171                      loc.line as c_uint,
  172                      type_metadata,
  173                      false,
  174                      0,
  175                      0)
  176              }
  177          };
  178  
  179          let llptr = match bcx.fcx.lllocals.find_copy(&node_id) {
  180              Some(v) => v,
  181              None => {
  182                  bcx.tcx().sess.span_bug(span, fmt!("No entry in lllocals table for %?", node_id));
  183              }
  184          };
  185  
  186          set_debug_location(cx, lexical_block_metadata(bcx), loc.line, loc.col.to_uint());
  187          unsafe {
  188              let instr = llvm::LLVMDIBuilderInsertDeclareAtEnd(
  189                  DIB(cx),
  190                  llptr,
  191                  var_metadata,
  192                  bcx.llbb);
  193  
  194              llvm::LLVMSetInstDebugLocation(trans::build::B(bcx).llbuilder, instr);
  195          }
  196      }
  197  }

  199  /// Creates debug information for the given function argument.
  200  ///
  201  /// Adds the created metadata nodes directly to the crate's IR.
  202  pub fn create_argument_metadata(bcx: @mut Block,
  203                                  arg: &ast::arg) {
  204      let fcx = bcx.fcx;
  205      let cx = fcx.ccx;
  206  
  207      let pattern = arg.pat;
  208      let filename = span_start(cx, pattern.span).file.name;
  209  
  210      if fcx.id == -1 ||
  211         fcx.span.is_none() ||
  212         "<intrinsic>" == filename {
  213          return;
  214      }
  215  
  216      // Limited the scope within which `debug_context` is live,
  217      // otherwise => borrowing errors
  218      {
  219          let debug_context = dbg_cx(cx);
  220  
  221          // If this is a new function, reset the counter. llvm::DIBuilder
  222          // wants arguments to be indexed starting from 1.
  223          if fcx.id != debug_context.last_function_context_id {
  224                      debug_context.argument_counter = 1;
  225          }
  226          // Keep track of the function we are in
  227          debug_context.last_function_context_id = fcx.id;
  228      }
  229  
  230      let def_map = cx.tcx.def_map;
  231      let file_metadata = file_metadata(cx, filename);
  232      let scope = create_function_metadata(fcx);
  233  
  234      do pat_util::pat_bindings(def_map, pattern) |_, node_id, span, path_ref| {
  235  
  236          let ty = node_id_type(bcx, node_id);
  237          let type_metadata = type_metadata(cx, ty, codemap::dummy_sp());
  238          let loc = span_start(cx, span);
  239          let ident = ast_util::path_to_ident(path_ref);
  240          let name: &str = cx.sess.str_of(ident);
  241          debug!("create_argument_metadata: %s", name);
  242  
  243          let argument_index = {
  244              let debug_context = dbg_cx(cx);
  245              let argument_index = debug_context.argument_counter;
  246              debug_context.argument_counter += 1;
  247              argument_index as c_uint
  248          };
  249  
  250          let arg_metadata = do name.to_c_str().with_ref |name| {
  251              unsafe {
  252                  llvm::LLVMDIBuilderCreateLocalVariable(
  253                      DIB(cx),
  254                      DW_TAG_arg_variable,
  255                      scope,
  256                      name,
  257                      file_metadata,
  258                      loc.line as c_uint,
  259                      type_metadata,
  260                      false,
  261                      0,
  262                      argument_index)
  263              }
  264          };
  265  
  266          let llptr = match bcx.fcx.llargs.find_copy(&node_id) {
  267              Some(v) => v,
  268              None => {
  269                  bcx.tcx().sess.span_bug(span, fmt!("No entry in llargs table for %?", node_id));
  270              }
  271          };
  272  
  273          set_debug_location(cx, lexical_block_metadata(bcx), loc.line, loc.col.to_uint());
  274          unsafe {
  275              let instr = llvm::LLVMDIBuilderInsertDeclareAtEnd(
  276                  DIB(cx),
  277                  llptr,
  278                  arg_metadata,
  279                  bcx.llbb);
  280  
  281              llvm::LLVMSetInstDebugLocation(trans::build::B(bcx).llbuilder, instr);
  282          }
  283      }
  284  }

  298  /// Creates debug information for the given function.
  299  ///
  300  /// Adds the created metadata nodes directly to the crate's IR.
  301  /// The return value should be ignored if called from outside of the debuginfo module.
  302  pub fn create_function_metadata(fcx: &FunctionContext) -> DISubprogram {
  303      let cx = fcx.ccx;
  304  
  305      let fnitem = cx.tcx.items.get_copy(&fcx.id);
  306      let (ident, ret_ty, id) = match fnitem {
  307          ast_map::node_item(ref item, _) => {
  308              match item.node {
  309                  ast::item_fn(ast::fn_decl { output: ref ty, _}, _, _, _, _) => {
  310                      (item.ident, ty, item.id)
  311                  }
  312                  _ => fcx.ccx.sess.span_bug(item.span,
  313                                             "create_function_metadata: item bound to non-function")
  314              }
  315          }
  316          ast_map::node_method(
  317              @ast::method {
  318                  decl: ast::fn_decl { output: ref ty, _ },
  319                  id: id,
  320                  ident: ident,
  321                  _
  322              },
  323              _,
  324              _) => {
  325              (ident, ty, id)
  326          }
  327          ast_map::node_expr(ref expr) => {
  328              match expr.node {
  329                  ast::expr_fn_block(ref decl, _) => {
  330                      let name = gensym_name("fn");
  331                      (name, &decl.output, expr.id)
  332                  }
  333                  _ => fcx.ccx.sess.span_bug(expr.span,
  334                          "create_function_metadata: expected an expr_fn_block here")
  335              }
  336          }
  337          ast_map::node_trait_method(
  338              @ast::provided(
  339                  @ast::method {
  340                      decl: ast::fn_decl { output: ref ty, _ },
  341                      id: id,
  342                      ident: ident,
  343                      _
  344                  }),
  345              _,
  346              _) => {
  347              (ident, ty, id)
  348          }
  349          _ => fcx.ccx.sess.bug(fmt!("create_function_metadata: unexpected sort of node: %?", fnitem))
  350      };
  351  
  352      match dbg_cx(cx).created_functions.find(&id) {
  353          Some(fn_metadata) => return *fn_metadata,
  354          None => ()
  355      }
  356  
  357      let span = match fcx.span {
  358          Some(value) => value,
  359          None => codemap::dummy_sp()
  360      };
  361  
  362      debug!("create_function_metadata: %s, %s",
  363             cx.sess.str_of(ident),
  364             cx.sess.codemap.span_to_str(span));
  365  
  366      let loc = span_start(cx, span);
  367      let file_metadata = file_metadata(cx, loc.file.name);
  368  
  369      let return_type_metadata = if cx.sess.opts.extra_debuginfo {
  370          match ret_ty.node {
  371            ast::ty_nil => ptr::null(),
  372            _ => type_metadata(cx, ty::node_id_to_type(cx.tcx, id), ret_ty.span)
  373          }
  374      } else {
  375          ptr::null()
  376      };
  377  
  378      let fn_ty = unsafe {
  379          llvm::LLVMDIBuilderCreateSubroutineType(
  380              DIB(cx),
  381              file_metadata,
  382              create_DIArray(DIB(cx), [return_type_metadata]))
  383      };
  384  
  385      let fn_metadata =
  386          do cx.sess.str_of(ident).to_c_str().with_ref |name| {
  387          do cx.sess.str_of(ident).to_c_str().with_ref |linkage| {
  388              unsafe {
  389                  llvm::LLVMDIBuilderCreateFunction(
  390                      DIB(cx),
  391                      file_metadata,
  392                      name,
  393                      linkage,
  394                      file_metadata,
  395                      loc.line as c_uint,
  396                      fn_ty,
  397                      false,
  398                      true,
  399                      loc.line as c_uint,
  400                      FlagPrototyped as c_uint,
  401                      cx.sess.opts.optimize != session::No,
  402                      fcx.llfn,
  403                      ptr::null(),
  404                      ptr::null())
  405              }
  406          }};
  407  
  408      dbg_cx(cx).created_functions.insert(id, fn_metadata);
  409      return fn_metadata;
  410  }

  425  fn compile_unit_metadata(cx: @mut CrateContext) {
  426      let dcx = dbg_cx(cx);
  427      let crate_name: &str = dcx.crate_file;
  428  
  429      debug!("compile_unit_metadata: %?", crate_name);
  430  
  431      let work_dir = cx.sess.working_dir.to_str();
  432      let producer = fmt!("rustc version %s", env!("CFG_VERSION"));
  433

  476  /// Get or create the lexical block metadata node for the given LLVM basic block.
  477  fn lexical_block_metadata(bcx: @mut Block) -> DILexicalBlock {
  478      let cx = bcx.ccx();
  479      let mut bcx = bcx;
  480  
  481      // Search up the tree of basic blocks until we find one that knows the containing lexical block.
  482      while bcx.node_info.is_none() {
  483          match bcx.parent {
  484              Some(b) => bcx = b,
  485              None => cx.sess.bug("debuginfo: Could not find lexical block for LLVM basic block.")
  486          }
  487      }
  488  
  489      let span = bcx.node_info.unwrap().span;
  490      let id = bcx.node_info.unwrap().id;
  491  
  492      // Check whether we already have a cache entry for this node id
  493      match dbg_cx(cx).created_blocks.find(&id) {
  494          Some(block) => return *block,
  495          None => ()
  496      }
  497  
  498      debug!("lexical_block_metadata: %s", bcx.sess().codemap.span_to_str(span));
  499  
  500      let parent = match bcx.parent {
  501          None => create_function_metadata(bcx.fcx),
  502          Some(b) => lexical_block_metadata(b)
  503      };
  504  
  505      let loc = span_start(cx, span);
  506      let file_metadata = file_metadata(cx, loc.file.name);
  507  
  508      let lexical_block_metadata = unsafe {
  509          llvm::LLVMDIBuilderCreateLexicalBlock(
  510              DIB(cx),
  511              parent,
  512              file_metadata,
  513              loc.line as c_uint,
  514              loc.col.to_uint() as c_uint)
  515      };
  516  
  517      dbg_cx(cx).created_blocks.insert(id, lexical_block_metadata);
  518  
  519      return lexical_block_metadata;
  520  }

 238     fn fixup(ty: Type, cls: &mut [RegClass]) {
 239         let mut i = 0u;

 244                 i += 1u;

 247                         all_mem(cls);
 248                         return;
 249                     }
 250                     i += 1u;
 251                 }
 252             } else {
 253                 all_mem(cls);
 254                 return
 255             }
 256         } else {
 257             while i < e {

 190      fn classify(ty: Type,
 191                  cls: &mut [RegClass], ix: uint,
 192                  off: uint) {
 193          let t_align = ty_align(ty);

 239      fn fixup(ty: Type, cls: &mut [RegClass]) {
 240          let mut i = 0u;
 241          let ty_kind = ty.kind();

 244              if cls[i].is_sse() {
 245                  i += 1u;
 246                  while i < e {
 247                      if cls[i] != SSEUp {
 248                          all_mem(cls);
 249                          return;
 250                      }
 251                      i += 1u;
 252                  }
 253              } else {
 254                  all_mem(cls);
 255                  return
 256              }
 257          } else {
 258              while i < e {
 259                  if cls[i] == Memory {
 260                      all_mem(cls);
 261                      return;
 262                  }
 263                  if cls[i] == X87Up {
 264                      // for darwin

 266                      all_mem(cls);

 269                  if cls[i] == SSEUp {
 270                      cls[i] = SSEDv;
 271                  } else if cls[i].is_sse() {
 272                      i += 1;
 273                      while i != e && cls[i] == SSEUp { i += 1u; }
 274                  } else if cls[i] == X87 {
 275                      i += 1;
 276                      while i != e && cls[i] == X87Up { i += 1u; }
 277                  } else {
 278                      i += 1;

 280              }

 282      }

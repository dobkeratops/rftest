  36 pub fn trans_block(bcx: @mut Block, b: &ast::Block, dest: expr::Dest) -> @mut Block {
  37     let _icx = push_ctxt("trans_block");
  38     let mut bcx = bcx;
  39     for s in b.stmts.iter() {
  40         debuginfo::update_source_pos(bcx, b.span);
  41         bcx = trans_stmt(bcx, *s);
  42     }
  43     match b.expr {
  44         Some(e) => {
  45             debuginfo::update_source_pos(bcx, e.span);
  46             bcx = expr::trans_into(bcx, e, dest);
  47         }
  48         None => {
  49             assert!(dest == expr::Ignore || bcx.unreachable);
  50         }
  51     }
  52     return bcx;
  53 }

  55 pub fn trans_if(bcx: @mut Block,
  56             cond: @ast::expr,
  57             thn: &ast::Block,
  58             els: Option<@ast::expr>,
  59             dest: expr::Dest)
  60          -> @mut Block {
  61     debug!("trans_if(bcx=%s, cond=%s, thn=%?, dest=%s)",
  62            bcx.to_str(), bcx.expr_to_str(cond), thn.id,
  63            dest.to_str(bcx.ccx()));
  64     let _indenter = indenter();
  65 
  66     let _icx = push_ctxt("trans_if");
  67 
  68     let Result {bcx, val: cond_val} =
  69         expr::trans_to_datum(bcx, cond).to_result();
  70 
  71     let cond_val = bool_to_i1(bcx, cond_val);
  72 
  73     // Drop branches that are known to be impossible
  74     if is_const(cond_val) && !is_undef(cond_val) {
  75         if const_to_uint(cond_val) == 1 {
  76             // if true { .. } [else { .. }]
  77             return do with_scope(bcx, thn.info(), "if_true_then") |bcx| {
  78                 let bcx_out = trans_block(bcx, thn, dest);
  79                 trans_block_cleanups(bcx_out, block_cleanups(bcx))
  80             }
  81         } else {
  82             match els {
  83                 // if false { .. } else { .. }
  84                 Some(elexpr) => {
  85                     return do with_scope(bcx, elexpr.info(), "if_false_then") |bcx| {
  86                         let bcx_out = trans_if_else(bcx, elexpr, dest);
  87                         trans_block_cleanups(bcx_out, block_cleanups(bcx))
  88                     }
  89                 }
  90                 // if false { .. }
  91                 None => return bcx,
  92             }
  93         }
  94     }
  95 
  96     let then_bcx_in = scope_block(bcx, thn.info(), "then");
  97 
  98     let then_bcx_out = trans_block(then_bcx_in, thn, dest);
  99     let then_bcx_out = trans_block_cleanups(then_bcx_out,
 100                                             block_cleanups(then_bcx_in));
 101 
 102     // Calling trans_block directly instead of trans_expr
 103     // because trans_expr will create another scope block
 104     // context for the block, but we've already got the
 105     // 'else' context
 106     let (else_bcx_in, next_bcx) = match els {
 107       Some(elexpr) => {
 108           let else_bcx_in = scope_block(bcx, elexpr.info(), "else");
 109           let else_bcx_out = trans_if_else(else_bcx_in, elexpr, dest);
 110           (else_bcx_in, join_blocks(bcx, [then_bcx_out, else_bcx_out]))
 111       }
 112       _ => {
 113           let next_bcx = sub_block(bcx, "next");
 114           Br(then_bcx_out, next_bcx.llbb);
 115 
 116           (next_bcx, next_bcx)
 117       }
 118     };
 119 
 120     debug!("then_bcx_in=%s, else_bcx_in=%s",
 121            then_bcx_in.to_str(), else_bcx_in.to_str());
 122 
 123     CondBr(bcx, cond_val, then_bcx_in.llbb, else_bcx_in.llbb);
 124     return next_bcx;
 125 
 126     // trans `else [ if { .. } ... | { .. } ]`
 127     fn trans_if_else(else_bcx_in: @mut Block, elexpr: @ast::expr,
 128                      dest: expr::Dest) -> @mut Block {
 129         let else_bcx_out = match elexpr.node {
 130             ast::expr_if(_, _, _) => {
 131                 let elseif_blk = ast_util::block_from_expr(elexpr);
 132                 trans_block(else_bcx_in, &elseif_blk, dest)
 133             }
 134             ast::expr_block(ref blk) => {
 135                 trans_block(else_bcx_in, blk, dest)
 136             }
 137             // would be nice to have a constraint on ifs
 138             _ => else_bcx_in.tcx().sess.bug("strange alternative in if")
 139         };
 140         trans_block_cleanups(else_bcx_out, block_cleanups(else_bcx_in))
 141     }
 142 }

 144 pub fn join_blocks(parent_bcx: @mut Block, in_cxs: &[@mut Block]) -> @mut Block {
 145     let out = sub_block(parent_bcx, "join");
 146     let mut reachable = false;
 147     for bcx in in_cxs.iter() {
 148         if !bcx.unreachable {
 149             Br(*bcx, out.llbb);
 150             reachable = true;
 151         }
 152     }
 153     if !reachable {
 154         Unreachable(out);
 155     }
 156     return out;
 157 }

 159 pub fn trans_while(bcx: @mut Block, cond: @ast::expr, body: &ast::Block) -> @mut Block {
 160     let _icx = push_ctxt("trans_while");
 161     let next_bcx = sub_block(bcx, "while next");
 162 
 163     //            bcx
 164     //             |
 165     //          loop_bcx
 166     //             |
 167     //         cond_bcx_in  <--------+
 168     //             |                 |
 169     //         cond_bcx_out          |
 170     //           |      |            |
 171     //           |    body_bcx_in    |
 172     //    +------+      |            |
 173     //    |           body_bcx_out --+
 174     // next_bcx
 175 
 176     let loop_bcx = loop_scope_block(bcx, next_bcx, None, "`while`",
 177                                     body.info());
 178     let cond_bcx_in = scope_block(loop_bcx, cond.info(), "while loop cond");
 179     let body_bcx_in = scope_block(loop_bcx, body.info(), "while loop body");
 180     Br(bcx, loop_bcx.llbb);
 181     Br(loop_bcx, cond_bcx_in.llbb);
 182 
 183     // compile the condition
 184     let Result {bcx: cond_bcx_out, val: cond_val} =
 185         expr::trans_to_datum(cond_bcx_in, cond).to_result();
 186     let cond_val = bool_to_i1(cond_bcx_out, cond_val);
 187     let cond_bcx_out =
 188         trans_block_cleanups(cond_bcx_out, block_cleanups(cond_bcx_in));
 189     CondBr(cond_bcx_out, cond_val, body_bcx_in.llbb, next_bcx.llbb);
 190 
 191     // loop body:
 192     let body_bcx_out = trans_block(body_bcx_in, body, expr::Ignore);
 193     cleanup_and_Br(body_bcx_out, body_bcx_in, cond_bcx_in.llbb);
 194 
 195     return next_bcx;
 196 }

 198 pub fn trans_loop(bcx:@mut Block,
 199                   body: &ast::Block,
 200                   opt_label: Option<ident>)
 201                -> @mut Block {
 202     let _icx = push_ctxt("trans_loop");
 203     let next_bcx = sub_block(bcx, "next");
 204     let body_bcx_in = loop_scope_block(bcx, next_bcx, opt_label, "`loop`",
 205                                        body.info());
 206     Br(bcx, body_bcx_in.llbb);
 207     let body_bcx_out = trans_block(body_bcx_in, body, expr::Ignore);
 208     cleanup_and_Br(body_bcx_out, body_bcx_in, body_bcx_in.llbb);
 209     return next_bcx;
 210 }

 212 pub fn trans_log(log_ex: &ast::expr,
 213                  lvl: @ast::expr,
 214                  bcx: @mut Block,
 215                  e: @ast::expr) -> @mut Block {
 216     let _icx = push_ctxt("trans_log");
 217     let ccx = bcx.ccx();
 218     let mut bcx = bcx;
 219     if ty::type_is_bot(expr_ty(bcx, lvl)) {
 220        return expr::trans_into(bcx, lvl, expr::Ignore);
 221     }
 222 
 223     let (modpath, modname) = {
 224         let path = &mut bcx.fcx.path;
 225         let mut modpath = ~[path_mod(ccx.sess.ident_of(ccx.link_meta.name))];
 226         for e in path.iter() {
 227             match *e {
 228                 path_mod(_) => { modpath.push(*e) }
 229                 _ => {}
 230             }
 231         }
 232         let modname = path_str(ccx.sess, modpath);
 233         (modpath, modname)
 234     };
 235 
 236     let global = if ccx.module_data.contains_key(&modname) {
 237         ccx.module_data.get_copy(&modname)
 238     } else {
 239         let s = link::mangle_internal_name_by_path_and_seq(
 240             ccx, modpath, "loglevel");
 241         let global;
 242         unsafe {
 243             global = do s.as_c_str |buf| {
 244                 llvm::LLVMAddGlobal(ccx.llmod, Type::i32().to_ref(), buf)
 245             };
 246             llvm::LLVMSetGlobalConstant(global, False);
 247             llvm::LLVMSetInitializer(global, C_null(Type::i32()));
 248             lib::llvm::SetLinkage(global, lib::llvm::InternalLinkage);
 249         }
 250         ccx.module_data.insert(modname, global);
 251         global
 252     };
 253     let current_level = Load(bcx, global);
 254     let level = unpack_result!(bcx, {
 255         do with_scope_result(bcx, lvl.info(), "level") |bcx| {
 256             expr::trans_to_datum(bcx, lvl).to_result()
 257         }
 258     });
 259 
 260     let llenabled = ICmp(bcx, lib::llvm::IntUGE, current_level, level);
 261     do with_cond(bcx, llenabled) |bcx| {
 262         do with_scope(bcx, log_ex.info(), "log") |bcx| {
 263             let mut bcx = bcx;
 264 
 265             // Translate the value to be logged
 266             let val_datum = unpack_datum!(bcx, expr::trans_to_datum(bcx, e));
 267 
 268             // Call the polymorphic log function
 269             let val = val_datum.to_ref_llval(bcx);
 270             let did = langcall(bcx, Some(e.span), "", LogTypeFnLangItem);
 271             let bcx = callee::trans_lang_call_with_type_params(
 272                 bcx, did, [level, val], [val_datum.ty], expr::Ignore);
 273             bcx
 274         }
 275     }
 276 }

 278 pub fn trans_break_cont(bcx: @mut Block,
 279                         opt_label: Option<ident>,
 280                         to_end: bool)
 281                      -> @mut Block {
 282     let _icx = push_ctxt("trans_break_cont");
 283     // Locate closest loop block, outputting cleanup as we go.
 284     let mut unwind = bcx;
 285     let mut cur_scope = unwind.scope;
 286     let mut target;
 287     loop {
 288         cur_scope = match cur_scope {
 289             Some(@ScopeInfo {
 290                 loop_break: Some(brk),
 291                 loop_label: l,
 292                 parent,
 293                 _
 294             }) => {
 295                 // If we're looking for a labeled loop, check the label...
 296                 target = if to_end {
 297                     brk
 298                 } else {
 299                     unwind
 300                 };
 301                 match opt_label {
 302                     Some(desired) => match l {
 303                         Some(actual) if actual == desired => break,
 304                         // If it doesn't match the one we want,
 305                         // don't break
 306                         _ => parent,
 307                     },
 308                     None => break,
 309                 }
 310             }
 311             Some(inf) => inf.parent,
 312             None => {
 313                 unwind = match unwind.parent {
 314                     Some(bcx) => bcx,
 315                         // This is a return from a loop body block
 316                         None => {
 317                             Store(bcx, C_bool(!to_end), bcx.fcx.llretptr.get());
 318                             cleanup_and_leave(bcx, None, Some(bcx.fcx.get_llreturn()));
 319                             Unreachable(bcx);
 320                             return bcx;
 321                         }
 322                 };
 323                 unwind.scope
 324             }
 325         }
 326     }
 327     cleanup_and_Br(bcx, unwind, target.llbb);
 328     Unreachable(bcx);
 329     return bcx;
 330 }

 340 pub fn trans_ret(bcx: @mut Block, e: Option<@ast::expr>) -> @mut Block {
 341     let _icx = push_ctxt("trans_ret");
 342     let mut bcx = bcx;
 343     let dest = match bcx.fcx.loop_ret {
 344       Some((flagptr, retptr)) => {
 345         // This is a loop body return. Must set continue flag (our retptr)
 346         // to false, return flag to true, and then store the value in the
 347         // parent's retptr.
 348         Store(bcx, C_bool(true), flagptr);
 349         Store(bcx, C_bool(false), bcx.fcx.llretptr.get());
 350         expr::SaveIn(match e {
 351           Some(x) => PointerCast(bcx, retptr,
 352                                  type_of(bcx.ccx(), expr_ty(bcx, x)).ptr_to()),
 353           None => retptr
 354         })
 355       }
 356       None => match bcx.fcx.llretptr {
 357         None => expr::Ignore,
 358         Some(retptr) => expr::SaveIn(retptr),
 359       }
 360     };
 361     match e {
 362       Some(x) => {
 363         bcx = expr::trans_into(bcx, x, dest);
 364       }
 365       _ => ()
 366     }
 367     cleanup_and_leave(bcx, None, Some(bcx.fcx.get_llreturn()));
 368     Unreachable(bcx);
 369     return bcx;
 370 }

 372 pub fn trans_fail_expr(bcx: @mut Block,
 373                        sp_opt: Option<span>,
 374                        fail_expr: Option<@ast::expr>)
 375                     -> @mut Block {
 376     let _icx = push_ctxt("trans_fail_expr");
 377     let mut bcx = bcx;
 378     match fail_expr {
 379         Some(arg_expr) => {
 380             let ccx = bcx.ccx();
 381             let tcx = ccx.tcx;
 382             let arg_datum = unpack_datum!(
 383                 bcx, expr::trans_to_datum(bcx, arg_expr));
 384 
 385             if ty::type_is_str(arg_datum.ty) {
 386                 let (lldata, _) = arg_datum.get_vec_base_and_len_no_root(bcx);
 387                 return trans_fail_value(bcx, sp_opt, lldata);
 388             } else if bcx.unreachable || ty::type_is_bot(arg_datum.ty) {
 389                 return bcx;
 390             } else {
 391                 bcx.sess().span_bug(
 392                     arg_expr.span, ~"fail called with unsupported type " +
 393                     ppaux::ty_to_str(tcx, arg_datum.ty));
 394             }
 395         }
 396         _ => trans_fail(bcx, sp_opt, @"explicit failure")
 397     }
 398 }

 409 fn trans_fail_value(bcx: @mut Block,
 410                     sp_opt: Option<span>,
 411                     V_fail_str: ValueRef)
 412                  -> @mut Block {
 413     let _icx = push_ctxt("trans_fail_value");
 414     let ccx = bcx.ccx();
 415     let (V_filename, V_line) = match sp_opt {
 416       Some(sp) => {
 417         let sess = bcx.sess();
 418         let loc = sess.parse_sess.cm.lookup_char_pos(sp.lo);
 419         (C_cstr(bcx.ccx(), loc.file.name),
 420          loc.line as int)
 421       }
 422       None => {
 423         (C_cstr(bcx.ccx(), @"<runtime>"), 0)
 424       }
 425     };
 426     let V_str = PointerCast(bcx, V_fail_str, Type::i8p());
 427     let V_filename = PointerCast(bcx, V_filename, Type::i8p());
 428     let args = ~[V_str, V_filename, C_int(ccx, V_line)];
 429     let did = langcall(bcx, sp_opt, "", FailFnLangItem);
 430     let bcx = callee::trans_lang_call(bcx, did, args, Some(expr::Ignore)).bcx;
 431     Unreachable(bcx);
 432     return bcx;
 433 }

 435 pub fn trans_fail_bounds_check(bcx: @mut Block, sp: span,
 436                                index: ValueRef, len: ValueRef) -> @mut Block {
 437     let _icx = push_ctxt("trans_fail_bounds_check");
 438     let (filename, line) = filename_and_line_num_from_span(bcx, sp);

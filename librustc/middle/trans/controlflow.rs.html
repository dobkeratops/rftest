  37  pub fn trans_block(bcx: @mut Block, b: &ast::Block, dest: expr::Dest) -> @mut Block {
  38      let _icx = push_ctxt("trans_block");
  39      let mut bcx = bcx;
  40      for s in b.stmts.iter() {
  41          debuginfo::update_source_pos(bcx, b.span);
  42          bcx = trans_stmt(bcx, *s);
  43      }
  44      match b.expr {
  45          Some(e) => {
  46              debuginfo::update_source_pos(bcx, e.span);
  47              bcx = expr::trans_into(bcx, e, dest);
  48          }
  49          None => {
  50              assert!(dest == expr::Ignore || bcx.unreachable);
  51          }
  52      }
  53      return bcx;
  54  }

  56  pub fn trans_if(bcx: @mut Block,
  57              cond: @ast::expr,
  58              thn: &ast::Block,
  59              els: Option<@ast::expr>,
  60              dest: expr::Dest)
  61           -> @mut Block {
  62      debug!("trans_if(bcx=%s, cond=%s, thn=%?, dest=%s)",
  63             bcx.to_str(), bcx.expr_to_str(cond), thn.id,
  64             dest.to_str(bcx.ccx()));
  65      let _indenter = indenter();
  66  
  67      let _icx = push_ctxt("trans_if");
  68  
  69      let Result {bcx, val: cond_val} =
  70          expr::trans_to_datum(bcx, cond).to_result();
  71  
  72      let cond_val = bool_to_i1(bcx, cond_val);
  73  
  74      // Drop branches that are known to be impossible
  75      if is_const(cond_val) && !is_undef(cond_val) {
  76          if const_to_uint(cond_val) == 1 {
  77              // if true { .. } [else { .. }]
  78              return do with_scope(bcx, thn.info(), "if_true_then") |bcx| {
  79                  let bcx_out = trans_block(bcx, thn, dest);
  80                  trans_block_cleanups(bcx_out, block_cleanups(bcx))
  81              }
  82          } else {
  83              match els {
  84                  // if false { .. } else { .. }
  85                  Some(elexpr) => {
  86                      return do with_scope(bcx, elexpr.info(), "if_false_then") |bcx| {
  87                          let bcx_out = trans_if_else(bcx, elexpr, dest);
  88                          trans_block_cleanups(bcx_out, block_cleanups(bcx))
  89                      }
  90                  }
  91                  // if false { .. }
  92                  None => return bcx,
  93              }
  94          }
  95      }
  96  
  97      let then_bcx_in = scope_block(bcx, thn.info(), "then");
  98  
  99      let then_bcx_out = trans_block(then_bcx_in, thn, dest);
 100      let then_bcx_out = trans_block_cleanups(then_bcx_out,
 101                                              block_cleanups(then_bcx_in));
 102  
 103      // Calling trans_block directly instead of trans_expr
 104      // because trans_expr will create another scope block
 105      // context for the block, but we've already got the
 106      // 'else' context
 107      let (else_bcx_in, next_bcx) = match els {
 108        Some(elexpr) => {
 109            let else_bcx_in = scope_block(bcx, elexpr.info(), "else");
 110            let else_bcx_out = trans_if_else(else_bcx_in, elexpr, dest);
 111            (else_bcx_in, join_blocks(bcx, [then_bcx_out, else_bcx_out]))
 112        }
 113        _ => {
 114            let next_bcx = sub_block(bcx, "next");
 115            Br(then_bcx_out, next_bcx.llbb);
 116  
 117            (next_bcx, next_bcx)
 118        }
 119      };
 120  
 121      debug!("then_bcx_in=%s, else_bcx_in=%s",
 122             then_bcx_in.to_str(), else_bcx_in.to_str());
 123  
 124      CondBr(bcx, cond_val, then_bcx_in.llbb, else_bcx_in.llbb);
 125      return next_bcx;
 126  
 127      // trans `else [ if { .. } ... | { .. } ]`
 128      fn trans_if_else(else_bcx_in: @mut Block, elexpr: @ast::expr,
 129                       dest: expr::Dest) -> @mut Block {
 130          let else_bcx_out = match elexpr.node {
 131              ast::expr_if(_, _, _) => {
 132                  let elseif_blk = ast_util::block_from_expr(elexpr);
 133                  trans_block(else_bcx_in, &elseif_blk, dest)
 134              }
 135              ast::expr_block(ref blk) => {
 136                  trans_block(else_bcx_in, blk, dest)
 137              }
 138              // would be nice to have a constraint on ifs
 139              _ => else_bcx_in.tcx().sess.bug("strange alternative in if")
 140          };
 141          trans_block_cleanups(else_bcx_out, block_cleanups(else_bcx_in))
 142      }
 143  }

 145  pub fn join_blocks(parent_bcx: @mut Block, in_cxs: &[@mut Block]) -> @mut Block {
 146      let out = sub_block(parent_bcx, "join");
 147      let mut reachable = false;
 148      for bcx in in_cxs.iter() {
 149          if !bcx.unreachable {
 150              Br(*bcx, out.llbb);
 151              reachable = true;
 152          }
 153      }
 154      if !reachable {
 155          Unreachable(out);
 156      }
 157      return out;
 158  }

 160  pub fn trans_while(bcx: @mut Block, cond: @ast::expr, body: &ast::Block) -> @mut Block {
 161      let _icx = push_ctxt("trans_while");
 162      let next_bcx = sub_block(bcx, "while next");
 163  
 164      //            bcx
 165      //             |
 166      //          loop_bcx
 167      //             |
 168      //         cond_bcx_in  <--------+
 169      //             |                 |
 170      //         cond_bcx_out          |
 171      //           |      |            |
 172      //           |    body_bcx_in    |
 173      //    +------+      |            |
 174      //    |           body_bcx_out --+
 175      // next_bcx
 176  
 177      let loop_bcx = loop_scope_block(bcx, next_bcx, None, "`while`",
 178                                      body.info());
 179      let cond_bcx_in = scope_block(loop_bcx, cond.info(), "while loop cond");
 180      let body_bcx_in = scope_block(loop_bcx, body.info(), "while loop body");
 181      Br(bcx, loop_bcx.llbb);
 182      Br(loop_bcx, cond_bcx_in.llbb);
 183  
 184      // compile the condition
 185      let Result {bcx: cond_bcx_out, val: cond_val} =
 186          expr::trans_to_datum(cond_bcx_in, cond).to_result();
 187      let cond_val = bool_to_i1(cond_bcx_out, cond_val);
 188      let cond_bcx_out =
 189          trans_block_cleanups(cond_bcx_out, block_cleanups(cond_bcx_in));
 190      CondBr(cond_bcx_out, cond_val, body_bcx_in.llbb, next_bcx.llbb);
 191  
 192      // loop body:
 193      let body_bcx_out = trans_block(body_bcx_in, body, expr::Ignore);
 194      cleanup_and_Br(body_bcx_out, body_bcx_in, cond_bcx_in.llbb);
 195  
 196      return next_bcx;
 197  }

 199  pub fn trans_loop(bcx:@mut Block,
 200                    body: &ast::Block,
 201                    opt_label: Option<ident>)
 202                 -> @mut Block {
 203      let _icx = push_ctxt("trans_loop");
 204      let next_bcx = sub_block(bcx, "next");
 205      let body_bcx_in = loop_scope_block(bcx, next_bcx, opt_label, "`loop`",
 206                                         body.info());
 207      Br(bcx, body_bcx_in.llbb);
 208      let body_bcx_out = trans_block(body_bcx_in, body, expr::Ignore);
 209      cleanup_and_Br(body_bcx_out, body_bcx_in, body_bcx_in.llbb);
 210      return next_bcx;
 211  }

 213  pub fn trans_log(log_ex: &ast::expr,
 214                   lvl: @ast::expr,
 215                   bcx: @mut Block,
 216                   e: @ast::expr) -> @mut Block {
 217      let _icx = push_ctxt("trans_log");
 218      let ccx = bcx.ccx();
 219      let mut bcx = bcx;
 220      if ty::type_is_bot(expr_ty(bcx, lvl)) {
 221         return expr::trans_into(bcx, lvl, expr::Ignore);
 222      }
 223  
 224      let (modpath, modname) = {
 225          let path = &mut bcx.fcx.path;
 226          let mut modpath = ~[path_mod(ccx.sess.ident_of(ccx.link_meta.name))];
 227          for e in path.iter() {
 228              match *e {
 229                  path_mod(_) => { modpath.push(*e) }
 230                  _ => {}
 231              }
 232          }
 233          let modname = path_str(ccx.sess, modpath);
 234          (modpath, modname)
 235      };
 236  
 237      let global = if ccx.module_data.contains_key(&modname) {
 238          ccx.module_data.get_copy(&modname)
 239      } else {
 240          let s = link::mangle_internal_name_by_path_and_seq(
 241              ccx, modpath, "loglevel");
 242          let global;
 243          unsafe {
 244              global = do s.to_c_str().with_ref |buf| {
 245                  llvm::LLVMAddGlobal(ccx.llmod, Type::i32().to_ref(), buf)
 246              };
 247              llvm::LLVMSetGlobalConstant(global, False);
 248              llvm::LLVMSetInitializer(global, C_null(Type::i32()));
 249              lib::llvm::SetLinkage(global, lib::llvm::InternalLinkage);
 250          }
 251          ccx.module_data.insert(modname, global);
 252          global
 253      };
 254      let current_level = Load(bcx, global);
 255      let level = unpack_result!(bcx, {
 256          do with_scope_result(bcx, lvl.info(), "level") |bcx| {
 257              expr::trans_to_datum(bcx, lvl).to_result()
 258          }
 259      });
 260  
 261      let llenabled = ICmp(bcx, lib::llvm::IntUGE, current_level, level);
 262      do with_cond(bcx, llenabled) |bcx| {
 263          do with_scope(bcx, log_ex.info(), "log") |bcx| {
 264              let mut bcx = bcx;
 265  
 266              // Translate the value to be logged
 267              let val_datum = unpack_datum!(bcx, expr::trans_to_datum(bcx, e));
 268  
 269              // Call the polymorphic log function
 270              let val = val_datum.to_ref_llval(bcx);
 271              let did = langcall(bcx, Some(e.span), "", LogTypeFnLangItem);
 272              let bcx = callee::trans_lang_call_with_type_params(
 273                  bcx, did, [level, val], [val_datum.ty], expr::Ignore);
 274              bcx
 275          }
 276      }
 277  }

 279  pub fn trans_break_cont(bcx: @mut Block,
 280                          opt_label: Option<ident>,
 281                          to_end: bool)
 282                       -> @mut Block {
 283      let _icx = push_ctxt("trans_break_cont");
 284      // Locate closest loop block, outputting cleanup as we go.
 285      let mut unwind = bcx;
 286      let mut cur_scope = unwind.scope;
 287      let mut target;
 288      loop {
 289          cur_scope = match cur_scope {
 290              Some(@ScopeInfo {
 291                  loop_break: Some(brk),
 292                  loop_label: l,
 293                  parent,
 294                  _
 295              }) => {
 296                  // If we're looking for a labeled loop, check the label...
 297                  target = if to_end {
 298                      brk
 299                  } else {
 300                      unwind
 301                  };
 302                  match opt_label {
 303                      Some(desired) => match l {
 304                          Some(actual) if actual == desired => break,
 305                          // If it doesn't match the one we want,
 306                          // don't break
 307                          _ => parent,
 308                      },
 309                      None => break,
 310                  }
 311              }
 312              Some(inf) => inf.parent,
 313              None => {
 314                  unwind = match unwind.parent {
 315                      Some(bcx) => bcx,
 316                          // This is a return from a loop body block
 317                          None => {
 318                              Store(bcx, C_bool(!to_end), bcx.fcx.llretptr.unwrap());
 319                              cleanup_and_leave(bcx, None, Some(bcx.fcx.get_llreturn()));
 320                              Unreachable(bcx);
 321                              return bcx;
 322                          }
 323                  };
 324                  unwind.scope
 325              }
 326          }
 327      }
 328      cleanup_and_Br(bcx, unwind, target.llbb);
 329      Unreachable(bcx);
 330      return bcx;
 331  }

 341  pub fn trans_ret(bcx: @mut Block, e: Option<@ast::expr>) -> @mut Block {
 342      let _icx = push_ctxt("trans_ret");
 343      let mut bcx = bcx;
 344      let dest = match bcx.fcx.loop_ret {
 345        Some((flagptr, retptr)) => {
 346          // This is a loop body return. Must set continue flag (our retptr)
 347          // to false, return flag to true, and then store the value in the
 348          // parent's retptr.
 349          Store(bcx, C_bool(true), flagptr);
 350          Store(bcx, C_bool(false), bcx.fcx.llretptr.unwrap());
 351          expr::SaveIn(match e {
 352            Some(x) => PointerCast(bcx, retptr,
 353                                   type_of(bcx.ccx(), expr_ty(bcx, x)).ptr_to()),
 354            None => retptr
 355          })
 356        }
 357        None => match bcx.fcx.llretptr {
 358          None => expr::Ignore,
 359          Some(retptr) => expr::SaveIn(retptr),
 360        }
 361      };
 362      match e {
 363        Some(x) => {
 364          bcx = expr::trans_into(bcx, x, dest);
 365        }
 366        _ => ()
 367      }
 368      cleanup_and_leave(bcx, None, Some(bcx.fcx.get_llreturn()));
 369      Unreachable(bcx);
 370      return bcx;
 371  }

 373  pub fn trans_fail_expr(bcx: @mut Block,
 374                         sp_opt: Option<span>,
 375                         fail_expr: Option<@ast::expr>)
 376                      -> @mut Block {
 377      let _icx = push_ctxt("trans_fail_expr");
 378      let mut bcx = bcx;
 379      match fail_expr {
 380          Some(arg_expr) => {
 381              let ccx = bcx.ccx();
 382              let tcx = ccx.tcx;
 383              let arg_datum = unpack_datum!(
 384                  bcx, expr::trans_to_datum(bcx, arg_expr));
 385  
 386              if ty::type_is_str(arg_datum.ty) {
 387                  let (lldata, _) = arg_datum.get_vec_base_and_len_no_root(bcx);
 388                  return trans_fail_value(bcx, sp_opt, lldata);
 389              } else if bcx.unreachable || ty::type_is_bot(arg_datum.ty) {
 390                  return bcx;
 391              } else {
 392                  bcx.sess().span_bug(
 393                      arg_expr.span, ~"fail called with unsupported type " +
 394                      ppaux::ty_to_str(tcx, arg_datum.ty));
 395              }
 396          }
 397          _ => trans_fail(bcx, sp_opt, @"explicit failure")
 398      }
 399  }

 401  pub fn trans_fail(bcx: @mut Block,
 402                    sp_opt: Option<span>,
 403                    fail_str: @str)
 404                 -> @mut Block {
 405      let _icx = push_ctxt("trans_fail");
 406      let V_fail_str = C_cstr(bcx.ccx(), fail_str);
 407      return trans_fail_value(bcx, sp_opt, V_fail_str);
 408  }

 410  fn trans_fail_value(bcx: @mut Block,
 411                      sp_opt: Option<span>,
 412                      V_fail_str: ValueRef)
 413                   -> @mut Block {
 414      let _icx = push_ctxt("trans_fail_value");
 415      let ccx = bcx.ccx();
 416      let (V_filename, V_line) = match sp_opt {
 417        Some(sp) => {
 418          let sess = bcx.sess();
 419          let loc = sess.parse_sess.cm.lookup_char_pos(sp.lo);
 420          (C_cstr(bcx.ccx(), loc.file.name),
 421           loc.line as int)

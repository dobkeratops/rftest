  29  pub fn trans_block(bcx: @mut Block, b: &ast::Block, dest: expr::Dest) -> @mut Block {
  30      let _icx = push_ctxt("trans_block");
  31      let mut bcx = bcx;
  32      for s in b.stmts.iter() {
  33          bcx = trans_stmt(bcx, *s);
  34      }
  35      match b.expr {
  36          Some(e) => {
  37              bcx = expr::trans_into(bcx, e, dest);
  38          }
  39          None => {
  40              assert!(dest == expr::Ignore || bcx.unreachable);
  41          }
  42      }
  43      return bcx;
  44  }

  46  pub fn trans_if(bcx: @mut Block,
  47              cond: @ast::Expr,
  48              thn: &ast::Block,
  49              els: Option<@ast::Expr>,
  50              dest: expr::Dest)
  51           -> @mut Block {
  52      debug!("trans_if(bcx=%s, cond=%s, thn=%?, dest=%s)",
  53             bcx.to_str(), bcx.expr_to_str(cond), thn.id,
  54             dest.to_str(bcx.ccx()));
  55      let _indenter = indenter();
  56  
  57      let _icx = push_ctxt("trans_if");
  58  
  59      let Result {bcx, val: cond_val} =
  60          expr::trans_to_datum(bcx, cond).to_result();
  61  
  62      let cond_val = bool_to_i1(bcx, cond_val);
  63  
  64      // Drop branches that are known to be impossible
  65      if is_const(cond_val) && !is_undef(cond_val) {
  66          if const_to_uint(cond_val) == 1 {
  67              // if true { .. } [else { .. }]
  68              return do with_scope(bcx, thn.info(), "if_true_then") |bcx| {
  69                  let bcx_out = trans_block(bcx, thn, dest);
  70                  trans_block_cleanups(bcx_out, block_cleanups(bcx))
  71              }
  72          } else {
  73              match els {
  74                  // if false { .. } else { .. }
  75                  Some(elexpr) => {
  76                      return do with_scope(bcx, elexpr.info(), "if_false_then") |bcx| {
  77                          let bcx_out = trans_if_else(bcx, elexpr, dest);
  78                          trans_block_cleanups(bcx_out, block_cleanups(bcx))
  79                      }
  80                  }
  81                  // if false { .. }
  82                  None => return bcx,
  83              }
  84          }
  85      }
  86  
  87      let then_bcx_in = scope_block(bcx, thn.info(), "then");
  88  
  89      let then_bcx_out = trans_block(then_bcx_in, thn, dest);
  90      let then_bcx_out = trans_block_cleanups(then_bcx_out,
  91                                              block_cleanups(then_bcx_in));
  92  
  93      // Calling trans_block directly instead of trans_expr
  94      // because trans_expr will create another scope block
  95      // context for the block, but we've already got the
  96      // 'else' context
  97      let (else_bcx_in, next_bcx) = match els {
  98        Some(elexpr) => {
  99            let else_bcx_in = scope_block(bcx, elexpr.info(), "else");
 100            let else_bcx_out = trans_if_else(else_bcx_in, elexpr, dest);
 101            (else_bcx_in, join_blocks(bcx, [then_bcx_out, else_bcx_out]))
 102        }
 103        _ => {
 104            let next_bcx = sub_block(bcx, "next");
 105            Br(then_bcx_out, next_bcx.llbb);
 106  
 107            (next_bcx, next_bcx)
 108        }
 109      };
 110  
 111      debug!("then_bcx_in=%s, else_bcx_in=%s",
 112             then_bcx_in.to_str(), else_bcx_in.to_str());
 113  
 114      CondBr(bcx, cond_val, then_bcx_in.llbb, else_bcx_in.llbb);
 115      return next_bcx;
 116  
 117      // trans `else [ if { .. } ... | { .. } ]`
 118      fn trans_if_else(else_bcx_in: @mut Block, elexpr: @ast::Expr,
 119                       dest: expr::Dest) -> @mut Block {
 120          let else_bcx_out = match elexpr.node {
 121              ast::ExprIf(_, _, _) => {
 122                  let elseif_blk = ast_util::block_from_expr(elexpr);
 123                  trans_block(else_bcx_in, &elseif_blk, dest)
 124              }
 125              ast::ExprBlock(ref blk) => {
 126                  trans_block(else_bcx_in, blk, dest)
 127              }
 128              // would be nice to have a constraint on ifs
 129              _ => else_bcx_in.tcx().sess.bug("strange alternative in if")
 130          };
 131          trans_block_cleanups(else_bcx_out, block_cleanups(else_bcx_in))
 132      }
 133  }

 135  pub fn join_blocks(parent_bcx: @mut Block, in_cxs: &[@mut Block]) -> @mut Block {
 136      let out = sub_block(parent_bcx, "join");
 137      let mut reachable = false;
 138      for bcx in in_cxs.iter() {
 139          if !bcx.unreachable {
 140              Br(*bcx, out.llbb);
 141              reachable = true;
 142          }
 143      }
 144      if !reachable {
 145          Unreachable(out);
 146      }
 147      return out;
 148  }

 150  pub fn trans_while(bcx: @mut Block, cond: @ast::Expr, body: &ast::Block) -> @mut Block {
 151      let _icx = push_ctxt("trans_while");
 152      let next_bcx = sub_block(bcx, "while next");
 153  
 154      //            bcx
 155      //             |
 156      //          loop_bcx
 157      //             |
 158      //         cond_bcx_in  <--------+
 159      //             |                 |
 160      //         cond_bcx_out          |
 161      //           |      |            |
 162      //           |    body_bcx_in    |
 163      //    +------+      |            |
 164      //    |           body_bcx_out --+
 165      // next_bcx
 166  
 167      let loop_bcx = loop_scope_block(bcx, next_bcx, None, "`while`",
 168                                      body.info());
 169      let cond_bcx_in = scope_block(loop_bcx, cond.info(), "while loop cond");
 170      let body_bcx_in = scope_block(loop_bcx, body.info(), "while loop body");
 171      Br(bcx, loop_bcx.llbb);
 172      Br(loop_bcx, cond_bcx_in.llbb);
 173  
 174      // compile the condition
 175      let Result {bcx: cond_bcx_out, val: cond_val} =
 176          expr::trans_to_datum(cond_bcx_in, cond).to_result();
 177      let cond_val = bool_to_i1(cond_bcx_out, cond_val);
 178      let cond_bcx_out =
 179          trans_block_cleanups(cond_bcx_out, block_cleanups(cond_bcx_in));
 180      CondBr(cond_bcx_out, cond_val, body_bcx_in.llbb, next_bcx.llbb);
 181  
 182      // loop body:
 183      let body_bcx_out = trans_block(body_bcx_in, body, expr::Ignore);
 184      cleanup_and_Br(body_bcx_out, body_bcx_in, cond_bcx_in.llbb);
 185  
 186      return next_bcx;
 187  }

 203  pub fn trans_break_cont(bcx: @mut Block,
 204                          opt_label: Option<Name>,
 205                          to_end: bool)
 206                       -> @mut Block {
 207      let _icx = push_ctxt("trans_break_cont");
 208      // Locate closest loop block, outputting cleanup as we go.
 209      let mut unwind = bcx;
 210      let mut cur_scope = unwind.scope;
 211      let mut target;
 212      loop {
 213          cur_scope = match cur_scope {
 214              Some(@ScopeInfo {
 215                  loop_break: Some(brk),
 216                  loop_label: l,
 217                  parent,
 218                  _
 219              }) => {
 220                  // If we're looking for a labeled loop, check the label...
 221                  target = if to_end {
 222                      brk
 223                  } else {
 224                      unwind
 225                  };
 226                  match opt_label {
 227                      Some(desired) => match l {
 228                          Some(actual) if actual == desired => break,
 229                          // If it doesn't match the one we want,
 230                          // don't break
 231                          _ => parent,
 232                      },
 233                      None => break,
 234                  }
 235              }
 236              Some(inf) => inf.parent,
 237              None => {
 238                  unwind = match unwind.parent {
 239                      Some(bcx) => bcx,
 240                          // This is a return from a loop body block
 241                          None => {
 242                              Store(bcx, C_bool(!to_end), bcx.fcx.llretptr.unwrap());
 243                              cleanup_and_leave(bcx, None, Some(bcx.fcx.get_llreturn()));
 244                              Unreachable(bcx);
 245                              return bcx;
 246                          }
 247                  };
 248                  unwind.scope
 249              }
 250          }
 251      }
 252      cleanup_and_Br(bcx, unwind, target.llbb);
 253      Unreachable(bcx);
 254      return bcx;
 255  }

   12  /*!
   13   * A module for propagating forward dataflow information. The analysis
   14   * assumes that the items to be propagated can be represented as bits
   15   * and thus uses bitvectors. Your job is simply to specify the so-called
   16   * GEN and KILL bits for each expression.
   17   */

   33  #[deriving(Clone)]
   34  pub struct DataFlowContext<O> {
   35      priv tcx: ty::ctxt,
   36      priv method_map: typeck::method_map,
   37  
   38      /// the data flow operator
   39      priv oper: O,
   40  
   41      /// number of bits to propagate per id
   42      priv bits_per_id: uint,
   43  
   44      /// number of words we will use to store bits_per_id.
   45      /// equal to bits_per_id/uint::bits rounded up.
   46      priv words_per_id: uint,
   47  
   48      // mapping from node to bitset index.
   49      priv nodeid_to_bitset: HashMap<ast::NodeId,uint>,
   50  
   51      // Bit sets per id.  The following three fields (`gens`, `kills`,
   52      // and `on_entry`) all have the same structure. For each id in
   53      // `id_range`, there is a range of words equal to `words_per_id`.
   54      // So, to access the bits for any given id, you take a slice of
   55      // the full vector (see the method `compute_id_range()`).
   56  
   57      /// bits generated as we exit the scope `id`. Updated by `add_gen()`.
   58      priv gens: ~[uint],
   59  
   60      /// bits killed as we exit the scope `id`. Updated by `add_kill()`.
   61      priv kills: ~[uint],
   62  
   63      /// bits that are valid on entry to the scope `id`. Updated by
   64      /// `propagate()`.
   65      priv on_entry: ~[uint]
   66  }
   67  
   68  /// Parameterization for the precise form of data flow that is used.
   69  pub trait DataFlowOperator {
   70      /// Specifies the initial value for each bit in the `on_entry` set
   71      fn initial_value(&self) -> bool;
   72  
   73      /// Joins two predecessor bits together, typically either `|` or `&`
   74      fn join(&self, succ: uint, pred: uint) -> uint;
   75  
   76      /// True if we should propagate through closures
   77      fn walk_closures(&self) -> bool;
   78  }

   90  impl<O:DataFlowOperator> pprust::pp_ann for DataFlowContext<O> {
   91      fn pre(&self, node: pprust::ann_node) {
   92          let (ps, id) = match node {
   93              pprust::node_expr(ps, expr) => (ps, expr.id),
   94              pprust::node_block(ps, blk) => (ps, blk.id),
   95              pprust::node_item(ps, _) => (ps, 0),
   96              pprust::node_pat(ps, pat) => (ps, pat.id)
   97          };
   98  
   99          if self.nodeid_to_bitset.contains_key(&id) {
  100              let (start, end) = self.compute_id_range_frozen(id);
  101              let on_entry = self.on_entry.slice(start, end);
  102              let entry_str = bits_to_str(on_entry);
  103  
  104              let gens = self.gens.slice(start, end);
  105              let gens_str = if gens.iter().any(|&u| u != 0) {
  106                  fmt!(" gen: %s", bits_to_str(gens))
  107              } else {
  108                  ~""
  109              };
  110  
  111              let kills = self.kills.slice(start, end);
  112              let kills_str = if kills.iter().any(|&u| u != 0) {
  113                  fmt!(" kill: %s", bits_to_str(kills))
  114              } else {
  115                  ~""
  116              };
  117  
  118              let comment_str = fmt!("id %d: %s%s%s",
  119                                     id, entry_str, gens_str, kills_str);
  120              pprust::synth_comment(ps, comment_str);
  121              pp::space(ps.s);
  122          }
  123      }
  124  }

  163      }

  176      fn apply_gen_kill(&mut self, id: ast::NodeId, bits: &mut [uint]) {
  177          //! Applies the gen and kill sets for `id` to `bits`
  178  
  179          debug!("apply_gen_kill(id=%?, bits=%s) [before]",
  180                 id, mut_bits_to_str(bits));
  181          let (start, end) = self.compute_id_range(id);
  182          let gens = self.gens.slice(start, end);
  183          bitwise(bits, gens, |a, b| a | b);
  184          let kills = self.kills.slice(start, end);
  185          bitwise(bits, kills, |a, b| a & !b);
  186  
  187          debug!("apply_gen_kill(id=%?, bits=%s) [after]",
  188                 id, mut_bits_to_str(bits));
  189      }

  191      fn apply_kill(&mut self, id: ast::NodeId, bits: &mut [uint]) {

  194          let (start, end) = self.compute_id_range(id);
  195          let kills = self.kills.slice(start, end);
  196          bitwise(bits, kills, |a, b| a & !b);
  197          debug!("apply_kill(id=%?, bits=%s) [after]",

  199      }

  201      fn compute_id_range_frozen(&self, id: ast::NodeId) -> (uint, uint) {
  202          let n = *self.nodeid_to_bitset.get(&id);
  203          let start = n * self.words_per_id;
  204          let end = start + self.words_per_id;
  205          (start, end)
  206      }
  207  
  208      fn compute_id_range(&mut self, id: ast::NodeId) -> (uint, uint) {
  209          let mut expanded = false;
  210          let len = self.nodeid_to_bitset.len();
  211          let n = do self.nodeid_to_bitset.find_or_insert_with(id) |_| {
  212              expanded = true;
  213              len
  214          };
  215          if expanded {
  216              let entry = if self.oper.initial_value() { uint::max_value } else {0};
  217              do self.words_per_id.times {
  218                  self.gens.push(0);
  219                  self.kills.push(0);
  220                  self.on_entry.push(entry);
  221              }
  222          }
  223          let start = *n * self.words_per_id;
  224          let end = start + self.words_per_id;
  225  
  226          assert!(start < self.gens.len());
  227          assert!(end <= self.gens.len());
  228          assert!(self.gens.len() == self.kills.len());
  229          assert!(self.gens.len() == self.on_entry.len());
  230  
  231          (start, end)
  232      }

  235      pub fn each_bit_on_entry_frozen(&self,
  236                                      id: ast::NodeId,
  237                                      f: &fn(uint) -> bool) -> bool {
  238          //! Iterates through each bit that is set on entry to `id`.
  239          //! Only useful after `propagate()` has been called.
  240          if !self.nodeid_to_bitset.contains_key(&id) {
  241              return true;
  242          }
  243          let (start, end) = self.compute_id_range_frozen(id);
  244          let on_entry = self.on_entry.slice(start, end);
  245          debug!("each_bit_on_entry_frozen(id=%?, on_entry=%s)",
  246                 id, bits_to_str(on_entry));
  247          self.each_bit(on_entry, f)
  248      }

  278          //! Iterates through each bit in the gen set for `id`.
  279          if !self.nodeid_to_bitset.contains_key(&id) {
  280              return true;
  281          }
  282          let (start, end) = self.compute_id_range_frozen(id);
  283          let gens = self.gens.slice(start, end);
  284          debug!("each_gen_bit(id=%?, gens=%s)",

  289      fn each_bit(&self,
  290                  words: &[uint],
  291                  f: &fn(uint) -> bool) -> bool {
  292          //! Helper for iterating over the bits in a bit set.
  293  
  294          for (word_index, &word) in words.iter().enumerate() {
  295              if word != 0 {
  296                  let base_index = word_index * uint::bits;
  297                  for offset in range(0u, uint::bits) {
  298                      let bit = 1 << offset;
  299                      if (word & bit) != 0 {
  300                          // NB: we round up the total number of bits
  301                          // that we store in any given bit set so that
  302                          // it is an even multiple of uint::bits.  This
  303                          // means that there may be some stray bits at
  304                          // the end that do not correspond to any
  305                          // actual value.  So before we callback, check
  306                          // whether the bit_index is greater than the
  307                          // actual value the user specified and stop
  308                          // iterating if so.
  309                          let bit_index = base_index + offset;
  310                          if bit_index >= self.bits_per_id {
  311                              return true;
  312                          } else if !f(bit_index) {
  313                              return false;
  314                          }
  315                      }
  316                  }
  317              }
  318          }
  319          return true;
  320      }
  321  }

  323  impl<O:DataFlowOperator+Clone+'static> DataFlowContext<O> {
  324  //                      ^^^^^^^^^^^^^ only needed for pretty printing
  325      pub fn propagate(&mut self, blk: &ast::Block) {
  326          //! Performs the data flow analysis.
  327  
  328          if self.bits_per_id == 0 {
  329              // Optimize the surprisingly common degenerate case.
  330              return;
  331          }
  332  
  333          {
  334              let mut propcx = PropagationContext {
  335                  dfcx: self,
  336                  changed: true
  337              };
  338  
  339              let mut temp = vec::from_elem(self.words_per_id, 0u);
  340              let mut loop_scopes = ~[];
  341  
  342              while propcx.changed {
  343                  propcx.changed = false;
  344                  propcx.reset(temp);
  345                  propcx.walk_block(blk, temp, &mut loop_scopes);
  346              }
  347          }
  348  
  349          debug!("Dataflow result:");
  350          debug!("%s", {
  351              let this = @(*self).clone();
  352              this.pretty_print_to(io::stderr(), blk);
  353              ""
  354          });
  355      }
  356  
  357      fn pretty_print_to(@self, wr: @io::Writer, blk: &ast::Block) {
  358          let ps = pprust::rust_printer_annotated(wr,
  359                                                  self.tcx.sess.intr(),
  360                                                  self as @pprust::pp_ann);
  361          pprust::cbox(ps, pprust::indent_unit);
  362          pprust::ibox(ps, 0u);
  363          pprust::print_block(ps, blk);
  364          pp::eof(ps.s);
  365      }
  366  }

  368  impl<'self, O:DataFlowOperator> PropagationContext<'self, O> {
  369      fn tcx(&self) -> ty::ctxt {
  370          self.dfcx.tcx
  371      }
  372  
  373      fn walk_block(&mut self,
  374                    blk: &ast::Block,
  375                    in_out: &mut [uint],
  376                    loop_scopes: &mut ~[LoopScope]) {
  377          debug!("DataFlowContext::walk_block(blk.id=%?, in_out=%s)",
  378                 blk.id, bits_to_str(reslice(in_out)));
  379  
  380          self.merge_with_entry_set(blk.id, in_out);
  381  
  382          for &stmt in blk.stmts.iter() {
  383              self.walk_stmt(stmt, in_out, loop_scopes);
  384          }
  385  
  386          self.walk_opt_expr(blk.expr, in_out, loop_scopes);
  387  
  388          self.dfcx.apply_gen_kill(blk.id, in_out);
  389      }
  390  
  391      fn walk_stmt(&mut self,
  392                   stmt: @ast::Stmt,
  393                   in_out: &mut [uint],
  394                   loop_scopes: &mut ~[LoopScope]) {
  395          match stmt.node {
  396              ast::StmtDecl(decl, _) => {
  397                  self.walk_decl(decl, in_out, loop_scopes);
  398              }
  399  
  400              ast::StmtExpr(expr, _) | ast::StmtSemi(expr, _) => {
  401                  self.walk_expr(expr, in_out, loop_scopes);
  402              }
  403  
  404              ast::StmtMac(*) => {
  405                  self.tcx().sess.span_bug(stmt.span, "unexpanded macro");
  406              }
  407          }
  408      }
  409  
  410      fn walk_decl(&mut self,
  411                   decl: @ast::Decl,
  412                   in_out: &mut [uint],
  413                   loop_scopes: &mut ~[LoopScope]) {
  414          match decl.node {
  415              ast::DeclLocal(local) => {
  416                  self.walk_opt_expr(local.init, in_out, loop_scopes);
  417                  self.walk_pat(local.pat, in_out, loop_scopes);
  418              }
  419  
  420              ast::DeclItem(_) => {}
  421          }
  422      }
  423  
  424      fn walk_expr(&mut self,
  425                   expr: @ast::Expr,
  426                   in_out: &mut [uint],
  427                   loop_scopes: &mut ~[LoopScope]) {
  428          debug!("DataFlowContext::walk_expr(expr=%s, in_out=%s)",
  429                 expr.repr(self.dfcx.tcx), bits_to_str(reslice(in_out)));
  430  
  431          self.merge_with_entry_set(expr.id, in_out);
  432  
  433          match expr.node {
  434              ast::ExprFnBlock(ref decl, ref body) => {
  435                  if self.dfcx.oper.walk_closures() {
  436                      // In the absence of once fns, we must assume that
  437                      // every function body will execute more than
  438                      // once. Thus we treat every function body like a
  439                      // loop.
  440                      //
  441                      // What is subtle and a bit tricky, also, is how
  442                      // to deal with the "output" bits---that is, what
  443                      // do we consider to be the successor of a
  444                      // function body, given that it could be called
  445                      // from any point within its lifetime? What we do
  446                      // is to add their effects immediately as of the
  447                      // point of creation. Of course we have to ensure
  448                      // that this is sound for the analyses which make
  449                      // use of dataflow.
  450                      //
  451                      // In the case of the initedness checker (which
  452                      // does not currently use dataflow, but I hope to
  453                      // convert at some point), we will simply not walk
  454                      // closures at all, so it's a moot point.
  455                      //
  456                      // In the case of the borrow checker, this means
  457                      // the loans which would be created by calling a
  458                      // function come into effect immediately when the
  459                      // function is created. This is guaranteed to be
  460                      // earlier than the point at which the loan
  461                      // actually comes into scope (which is the point
  462                      // at which the closure is *called*). Because
  463                      // loans persist until the scope of the loans is
  464                      // exited, it is always a safe approximation to
  465                      // have a loan begin earlier than it actually will
  466                      // at runtime, so this should be sound.
  467                      //
  468                      // We stil have to be careful in the region
  469                      // checker and borrow checker to treat function
  470                      // bodies like loops, which implies some
  471                      // limitations. For example, a closure cannot root
  472                      // a managed box for longer than its body.
  473                      //
  474                      // General control flow looks like this:
  475                      //
  476                      //  +- (expr) <----------+
  477                      //  |    |               |
  478                      //  |    v               |
  479                      //  |  (body) -----------+--> (exit)
  480                      //  |    |               |
  481                      //  |    + (break/loop) -+
  482                      //  |                    |
  483                      //  +--------------------+
  484                      //
  485                      // This is a bit more conservative than a loop.
  486                      // Note that we must assume that even after a
  487                      // `break` occurs (e.g., in a `for` loop) that the
  488                      // closure may be reinvoked.
  489                      //
  490                      // One difference from other loops is that `loop`
  491                      // and `break` statements which target a closure
  492                      // both simply add to the `break_bits`.
  493  
  494                      // func_bits represents the state when the function
  495                      // returns
  496                      let mut func_bits = reslice(in_out).to_owned();
  497  
  498                      loop_scopes.push(LoopScope {
  499                          loop_id: expr.id,
  500                          break_bits: reslice(in_out).to_owned()
  501                      });
  502                      for input in decl.inputs.iter() {
  503                          self.walk_pat(input.pat, func_bits, loop_scopes);
  504                      }
  505                      self.walk_block(body, func_bits, loop_scopes);
  506  
  507                      // add the bits from any early return via `break`,
  508                      // `continue`, or `return` into `func_bits`
  509                      let loop_scope = loop_scopes.pop();
  510                      join_bits(&self.dfcx.oper, loop_scope.break_bits, func_bits);
  511  
  512                      // add `func_bits` to the entry bits for `expr`,
  513                      // since we must assume the function may be called
  514                      // more than once
  515                      self.add_to_entry_set(expr.id, reslice(func_bits));
  516  
  517                      // the final exit bits include whatever was present
  518                      // in the original, joined with the bits from the function
  519                      join_bits(&self.dfcx.oper, func_bits, in_out);
  520                  }
  521              }
  522  
  523              ast::ExprIf(cond, ref then, els) => {
  524                  //
  525                  //     (cond)
  526                  //       |
  527                  //       v
  528                  //      ( )
  529                  //     /   \
  530                  //    |     |
  531                  //    v     v
  532                  //  (then)(els)
  533                  //    |     |
  534                  //    v     v
  535                  //   (  succ  )
  536                  //
  537                  self.walk_expr(cond, in_out, loop_scopes);
  538  
  539                  let mut then_bits = reslice(in_out).to_owned();
  540                  self.walk_block(then, then_bits, loop_scopes);
  541  
  542                  self.walk_opt_expr(els, in_out, loop_scopes);
  543                  join_bits(&self.dfcx.oper, then_bits, in_out);
  544              }
  545  
  546              ast::ExprWhile(cond, ref blk) => {
  547                  //
  548                  //     (expr) <--+
  549                  //       |       |
  550                  //       v       |
  551                  //  +--(cond)    |
  552                  //  |    |       |
  553                  //  |    v       |
  554                  //  v  (blk) ----+
  555                  //       |
  556                  //    <--+ (break)
  557                  //
  558  
  559                  self.walk_expr(cond, in_out, loop_scopes);
  560  
  561                  let mut body_bits = reslice(in_out).to_owned();
  562                  loop_scopes.push(LoopScope {
  563                      loop_id: expr.id,
  564                      break_bits: reslice(in_out).to_owned()
  565                  });
  566                  self.walk_block(blk, body_bits, loop_scopes);
  567                  self.add_to_entry_set(expr.id, body_bits);
  568                  let new_loop_scope = loop_scopes.pop();
  569                  copy_bits(new_loop_scope.break_bits, in_out);
  570              }
  571  
  572              ast::ExprForLoop(*) => fail!("non-desugared expr_for_loop"),
  573  
  574              ast::ExprLoop(ref blk, _) => {
  575                  //
  576                  //     (expr) <--+
  577                  //       |       |
  578                  //       v       |
  579                  //     (blk) ----+
  580                  //       |
  581                  //    <--+ (break)
  582                  //
  583  
  584                  let mut body_bits = reslice(in_out).to_owned();

  586                  loop_scopes.push(LoopScope {
  587                      loop_id: expr.id,
  588                      break_bits: reslice(in_out).to_owned()
  589                  });
  590                  self.walk_block(blk, body_bits, loop_scopes);
  591                  self.add_to_entry_set(expr.id, body_bits);
  592  
  593                  let new_loop_scope = loop_scopes.pop();
  594                  assert_eq!(new_loop_scope.loop_id, expr.id);
  595                  copy_bits(new_loop_scope.break_bits, in_out);
  596              }

  598              ast::ExprMatch(discr, ref arms) => {
  599                  //
  600                  //    (discr)
  601                  //     / | \
  602                  //    |  |  |
  603                  //    v  v  v
  604                  //   (..arms..)
  605                  //    |  |  |
  606                  //    v  v  v
  607                  //   (  succ  )
  608                  //
  609                  //
  610                  self.walk_expr(discr, in_out, loop_scopes);
  611  
  612                  let mut guards = reslice(in_out).to_owned();
  613  
  614                  // We know that exactly one arm will be taken, so we
  615                  // can start out with a blank slate and just union
  616                  // together the bits from each arm:
  617                  self.reset(in_out);
  618  
  619                  for arm in arms.iter() {
  620                      // in_out reflects the discr and all guards to date
  621                      self.walk_opt_expr(arm.guard, guards, loop_scopes);
  622  
  623                      // determine the bits for the body and then union
  624                      // them into `in_out`, which reflects all bodies to date
  625                      let mut body = reslice(guards).to_owned();
  626                      self.walk_pat_alternatives(arm.pats, body, loop_scopes);
  627                      self.walk_block(&arm.body, body, loop_scopes);
  628                      join_bits(&self.dfcx.oper, body, in_out);
  629                  }
  630              }
  631  
  632              ast::ExprRet(o_e) => {

  635              }

  746      fn pop_scopes(&mut self,
  747                    from_expr: @ast::Expr,
  748                    to_scope: &mut LoopScope,
  749                    in_out: &mut [uint]) {
  750          //! Whenever you have a `break` or a `loop` statement, flow
  751          //! exits through any number of enclosing scopes on its
  752          //! way to the new destination. This function applies the kill
  753          //! sets of those enclosing scopes to `in_out` (those kill sets
  754          //! concern items that are going out of scope).
  755  
  756          let tcx = self.tcx();
  757          let region_maps = tcx.region_maps;
  758  
  759          debug!("pop_scopes(from_expr=%s, to_scope=%?, in_out=%s)",
  760                 from_expr.repr(tcx), to_scope.loop_id,
  761                 bits_to_str(reslice(in_out)));
  762  
  763          let mut id = from_expr.id;
  764          while id != to_scope.loop_id {
  765              self.dfcx.apply_kill(id, in_out);
  766  
  767              match region_maps.opt_encl_scope(id) {
  768                  Some(i) => { id = i; }
  769                  None => {
  770                      tcx.sess.span_bug(
  771                          from_expr.span,
  772                          fmt!("pop_scopes(from_expr=%s, to_scope=%?) \
  773                                to_scope does not enclose from_expr",
  774                               from_expr.repr(tcx), to_scope.loop_id));
  775                  }
  776              }
  777          }
  778      }

  780      fn break_from_to(&mut self,
  781                       from_expr: @ast::Expr,
  782                       to_scope: &mut LoopScope,
  783                       in_out: &mut [uint]) {
  784          self.pop_scopes(from_expr, to_scope, in_out);
  785          self.dfcx.apply_kill(from_expr.id, in_out);
  786          join_bits(&self.dfcx.oper, reslice(in_out), to_scope.break_bits);
  787          debug!("break_from_to(from_expr=%s, to_scope=%?) final break_bits=%s",
  788                 from_expr.repr(self.tcx()),
  789                 to_scope.loop_id,
  790                 bits_to_str(reslice(in_out)));
  791      }

  832      fn walk_pat(&mut self,
  833                  pat: @ast::Pat,
  834                  in_out: &mut [uint],

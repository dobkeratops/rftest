  12  /*!
  13   * A module for propagating forward dataflow information. The analysis
  14   * assumes that the items to be propagated can be represented as bits
  15   * and thus uses bitvectors. Your job is simply to specify the so-called
  16   * GEN and KILL bits for each expression.
  17   */

  32  #[deriving(Clone)]
  33  pub struct DataFlowContext<'a, O> {
  34      tcx: &'a ty::ctxt,
  35      method_map: typeck::MethodMap,
  36  
  37      /// the data flow operator
  38      oper: O,
  39  
  40      /// number of bits to propagate per id
  41      bits_per_id: uint,
  42  
  43      /// number of words we will use to store bits_per_id.
  44      /// equal to bits_per_id/uint::BITS rounded up.
  45      words_per_id: uint,
  46  
  47      // mapping from node to bitset index.
  48      nodeid_to_bitset: NodeMap<uint>,
  49  
  50      // Bit sets per id.  The following three fields (`gens`, `kills`,
  51      // and `on_entry`) all have the same structure. For each id in
  52      // `id_range`, there is a range of words equal to `words_per_id`.
  53      // So, to access the bits for any given id, you take a slice of
  54      // the full vector (see the method `compute_id_range()`).
  55  
  56      /// bits generated as we exit the scope `id`. Updated by `add_gen()`.
  57      gens: Vec<uint>,
  58  
  59      /// bits killed as we exit the scope `id`. Updated by `add_kill()`.
  60      kills: Vec<uint>,
  61  
  62      /// bits that are valid on entry to the scope `id`. Updated by
  63      /// `propagate()`.
  64      on_entry: Vec<uint>,
  65  }

  74  }

  86  impl<'a, O:DataFlowOperator> pprust::PpAnn for DataFlowContext<'a, O> {
  87      fn pre(&self,
  88             ps: &mut pprust::State,
  89             node: pprust::AnnNode) -> io::IoResult<()> {
  90          let id = match node {
  91              pprust::NodeExpr(expr) => expr.id,
  92              pprust::NodeBlock(blk) => blk.id,
  93              pprust::NodeItem(_) => 0,
  94              pprust::NodePat(pat) => pat.id
  95          };
  96  
  97          if self.nodeid_to_bitset.contains_key(&id) {
  98              let (start, end) = self.compute_id_range_frozen(id);
  99              let on_entry = self.on_entry.slice(start, end);
 100              let entry_str = bits_to_str(on_entry);
 101  
 102              let gens = self.gens.slice(start, end);
 103              let gens_str = if gens.iter().any(|&u| u != 0) {
 104                  format!(" gen: {}", bits_to_str(gens))
 105              } else {
 106                  "".to_owned()
 107              };
 108  
 109              let kills = self.kills.slice(start, end);
 110              let kills_str = if kills.iter().any(|&u| u != 0) {
 111                  format!(" kill: {}", bits_to_str(kills))
 112              } else {
 113                  "".to_owned()
 114              };
 115  
 116              try!(ps.synth_comment(format!("id {}: {}{}{}", id, entry_str,
 117                                            gens_str, kills_str)));
 118              try!(pp::space(&mut ps.s));
 119          }
 120          Ok(())
 121      }
 122  }

 174      fn apply_gen_kill(&mut self, id: ast::NodeId, bits: &mut [uint]) {
 175          //! Applies the gen and kill sets for `id` to `bits`
 176  
 177          debug!("apply_gen_kill(id={:?}, bits={}) [before]",
 178                 id, mut_bits_to_str(bits));
 179          let (start, end) = self.compute_id_range(id);
 180          let gens = self.gens.slice(start, end);
 181          bitwise(bits, gens, |a, b| a | b);
 182          let kills = self.kills.slice(start, end);
 183          bitwise(bits, kills, |a, b| a & !b);
 184  
 185          debug!("apply_gen_kill(id={:?}, bits={}) [after]",
 186                 id, mut_bits_to_str(bits));
 187      }

 199      fn compute_id_range_frozen(&self, id: ast::NodeId) -> (uint, uint) {
 200          let n = *self.nodeid_to_bitset.get(&id);
 201          let start = n * self.words_per_id;
 202          let end = start + self.words_per_id;
 203          (start, end)
 204      }
 205  
 206      fn compute_id_range(&mut self, id: ast::NodeId) -> (uint, uint) {
 207          let mut expanded = false;
 208          let len = self.nodeid_to_bitset.len();
 209          let n = self.nodeid_to_bitset.find_or_insert_with(id, |_| {
 210              expanded = true;
 211              len
 212          });
 213          if expanded {
 214              let entry = if self.oper.initial_value() { uint::MAX } else {0};
 215              for _ in range(0, self.words_per_id) {
 216                  self.gens.push(0);
 217                  self.kills.push(0);
 218                  self.on_entry.push(entry);
 219              }
 220          }
 221          let start = *n * self.words_per_id;
 222          let end = start + self.words_per_id;
 223  
 224          assert!(start < self.gens.len());
 225          assert!(end <= self.gens.len());
 226          assert!(self.gens.len() == self.kills.len());
 227          assert!(self.gens.len() == self.on_entry.len());
 228  
 229          (start, end)
 230      }

 233      pub fn each_bit_on_entry_frozen(&self,
 234                                      id: ast::NodeId,
 235                                      f: |uint| -> bool)
 236                                      -> bool {
 237          //! Iterates through each bit that is set on entry to `id`.
 238          //! Only useful after `propagate()` has been called.
 239          if !self.nodeid_to_bitset.contains_key(&id) {
 240              return true;
 241          }
 242          let (start, end) = self.compute_id_range_frozen(id);
 243          let on_entry = self.on_entry.slice(start, end);
 244          debug!("each_bit_on_entry_frozen(id={:?}, on_entry={})",
 245                 id, bits_to_str(on_entry));
 246          self.each_bit(on_entry, f)
 247      }

 249      pub fn each_gen_bit_frozen(&self, id: ast::NodeId, f: |uint| -> bool)
 250                                 -> bool {
 251          //! Iterates through each bit in the gen set for `id`.
 252          if !self.nodeid_to_bitset.contains_key(&id) {
 253              return true;
 254          }
 255          let (start, end) = self.compute_id_range_frozen(id);
 256          let gens = self.gens.slice(start, end);
 257          debug!("each_gen_bit(id={:?}, gens={})",
 258                 id, bits_to_str(gens));
 259          self.each_bit(gens, f)
 260      }

 262      fn each_bit(&self, words: &[uint], f: |uint| -> bool) -> bool {
 263          //! Helper for iterating over the bits in a bit set.
 264  
 265          for (word_index, &word) in words.iter().enumerate() {
 266              if word != 0 {
 267                  let base_index = word_index * uint::BITS;
 268                  for offset in range(0u, uint::BITS) {
 269                      let bit = 1 << offset;
 270                      if (word & bit) != 0 {
 271                          // NB: we round up the total number of bits
 272                          // that we store in any given bit set so that
 273                          // it is an even multiple of uint::BITS.  This
 274                          // means that there may be some stray bits at
 275                          // the end that do not correspond to any
 276                          // actual value.  So before we callback, check
 277                          // whether the bit_index is greater than the
 278                          // actual value the user specified and stop
 279                          // iterating if so.
 280                          let bit_index = base_index + offset;
 281                          if bit_index >= self.bits_per_id {
 282                              return true;
 283                          } else if !f(bit_index) {
 284                              return false;
 285                          }
 286                      }
 287                  }
 288              }
 289          }
 290          return true;
 291      }
 292  }

 294  impl<'a, O:DataFlowOperator+Clone+'static> DataFlowContext<'a, O> {
 295  //                          ^^^^^^^^^^^^^ only needed for pretty printing
 296      pub fn propagate(&mut self, blk: &ast::Block) {
 297          //! Performs the data flow analysis.
 298  
 299          if self.bits_per_id == 0 {
 300              // Optimize the surprisingly common degenerate case.
 301              return;
 302          }
 303  
 304          {
 305              let mut propcx = PropagationContext {
 306                  dfcx: &mut *self,
 307                  changed: true
 308              };
 309  
 310              let mut temp = Vec::from_elem(self.words_per_id, 0u);
 311              let mut loop_scopes = Vec::new();
 312  
 313              while propcx.changed {
 314                  propcx.changed = false;
 315                  propcx.reset(temp.as_mut_slice());
 316                  propcx.walk_block(blk, temp.as_mut_slice(), &mut loop_scopes);
 317              }
 318          }
 319  
 320          debug!("Dataflow result:");
 321          debug!("{}", {
 322              self.pretty_print_to(~io::stderr(), blk).unwrap();
 323              ""
 324          });
 325      }
 326  
 327      fn pretty_print_to(&self, wr: ~io::Writer,
 328                         blk: &ast::Block) -> io::IoResult<()> {
 329          let mut ps = pprust::rust_printer_annotated(wr, self);
 330          try!(ps.cbox(pprust::indent_unit));
 331          try!(ps.ibox(0u));
 332          try!(ps.print_block(blk));
 333          pp::eof(&mut ps.s)
 334      }
 335  }

 337  impl<'a, 'b, O:DataFlowOperator> PropagationContext<'a, 'b, O> {
 338      fn tcx(&self) -> &'b ty::ctxt {
 339          self.dfcx.tcx
 340      }
 341  
 342      fn walk_block(&mut self,
 343                    blk: &ast::Block,
 344                    in_out: &mut [uint],
 345                    loop_scopes: &mut Vec<LoopScope> ) {
 346          debug!("DataFlowContext::walk_block(blk.id={}, in_out={})",
 347                 blk.id, bits_to_str(in_out));
 348  
 349          self.merge_with_entry_set(blk.id, in_out);
 350  
 351          for &stmt in blk.stmts.iter() {
 352              self.walk_stmt(stmt, in_out, loop_scopes);
 353          }
 354  
 355          self.walk_opt_expr(blk.expr, in_out, loop_scopes);
 356  
 357          self.dfcx.apply_gen_kill(blk.id, in_out);
 358      }
 359  
 360      fn walk_stmt(&mut self,
 361                   stmt: @ast::Stmt,
 362                   in_out: &mut [uint],
 363                   loop_scopes: &mut Vec<LoopScope> ) {
 364          match stmt.node {
 365              ast::StmtDecl(decl, _) => {
 366                  self.walk_decl(decl, in_out, loop_scopes);
 367              }
 368  
 369              ast::StmtExpr(expr, _) | ast::StmtSemi(expr, _) => {
 370                  self.walk_expr(expr, in_out, loop_scopes);
 371              }
 372  
 373              ast::StmtMac(..) => {
 374                  self.tcx().sess.span_bug(stmt.span, "unexpanded macro");
 375              }
 376          }
 377      }
 378  
 379      fn walk_decl(&mut self,
 380                   decl: @ast::Decl,
 381                   in_out: &mut [uint],
 382                   loop_scopes: &mut Vec<LoopScope> ) {
 383          match decl.node {
 384              ast::DeclLocal(local) => {
 385                  self.walk_opt_expr(local.init, in_out, loop_scopes);
 386                  self.walk_pat(local.pat, in_out, loop_scopes);
 387              }
 388  
 389              ast::DeclItem(_) => {}
 390          }
 391      }
 392  
 393      fn walk_expr(&mut self,
 394                   expr: &ast::Expr,
 395                   in_out: &mut [uint],
 396                   loop_scopes: &mut Vec<LoopScope> ) {
 397          debug!("DataFlowContext::walk_expr(expr={}, in_out={})",
 398                 expr.repr(self.dfcx.tcx), bits_to_str(in_out));
 399  
 400          self.merge_with_entry_set(expr.id, in_out);
 401  
 402          match expr.node {
 403              ast::ExprFnBlock(..) | ast::ExprProc(..) => {
 404              }
 405  
 406              ast::ExprIf(cond, then, els) => {
 407                  //
 408                  //     (cond)
 409                  //       |
 410                  //       v
 411                  //      ( )
 412                  //     /   \
 413                  //    |     |
 414                  //    v     v
 415                  //  (then)(els)
 416                  //    |     |
 417                  //    v     v
 418                  //   (  succ  )
 419                  //
 420                  self.walk_expr(cond, in_out, loop_scopes);
 421  
 422                  let mut then_bits = in_out.to_owned();
 423                  self.walk_block(then, then_bits, loop_scopes);
 424  
 425                  self.walk_opt_expr(els, in_out, loop_scopes);
 426                  join_bits(&self.dfcx.oper, then_bits, in_out);
 427              }
 428  
 429              ast::ExprWhile(cond, blk) => {
 430                  //
 431                  //     (expr) <--+
 432                  //       |       |
 433                  //       v       |
 434                  //  +--(cond)    |
 435                  //  |    |       |
 436                  //  |    v       |
 437                  //  v  (blk) ----+
 438                  //       |
 439                  //    <--+ (break)
 440                  //
 441  
 442                  self.walk_expr(cond, in_out, loop_scopes);
 443  
 444                  let mut body_bits = in_out.to_owned();
 445                  loop_scopes.push(LoopScope {
 446                      loop_id: expr.id,

 629      fn pop_scopes(&mut self,
 630                    from_expr: &ast::Expr,
 631                    to_scope: &mut LoopScope,
 632                    in_out: &mut [uint]) {
 633          //! Whenever you have a `break` or a `loop` statement, flow
 634          //! exits through any number of enclosing scopes on its
 635          //! way to the new destination. This function applies the kill
 636          //! sets of those enclosing scopes to `in_out` (those kill sets
 637          //! concern items that are going out of scope).
 638  
 639          let tcx = self.tcx();
 640  
 641          debug!("pop_scopes(from_expr={}, to_scope={:?}, in_out={})",
 642                 from_expr.repr(tcx), to_scope.loop_id,
 643                 bits_to_str(in_out));
 644  
 645          let mut id = from_expr.id;
 646          while id != to_scope.loop_id {
 647              self.dfcx.apply_kill(id, in_out);
 648  
 649              match tcx.region_maps.opt_encl_scope(id) {
 650                  Some(i) => { id = i; }
 651                  None => {
 652                      tcx.sess.span_bug(
 653                          from_expr.span,
 654                          format!("pop_scopes(from_expr={}, to_scope={:?}) \
 655                                to_scope does not enclose from_expr",
 656                               from_expr.repr(tcx), to_scope.loop_id));
 657                  }
 658              }
 659          }
 660      }

 664                       to_scope: &mut LoopScope,

 666          self.pop_scopes(from_expr, to_scope, in_out);
 667          self.dfcx.apply_kill(from_expr.id, in_out);
 668          join_bits(&self.dfcx.oper,
 669                    in_out,
 670                    to_scope.break_bits.as_mut_slice());
 671          debug!("break_from_to(from_expr={}, to_scope={}) final break_bits={}",
 672                 from_expr.repr(self.tcx()),
 673                 to_scope.loop_id,

 675      }

 728      fn walk_pat_alternatives(&mut self,
 729                               pats: &[@ast::Pat],
 730                               in_out: &mut [uint],
 731                               loop_scopes: &mut Vec<LoopScope> ) {
 732          if pats.len() == 1 {
 733              // Common special case:
 734              return self.walk_pat(pats[0], in_out, loop_scopes);
 735          }
 736  
 737          // In the general case, the patterns in `pats` are
 738          // alternatives, so we must treat this like an N-way select
 739          // statement.
 740          let initial_state = in_out.to_owned();
 741          for &pat in pats.iter() {
 742              let mut temp = initial_state.clone();
 743              self.walk_pat(pat, temp, loop_scopes);
 744              join_bits(&self.dfcx.oper, temp, in_out);
 745          }
 746      }

 748      fn find_scope<'a,'b>(
 749                    &self,
 750                    expr: &ast::Expr,
 751                    label: Option<ast::Ident>,
 752                    loop_scopes: &'a mut Vec<LoopScope<'b>>)
 753                    -> &'a mut LoopScope<'b> {
 754          let index = match label {
 755              None => {
 756                  let len = loop_scopes.len();
 757                  len - 1
 758              }
 759  
 760              Some(_) => {
 761                  match self.tcx().def_map.borrow().find(&expr.id) {
 762                      Some(&ast::DefLabel(loop_id)) => {
 763                          match loop_scopes.iter().position(|l| l.loop_id == loop_id) {
 764                              Some(i) => i,
 765                              None => {
 766                                  self.tcx().sess.span_bug(
 767                                      expr.span,
 768                                      format!("no loop scope for id {:?}", loop_id));
 769                              }
 770                          }
 771                      }
 772  
 773                      r => {
 774                          self.tcx().sess.span_bug(
 775                              expr.span,
 776                              format!("bad entry `{:?}` in def_map for label", r));
 777                      }
 778                  }
 779              }
 780          };
 781  
 782          loop_scopes.get_mut(index)
 783      }

 834  fn bits_to_str(words: &[uint]) -> ~str {
 835      let mut result = StrBuf::new();
 836      let mut sep = '[';
 837  
 838      // Note: this is a little endian printout of bytes.
 839  
 840      for &word in words.iter() {
 841          let mut v = word;
 842          for _ in range(0u, uint::BYTES) {
 843              result.push_char(sep);
 844              result.push_str(format!("{:02x}", v & 0xFF));
 845              v >>= 8;
 846              sep = '-';
 847          }
 848      }
 849      result.push_char(']');
 850      return result.into_owned();
 851  }

 854      bitwise(out_vec, in_vec, |_, b| b)

 857  fn join_bits<O:DataFlowOperator>(oper: &O,
 858                                   in_vec: &[uint],
 859                                   out_vec: &mut [uint]) -> bool {
 860      bitwise(out_vec, in_vec, |a, b| oper.join(a, b))
 861  }

 863  #[inline]
 864  fn bitwise(out_vec: &mut [uint], in_vec: &[uint], op: |uint, uint| -> uint)
 865             -> bool {
 866      assert_eq!(out_vec.len(), in_vec.len());
 867      let mut changed = false;
 868      for (out_elt, in_elt) in out_vec.mut_iter().zip(in_vec.iter()) {
 869          let old_val = *out_elt;
 870          let new_val = op(old_val, *in_elt);
 871          *out_elt = new_val;
 872          changed |= old_val != new_val;
 873      }
 874      changed
 875  }

 877  fn set_bit(words: &mut [uint], bit: uint) -> bool {
 878      debug!("set_bit: words={} bit={}",
 879             mut_bits_to_str(words), bit_str(bit));
 880      let word = bit / uint::BITS;
 881      let bit_in_word = bit % uint::BITS;
 882      let bit_mask = 1 << bit_in_word;
 883      debug!("word={} bit_in_word={} bit_mask={}", word, bit_in_word, word);
 884      let oldv = words[word];
 885      let newv = oldv | bit_mask;
 886      words[word] = newv;
 887      oldv != newv
 888  }

 890  fn bit_str(bit: uint) -> ~str {

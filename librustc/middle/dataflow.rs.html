  12  /*!
  13   * A module for propagating forward dataflow information. The analysis
  14   * assumes that the items to be propagated can be represented as bits
  15   * and thus uses bitvectors. Your job is simply to specify the so-called
  16   * GEN and KILL bits for each expression.
  17   */

  32  #[deriving(Clone)]
  33  pub struct DataFlowContext<'a, O> {
  34      priv tcx: &'a ty::ctxt,
  35      priv method_map: typeck::MethodMap,
  36  
  37      /// the data flow operator
  38      priv oper: O,
  39  
  40      /// number of bits to propagate per id
  41      priv bits_per_id: uint,
  42  
  43      /// number of words we will use to store bits_per_id.
  44      /// equal to bits_per_id/uint::BITS rounded up.
  45      priv words_per_id: uint,
  46  
  47      // mapping from node to bitset index.
  48      priv nodeid_to_bitset: NodeMap<uint>,
  49  
  50      // Bit sets per id.  The following three fields (`gens`, `kills`,
  51      // and `on_entry`) all have the same structure. For each id in
  52      // `id_range`, there is a range of words equal to `words_per_id`.
  53      // So, to access the bits for any given id, you take a slice of
  54      // the full vector (see the method `compute_id_range()`).
  55  
  56      /// bits generated as we exit the scope `id`. Updated by `add_gen()`.
  57      priv gens: Vec<uint> ,
  58  
  59      /// bits killed as we exit the scope `id`. Updated by `add_kill()`.
  60      priv kills: Vec<uint> ,
  61  
  62      /// bits that are valid on entry to the scope `id`. Updated by
  63      /// `propagate()`.
  64      priv on_entry: Vec<uint> }
  65  
  66  /// Parameterization for the precise form of data flow that is used.
  67  pub trait DataFlowOperator {
  68      /// Specifies the initial value for each bit in the `on_entry` set
  69      fn initial_value(&self) -> bool;
  70  
  71      /// Joins two predecessor bits together, typically either `|` or `&`
  72      fn join(&self, succ: uint, pred: uint) -> uint;
  73  }

  75  struct PropagationContext<'a, 'b, O> {
  76      dfcx: &'a mut DataFlowContext<'b, O>,
  77      changed: bool
  78  }

  81      loop_id: ast::NodeId,
  82      break_bits: Vec<uint>
  83  }

  85  impl<'a, O:DataFlowOperator> pprust::PpAnn for DataFlowContext<'a, O> {
  86      fn pre(&self,
  87             ps: &mut pprust::State,
  88             node: pprust::AnnNode) -> io::IoResult<()> {
  89          let id = match node {
  90              pprust::NodeExpr(expr) => expr.id,
  91              pprust::NodeBlock(blk) => blk.id,
  92              pprust::NodeItem(_) => 0,
  93              pprust::NodePat(pat) => pat.id
  94          };
  95  
  96          if self.nodeid_to_bitset.contains_key(&id) {
  97              let (start, end) = self.compute_id_range_frozen(id);
  98              let on_entry = self.on_entry.slice(start, end);
  99              let entry_str = bits_to_str(on_entry);
 100  
 101              let gens = self.gens.slice(start, end);
 102              let gens_str = if gens.iter().any(|&u| u != 0) {
 103                  format!(" gen: {}", bits_to_str(gens))
 104              } else {
 105                  ~""
 106              };
 107  
 108              let kills = self.kills.slice(start, end);
 109              let kills_str = if kills.iter().any(|&u| u != 0) {
 110                  format!(" kill: {}", bits_to_str(kills))
 111              } else {
 112                  ~""
 113              };
 114  
 115              try!(ps.synth_comment(format!("id {}: {}{}{}", id, entry_str,
 116                                            gens_str, kills_str)));
 117              try!(pp::space(&mut ps.s));
 118          }
 119          Ok(())
 120      }
 121  }

 133

 139              tcx: tcx,
 140              method_map: method_map,
 141              words_per_id: words_per_id,
 142              nodeid_to_bitset: NodeMap::new(),
 143              bits_per_id: bits_per_id,
 144              oper: oper,
 145              gens: gens,
 146              kills: kills,
 147              on_entry: on_entry
 148          }
 149      }

 151      pub fn add_gen(&mut self, id: ast::NodeId, bit: uint) {
 152          //! Indicates that `id` generates `bit`
 153  
 154          debug!("add_gen(id={:?}, bit={:?})", id, bit);
 155          let (start, end) = self.compute_id_range(id);
 156          {
 157              let gens = self.gens.mut_slice(start, end);
 158              set_bit(gens, bit);
 159          }
 160      }

 167          {
 168              let kills = self.kills.mut_slice(start, end);
 169              set_bit(kills, bit);
 170          }
 171      }

 173      fn apply_gen_kill(&mut self, id: ast::NodeId, bits: &mut [uint]) {
 174          //! Applies the gen and kill sets for `id` to `bits`
 175  
 176          debug!("apply_gen_kill(id={:?}, bits={}) [before]",
 177                 id, mut_bits_to_str(bits));
 178          let (start, end) = self.compute_id_range(id);
 179          let gens = self.gens.slice(start, end);
 180          bitwise(bits, gens, |a, b| a | b);
 181          let kills = self.kills.slice(start, end);
 182          bitwise(bits, kills, |a, b| a & !b);
 183  
 184          debug!("apply_gen_kill(id={:?}, bits={}) [after]",
 185                 id, mut_bits_to_str(bits));
 186      }

 188      fn apply_kill(&mut self, id: ast::NodeId, bits: &mut [uint]) {
 189          debug!("apply_kill(id={:?}, bits={}) [before]",
 190                 id, mut_bits_to_str(bits));
 191          let (start, end) = self.compute_id_range(id);
 192          let kills = self.kills.slice(start, end);
 193          bitwise(bits, kills, |a, b| a & !b);
 194          debug!("apply_kill(id={:?}, bits={}) [after]",
 195                 id, mut_bits_to_str(bits));
 196      }

 198      fn compute_id_range_frozen(&self, id: ast::NodeId) -> (uint, uint) {
 199          let n = *self.nodeid_to_bitset.get(&id);
 200          let start = n * self.words_per_id;
 201          let end = start + self.words_per_id;
 202          (start, end)
 203      }
 204  
 205      fn compute_id_range(&mut self, id: ast::NodeId) -> (uint, uint) {
 206          let mut expanded = false;
 207          let len = self.nodeid_to_bitset.len();
 208          let n = self.nodeid_to_bitset.find_or_insert_with(id, |_| {
 209              expanded = true;
 210              len
 211          });
 212          if expanded {
 213              let entry = if self.oper.initial_value() { uint::MAX } else {0};
 214              for _ in range(0, self.words_per_id) {
 215                  self.gens.push(0);
 216                  self.kills.push(0);
 217                  self.on_entry.push(entry);
 218              }
 219          }
 220          let start = *n * self.words_per_id;
 221          let end = start + self.words_per_id;
 222  
 223          assert!(start < self.gens.len());
 224          assert!(end <= self.gens.len());
 225          assert!(self.gens.len() == self.kills.len());
 226          assert!(self.gens.len() == self.on_entry.len());
 227  
 228          (start, end)
 229      }

 232      pub fn each_bit_on_entry_frozen(&self,
 233                                      id: ast::NodeId,
 234                                      f: |uint| -> bool)
 235                                      -> bool {
 236          //! Iterates through each bit that is set on entry to `id`.
 237          //! Only useful after `propagate()` has been called.
 238          if !self.nodeid_to_bitset.contains_key(&id) {
 239              return true;
 240          }
 241          let (start, end) = self.compute_id_range_frozen(id);
 242          let on_entry = self.on_entry.slice(start, end);
 243          debug!("each_bit_on_entry_frozen(id={:?}, on_entry={})",
 244                 id, bits_to_str(on_entry));
 245          self.each_bit(on_entry, f)
 246      }

 277              return true;
 278          }

 286      fn each_bit(&self, words: &[uint], f: |uint| -> bool) -> bool {
 287          //! Helper for iterating over the bits in a bit set.
 288  
 289          for (word_index, &word) in words.iter().enumerate() {
 290              if word != 0 {
 291                  let base_index = word_index * uint::BITS;
 292                  for offset in range(0u, uint::BITS) {
 293                      let bit = 1 << offset;
 294                      if (word & bit) != 0 {
 295                          // NB: we round up the total number of bits
 296                          // that we store in any given bit set so that
 297                          // it is an even multiple of uint::BITS.  This
 298                          // means that there may be some stray bits at
 299                          // the end that do not correspond to any
 300                          // actual value.  So before we callback, check
 301                          // whether the bit_index is greater than the
 302                          // actual value the user specified and stop
 303                          // iterating if so.
 304                          let bit_index = base_index + offset;
 305                          if bit_index >= self.bits_per_id {
 306                              return true;
 307                          } else if !f(bit_index) {
 308                              return false;
 309                          }
 310                      }
 311                  }
 312              }
 313          }
 314          return true;
 315      }
 316  }

 318  impl<'a, O:DataFlowOperator+Clone+'static> DataFlowContext<'a, O> {
 319  //                          ^^^^^^^^^^^^^ only needed for pretty printing
 320      pub fn propagate(&mut self, blk: &ast::Block) {
 321          //! Performs the data flow analysis.
 322  
 323          if self.bits_per_id == 0 {
 324              // Optimize the surprisingly common degenerate case.
 325              return;
 326          }
 327  
 328          {
 329              let mut propcx = PropagationContext {
 330                  dfcx: &mut *self,
 331                  changed: true
 332              };
 333  
 334              let mut temp = slice::from_elem(self.words_per_id, 0u);
 335              let mut loop_scopes = Vec::new();
 336  
 337              while propcx.changed {
 338                  propcx.changed = false;
 339                  propcx.reset(temp);
 340                  propcx.walk_block(blk, temp, &mut loop_scopes);
 341              }
 342          }
 343  
 344          debug!("Dataflow result:");
 345          debug!("{}", {
 346              self.pretty_print_to(~io::stderr(), blk).unwrap();
 347              ""
 348          });
 349      }
 350  
 351      fn pretty_print_to(&self, wr: ~io::Writer,
 352                         blk: &ast::Block) -> io::IoResult<()> {
 353          let mut ps = pprust::rust_printer_annotated(wr, self);
 354          try!(ps.cbox(pprust::indent_unit));
 355          try!(ps.ibox(0u));
 356          try!(ps.print_block(blk));
 357          pp::eof(&mut ps.s)
 358      }
 359  }

 361  impl<'a, 'b, O:DataFlowOperator> PropagationContext<'a, 'b, O> {
 362      fn tcx(&self) -> &'b ty::ctxt {
 363          self.dfcx.tcx
 364      }
 365  
 366      fn walk_block(&mut self,
 367                    blk: &ast::Block,
 368                    in_out: &mut [uint],
 369                    loop_scopes: &mut Vec<LoopScope> ) {
 370          debug!("DataFlowContext::walk_block(blk.id={}, in_out={})",
 371                 blk.id, bits_to_str(in_out));
 372  
 373          self.merge_with_entry_set(blk.id, in_out);
 374  
 375          for &stmt in blk.stmts.iter() {
 376              self.walk_stmt(stmt, in_out, loop_scopes);
 377          }
 378  
 379          self.walk_opt_expr(blk.expr, in_out, loop_scopes);
 380  
 381          self.dfcx.apply_gen_kill(blk.id, in_out);
 382      }
 383  
 384      fn walk_stmt(&mut self,
 385                   stmt: @ast::Stmt,
 386                   in_out: &mut [uint],
 387                   loop_scopes: &mut Vec<LoopScope> ) {
 388          match stmt.node {
 389              ast::StmtDecl(decl, _) => {
 390                  self.walk_decl(decl, in_out, loop_scopes);
 391              }
 392  
 393              ast::StmtExpr(expr, _) | ast::StmtSemi(expr, _) => {
 394                  self.walk_expr(expr, in_out, loop_scopes);
 395              }
 396  
 397              ast::StmtMac(..) => {
 398                  self.tcx().sess.span_bug(stmt.span, "unexpanded macro");
 399              }
 400          }
 401      }
 402  
 403      fn walk_decl(&mut self,
 404                   decl: @ast::Decl,
 405                   in_out: &mut [uint],
 406                   loop_scopes: &mut Vec<LoopScope> ) {
 407          match decl.node {
 408              ast::DeclLocal(local) => {
 409                  self.walk_opt_expr(local.init, in_out, loop_scopes);
 410                  self.walk_pat(local.pat, in_out, loop_scopes);
 411              }
 412  
 413              ast::DeclItem(_) => {}
 414          }
 415      }
 416  
 417      fn walk_expr(&mut self,
 418                   expr: &ast::Expr,
 419                   in_out: &mut [uint],
 420                   loop_scopes: &mut Vec<LoopScope> ) {
 421          debug!("DataFlowContext::walk_expr(expr={}, in_out={})",
 422                 expr.repr(self.dfcx.tcx), bits_to_str(in_out));
 423  
 424          self.merge_with_entry_set(expr.id, in_out);
 425  
 426          match expr.node {
 427              ast::ExprFnBlock(..) | ast::ExprProc(..) => {
 428              }
 429  
 430              ast::ExprIf(cond, then, els) => {
 431                  //
 432                  //     (cond)
 433                  //       |
 434                  //       v
 435                  //      ( )
 436                  //     /   \
 437                  //    |     |
 438                  //    v     v
 439                  //  (then)(els)
 440                  //    |     |
 441                  //    v     v
 442                  //   (  succ  )
 443                  //
 444                  self.walk_expr(cond, in_out, loop_scopes);

 653      fn pop_scopes(&mut self,
 654                    from_expr: &ast::Expr,
 655                    to_scope: &mut LoopScope,
 656                    in_out: &mut [uint]) {
 657          //! Whenever you have a `break` or a `loop` statement, flow
 658          //! exits through any number of enclosing scopes on its
 659          //! way to the new destination. This function applies the kill
 660          //! sets of those enclosing scopes to `in_out` (those kill sets
 661          //! concern items that are going out of scope).
 662  
 663          let tcx = self.tcx();
 664  
 665          debug!("pop_scopes(from_expr={}, to_scope={:?}, in_out={})",
 666                 from_expr.repr(tcx), to_scope.loop_id,
 667                 bits_to_str(in_out));
 668  
 669          let mut id = from_expr.id;
 670          while id != to_scope.loop_id {
 671              self.dfcx.apply_kill(id, in_out);
 672  
 673              match tcx.region_maps.opt_encl_scope(id) {
 674                  Some(i) => { id = i; }
 675                  None => {
 676                      tcx.sess.span_bug(
 677                          from_expr.span,
 678                          format!("pop_scopes(from_expr={}, to_scope={:?}) \
 679                                to_scope does not enclose from_expr",
 680                               from_expr.repr(tcx), to_scope.loop_id));
 681                  }
 682              }
 683          }
 684      }

 686      fn break_from_to(&mut self,
 687                       from_expr: &ast::Expr,
 688                       to_scope: &mut LoopScope,
 689                       in_out: &mut [uint]) {
 690          self.pop_scopes(from_expr, to_scope, in_out);
 691          self.dfcx.apply_kill(from_expr.id, in_out);
 692          join_bits(&self.dfcx.oper,
 693                    in_out,
 694                    to_scope.break_bits.as_mut_slice());
 695          debug!("break_from_to(from_expr={}, to_scope={}) final break_bits={}",
 696                 from_expr.repr(self.tcx()),
 697                 to_scope.loop_id,
 698                 bits_to_str(in_out));
 699      }

 752      fn walk_pat_alternatives(&mut self,
 753                               pats: &[@ast::Pat],
 754                               in_out: &mut [uint],
 755                               loop_scopes: &mut Vec<LoopScope> ) {
 756          if pats.len() == 1 {
 757              // Common special case:
 758              return self.walk_pat(pats[0], in_out, loop_scopes);
 759          }
 760  
 761          // In the general case, the patterns in `pats` are
 762          // alternatives, so we must treat this like an N-way select
 763          // statement.
 764          let initial_state = in_out.to_owned();
 765          for &pat in pats.iter() {
 766              let mut temp = initial_state.clone();
 767              self.walk_pat(pat, temp, loop_scopes);
 768              join_bits(&self.dfcx.oper, temp, in_out);
 769          }
 770      }

 772      fn find_scope<'a,'b>(
 773                    &self,
 774                    expr: &ast::Expr,
 775                    label: Option<ast::Ident>,
 776                    loop_scopes: &'a mut Vec<LoopScope<'b>>)
 777                    -> &'a mut LoopScope<'b> {
 778          let index = match label {
 779              None => {
 780                  let len = loop_scopes.len();
 781                  len - 1
 782              }
 783  
 784              Some(_) => {
 785                  match self.tcx().def_map.borrow().find(&expr.id) {
 786                      Some(&ast::DefLabel(loop_id)) => {
 787                          match loop_scopes.iter().position(|l| l.loop_id == loop_id) {
 788                              Some(i) => i,
 789                              None => {
 790                                  self.tcx().sess.span_bug(
 791                                      expr.span,
 792                                      format!("no loop scope for id {:?}", loop_id));
 793                              }
 794                          }
 795                      }
 796  
 797                      r => {
 798                          self.tcx().sess.span_bug(
 799                              expr.span,
 800                              format!("bad entry `{:?}` in def_map for label", r));
 801                      }
 802                  }
 803              }
 804          };
 805  
 806          loop_scopes.get_mut(index)
 807      }

 858  fn bits_to_str(words: &[uint]) -> ~str {
 859      let mut result = ~"";
 860      let mut sep = '[';
 861  
 862      // Note: this is a little endian printout of bytes.
 863  
 864      for &word in words.iter() {
 865          let mut v = word;
 866          for _ in range(0u, uint::BYTES) {
 867              result.push_char(sep);
 868              result.push_str(format!("{:02x}", v & 0xFF));
 869              v >>= 8;
 870              sep = '-';
 871          }
 872      }
 873      result.push_char(']');
 874      return result;
 875  }

 877  fn copy_bits(in_vec: &[uint], out_vec: &mut [uint]) -> bool {
 878      bitwise(out_vec, in_vec, |_, b| b)
 879  }

 881  fn join_bits<O:DataFlowOperator>(oper: &O,
 882                                   in_vec: &[uint],
 883                                   out_vec: &mut [uint]) -> bool {
 884      bitwise(out_vec, in_vec, |a, b| oper.join(a, b))
 885  }
 886  
 887  #[inline]
 888  fn bitwise(out_vec: &mut [uint], in_vec: &[uint], op: |uint, uint| -> uint)
 889             -> bool {
 890      assert_eq!(out_vec.len(), in_vec.len());
 891      let mut changed = false;
 892      for (out_elt, in_elt) in out_vec.mut_iter().zip(in_vec.iter()) {
 893          let old_val = *out_elt;

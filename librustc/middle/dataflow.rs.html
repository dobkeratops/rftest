   12 /*!
   13  * A module for propagating forward dataflow information. The analysis
   14  * assumes that the items to be propagated can be represented as bits
   15  * and thus uses bitvectors. Your job is simply to specify the so-called
   16  * GEN and KILL bits for each expression.
   17  */

   33 #[deriving(Clone)]
   34 pub struct DataFlowContext<O> {
   35     priv tcx: ty::ctxt,
   36     priv method_map: typeck::method_map,
   37 
   38     /// the data flow operator
   39     priv oper: O,
   40 
   41     /// number of bits to propagate per id
   42     priv bits_per_id: uint,
   43 
   44     /// number of words we will use to store bits_per_id.
   45     /// equal to bits_per_id/uint::bits rounded up.
   46     priv words_per_id: uint,
   47 
   48     // mapping from node to bitset index.
   49     priv nodeid_to_bitset: HashMap<ast::NodeId,uint>,
   50 
   51     // Bit sets per id.  The following three fields (`gens`, `kills`,
   52     // and `on_entry`) all have the same structure. For each id in
   53     // `id_range`, there is a range of words equal to `words_per_id`.
   54     // So, to access the bits for any given id, you take a slice of
   55     // the full vector (see the method `compute_id_range()`).
   56 
   57     /// bits generated as we exit the scope `id`. Updated by `add_gen()`.
   58     priv gens: ~[uint],
   59 
   60     /// bits killed as we exit the scope `id`. Updated by `add_kill()`.
   61     priv kills: ~[uint],
   62 
   63     /// bits that are valid on entry to the scope `id`. Updated by
   64     /// `propagate()`.
   65     priv on_entry: ~[uint]
   66 }
   67 
   68 /// Parameterization for the precise form of data flow that is used.
   69 pub trait DataFlowOperator {
   70     /// Specifies the initial value for each bit in the `on_entry` set
   71     fn initial_value(&self) -> bool;
   72 
   73     /// Joins two predecessor bits together, typically either `|` or `&`
   74     fn join(&self, succ: uint, pred: uint) -> uint;
   75 
   76     /// True if we should propagate through closures
   77     fn walk_closures(&self) -> bool;
   78 }
   79

   81     dfcx: &'self mut DataFlowContext<O>,
   82     changed: bool
   83 }

   85 struct LoopScope<'self> {
   86     loop_id: ast::NodeId,
   87     break_bits: ~[uint]
   88 }

   90 impl<O:DataFlowOperator> DataFlowContext<O> {
   91     pub fn new(tcx: ty::ctxt,

   94                id_range: id_range,
   95                bits_per_id: uint) -> DataFlowContext<O> {
   96         let words_per_id = (bits_per_id + uint::bits - 1) / uint::bits;
   97 
   98         debug!("DataFlowContext::new(id_range=%?, bits_per_id=%?, words_per_id=%?)",
   99                id_range, bits_per_id, words_per_id);
  100 
  101         let gens = ~[];
  102         let kills = ~[];
  103         let on_entry = ~[];
  104 
  105         DataFlowContext {
  106             tcx: tcx,
  107             method_map: method_map,
  108             words_per_id: words_per_id,
  109             nodeid_to_bitset: HashMap::new(),
  110             bits_per_id: bits_per_id,
  111             oper: oper,
  112             gens: gens,
  113             kills: kills,
  114             on_entry: on_entry
  115         }
  116     }

  118     pub fn add_gen(&mut self, id: ast::NodeId, bit: uint) {
  119         //! Indicates that `id` generates `bit`
  120 
  121         debug!("add_gen(id=%?, bit=%?)", id, bit);
  122         let (start, end) = self.compute_id_range(id);
  123         {
  124             let gens = self.gens.mut_slice(start, end);
  125             set_bit(gens, bit);
  126         }
  127     }

  129     pub fn add_kill(&mut self, id: ast::NodeId, bit: uint) {

  131 
  132         debug!("add_kill(id=%?, bit=%?)", id, bit);
  133         let (start, end) = self.compute_id_range(id);
  134         {

  136             set_bit(kills, bit);
  137         }
  138     }

  140     fn apply_gen_kill(&mut self, id: ast::NodeId, bits: &mut [uint]) {
  141         //! Applies the gen and kill sets for `id` to `bits`
  142 
  143         debug!("apply_gen_kill(id=%?, bits=%s) [before]",
  144                id, mut_bits_to_str(bits));
  145         let (start, end) = self.compute_id_range(id);
  146         let gens = self.gens.slice(start, end);
  147         bitwise(bits, gens, |a, b| a | b);
  148         let kills = self.kills.slice(start, end);
  149         bitwise(bits, kills, |a, b| a & !b);
  150 
  151         debug!("apply_gen_kill(id=%?, bits=%s) [after]",
  152                id, mut_bits_to_str(bits));
  153     }

  155     fn apply_kill(&mut self, id: ast::NodeId, bits: &mut [uint]) {
  156         debug!("apply_kill(id=%?, bits=%s) [before]",
  157                id, mut_bits_to_str(bits));
  158         let (start, end) = self.compute_id_range(id);
  159         let kills = self.kills.slice(start, end);
  160         bitwise(bits, kills, |a, b| a & !b);
  161         debug!("apply_kill(id=%?, bits=%s) [after]",
  162                id, mut_bits_to_str(bits));
  163     }

  165     fn compute_id_range_frozen(&self, id: ast::NodeId) -> (uint, uint) {
  166         let n = *self.nodeid_to_bitset.get(&id);
  167         let start = n * self.words_per_id;
  168         let end = start + self.words_per_id;
  169         (start, end)
  170     }
  171 
  172     fn compute_id_range(&mut self, id: ast::NodeId) -> (uint, uint) {
  173         let mut expanded = false;
  174         let len = self.nodeid_to_bitset.len();
  175         let n = do self.nodeid_to_bitset.find_or_insert_with(id) |_| {
  176             expanded = true;
  177             len
  178         };
  179         if expanded {
  180             let entry = if self.oper.initial_value() { uint::max_value } else {0};
  181             do self.words_per_id.times {
  182                 self.gens.push(0);
  183                 self.kills.push(0);
  184                 self.on_entry.push(entry);
  185             }
  186         }
  187         let start = *n * self.words_per_id;
  188         let end = start + self.words_per_id;
  189 
  190         assert!(start < self.gens.len());
  191         assert!(end <= self.gens.len());
  192         assert!(self.gens.len() == self.kills.len());
  193         assert!(self.gens.len() == self.on_entry.len());
  194 
  195         (start, end)
  196     }

  199     pub fn each_bit_on_entry_frozen(&self,
  200                                     id: ast::NodeId,
  201                                     f: &fn(uint) -> bool) -> bool {
  202         //! Iterates through each bit that is set on entry to `id`.
  203         //! Only useful after `propagate()` has been called.
  204         if !self.nodeid_to_bitset.contains_key(&id) {
  205             return true;
  206         }
  207         let (start, end) = self.compute_id_range_frozen(id);
  208         let on_entry = self.on_entry.slice(start, end);
  209         debug!("each_bit_on_entry_frozen(id=%?, on_entry=%s)",
  210                id, bits_to_str(on_entry));
  211         self.each_bit(on_entry, f)
  212     }

  228                         id: ast::NodeId,

  233         let gens = self.gens.slice(start, end);

  242         //! Iterates through each bit in the gen set for `id`.
  243         if !self.nodeid_to_bitset.contains_key(&id) {
  244             return true;
  245         }
  246         let (start, end) = self.compute_id_range_frozen(id);
  247         let gens = self.gens.slice(start, end);
  248         debug!("each_gen_bit(id=%?, gens=%s)",

  253     fn each_bit(&self,
  254                 words: &[uint],
  255                 f: &fn(uint) -> bool) -> bool {
  256         //! Helper for iterating over the bits in a bit set.
  257 
  258         for (word_index, &word) in words.iter().enumerate() {
  259             if word != 0 {
  260                 let base_index = word_index * uint::bits;
  261                 for offset in range(0u, uint::bits) {
  262                     let bit = 1 << offset;
  263                     if (word & bit) != 0 {
  264                         // NB: we round up the total number of bits
  265                         // that we store in any given bit set so that
  266                         // it is an even multiple of uint::bits.  This
  267                         // means that there may be some stray bits at
  268                         // the end that do not correspond to any
  269                         // actual value.  So before we callback, check
  270                         // whether the bit_index is greater than the
  271                         // actual value the user specified and stop
  272                         // iterating if so.
  273                         let bit_index = base_index + offset;
  274                         if bit_index >= self.bits_per_id {
  275                             return true;
  276                         } else if !f(bit_index) {
  277                             return false;
  278                         }
  279                     }
  280                 }
  281             }
  282         }
  283         return true;
  284     }
  285 }

  287 impl<O:DataFlowOperator+Clone+'static> DataFlowContext<O> {
  288 //                      ^^^^^^^^^^^^^ only needed for pretty printing
  289     pub fn propagate(&mut self, blk: &ast::Block) {
  290         //! Performs the data flow analysis.
  291 
  292         if self.bits_per_id == 0 {
  293             // Optimize the surprisingly common degenerate case.
  294             return;
  295         }
  296 
  297         {
  298             let mut propcx = PropagationContext {
  299                 dfcx: self,
  300                 changed: true
  301             };
  302 
  303             let mut temp = vec::from_elem(self.words_per_id, 0u);
  304             let mut loop_scopes = ~[];
  305 
  306             while propcx.changed {
  307                 propcx.changed = false;
  308                 propcx.reset(temp);
  309                 propcx.walk_block(blk, temp, &mut loop_scopes);
  310             }
  311         }
  312 
  313         debug!("Dataflow result:");
  314         debug!("%s", {
  315             let this = @(*self).clone();
  316             this.pretty_print_to(io::stderr(), blk);
  317             ""
  318         });
  319     }
  320 
  321     fn pretty_print_to(@self, wr: @io::Writer, blk: &ast::Block) {
  322         let pre: @fn(pprust::ann_node) = |node| {
  323             let (ps, id) = match node {
  324                 pprust::node_expr(ps, expr) => (ps, expr.id),
  325                 pprust::node_block(ps, blk) => (ps, blk.id),
  326                 pprust::node_item(ps, _) => (ps, 0),
  327                 pprust::node_pat(ps, pat) => (ps, pat.id)
  328             };
  329 
  330             if self.nodeid_to_bitset.contains_key(&id) {
  331                 let (start, end) = self.compute_id_range_frozen(id);
  332                 let on_entry = self.on_entry.slice(start, end);
  333                 let entry_str = bits_to_str(on_entry);
  334 
  335                 let gens = self.gens.slice(start, end);
  336                 let gens_str = if gens.iter().any(|&u| u != 0) {
  337                     fmt!(" gen: %s", bits_to_str(gens))
  338                 } else {
  339                     ~""
  340                 };
  341 
  342                 let kills = self.kills.slice(start, end);
  343                 let kills_str = if kills.iter().any(|&u| u != 0) {
  344                     fmt!(" kill: %s", bits_to_str(kills))
  345                 } else {
  346                     ~""
  347                 };
  348 
  349                 let comment_str = fmt!("id %d: %s%s%s",
  350                                        id, entry_str, gens_str, kills_str);
  351                 pprust::synth_comment(ps, comment_str);
  352                 pp::space(ps.s);
  353             }
  354         };
  355 
  356         let post: @fn(pprust::ann_node) = |_| {
  357         };
  358 
  359         let ps = pprust::rust_printer_annotated(
  360             wr, self.tcx.sess.intr(),
  361             pprust::pp_ann {pre:pre, post:post});
  362         pprust::cbox(ps, pprust::indent_unit);
  363         pprust::ibox(ps, 0u);
  364         pprust::print_block(ps, blk);
  365         pp::eof(ps.s);
  366     }
  367 }

  369 impl<'self, O:DataFlowOperator> PropagationContext<'self, O> {
  370     fn tcx(&self) -> ty::ctxt {
  371         self.dfcx.tcx
  372     }
  373 
  374     fn walk_block(&mut self,
  375                   blk: &ast::Block,
  376                   in_out: &mut [uint],
  377                   loop_scopes: &mut ~[LoopScope]) {
  378         debug!("DataFlowContext::walk_block(blk.id=%?, in_out=%s)",
  379                blk.id, bits_to_str(reslice(in_out)));
  380 
  381         self.merge_with_entry_set(blk.id, in_out);
  382 
  383         for &stmt in blk.stmts.iter() {
  384             self.walk_stmt(stmt, in_out, loop_scopes);
  385         }
  386 
  387         self.walk_opt_expr(blk.expr, in_out, loop_scopes);
  388 
  389         self.dfcx.apply_gen_kill(blk.id, in_out);
  390     }
  391 
  392     fn walk_stmt(&mut self,
  393                  stmt: @ast::stmt,
  394                  in_out: &mut [uint],
  395                  loop_scopes: &mut ~[LoopScope]) {
  396         match stmt.node {
  397             ast::stmt_decl(decl, _) => {
  398                 self.walk_decl(decl, in_out, loop_scopes);
  399             }
  400 
  401             ast::stmt_expr(expr, _) | ast::stmt_semi(expr, _) => {
  402                 self.walk_expr(expr, in_out, loop_scopes);
  403             }
  404 
  405             ast::stmt_mac(*) => {
  406                 self.tcx().sess.span_bug(stmt.span, "unexpanded macro");
  407             }
  408         }
  409     }
  410 
  411     fn walk_decl(&mut self,
  412                  decl: @ast::decl,
  413                  in_out: &mut [uint],
  414                  loop_scopes: &mut ~[LoopScope]) {
  415         match decl.node {
  416             ast::decl_local(local) => {
  417                 self.walk_opt_expr(local.init, in_out, loop_scopes);
  418                 self.walk_pat(local.pat, in_out, loop_scopes);
  419             }
  420 
  421             ast::decl_item(_) => {}
  422         }
  423     }
  424 
  425     fn walk_expr(&mut self,
  426                  expr: @ast::expr,
  427                  in_out: &mut [uint],
  428                  loop_scopes: &mut ~[LoopScope]) {
  429         debug!("DataFlowContext::walk_expr(expr=%s, in_out=%s)",
  430                expr.repr(self.dfcx.tcx), bits_to_str(reslice(in_out)));
  431 
  432         self.merge_with_entry_set(expr.id, in_out);
  433 
  434         match expr.node {
  435             ast::expr_fn_block(ref decl, ref body) => {
  436                 if self.dfcx.oper.walk_closures() {
  437                     // In the absence of once fns, we must assume that
  438                     // every function body will execute more than
  439                     // once. Thus we treat every function body like a
  440                     // loop.
  441                     //
  442                     // What is subtle and a bit tricky, also, is how
  443                     // to deal with the "output" bits---that is, what
  444                     // do we consider to be the successor of a
  445                     // function body, given that it could be called
  446                     // from any point within its lifetime? What we do
  447                     // is to add their effects immediately as of the
  448                     // point of creation. Of course we have to ensure
  449                     // that this is sound for the analyses which make
  450                     // use of dataflow.
  451                     //
  452                     // In the case of the initedness checker (which
  453                     // does not currently use dataflow, but I hope to
  454                     // convert at some point), we will simply not walk
  455                     // closures at all, so it's a moot point.
  456                     //
  457                     // In the case of the borrow checker, this means
  458                     // the loans which would be created by calling a
  459                     // function come into effect immediately when the
  460                     // function is created. This is guaranteed to be
  461                     // earlier than the point at which the loan
  462                     // actually comes into scope (which is the point
  463                     // at which the closure is *called*). Because
  464                     // loans persist until the scope of the loans is
  465                     // exited, it is always a safe approximation to
  466                     // have a loan begin earlier than it actually will
  467                     // at runtime, so this should be sound.
  468                     //
  469                     // We stil have to be careful in the region
  470                     // checker and borrow checker to treat function
  471                     // bodies like loops, which implies some
  472                     // limitations. For example, a closure cannot root
  473                     // a managed box for longer than its body.
  474                     //
  475                     // General control flow looks like this:
  476                     //
  477                     //  +- (expr) <----------+
  478                     //  |    |               |
  479                     //  |    v               |
  480                     //  |  (body) -----------+--> (exit)
  481                     //  |    |               |
  482                     //  |    + (break/loop) -+
  483                     //  |                    |
  484                     //  +--------------------+
  485                     //
  486                     // This is a bit more conservative than a loop.
  487                     // Note that we must assume that even after a
  488                     // `break` occurs (e.g., in a `for` loop) that the
  489                     // closure may be reinvoked.
  490                     //
  491                     // One difference from other loops is that `loop`
  492                     // and `break` statements which target a closure
  493                     // both simply add to the `break_bits`.
  494 
  495                     // func_bits represents the state when the function
  496                     // returns
  497                     let mut func_bits = reslice(in_out).to_owned();
  498 
  499                     loop_scopes.push(LoopScope {
  500                         loop_id: expr.id,
  501                         break_bits: reslice(in_out).to_owned()
  502                     });
  503                     for input in decl.inputs.iter() {
  504                         self.walk_pat(input.pat, func_bits, loop_scopes);
  505                     }
  506                     self.walk_block(body, func_bits, loop_scopes);
  507 
  508                     // add the bits from any early return via `break`,
  509                     // `continue`, or `return` into `func_bits`
  510                     let loop_scope = loop_scopes.pop();
  511                     join_bits(&self.dfcx.oper, loop_scope.break_bits, func_bits);
  512 
  513                     // add `func_bits` to the entry bits for `expr`,
  514                     // since we must assume the function may be called
  515                     // more than once
  516                     self.add_to_entry_set(expr.id, reslice(func_bits));
  517 
  518                     // the final exit bits include whatever was present
  519                     // in the original, joined with the bits from the function
  520                     join_bits(&self.dfcx.oper, func_bits, in_out);
  521                 }
  522             }

  747     fn pop_scopes(&mut self,
  748                   from_expr: @ast::expr,
  749                   to_scope: &mut LoopScope,
  750                   in_out: &mut [uint]) {
  751         //! Whenever you have a `break` or a `loop` statement, flow
  752         //! exits through any number of enclosing scopes on its
  753         //! way to the new destination. This function applies the kill
  754         //! sets of those enclosing scopes to `in_out` (those kill sets
  755         //! concern items that are going out of scope).
  756 
  757         let tcx = self.tcx();
  758         let region_maps = tcx.region_maps;
  759 
  760         debug!("pop_scopes(from_expr=%s, to_scope=%?, in_out=%s)",
  761                from_expr.repr(tcx), to_scope.loop_id,
  762                bits_to_str(reslice(in_out)));
  763 
  764         let mut id = from_expr.id;
  765         while id != to_scope.loop_id {
  766             self.dfcx.apply_kill(id, in_out);
  767 
  768             match region_maps.opt_encl_scope(id) {
  769                 Some(i) => { id = i; }
  770                 None => {
  771                     tcx.sess.span_bug(
  772                         from_expr.span,
  773                         fmt!("pop_scopes(from_expr=%s, to_scope=%?) \
  774                               to_scope does not enclose from_expr",
  775                              from_expr.repr(tcx), to_scope.loop_id));
  776                 }
  777             }
  778         }
  779     }

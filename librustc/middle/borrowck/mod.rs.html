 124  fn borrowck_fn(this: &mut BorrowckCtxt,
 125                 fk: &visit::fn_kind,
 126                 decl: &ast::fn_decl,
 127                 body: &ast::Block,
 128                 sp: Span,
 129                 id: ast::NodeId) {
 130      match fk {
 131          &visit::fk_anon(*) |
 132          &visit::fk_fn_block(*) => {
 133              // Closures are checked as part of their containing fn item.
 134          }
 135  
 136          &visit::fk_item_fn(*) |
 137          &visit::fk_method(*) => {
 138              debug!("borrowck_fn(id=%?)", id);
 139  
 140              // Check the body of fn items.
 141              let (id_range, all_loans, move_data) =
 142                  gather_loans::gather_loans(this, decl, body);
 143              let mut loan_dfcx =
 144                  DataFlowContext::new(this.tcx,
 145                                       this.method_map,
 146                                       LoanDataFlowOperator,
 147                                       id_range,
 148                                       all_loans.len());
 149              for (loan_idx, loan) in all_loans.iter().enumerate() {
 150                  loan_dfcx.add_gen(loan.gen_scope, loan_idx);
 151                  loan_dfcx.add_kill(loan.kill_scope, loan_idx);
 152              }
 153              loan_dfcx.propagate(body);
 154  
 155              let flowed_moves = move_data::FlowedMoveData::new(move_data,
 156                                                                this.tcx,
 157                                                                this.method_map,
 158                                                                id_range,
 159                                                                body);
 160  
 161              check_loans::check_loans(this, &loan_dfcx, flowed_moves,
 162                                       *all_loans, body);
 163          }
 164      }
 165  
 166      visit::walk_fn(this, fk, decl, body, sp, id, ());
 167  }

 197  // The keys to the root map combine the `id` of the deref expression
 198  // with the number of types that it is *autodereferenced*. So, for
 199  // example, imagine I have a variable `x: @@@T` and an expression
 200  // `(*x).f`.  This will have 3 derefs, one explicit and then two
 201  // autoderefs. These are the relevant `root_map_key` values that could
 202  // appear:
 203  //
 204  //    {id:*x, derefs:0} --> roots `x` (type: @@@T, due to explicit deref)
 205  //    {id:*x, derefs:1} --> roots `*x` (type: @@T, due to autoderef #1)
 206  //    {id:*x, derefs:2} --> roots `**x` (type: @T, due to autoderef #2)
 207  //
 208  // Note that there is no entry with derefs:3---the type of that expression
 209  // is T, which is not a box.
 210  //
 211  // Note that implicit dereferences also occur with indexing of `@[]`,
 212  // `@str`, etc.  The same rules apply. So, for example, given a
 213  // variable `x` of type `@[@[...]]`, if I have an instance of the
 214  // expression `x[0]` which is then auto-slice'd, there would be two
 215  // potential entries in the root map, both with the id of the `x[0]`
 216  // expression. The entry with `derefs==0` refers to the deref of `x`
 217  // used as part of evaluating `x[0]`. The entry with `derefs==1`
 218  // refers to the deref of the `x[0]` that occurs as part of the
 219  // auto-slice.
 220  #[deriving(Eq, IterBytes)]
 221  pub struct root_map_key {
 222      id: ast::NodeId,
 223      derefs: uint
 224  }

 301  pub fn opt_loan_path(cmt: mc::cmt) -> Option<@LoanPath> {
 302      //! Computes the `LoanPath` (if any) for a `cmt`.
 303      //! Note that this logic is somewhat duplicated in
 304      //! the method `compute()` found in `gather_loans::restrictions`,
 305      //! which allows it to share common loan path pieces as it
 306      //! traverses the CMT.
 307  
 308      match cmt.cat {
 309          mc::cat_rvalue(*) |
 310          mc::cat_static_item |
 311          mc::cat_copied_upvar(_) => {
 312              None
 313          }
 314  
 315          mc::cat_local(id) |
 316          mc::cat_arg(id) |
 317          mc::cat_self(id) => {
 318              Some(@LpVar(id))
 319          }
 320  
 321          mc::cat_deref(cmt_base, _, pk) => {
 322              do opt_loan_path(cmt_base).map_move |lp| {
 323                  @LpExtend(lp, cmt.mutbl, LpDeref(pk))
 324              }
 325          }
 326  
 327          mc::cat_interior(cmt_base, ik) => {
 328              do opt_loan_path(cmt_base).map_move |lp| {
 329                  @LpExtend(lp, cmt.mutbl, LpInterior(ik))
 330              }
 331          }
 332  
 333          mc::cat_downcast(cmt_base) |
 334          mc::cat_stack_upvar(cmt_base) |
 335          mc::cat_discr(cmt_base, _) => {
 336              opt_loan_path(cmt_base)
 337          }
 338      }
 339  }

 342  // Restrictions
 343  //
 344  // Borrowing an lvalue often results in *restrictions* that limit what
 345  // can be done with this lvalue during the scope of the loan:
 346  //
 347  // - `RESTR_MUTATE`: The lvalue may not be modified.
 348  // - `RESTR_CLAIM`: `&mut` borrows of the lvalue are forbidden.
 349  // - `RESTR_FREEZE`: `&` borrows of the lvalue are forbidden.
 350  // - `RESTR_ALIAS`: All borrows of the lvalue are forbidden.
 351  //
 352  // In addition, no value which is restricted may be moved. Therefore,
 353  // restrictions are meaningful even if the RestrictionSet is empty,
 354  // because the restriction against moves is implied.
 355  
 356  pub struct Restriction {
 357      loan_path: @LoanPath,
 358      set: RestrictionSet
 359  }
 360  
 361  #[deriving(Eq)]
 362  pub struct RestrictionSet {
 363      bits: u32
 364  }
 365  
 366  pub static RESTR_EMPTY: RestrictionSet  = RestrictionSet {bits: 0b0000};
 367  pub static RESTR_MUTATE: RestrictionSet = RestrictionSet {bits: 0b0001};
 368  pub static RESTR_CLAIM: RestrictionSet  = RestrictionSet {bits: 0b0010};
 369  pub static RESTR_FREEZE: RestrictionSet = RestrictionSet {bits: 0b0100};
 370  pub static RESTR_ALIAS: RestrictionSet  = RestrictionSet {bits: 0b1000};
 371  
 372  impl RestrictionSet {
 373      pub fn intersects(&self, restr: RestrictionSet) -> bool {
 374          (self.bits & restr.bits) != 0
 375      }
 376  
 377      pub fn contains_all(&self, restr: RestrictionSet) -> bool {
 378          (self.bits & restr.bits) == restr.bits
 379      }
 380  }
 381  
 382  impl BitOr<RestrictionSet,RestrictionSet> for RestrictionSet {
 383      fn bitor(&self, rhs: &RestrictionSet) -> RestrictionSet {
 384          RestrictionSet {bits: self.bits | rhs.bits}
 385      }
 386  }
 387  
 388  impl BitAnd<RestrictionSet,RestrictionSet> for RestrictionSet {
 389      fn bitand(&self, rhs: &RestrictionSet) -> RestrictionSet {
 390          RestrictionSet {bits: self.bits & rhs.bits}
 391      }
 392  }

 394  ///////////////////////////////////////////////////////////////////////////
 395  // Rooting of managed boxes
 396  //
 397  // When we borrow the interior of a managed box, it is sometimes
 398  // necessary to *root* the box, meaning to stash a copy of the box
 399  // somewhere that the garbage collector will find it. This ensures
 400  // that the box is not collected for the lifetime of the borrow.
 401  //
 402  // As part of this rooting, we sometimes also freeze the box at
 403  // runtime, meaning that we dynamically detect when the box is
 404  // borrowed in incompatible ways.
 405  //
 406  // Both of these actions are driven through the `root_map`, which maps
 407  // from a node to the dynamic rooting action that should be taken when
 408  // that node executes. The node is identified through a
 409  // `root_map_key`, which pairs a node-id and a deref count---the
 410  // problem is that sometimes the box that needs to be rooted is only
 411  // uncovered after a certain number of auto-derefs.
 412  
 413  pub struct RootInfo {
 414      scope: ast::NodeId,
 415      freeze: Option<DynaFreezeKind> // Some() if we should freeze box at runtime
 416  }
 417  
 418  pub type root_map = @mut HashMap<root_map_key, RootInfo>;
 419  
 420  pub fn root_map() -> root_map {
 421      return @mut HashMap::new();
 422  }

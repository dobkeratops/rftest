  80         stats: @mut BorrowStats {
  81             loaned_paths_same: 0,
  82             loaned_paths_imm: 0,
  83             stable_paths: 0,
  84             req_pure_paths: 0,

  87     };

 116 fn borrowck_fn(fk: &oldvisit::fn_kind,
 117                decl: &ast::fn_decl,
 118                body: &ast::Block,
 119                sp: span,
 120                id: ast::NodeId,
 121                (this, v): (@BorrowckCtxt,
 122                            oldvisit::vt<@BorrowckCtxt>)) {
 123     match fk {
 124         &oldvisit::fk_anon(*) |
 125         &oldvisit::fk_fn_block(*) => {
 126             // Closures are checked as part of their containing fn item.
 127         }
 128 
 129         &oldvisit::fk_item_fn(*) |
 130         &oldvisit::fk_method(*) => {
 131             debug!("borrowck_fn(id=%?)", id);
 132 
 133             // Check the body of fn items.
 134             let (id_range, all_loans, move_data) =
 135                 gather_loans::gather_loans(this, decl, body);
 136             let mut loan_dfcx =
 137                 DataFlowContext::new(this.tcx,
 138                                      this.method_map,
 139                                      LoanDataFlowOperator,
 140                                      id_range,
 141                                      all_loans.len());
 142             for (loan_idx, loan) in all_loans.iter().enumerate() {
 143                 loan_dfcx.add_gen(loan.gen_scope, loan_idx);
 144                 loan_dfcx.add_kill(loan.kill_scope, loan_idx);
 145             }
 146             loan_dfcx.propagate(body);
 147 
 148             let flowed_moves = move_data::FlowedMoveData::new(move_data,
 149                                                               this.tcx,
 150                                                               this.method_map,
 151                                                               id_range,
 152                                                               body);
 153 
 154             check_loans::check_loans(this, &loan_dfcx, flowed_moves,
 155                                      *all_loans, body);
 156         }
 157     }
 158 
 159     oldvisit::visit_fn(fk, decl, body, sp, id, (this, v));
 160 }

 190 // The keys to the root map combine the `id` of the deref expression
 191 // with the number of types that it is *autodereferenced*. So, for
 192 // example, imagine I have a variable `x: @@@T` and an expression
 193 // `(*x).f`.  This will have 3 derefs, one explicit and then two
 194 // autoderefs. These are the relevant `root_map_key` values that could
 195 // appear:
 196 //
 197 //    {id:*x, derefs:0} --> roots `x` (type: @@@T, due to explicit deref)
 198 //    {id:*x, derefs:1} --> roots `*x` (type: @@T, due to autoderef #1)
 199 //    {id:*x, derefs:2} --> roots `**x` (type: @T, due to autoderef #2)
 200 //
 201 // Note that there is no entry with derefs:3---the type of that expression
 202 // is T, which is not a box.
 203 //
 204 // Note that implicit dereferences also occur with indexing of `@[]`,
 205 // `@str`, etc.  The same rules apply. So, for example, given a
 206 // variable `x` of type `@[@[...]]`, if I have an instance of the
 207 // expression `x[0]` which is then auto-slice'd, there would be two
 208 // potential entries in the root map, both with the id of the `x[0]`
 209 // expression. The entry with `derefs==0` refers to the deref of `x`
 210 // used as part of evaluating `x[0]`. The entry with `derefs==1`
 211 // refers to the deref of the `x[0]` that occurs as part of the
 212 // auto-slice.
 213 #[deriving(Eq, IterBytes)]
 214 pub struct root_map_key {
 215     id: ast::NodeId,
 216     derefs: uint
 217 }

 267 pub fn opt_loan_path(cmt: mc::cmt) -> Option<@LoanPath> {
 268     //! Computes the `LoanPath` (if any) for a `cmt`.
 269     //! Note that this logic is somewhat duplicated in
 270     //! the method `compute()` found in `gather_loans::restrictions`,
 271     //! which allows it to share common loan path pieces as it
 272     //! traverses the CMT.
 273 
 274     match cmt.cat {
 275         mc::cat_rvalue(*) |
 276         mc::cat_static_item |
 277         mc::cat_copied_upvar(_) |
 278         mc::cat_implicit_self => {
 279             None
 280         }
 281 
 282         mc::cat_local(id) |
 283         mc::cat_arg(id) |
 284         mc::cat_self(id) => {
 285             Some(@LpVar(id))
 286         }
 287 
 288         mc::cat_deref(cmt_base, _, _) => {
 289             opt_loan_path(cmt_base).map(
 290                 |&lp| @LpExtend(lp, cmt.mutbl, LpDeref))
 291         }
 292 
 293         mc::cat_interior(cmt_base, ik) => {
 294             opt_loan_path(cmt_base).map(
 295                 |&lp| @LpExtend(lp, cmt.mutbl, LpInterior(ik)))
 296         }
 297 
 298         mc::cat_downcast(cmt_base) |
 299         mc::cat_stack_upvar(cmt_base) |
 300         mc::cat_discr(cmt_base, _) => {
 301             opt_loan_path(cmt_base)
 302         }
 303     }
 304 }

 307 // Restrictions
 308 //
 309 // Borrowing an lvalue often results in *restrictions* that limit what
 310 // can be done with this lvalue during the scope of the loan:
 311 //
 312 // - `RESTR_MUTATE`: The lvalue may not be modified.
 313 // - `RESTR_CLAIM`: `&mut` borrows of the lvalue are forbidden.
 314 // - `RESTR_FREEZE`: `&` borrows of the lvalue are forbidden.
 315 // - `RESTR_ALIAS`: All borrows of the lvalue are forbidden.
 316 //
 317 // In addition, no value which is restricted may be moved. Therefore,
 318 // restrictions are meaningful even if the RestrictionSet is empty,
 319 // because the restriction against moves is implied.
 320 
 321 pub struct Restriction {
 322     loan_path: @LoanPath,
 323     set: RestrictionSet
 324 }
 325 
 326 #[deriving(Eq)]
 327 pub struct RestrictionSet {
 328     bits: u32
 329 }
 330 
 331 pub static RESTR_EMPTY: RestrictionSet  = RestrictionSet {bits: 0b0000};
 332 pub static RESTR_MUTATE: RestrictionSet = RestrictionSet {bits: 0b0001};
 333 pub static RESTR_CLAIM: RestrictionSet  = RestrictionSet {bits: 0b0010};
 334 pub static RESTR_FREEZE: RestrictionSet = RestrictionSet {bits: 0b0100};
 335 pub static RESTR_ALIAS: RestrictionSet  = RestrictionSet {bits: 0b1000};
 336 
 337 impl RestrictionSet {

 359 ///////////////////////////////////////////////////////////////////////////
 360 // Rooting of managed boxes
 361 //
 362 // When we borrow the interior of a managed box, it is sometimes
 363 // necessary to *root* the box, meaning to stash a copy of the box
 364 // somewhere that the garbage collector will find it. This ensures
 365 // that the box is not collected for the lifetime of the borrow.
 366 //
 367 // As part of this rooting, we sometimes also freeze the box at
 368 // runtime, meaning that we dynamically detect when the box is
 369 // borrowed in incompatible ways.
 370 //
 371 // Both of these actions are driven through the `root_map`, which maps
 372 // from a node to the dynamic rooting action that should be taken when
 373 // that node executes. The node is identified through a
 374 // `root_map_key`, which pairs a node-id and a deref count---the
 375 // problem is that sometimes the box that needs to be rooted is only
 376 // uncovered after a certain number of auto-derefs.
 377 
 378 pub struct RootInfo {
 379     scope: ast::NodeId,
 380     freeze: Option<DynaFreezeKind> // Some() if we should freeze box at runtime
 381 }
 382 
 383 pub type root_map = @mut HashMap<root_map_key, RootInfo>;
 384 
 385 pub fn root_map() -> root_map {
 386     return @mut HashMap::new();
 387 }

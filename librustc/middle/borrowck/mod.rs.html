 123  fn borrowck_item(this: &mut BorrowckCtxt, item: &ast::Item) {
 124      // Gather loans for items. Note that we don't need
 125      // to check loans for single expressions. The check
 126      // loan step is intended for things that have a data
 127      // flow dependent conditions.
 128      match item.node {
 129          ast::ItemStatic(_, _, ex) => {
 130              gather_loans::gather_loans_in_static_initializer(this, ex);
 131          }
 132          _ => {
 133              visit::walk_item(this, item, ());
 134          }
 135      }
 136  }

 138  fn borrowck_fn(this: &mut BorrowckCtxt,
 139                 fk: &FnKind,
 140                 decl: &ast::FnDecl,
 141                 body: &ast::Block,
 142                 sp: Span,
 143                 id: ast::NodeId) {
 144      debug!("borrowck_fn(id={})", id);
 145  
 146      // Check the body of fn items.
 147      let (id_range, all_loans, move_data) =
 148          gather_loans::gather_loans_in_fn(this, decl, body);
 149      let mut loan_dfcx =
 150          DataFlowContext::new(this.tcx,
 151                               this.method_map,
 152                               LoanDataFlowOperator,
 153                               id_range,
 154                               all_loans.len());
 155      for (loan_idx, loan) in all_loans.iter().enumerate() {
 156          loan_dfcx.add_gen(loan.gen_scope, loan_idx);
 157          loan_dfcx.add_kill(loan.kill_scope, loan_idx);
 158      }
 159      loan_dfcx.propagate(body);
 160  
 161      let flowed_moves = move_data::FlowedMoveData::new(move_data,
 162                                                        this.tcx,
 163                                                        this.method_map,
 164                                                        id_range,
 165                                                        body);
 166  
 167      check_loans::check_loans(this, &loan_dfcx, flowed_moves,
 168                               all_loans.as_slice(), body);
 169  
 170      visit::walk_fn(this, fk, decl, body, sp, id, ());
 171  }

 195  // The keys to the root map combine the `id` of the deref expression
 196  // with the number of types that it is *autodereferenced*. So, for
 197  // example, imagine I have a variable `x: @@@T` and an expression
 198  // `(*x).f`.  This will have 3 derefs, one explicit and then two
 199  // autoderefs. These are the relevant `root_map_key` values that could
 200  // appear:
 201  //
 202  //    {id:*x, derefs:0} --> roots `x` (type: @@@T, due to explicit deref)
 203  //    {id:*x, derefs:1} --> roots `*x` (type: @@T, due to autoderef #1)
 204  //    {id:*x, derefs:2} --> roots `**x` (type: @T, due to autoderef #2)
 205  //
 206  // Note that there is no entry with derefs:3---the type of that expression
 207  // is T, which is not a box.
 208  #[deriving(Eq, TotalEq, Hash)]
 209  pub struct root_map_key {
 210      pub id: ast::NodeId,
 211      pub derefs: uint
 212  }

 267  pub fn opt_loan_path(cmt: mc::cmt) -> Option<@LoanPath> {
 268      //! Computes the `LoanPath` (if any) for a `cmt`.
 269      //! Note that this logic is somewhat duplicated in
 270      //! the method `compute()` found in `gather_loans::restrictions`,
 271      //! which allows it to share common loan path pieces as it
 272      //! traverses the CMT.
 273  
 274      match cmt.cat {
 275          mc::cat_rvalue(..) |
 276          mc::cat_static_item |
 277          mc::cat_copied_upvar(_) => {
 278              None
 279          }
 280  
 281          mc::cat_local(id) |
 282          mc::cat_arg(id) |
 283          mc::cat_upvar(ty::UpvarId {var_id: id, ..}, _) => {
 284              Some(@LpVar(id))
 285          }
 286  
 287          mc::cat_deref(cmt_base, _, pk) => {
 288              opt_loan_path(cmt_base).map(|lp| {
 289                  @LpExtend(lp, cmt.mutbl, LpDeref(pk))
 290              })
 291          }
 292  
 293          mc::cat_interior(cmt_base, ik) => {
 294              opt_loan_path(cmt_base).map(|lp| {
 295                  @LpExtend(lp, cmt.mutbl, LpInterior(ik))
 296              })
 297          }
 298  
 299          mc::cat_downcast(cmt_base) |
 300          mc::cat_discr(cmt_base, _) => {
 301              opt_loan_path(cmt_base)
 302          }
 303      }
 304  }

 307  // Restrictions
 308  //
 309  // Borrowing an lvalue often results in *restrictions* that limit what
 310  // can be done with this lvalue during the scope of the loan:
 311  //
 312  // - `RESTR_MUTATE`: The lvalue may not be modified or `&mut` borrowed.
 313  // - `RESTR_FREEZE`: `&` borrows of the lvalue are forbidden.
 314  //
 315  // In addition, no value which is restricted may be moved. Therefore,
 316  // restrictions are meaningful even if the RestrictionSet is empty,
 317  // because the restriction against moves is implied.
 318  
 319  pub struct Restriction {
 320      loan_path: @LoanPath,
 321      set: RestrictionSet
 322  }
 323  
 324  #[deriving(Eq)]
 325  pub struct RestrictionSet {
 326      bits: u32
 327  }
 328  
 329  #[allow(dead_code)] // potentially useful
 330  pub static RESTR_EMPTY: RestrictionSet  = RestrictionSet {bits: 0b0000};
 331  pub static RESTR_MUTATE: RestrictionSet = RestrictionSet {bits: 0b0001};
 332  pub static RESTR_FREEZE: RestrictionSet = RestrictionSet {bits: 0b0010};
 333

 335      pub fn intersects(&self, restr: RestrictionSet) -> bool {

 337      }
 338  }

 358  ///////////////////////////////////////////////////////////////////////////
 359  // Rooting of managed boxes
 360  //
 361  // When we borrow the interior of a managed box, it is sometimes
 362  // necessary to *root* the box, meaning to stash a copy of the box
 363  // somewhere that the garbage collector will find it. This ensures
 364  // that the box is not collected for the lifetime of the borrow.
 365  //
 366  // As part of this rooting, we sometimes also freeze the box at
 367  // runtime, meaning that we dynamically detect when the box is
 368  // borrowed in incompatible ways.
 369  //
 370  // Both of these actions are driven through the `root_map`, which maps
 371  // from a node to the dynamic rooting action that should be taken when
 372  // that node executes. The node is identified through a
 373  // `root_map_key`, which pairs a node-id and a deref count---the
 374  // problem is that sometimes the box that needs to be rooted is only
 375  // uncovered after a certain number of auto-derefs.
 376  
 377  pub struct RootInfo {
 378      pub scope: ast::NodeId,
 379  }
 380  
 381  pub type root_map = @RefCell<HashMap<root_map_key, RootInfo>>;
 382  
 383  pub fn root_map() -> root_map {
 384      return @RefCell::new(HashMap::new());
 385  }

 657      pub fn bckerr_to_str(&self, err: BckError) -> ~str {
 658          match err.code {
 659              err_mutbl => {
 660                  let descr = match opt_loan_path(err.cmt) {
 661                      None => format!("{} {}",
 662                                      err.cmt.mutbl.to_user_str(),
 663                                      self.cmt_to_str(err.cmt)),
 664                      Some(lp) => format!("{} {} `{}`",
 665                                          err.cmt.mutbl.to_user_str(),
 666                                          self.cmt_to_str(err.cmt),
 667                                          self.loan_path_to_str(lp)),
 668                  };
 669  
 670                  match err.cause {
 671                      ClosureCapture(_) => {
 672                          format!("closure cannot assign to {}", descr)
 673                      }
 674                      AddrOf | RefBinding | AutoRef => {
 675                          format!("cannot borrow {} as mutable", descr)

 679              err_out_of_root_scope(..) => {
 680                  format!("cannot root managed value long enough")
 681              }
 682              err_out_of_scope(..) => {
 683                  let msg = match opt_loan_path(err.cmt) {
 684                      None => format!("borrowed value"),
 685                      Some(lp) => format!("`{}`", self.loan_path_to_str(lp)),
 686                  };
 687                  format!("{} does not live long enough", msg)
 688              }
 689              err_borrowed_pointer_too_short(..) => {
 690                  let descr = match opt_loan_path(err.cmt) {
 691                      Some(lp) => format!("`{}`", self.loan_path_to_str(lp)),
 692                      None => self.cmt_to_str(err.cmt),
 693                  };
 694  
 695                  format!("lifetime of {} is too short to guarantee \
 696                          its contents can be safely reborrowed",
 697                          descr)
 698              }
 699          }
 700      }

 702      pub fn report_aliasability_violation(&self,
 703                                           span: Span,
 704                                           kind: AliasableViolationKind,
 705                                           cause: mc::AliasableReason) {
 706          let prefix = match kind {
 707              MutabilityViolation => {
 708                  "cannot assign to data"
 709              }
 710              BorrowViolation(ClosureCapture(_)) => {
 711                  // I don't think we can get aliasability violations
 712                  // with closure captures, so no need to come up with a
 713                  // good error message. The reason this cannot happen
 714                  // is because we only capture local variables in
 715                  // closures, and those are never aliasable.
 716                  self.tcx.sess.span_bug(
 717                      span,
 718                      "aliasability violation with closure");
 719              }
 720              BorrowViolation(AddrOf) |
 721              BorrowViolation(AutoRef) |
 722              BorrowViolation(RefBinding) => {
 723                  "cannot borrow data mutably"
 724              }
 725          };
 726  
 727          match cause {
 728              mc::AliasableOther => {
 729                  self.tcx.sess.span_err(
 730                      span,
 731                      format!("{} in an aliasable location", prefix));
 732              }
 733              mc::AliasableStatic(..) |
 734              mc::AliasableStaticMut(..) => {
 735                  self.tcx.sess.span_err(
 736                      span,
 737                      format!("{} in a static location", prefix));
 738              }
 739              mc::AliasableManaged => {
 740                  self.tcx.sess.span_err(
 741                      span,
 742                      format!("{} in a `@` pointer", prefix));
 743              }
 744              mc::AliasableBorrowed => {
 745                  self.tcx.sess.span_err(
 746                      span,
 747                      format!("{} in a `&` reference", prefix));

 749          }

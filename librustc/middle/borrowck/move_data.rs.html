  33  pub struct MoveData {

  36

 197      pub fn move_path(&mut self,
 198                       tcx: ty::ctxt,
 199                       lp: @LoanPath) -> MovePathIndex {
 200          /*!
 201           * Returns the existing move path index for `lp`, if any,
 202           * and otherwise adds a new index for `lp` and any of its
 203           * base paths that do not yet have an index.
 204           */
 205  
 206          match self.path_map.find(&lp) {
 207              Some(&index) => {
 208                  return index;
 209              }
 210              None => {}
 211          }
 212  
 213          let index = match *lp {
 214              LpVar(*) => {
 215                  let index = MovePathIndex(self.paths.len());
 216  
 217                  self.paths.push(MovePath {
 218                      loan_path: lp,
 219                      parent: InvalidMovePathIndex,
 220                      first_move: InvalidMoveIndex,
 221                      first_child: InvalidMovePathIndex,
 222                      next_sibling: InvalidMovePathIndex,
 223                  });
 224  
 225                  index
 226              }
 227  
 228              LpExtend(base, _, _) => {
 229                  let parent_index = self.move_path(tcx, base);
 230                  let index = MovePathIndex(self.paths.len());
 231  
 232                  let next_sibling = self.path(parent_index).first_child;
 233                  self.mut_path(parent_index).first_child = index;
 234  
 235                  self.paths.push(MovePath {
 236                      loan_path: lp,
 237                      parent: parent_index,
 238                      first_move: InvalidMoveIndex,
 239                      first_child: InvalidMovePathIndex,
 240                      next_sibling: next_sibling,
 241                  });
 242  
 243                  index
 244              }
 245          };
 246  
 247          debug!("move_path(lp=%s, index=%?)",
 248                 lp.repr(tcx),
 249                 index);
 250  
 251          assert_eq!(*index, self.paths.len() - 1);
 252          self.path_map.insert(lp, index);
 253          return index;
 254      }

 263                             lp: @LoanPath)
 264                             -> OptVec<MovePathIndex> {

 266          self.add_existing_base_paths(lp, &mut result);

 270      fn add_existing_base_paths(&self,
 271                                 lp: @LoanPath,
 272                                 result: &mut OptVec<MovePathIndex>) {
 273          /*!
 274           * Adds any existing move path indices for `lp` and any base
 275           * paths of `lp` to `result`, but does not add new move paths
 276           */
 277  
 278          match self.path_map.find_copy(&lp) {
 279              Some(index) => {
 280                  do self.each_base_path(index) |p| {
 281                      result.push(p);
 282                      true
 283                  };
 284              }
 285              None => {
 286                  match *lp {
 287                      LpVar(*) => { }
 288                      LpExtend(b, _, _) => {
 289                          self.add_existing_base_paths(b, result);
 290                      }
 291                  }
 292              }
 293          }
 294  
 295      }

 300                      id: ast::NodeId,
 301                      kind: MoveKind) {
 302          /*!
 303           * Adds a new move entry for a move of `lp` that occurs at

 326      pub fn add_assignment(&mut self,
 327                            tcx: ty::ctxt,
 328                            lp: @LoanPath,
 329                            assign_id: ast::NodeId,
 330                            span: span,
 331                            assignee_id: ast::NodeId) {
 332          /*!
 333           * Adds a new record for an assignment to `lp` that occurs at
 334           * location `id` with the given `span`.
 335           */
 336  
 337          debug!("add_assignment(lp=%s, assign_id=%?, assignee_id=%?",
 338                 lp.repr(tcx), assign_id, assignee_id);
 339  
 340          let path_index = self.move_path(tcx, lp);
 341  
 342          self.assignee_ids.insert(assignee_id);
 343  
 344          let assignment = Assignment {
 345              path: path_index,
 346              id: assign_id,
 347              span: span,
 348          };
 349  
 350          if self.is_var_path(path_index) {
 351              debug!("add_assignment[var](lp=%s, assignment=%u, path_index=%?)",
 352                     lp.repr(tcx), self.var_assignments.len(), path_index);
 353  
 354              self.var_assignments.push(assignment);
 355          } else {
 356              debug!("add_assignment[path](lp=%s, path_index=%?)",
 357                     lp.repr(tcx), path_index);
 358  
 359              self.path_assignments.push(assignment);
 360          }
 361      }

 363      fn add_gen_kills(&self,
 364                       tcx: ty::ctxt,
 365                       dfcx_moves: &mut MoveDataFlow,
 366                       dfcx_assign: &mut AssignDataFlow) {
 367          /*!
 368           * Adds the gen/kills for the various moves and
 369           * assignments into the provided data flow contexts.
 370           * Moves are generated by moves and killed by assignments and
 371           * scoping. Assignments are generated by assignment to variables and
 372           * killed by scoping. See `doc.rs` for more details.
 373           */
 374  
 375          for (i, move) in self.moves.iter().enumerate() {
 376              dfcx_moves.add_gen(move.id, i);
 377          }
 378  
 379          for (i, assignment) in self.var_assignments.iter().enumerate() {
 380              dfcx_assign.add_gen(assignment.id, i);
 381              self.kill_moves(assignment.path, assignment.id, dfcx_moves);
 382          }
 383  
 384          for assignment in self.path_assignments.iter() {
 385              self.kill_moves(assignment.path, assignment.id, dfcx_moves);
 386          }
 387  
 388          // Kill all moves related to a variable `x` when it goes out
 389          // of scope:
 390          for path in self.paths.iter() {
 391              match *path.loan_path {
 392                  LpVar(id) => {
 393                      let kill_id = tcx.region_maps.encl_scope(id);
 394                      let path = *self.path_map.get(&path.loan_path);
 395                      self.kill_moves(path, kill_id, dfcx_moves);
 396                  }
 397                  LpExtend(*) => {}
 398              }
 399          }
 400  
 401          // Kill all assignments when the variable goes out of scope:
 402          for (assignment_index, assignment) in self.var_assignments.iter().enumerate() {
 403              match *self.path(assignment.path).loan_path {
 404                  LpVar(id) => {
 405                      let kill_id = tcx.region_maps.encl_scope(id);
 406                      dfcx_assign.add_kill(kill_id, assignment_index);
 407                  }
 408                  LpExtend(*) => {
 409                      tcx.sess.bug("Var assignment for non var path");
 410                  }
 411              }
 412          }
 413      }

 415      fn each_base_path(&self,
 416                        index: MovePathIndex,
 417                        f: &fn(MovePathIndex) -> bool)
 418                        -> bool {
 419          let mut p = index;
 420          while p != InvalidMovePathIndex {
 421              if !f(p) {
 422                  return false;
 423              }
 424              p = self.path(p).parent;
 425          }
 426          return true;
 427      }
 428  
 429      fn each_extending_path(&self,
 430                             index: MovePathIndex,
 431                             f: &fn(MovePathIndex) -> bool) -> bool {
 432          if !f(index) {
 433              return false;
 434          }
 435  
 436          let mut p = self.path(index).first_child;
 437          while p != InvalidMovePathIndex {
 438              if !self.each_extending_path(p, |x| f(x)) {
 439                  return false;
 440              }
 441              p = self.path(p).next_sibling;
 442          }
 443  
 444          return true;
 445      }
 446  
 447      fn each_applicable_move(&self,
 448                              index0: MovePathIndex,
 449                              f: &fn(MoveIndex) -> bool) -> bool {
 450          let mut ret = true;
 451          do self.each_extending_path(index0) |index| {
 452              let mut p = self.path(index).first_move;
 453              while p != InvalidMoveIndex {
 454                  if !f(p) {
 455                      ret = false;
 456                      break;
 457                  }
 458                  p = self.move(p).next_move;
 459              }
 460              ret
 461          };
 462          ret
 463      }
 464  
 465      fn kill_moves(&self,
 466                    path: MovePathIndex,
 467                    kill_id: ast::NodeId,
 468                    dfcx_moves: &mut MoveDataFlow) {
 469          do self.each_applicable_move(path) |move_index| {
 470              dfcx_moves.add_kill(kill_id, *move_index);
 471              true
 472          };
 473      }
 474  }

 476  impl FlowedMoveData {
 477      pub fn new(move_data: @mut MoveData,
 478                 tcx: ty::ctxt,
 479                 method_map: typeck::method_map,
 480                 id_range: ast_util::id_range,
 481                 body: &ast::Block)
 482                 -> FlowedMoveData
 483      {
 484          let mut dfcx_moves =
 485              DataFlowContext::new(tcx,
 486                                   method_map,
 487                                   MoveDataFlowOperator,
 488                                   id_range,
 489                                   move_data.moves.len());
 490          let mut dfcx_assign =
 491              DataFlowContext::new(tcx,
 492                                   method_map,
 493                                   AssignDataFlowOperator,
 494                                   id_range,
 495                                   move_data.var_assignments.len());
 496          move_data.add_gen_kills(tcx, &mut dfcx_moves, &mut dfcx_assign);
 497          dfcx_moves.propagate(body);
 498          dfcx_assign.propagate(body);
 499          FlowedMoveData {
 500              move_data: move_data,
 501              dfcx_moves: dfcx_moves,
 502              dfcx_assign: dfcx_assign,
 503          }
 504      }

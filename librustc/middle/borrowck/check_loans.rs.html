 127      pub fn each_issued_loan(&self,
 128                              scope_id: ast::NodeId,
 129                              op: &fn(&Loan) -> bool)
 130                              -> bool {
 131          //! Iterates over each loan that has been issued
 132          //! on entrance to `scope_id`, regardless of whether it is
 133          //! actually *in scope* at that point.  Sometimes loans
 134          //! are issued for future scopes and thus they may have been
 135          //! *issued* but not yet be in effect.
 136  
 137          do self.dfcx_loans.each_bit_on_entry_frozen(scope_id) |loan_index| {
 138              let loan = &self.all_loans[loan_index];
 139              op(loan)
 140          }
 141      }

 143      pub fn each_in_scope_loan(&self,
 144                                scope_id: ast::NodeId,
 145                                op: &fn(&Loan) -> bool)
 146                                -> bool {
 147          //! Like `each_issued_loan()`, but only considers loans that are
 148          //! currently in scope.
 149  
 150          let region_maps = self.tcx().region_maps;
 151          do self.each_issued_loan(scope_id) |loan| {
 152              if region_maps.is_subscope_of(scope_id, loan.kill_scope) {
 153                  op(loan)
 154              } else {
 155                  true
 156              }
 157          }
 158      }

 160      pub fn each_in_scope_restriction(&self,
 161                                       scope_id: ast::NodeId,
 162                                       loan_path: @LoanPath,
 163                                       op: &fn(&Loan, &Restriction) -> bool)
 164                                       -> bool {
 165          //! Iterates through all the in-scope restrictions for the
 166          //! given `loan_path`
 167  
 168          do self.each_in_scope_loan(scope_id) |loan| {
 169              let mut ret = true;
 170              for restr in loan.restrictions.iter() {
 171                  if restr.loan_path == loan_path {
 172                      if !op(loan, restr) {
 173                          ret = false;
 174                          break;
 175                      }
 176                  }
 177              }
 178              ret
 179          }
 180      }

 194      pub fn check_for_conflicting_loans(&self, scope_id: ast::NodeId) {
 195          //! Checks to see whether any of the loans that are issued
 196          //! by `scope_id` conflict with loans that have already been
 197          //! issued when we enter `scope_id` (for example, we do not
 198          //! permit two `&mut` borrows of the same variable).
 199  
 200          debug!("check_for_conflicting_loans(scope_id=%?)", scope_id);
 201  
 202          let new_loan_indices = self.loans_generated_by(scope_id);
 203          debug!("new_loan_indices = %?", new_loan_indices);
 204  
 205          do self.each_issued_loan(scope_id) |issued_loan| {
 206              for &new_loan_index in new_loan_indices.iter() {
 207                  let new_loan = &self.all_loans[new_loan_index];
 208                  self.report_error_if_loans_conflict(issued_loan, new_loan);
 209              }
 210              true
 211          };
 212  
 213          for (i, &x) in new_loan_indices.iter().enumerate() {
 214              let old_loan = &self.all_loans[x];
 215              for &y in new_loan_indices.slice_from(i+1).iter() {
 216                  let new_loan = &self.all_loans[y];
 217                  self.report_error_if_loans_conflict(old_loan, new_loan);
 218              }
 219          }
 220      }

 222      pub fn report_error_if_loans_conflict(&self,
 223                                            old_loan: &Loan,
 224                                            new_loan: &Loan) {
 225          //! Checks whether `old_loan` and `new_loan` can safely be issued
 226          //! simultaneously.
 227  
 228          debug!("report_error_if_loans_conflict(old_loan=%s, new_loan=%s)",
 229                 old_loan.repr(self.tcx()),
 230                 new_loan.repr(self.tcx()));
 231  
 232          // Should only be called for loans that are in scope at the same time.
 233          let region_maps = self.tcx().region_maps;
 234          assert!(region_maps.scopes_intersect(old_loan.kill_scope,
 235                                               new_loan.kill_scope));
 236  
 237          self.report_error_if_loan_conflicts_with_restriction(
 238              old_loan, new_loan, old_loan, new_loan) &&
 239          self.report_error_if_loan_conflicts_with_restriction(
 240              new_loan, old_loan, old_loan, new_loan);
 241      }

 243      pub fn report_error_if_loan_conflicts_with_restriction(&self,
 244                                                             loan1: &Loan,
 245                                                             loan2: &Loan,
 246                                                             old_loan: &Loan,
 247                                                             new_loan: &Loan)
 248                                                             -> bool {
 249          //! Checks whether the restrictions introduced by `loan1` would
 250          //! prohibit `loan2`. Returns false if an error is reported.
 251  
 252          debug!("report_error_if_loan_conflicts_with_restriction(\
 253                  loan1=%s, loan2=%s)",
 254                 loan1.repr(self.tcx()),
 255                 loan2.repr(self.tcx()));
 256  
 257          // Restrictions that would cause the new loan to be illegal:
 258          let illegal_if = match loan2.mutbl {
 259              MutableMutability   => RESTR_ALIAS | RESTR_FREEZE | RESTR_CLAIM,
 260              ImmutableMutability => RESTR_ALIAS | RESTR_FREEZE,
 261              ConstMutability     => RESTR_ALIAS,
 262          };
 263          debug!("illegal_if=%?", illegal_if);
 264  
 265          for restr in loan1.restrictions.iter() {
 266              if !restr.set.intersects(illegal_if) { loop; }
 267              if restr.loan_path != loan2.loan_path { loop; }
 268  
 269              match (new_loan.mutbl, old_loan.mutbl) {
 270                  (MutableMutability, MutableMutability) => {
 271                      self.bccx.span_err(
 272                          new_loan.span,
 273                          fmt!("cannot borrow `%s` as mutable \
 274                                more than once at a time",
 275                               self.bccx.loan_path_to_str(new_loan.loan_path)));
 276                      self.bccx.span_note(
 277                          old_loan.span,
 278                          fmt!("second borrow of `%s` as mutable occurs here",
 279                               self.bccx.loan_path_to_str(new_loan.loan_path)));
 280                      return false;
 281                  }
 282  
 283                  _ => {
 284                      self.bccx.span_err(
 285                          new_loan.span,
 286                          fmt!("cannot borrow `%s` as %s because \
 287                                it is also borrowed as %s",
 288                               self.bccx.loan_path_to_str(new_loan.loan_path),
 289                               self.bccx.mut_to_str(new_loan.mutbl),
 290                               self.bccx.mut_to_str(old_loan.mutbl)));
 291                      self.bccx.span_note(
 292                          old_loan.span,
 293                          fmt!("second borrow of `%s` occurs here",
 294                               self.bccx.loan_path_to_str(new_loan.loan_path)));
 295                      return false;
 296                  }
 297              }
 298          }
 299  
 300          true
 301      }

 333      pub fn check_assignment(&self, expr: @ast::Expr) {
 334          // We don't use cat_expr() here because we don't want to treat
 335          // auto-ref'd parameters in overloaded operators as rvalues.
 336          let cmt = match self.bccx.tcx.adjustments.find(&expr.id) {
 337              None => self.bccx.cat_expr_unadjusted(expr),
 338              Some(&adj) => self.bccx.cat_expr_autoderefd(expr, adj)
 339          };
 340  
 341          debug!("check_assignment(cmt=%s)", cmt.repr(self.tcx()));
 342  
 343          // Mutable values can be assigned, as long as they obey loans
 344          // and aliasing restrictions:
 345          if cmt.mutbl.is_mutable() {
 346              if check_for_aliasable_mutable_writes(self, expr, cmt) {
 347                  if check_for_assignment_to_restricted_or_frozen_location(
 348                      self, expr, cmt)
 349                  {
 350                      // Safe, but record for lint pass later:
 351                      mark_variable_as_used_mut(self, cmt);
 352                  }
 353              }
 354              return;
 355          }
 356  
 357          // For immutable local variables, assignments are legal
 358          // if they cannot already have been assigned
 359          if self.is_local_variable(cmt) {
 360              assert!(cmt.mutbl.is_immutable()); // no "const" locals
 361              let lp = opt_loan_path(cmt).unwrap();
 362              do self.move_data.each_assignment_of(expr.id, lp) |assign| {
 363                  self.bccx.report_reassigned_immutable_variable(
 364                      expr.span,
 365                      lp,
 366                      assign);
 367                  false
 368              };
 369              return;
 370          }
 371  
 372          // Otherwise, just a plain error.
 373          self.bccx.span_err(
 374              expr.span,
 375              fmt!("cannot assign to %s %s",
 376                   cmt.mutbl.to_user_str(),
 377                   self.bccx.cmt_to_str(cmt)));
 378          return;

 380          fn mark_variable_as_used_mut(this: &CheckLoanCtxt,
 381                                       cmt: mc::cmt) {
 382              //! If the mutability of the `cmt` being written is inherited
 383              //! from a local variable, liveness will
 384              //! not have been able to detect that this variable's mutability
 385              //! is important, so we must add the variable to the
 386              //! `used_mut_nodes` table here.
 387  
 388              let mut cmt = cmt;
 389              loop {
 390                  debug!("mark_writes_through_upvars_as_used_mut(cmt=%s)",
 391                         cmt.repr(this.tcx()));
 392                  match cmt.cat {
 393                      mc::cat_local(id) |
 394                      mc::cat_arg(id) |
 395                      mc::cat_self(id) => {
 396                          this.tcx().used_mut_nodes.insert(id);
 397                          return;
 398                      }
 399  
 400                      mc::cat_stack_upvar(b) => {
 401                          cmt = b;
 402                      }
 403  
 404                      mc::cat_rvalue(*) |
 405                      mc::cat_static_item |
 406                      mc::cat_copied_upvar(*) |
 407                      mc::cat_deref(_, _, mc::unsafe_ptr(*)) |
 408                      mc::cat_deref(_, _, mc::gc_ptr(*)) |
 409                      mc::cat_deref(_, _, mc::region_ptr(*)) => {
 410                          assert_eq!(cmt.mutbl, mc::McDeclared);
 411                          return;
 412                      }
 413  
 414                      mc::cat_discr(b, _) |
 415                      mc::cat_deref(b, _, mc::uniq_ptr) => {
 416                          assert_eq!(cmt.mutbl, mc::McInherited);
 417                          cmt = b;
 418                      }
 419  
 420                      mc::cat_downcast(b) |
 421                      mc::cat_interior(b, _) => {
 422                          if cmt.mutbl == mc::McInherited {
 423                              cmt = b;
 424                          } else {
 425                              return; // field declared as mutable or some such
 426                          }
 427                      }
 428                  }
 429              }
 430          }

 432          fn check_for_aliasable_mutable_writes(this: &CheckLoanCtxt,
 433                                                expr: @ast::Expr,
 434                                                cmt: mc::cmt) -> bool {
 435              //! Safety checks related to writes to aliasable, mutable locations
 436

 508          fn check_for_assignment_to_restricted_or_frozen_location(
 509              this: &CheckLoanCtxt,
 510              expr: @ast::Expr,
 511              cmt: mc::cmt) -> bool

 513              //! Check for assignments that violate the terms of an
 514              //! outstanding loan.

 517                  Some(lp) => lp,

 522              // location. Here is one example of the kind of error caught
 523              // by this check:
 524              //
 525              //    let mut v = ~[1, 2, 3];

 530              // restriction on the path `v`.

 532              // Here is a second, more subtle example:
 533              //
 534              //    let mut v = ~[1, 2, 3];
 535              //    let p = &const v[0];

 537              //    v[1] = 5;                   // OK
 538              //    v = ~[4, 5, 3];             // Error
 539              //
 540              // In this case, `p` is pointing to `v[0]`, and it is a
 541              // `const` pointer in any case. So the first two
 542              // assignments are legal (and would be permitted by this
 543              // check). However, the final assignment (which is
 544              // logically equivalent) is forbidden, because it would
 545              // cause the existing `v` array to be freed, thus
 546              // invalidating `p`. In the code, this error results
 547              // because `gather_loans::restrictions` adds a
 548              // `RESTR_MUTATE` restriction whenever the contents of an
 549              // owned pointer are borrowed, and hence while `v[*]` is not
 550              // restricted from being written, `v` is.
 551              let cont = do this.each_in_scope_restriction(expr.id, loan_path)
 552                  |loan, restr|
 553              {
 554                  if restr.set.intersects(RESTR_MUTATE) {
 555                      this.report_illegal_mutation(expr, loan_path, loan);
 556                      false
 557                  } else {
 558                      true
 559                  }
 560              };

 565              // have been restricted. This covers paths that have been
 566              // directly lent out and their base paths, but does not
 567              // cover random extensions of those paths. For example,
 568              // the following program is not declared illegal by the
 569              // previous check:
 570              //
 571              //    let mut v = ~[1, 2, 3];
 572              //    let p = &v;
 573              //    v[0] = 4; // declared error by loop below, not code above
 574              //
 575              // The reason that this passes the previous check whereas
 576              // an assignment like `v = ~[4]` fails is because the assignment
 577              // here is to `v[*]`, and the existing restrictions were issued
 578              // for `v`, not `v[*]`.
 579              //
 580              // So in this loop, we walk back up the loan path so long

 583              // mutable or immutable (a const loan is ok).
 584              //
 585              // Mutability of a path can be dependent on the super path
 586              // in two ways. First, it might be inherited mutability.
 587              // Second, the pointee of an `&mut` pointer can only be
 588              // mutated if it is found in an unaliased location, so we
 589              // have to check that the owner location is not borrowed.
 590              //

 592              // restrictions.  We are only interested in the pointers

 595              // for all restrictions, and not just loans, then the following
 596              // valid program would be considered illegal:
 597              //

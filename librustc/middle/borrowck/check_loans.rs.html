  76      pub fn each_issued_loan(&self,
  77                              scope_id: ast::NodeId,
  78                              op: &fn(&Loan) -> bool)
  79                              -> bool {
  80          //! Iterates over each loan that has been issued
  81          //! on entrance to `scope_id`, regardless of whether it is
  82          //! actually *in scope* at that point.  Sometimes loans
  83          //! are issued for future scopes and thus they may have been
  84          //! *issued* but not yet be in effect.
  85  
  86          do self.dfcx_loans.each_bit_on_entry_frozen(scope_id) |loan_index| {
  87              let loan = &self.all_loans[loan_index];
  88              op(loan)
  89          }
  90      }

  92      pub fn each_in_scope_loan(&self,
  93                                scope_id: ast::NodeId,
  94                                op: &fn(&Loan) -> bool)
  95                                -> bool {
  96          //! Like `each_issued_loan()`, but only considers loans that are
  97          //! currently in scope.
  98  
  99          let region_maps = self.tcx().region_maps;
 100          do self.each_issued_loan(scope_id) |loan| {
 101              if region_maps.is_subscope_of(scope_id, loan.kill_scope) {
 102                  op(loan)
 103              } else {
 104                  true
 105              }
 106          }
 107      }

 109      pub fn each_in_scope_restriction(&self,
 110                                       scope_id: ast::NodeId,
 111                                       loan_path: @LoanPath,
 112                                       op: &fn(&Loan, &Restriction) -> bool)
 113                                       -> bool {
 114          //! Iterates through all the in-scope restrictions for the
 115          //! given `loan_path`
 116  
 117          do self.each_in_scope_loan(scope_id) |loan| {
 118              let mut ret = true;
 119              for restr in loan.restrictions.iter() {
 120                  if restr.loan_path == loan_path {
 121                      if !op(loan, restr) {
 122                          ret = false;
 123                          break;
 124                      }
 125                  }
 126              }
 127              ret
 128          }
 129      }

 143      pub fn check_for_conflicting_loans(&self, scope_id: ast::NodeId) {
 144          //! Checks to see whether any of the loans that are issued
 145          //! by `scope_id` conflict with loans that have already been
 146          //! issued when we enter `scope_id` (for example, we do not
 147          //! permit two `&mut` borrows of the same variable).
 148  
 149          debug!("check_for_conflicting_loans(scope_id=%?)", scope_id);
 150  
 151          let new_loan_indices = self.loans_generated_by(scope_id);
 152          debug!("new_loan_indices = %?", new_loan_indices);
 153  
 154          do self.each_issued_loan(scope_id) |issued_loan| {
 155              for &new_loan_index in new_loan_indices.iter() {
 156                  let new_loan = &self.all_loans[new_loan_index];
 157                  self.report_error_if_loans_conflict(issued_loan, new_loan);
 158              }
 159              true
 160          };
 161  
 162          for (i, &x) in new_loan_indices.iter().enumerate() {
 163              let old_loan = &self.all_loans[x];
 164              for &y in new_loan_indices.slice_from(i+1).iter() {
 165                  let new_loan = &self.all_loans[y];
 166                  self.report_error_if_loans_conflict(old_loan, new_loan);
 167              }
 168          }
 169      }

 171      pub fn report_error_if_loans_conflict(&self,
 172                                            old_loan: &Loan,
 173                                            new_loan: &Loan) {
 174          //! Checks whether `old_loan` and `new_loan` can safely be issued
 175          //! simultaneously.
 176  
 177          debug!("report_error_if_loans_conflict(old_loan=%s, new_loan=%s)",
 178                 old_loan.repr(self.tcx()),
 179                 new_loan.repr(self.tcx()));
 180  
 181          // Should only be called for loans that are in scope at the same time.
 182          let region_maps = self.tcx().region_maps;
 183          assert!(region_maps.scopes_intersect(old_loan.kill_scope,
 184                                               new_loan.kill_scope));
 185  
 186          self.report_error_if_loan_conflicts_with_restriction(
 187              old_loan, new_loan, old_loan, new_loan) &&
 188          self.report_error_if_loan_conflicts_with_restriction(
 189              new_loan, old_loan, old_loan, new_loan);
 190      }

 192      pub fn report_error_if_loan_conflicts_with_restriction(&self,
 193                                                             loan1: &Loan,
 194                                                             loan2: &Loan,
 195                                                             old_loan: &Loan,
 196                                                             new_loan: &Loan)
 197                                                             -> bool {
 198          //! Checks whether the restrictions introduced by `loan1` would
 199          //! prohibit `loan2`. Returns false if an error is reported.
 200  
 201          debug!("report_error_if_loan_conflicts_with_restriction(\
 202                  loan1=%s, loan2=%s)",
 203                 loan1.repr(self.tcx()),
 204                 loan2.repr(self.tcx()));
 205  
 206          // Restrictions that would cause the new loan to be illegal:
 207          let illegal_if = match loan2.mutbl {
 208              m_mutbl => RESTR_ALIAS | RESTR_FREEZE | RESTR_CLAIM,
 209              m_imm =>   RESTR_ALIAS | RESTR_FREEZE,
 210              m_const => RESTR_ALIAS,
 211          };
 212          debug!("illegal_if=%?", illegal_if);
 213  
 214          for restr in loan1.restrictions.iter() {
 215              if !restr.set.intersects(illegal_if) { loop; }
 216              if restr.loan_path != loan2.loan_path { loop; }
 217  
 218              match (new_loan.mutbl, old_loan.mutbl) {
 219                  (m_mutbl, m_mutbl) => {
 220                      self.bccx.span_err(
 221                          new_loan.span,
 222                          fmt!("cannot borrow `%s` as mutable \
 223                                more than once at a time",
 224                               self.bccx.loan_path_to_str(new_loan.loan_path)));
 225                      self.bccx.span_note(
 226                          old_loan.span,
 227                          fmt!("second borrow of `%s` as mutable occurs here",
 228                               self.bccx.loan_path_to_str(new_loan.loan_path)));
 229                      return false;
 230                  }
 231  
 232                  _ => {
 233                      self.bccx.span_err(
 234                          new_loan.span,
 235                          fmt!("cannot borrow `%s` as %s because \
 236                                it is also borrowed as %s",
 237                               self.bccx.loan_path_to_str(new_loan.loan_path),
 238                               self.bccx.mut_to_str(new_loan.mutbl),
 239                               self.bccx.mut_to_str(old_loan.mutbl)));
 240                      self.bccx.span_note(
 241                          old_loan.span,
 242                          fmt!("second borrow of `%s` occurs here",
 243                               self.bccx.loan_path_to_str(new_loan.loan_path)));
 244                      return false;
 245                  }
 246              }
 247          }
 248  
 249          true
 250      }

 282      pub fn check_assignment(&self, expr: @ast::expr) {
 283          // We don't use cat_expr() here because we don't want to treat
 284          // auto-ref'd parameters in overloaded operators as rvalues.
 285          let cmt = match self.bccx.tcx.adjustments.find(&expr.id) {
 286              None => self.bccx.cat_expr_unadjusted(expr),
 287              Some(&adj) => self.bccx.cat_expr_autoderefd(expr, adj)
 288          };
 289  
 290          debug!("check_assignment(cmt=%s)", cmt.repr(self.tcx()));
 291  
 292          // Mutable values can be assigned, as long as they obey loans
 293          // and aliasing restrictions:
 294          if cmt.mutbl.is_mutable() {
 295              if check_for_aliasable_mutable_writes(self, expr, cmt) {
 296                  if check_for_assignment_to_restricted_or_frozen_location(
 297                      self, expr, cmt)
 298                  {
 299                      // Safe, but record for lint pass later:
 300                      mark_variable_as_used_mut(self, cmt);
 301                  }
 302              }
 303              return;
 304          }
 305  
 306          // For immutable local variables, assignments are legal
 307          // if they cannot already have been assigned
 308          if self.is_local_variable(cmt) {
 309              assert!(cmt.mutbl.is_immutable()); // no "const" locals
 310              let lp = opt_loan_path(cmt).unwrap();
 311              do self.move_data.each_assignment_of(expr.id, lp) |assign| {
 312                  self.bccx.report_reassigned_immutable_variable(
 313                      expr.span,
 314                      lp,
 315                      assign);
 316                  false
 317              };
 318              return;
 319          }
 320  
 321          // Otherwise, just a plain error.
 322          self.bccx.span_err(
 323              expr.span,
 324              fmt!("cannot assign to %s %s",
 325                   cmt.mutbl.to_user_str(),
 326                   self.bccx.cmt_to_str(cmt)));
 327          return;

 329          fn mark_variable_as_used_mut(this: &CheckLoanCtxt,
 330                                       cmt: mc::cmt) {
 331              //! If the mutability of the `cmt` being written is inherited
 332              //! from a local variable, liveness will
 333              //! not have been able to detect that this variable's mutability
 334              //! is important, so we must add the variable to the
 335              //! `used_mut_nodes` table here.
 336  
 337              let mut cmt = cmt;
 338              loop {
 339                  debug!("mark_writes_through_upvars_as_used_mut(cmt=%s)",
 340                         cmt.repr(this.tcx()));
 341                  match cmt.cat {
 342                      mc::cat_local(id) |
 343                      mc::cat_arg(id) |
 344                      mc::cat_self(id) => {
 345                          this.tcx().used_mut_nodes.insert(id);
 346                          return;
 347                      }
 348  
 349                      mc::cat_stack_upvar(b) => {
 350                          cmt = b;
 351                      }
 352  
 353                      mc::cat_rvalue(*) |
 354                      mc::cat_static_item |
 355                      mc::cat_implicit_self |
 356                      mc::cat_copied_upvar(*) |
 357                      mc::cat_deref(_, _, mc::unsafe_ptr(*)) |
 358                      mc::cat_deref(_, _, mc::gc_ptr(*)) |
 359                      mc::cat_deref(_, _, mc::region_ptr(*)) => {
 360                          assert_eq!(cmt.mutbl, mc::McDeclared);
 361                          return;
 362                      }
 363  
 364                      mc::cat_discr(b, _) |
 365                      mc::cat_deref(b, _, mc::uniq_ptr) => {
 366                          assert_eq!(cmt.mutbl, mc::McInherited);
 367                          cmt = b;
 368                      }
 369  
 370                      mc::cat_downcast(b) |
 371                      mc::cat_interior(b, _) => {
 372                          if cmt.mutbl == mc::McInherited {
 373                              cmt = b;
 374                          } else {
 375                              return; // field declared as mutable or some such
 376                          }
 377                      }
 378                  }
 379              }
 380          }

 382          fn check_for_aliasable_mutable_writes(this: &CheckLoanCtxt,
 383                                                expr: @ast::expr,
 384                                                cmt: mc::cmt) -> bool {
 385              //! Safety checks related to writes to aliasable, mutable locations
 386  
 387              let guarantor = cmt.guarantor();
 388              debug!("check_for_aliasable_mutable_writes(cmt=%s, guarantor=%s)",
 389                     cmt.repr(this.tcx()), guarantor.repr(this.tcx()));
 390              match guarantor.cat {
 391                  mc::cat_deref(b, _, mc::region_ptr(m_mutbl, _)) => {
 392                      // Statically prohibit writes to `&mut` when aliasable
 393  
 394                      match b.freely_aliasable() {
 395                          None => {}
 396                          Some(cause) => {
 397                              this.bccx.report_aliasability_violation(
 398                                  expr.span,
 399                                  MutabilityViolation,
 400                                  cause);
 401                          }
 402                      }
 403                  }
 404  
 405                  mc::cat_deref(_, deref_count, mc::gc_ptr(ast::m_mutbl)) => {
 406                      // Dynamically check writes to `@mut`
 407  
 408                      let key = root_map_key {
 409                          id: guarantor.id,
 410                          derefs: deref_count
 411                      };
 412                      debug!("Inserting write guard at %?", key);
 413                      this.bccx.write_guard_map.insert(key);
 414                  }
 415  
 416                  _ => {}
 417              }
 418  
 419              return true; // no errors reported
 420          }

 422          fn check_for_assignment_to_restricted_or_frozen_location(
 423              this: &CheckLoanCtxt,
 424              expr: @ast::expr,
 425              cmt: mc::cmt) -> bool
 426          {
 427              //! Check for assignments that violate the terms of an
 428              //! outstanding loan.
 429  
 430              let loan_path = match opt_loan_path(cmt) {
 431                  Some(lp) => lp,
 432                  None => { return true; /* no loan path, can't be any loans */ }
 433              };
 434  
 435              // Start by searching for an assignment to a *restricted*
 436              // location. Here is one example of the kind of error caught
 437              // by this check:
 438              //
 439              //    let mut v = ~[1, 2, 3];
 440              //    let p = &v;
 441              //    v = ~[4];
 442              //
 443              // In this case, creating `p` triggers a RESTR_MUTATE
 444              // restriction on the path `v`.
 445              //
 446              // Here is a second, more subtle example:
 447              //
 448              //    let mut v = ~[1, 2, 3];
 449              //    let p = &const v[0];
 450              //    v[0] = 4;                   // OK
 451              //    v[1] = 5;                   // OK
 452              //    v = ~[4, 5, 3];             // Error
 453              //
 454              // In this case, `p` is pointing to `v[0]`, and it is a
 455              // `const` pointer in any case. So the first two
 456              // assignments are legal (and would be permitted by this
 457              // check). However, the final assignment (which is
 458              // logically equivalent) is forbidden, because it would
 459              // cause the existing `v` array to be freed, thus
 460              // invalidating `p`. In the code, this error results
 461              // because `gather_loans::restrictions` adds a
 462              // `RESTR_MUTATE` restriction whenever the contents of an
 463              // owned pointer are borrowed, and hence while `v[*]` is not
 464              // restricted from being written, `v` is.
 465              let cont = do this.each_in_scope_restriction(expr.id, loan_path)
 466                  |loan, restr|
 467              {
 468                  if restr.set.intersects(RESTR_MUTATE) {
 469                      this.report_illegal_mutation(expr, loan_path, loan);
 470                      false
 471                  } else {
 472                      true
 473                  }
 474              };
 475  
 476              if !cont { return false }
 477  
 478              // The previous code handled assignments to paths that
 479              // have been restricted. This covers paths that have been
 480              // directly lent out and their base paths, but does not
 481              // cover random extensions of those paths. For example,
 482              // the following program is not declared illegal by the
 483              // previous check:
 484              //
 485              //    let mut v = ~[1, 2, 3];
 486              //    let p = &v;
 487              //    v[0] = 4; // declared error by loop below, not code above
 488              //
 489              // The reason that this passes the previous check whereas
 490              // an assignment like `v = ~[4]` fails is because the assignment
 491              // here is to `v[*]`, and the existing restrictions were issued
 492              // for `v`, not `v[*]`.
 493              //
 494              // So in this loop, we walk back up the loan path so long
 495              // as the mutability of the path is dependent on a super
 496              // path, and check that the super path was not lent out as
 497              // mutable or immutable (a const loan is ok).
 498              //
 499              // Note that we are *not* checking for any and all
 500              // restrictions.  We are only interested in the pointers
 501              // that the user created, whereas we add restrictions for
 502              // all kinds of paths that are not directly aliased. If we checked
 503              // for all restrictions, and not just loans, then the following
 504              // valid program would be considered illegal:
 505              //
 506              //    let mut v = ~[1, 2, 3];
 507              //    let p = &const v[0];
 508              //    v[1] = 5; // ok
 509              //
 510              // Here the restriction that `v` not be mutated would be misapplied
 511              // to block the subpath `v[1]`.
 512              let full_loan_path = loan_path;
 513              let mut loan_path = loan_path;
 514              loop {
 515                  match *loan_path {
 516                      // Peel back one layer if `loan_path` has
 517                      // inherited mutability
 518                      LpExtend(lp_base, mc::McInherited, _) => {
 519                          loan_path = lp_base;
 520                      }
 521  
 522                      // Otherwise stop iterating
 523                      LpExtend(_, mc::McDeclared, _) |
 524                      LpExtend(_, mc::McImmutable, _) |
 525                      LpExtend(_, mc::McReadOnly, _) |
 526                      LpVar(_) => {
 527                          return true;
 528                      }
 529                  }
 530  
 531                  // Check for a non-const loan of `loan_path`
 532                  let cont = do this.each_in_scope_loan(expr.id) |loan| {
 533                      if loan.loan_path == loan_path && loan.mutbl != m_const {
 534                          this.report_illegal_mutation(expr, full_loan_path, loan);
 535                          false
 536                      } else {
 537                          true
 538                      }
 539                  };
 540  
 541                  if !cont { return false }
 542              }
 543          }
 544      }

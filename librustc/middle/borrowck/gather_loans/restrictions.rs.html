  53      fn restrict(&self,
  54                  cmt: mc::cmt,
  55                  restrictions: RestrictionSet) -> RestrictionResult {
  56  
  57          // Check for those cases where we cannot control the aliasing
  58          // and make sure that we are not being asked to.
  59          match cmt.freely_aliasable() {
  60              None => {}
  61              Some(cause) => {
  62                  self.check_aliasing_permitted(cause, restrictions);
  63              }
  64          }
  65  
  66          match cmt.cat {
  67              mc::cat_rvalue(*) => {
  68                  // Effectively, rvalues are stored into a
  69                  // non-aliasable temporary on the stack. Since they
  70                  // are inherently non-aliasable, they can only be
  71                  // accessed later through the borrow itself and hence
  72                  // must inherently comply with its terms.
  73                  Safe
  74              }
  75  
  76              mc::cat_local(local_id) |
  77              mc::cat_arg(local_id) |
  78              mc::cat_self(local_id) => {
  79                  // R-Variable
  80                  let lp = @LpVar(local_id);
  81                  SafeIf(lp, ~[Restriction {loan_path: lp,
  82                                            set: restrictions}])
  83              }
  84  
  85              mc::cat_downcast(cmt_base) => {
  86                  // When we borrow the interior of an enum, we have to
  87                  // ensure the enum itself is not mutated, because that
  88                  // could cause the type of the memory to change.
  89                  self.restrict(
  90                      cmt_base,
  91                      restrictions | RESTR_MUTATE | RESTR_CLAIM)
  92              }
  93  
  94              mc::cat_interior(cmt_base, i) => {
  95                  // R-Field
  96                  //
  97                  // Overwriting the base would not change the type of
  98                  // the memory, so no additional restrictions are
  99                  // needed.
 100                  let result = self.restrict(cmt_base, restrictions);
 101                  self.extend(result, cmt.mutbl, LpInterior(i), restrictions)
 102              }
 103  
 104              mc::cat_deref(cmt_base, _, mc::uniq_ptr) => {
 105                  // R-Deref-Send-Pointer
 106                  //
 107                  // When we borrow the interior of an owned pointer, we
 108                  // cannot permit the base to be mutated, because that
 109                  // would cause the unique pointer to be freed.
 110                  let result = self.restrict(
 111                      cmt_base,
 112                      restrictions | RESTR_MUTATE | RESTR_CLAIM);
 113                  self.extend(result, cmt.mutbl, LpDeref, restrictions)
 114              }
 115  
 116              mc::cat_copied_upvar(*) | // FIXME(#2152) allow mutation of upvars
 117              mc::cat_static_item(*) |
 118              mc::cat_implicit_self(*) |
 119              mc::cat_deref(_, _, mc::region_ptr(m_imm, _)) |
 120              mc::cat_deref(_, _, mc::gc_ptr(m_imm)) => {
 121                  // R-Deref-Imm-Borrowed
 122                  Safe
 123              }
 124  
 125              mc::cat_deref(_, _, mc::region_ptr(m_const, _)) |
 126              mc::cat_deref(_, _, mc::gc_ptr(m_const)) => {
 127                  // R-Deref-Freeze-Borrowed
 128                  self.check_no_mutability_control(cmt, restrictions);
 129                  Safe
 130              }
 131  
 132              mc::cat_deref(cmt_base, _, mc::gc_ptr(m_mutbl)) => {
 133                  // R-Deref-Managed-Borrowed
 134                  //
 135                  // Technically, no restrictions are *necessary* here.
 136                  // The validity of the borrow is guaranteed
 137                  // dynamically.  However, nonetheless we add a
 138                  // restriction to make a "best effort" to report
 139                  // static errors. For example, if there is code like
 140                  //
 141                  //    let v = @mut ~[1, 2, 3];
 142                  //    for e in v.iter() {
 143                  //        v.push(e + 1);
 144                  //    }
 145                  //
 146                  // Then the code below would add restrictions on `*v`,
 147                  // which means that an error would be reported
 148                  // here. This of course is not perfect. For example,
 149                  // a function like the following would not report an error
 150                  // at compile-time but would fail dynamically:
 151                  //
 152                  //    let v = @mut ~[1, 2, 3];
 153                  //    let w = v;
 154                  //    for e in v.iter() {
 155                  //        w.push(e + 1);
 156                  //    }
 157                  //
 158                  // In addition, we only add a restriction for those cases
 159                  // where we can construct a sensible loan path, so an
 160                  // example like the following will fail dynamically:
 161                  //
 162                  //    impl V {
 163                  //      fn get_list(&self) -> @mut ~[int];
 164                  //    }
 165                  //    ...
 166                  //    let v: &V = ...;
 167                  //    for e in v.get_list().iter() {
 168                  //        v.get_list().push(e + 1);
 169                  //    }
 170                  match opt_loan_path(cmt_base) {
 171                      None => Safe,
 172                      Some(lp_base) => {
 173                          let lp = @LpExtend(lp_base, cmt.mutbl, LpDeref);
 174                          SafeIf(lp, ~[Restriction {loan_path: lp,
 175                                                    set: restrictions}])
 176                      }
 177                  }
 178              }
 179  
 180              mc::cat_deref(cmt_base, _, mc::region_ptr(m_mutbl, _)) => {
 181                  // Because an `&mut` pointer does not inherit its
 182                  // mutability, we can only prevent mutation or prevent
 183                  // freezing if it is not aliased. Therefore, in such
 184                  // cases we restrict aliasing on `cmt_base`.
 185                  if restrictions != RESTR_EMPTY {
 186                      // R-Deref-Mut-Borrowed-1
 187                      let result = self.restrict(
 188                          cmt_base,
 189                          RESTR_ALIAS | RESTR_MUTATE | RESTR_CLAIM);
 190                      self.extend(result, cmt.mutbl, LpDeref, restrictions)
 191                  } else {
 192                      // R-Deref-Mut-Borrowed-2
 193                      Safe
 194                  }
 195              }
 196  
 197              mc::cat_deref(_, _, mc::unsafe_ptr(*)) => {
 198                  // We are very trusting when working with unsafe pointers.
 199                  Safe
 200              }
 201  
 202              mc::cat_stack_upvar(cmt_base) |
 203              mc::cat_discr(cmt_base, _) => {
 204                  self.restrict(cmt_base, restrictions)
 205              }
 206          }
 207      }

 209      fn extend(&self,
 210                result: RestrictionResult,
 211                mc: mc::MutabilityCategory,
 212                elem: LoanPathElem,
 213                restrictions: RestrictionSet) -> RestrictionResult {
 214          match result {
 215              Safe => Safe,
 216              SafeIf(base_lp, base_vec) => {
 217                  let lp = @LpExtend(base_lp, mc, elem);
 218                  SafeIf(lp, vec::append_one(base_vec,
 219                                             Restriction {loan_path: lp,
 220                                                          set: restrictions}))
 221              }
 222          }
 223      }

 225      fn check_aliasing_permitted(&self,
 226                                  cause: mc::AliasableReason,
 227                                  restrictions: RestrictionSet) {
 228          //! This method is invoked when the current `cmt` is something
 229          //! where aliasing cannot be controlled. It reports an error if
 230          //! the restrictions required that it not be aliased; currently
 231          //! this only occurs when re-borrowing an `&mut` pointer.
 232          //!
 233          //! NB: To be 100% consistent, we should report an error if
 234          //! RESTR_FREEZE is found, because we cannot prevent freezing,
 235          //! nor would we want to. However, we do not report such an
 236          //! error, because this restriction only occurs when the user
 237          //! is creating an `&mut` pointer to immutable or read-only
 238          //! data, and there is already another piece of code that
 239          //! checks for this condition.
 240  
 241          if restrictions.intersects(RESTR_ALIAS) {
 242              self.bccx.report_aliasability_violation(
 243                  self.span,
 244                  BorrowViolation,
 245                  cause);
 246          }
 247      }
 248  
 249      fn check_no_mutability_control(&self,
 250                                     cmt: mc::cmt,
 251                                     restrictions: RestrictionSet) {
 252          if restrictions.intersects(RESTR_MUTATE | RESTR_FREEZE | RESTR_CLAIM) {
 253              self.bccx.report(BckError {span: self.span,
 254                                         cmt: cmt,
 255                                         code: err_freeze_aliasable_const});
 256          }
 257      }

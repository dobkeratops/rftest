  53      fn restrict(&self,
  54                  cmt: mc::cmt,
  55                  restrictions: RestrictionSet) -> RestrictionResult {
  56  
  57          // Check for those cases where we cannot control the aliasing
  58          // and make sure that we are not being asked to.
  59          match cmt.freely_aliasable() {
  60              None => {}
  61              Some(cause) => {
  62                  self.check_aliasing_permitted(cause, restrictions);
  63              }
  64          }
  65  
  66          match cmt.cat {
  67              mc::cat_rvalue(*) => {
  68                  // Effectively, rvalues are stored into a
  69                  // non-aliasable temporary on the stack. Since they
  70                  // are inherently non-aliasable, they can only be
  71                  // accessed later through the borrow itself and hence
  72                  // must inherently comply with its terms.
  73                  Safe
  74              }
  75  
  76              mc::cat_local(local_id) |
  77              mc::cat_arg(local_id) |
  78              mc::cat_self(local_id) => {
  79                  // R-Variable
  80                  let lp = @LpVar(local_id);
  81                  SafeIf(lp, ~[Restriction {loan_path: lp,
  82                                            set: restrictions}])
  83              }
  84  
  85              mc::cat_downcast(cmt_base) => {
  86                  // When we borrow the interior of an enum, we have to
  87                  // ensure the enum itself is not mutated, because that
  88                  // could cause the type of the memory to change.
  89                  self.restrict(
  90                      cmt_base,
  91                      restrictions | RESTR_MUTATE | RESTR_CLAIM)
  92              }
  93  
  94              mc::cat_interior(cmt_base, i) => {
  95                  // R-Field
  96                  //
  97                  // Overwriting the base would not change the type of
  98                  // the memory, so no additional restrictions are
  99                  // needed.
 100                  let result = self.restrict(cmt_base, restrictions);
 101                  self.extend(result, cmt.mutbl, LpInterior(i), restrictions)
 102              }
 103  
 104              mc::cat_deref(cmt_base, _, pk @ mc::uniq_ptr) => {
 105                  // R-Deref-Send-Pointer
 106                  //
 107                  // When we borrow the interior of an owned pointer, we
 108                  // cannot permit the base to be mutated, because that
 109                  // would cause the unique pointer to be freed.
 110                  let result = self.restrict(
 111                      cmt_base,
 112                      restrictions | RESTR_MUTATE | RESTR_CLAIM);
 113                  self.extend(result, cmt.mutbl, LpDeref(pk), restrictions)
 114              }
 115  
 116              mc::cat_copied_upvar(*) | // FIXME(#2152) allow mutation of upvars
 117              mc::cat_static_item(*) |
 118              mc::cat_deref(_, _, mc::region_ptr(MutImmutable, _)) |
 119              mc::cat_deref(_, _, mc::gc_ptr(MutImmutable)) => {
 120                  // R-Deref-Imm-Borrowed
 121                  Safe
 122              }
 123  
 124              mc::cat_deref(cmt_base, _, pk @ mc::gc_ptr(MutMutable)) => {
 125                  // R-Deref-Managed-Borrowed
 126                  //
 127                  // Technically, no restrictions are *necessary* here.
 128                  // The validity of the borrow is guaranteed
 129                  // dynamically.  However, nonetheless we add a
 130                  // restriction to make a "best effort" to report
 131                  // static errors. For example, if there is code like
 132                  //
 133                  //    let v = @mut ~[1, 2, 3];
 134                  //    for e in v.iter() {
 135                  //        v.push(e + 1);
 136                  //    }
 137                  //
 138                  // Then the code below would add restrictions on `*v`,
 139                  // which means that an error would be reported
 140                  // here. This of course is not perfect. For example,
 141                  // a function like the following would not report an error
 142                  // at compile-time but would fail dynamically:
 143                  //
 144                  //    let v = @mut ~[1, 2, 3];
 145                  //    let w = v;
 146                  //    for e in v.iter() {
 147                  //        w.push(e + 1);
 148                  //    }
 149                  //
 150                  // In addition, we only add a restriction for those cases
 151                  // where we can construct a sensible loan path, so an
 152                  // example like the following will fail dynamically:
 153                  //
 154                  //    impl V {
 155                  //      fn get_list(&self) -> @mut ~[int];
 156                  //    }
 157                  //    ...
 158                  //    let v: &V = ...;
 159                  //    for e in v.get_list().iter() {
 160                  //        v.get_list().push(e + 1);
 161                  //    }
 162                  match opt_loan_path(cmt_base) {
 163                      None => Safe,
 164                      Some(lp_base) => {
 165                          let lp = @LpExtend(lp_base, cmt.mutbl, LpDeref(pk));
 166                          SafeIf(lp, ~[Restriction {loan_path: lp,
 167                                                    set: restrictions}])
 168                      }
 169                  }
 170              }
 171  
 172              mc::cat_deref(cmt_base, _, pk @ mc::region_ptr(MutMutable, _)) => {
 173                  // Because an `&mut` pointer does not inherit its
 174                  // mutability, we can only prevent mutation or prevent
 175                  // freezing if it is not aliased. Therefore, in such
 176                  // cases we restrict aliasing on `cmt_base`.
 177                  if restrictions != RESTR_EMPTY {
 178                      // R-Deref-Mut-Borrowed-1
 179                      let result = self.restrict(
 180                          cmt_base,
 181                          RESTR_ALIAS | RESTR_MUTATE | RESTR_CLAIM);
 182                      self.extend(result, cmt.mutbl, LpDeref(pk), restrictions)
 183                  } else {
 184                      // R-Deref-Mut-Borrowed-2
 185                      Safe
 186                  }
 187              }
 188  
 189              mc::cat_deref(_, _, mc::unsafe_ptr(*)) => {
 190                  // We are very trusting when working with unsafe pointers.
 191                  Safe
 192              }
 193  
 194              mc::cat_stack_upvar(cmt_base) |
 195              mc::cat_discr(cmt_base, _) => {
 196                  self.restrict(cmt_base, restrictions)
 197              }
 198          }
 199      }

 201      fn extend(&self,
 202                result: RestrictionResult,
 203                mc: mc::MutabilityCategory,
 204                elem: LoanPathElem,
 205                restrictions: RestrictionSet) -> RestrictionResult {
 206          match result {
 207              Safe => Safe,
 208              SafeIf(base_lp, base_vec) => {
 209                  let lp = @LpExtend(base_lp, mc, elem);
 210                  SafeIf(lp, vec::append_one(base_vec,
 211                                             Restriction {loan_path: lp,
 212                                                          set: restrictions}))
 213              }
 214          }
 215      }

 217      fn check_aliasing_permitted(&self,
 218                                  cause: mc::AliasableReason,
 219                                  restrictions: RestrictionSet) {
 220          //! This method is invoked when the current `cmt` is something
 221          //! where aliasing cannot be controlled. It reports an error if
 222          //! the restrictions required that it not be aliased; currently
 223          //! this only occurs when re-borrowing an `&mut` pointer.
 224          //!
 225          //! NB: To be 100% consistent, we should report an error if
 226          //! RESTR_FREEZE is found, because we cannot prevent freezing,
 227          //! nor would we want to. However, we do not report such an
 228          //! error, because this restriction only occurs when the user
 229          //! is creating an `&mut` pointer to immutable or read-only
 230          //! data, and there is already another piece of code that
 231          //! checks for this condition.
 232  
 233          if restrictions.intersects(RESTR_ALIAS) {
 234              self.bccx.report_aliasability_violation(
 235                  self.span,
 236                  BorrowViolation,
 237                  cause);
 238          }
 239      }
 240  
 241      fn check_no_mutability_control(&self,
 242                                     cmt: mc::cmt,
 243                                     restrictions: RestrictionSet) {
 244          if restrictions.intersects(RESTR_MUTATE | RESTR_FREEZE | RESTR_CLAIM) {
 245              self.bccx.report(BckError {span: self.span,
 246                                         cmt: cmt,
 247                                         code: err_freeze_aliasable_const});
 248          }
 249      }

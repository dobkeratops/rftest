  12  // Gathering loans
  13  //
  14  // The borrow check proceeds in two phases. In phase one, we gather the full
  15  // set of loans that are required at any point.  These are sorted according to
  16  // their associated scopes.  In phase two, checking loans, we will then make
  17  // sure that all of these loans are honored.
  18

  40  /// Context used while gathering loans:
  41  ///
  42  /// - `bccx`: the borrow check context
  43  /// - `item_ub`: the id of the block for the enclosing fn/method item
  44  /// - `root_ub`: the id of the outermost block for which we can root
  45  ///   an `@T`.  This is the id of the innermost enclosing
  46  ///   loop or function body.
  47  ///
  48  /// The role of `root_ub` is to prevent us from having to accumulate
  49  /// vectors of rooted items at runtime.  Consider this case:
  50  ///
  51  ///     fn foo(...) -> int {
  52  ///         let mut ptr: &int;
  53  ///         while some_cond {
  54  ///             let x: @int = ...;
  55  ///             ptr = &*x;
  56  ///         }
  57  ///         *ptr
  58  ///     }
  59  ///
  60  /// If we are not careful here, we would infer the scope of the borrow `&*x`
  61  /// to be the body of the function `foo()` as a whole.  We would then
  62  /// have root each `@int` that is produced, which is an unbounded number.
  63  /// No good.  Instead what will happen is that `root_ub` will be set to the
  64  /// body of the while loop and we will refuse to root the pointer `&*x`
  65  /// because it would have to be rooted for a region greater than `root_ub`.
  66  struct GatherLoanCtxt {
  67      bccx: @BorrowckCtxt,
  68      id_range: id_range,
  69      move_data: @mut move_data::MoveData,
  70      all_loans: @mut ~[Loan],
  71      item_ub: ast::NodeId,
  72      repeating_ids: ~[ast::NodeId]
  73  }
  74  
  75  pub fn gather_loans(bccx: @BorrowckCtxt,
  76                      decl: &ast::fn_decl,
  77                      body: &ast::Block)
  78                      -> (id_range, @mut ~[Loan], @mut move_data::MoveData) {
  79      let glcx = @mut GatherLoanCtxt {
  80          bccx: bccx,
  81          id_range: id_range::max(),
  82          all_loans: @mut ~[],
  83          item_ub: body.id,
  84          repeating_ids: ~[body.id],
  85          move_data: @mut MoveData::new()
  86      };
  87      glcx.gather_fn_arg_patterns(decl, body);
  88      let v = oldvisit::mk_vt(@oldvisit::Visitor {
  89          visit_expr: gather_loans_in_expr,
  90          visit_block: gather_loans_in_block,
  91          visit_fn: gather_loans_in_fn,
  92          visit_stmt: add_stmt_to_map,
  93          visit_pat: add_pat_to_id_range,
  94          visit_local: gather_loans_in_local,
  95          .. *oldvisit::default_visitor()
  96      });
  97      (v.visit_block)(body, (glcx, v));
  98      return (glcx.id_range, glcx.all_loans, glcx.move_data);
  99  }

 101  fn add_pat_to_id_range(p: @ast::pat,
 102                         (this, v): (@mut GatherLoanCtxt,
 103                                     oldvisit::vt<@mut GatherLoanCtxt>)) {
 104      // NB: This visitor function just adds the pat ids into the id
 105      // range. We gather loans that occur in patterns using the
 106      // `gather_pat()` method below. Eventually these two should be
 107      // brought together.
 108      this.id_range.add(p.id);
 109      oldvisit::visit_pat(p, (this, v));
 110  }

 112  fn gather_loans_in_fn(fk: &oldvisit::fn_kind,
 113                        decl: &ast::fn_decl,
 114                        body: &ast::Block,
 115                        sp: span,
 116                        id: ast::NodeId,
 117                        (this, v): (@mut GatherLoanCtxt,
 118                                    oldvisit::vt<@mut GatherLoanCtxt>)) {
 119      match fk {
 120          // Do not visit items here, the outer loop in borrowck/mod
 121          // will visit them for us in turn.
 122          &oldvisit::fk_item_fn(*) | &oldvisit::fk_method(*) => {
 123              return;
 124          }
 125  
 126          // Visit closures as part of the containing item.
 127          &oldvisit::fk_anon(*) | &oldvisit::fk_fn_block(*) => {
 128              this.push_repeating_id(body.id);
 129              oldvisit::visit_fn(fk, decl, body, sp, id, (this, v));
 130              this.pop_repeating_id(body.id);
 131              this.gather_fn_arg_patterns(decl, body);
 132          }
 133      }
 134  }

 143  fn gather_loans_in_local(local: @ast::Local,
 144                           (this, vt): (@mut GatherLoanCtxt,
 145                                        oldvisit::vt<@mut GatherLoanCtxt>)) {
 146      match local.init {
 147          None => {
 148              // Variable declarations without initializers are considered "moves":
 149              let tcx = this.bccx.tcx;
 150              do pat_util::pat_bindings(tcx.def_map, local.pat)
 151                  |_, id, span, _| {
 152                  gather_moves::gather_decl(this.bccx,
 153                                            this.move_data,
 154                                            id,
 155                                            span,
 156                                            id);
 157              }
 158          }
 159          Some(init) => {
 160              // Variable declarations with initializers are considered "assigns":
 161              let tcx = this.bccx.tcx;
 162              do pat_util::pat_bindings(tcx.def_map, local.pat)
 163                  |_, id, span, _| {
 164                  gather_moves::gather_assignment(this.bccx,
 165                                                  this.move_data,
 166                                                  id,
 167                                                  span,
 168                                                  @LpVar(id),
 169                                                  id);
 170              }
 171              let init_cmt = this.bccx.cat_expr(init);
 172              this.gather_pat(init_cmt, local.pat, None);
 173          }
 174      }
 175  
 176      oldvisit::visit_local(local, (this, vt));
 177  }

 179  fn gather_loans_in_expr(ex: @ast::expr,
 180                          (this, vt): (@mut GatherLoanCtxt,
 181                                       oldvisit::vt<@mut GatherLoanCtxt>)) {
 182      let bccx = this.bccx;
 183      let tcx = bccx.tcx;
 184  
 185      debug!("gather_loans_in_expr(expr=%?/%s)",
 186             ex.id, pprust::expr_to_str(ex, tcx.sess.intr()));
 187  
 188      this.id_range.add(ex.id);
 189  
 190      {
 191          let r = ex.get_callee_id();
 192          for callee_id in r.iter() {
 193              this.id_range.add(*callee_id);
 194          }
 195      }
 196  
 197      // If this expression is borrowed, have to ensure it remains valid:
 198      {
 199          let r = tcx.adjustments.find(&ex.id);
 200          for &adjustments in r.iter() {
 201              this.guarantee_adjustments(ex, *adjustments);
 202          }
 203      }
 204  
 205      // If this expression is a move, gather it:
 206      if this.bccx.is_move(ex.id) {
 207          let cmt = this.bccx.cat_expr(ex);
 208          gather_moves::gather_move_from_expr(
 209              this.bccx, this.move_data, ex, cmt);
 210      }
 211  
 212      // Special checks for various kinds of expressions:
 213      match ex.node {
 214        ast::expr_addr_of(mutbl, base) => {
 215          let base_cmt = this.bccx.cat_expr(base);
 216  
 217          // make sure that the thing we are pointing out stays valid
 218          // for the lifetime `scope_r` of the resulting ptr:
 219          let scope_r = ty_region(tcx, ex.span, ty::expr_ty(tcx, ex));
 220          this.guarantee_valid(ex.id, ex.span, base_cmt, mutbl, scope_r);
 221          oldvisit::visit_expr(ex, (this, vt));
 222        }
 223  
 224        ast::expr_assign(l, _) | ast::expr_assign_op(_, _, l, _) => {
 225            let l_cmt = this.bccx.cat_expr(l);
 226            match opt_loan_path(l_cmt) {
 227                Some(l_lp) => {
 228                    gather_moves::gather_assignment(this.bccx, this.move_data,
 229                                                    ex.id, ex.span,
 230                                                    l_lp, l.id);
 231                }
 232                None => {
 233                    // This can occur with e.g. `*foo() = 5`.  In such
 234                    // cases, there is no need to check for conflicts
 235                    // with moves etc, just ignore.
 236                }
 237            }
 238            oldvisit::visit_expr(ex, (this, vt));
 239        }
 240  
 241        ast::expr_match(ex_v, ref arms) => {
 242          let cmt = this.bccx.cat_expr(ex_v);
 243          for arm in arms.iter() {
 244              for pat in arm.pats.iter() {
 245                  this.gather_pat(cmt, *pat, Some((arm.body.id, ex.id)));
 246              }
 247          }
 248          oldvisit::visit_expr(ex, (this, vt));
 249        }
 250  
 251        ast::expr_index(_, _, arg) |
 252        ast::expr_binary(_, _, _, arg)
 253        if this.bccx.method_map.contains_key(&ex.id) => {
 254            // Arguments in method calls are always passed by ref.
 255            //
 256            // Currently these do not use adjustments, so we have to
 257            // hardcode this check here (note that the receiver DOES use
 258            // adjustments).
 259            let scope_r = ty::re_scope(ex.id);
 260            let arg_cmt = this.bccx.cat_expr(arg);
 261            this.guarantee_valid(arg.id, arg.span, arg_cmt, m_imm, scope_r);
 262            oldvisit::visit_expr(ex, (this, vt));
 263        }
 264  
 265        // see explanation attached to the `root_ub` field:
 266        ast::expr_while(cond, ref body) => {
 267            // during the condition, can only root for the condition
 268            this.push_repeating_id(cond.id);
 269            (vt.visit_expr)(cond, (this, vt));
 270            this.pop_repeating_id(cond.id);
 271  
 272            // during body, can only root for the body
 273            this.push_repeating_id(body.id);
 274            (vt.visit_block)(body, (this, vt));
 275            this.pop_repeating_id(body.id);
 276        }
 277  
 278        // see explanation attached to the `root_ub` field:
 279        ast::expr_loop(ref body, _) => {
 280            this.push_repeating_id(body.id);
 281            oldvisit::visit_expr(ex, (this, vt));
 282            this.pop_repeating_id(body.id);
 283        }
 284  
 285        ast::expr_fn_block(*) => {
 286            gather_moves::gather_captures(this.bccx, this.move_data, ex);
 287            oldvisit::visit_expr(ex, (this, vt));
 288        }
 289  
 290        _ => {
 291          oldvisit::visit_expr(ex, (this, vt));
 292        }
 293      }
 294  }

 308      pub fn guarantee_adjustments(&mut self,
 309                                   expr: @ast::expr,
 310                                   adjustment: &ty::AutoAdjustment) {
 311          debug!("guarantee_adjustments(expr=%s, adjustment=%?)",
 312                 expr.repr(self.tcx()), adjustment);
 313          let _i = indenter();
 314  
 315          match *adjustment {
 316              ty::AutoAddEnv(*) => {
 317                  debug!("autoaddenv -- no autoref");
 318                  return;
 319              }
 320  
 321              ty::AutoDerefRef(
 322                  ty::AutoDerefRef {
 323                      autoref: None, _ }) => {
 324                  debug!("no autoref");
 325                  return;
 326              }
 327  
 328              ty::AutoDerefRef(
 329                  ty::AutoDerefRef {
 330                      autoref: Some(ref autoref),
 331                      autoderefs: autoderefs}) => {
 332                  let mcx = &mc::mem_categorization_ctxt {
 333                      tcx: self.tcx(),
 334                      method_map: self.bccx.method_map};
 335                  let cmt = mcx.cat_expr_autoderefd(expr, autoderefs);
 336                  debug!("after autoderef, cmt=%s", cmt.repr(self.tcx()));
 337  
 338                  match *autoref {
 339                      ty::AutoPtr(r, m) => {
 340                          self.guarantee_valid(expr.id,
 341                                               expr.span,
 342                                               cmt,
 343                                               m,
 344                                               r)
 345                      }
 346                      ty::AutoBorrowVec(r, m) | ty::AutoBorrowVecRef(r, m) => {
 347                          let cmt_index = mcx.cat_index(expr, cmt, autoderefs+1);
 348                          self.guarantee_valid(expr.id,
 349                                               expr.span,
 350                                               cmt_index,
 351                                               m,
 352                                               r)
 353                      }
 354                      ty::AutoBorrowFn(r) => {
 355                          let cmt_deref = mcx.cat_deref_fn_or_obj(expr, cmt, 0);
 356                          self.guarantee_valid(expr.id,
 357                                               expr.span,
 358                                               cmt_deref,
 359                                               m_imm,
 360                                               r)
 361                      }
 362                      ty::AutoBorrowObj(r, m) => {
 363                          let cmt_deref = mcx.cat_deref_fn_or_obj(expr, cmt, 0);
 364                          self.guarantee_valid(expr.id,
 365                                               expr.span,
 366                                               cmt_deref,
 367                                               m,
 368                                               r)
 369                      }
 370                      ty::AutoUnsafe(_) => {}
 371                  }
 372              }
 373          }
 374      }

 376      // Guarantees that addr_of(cmt) will be valid for the duration of
 377      // `static_scope_r`, or reports an error.  This may entail taking
 378      // out loans, which will be added to the `req_loan_map`.  This can
 379      // also entail "rooting" GC'd pointers, which means ensuring
 380      // dynamically that they are not freed.
 381      pub fn guarantee_valid(&mut self,
 382                             borrow_id: ast::NodeId,
 383                             borrow_span: span,
 384                             cmt: mc::cmt,
 385                             req_mutbl: ast::mutability,
 386                             loan_region: ty::Region) {
 387          debug!("guarantee_valid(borrow_id=%?, cmt=%s, \
 388                  req_mutbl=%?, loan_region=%?)",
 389                 borrow_id,
 390                 cmt.repr(self.tcx()),
 391                 req_mutbl,
 392                 loan_region);
 393  
 394          // a loan for the empty region can never be dereferenced, so
 395          // it is always safe
 396          if loan_region == ty::re_empty {
 397              return;
 398          }
 399  
 400          let root_ub = { *self.repeating_ids.last() }; // FIXME(#5074)
 401  
 402          // Check that the lifetime of the borrow does not exceed
 403          // the lifetime of the data being borrowed.
 404          lifetime::guarantee_lifetime(self.bccx, self.item_ub, root_ub,
 405                                       borrow_span, cmt, loan_region, req_mutbl);
 406  
 407          // Check that we don't allow mutable borrows of non-mutable data.
 408          check_mutability(self.bccx, borrow_span, cmt, req_mutbl);
 409  
 410          // Compute the restrictions that are required to enforce the
 411          // loan is safe.
 412          let restr = restrictions::compute_restrictions(
 413              self.bccx, borrow_span,
 414              cmt, self.restriction_set(req_mutbl));
 415  
 416          // Create the loan record (if needed).
 417          let loan = match restr {
 418              restrictions::Safe => {
 419                  // No restrictions---no loan record necessary
 420                  return;
 421              }
 422  
 423              restrictions::SafeIf(loan_path, restrictions) => {
 424                  let loan_scope = match loan_region {
 425                      ty::re_scope(id) => id,

 588      pub fn compute_kill_scope(&self, loan_scope: ast::NodeId, lp: @LoanPath)

 591          //! This is either when the lifetime expires or when the
 592          //! local variable which roots the loan-path goes out of scope,

 595          //! It may seem surprising that we might have a loan region

 599          //!

 605          //! variable `v` that hosts it. Note that this doesn't come up
 606          //! with immutable `&` pointers, because borrows of such pointers
 607          //! do not require restrictions and hence do not cause a loan.

 610          let lexical_scope = rm.encl_scope(lp.node_id());

 612              lexical_scope
 613          } else {
 614              assert!(rm.is_subscope_of(loan_scope, lexical_scope));
 615              loan_scope
 616          }
 617      }

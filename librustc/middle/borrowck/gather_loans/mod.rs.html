  12  // Gathering loans
  13  //
  14  // The borrow check proceeds in two phases. In phase one, we gather the full
  15  // set of loans that are required at any point.  These are sorted according to
  16  // their associated scopes.  In phase two, checking loans, we will then make
  17  // sure that all of these loans are honored.

  44  /// Context used while gathering loans:
  45  ///
  46  /// - `bccx`: the borrow check context
  47  /// - `item_ub`: the id of the block for the enclosing fn/method item
  48  /// - `root_ub`: the id of the outermost block for which we can root
  49  ///   an `@T`.  This is the id of the innermost enclosing
  50  ///   loop or function body.
  51  ///
  52  /// The role of `root_ub` is to prevent us from having to accumulate
  53  /// vectors of rooted items at runtime.  Consider this case:
  54  ///
  55  ///     fn foo(...) -> int {
  56  ///         let mut ptr: &int;
  57  ///         while some_cond {
  58  ///             let x: @int = ...;
  59  ///             ptr = &*x;
  60  ///         }
  61  ///         *ptr
  62  ///     }
  63  ///
  64  /// If we are not careful here, we would infer the scope of the borrow `&*x`
  65  /// to be the body of the function `foo()` as a whole.  We would then
  66  /// have root each `@int` that is produced, which is an unbounded number.
  67  /// No good.  Instead what will happen is that `root_ub` will be set to the
  68  /// body of the while loop and we will refuse to root the pointer `&*x`
  69  /// because it would have to be rooted for a region greater than `root_ub`.
  70  struct GatherLoanCtxt<'a> {
  71      bccx: &'a BorrowckCtxt<'a>,
  72      id_range: IdRange,
  73      move_data: move_data::MoveData,
  74      move_error_collector: move_error::MoveErrorCollector,
  75      all_loans: Vec<Loan>,
  76      item_ub: ast::NodeId,
  77      repeating_ids: Vec<ast::NodeId> }

 143  fn gather_loans_in_local(this: &mut GatherLoanCtxt,
 144                           local: &ast::Local) {
 145      match local.init {
 146          None => {
 147              // Variable declarations without initializers are considered "moves":
 148              let tcx = this.bccx.tcx;
 149              pat_util::pat_bindings(tcx.def_map, local.pat, |_, id, span, _| {
 150                  gather_moves::gather_decl(this.bccx,
 151                                            &this.move_data,
 152                                            id,
 153                                            span,
 154                                            id);
 155              })
 156          }
 157          Some(init) => {
 158              // Variable declarations with initializers are considered "assigns",
 159              // which is handled by `gather_pat`:
 160              let init_cmt = this.bccx.cat_expr(init);
 161              this.gather_pat(init_cmt, local.pat, None);
 162          }
 163      }
 164  
 165      visit::walk_local(this, local, ());
 166  }

 168  fn gather_loans_in_expr(this: &mut GatherLoanCtxt,
 169                          ex: &ast::Expr) {
 170      let bccx = this.bccx;
 171      let tcx = bccx.tcx;
 172  
 173      debug!("gather_loans_in_expr(expr={:?}/{})",
 174             ex.id, pprust::expr_to_str(ex));
 175  
 176      this.id_range.add(ex.id);
 177  
 178      // If this expression is borrowed, have to ensure it remains valid:
 179      for &adjustments in tcx.adjustments.borrow().find(&ex.id).iter() {
 180          this.guarantee_adjustments(ex, *adjustments);
 181      }
 182  
 183      // If this expression is a move, gather it:
 184      if this.bccx.is_move(ex.id) {
 185          let cmt = this.bccx.cat_expr(ex);
 186          gather_moves::gather_move_from_expr(
 187              this.bccx, &this.move_data, &this.move_error_collector, ex, cmt);
 188      }
 189  
 190      // Special checks for various kinds of expressions:
 191      let method_map = this.bccx.method_map.borrow();
 192      match ex.node {
 193        ast::ExprAddrOf(mutbl, base) => {
 194          let base_cmt = this.bccx.cat_expr(base);
 195  
 196          // make sure that the thing we are pointing out stays valid
 197          // for the lifetime `scope_r` of the resulting ptr:
 198          let expr_ty = ty::expr_ty(tcx, ex);
 199          if !ty::type_is_bot(expr_ty) {
 200              let scope_r = ty_region(tcx, ex.span, expr_ty);
 201              this.guarantee_valid(ex.id,
 202                                   ex.span,
 203                                   base_cmt,
 204                                   mutbl,
 205                                   scope_r,
 206                                   AddrOf);
 207          }
 208          visit::walk_expr(this, ex, ());
 209        }
 210  
 211        ast::ExprAssign(l, _) => {
 212            with_assignee_loan_path(
 213                this.bccx, l,
 214                |lp| gather_moves::gather_assignment(this.bccx, &this.move_data,
 215                                                     ex.id, ex.span, lp, l.id));
 216            visit::walk_expr(this, ex, ());
 217        }
 218  
 219        ast::ExprAssignOp(_, l, _) => {
 220            with_assignee_loan_path(
 221                this.bccx, l,
 222                |lp| gather_moves::gather_move_and_assignment(this.bccx, &this.move_data,
 223                                                              ex.id, ex.span, lp, l.id));
 224            visit::walk_expr(this, ex, ());
 225        }
 226  
 227        ast::ExprMatch(ex_v, ref arms) => {
 228          let cmt = this.bccx.cat_expr(ex_v);
 229          for arm in arms.iter() {
 230              for pat in arm.pats.iter() {
 231                  this.gather_pat(cmt, *pat, Some((arm.body.id, ex.id)));
 232              }
 233          }
 234          visit::walk_expr(this, ex, ());
 235        }
 236  
 237        ast::ExprIndex(_, arg) |
 238        ast::ExprBinary(_, _, arg)
 239        if method_map.contains_key(&MethodCall::expr(ex.id)) => {
 240            // Arguments in method calls are always passed by ref.
 241            //
 242            // Currently these do not use adjustments, so we have to
 243            // hardcode this check here (note that the receiver DOES use
 244            // adjustments).
 245            let scope_r = ty::ReScope(ex.id);
 246            let arg_cmt = this.bccx.cat_expr(arg);
 247            this.guarantee_valid(arg.id,
 248                                 arg.span,
 249                                 arg_cmt,
 250                                 ast::MutImmutable,
 251                                 scope_r,
 252                                 AutoRef);
 253            visit::walk_expr(this, ex, ());
 254        }
 255  
 256        // see explanation attached to the `root_ub` field:
 257        ast::ExprWhile(cond, body) => {
 258            // during the condition, can only root for the condition
 259            this.push_repeating_id(cond.id);
 260            this.visit_expr(cond, ());
 261            this.pop_repeating_id(cond.id);
 262  
 263            // during body, can only root for the body
 264            this.push_repeating_id(body.id);
 265            this.visit_block(body, ());
 266            this.pop_repeating_id(body.id);
 267        }
 268  
 269        // see explanation attached to the `root_ub` field:
 270        ast::ExprLoop(body, _) => {
 271            this.push_repeating_id(body.id);
 272            visit::walk_expr(this, ex, ());
 273            this.pop_repeating_id(body.id);
 274        }
 275  
 276        ast::ExprFnBlock(..) | ast::ExprProc(..) => {
 277            gather_moves::gather_captures(this.bccx, &this.move_data,
 278                                          &this.move_error_collector, ex);
 279            this.guarantee_captures(ex);
 280            visit::walk_expr(this, ex, ());
 281        }
 282  
 283        ast::ExprInlineAsm(ref ia) => {
 284            for &(_, out) in ia.outputs.iter() {
 285                with_assignee_loan_path(
 286                    this.bccx, out,
 287                    |lp| gather_moves::gather_assignment(this.bccx, &this.move_data,
 288                                                         ex.id, ex.span, lp, out.id));
 289            }
 290            visit::walk_expr(this, ex, ());
 291        }
 292  
 293        _ => {
 294            visit::walk_expr(this, ex, ());
 295        }
 296      }
 297  }

 312  /// Implements the A-* rules in doc.rs.
 313  fn check_aliasability(bccx: &BorrowckCtxt,
 314                        borrow_span: Span,
 315                        loan_cause: LoanCause,
 316                        cmt: mc::cmt,
 317                        req_kind: ty::BorrowKind)
 318                        -> Result<(),()> {
 319  
 320      match (cmt.freely_aliasable(bccx.tcx), req_kind) {
 321          (None, _) => {
 322              /* Uniquely accessible path -- OK for `&` and `&mut` */
 323              Ok(())
 324          }
 325          (Some(mc::AliasableStatic(safety)), ty::ImmBorrow) => {
 326              // Borrow of an immutable static item:
 327              match safety {
 328                  mc::InteriorUnsafe => {
 329                      // If the static item contains an Unsafe<T>, it has interior mutability.
 330                      // In such cases, we cannot permit it to be borrowed, because the
 331                      // static item resides in immutable memory and mutating it would
 332                      // cause segfaults.
 333                      bccx.tcx.sess.span_err(borrow_span,
 334                                             format!("borrow of immutable static items with \
 335                                                      unsafe interior is not allowed"));
 336                      Err(())
 337                  }
 338                  mc::InteriorSafe => {
 339                      // Immutable static can be borrowed, no problem.
 340                      Ok(())
 341                  }
 342              }
 343          }
 344          (Some(mc::AliasableStaticMut(..)), _) => {
 345              // Even touching a static mut is considered unsafe. We assume the
 346              // user knows what they're doing in these cases.
 347              Ok(())
 348          }
 349          (Some(alias_cause), ty::UniqueImmBorrow) |
 350          (Some(alias_cause), ty::MutBorrow) => {
 351              bccx.report_aliasability_violation(
 352                          borrow_span,
 353                          BorrowViolation(loan_cause),
 354                          alias_cause);
 355              Err(())
 356          }
 357          (_, _) => {
 358              Ok(())
 359          }
 360      }
 361  }

 375      pub fn guarantee_autoderefs(&mut self,
 376                                  expr: &ast::Expr,
 377                                  autoderefs: uint) {
 378          let method_map = self.bccx.method_map.borrow();
 379          for i in range(0, autoderefs) {
 380              match method_map.find(&MethodCall::autoderef(expr.id, i as u32)) {
 381                  Some(method) => {
 382                      // Treat overloaded autoderefs as if an AutoRef adjustment
 383                      // was applied on the base type, as that is always the case.
 384                      let mut mc = self.bccx.mc();
 385                      let cmt = match mc.cat_expr_autoderefd(expr, i) {
 386                          Ok(v) => v,
 387                          Err(()) => self.tcx().sess.span_bug(expr.span, "Err from mc")
 388                      };
 389                      let self_ty = *ty::ty_fn_args(method.ty).get(0);
 390                      let (m, r) = match ty::get(self_ty).sty {
 391                          ty::ty_rptr(r, ref m) => (m.mutbl, r),
 392                          _ => self.tcx().sess.span_bug(expr.span,
 393                                  format!("bad overloaded deref type {}",
 394                                      method.ty.repr(self.tcx())))
 395                      };
 396                      self.guarantee_valid(expr.id,
 397                                           expr.span,
 398                                           cmt,
 399                                           m,
 400                                           r,
 401                                           AutoRef);
 402                  }
 403                  None => {}
 404              }
 405          }
 406      }

 408      pub fn guarantee_adjustments(&mut self,
 409                                   expr: &ast::Expr,
 410                                   adjustment: &ty::AutoAdjustment) {
 411          debug!("guarantee_adjustments(expr={}, adjustment={:?})",
 412                 expr.repr(self.tcx()), adjustment);
 413          let _i = indenter();
 414  
 415          match *adjustment {
 416              ty::AutoAddEnv(..) => {
 417                  debug!("autoaddenv -- no autoref");
 418                  return;
 419              }
 420

 526                              loan_region: ty::Region,
 527                              cause: LoanCause) {
 528          /*!
 529           * Guarantees that `addr_of(cmt)` will be valid for the duration of
 530           * `static_scope_r`, or reports an error.  This may entail taking
 531           * out loans, which will be added to the `req_loan_map`.  This can
 532           * also entail "rooting" GC'd pointers, which means ensuring
 533           * dynamically that they are not freed.
 534           */
 535  
 536          debug!("guarantee_valid(borrow_id={:?}, cmt={}, \
 537                  req_mutbl={:?}, loan_region={:?})",
 538                 borrow_id,

 540                 req_kind,
 541                 loan_region);

 543          // a loan for the empty region can never be dereferenced, so
 544          // it is always safe
 545          if loan_region == ty::ReEmpty {
 546              return;
 547          }
 548  
 549          let root_ub = { *self.repeating_ids.last().unwrap() }; // FIXME(#5074)

 552          // the lifetime of the data being borrowed.

 554                                          borrow_span, cause, cmt, loan_region,

 556              return; // reported an error, no sense in reporting more.
 557          }
 558  
 559          // Check that we don't allow mutable borrows of non-mutable data.

 572          // loan is safe.

 579              restrictions::Safe => {

 581                  return;

 588  
 589                      ty::ReStatic => {
 590                          // If we get here, an error must have been
 591                          // reported in

 593                          // the only legal ways to have a borrow with a
 594                          // static lifetime should not require
 595                          // restrictions. To avoid reporting derived

 597                          // any loans.
 598                          return;
 599                      }

 602                      ty::ReLateBound(..) |
 603                      ty::ReEarlyBound(..) |
 604                      ty::ReInfer(..) => {

 623                      index: self.all_loans.len(),

 627                      gen_scope: gen_scope,
 628                      kill_scope: kill_scope,
 629                      span: borrow_span,

 632                  }
 633              }
 634          };

 670                              borrow_span: Span,
 671                              cause: LoanCause,
 672                              cmt: mc::cmt,
 673                              req_kind: ty::BorrowKind)
 674                              -> Result<(),()> {
 675              //! Implements the M-* rules in doc.rs.
 676  
 677              match req_kind {
 678                  ty::UniqueImmBorrow | ty::ImmBorrow => {
 679                      match cmt.mutbl {
 680                          // I am intentionally leaving this here to help
 681                          // refactoring if, in the future, we should add new

 683                          mc::McImmutable | mc::McDeclared | mc::McInherited => {

 685                              // for mutable data, this is a freeze
 686                              Ok(())
 687                          }
 688                      }
 689                  }
 690  
 691                  ty::MutBorrow => {
 692                      // Only mutable data can be lent as mutable.
 693                      if !cmt.mutbl.is_mutable() {

 698                      } else {
 699                          Ok(())
 700                      }
 701                  }
 702              }
 703          }

 739      pub fn compute_gen_scope(&self,
 740                               borrow_id: ast::NodeId,

 745          //! notably method arguments, the loan may be introduced only
 746          //! later, once it comes into scope.
 747

 749              borrow_id
 750          } else {
 751              loan_scope

 755      pub fn compute_kill_scope(&self, loan_scope: ast::NodeId, lp: @LoanPath)
 756                                -> ast::NodeId {
 757          //! Determine when the loan restrictions go out of scope.
 758          //! This is either when the lifetime expires or when the
 759          //! local variable which roots the loan-path goes out of scope,
 760          //! whichever happens faster.
 761          //!
 762          //! It may seem surprising that we might have a loan region
 763          //! larger than the variable which roots the loan-path; this can
 764          //! come about when variables of `&mut` type are re-borrowed,
 765          //! as in this example:
 766          //!
 767          //!     fn counter<'a>(v: &'a mut Foo) -> &'a mut uint {
 768          //!         &mut v.counter
 769          //!     }
 770          //!
 771          //! In this case, the reference (`'a`) outlives the
 772          //! variable `v` that hosts it. Note that this doesn't come up
 773          //! with immutable `&` pointers, because borrows of such pointers
 774          //! do not require restrictions and hence do not cause a loan.
 775  
 776          let rm = &self.bccx.tcx.region_maps;
 777          let lexical_scope = rm.var_scope(lp.node_id());
 778          if rm.is_subscope_of(lexical_scope, loan_scope) {
 779              lexical_scope
 780          } else {
 781              assert!(self.bccx.tcx.region_maps.is_subscope_of(loan_scope, lexical_scope));
 782              loan_scope
 783          }
 784      }

 786      fn gather_fn_arg_patterns(&mut self,
 787                                decl: &ast::FnDecl,
 788                                body: &ast::Block) {
 789          /*!
 790           * Walks the patterns for fn arguments, checking that they
 791           * do not attempt illegal moves or create refs that outlive
 792           * the arguments themselves. Just a shallow wrapper around
 793           * `gather_pat()`.
 794           */
 795  
 796          let mut mc = self.bccx.mc();
 797          for arg in decl.inputs.iter() {
 798              let arg_ty = ty::node_id_to_type(self.tcx(), arg.pat.id);
 799  
 800              let arg_cmt = mc.cat_rvalue(
 801                  arg.id,
 802                  arg.pat.span,
 803                  ty::ReScope(body.id), // Args live only as long as the fn body.
 804                  arg_ty);
 805  
 806              self.gather_pat(arg_cmt, arg.pat, None);
 807          }
 808      }

 810      fn gather_pat(&mut self,
 811                    discr_cmt: mc::cmt,
 812                    root_pat: @ast::Pat,
 813                    arm_match_ids: Option<(ast::NodeId, ast::NodeId)>) {
 814          /*!
 815           * Walks patterns, examining the bindings to determine if they
 816           * cause borrows (`ref` bindings, vector patterns) or
 817           * moves (non-`ref` bindings with linear type).
 818           */
 819  
 820          self.bccx.cat_pattern(discr_cmt, root_pat, |cmt, pat| {
 821              match pat.node {
 822                ast::PatIdent(bm, _, _) if self.pat_is_binding(pat) => {
 823                  // Each match binding is effectively an assignment.
 824                  let tcx = self.bccx.tcx;
 825                  pat_util::pat_bindings(tcx.def_map, pat, |_, id, span, _| {
 826                      gather_moves::gather_assignment(self.bccx,
 827                                                      &self.move_data,
 828                                                      id,
 829                                                      span,
 830                                                      @LpVar(id),
 831                                                      id);
 832                  });
 833  
 834                  match bm {
 835                    ast::BindByRef(mutbl) => {
 836                      // ref x or ref x @ p --- creates a ptr which must
 837                      // remain valid for the scope of the match
 838  
 839                      // find the region of the resulting pointer (note that
 840                      // the type of such a pattern will *always* be a
 841                      // region pointer)
 842                      let scope_r =
 843                          ty_region(self.tcx(), pat.span,
 844                                    ty::node_id_to_type(self.tcx(), pat.id));
 845  
 846                      // if the scope of the region ptr turns out to be
 847                      // specific to this arm, wrap the categorization
 848                      // with a cat_discr() node.  There is a detailed
 849                      // discussion of the function of this node in
 850                      // `lifetime.rs`:
 851                      let cmt_discr = match arm_match_ids {
 852                          None => cmt,
 853                          Some((arm_id, match_id)) => {
 854                              let arm_scope = ty::ReScope(arm_id);

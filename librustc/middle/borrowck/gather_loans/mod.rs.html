  12  // Gathering loans
  13  //
  14  // The borrow check proceeds in two phases. In phase one, we gather the full
  15  // set of loans that are required at any point.  These are sorted according to
  16  // their associated scopes.  In phase two, checking loans, we will then make
  17  // sure that all of these loans are honored.

  43  /// Context used while gathering loans:
  44  ///
  45  /// - `bccx`: the borrow check context
  46  /// - `item_ub`: the id of the block for the enclosing fn/method item
  47  /// - `root_ub`: the id of the outermost block for which we can root
  48  ///   an `@T`.  This is the id of the innermost enclosing
  49  ///   loop or function body.
  50  ///
  51  /// The role of `root_ub` is to prevent us from having to accumulate
  52  /// vectors of rooted items at runtime.  Consider this case:
  53  ///
  54  ///     fn foo(...) -> int {
  55  ///         let mut ptr: &int;
  56  ///         while some_cond {
  57  ///             let x: @int = ...;
  58  ///             ptr = &*x;
  59  ///         }
  60  ///         *ptr
  61  ///     }
  62  ///
  63  /// If we are not careful here, we would infer the scope of the borrow `&*x`
  64  /// to be the body of the function `foo()` as a whole.  We would then
  65  /// have root each `@int` that is produced, which is an unbounded number.
  66  /// No good.  Instead what will happen is that `root_ub` will be set to the
  67  /// body of the while loop and we will refuse to root the pointer `&*x`
  68  /// because it would have to be rooted for a region greater than `root_ub`.
  69  struct GatherLoanCtxt<'a> {
  70      bccx: &'a BorrowckCtxt<'a>,
  71      id_range: IdRange,
  72      move_data: move_data::MoveData,
  73      all_loans: Vec<Loan>,
  74      item_ub: ast::NodeId,
  75      repeating_ids: Vec<ast::NodeId> }
  76  
  77  impl<'a> visit::Visitor<()> for GatherLoanCtxt<'a> {

  98      }

 104  }

 108      // NB: This visitor function just adds the pat ids into the id

 116  pub fn gather_loans_in_fn(bccx: &BorrowckCtxt, decl: &ast::FnDecl, body: &ast::Block)
 117                      -> (IdRange, Vec<Loan>, move_data::MoveData) {
 118      let mut glcx = GatherLoanCtxt {
 119          bccx: bccx,
 120          id_range: IdRange::max(),
 121          all_loans: Vec::new(),
 122          item_ub: body.id,
 123          repeating_ids: vec!(body.id),
 124          move_data: MoveData::new()
 125      };
 126      glcx.gather_fn_arg_patterns(decl, body);
 127  
 128      glcx.visit_block(body, ());
 129      let GatherLoanCtxt { id_range, all_loans, move_data, .. } = glcx;
 130      (id_range, all_loans, move_data)
 131  }

 139  fn gather_loans_in_local(this: &mut GatherLoanCtxt,
 140                           local: &ast::Local) {
 141      match local.init {
 142          None => {
 143              // Variable declarations without initializers are considered "moves":
 144              let tcx = this.bccx.tcx;
 145              pat_util::pat_bindings(tcx.def_map, local.pat, |_, id, span, _| {
 146                  gather_moves::gather_decl(this.bccx,
 147                                            &this.move_data,
 148                                            id,
 149                                            span,
 150                                            id);
 151              })
 152          }
 153          Some(init) => {
 154              // Variable declarations with initializers are considered "assigns",
 155              // which is handled by `gather_pat`:
 156              let init_cmt = this.bccx.cat_expr(init);
 157              this.gather_pat(init_cmt, local.pat, None);
 158          }
 159      }
 160  
 161      visit::walk_local(this, local, ());
 162  }

 164  fn gather_loans_in_expr(this: &mut GatherLoanCtxt,
 165                          ex: &ast::Expr) {
 166      let bccx = this.bccx;
 167      let tcx = bccx.tcx;
 168  
 169      debug!("gather_loans_in_expr(expr={:?}/{})",
 170             ex.id, pprust::expr_to_str(ex));
 171  
 172      this.id_range.add(ex.id);
 173  
 174      // If this expression is borrowed, have to ensure it remains valid:
 175      for &adjustments in tcx.adjustments.borrow().find(&ex.id).iter() {
 176          this.guarantee_adjustments(ex, *adjustments);
 177      }
 178  
 179      // If this expression is a move, gather it:
 180      if this.bccx.is_move(ex.id) {
 181          let cmt = this.bccx.cat_expr(ex);
 182          gather_moves::gather_move_from_expr(
 183              this.bccx, &this.move_data, ex, cmt);
 184      }
 185  
 186      // Special checks for various kinds of expressions:
 187      let method_map = this.bccx.method_map.borrow();
 188      match ex.node {
 189        ast::ExprAddrOf(mutbl, base) => {
 190          let base_cmt = this.bccx.cat_expr(base);
 191  
 192          // make sure that the thing we are pointing out stays valid
 193          // for the lifetime `scope_r` of the resulting ptr:
 194          let expr_ty = ty::expr_ty(tcx, ex);
 195          if !ty::type_is_bot(expr_ty) {
 196              let scope_r = ty_region(tcx, ex.span, expr_ty);
 197              this.guarantee_valid(ex.id,
 198                                   ex.span,
 199                                   base_cmt,
 200                                   mutbl,
 201                                   scope_r,
 202                                   AddrOf);
 203          }
 204          visit::walk_expr(this, ex, ());
 205        }
 206  
 207        ast::ExprAssign(l, _) => {
 208            with_assignee_loan_path(
 209                this.bccx, l,
 210                |lp| gather_moves::gather_assignment(this.bccx, &this.move_data,
 211                                                     ex.id, ex.span, lp, l.id));
 212            visit::walk_expr(this, ex, ());
 213        }
 214  
 215        ast::ExprAssignOp(_, l, _) => {
 216            with_assignee_loan_path(
 217                this.bccx, l,
 218                |lp| gather_moves::gather_move_and_assignment(this.bccx, &this.move_data,
 219                                                              ex.id, ex.span, lp, l.id));
 220            visit::walk_expr(this, ex, ());
 221        }
 222  
 223        ast::ExprMatch(ex_v, ref arms) => {
 224          let cmt = this.bccx.cat_expr(ex_v);
 225          for arm in arms.iter() {
 226              for pat in arm.pats.iter() {
 227                  this.gather_pat(cmt, *pat, Some((arm.body.id, ex.id)));
 228              }
 229          }
 230          visit::walk_expr(this, ex, ());
 231        }
 232  
 233        ast::ExprIndex(_, arg) |
 234        ast::ExprBinary(_, _, arg)
 235        if method_map.contains_key(&MethodCall::expr(ex.id)) => {
 236            // Arguments in method calls are always passed by ref.
 237            //
 238            // Currently these do not use adjustments, so we have to
 239            // hardcode this check here (note that the receiver DOES use
 240            // adjustments).
 241            let scope_r = ty::ReScope(ex.id);
 242            let arg_cmt = this.bccx.cat_expr(arg);
 243            this.guarantee_valid(arg.id,
 244                                 arg.span,
 245                                 arg_cmt,
 246                                 ast::MutImmutable,
 247                                 scope_r,
 248                                 AutoRef);
 249            visit::walk_expr(this, ex, ());
 250        }
 251  
 252        // see explanation attached to the `root_ub` field:
 253        ast::ExprWhile(cond, body) => {
 254            // during the condition, can only root for the condition
 255            this.push_repeating_id(cond.id);
 256            this.visit_expr(cond, ());
 257            this.pop_repeating_id(cond.id);
 258  
 259            // during body, can only root for the body
 260            this.push_repeating_id(body.id);
 261            this.visit_block(body, ());
 262            this.pop_repeating_id(body.id);
 263        }
 264  
 265        // see explanation attached to the `root_ub` field:
 266        ast::ExprLoop(body, _) => {
 267            this.push_repeating_id(body.id);
 268            visit::walk_expr(this, ex, ());
 269            this.pop_repeating_id(body.id);
 270        }
 271  
 272        ast::ExprFnBlock(..) | ast::ExprProc(..) => {
 273            gather_moves::gather_captures(this.bccx, &this.move_data, ex);
 274            this.guarantee_captures(ex);
 275            visit::walk_expr(this, ex, ());
 276        }
 277  
 278        ast::ExprInlineAsm(ref ia) => {
 279            for &(_, out) in ia.outputs.iter() {
 280                with_assignee_loan_path(
 281                    this.bccx, out,
 282                    |lp| gather_moves::gather_assignment(this.bccx, &this.move_data,
 283                                                         ex.id, ex.span, lp, out.id));
 284            }
 285            visit::walk_expr(this, ex, ());
 286        }
 287  
 288        _ => {
 289            visit::walk_expr(this, ex, ());
 290        }
 291      }
 292  }

 294  fn with_assignee_loan_path(bccx: &BorrowckCtxt, expr: &ast::Expr, op: |@LoanPath|) {
 295      let cmt = bccx.cat_expr(expr);
 296      match opt_loan_path(cmt) {
 297          Some(lp) => op(lp),
 298          None => {
 299              // This can occur with e.g. `*foo() = 5`.  In such
 300              // cases, there is no need to check for conflicts
 301              // with moves etc, just ignore.
 302          }
 303      }
 304  }

 307  /// Implements the A-* rules in doc.rs.
 308  fn check_aliasability(bccx: &BorrowckCtxt,
 309                        borrow_span: Span,
 310                        loan_cause: LoanCause,
 311                        cmt: mc::cmt,
 312                        req_kind: ty::BorrowKind)
 313                        -> Result<(),()> {
 314  
 315      match (cmt.freely_aliasable(bccx.tcx), req_kind) {
 316          (None, _) => {
 317              /* Uniquely accessible path -- OK for `&` and `&mut` */
 318              Ok(())
 319          }
 320          (Some(mc::AliasableStatic(safety)), ty::ImmBorrow) => {
 321              // Borrow of an immutable static item:
 322              match safety {
 323                  mc::InteriorUnsafe => {
 324                      // If the static item contains an Unsafe<T>, it has interior mutability.
 325                      // In such cases, we cannot permit it to be borrowed, because the
 326                      // static item resides in immutable memory and mutating it would
 327                      // cause segfaults.
 328                      bccx.tcx.sess.span_err(borrow_span,
 329                                             format!("borrow of immutable static items with \
 330                                                      unsafe interior is not allowed"));
 331                      Err(())
 332                  }
 333                  mc::InteriorSafe => {
 334                      // Immutable static can be borrowed, no problem.
 335                      Ok(())
 336                  }
 337              }
 338          }
 339          (Some(mc::AliasableStaticMut(..)), _) => {
 340              // Even touching a static mut is considered unsafe. We assume the
 341              // user knows what they're doing in these cases.
 342              Ok(())
 343          }
 344          (Some(alias_cause), ty::UniqueImmBorrow) |
 345          (Some(alias_cause), ty::MutBorrow) => {
 346              bccx.report_aliasability_violation(
 347                          borrow_span,
 348                          BorrowViolation(loan_cause),
 349                          alias_cause);
 350              Err(())
 351          }
 352          (_, _) => {
 353              Ok(())
 354          }
 355      }
 356  }

 370      pub fn guarantee_autoderefs(&mut self,
 371                                  expr: &ast::Expr,
 372                                  autoderefs: uint) {
 373          let method_map = self.bccx.method_map.borrow();
 374          for i in range(0, autoderefs) {
 375              match method_map.find(&MethodCall::autoderef(expr.id, i as u32)) {
 376                  Some(method) => {
 377                      // Treat overloaded autoderefs as if an AutoRef adjustment
 378                      // was applied on the base type, as that is always the case.
 379                      let mut mc = self.bccx.mc();
 380                      let cmt = match mc.cat_expr_autoderefd(expr, i) {
 381                          Ok(v) => v,
 382                          Err(()) => self.tcx().sess.span_bug(expr.span, "Err from mc")
 383                      };
 384                      let self_ty = *ty::ty_fn_args(method.ty).get(0);
 385                      let (m, r) = match ty::get(self_ty).sty {
 386                          ty::ty_rptr(r, ref m) => (m.mutbl, r),
 387                          _ => self.tcx().sess.span_bug(expr.span,
 388                                  format!("bad overloaded deref type {}",
 389                                      method.ty.repr(self.tcx())))
 390                      };
 391                      self.guarantee_valid(expr.id,
 392                                           expr.span,
 393                                           cmt,
 394                                           m,
 395                                           r,
 396                                           AutoRef);
 397                  }
 398                  None => {}
 399              }
 400          }
 401      }

 403      pub fn guarantee_adjustments(&mut self,
 404                                   expr: &ast::Expr,
 405                                   adjustment: &ty::AutoAdjustment) {
 406          debug!("guarantee_adjustments(expr={}, adjustment={:?})",
 407                 expr.repr(self.tcx()), adjustment);
 408          let _i = indenter();
 409  
 410          match *adjustment {
 411              ty::AutoAddEnv(..) => {
 412                  debug!("autoaddenv -- no autoref");
 413                  return;
 414              }
 415  
 416              ty::AutoDerefRef(
 417                  ty::AutoDerefRef {
 418                      autoref: None, autoderefs }) => {
 419                  debug!("no autoref");
 420                  self.guarantee_autoderefs(expr, autoderefs);
 421                  return;
 422              }
 423  
 424              ty::AutoDerefRef(
 425                  ty::AutoDerefRef {
 426                      autoref: Some(ref autoref),
 427                      autoderefs}) => {
 428                  self.guarantee_autoderefs(expr, autoderefs);

 525      fn guarantee_valid_kind(&mut self,
 526                              borrow_id: ast::NodeId,

 529                              req_kind: ty::BorrowKind,
 530                              loan_region: ty::Region,
 531                              cause: LoanCause) {
 532          /*!
 533           * Guarantees that `addr_of(cmt)` will be valid for the duration of
 534           * `static_scope_r`, or reports an error.  This may entail taking
 535           * out loans, which will be added to the `req_loan_map`.  This can
 536           * also entail "rooting" GC'd pointers, which means ensuring
 537           * dynamically that they are not freed.
 538           */
 539  
 540          debug!("guarantee_valid(borrow_id={:?}, cmt={}, \
 541                  req_mutbl={:?}, loan_region={:?})",
 542                 borrow_id,
 543                 cmt.repr(self.tcx()),
 544                 req_kind,
 545                 loan_region);
 546  
 547          // a loan for the empty region can never be dereferenced, so
 548          // it is always safe
 549          if loan_region == ty::ReEmpty {
 550              return;
 551          }
 552  
 553          let root_ub = { *self.repeating_ids.last().unwrap() }; // FIXME(#5074)
 554  
 555          // Check that the lifetime of the borrow does not exceed
 556          // the lifetime of the data being borrowed.
 557          if lifetime::guarantee_lifetime(self.bccx, self.item_ub, root_ub,
 558                                          borrow_span, cause, cmt, loan_region,
 559                                          req_kind).is_err() {
 560              return; // reported an error, no sense in reporting more.
 561          }
 562  
 563          // Check that we don't allow mutable borrows of non-mutable data.
 564          if check_mutability(self.bccx, borrow_span, cause,
 565                              cmt, req_kind).is_err() {
 566              return; // reported an error, no sense in reporting more.
 567          }
 568  
 569          // Check that we don't allow mutable borrows of aliasable data.
 570          if check_aliasability(self.bccx, borrow_span, cause,
 571                                cmt, req_kind).is_err() {
 572              return; // reported an error, no sense in reporting more.
 573          }
 574

 576          // loan is safe.
 577          let restr = restrictions::compute_restrictions(
 578              self.bccx, borrow_span, cause,
 579              cmt, loan_region, self.restriction_set(req_kind));
 580

 582          let loan = match restr {
 583              restrictions::Safe => {
 584                  // No restrictions---no loan record necessary
 585                  return;
 586              }
 587  
 588              restrictions::SafeIf(loan_path, restrictions) => {
 589                  let loan_scope = match loan_region {
 590                      ty::ReScope(id) => id,
 591                      ty::ReFree(ref fr) => fr.scope_id,
 592  
 593                      ty::ReStatic => {
 594                          // If we get here, an error must have been
 595                          // reported in
 596                          // `lifetime::guarantee_lifetime()`, because
 597                          // the only legal ways to have a borrow with a
 598                          // static lifetime should not require
 599                          // restrictions. To avoid reporting derived
 600                          // errors, we just return here without adding
 601                          // any loans.
 602                          return;
 603                      }
 604

 606                      ty::ReLateBound(..) |
 607                      ty::ReEarlyBound(..) |
 608                      ty::ReInfer(..) => {
 609                          self.tcx().sess.span_bug(
 610                              cmt.span,
 611                              format!("invalid borrow lifetime: {:?}", loan_region));
 612                      }
 613                  };

 616                  let gen_scope = self.compute_gen_scope(borrow_id, loan_scope);

 621  
 622                  if req_kind == ty::MutBorrow {
 623                      self.mark_loan_path_as_mutated(loan_path);
 624                  }

 626                  Loan {
 627                      index: self.all_loans.len(),
 628                      loan_path: loan_path,
 629                      cmt: cmt,
 630                      kind: req_kind,
 631                      gen_scope: gen_scope,
 632                      kill_scope: kill_scope,
 633                      span: borrow_span,
 634                      restrictions: restrictions,
 635                      cause: cause,
 636                  }
 637              }
 638          };
 639  
 640          debug!("guarantee_valid(borrow_id={:?}), loan={}",
 641                 borrow_id, loan.repr(self.tcx()));
 642  
 643          // let loan_path = loan.loan_path;
 644          // let loan_gen_scope = loan.gen_scope;

 647

 649              // FIXME(#6268) Nested method calls

 653              // This is a subtle case. See the test case
 654              // <compile-fail/borrowck-bad-nested-calls-free.rs>
 655              // to see what we are guarding against.
 656  
 657              //let restr = restrictions::compute_restrictions(

 659              //let loan = {

 661              //    Loan {

 673          fn check_mutability(bccx: &BorrowckCtxt,
 674                              borrow_span: Span,
 675                              cause: LoanCause,
 676                              cmt: mc::cmt,
 677                              req_kind: ty::BorrowKind)
 678                              -> Result<(),()> {
 679              //! Implements the M-* rules in doc.rs.
 680  
 681              match req_kind {
 682                  ty::UniqueImmBorrow | ty::ImmBorrow => {
 683                      match cmt.mutbl {
 684                          // I am intentionally leaving this here to help
 685                          // refactoring if, in the future, we should add new
 686                          // kinds of mutability.
 687                          mc::McImmutable | mc::McDeclared | mc::McInherited => {
 688                              // both imm and mut data can be lent as imm;
 689                              // for mutable data, this is a freeze
 690                              Ok(())
 691                          }
 692                      }
 693                  }
 694  
 695                  ty::MutBorrow => {
 696                      // Only mutable data can be lent as mutable.
 697                      if !cmt.mutbl.is_mutable() {

 699                                                     cause: cause,

 701                                                     code: err_mutbl }))
 702                      } else {
 703                          Ok(())
 704                      }
 705                  }
 706              }
 707          }

 743      pub fn compute_gen_scope(&self,
 744                               borrow_id: ast::NodeId,
 745                               loan_scope: ast::NodeId)
 746                               -> ast::NodeId {
 747          //! Determine when to introduce the loan. Typically the loan
 748          //! is introduced at the point of the borrow, but in some cases,
 749          //! notably method arguments, the loan may be introduced only
 750          //! later, once it comes into scope.
 751  
 752          if self.bccx.tcx.region_maps.is_subscope_of(borrow_id, loan_scope) {
 753              borrow_id
 754          } else {
 755              loan_scope
 756          }
 757      }

 759      pub fn compute_kill_scope(&self, loan_scope: ast::NodeId, lp: @LoanPath)
 760                                -> ast::NodeId {
 761          //! Determine when the loan restrictions go out of scope.
 762          //! This is either when the lifetime expires or when the
 763          //! local variable which roots the loan-path goes out of scope,
 764          //! whichever happens faster.
 765          //!
 766          //! It may seem surprising that we might have a loan region
 767          //! larger than the variable which roots the loan-path; this can
 768          //! come about when variables of `&mut` type are re-borrowed,
 769          //! as in this example:
 770          //!
 771          //!     fn counter<'a>(v: &'a mut Foo) -> &'a mut uint {
 772          //!         &mut v.counter
 773          //!     }
 774          //!
 775          //! In this case, the reference (`'a`) outlives the
 776          //! variable `v` that hosts it. Note that this doesn't come up
 777          //! with immutable `&` pointers, because borrows of such pointers
 778          //! do not require restrictions and hence do not cause a loan.
 779  
 780          let rm = &self.bccx.tcx.region_maps;
 781          let lexical_scope = rm.var_scope(lp.node_id());
 782          if rm.is_subscope_of(lexical_scope, loan_scope) {
 783              lexical_scope
 784          } else {
 785              assert!(self.bccx.tcx.region_maps.is_subscope_of(loan_scope, lexical_scope));
 786              loan_scope
 787          }
 788      }

 790      fn gather_fn_arg_patterns(&mut self,
 791                                decl: &ast::FnDecl,
 792                                body: &ast::Block) {
 793          /*!
 794           * Walks the patterns for fn arguments, checking that they
 795           * do not attempt illegal moves or create refs that outlive
 796           * the arguments themselves. Just a shallow wrapper around
 797           * `gather_pat()`.
 798           */
 799  
 800          let mut mc = self.bccx.mc();
 801          for arg in decl.inputs.iter() {
 802              let arg_ty = ty::node_id_to_type(self.tcx(), arg.pat.id);
 803  
 804              let arg_cmt = mc.cat_rvalue(
 805                  arg.id,
 806                  arg.pat.span,
 807                  ty::ReScope(body.id), // Args live only as long as the fn body.
 808                  arg_ty);
 809  
 810              self.gather_pat(arg_cmt, arg.pat, None);
 811          }
 812      }

 814      fn gather_pat(&mut self,
 815                    discr_cmt: mc::cmt,
 816                    root_pat: @ast::Pat,
 817                    arm_match_ids: Option<(ast::NodeId, ast::NodeId)>) {
 818          /*!
 819           * Walks patterns, examining the bindings to determine if they
 820           * cause borrows (`ref` bindings, vector patterns) or
 821           * moves (non-`ref` bindings with linear type).
 822           */
 823  
 824          self.bccx.cat_pattern(discr_cmt, root_pat, |cmt, pat| {
 825              match pat.node {
 826                ast::PatIdent(bm, _, _) if self.pat_is_binding(pat) => {
 827                  // Each match binding is effectively an assignment.
 828                  let tcx = self.bccx.tcx;
 829                  pat_util::pat_bindings(tcx.def_map, pat, |_, id, span, _| {
 830                      gather_moves::gather_assignment(self.bccx,
 831                                                      &self.move_data,
 832                                                      id,
 833                                                      span,
 834                                                      @LpVar(id),
 835                                                      id);
 836                  });
 837  
 838                  match bm {
 839                    ast::BindByRef(mutbl) => {
 840                      // ref x or ref x @ p --- creates a ptr which must
 841                      // remain valid for the scope of the match
 842  
 843                      // find the region of the resulting pointer (note that
 844                      // the type of such a pattern will *always* be a
 845                      // region pointer)
 846                      let scope_r =
 847                          ty_region(self.tcx(), pat.span,
 848                                    ty::node_id_to_type(self.tcx(), pat.id));
 849  
 850                      // if the scope of the region ptr turns out to be
 851                      // specific to this arm, wrap the categorization
 852                      // with a cat_discr() node.  There is a detailed
 853                      // discussion of the function of this node in
 854                      // `lifetime.rs`:
 855                      let cmt_discr = match arm_match_ids {
 856                          None => cmt,
 857                          Some((arm_id, match_id)) => {
 858                              let arm_scope = ty::ReScope(arm_id);
 859                              if self.bccx.is_subregion_of(scope_r, arm_scope) {

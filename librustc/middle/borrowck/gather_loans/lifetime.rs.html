  66     fn check(&self, cmt: mc::cmt, discr_scope: Option<ast::NodeId>) {
  67         //! Main routine. Walks down `cmt` until we find the "guarantor".
  68 
  69         match cmt.cat {
  70             mc::cat_rvalue(*) |
  71             mc::cat_implicit_self |
  72             mc::cat_copied_upvar(*) |                  // L-Local
  73             mc::cat_local(*) |                         // L-Local
  74             mc::cat_arg(*) |                           // L-Local
  75             mc::cat_self(*) |                          // L-Local
  76             mc::cat_deref(_, _, mc::region_ptr(*)) |   // L-Deref-Borrowed
  77             mc::cat_deref(_, _, mc::unsafe_ptr) => {
  78                 let scope = self.scope(cmt);
  79                 self.check_scope(scope)
  80             }
  81 
  82             mc::cat_stack_upvar(cmt) => {
  83                 self.check(cmt, discr_scope)
  84             }
  85 
  86             mc::cat_static_item => {
  87             }
  88 
  89             mc::cat_deref(base, derefs, mc::gc_ptr(ptr_mutbl)) => {
  90                 let base_scope = self.scope(base);
  91 
  92                 // L-Deref-Managed-Imm-User-Root
  93                 let omit_root = (
  94                     ptr_mutbl == m_imm &&
  95                     self.bccx.is_subregion_of(self.loan_region, base_scope) &&
  96                     self.is_rvalue_or_immutable(base) &&
  97                     !self.is_moved(base)
  98                 );
  99 
 100                 if !omit_root {
 101                     // L-Deref-Managed-Imm-Compiler-Root
 102                     // L-Deref-Managed-Mut-Compiler-Root
 103                     self.check_root(cmt, base, derefs, ptr_mutbl, discr_scope);
 104                 } else {
 105                     debug!("omitting root, base=%s, base_scope=%?",
 106                            base.repr(self.tcx()), base_scope);
 107                 }
 108             }
 109 
 110             mc::cat_downcast(base) |
 111             mc::cat_deref(base, _, mc::uniq_ptr(*)) |  // L-Deref-Send
 112             mc::cat_interior(base, _) => {             // L-Field
 113                 self.check(base, discr_scope)
 114             }
 115 
 116             mc::cat_discr(base, new_discr_scope) => {
 117                 // Subtle: in a match, we must ensure that each binding
 118                 // variable remains valid for the duration of the arm in
 119                 // which it appears, presuming that this arm is taken.
 120                 // But it is inconvenient in trans to root something just
 121                 // for one arm.  Therefore, we insert a cat_discr(),
 122                 // basically a special kind of category that says "if this
 123                 // value must be dynamically rooted, root it for the scope
 124                 // `match_id`.
 125                 //
 126                 // As an example, consider this scenario:
 127                 //
 128                 //    let mut x = @Some(3);
 129                 //    match *x { Some(y) {...} None {...} }
 130                 //
 131                 // Technically, the value `x` need only be rooted
 132                 // in the `some` arm.  However, we evaluate `x` in trans
 133                 // before we know what arm will be taken, so we just
 134                 // always root it for the duration of the match.
 135                 //
 136                 // As a second example, consider *this* scenario:
 137                 //
 138                 //    let x = @mut @Some(3);
 139                 //    match x { @@Some(y) {...} @@None {...} }
 140                 //
 141                 // Here again, `x` need only be rooted in the `some` arm.
 142                 // In this case, the value which needs to be rooted is
 143                 // found only when checking which pattern matches: but
 144                 // this check is done before entering the arm.  Therefore,
 145                 // even in this case we just choose to keep the value
 146                 // rooted for the entire match.  This means the value will be
 147                 // rooted even if the none arm is taken.  Oh well.
 148                 //
 149                 // At first, I tried to optimize the second case to only
 150                 // root in one arm, but the result was suboptimal: first,
 151                 // it interfered with the construction of phi nodes in the
 152                 // arm, as we were adding code to root values before the
 153                 // phi nodes were added.  This could have been addressed
 154                 // with a second basic block.  However, the naive approach
 155                 // also yielded suboptimal results for patterns like:
 156                 //
 157                 //    let x = @mut @...;
 158                 //    match x { @@some_variant(y) | @@some_other_variant(y) =>
 159                 //
 160                 // The reason is that we would root the value once for
 161                 // each pattern and not once per arm.  This is also easily
 162                 // fixed, but it's yet more code for what is really quite
 163                 // the corner case.
 164                 //
 165                 // Nonetheless, if you decide to optimize this case in the
 166                 // future, you need only adjust where the cat_discr()
 167                 // node appears to draw the line between what will be rooted
 168                 // in the *arm* vs the *match*.
 169                 self.check(base, Some(new_discr_scope))
 170             }
 171         }
 172     }

 174     fn is_rvalue_or_immutable(&self,
 175                               cmt: mc::cmt) -> bool {
 176         //! We can omit the root on an `@T` value if the location
 177         //! that holds the box is either (1) an rvalue, in which case
 178         //! it is in a non-user-accessible temporary, or (2) an immutable
 179         //! lvalue.
 180 
 181         cmt.mutbl.is_immutable() || match cmt.guarantor().cat {
 182             mc::cat_rvalue(*) => true,
 183             _ => false
 184         }
 185     }

 187     fn check_root(&self,
 188                   cmt_deref: mc::cmt,
 189                   cmt_base: mc::cmt,
 190                   derefs: uint,
 191                   ptr_mutbl: ast::mutability,
 192                   discr_scope: Option<ast::NodeId>) {
 193         debug!("check_root(cmt_deref=%s, cmt_base=%s, derefs=%?, ptr_mutbl=%?, \
 194                 discr_scope=%?)",
 195                cmt_deref.repr(self.tcx()),
 196                cmt_base.repr(self.tcx()),
 197                derefs,
 198                ptr_mutbl,
 199                discr_scope);
 200 
 201         // Make sure that the loan does not exceed the maximum time
 202         // that we can root the value, dynamically.
 203         let root_region = ty::re_scope(self.root_scope_id);
 204         if !self.bccx.is_subregion_of(self.loan_region, root_region) {
 205             self.report_error(
 206                 err_out_of_root_scope(root_region, self.loan_region));
 207             return;
 208         }
 209 
 210         // Extract the scope id that indicates how long the rooting is required
 211         let root_scope = match self.loan_region {
 212             ty::re_scope(id) => id,
 213             _ => {
 214                 // the check above should fail for anything is not re_scope
 215                 self.bccx.tcx.sess.span_bug(
 216                     cmt_base.span,
 217                     fmt!("Cannot issue root for scope region: %?",
 218                          self.loan_region));
 219             }
 220         };
 221 
 222         // If inside of a match arm, expand the rooting to the entire
 223         // match. See the detailed discussion in `check()` above.
 224         let mut root_scope = match discr_scope {
 225             None => root_scope,
 226             Some(id) => {
 227                 if self.bccx.is_subscope_of(root_scope, id) {
 228                     id
 229                 } else {
 230                     root_scope
 231                 }
 232             }
 233         };
 234 
 235         // If we are borrowing the inside of an `@mut` box,
 236         // we need to dynamically mark it to prevent incompatible
 237         // borrows from happening later.
 238         let opt_dyna = match ptr_mutbl {
 239             m_imm | m_const => None,
 240             m_mutbl => {
 241                 match self.loan_mutbl {
 242                     m_mutbl => Some(DynaMut),
 243                     m_imm | m_const => Some(DynaImm)
 244                 }
 245             }
 246         };
 247 
 248         // FIXME(#3511) grow to the nearest cleanup scope---this can
 249         // cause observable errors if freezing!
 250         if !self.bccx.tcx.region_maps.is_cleanup_scope(root_scope) {
 251             debug!("%? is not a cleanup scope, adjusting", root_scope);
 252 
 253             let cleanup_scope =
 254                 self.bccx.tcx.region_maps.cleanup_scope(root_scope);
 255 
 256             if opt_dyna.is_some() {
 257                 self.tcx().sess.span_warn(
 258                     self.span,
 259                     fmt!("Dynamic freeze scope artifically extended \
 260                           (see Issue #6248)"));
 261                 note_and_explain_region(
 262                     self.bccx.tcx,
 263                     "managed value only needs to be frozen for ",
 264                     ty::re_scope(root_scope),
 265                     "...");
 266                 note_and_explain_region(
 267                     self.bccx.tcx,
 268                     "...but due to Issue #6248, it will be frozen for ",
 269                     ty::re_scope(cleanup_scope),
 270                     "");
 271             }
 272 
 273             root_scope = cleanup_scope;
 274         }
 275 
 276         // Add a record of what is required
 277         let rm_key = root_map_key {id: cmt_deref.id, derefs: derefs};
 278         let root_info = RootInfo {scope: root_scope, freeze: opt_dyna};
 279         self.bccx.root_map.insert(rm_key, root_info);
 280 
 281         debug!("root_key: %? root_info: %?", rm_key, root_info);
 282     }

 304             mc::cat_implicit_self |

 320     fn scope(&self, cmt: mc::cmt) -> ty::Region {
 321         //! Returns the maximal region scope for the which the
 322         //! lvalue `cmt` is guaranteed to be valid without any
 323         //! rooting etc, and presuming `cmt` is not mutated.
 324 
 325         // See the SCOPE(LV) function in doc.rs
 326 
 327         match cmt.cat {
 328             mc::cat_rvalue(cleanup_scope_id) => {
 329                 ty::re_scope(cleanup_scope_id)
 330             }
 331             mc::cat_implicit_self |
 332             mc::cat_copied_upvar(_) => {
 333                 ty::re_scope(self.item_scope_id)
 334             }
 335             mc::cat_static_item => {
 336                 ty::re_static
 337             }
 338             mc::cat_local(local_id) |
 339             mc::cat_arg(local_id) |
 340             mc::cat_self(local_id) => {
 341                 self.bccx.tcx.region_maps.encl_region(local_id)
 342             }
 343             mc::cat_deref(_, _, mc::unsafe_ptr(*)) => {
 344                 ty::re_static
 345             }
 346             mc::cat_deref(_, _, mc::region_ptr(_, r)) => {
 347                 r
 348             }
 349             mc::cat_downcast(cmt) |
 350             mc::cat_deref(cmt, _, mc::uniq_ptr(*)) |
 351             mc::cat_deref(cmt, _, mc::gc_ptr(*)) |
 352             mc::cat_interior(cmt, _) |
 353             mc::cat_stack_upvar(cmt) |
 354             mc::cat_discr(cmt, _) => {
 355                 self.scope(cmt)
 356             }
 357         }
 358     }

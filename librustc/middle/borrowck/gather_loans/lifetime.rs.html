  66      fn check(&self, cmt: mc::cmt, discr_scope: Option<ast::NodeId>) {
  67          //! Main routine. Walks down `cmt` until we find the "guarantor".
  68  
  69          match cmt.cat {
  70              mc::cat_rvalue(*) |
  71              mc::cat_copied_upvar(*) |                  // L-Local
  72              mc::cat_local(*) |                         // L-Local
  73              mc::cat_arg(*) |                           // L-Local
  74              mc::cat_self(*) |                          // L-Local
  75              mc::cat_deref(_, _, mc::region_ptr(*)) |   // L-Deref-Borrowed
  76              mc::cat_deref(_, _, mc::unsafe_ptr(*)) => {
  77                  let scope = self.scope(cmt);
  78                  self.check_scope(scope)
  79              }
  80  
  81              mc::cat_stack_upvar(cmt) => {
  82                  self.check(cmt, discr_scope)
  83              }
  84  
  85              mc::cat_static_item => {
  86              }
  87  
  88              mc::cat_deref(base, derefs, mc::gc_ptr(ptr_mutbl)) => {
  89                  let base_scope = self.scope(base);
  90  
  91                  // L-Deref-Managed-Imm-User-Root
  92                  let omit_root = (
  93                      ptr_mutbl == MutImmutable &&
  94                      self.bccx.is_subregion_of(self.loan_region, base_scope) &&
  95                      self.is_rvalue_or_immutable(base) &&
  96                      !self.is_moved(base)
  97                  );
  98  
  99                  if !omit_root {
 100                      // L-Deref-Managed-Imm-Compiler-Root
 101                      // L-Deref-Managed-Mut-Compiler-Root
 102                      self.check_root(cmt, base, derefs, ptr_mutbl, discr_scope);
 103                  } else {
 104                      debug!("omitting root, base=%s, base_scope=%?",
 105                             base.repr(self.tcx()), base_scope);
 106                  }
 107              }
 108  
 109              mc::cat_downcast(base) |
 110              mc::cat_deref(base, _, mc::uniq_ptr) |     // L-Deref-Send
 111              mc::cat_interior(base, _) => {             // L-Field
 112                  self.check(base, discr_scope)
 113              }
 114  
 115              mc::cat_discr(base, new_discr_scope) => {
 116                  // Subtle: in a match, we must ensure that each binding
 117                  // variable remains valid for the duration of the arm in
 118                  // which it appears, presuming that this arm is taken.
 119                  // But it is inconvenient in trans to root something just
 120                  // for one arm.  Therefore, we insert a cat_discr(),
 121                  // basically a special kind of category that says "if this
 122                  // value must be dynamically rooted, root it for the scope
 123                  // `match_id`.
 124                  //
 125                  // As an example, consider this scenario:
 126                  //
 127                  //    let mut x = @Some(3);
 128                  //    match *x { Some(y) {...} None {...} }
 129                  //
 130                  // Technically, the value `x` need only be rooted
 131                  // in the `some` arm.  However, we evaluate `x` in trans
 132                  // before we know what arm will be taken, so we just
 133                  // always root it for the duration of the match.
 134                  //
 135                  // As a second example, consider *this* scenario:
 136                  //
 137                  //    let x = @mut @Some(3);
 138                  //    match x { @@Some(y) {...} @@None {...} }
 139                  //
 140                  // Here again, `x` need only be rooted in the `some` arm.
 141                  // In this case, the value which needs to be rooted is
 142                  // found only when checking which pattern matches: but
 143                  // this check is done before entering the arm.  Therefore,
 144                  // even in this case we just choose to keep the value
 145                  // rooted for the entire match.  This means the value will be
 146                  // rooted even if the none arm is taken.  Oh well.
 147                  //
 148                  // At first, I tried to optimize the second case to only
 149                  // root in one arm, but the result was suboptimal: first,
 150                  // it interfered with the construction of phi nodes in the
 151                  // arm, as we were adding code to root values before the
 152                  // phi nodes were added.  This could have been addressed
 153                  // with a second basic block.  However, the naive approach
 154                  // also yielded suboptimal results for patterns like:
 155                  //
 156                  //    let x = @mut @...;
 157                  //    match x { @@some_variant(y) | @@some_other_variant(y) =>
 158                  //
 159                  // The reason is that we would root the value once for
 160                  // each pattern and not once per arm.  This is also easily
 161                  // fixed, but it's yet more code for what is really quite
 162                  // the corner case.
 163                  //
 164                  // Nonetheless, if you decide to optimize this case in the
 165                  // future, you need only adjust where the cat_discr()
 166                  // node appears to draw the line between what will be rooted
 167                  // in the *arm* vs the *match*.
 168                  self.check(base, Some(new_discr_scope))
 169              }
 170          }
 171      }

 173      fn is_rvalue_or_immutable(&self,
 174                                cmt: mc::cmt) -> bool {
 175          //! We can omit the root on an `@T` value if the location
 176          //! that holds the box is either (1) an rvalue, in which case
 177          //! it is in a non-user-accessible temporary, or (2) an immutable
 178          //! lvalue.
 179  
 180          cmt.mutbl.is_immutable() || match cmt.guarantor().cat {
 181              mc::cat_rvalue(*) => true,
 182              _ => false
 183          }
 184      }

 186      fn check_root(&self,
 187                    cmt_deref: mc::cmt,
 188                    cmt_base: mc::cmt,
 189                    derefs: uint,
 190                    ptr_mutbl: ast::Mutability,
 191                    discr_scope: Option<ast::NodeId>) {
 192          debug!("check_root(cmt_deref=%s, cmt_base=%s, derefs=%?, ptr_mutbl=%?, \
 193                  discr_scope=%?)",
 194                 cmt_deref.repr(self.tcx()),
 195                 cmt_base.repr(self.tcx()),
 196                 derefs,
 197                 ptr_mutbl,
 198                 discr_scope);
 199  
 200          // Make sure that the loan does not exceed the maximum time
 201          // that we can root the value, dynamically.
 202          let root_region = ty::re_scope(self.root_scope_id);
 203          if !self.bccx.is_subregion_of(self.loan_region, root_region) {
 204              self.report_error(
 205                  err_out_of_root_scope(root_region, self.loan_region));
 206              return;
 207          }
 208  
 209          // Extract the scope id that indicates how long the rooting is required
 210          let root_scope = match self.loan_region {
 211              ty::re_scope(id) => id,
 212              _ => {
 213                  // the check above should fail for anything is not re_scope
 214                  self.bccx.tcx.sess.span_bug(
 215                      cmt_base.span,
 216                      fmt!("Cannot issue root for scope region: %?",
 217                           self.loan_region));
 218              }
 219          };
 220  
 221          // If inside of a match arm, expand the rooting to the entire
 222          // match. See the detailed discussion in `check()` above.
 223          let mut root_scope = match discr_scope {
 224              None => root_scope,
 225              Some(id) => {
 226                  if self.bccx.is_subscope_of(root_scope, id) {
 227                      id
 228                  } else {
 229                      root_scope
 230                  }
 231              }
 232          };
 233  
 234          // If we are borrowing the inside of an `@mut` box,
 235          // we need to dynamically mark it to prevent incompatible
 236          // borrows from happening later.
 237          let opt_dyna = match ptr_mutbl {
 238              MutImmutable => None,
 239              MutMutable => {
 240                  match self.loan_mutbl {
 241                      MutableMutability => Some(DynaMut),
 242                      ImmutableMutability | ConstMutability => Some(DynaImm)
 243                  }
 244              }
 245          };
 246  
 247          // FIXME(#3511) grow to the nearest cleanup scope---this can
 248          // cause observable errors if freezing!
 249          if !self.bccx.tcx.region_maps.is_cleanup_scope(root_scope) {
 250              debug!("%? is not a cleanup scope, adjusting", root_scope);
 251  
 252              let cleanup_scope =
 253                  self.bccx.tcx.region_maps.cleanup_scope(root_scope);
 254  
 255              if opt_dyna.is_some() {
 256                  self.tcx().sess.span_warn(
 257                      self.span,
 258                      fmt!("Dynamic freeze scope artifically extended \
 259                            (see Issue #6248)"));
 260                  note_and_explain_region(
 261                      self.bccx.tcx,
 262                      "managed value only needs to be frozen for ",
 263                      ty::re_scope(root_scope),
 264                      "...");
 265                  note_and_explain_region(
 266                      self.bccx.tcx,
 267                      "...but due to Issue #6248, it will be frozen for ",
 268                      ty::re_scope(cleanup_scope),
 269                      "");
 270              }
 271  
 272              root_scope = cleanup_scope;
 273          }
 274  
 275          // Add a record of what is required
 276          let rm_key = root_map_key {id: cmt_deref.id, derefs: derefs};
 277          let root_info = RootInfo {scope: root_scope, freeze: opt_dyna};
 278          self.bccx.root_map.insert(rm_key, root_info);
 279  
 280          debug!("root_key: %? root_info: %?", rm_key, root_info);
 281      }

 318      fn scope(&self, cmt: mc::cmt) -> ty::Region {
 319          //! Returns the maximal region scope for the which the
 320          //! lvalue `cmt` is guaranteed to be valid without any
 321          //! rooting etc, and presuming `cmt` is not mutated.
 322  
 323          // See the SCOPE(LV) function in doc.rs
 324  
 325          match cmt.cat {
 326              mc::cat_rvalue(cleanup_scope_id) => {
 327                  ty::re_scope(cleanup_scope_id)
 328              }
 329              mc::cat_copied_upvar(_) => {
 330                  ty::re_scope(self.item_scope_id)
 331              }
 332              mc::cat_static_item => {
 333                  ty::re_static
 334              }
 335              mc::cat_local(local_id) |
 336              mc::cat_arg(local_id) |
 337              mc::cat_self(local_id) => {
 338                  self.bccx.tcx.region_maps.encl_region(local_id)
 339              }
 340              mc::cat_deref(_, _, mc::unsafe_ptr(*)) => {
 341                  ty::re_static
 342              }
 343              mc::cat_deref(_, _, mc::region_ptr(_, r)) => {
 344                  r
 345              }
 346              mc::cat_downcast(cmt) |
 347              mc::cat_deref(cmt, _, mc::uniq_ptr) |
 348              mc::cat_deref(cmt, _, mc::gc_ptr(*)) |
 349              mc::cat_interior(cmt, _) |
 350              mc::cat_stack_upvar(cmt) |
 351              mc::cat_discr(cmt, _) => {
 352                  self.scope(cmt)
 353              }
 354          }
 355      }

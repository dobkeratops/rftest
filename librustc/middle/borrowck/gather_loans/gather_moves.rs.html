  99  fn check_is_legal_to_move_from(bccx: &BorrowckCtxt,
 100                                 cmt0: mc::cmt,
 101                                 cmt: mc::cmt) -> bool {
 102      match cmt.cat {
 103          mc::cat_deref(_, _, mc::region_ptr(*)) |
 104          mc::cat_deref(_, _, mc::gc_ptr(*)) |
 105          mc::cat_deref(_, _, mc::unsafe_ptr(*)) => {
 106              bccx.span_err(
 107                  cmt0.span,
 108                  fmt!("cannot move out of %s",
 109                       bccx.cmt_to_str(cmt)));
 110              false
 111          }
 112  
 113          // These are separate from the above cases for a better error message.
 114          mc::cat_stack_upvar(*) |
 115          mc::cat_copied_upvar(mc::CopiedUpvar { onceness: ast::Many, _ }) => {
 116              let once_hint = if bccx.tcx.sess.once_fns() {
 117                  " (unless the destination closure type is `once fn')"
 118              } else {
 119                  ""
 120              };
 121              bccx.span_err(
 122                  cmt0.span,
 123                  fmt!("cannot move out of %s%s", bccx.cmt_to_str(cmt), once_hint));
 124              false
 125          }
 126  
 127          // Can move out of captured upvars only if the destination closure
 128          // type is 'once'. 1-shot stack closures emit the copied_upvar form
 129          // (see mem_categorization.rs).
 130          mc::cat_copied_upvar(mc::CopiedUpvar { onceness: ast::Once, _ }) => {
 131              true
 132          }
 133  
 134          // It seems strange to allow a move out of a static item,
 135          // but what happens in practice is that you have a
 136          // reference to a constant with a type that should be
 137          // moved, like `None::<~int>`.  The type of this constant
 138          // is technically `Option<~int>`, which moves, but we know
 139          // that the content of static items will never actually
 140          // contain allocated pointers, so we can just memcpy it.
 141          // Since static items can never have allocated memory,
 142          // this is ok. For now anyhow.
 143          mc::cat_static_item => {
 144              true
 145          }
 146  
 147          mc::cat_rvalue(*) |
 148          mc::cat_local(*) |
 149          mc::cat_arg(*) |
 150          mc::cat_self(*) => {
 151              true
 152          }
 153  
 154          mc::cat_downcast(b) |
 155          mc::cat_interior(b, _) => {
 156              match ty::get(b.ty).sty {
 157                  ty::ty_struct(did, _) | ty::ty_enum(did, _) => {
 158                      if ty::has_dtor(bccx.tcx, did) {
 159                          bccx.span_err(
 160                              cmt0.span,
 161                              fmt!("cannot move out of type `%s`, \
 162                                    which defines the `Drop` trait",
 163                                   b.ty.user_string(bccx.tcx)));
 164                          false
 165                      } else {
 166                          check_is_legal_to_move_from(bccx, cmt0, b)
 167                      }
 168                  }
 169                  _ => {
 170                      check_is_legal_to_move_from(bccx, cmt0, b)
 171                  }
 172              }
 173          }
 174  
 175          mc::cat_deref(b, _, mc::uniq_ptr) |
 176          mc::cat_discr(b, _) => {
 177              check_is_legal_to_move_from(bccx, cmt0, b)
 178          }
 179      }
 180  }

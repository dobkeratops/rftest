  99  fn check_is_legal_to_move_from(bccx: @BorrowckCtxt,
 100                                 cmt0: mc::cmt,
 101                                 cmt: mc::cmt) -> bool {
 102      match cmt.cat {
 103          mc::cat_implicit_self(*) |
 104          mc::cat_deref(_, _, mc::region_ptr(*)) |
 105          mc::cat_deref(_, _, mc::gc_ptr(*)) |
 106          mc::cat_deref(_, _, mc::unsafe_ptr(*)) => {
 107              bccx.span_err(
 108                  cmt0.span,
 109                  fmt!("cannot move out of %s",
 110                       bccx.cmt_to_str(cmt)));
 111              false
 112          }
 113  
 114          // These are separate from the above cases for a better error message.
 115          mc::cat_stack_upvar(*) |
 116          mc::cat_copied_upvar(mc::CopiedUpvar { onceness: ast::Many, _ }) => {
 117              let once_hint = if bccx.tcx.sess.once_fns() {
 118                  " (unless the destination closure type is `once fn')"
 119              } else {
 120                  ""
 121              };
 122              bccx.span_err(
 123                  cmt0.span,
 124                  fmt!("cannot move out of %s%s", bccx.cmt_to_str(cmt), once_hint));
 125              false
 126          }
 127  
 128          // Can move out of captured upvars only if the destination closure
 129          // type is 'once'. 1-shot stack closures emit the copied_upvar form
 130          // (see mem_categorization.rs).
 131          mc::cat_copied_upvar(mc::CopiedUpvar { onceness: ast::Once, _ }) => {
 132              true
 133          }
 134  
 135          // It seems strange to allow a move out of a static item,
 136          // but what happens in practice is that you have a
 137          // reference to a constant with a type that should be
 138          // moved, like `None::<~int>`.  The type of this constant
 139          // is technically `Option<~int>`, which moves, but we know
 140          // that the content of static items will never actually
 141          // contain allocated pointers, so we can just memcpy it.
 142          // Since static items can never have allocated memory,
 143          // this is ok. For now anyhow.
 144          mc::cat_static_item => {
 145              true
 146          }
 147  
 148          mc::cat_rvalue(*) |
 149          mc::cat_local(*) |
 150          mc::cat_arg(*) |
 151          mc::cat_self(*) => {
 152              true
 153          }
 154  
 155          mc::cat_downcast(b) |
 156          mc::cat_interior(b, _) => {
 157              match ty::get(b.ty).sty {
 158                  ty::ty_struct(did, _) | ty::ty_enum(did, _) => {
 159                      if ty::has_dtor(bccx.tcx, did) {
 160                          bccx.span_err(
 161                              cmt0.span,
 162                              fmt!("cannot move out of type `%s`, \
 163                                    which defines the `Drop` trait",
 164                                   b.ty.user_string(bccx.tcx)));
 165                          false
 166                      } else {
 167                          check_is_legal_to_move_from(bccx, cmt0, b)
 168                      }
 169                  }
 170                  _ => {
 171                      check_is_legal_to_move_from(bccx, cmt0, b)
 172                  }
 173              }
 174          }
 175  
 176          mc::cat_deref(b, _, mc::uniq_ptr(*)) |
 177          mc::cat_discr(b, _) => {
 178              check_is_legal_to_move_from(bccx, cmt0, b)
 179          }
 180      }
 181  }

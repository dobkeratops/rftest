 113  fn check_is_legal_to_move_from(bccx: &BorrowckCtxt,
 114                                 cmt0: mc::cmt,
 115                                 cmt: mc::cmt) -> bool {
 116      match cmt.cat {
 117          mc::cat_deref(_, _, mc::BorrowedPtr(..)) |
 118          mc::cat_deref(_, _, mc::GcPtr) |
 119          mc::cat_deref(_, _, mc::UnsafePtr(..)) |
 120          mc::cat_upvar(..) | mc::cat_static_item |
 121          mc::cat_copied_upvar(mc::CopiedUpvar { onceness: ast::Many, .. }) => {
 122              bccx.span_err(
 123                  cmt0.span,
 124                  format!("cannot move out of {}",
 125                          bccx.cmt_to_str(cmt)));
 126              false
 127          }
 128  
 129          // Can move out of captured upvars only if the destination closure
 130          // type is 'once'. 1-shot stack closures emit the copied_upvar form
 131          // (see mem_categorization.rs).
 132          mc::cat_copied_upvar(mc::CopiedUpvar { onceness: ast::Once, .. }) => {
 133              true
 134          }
 135  
 136          mc::cat_rvalue(..) |
 137          mc::cat_local(..) |
 138          mc::cat_arg(..) => {
 139              true
 140          }
 141  
 142          mc::cat_downcast(b) |
 143          mc::cat_interior(b, _) => {
 144              match ty::get(b.ty).sty {
 145                  ty::ty_struct(did, _) | ty::ty_enum(did, _) => {
 146                      if ty::has_dtor(bccx.tcx, did) {
 147                          bccx.span_err(
 148                              cmt0.span,
 149                              format!("cannot move out of type `{}`, \
 150                                    which defines the `Drop` trait",
 151                                   b.ty.user_string(bccx.tcx)));
 152                          false
 153                      } else {
 154                          check_is_legal_to_move_from(bccx, cmt0, b)
 155                      }
 156                  }
 157                  _ => {
 158                      check_is_legal_to_move_from(bccx, cmt0, b)
 159                  }
 160              }
 161          }
 162  
 163          mc::cat_deref(b, _, mc::OwnedPtr) |
 164          mc::cat_discr(b, _) => {
 165              check_is_legal_to_move_from(bccx, cmt0, b)
 166          }
 167      }
 168  }

   11 /*!
   12  * # Categorization
   13  *
   14  * The job of the categorization module is to analyze an expression to
   15  * determine what kind of memory is used in evaluating it (for example,
   16  * where dereferences occur and what kind of pointer is dereferenced;
   17  * whether the memory is mutable; etc)
   18  *
   19  * Categorization effectively transforms all of our expressions into
   20  * expressions of the following forms (the actual enum has many more
   21  * possibilities, naturally, but they are all variants of these base
   22  * forms):
   23  *
   24  *     E = rvalue    // some computed rvalue
   25  *       | x         // address of a local variable, arg, or upvar
   26  *       | *E        // deref of a ptr
   27  *       | E.comp    // access to an interior component
   28  *
   29  * Imagine a routine ToAddr(Expr) that evaluates an expression and returns an
   30  * address where the result is to be found.  If Expr is an lvalue, then this
   31  * is the address of the lvalue.  If Expr is an rvalue, this is the address of
   32  * some temporary spot in memory where the result is stored.
   33  *
   34  * Now, cat_expr() classies the expression Expr and the address A=ToAddr(Expr)
   35  * as follows:
   36  *
   37  * - cat: what kind of expression was this?  This is a subset of the
   38  *   full expression forms which only includes those that we care about
   39  *   for the purpose of the analysis.
   40  * - mutbl: mutability of the address A
   41  * - ty: the type of data found at the address A
   42  *
   43  * The resulting categorization tree differs somewhat from the expressions
   44  * themselves.  For example, auto-derefs are explicit.  Also, an index a[b] is
   45  * decomposed into two operations: a derefence to reach the array data and
   46  * then an index to jump forward to the relevant item.
   47  */

  122 // `cmt`: "Category, Mutability, and Type".
  123 //
  124 // a complete categorization of a value indicating where it originated
  125 // and how it is located, as well as the mutability of the memory in
  126 // which the value is stored.
  127 //
  128 // *WARNING* The field `cmt.type` is NOT necessarily the same as the
  129 // result of `node_id_to_type(cmt.id)`. This is because the `id` is
  130 // always the `id` of the node producing the type; in an expression
  131 // like `*x`, the type of this deref node is the deref'd type (`T`),
  132 // but in a pattern like `@x`, the `@x` pattern is again a
  133 // dereference, but its type is the type *before* the dereference
  134 // (`@T`). So use `cmt.type` to find the type of the value in a consistent
  135 // fashion. For more details, see the method `cat_pattern`
  136 #[deriving(Eq)]
  137 pub struct cmt_ {
  138     id: ast::NodeId,          // id of expr/pat producing this value
  139     span: span,                // span of same expr/pat
  140     cat: categorization,       // categorization of expr
  141     mutbl: MutabilityCategory, // mutability of expr as lvalue
  142     ty: ty::t                  // type of the expr (*see WARNING above*)
  143 }
  144

  147 // We pun on *T to mean both actual deref of a ptr as well
  148 // as accessing of components:

  150     deref_ptr(ptr_kind),
  151     deref_interior(InteriorKind),
  152 }

  154 // Categorizes a derefable type.  Note that we include vectors and strings as
  155 // derefable (we model an index as the combination of a deref and then a
  156 // pointer adjustment).
  157 pub fn opt_deref_kind(t: ty::t) -> Option<deref_kind> {
  158     match ty::get(t).sty {
  159         ty::ty_uniq(mt) => {
  160             Some(deref_ptr(uniq_ptr(mt.mutbl)))
  161         }
  162 
  163         ty::ty_evec(_, ty::vstore_uniq) |
  164         ty::ty_estr(ty::vstore_uniq) |
  165         ty::ty_closure(ty::ClosureTy {sigil: ast::OwnedSigil, _}) => {
  166             Some(deref_ptr(uniq_ptr(m_imm)))
  167         }
  168 
  169         ty::ty_rptr(r, mt) |
  170         ty::ty_evec(mt, ty::vstore_slice(r)) => {
  171             Some(deref_ptr(region_ptr(mt.mutbl, r)))
  172         }
  173 
  174         ty::ty_estr(ty::vstore_slice(r)) |
  175         ty::ty_closure(ty::ClosureTy {sigil: ast::BorrowedSigil,
  176                                       region: r, _}) => {
  177             Some(deref_ptr(region_ptr(ast::m_imm, r)))
  178         }
  179 
  180         ty::ty_box(mt) |
  181         ty::ty_evec(mt, ty::vstore_box) => {
  182             Some(deref_ptr(gc_ptr(mt.mutbl)))
  183         }
  184 
  185         ty::ty_estr(ty::vstore_box) |
  186         ty::ty_closure(ty::ClosureTy {sigil: ast::ManagedSigil, _}) => {
  187             Some(deref_ptr(gc_ptr(ast::m_imm)))
  188         }
  189 
  190         ty::ty_ptr(*) => {
  191             Some(deref_ptr(unsafe_ptr))
  192         }
  193 
  194         ty::ty_enum(*) |
  195         ty::ty_struct(*) => { // newtype
  196             Some(deref_interior(InteriorField(PositionalField(0))))
  197         }
  198 
  199         ty::ty_evec(_, ty::vstore_fixed(_)) |
  200         ty::ty_estr(ty::vstore_fixed(_)) => {
  201             Some(deref_interior(InteriorElement(element_kind(t))))
  202         }
  203 
  204         _ => None
  205     }
  206 }

  341     pub fn cat_expr(&self, expr: @ast::expr) -> cmt {
  342         match self.tcx.adjustments.find(&expr.id) {
  343             None => {
  344                 // No adjustments.
  345                 self.cat_expr_unadjusted(expr)
  346             }
  347 
  348             Some(&@ty::AutoAddEnv(*)) => {
  349                 // Convert a bare fn to a closure by adding NULL env.
  350                 // Result is an rvalue.
  351                 let expr_ty = ty::expr_ty_adjusted(self.tcx, expr);
  352                 self.cat_rvalue_node(expr, expr_ty)
  353             }
  354 
  355             Some(
  356                 &@ty::AutoDerefRef(
  357                     ty::AutoDerefRef {
  358                         autoref: Some(_), _})) => {
  359                 // Equivalent to &*expr or something similar.
  360                 // Result is an rvalue.
  361                 let expr_ty = ty::expr_ty_adjusted(self.tcx, expr);
  362                 self.cat_rvalue_node(expr, expr_ty)
  363             }
  364 
  365             Some(
  366                 &@ty::AutoDerefRef(
  367                     ty::AutoDerefRef {
  368                         autoref: None, autoderefs: autoderefs})) => {
  369                 // Equivalent to *expr or something similar.
  370                 self.cat_expr_autoderefd(expr, autoderefs)
  371             }
  372         }
  373     }

  375     pub fn cat_expr_autoderefd(&self, expr: @ast::expr, autoderefs: uint)
  376                                -> cmt {
  377         let mut cmt = self.cat_expr_unadjusted(expr);
  378         for deref in range(1u, autoderefs + 1) {
  379             cmt = self.cat_deref(expr, cmt, deref);
  380         }
  381         return cmt;
  382     }

  384     pub fn cat_expr_unadjusted(&self, expr: @ast::expr) -> cmt {
  385         debug!("cat_expr: id=%d expr=%s",
  386                expr.id, pprust::expr_to_str(expr, self.tcx.sess.intr()));
  387 
  388         let expr_ty = self.expr_ty(expr);
  389         match expr.node {
  390           ast::expr_unary(_, ast::deref, e_base) => {
  391             if self.method_map.contains_key(&expr.id) {
  392                 return self.cat_rvalue_node(expr, expr_ty);
  393             }
  394 
  395             let base_cmt = self.cat_expr(e_base);
  396             self.cat_deref(expr, base_cmt, 0)
  397           }
  398 
  399           ast::expr_field(base, f_name, _) => {
  400             // Method calls are now a special syntactic form,
  401             // so `a.b` should always be a field.
  402             assert!(!self.method_map.contains_key(&expr.id));
  403 
  404             let base_cmt = self.cat_expr(base);
  405             self.cat_field(expr, base_cmt, f_name, self.expr_ty(expr))
  406           }
  407 
  408           ast::expr_index(_, base, _) => {
  409             if self.method_map.contains_key(&expr.id) {
  410                 return self.cat_rvalue_node(expr, expr_ty);
  411             }
  412 
  413             let base_cmt = self.cat_expr(base);
  414             self.cat_index(expr, base_cmt, 0)
  415           }
  416 
  417           ast::expr_path(_) | ast::expr_self => {
  418             let def = self.tcx.def_map.get_copy(&expr.id);
  419             self.cat_def(expr.id, expr.span, expr_ty, def)
  420           }
  421 
  422           ast::expr_paren(e) => self.cat_expr_unadjusted(e),
  423 
  424           ast::expr_addr_of(*) | ast::expr_call(*) |
  425           ast::expr_assign(*) | ast::expr_assign_op(*) |
  426           ast::expr_fn_block(*) | ast::expr_ret(*) |
  427           ast::expr_do_body(*) | ast::expr_unary(*) |
  428           ast::expr_method_call(*) | ast::expr_cast(*) | ast::expr_vstore(*) |
  429           ast::expr_vec(*) | ast::expr_tup(*) | ast::expr_if(*) |
  430           ast::expr_log(*) | ast::expr_binary(*) | ast::expr_while(*) |
  431           ast::expr_block(*) | ast::expr_loop(*) | ast::expr_match(*) |
  432           ast::expr_lit(*) | ast::expr_break(*) | ast::expr_mac(*) |
  433           ast::expr_again(*) | ast::expr_struct(*) | ast::expr_repeat(*) |
  434           ast::expr_inline_asm(*) => {
  435             return self.cat_rvalue_node(expr, expr_ty);
  436           }
  437

  709     pub fn cat_index<N:ast_node>(&self,
  710                                  elt: N,
  711                                  base_cmt: cmt,
  712                                  derefs: uint)
  713                                  -> cmt {
  714         //! Creates a cmt for an indexing operation (`[]`); this
  715         //! indexing operation may occurs as part of an
  716         //! AutoBorrowVec, which when converting a `~[]` to an `&[]`
  717         //! effectively takes the address of the 0th element.
  718         //!
  719         //! One subtle aspect of indexing that may not be
  720         //! immediately obvious: for anything other than a fixed-length
  721         //! vector, an operation like `x[y]` actually consists of two
  722         //! disjoint (from the point of view of borrowck) operations.
  723         //! The first is a deref of `x` to create a pointer `p` that points
  724         //! at the first element in the array. The second operation is
  725         //! an index which adds `y*sizeof(T)` to `p` to obtain the
  726         //! pointer to `x[y]`. `cat_index` will produce a resulting
  727         //! cmt containing both this deref and the indexing,
  728         //! presuming that `base_cmt` is not of fixed-length type.
  729         //!
  730         //! In the event that a deref is needed, the "deref count"
  731         //! is taken from the parameter `derefs`. See the comment
  732         //! on the def'n of `root_map_key` in borrowck/mod.rs
  733         //! for more details about deref counts; the summary is
  734         //! that `derefs` should be 0 for an explicit indexing
  735         //! operation and N+1 for an indexing that is part of
  736         //! an auto-adjustment, where N is the number of autoderefs
  737         //! in that adjustment.
  738         //!
  739         //! # Parameters
  740         //! - `elt`: the AST node being indexed
  741         //! - `base_cmt`: the cmt of `elt`
  742         //! - `derefs`: the deref number to be used for
  743         //!   the implicit index deref, if any (see above)
  744 
  745         let mt = match ty::index(base_cmt.ty) {
  746           Some(mt) => mt,
  747           None => {
  748             self.tcx.sess.span_bug(
  749                 elt.span(),
  750                 fmt!("Explicit index of non-index type `%s`",
  751                      ty_to_str(self.tcx, base_cmt.ty)));
  752           }
  753         };
  754 
  755         return match deref_kind(self.tcx, base_cmt.ty) {
  756           deref_ptr(ptr) => {
  757             // for unique ptrs, we inherit mutability from the
  758             // owning reference.
  759             let m = match ptr {
  760               uniq_ptr(*) => {
  761                 self.inherited_mutability(base_cmt.mutbl, mt.mutbl)
  762               }
  763               gc_ptr(_) | region_ptr(_, _) | unsafe_ptr => {
  764                 MutabilityCategory::from_mutbl(mt.mutbl)
  765               }
  766             };
  767 
  768             // the deref is explicit in the resulting cmt
  769             let deref_cmt = @cmt_ {
  770                 id:elt.id(),
  771                 span:elt.span(),
  772                 cat:cat_deref(base_cmt, derefs, ptr),
  773                 mutbl:m,
  774                 ty:mt.ty
  775             };
  776 
  777             interior(elt, deref_cmt, base_cmt.ty, m, mt)
  778           }
  779 
  780           deref_interior(_) => {
  781             // fixed-length vectors have no deref
  782             let m = self.inherited_mutability(base_cmt.mutbl, mt.mutbl);
  783             interior(elt, base_cmt, base_cmt.ty, m, mt)
  784           }
  785         };

  832     pub fn cat_pattern(&self,
  833                        cmt: cmt,
  834                        pat: @ast::pat,
  835                        op: &fn(cmt, @ast::pat)) {
  836         // Here, `cmt` is the categorization for the value being
  837         // matched and pat is the pattern it is being matched against.
  838         //
  839         // In general, the way that this works is that we walk down
  840         // the pattern, constructing a cmt that represents the path
  841         // that will be taken to reach the value being matched.
  842         //
  843         // When we encounter named bindings, we take the cmt that has
  844         // been built up and pass it off to guarantee_valid() so that
  845         // we can be sure that the binding will remain valid for the
  846         // duration of the arm.
  847         //
  848         // (*) There is subtlety concerning the correspondence between
  849         // pattern ids and types as compared to *expression* ids and
  850         // types. This is explained briefly. on the definition of the
  851         // type `cmt`, so go off and read what it says there, then
  852         // come back and I'll dive into a bit more detail here. :) OK,
  853         // back?
  854         //
  855         // In general, the id of the cmt should be the node that
  856         // "produces" the value---patterns aren't executable code
  857         // exactly, but I consider them to "execute" when they match a
  858         // value. So if you have something like:
  859         //
  860         //     let x = @@3;
  861         //     match x {
  862         //       @@y { ... }
  863         //     }
  864         //
  865         // In this case, the cmt and the relevant ids would be:
  866         //
  867         //     CMT             Id                  Type of Id Type of cmt
  868         //
  869         //     local(x)->@->@
  870         //     ^~~~~~~^        `x` from discr      @@int      @@int
  871         //     ^~~~~~~~~~^     `@@y` pattern node  @@int      @int
  872         //     ^~~~~~~~~~~~~^  `@y` pattern node   @int       int
  873         //
  874         // You can see that the types of the id and the cmt are in
  875         // sync in the first line, because that id is actually the id
  876         // of an expression. But once we get to pattern ids, the types
  877         // step out of sync again. So you'll see below that we always

   11  /*!
   12   * # Categorization
   13   *
   14   * The job of the categorization module is to analyze an expression to
   15   * determine what kind of memory is used in evaluating it (for example,
   16   * where dereferences occur and what kind of pointer is dereferenced;
   17   * whether the memory is mutable; etc)
   18   *
   19   * Categorization effectively transforms all of our expressions into
   20   * expressions of the following forms (the actual enum has many more
   21   * possibilities, naturally, but they are all variants of these base
   22   * forms):
   23   *
   24   *     E = rvalue    // some computed rvalue
   25   *       | x         // address of a local variable, arg, or upvar
   26   *       | *E        // deref of a ptr
   27   *       | E.comp    // access to an interior component
   28   *
   29   * Imagine a routine ToAddr(Expr) that evaluates an expression and returns an
   30   * address where the result is to be found.  If Expr is an lvalue, then this
   31   * is the address of the lvalue.  If Expr is an rvalue, this is the address of
   32   * some temporary spot in memory where the result is stored.
   33   *
   34   * Now, cat_expr() classies the expression Expr and the address A=ToAddr(Expr)
   35   * as follows:
   36   *
   37   * - cat: what kind of expression was this?  This is a subset of the
   38   *   full expression forms which only includes those that we care about
   39   *   for the purpose of the analysis.
   40   * - mutbl: mutability of the address A
   41   * - ty: the type of data found at the address A
   42   *
   43   * The resulting categorization tree differs somewhat from the expressions
   44   * themselves.  For example, auto-derefs are explicit.  Also, an index a[b] is
   45   * decomposed into two operations: a derefence to reach the array data and
   46   * then an index to jump forward to the relevant item.
   47   */
   48

  122  // `cmt`: "Category, Mutability, and Type".
  123  //
  124  // a complete categorization of a value indicating where it originated
  125  // and how it is located, as well as the mutability of the memory in
  126  // which the value is stored.
  127  //
  128  // *WARNING* The field `cmt.type` is NOT necessarily the same as the
  129  // result of `node_id_to_type(cmt.id)`. This is because the `id` is
  130  // always the `id` of the node producing the type; in an expression
  131  // like `*x`, the type of this deref node is the deref'd type (`T`),
  132  // but in a pattern like `@x`, the `@x` pattern is again a
  133  // dereference, but its type is the type *before* the dereference
  134  // (`@T`). So use `cmt.type` to find the type of the value in a consistent
  135  // fashion. For more details, see the method `cat_pattern`
  136  #[deriving(Eq)]
  137  pub struct cmt_ {
  138      id: ast::NodeId,          // id of expr/pat producing this value
  139      span: span,                // span of same expr/pat
  140      cat: categorization,       // categorization of expr
  141      mutbl: MutabilityCategory, // mutability of expr as lvalue
  142      ty: ty::t                  // type of the expr (*see WARNING above*)
  143  }
  144

  154  // Categorizes a derefable type.  Note that we include vectors and strings as
  155  // derefable (we model an index as the combination of a deref and then a
  156  // pointer adjustment).
  157  pub fn opt_deref_kind(t: ty::t) -> Option<deref_kind> {
  158      match ty::get(t).sty {
  159          ty::ty_uniq(_) |
  160          ty::ty_trait(_, _, ty::UniqTraitStore, _, _) |
  161          ty::ty_evec(_, ty::vstore_uniq) |
  162          ty::ty_estr(ty::vstore_uniq) |
  163          ty::ty_closure(ty::ClosureTy {sigil: ast::OwnedSigil, _}) => {
  164              Some(deref_ptr(uniq_ptr))
  165          }
  166  
  167          ty::ty_rptr(r, mt) |
  168          ty::ty_evec(mt, ty::vstore_slice(r)) => {
  169              Some(deref_ptr(region_ptr(mt.mutbl, r)))
  170          }
  171  
  172          ty::ty_trait(_, _, ty::RegionTraitStore(r), m, _) => {
  173              Some(deref_ptr(region_ptr(m, r)))
  174          }
  175  
  176          ty::ty_estr(ty::vstore_slice(r)) |
  177          ty::ty_closure(ty::ClosureTy {sigil: ast::BorrowedSigil,
  178                                        region: r, _}) => {
  179              Some(deref_ptr(region_ptr(ast::m_imm, r)))
  180          }
  181  
  182          ty::ty_box(ref mt) |
  183          ty::ty_evec(ref mt, ty::vstore_box) => {
  184              Some(deref_ptr(gc_ptr(mt.mutbl)))
  185          }
  186  
  187          ty::ty_trait(_, _, ty::BoxTraitStore, m, _) => {
  188              Some(deref_ptr(gc_ptr(m)))
  189          }
  190  
  191          ty::ty_estr(ty::vstore_box) |
  192          ty::ty_closure(ty::ClosureTy {sigil: ast::ManagedSigil, _}) => {
  193              Some(deref_ptr(gc_ptr(ast::m_imm)))
  194          }
  195  
  196          ty::ty_ptr(ref mt) => {
  197              Some(deref_ptr(unsafe_ptr(mt.mutbl)))
  198          }
  199  
  200          ty::ty_enum(*) |
  201          ty::ty_struct(*) => { // newtype
  202              Some(deref_interior(InteriorField(PositionalField(0))))
  203          }
  204  
  205          ty::ty_evec(_, ty::vstore_fixed(_)) |
  206          ty::ty_estr(ty::vstore_fixed(_)) => {
  207              Some(deref_interior(InteriorElement(element_kind(t))))
  208          }
  209  
  210          _ => None
  211      }
  212  }

  347      pub fn cat_expr(&self, expr: @ast::expr) -> cmt {
  348          match self.tcx.adjustments.find(&expr.id) {
  349              None => {
  350                  // No adjustments.
  351                  self.cat_expr_unadjusted(expr)
  352              }
  353  
  354              Some(&@ty::AutoAddEnv(*)) => {
  355                  // Convert a bare fn to a closure by adding NULL env.
  356                  // Result is an rvalue.
  357                  let expr_ty = ty::expr_ty_adjusted(self.tcx, expr);
  358                  self.cat_rvalue_node(expr, expr_ty)
  359              }
  360  
  361              Some(
  362                  &@ty::AutoDerefRef(
  363                      ty::AutoDerefRef {
  364                          autoref: Some(_), _})) => {
  365                  // Equivalent to &*expr or something similar.
  366                  // Result is an rvalue.
  367                  let expr_ty = ty::expr_ty_adjusted(self.tcx, expr);
  368                  self.cat_rvalue_node(expr, expr_ty)
  369              }
  370  
  371              Some(
  372                  &@ty::AutoDerefRef(
  373                      ty::AutoDerefRef {
  374                          autoref: None, autoderefs: autoderefs})) => {
  375                  // Equivalent to *expr or something similar.
  376                  self.cat_expr_autoderefd(expr, autoderefs)
  377              }
  378          }
  379      }

  381      pub fn cat_expr_autoderefd(&self, expr: @ast::expr, autoderefs: uint)
  382                                 -> cmt {
  383          let mut cmt = self.cat_expr_unadjusted(expr);
  384          for deref in range(1u, autoderefs + 1) {
  385              cmt = self.cat_deref(expr, cmt, deref);
  386          }
  387          return cmt;
  388      }

  390      pub fn cat_expr_unadjusted(&self, expr: @ast::expr) -> cmt {
  391          debug!("cat_expr: id=%d expr=%s",
  392                 expr.id, pprust::expr_to_str(expr, self.tcx.sess.intr()));
  393  
  394          let expr_ty = self.expr_ty(expr);
  395          match expr.node {
  396            ast::expr_unary(_, ast::deref, e_base) => {
  397              if self.method_map.contains_key(&expr.id) {
  398                  return self.cat_rvalue_node(expr, expr_ty);
  399              }
  400  
  401              let base_cmt = self.cat_expr(e_base);
  402              self.cat_deref(expr, base_cmt, 0)
  403            }
  404  
  405            ast::expr_field(base, f_name, _) => {
  406              // Method calls are now a special syntactic form,
  407              // so `a.b` should always be a field.
  408              assert!(!self.method_map.contains_key(&expr.id));
  409  
  410              let base_cmt = self.cat_expr(base);
  411              self.cat_field(expr, base_cmt, f_name, self.expr_ty(expr))
  412            }
  413  
  414            ast::expr_index(_, base, _) => {
  415              if self.method_map.contains_key(&expr.id) {
  416                  return self.cat_rvalue_node(expr, expr_ty);
  417              }
  418  
  419              let base_cmt = self.cat_expr(base);
  420              self.cat_index(expr, base_cmt, 0)
  421            }
  422  
  423            ast::expr_path(_) | ast::expr_self => {
  424              let def = self.tcx.def_map.get_copy(&expr.id);

  448      pub fn cat_def(&self,
  449                     id: ast::NodeId,

  500                  cat_self(self_id)

  512            ast::def_upvar(upvar_id, inner, fn_node_id, _) => {
  513                let ty = ty::node_id_to_type(self.tcx, fn_node_id);
  514                match ty::get(ty).sty {
  515                    ty::ty_closure(ref closure_ty) => {
  516                        // Decide whether to use implicit reference or by copy/move
  517                        // capture for the upvar. This, combined with the onceness,
  518                        // determines whether the closure can move out of it.
  519                        let var_is_refd = match (closure_ty.sigil, closure_ty.onceness) {
  520                            // Many-shot stack closures can never move out.
  521                            (ast::BorrowedSigil, ast::Many) => true,
  522                            // 1-shot stack closures can move out with "-Z once-fns".
  523                            (ast::BorrowedSigil, ast::Once)
  524                                if self.tcx.sess.once_fns() => false,
  525                            (ast::BorrowedSigil, ast::Once) => true,
  526                            // Heap closures always capture by copy/move, and can
  527                            // move out iff they are once.
  528                            (ast::OwnedSigil, _) | (ast::ManagedSigil, _) => false,
  529  
  530                        };
  531                        if var_is_refd {
  532                            let upvar_cmt =
  533                                self.cat_def(id, span, expr_ty, *inner);

  535                                id:id,
  536                                span:span,
  537                                cat:cat_stack_upvar(upvar_cmt),
  538                                mutbl:upvar_cmt.mutbl.inherit(),
  539                                ty:upvar_cmt.ty
  540                            }
  541                        } else {
  542                            // FIXME #2152 allow mutation of moved upvars
  543                            @cmt_ {
  544                                id:id,
  545                                span:span,
  546                                cat:cat_copied_upvar(CopiedUpvar {
  547                                    upvar_id: upvar_id,
  548                                    onceness: closure_ty.onceness}),
  549                                mutbl:McImmutable,
  550                                ty:expr_ty
  551                            }
  552                        }
  553                    }
  554                    _ => {
  555                        self.tcx.sess.span_bug(
  556                            span,
  557                            fmt!("Upvar of non-closure %? - %s",
  558                                 fn_node_id, ty.repr(self.tcx)));
  559                    }
  560                }
  561            }
  562  
  563            ast::def_local(vid, mutbl) => {
  564              let m = if mutbl {McDeclared} else {McImmutable};

  714      pub fn cat_index<N:ast_node>(&self,
  715                                   elt: N,
  716                                   base_cmt: cmt,
  717                                   derefs: uint)
  718                                   -> cmt {
  719          //! Creates a cmt for an indexing operation (`[]`); this
  720          //! indexing operation may occurs as part of an
  721          //! AutoBorrowVec, which when converting a `~[]` to an `&[]`
  722          //! effectively takes the address of the 0th element.
  723          //!
  724          //! One subtle aspect of indexing that may not be
  725          //! immediately obvious: for anything other than a fixed-length
  726          //! vector, an operation like `x[y]` actually consists of two
  727          //! disjoint (from the point of view of borrowck) operations.
  728          //! The first is a deref of `x` to create a pointer `p` that points
  729          //! at the first element in the array. The second operation is
  730          //! an index which adds `y*sizeof(T)` to `p` to obtain the
  731          //! pointer to `x[y]`. `cat_index` will produce a resulting
  732          //! cmt containing both this deref and the indexing,
  733          //! presuming that `base_cmt` is not of fixed-length type.
  734          //!
  735          //! In the event that a deref is needed, the "deref count"
  736          //! is taken from the parameter `derefs`. See the comment
  737          //! on the def'n of `root_map_key` in borrowck/mod.rs
  738          //! for more details about deref counts; the summary is
  739          //! that `derefs` should be 0 for an explicit indexing
  740          //! operation and N+1 for an indexing that is part of
  741          //! an auto-adjustment, where N is the number of autoderefs
  742          //! in that adjustment.
  743          //!
  744          //! # Parameters
  745          //! - `elt`: the AST node being indexed
  746          //! - `base_cmt`: the cmt of `elt`
  747          //! - `derefs`: the deref number to be used for
  748          //!   the implicit index deref, if any (see above)
  749  
  750          let element_ty = match ty::index(base_cmt.ty) {
  751            Some(ref mt) => mt.ty,
  752            None => {
  753              self.tcx.sess.span_bug(
  754                  elt.span(),
  755                  fmt!("Explicit index of non-index type `%s`",
  756                       ty_to_str(self.tcx, base_cmt.ty)));
  757            }
  758          };
  759  
  760          return match deref_kind(self.tcx, base_cmt.ty) {
  761            deref_ptr(ptr) => {
  762              // for unique ptrs, we inherit mutability from the
  763              // owning reference.
  764              let m = match ptr {
  765                uniq_ptr => {
  766                  base_cmt.mutbl.inherit()
  767                }
  768                gc_ptr(m) | region_ptr(m, _) | unsafe_ptr(m) => {
  769                  MutabilityCategory::from_mutbl(m)
  770                }
  771              };
  772  
  773              // the deref is explicit in the resulting cmt
  774              let deref_cmt = @cmt_ {
  775                  id:elt.id(),
  776                  span:elt.span(),
  777                  cat:cat_deref(base_cmt, derefs, ptr),
  778                  mutbl:m,
  779                  ty:element_ty
  780              };
  781  
  782              interior(elt, deref_cmt, base_cmt.ty, m, element_ty)
  783            }
  784  
  785            deref_interior(_) => {
  786              // fixed-length vectors have no deref
  787              let m = base_cmt.mutbl.inherit();
  788              interior(elt, base_cmt, base_cmt.ty, m, element_ty)
  789            }
  790          };
  791

  793                                   of_cmt: cmt,
  794                                   vec_ty: ty::t,

  796                                   element_ty: ty::t) -> cmt
  797          {

  800                  span:elt.span(),
  801                  cat:cat_interior(of_cmt, InteriorElement(element_kind(vec_ty))),

  803                  ty:element_ty
  804              }
  805          }
  806      }

  837      pub fn cat_pattern(&self,
  838                         cmt: cmt,
  839                         pat: @ast::pat,
  840                         op: &fn(cmt, @ast::pat)) {
  841          // Here, `cmt` is the categorization for the value being
  842          // matched and pat is the pattern it is being matched against.
  843          //
  844          // In general, the way that this works is that we walk down
  845          // the pattern, constructing a cmt that represents the path
  846          // that will be taken to reach the value being matched.
  847          //
  848          // When we encounter named bindings, we take the cmt that has
  849          // been built up and pass it off to guarantee_valid() so that

   12   * # Categorization
   13   *
   14   * The job of the categorization module is to analyze an expression to
   15   * determine what kind of memory is used in evaluating it (for example,
   16   * where dereferences occur and what kind of pointer is dereferenced;
   17   * whether the memory is mutable; etc)
   18   *
   19   * Categorization effectively transforms all of our expressions into
   20   * expressions of the following forms (the actual enum has many more
   21   * possibilities, naturally, but they are all variants of these base
   22   * forms):
   23   *
   24   *     E = rvalue    // some computed rvalue
   25   *       | x         // address of a local variable or argument
   26   *       | *E        // deref of a ptr
   27   *       | E.comp    // access to an interior component
   28   *
   29   * Imagine a routine ToAddr(Expr) that evaluates an expression and returns an
   30   * address where the result is to be found.  If Expr is an lvalue, then this
   31   * is the address of the lvalue.  If Expr is an rvalue, this is the address of
   32   * some temporary spot in memory where the result is stored.
   33   *
   34   * Now, cat_expr() classies the expression Expr and the address A=ToAddr(Expr)
   35   * as follows:
   36   *
   37   * - cat: what kind of expression was this?  This is a subset of the
   38   *   full expression forms which only includes those that we care about
   39   *   for the purpose of the analysis.
   40   * - mutbl: mutability of the address A
   41   * - ty: the type of data found at the address A
   42   *
   43   * The resulting categorization tree differs somewhat from the expressions
   44   * themselves.  For example, auto-derefs are explicit.  Also, an index a[b] is
   45   * decomposed into two operations: a derefence to reach the array data and
   46   * then an index to jump forward to the relevant item.
   47   *
   48   * ## By-reference upvars
   49   *
   50   * One part of the translation which may be non-obvious is that we translate
   51   * closure upvars into the dereference of a borrowed pointer; this more closely
   52   * resembles the runtime translation. So, for example, if we had:
   53   *
   54   *     let mut x = 3;
   55   *     let y = 5;
   56   *     let inc = || x += y;
   57   *
   58   * Then when we categorize `x` (*within* the closure) we would yield a
   59   * result of `*x'`, effectively, where `x'` is a `cat_upvar` reference
   60   * tied to `x`. The type of `x'` will be a borrowed pointer.
   61   */

  134  // `cmt`: "Category, Mutability, and Type".
  135  //
  136  // a complete categorization of a value indicating where it originated
  137  // and how it is located, as well as the mutability of the memory in
  138  // which the value is stored.
  139  //
  140  // *WARNING* The field `cmt.type` is NOT necessarily the same as the
  141  // result of `node_id_to_type(cmt.id)`. This is because the `id` is
  142  // always the `id` of the node producing the type; in an expression
  143  // like `*x`, the type of this deref node is the deref'd type (`T`),
  144  // but in a pattern like `@x`, the `@x` pattern is again a
  145  // dereference, but its type is the type *before* the dereference
  146  // (`@T`). So use `cmt.type` to find the type of the value in a consistent
  147  // fashion. For more details, see the method `cat_pattern`

  154      pub ty: ty::t                  // type of the expr (*see WARNING above*)
  155  }

  166  // Categorizes a derefable type.  Note that we include vectors and strings as
  167  // derefable (we model an index as the combination of a deref and then a
  168  // pointer adjustment).
  169  pub fn opt_deref_kind(t: ty::t) -> Option<deref_kind> {
  170      match ty::get(t).sty {
  171          ty::ty_uniq(_) |
  172          ty::ty_trait(~ty::TyTrait { store: ty::UniqTraitStore, .. }) |
  173          ty::ty_vec(_, ty::VstoreUniq) |
  174          ty::ty_str(ty::VstoreUniq) |
  175          ty::ty_closure(~ty::ClosureTy {store: ty::UniqTraitStore, ..}) => {
  176              Some(deref_ptr(OwnedPtr))
  177          }
  178  
  179          ty::ty_rptr(r, mt) => {
  180              let kind = ty::BorrowKind::from_mutbl(mt.mutbl);
  181              Some(deref_ptr(BorrowedPtr(kind, r)))
  182          }
  183          ty::ty_vec(_, ty::VstoreSlice(r, mutbl)) |
  184          ty::ty_trait(~ty::TyTrait { store: ty::RegionTraitStore(r, mutbl), .. }) => {
  185              let kind = ty::BorrowKind::from_mutbl(mutbl);
  186              Some(deref_ptr(BorrowedPtr(kind, r)))
  187          }
  188  
  189          ty::ty_str(ty::VstoreSlice(r, ())) |
  190          ty::ty_closure(~ty::ClosureTy {store: ty::RegionTraitStore(r, _), ..}) => {
  191              Some(deref_ptr(BorrowedPtr(ty::ImmBorrow, r)))
  192          }
  193  
  194          ty::ty_box(..) => {
  195              Some(deref_ptr(GcPtr))
  196          }
  197  
  198          ty::ty_ptr(ref mt) => {
  199              Some(deref_ptr(UnsafePtr(mt.mutbl)))
  200          }
  201  
  202          ty::ty_enum(..) |
  203          ty::ty_struct(..) => { // newtype
  204              Some(deref_interior(InteriorField(PositionalField(0))))
  205          }
  206  
  207          ty::ty_vec(_, ty::VstoreFixed(_)) |
  208          ty::ty_str(ty::VstoreFixed(_)) => {
  209              Some(deref_interior(InteriorElement(element_kind(t))))
  210          }
  211  
  212          _ => None
  213      }
  214  }

  248  /**
  249   * The `Typer` trait provides the interface for the mem-categorization
  250   * module to the results of the type check. It can be used to query
  251   * the type assigned to an expression node, to inquire after adjustments,
  252   * and so on.
  253   *
  254   * This interface is needed because mem-categorization is used from
  255   * two places: `regionck` and `borrowck`. `regionck` executes before
  256   * type inference is complete, and hence derives types and so on from
  257   * intermediate tables.  This also implies that type errors can occur,
  258   * and hence `node_ty()` and friends return a `Result` type -- any
  259   * error will propagate back up through the mem-categorization
  260   * routines.
  261   *
  262   * In the borrow checker, in contrast, type checking is complete and we
  263   * know that no errors have occurred, so we simply consult the tcx and we
  264   * can be sure that only `Ok` results will occur.
  265   */

  267      fn tcx<'a>(&'a self) -> &'a ty::ctxt;
  268      fn node_ty(&mut self, id: ast::NodeId) -> McResult<ty::t>;
  269      fn node_method_ty(&self, method_call: typeck::MethodCall) -> Option<ty::t>;
  270      fn adjustment(&mut self, node_id: ast::NodeId) -> Option<@ty::AutoAdjustment>;
  271      fn is_method_call(&mut self, id: ast::NodeId) -> bool;
  272      fn temporary_scope(&mut self, rvalue_id: ast::NodeId) -> Option<ast::NodeId>;
  273      fn upvar_borrow(&mut self, upvar_id: ty::UpvarId) -> ty::UpvarBorrow;

  422      pub fn cat_expr_autoderefd(&mut self, expr: &ast::Expr, autoderefs: uint)
  423                                 -> McResult<cmt> {
  424          let mut cmt = if_ok!(self.cat_expr_unadjusted(expr));

  427          }
  428          return Ok(cmt);

  431      pub fn cat_expr_unadjusted(&mut self, expr: &ast::Expr) -> McResult<cmt> {
  432          debug!("cat_expr: id={} expr={}", expr.id, expr.repr(self.tcx()));
  433  
  434          let expr_ty = if_ok!(self.expr_ty(expr));
  435          match expr.node {
  436            ast::ExprUnary(ast::UnDeref, e_base) => {
  437              let base_cmt = if_ok!(self.cat_expr(e_base));
  438              Ok(self.cat_deref(expr, base_cmt, 0))
  439            }
  440  
  441            ast::ExprField(base, f_name, _) => {
  442              let base_cmt = if_ok!(self.cat_expr(base));
  443              Ok(self.cat_field(expr, base_cmt, f_name, expr_ty))
  444            }
  445  
  446            ast::ExprIndex(base, _) => {
  447              if self.typer.is_method_call(expr.id) {
  448                  return Ok(self.cat_rvalue_node(expr.id(), expr.span(), expr_ty));
  449              }
  450  
  451              let base_cmt = if_ok!(self.cat_expr(base));
  452              Ok(self.cat_index(expr, base_cmt, 0))
  453            }
  454  
  455            ast::ExprPath(_) => {
  456              let def = self.tcx().def_map.borrow().get_copy(&expr.id);
  457              self.cat_def(expr.id, expr.span, expr_ty, def)
  458            }
  459  
  460            ast::ExprParen(e) => self.cat_expr_unadjusted(e),
  461  
  462            ast::ExprAddrOf(..) | ast::ExprCall(..) |
  463            ast::ExprAssign(..) | ast::ExprAssignOp(..) |
  464            ast::ExprFnBlock(..) | ast::ExprProc(..) | ast::ExprRet(..) |
  465            ast::ExprUnary(..) |
  466            ast::ExprMethodCall(..) | ast::ExprCast(..) | ast::ExprVstore(..) |
  467            ast::ExprVec(..) | ast::ExprTup(..) | ast::ExprIf(..) |
  468            ast::ExprBinary(..) | ast::ExprWhile(..) |
  469            ast::ExprBlock(..) | ast::ExprLoop(..) | ast::ExprMatch(..) |
  470            ast::ExprLit(..) | ast::ExprBreak(..) | ast::ExprMac(..) |
  471            ast::ExprAgain(..) | ast::ExprStruct(..) | ast::ExprRepeat(..) |
  472            ast::ExprInlineAsm(..) | ast::ExprBox(..) => {
  473              Ok(self.cat_rvalue_node(expr.id(), expr.span(), expr_ty))
  474            }
  475  
  476            ast::ExprForLoop(..) => fail!("non-desugared expr_for_loop")

  478      }

  480      pub fn cat_def(&mut self,
  481                     id: ast::NodeId,

  485                     -> McResult<cmt> {

  490            ast::DefStruct(..) | ast::DefVariant(..) => {
  491                  Ok(self.cat_rvalue_node(id, span, expr_ty))
  492            }

  494            ast::DefForeignMod(_) | ast::DefStatic(_, false) |
  495            ast::DefUse(_) | ast::DefTrait(_) | ast::DefTy(_) | ast::DefPrimTy(_) |

  499                    id:id,
  500                    span:span,

  502                    mutbl: McImmutable,
  503                    ty:expr_ty
  504                })
  505            }
  506  
  507            ast::DefStatic(_, true) => {
  508                Ok(@cmt_ {
  509                    id:id,
  510                    span:span,
  511                    cat:cat_static_item,
  512                    mutbl: McDeclared,
  513                    ty:expr_ty
  514                })
  515            }
  516  
  517            ast::DefArg(vid, binding_mode) => {
  518              // Idea: make this could be rewritten to model by-ref
  519              // stuff as `&const` and `&mut`?
  520  
  521              // m: mutability of the argument
  522              let m = match binding_mode {
  523                  ast::BindByValue(ast::MutMutable) => McDeclared,
  524                  _ => McImmutable
  525              };
  526              Ok(@cmt_ {
  527                  id: id,
  528                  span: span,
  529                  cat: cat_arg(vid),
  530                  mutbl: m,
  531                  ty:expr_ty
  532              })
  533            }
  534  
  535            ast::DefUpvar(var_id, _, fn_node_id, _) => {
  536                let ty = if_ok!(self.node_ty(fn_node_id));
  537                match ty::get(ty).sty {
  538                    ty::ty_closure(ref closure_ty) => {
  539                        // Decide whether to use implicit reference or by copy/move
  540                        // capture for the upvar. This, combined with the onceness,
  541                        // determines whether the closure can move out of it.
  542                        let var_is_refd = match (closure_ty.store, closure_ty.onceness) {
  543                            // Many-shot stack closures can never move out.
  544                            (ty::RegionTraitStore(..), ast::Many) => true,
  545                            // 1-shot stack closures can move out.
  546                            (ty::RegionTraitStore(..), ast::Once) => false,
  547                            // Heap closures always capture by copy/move, and can
  548                            // move out if they are once.
  549                            (ty::UniqTraitStore, _) => false,
  550  
  551                        };
  552                        if var_is_refd {
  553                            self.cat_upvar(id, span, var_id, fn_node_id)
  554                        } else {
  555                            // FIXME #2152 allow mutation of moved upvars
  556                            Ok(@cmt_ {
  557                                id:id,
  558                                span:span,
  559                                cat:cat_copied_upvar(CopiedUpvar {
  560                                    upvar_id: var_id,
  561                                    onceness: closure_ty.onceness}),
  562                                mutbl:McImmutable,
  563                                ty:expr_ty
  564                            })
  565                        }
  566                    }
  567                    _ => {
  568                        self.tcx().sess.span_bug(
  569                            span,
  570                            format!("Upvar of non-closure {} - {}",
  571                                    fn_node_id, ty.repr(self.tcx())));
  572                    }
  573                }
  574            }
  575  
  576            ast::DefLocal(vid, binding_mode) |
  577            ast::DefBinding(vid, binding_mode) => {
  578              // by-value/by-ref bindings are local variables
  579              let m = match binding_mode {
  580                  ast::BindByValue(ast::MutMutable) => McDeclared,
  581                  _ => McImmutable
  582              };
  583  
  584              Ok(@cmt_ {
  585                  id: id,
  586                  span: span,
  587                  cat: cat_local(vid),
  588                  mutbl: m,
  589                  ty: expr_ty
  590              })
  591            }
  592          }
  593      }

  595      fn cat_upvar(&mut self,
  596                   id: ast::NodeId,
  597                   span: Span,
  598                   var_id: ast::NodeId,
  599                   fn_node_id: ast::NodeId)
  600                   -> McResult<cmt> {
  601          /*!
  602           * Upvars through a closure are in fact indirect
  603           * references. That is, when a closure refers to a
  604           * variable from a parent stack frame like `x = 10`,
  605           * that is equivalent to `*x_ = 10` where `x_` is a
  606           * borrowed pointer (`&mut x`) created when the closure
  607           * was created and store in the environment. This
  608           * equivalence is expose in the mem-categorization.
  609           */
  610  
  611          let upvar_id = ty::UpvarId { var_id: var_id,
  612                                       closure_expr_id: fn_node_id };
  613  
  614          let upvar_borrow = self.typer.upvar_borrow(upvar_id);
  615  
  616          let var_ty = if_ok!(self.node_ty(var_id));
  617  
  618          // We can't actually represent the types of all upvars
  619          // as user-describable types, since upvars support const
  620          // and unique-imm borrows! Therefore, we cheat, and just
  621          // give err type. Nobody should be inspecting this type anyhow.
  622          let upvar_ty = ty::mk_err();
  623  
  624          let base_cmt = @cmt_ {
  625              id:id,
  626              span:span,
  627              cat:cat_upvar(upvar_id, upvar_borrow),
  628              mutbl:McImmutable,
  629              ty:upvar_ty,
  630          };
  631  
  632          let ptr = BorrowedPtr(upvar_borrow.kind, upvar_borrow.region);
  633  
  634          let deref_cmt = @cmt_ {
  635              id:id,
  636              span:span,
  637              cat:cat_deref(base_cmt, 0, ptr),
  638              mutbl:MutabilityCategory::from_borrow_kind(upvar_borrow.kind),
  639              ty:var_ty,
  640          };
  641  
  642          Ok(deref_cmt)
  643      }

  751      pub fn cat_index<N:ast_node>(&mut self,
  752                                   elt: &N,
  753                                   base_cmt: cmt,
  754                                   derefs: uint)
  755                                   -> cmt {
  756          //! Creates a cmt for an indexing operation (`[]`); this
  757          //! indexing operation may occurs as part of an
  758          //! AutoBorrowVec, which when converting a `~[]` to an `&[]`
  759          //! effectively takes the address of the 0th element.
  760          //!
  761          //! One subtle aspect of indexing that may not be
  762          //! immediately obvious: for anything other than a fixed-length
  763          //! vector, an operation like `x[y]` actually consists of two
  764          //! disjoint (from the point of view of borrowck) operations.
  765          //! The first is a deref of `x` to create a pointer `p` that points
  766          //! at the first element in the array. The second operation is
  767          //! an index which adds `y*sizeof(T)` to `p` to obtain the
  768          //! pointer to `x[y]`. `cat_index` will produce a resulting
  769          //! cmt containing both this deref and the indexing,
  770          //! presuming that `base_cmt` is not of fixed-length type.
  771          //!
  772          //! In the event that a deref is needed, the "deref count"
  773          //! is taken from the parameter `derefs`. See the comment
  774          //! on the def'n of `root_map_key` in borrowck/mod.rs
  775          //! for more details about deref counts; the summary is
  776          //! that `derefs` should be 0 for an explicit indexing
  777          //! operation and N+1 for an indexing that is part of
  778          //! an auto-adjustment, where N is the number of autoderefs
  779          //! in that adjustment.
  780          //!
  781          //! # Parameters
  782          //! - `elt`: the AST node being indexed
  783          //! - `base_cmt`: the cmt of `elt`
  784          //! - `derefs`: the deref number to be used for
  785          //!   the implicit index deref, if any (see above)
  786  
  787          let element_ty = match ty::index(base_cmt.ty) {
  788            Some(ty) => ty,
  789            None => {
  790              self.tcx().sess.span_bug(
  791                  elt.span(),
  792                  format!("Explicit index of non-index type `{}`",
  793                       base_cmt.ty.repr(self.tcx())));
  794            }
  795          };
  796  
  797          return match deref_kind(self.tcx(), base_cmt.ty) {
  798            deref_ptr(ptr) => {
  799              // for unique ptrs, we inherit mutability from the
  800              // owning reference.
  801              let m = MutabilityCategory::from_pointer_kind(base_cmt.mutbl, ptr);
  802  
  803              // the deref is explicit in the resulting cmt
  804              let deref_cmt = @cmt_ {
  805                  id:elt.id(),
  806                  span:elt.span(),
  807                  cat:cat_deref(base_cmt, derefs, ptr),
  808                  mutbl:m,
  809                  ty:element_ty
  810              };
  811  
  812              interior(elt, deref_cmt, base_cmt.ty, m.inherit(), element_ty)
  813            }
  814  
  815            deref_interior(_) => {

  917      pub fn cat_pattern(&mut self,
  918                         cmt: cmt,
  919                         pat: @ast::Pat,
  920                         op: |&mut MemCategorizationContext<TYPER>,
  921                              cmt,
  922                              @ast::Pat|)
  923                         -> McResult<()> {
  924          // Here, `cmt` is the categorization for the value being
  925          // matched and pat is the pattern it is being matched against.
  926          //
  927          // In general, the way that this works is that we walk down
  928          // the pattern, constructing a cmt that represents the path
  929          // that will be taken to reach the value being matched.
  930          //
  931          // When we encounter named bindings, we take the cmt that has
  932          // been built up and pass it off to guarantee_valid() so that
  933          // we can be sure that the binding will remain valid for the
  934          // duration of the arm.
  935          //
  936          // (*2) There is subtlety concerning the correspondence between
  937          // pattern ids and types as compared to *expression* ids and
  938          // types. This is explained briefly. on the definition of the
  939          // type `cmt`, so go off and read what it says there, then
  940          // come back and I'll dive into a bit more detail here. :) OK,
  941          // back?
  942          //
  943          // In general, the id of the cmt should be the node that
  944          // "produces" the value---patterns aren't executable code
  945          // exactly, but I consider them to "execute" when they match a
  946          // value, and I consider them to produce the value that was
  947          // matched. So if you have something like:
  948          //
  949          //     let x = @@3;
  950          //     match x {
  951          //       @@y { ... }
  952          //     }
  953          //
  954          // In this case, the cmt and the relevant ids would be:
  955          //
  956          //     CMT             Id                  Type of Id Type of cmt
  957          //
  958          //     local(x)->@->@
  959          //     ^~~~~~~^        `x` from discr      @@int      @@int
  960          //     ^~~~~~~~~~^     `@@y` pattern node  @@int      @int
  961          //     ^~~~~~~~~~~~~^  `@y` pattern node   @int       int
  962          //
  963          // You can see that the types of the id and the cmt are in
  964          // sync in the first line, because that id is actually the id
  965          // of an expression. But once we get to pattern ids, the types
  966          // step out of sync again. So you'll see below that we always
  967          // get the type of the *subpattern* and use that.
  968  
  969          debug!("cat_pattern: id={} pat={} cmt={}",
  970                 pat.id, pprust::pat_to_str(pat),
  971                 cmt.repr(self.tcx()));
  972  
  973          op(self, cmt, pat);
  974  
  975          match pat.node {
  976            ast::PatWild | ast::PatWildMulti => {
  977              // _
  978            }
  979  
  980            ast::PatEnum(_, None) => {
  981              // variant(..)
  982            }
  983            ast::PatEnum(_, Some(ref subpats)) => {
  984              match self.tcx().def_map.borrow().find(&pat.id) {
  985                  Some(&ast::DefVariant(enum_did, _, _)) => {
  986                      // variant(x, y, z)
  987  
  988                      let downcast_cmt = {
  989                          if ty::enum_is_univariant(self.tcx(), enum_did) {
  990                              cmt // univariant, no downcast needed
  991                          } else {
  992                              self.cat_downcast(pat, cmt, cmt.ty)
  993                          }
  994                      };
  995  
  996                      for (i, &subpat) in subpats.iter().enumerate() {
  997                          let subpat_ty = if_ok!(self.pat_ty(subpat)); // see (*2)

 1000                              self.cat_imm_interior(

 1002                                  InteriorField(PositionalField(i)));
 1003  
 1004                          if_ok!(self.cat_pattern(subcmt, subpat, |x,y,z| op(x,y,z)));
 1005                      }
 1006                  }
 1007                  Some(&ast::DefFn(..)) |
 1008                  Some(&ast::DefStruct(..)) => {
 1009                      for (i, &subpat) in subpats.iter().enumerate() {

 1011                          let cmt_field =
 1012                              self.cat_imm_interior(
 1013                                  pat, cmt, subpat_ty,
 1014                                  InteriorField(PositionalField(i)));

 1018                  Some(&ast::DefStatic(..)) => {

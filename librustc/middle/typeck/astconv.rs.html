  11  /*!
  12   * Conversion from AST representation of types to the ty.rs
  13   * representation.  The main routine here is `ast_ty_to_ty()`: each use
  14   * is parameterized by an instance of `AstConv` and a `RegionScope`.
  15   *
  16   * The parameterization of `ast_ty_to_ty()` is because it behaves
  17   * somewhat differently during the collect and check phases,
  18   * particularly with respect to looking up the types of top-level
  19   * items.  In the collect phase, the crate context is used as the
  20   * `AstConv` instance; in this phase, the `get_item_ty()` function
  21   * triggers a recursive call to `ty_of_item()`  (note that
  22   * `ast_ty_to_ty()` will detect recursive types and report an error).
  23   * In the check phase, when the @FnCtxt is used as the `AstConv`,
  24   * `get_item_ty()` just looks up the item type in `tcx.tcache`.
  25   *
  26   * The `RegionScope` trait controls how region references are
  27   * handled.  It has two methods which are used to resolve anonymous
  28   * region references (e.g., `&T`) and named region references (e.g.,
  29   * `&a.T`).  There are numerous region scopes that can be used, but most
  30   * commonly you want either `EmptyRscope`, which permits only the static
  31   * region, or `TypeRscope`, which permits the self region if the type in
  32   * question is parameterized by a region.
  33   *
  34   * Unlike the `AstConv` trait, the region scope can change as we descend
  35   * the type.  This is to accommodate the fact that (a) fn types are binding
  36   * scopes and (b) the default region may change.  To understand case (a),
  37   * consider something like:
  38   *
  39   *   type foo = { x: &a.int, y: &fn(&a.int) }
  40   *
  41   * The type of `x` is an error because there is no region `a` in scope.
  42   * In the type of `y`, however, region `a` is considered a bound region
  43   * as it does not already appear in scope.
  44   *
  45   * Case (b) says that if you have a type:
  46   *   type foo<'self> = ...;
  47   *   type bar = fn(&foo, &a.foo)
  48   * The fully expanded version of type bar is:
  49   *   type bar = fn(&'foo &, &a.foo<'a>)
  50   * Note that the self region for the `foo` defaulted to `&` in the first
  51   * case but `&a` in the second.  Basically, defaults that appear inside
  52   * an rptr (`&r.T`) use the region `r` that appears in the rptr.
  53   */

  84  pub fn get_region_reporting_err(
  85      tcx: ty::ctxt,
  86      span: Span,
  87      a_r: &Option<ast::Lifetime>,
  88      res: Result<ty::Region, RegionError>) -> ty::Region
  89  {
  90      match res {
  91          result::Ok(r) => r,
  92          result::Err(ref e) => {
  93              let descr = match a_r {
  94                  &None => ~"anonymous lifetime",
  95                  &Some(ref a) => fmt!("lifetime %s",
  96                                  lifetime_to_str(a, tcx.sess.intr()))
  97              };
  98              tcx.sess.span_err(
  99                  span,
 100                  fmt!("Illegal %s: %s",
 101                       descr, e.msg));
 102              e.replacement
 103          }
 104      }
 105  }

 107  pub fn ast_region_to_region<AC:AstConv,RS:RegionScope + Clone + 'static>(
 108      this: &AC,
 109      rscope: &RS,
 110      default_span: Span,
 111      opt_lifetime: &Option<ast::Lifetime>) -> ty::Region
 112  {
 113      let (span, res) = match opt_lifetime {
 114          &None => {
 115              (default_span, rscope.anon_region(default_span))
 116          }
 117          &Some(ref lifetime) if lifetime.ident == special_idents::statik => {
 118              (lifetime.span, Ok(ty::re_static))
 119          }
 120          &Some(ref lifetime) if lifetime.ident == special_idents::self_ => {
 121              (lifetime.span, rscope.self_region(lifetime.span))
 122          }
 123          &Some(ref lifetime) => {
 124              (lifetime.span, rscope.named_region(lifetime.span,
 125                                                  lifetime.ident))
 126          }
 127      };
 128  
 129      get_region_reporting_err(this.tcx(), span, opt_lifetime, res)
 130  }

 132  fn ast_path_substs<AC:AstConv,RS:RegionScope + Clone + 'static>(
 133      this: &AC,
 134      rscope: &RS,
 135      def_id: ast::DefId,
 136      decl_generics: &ty::Generics,
 137      self_ty: Option<ty::t>,
 138      path: &ast::Path) -> ty::substs
 139  {
 140      /*!
 141       *
 142       * Given a path `path` that refers to an item `I` with the
 143       * declared generics `decl_generics`, returns an appropriate
 144       * set of substitutions for this particular reference to `I`.
 145       */
 146  
 147      let tcx = this.tcx();
 148  
 149      // If the type is parameterized by the this region, then replace this
 150      // region with the current anon region binding (in other words,
 151      // whatever & would get replaced with).
 152      let regions = match (&decl_generics.region_param,
 153                           &path.segments.last().lifetime) {
 154          (&None, &None) => {
 155              opt_vec::Empty
 156          }
 157          (&None, &Some(_)) => {
 158              tcx.sess.span_err(
 159                  path.span,
 160                  fmt!("no region bound is allowed on `%s`, \
 161                        which is not declared as containing region pointers",
 162                       ty::item_path_str(tcx, def_id)));
 163              opt_vec::Empty
 164          }
 165          (&Some(_), &None) => {
 166              let res = rscope.anon_region(path.span);
 167              let r = get_region_reporting_err(this.tcx(), path.span, &None, res);
 168              opt_vec::with(r)
 169          }
 170          (&Some(_), &Some(_)) => {
 171              opt_vec::with(
 172                  ast_region_to_region(this,
 173                                       rscope,
 174                                       path.span,
 175                                       &path.segments.last().lifetime))
 176          }
 177      };
 178  
 179      // Convert the type parameters supplied by the user.
 180      let supplied_type_parameter_count =
 181          path.segments.iter().flat_map(|s| s.types.iter()).len();
 182      if decl_generics.type_param_defs.len() != supplied_type_parameter_count {
 183          this.tcx().sess.span_fatal(
 184              path.span,
 185              fmt!("wrong number of type arguments: expected %u but found %u",
 186                   decl_generics.type_param_defs.len(),
 187                   supplied_type_parameter_count));
 188      }
 189      let tps = path.segments
 190                    .iter()
 191                    .flat_map(|s| s.types.iter())
 192                    .map(|a_t| ast_ty_to_ty(this, rscope, a_t))
 193                    .collect();
 194  
 195      substs {
 196          regions: ty::NonerasedRegions(regions),
 197          self_ty: self_ty,
 198          tps: tps
 199      }
 200  }

 202  pub fn ast_path_to_substs_and_ty<AC:AstConv,
 203                                   RS:RegionScope + Clone + 'static>(
 204                                   this: &AC,
 205                                   rscope: &RS,
 206                                   did: ast::DefId,
 207                                   path: &ast::Path)
 208                                   -> ty_param_substs_and_ty {
 209      let tcx = this.tcx();
 210      let ty::ty_param_bounds_and_ty {
 211          generics: generics,
 212          ty: decl_ty
 213      } = this.get_item_ty(did);
 214  
 215      let substs = ast_path_substs(this, rscope, did, &generics, None, path);
 216      let ty = ty::subst(tcx, &substs, decl_ty);
 217      ty_param_substs_and_ty { substs: substs, ty: ty }
 218  }

 220  pub fn ast_path_to_trait_ref<AC:AstConv,RS:RegionScope + Clone + 'static>(
 221      this: &AC,
 222      rscope: &RS,
 223      trait_def_id: ast::DefId,
 224      self_ty: Option<ty::t>,
 225      path: &ast::Path) -> @ty::TraitRef

 227      let trait_def =
 228          this.get_trait_def(trait_def_id);
 229      let substs =
 230          ast_path_substs(
 231              this,
 232              rscope,

 238          @ty::TraitRef {def_id: trait_def_id,

 268      fn ast_mt_to_mt<AC:AstConv, RS:RegionScope + Clone + 'static>(
 269          this: &AC, rscope: &RS, mt: &ast::mt) -> ty::mt {
 270  
 271          ty::mt {ty: ast_ty_to_ty(this, rscope, mt.ty), mutbl: mt.mutbl}
 272      }
 273  
 274      // Handle @, ~, and & being able to mean estrs and evecs.
 275      // If a_seq_ty is a str or a vec, make it an estr/evec.
 276      // Also handle first-class trait types.
 277      fn mk_pointer<AC:AstConv,RS:RegionScope + Clone + 'static>(
 278          this: &AC,
 279          rscope: &RS,
 280          a_seq_ty: &ast::mt,
 281          vst: ty::vstore,
 282          constr: &fn(ty::mt) -> ty::t) -> ty::t
 283      {
 284          let tcx = this.tcx();
 285  
 286          match a_seq_ty.ty.node {
 287              ast::ty_vec(ref mt) => {
 288                  let mut mt = ast_mt_to_mt(this, rscope, mt);
 289                  if a_seq_ty.mutbl == ast::MutMutable {
 290                      mt = ty::mt { ty: mt.ty, mutbl: a_seq_ty.mutbl };
 291                  }
 292                  return ty::mk_evec(tcx, mt, vst);
 293              }
 294              ast::ty_path(ref path, ref bounds, id) => {
 295                  // Note that the "bounds must be empty if path is not a trait"
 296                  // restriction is enforced in the below case for ty_path, which
 297                  // will run after this as long as the path isn't a trait.
 298                  match tcx.def_map.find(&id) {
 299                      Some(&ast::DefPrimTy(ast::ty_str)) if a_seq_ty.mutbl == ast::MutImmutable => {
 300                          check_path_args(tcx, path, NO_TPS | NO_REGIONS);
 301                          return ty::mk_estr(tcx, vst);
 302                      }
 303                      Some(&ast::DefTrait(trait_def_id)) => {
 304                          let result = ast_path_to_trait_ref(
 305                              this, rscope, trait_def_id, None, path);
 306                          let trait_store = match vst {
 307                              ty::vstore_box => ty::BoxTraitStore,
 308                              ty::vstore_uniq => ty::UniqTraitStore,
 309                              ty::vstore_slice(r) => {
 310                                  ty::RegionTraitStore(r)
 311                              }
 312                              ty::vstore_fixed(*) => {
 313                                  tcx.sess.span_err(
 314                                      path.span,
 315                                      "@trait, ~trait or &trait are the only supported \
 316                                       forms of casting-to-trait");
 317                                  ty::BoxTraitStore
 318                              }
 319                          };
 320                          let bounds = conv_builtin_bounds(this.tcx(), bounds, trait_store);
 321                          return ty::mk_trait(tcx,
 322                                              result.def_id,
 323                                              result.substs.clone(),
 324                                              trait_store,
 325                                              a_seq_ty.mutbl,
 326                                              bounds);
 327                      }
 328                      _ => {}
 329                  }
 330              }
 331              _ => {}
 332          }
 333  
 334          let seq_ty = ast_mt_to_mt(this, rscope, a_seq_ty);
 335          return constr(seq_ty);
 336      }

 338      fn check_path_args(tcx: ty::ctxt,
 339                         path: &ast::Path,
 340                         flags: uint) {
 341          if (flags & NO_TPS) != 0u {
 342              if !path.segments.iter().all(|s| s.types.is_empty()) {
 343                  tcx.sess.span_err(
 344                      path.span,
 345                      "type parameters are not allowed on this type");
 346              }
 347          }
 348  
 349          if (flags & NO_REGIONS) != 0u {
 350              if path.segments.last().lifetime.is_some() {
 351                  tcx.sess.span_err(
 352                      path.span,
 353                      "region parameters are not allowed on this type");
 354              }
 355          }
 356      }
 357  
 358      let tcx = this.tcx();
 359  
 360      match tcx.ast_ty_to_ty_cache.find(&ast_ty.id) {
 361        Some(&ty::atttce_resolved(ty)) => return ty,
 362        Some(&ty::atttce_unresolved) => {
 363          tcx.sess.span_fatal(ast_ty.span, "illegal recursive type; \
 364                                            insert an enum in the cycle, if this is desired");
 365        }
 366        None => { /* go on */ }
 367      }
 368  
 369      tcx.ast_ty_to_ty_cache.insert(ast_ty.id, ty::atttce_unresolved);
 370      let typ = match ast_ty.node {
 371        ast::ty_nil => ty::mk_nil(),
 372        ast::ty_bot => ty::mk_bot(),
 373        ast::ty_box(ref mt) => {
 374          mk_pointer(this, rscope, mt, ty::vstore_box,
 375                     |tmt| ty::mk_box(tcx, tmt))
 376        }
 377        ast::ty_uniq(ref mt) => {
 378          mk_pointer(this, rscope, mt, ty::vstore_uniq,
 379                     |tmt| ty::mk_uniq(tcx, tmt))
 380        }
 381        ast::ty_vec(ref mt) => {
 382          tcx.sess.span_err(ast_ty.span, "bare `[]` is not a type");
 383          // return /something/ so they can at least get more errors
 384          ty::mk_evec(tcx, ast_mt_to_mt(this, rscope, mt), ty::vstore_uniq)
 385        }
 386        ast::ty_ptr(ref mt) => {
 387          ty::mk_ptr(tcx, ast_mt_to_mt(this, rscope, mt))
 388        }
 389        ast::ty_rptr(ref region, ref mt) => {
 390          let r = ast_region_to_region(this, rscope, ast_ty.span, region);
 391          mk_pointer(this, rscope, mt, ty::vstore_slice(r),
 392                     |tmt| ty::mk_rptr(tcx, r, tmt))
 393        }
 394        ast::ty_tup(ref fields) => {
 395          let flds = fields.map(|t| ast_ty_to_ty(this, rscope, t));
 396          ty::mk_tup(tcx, flds)
 397        }
 398        ast::ty_bare_fn(ref bf) => {
 399            ty::mk_bare_fn(tcx, ty_of_bare_fn(this, rscope, bf.purity,
 400                                              bf.abis, &bf.lifetimes, &bf.decl))
 401        }
 402        ast::ty_closure(ref f) => {
 403          if f.sigil == ast::ManagedSigil {
 404              tcx.sess.span_err(ast_ty.span,
 405                                "managed closures are not supported");
 406          }
 407  
 408            let bounds = conv_builtin_bounds(this.tcx(), &f.bounds, match f.sigil {
 409                // Use corresponding trait store to figure out default bounds
 410                // if none were specified.
 411                ast::BorrowedSigil => ty::RegionTraitStore(ty::re_empty), // dummy region
 412                ast::OwnedSigil    => ty::UniqTraitStore,
 413                ast::ManagedSigil  => ty::BoxTraitStore,
 414            });
 415            let fn_decl = ty_of_closure(this,
 416                                        rscope,
 417                                        f.sigil,
 418                                        f.purity,
 419                                        f.onceness,
 420                                        bounds,
 421                                        &f.region,
 422                                        &f.decl,
 423                                        None,
 424                                        &f.lifetimes,
 425                                        ast_ty.span);
 426            ty::mk_closure(tcx, fn_decl)
 427        }
 428        ast::ty_path(ref path, ref bounds, id) => {
 429          let a_def = match tcx.def_map.find(&id) {
 430            None => tcx.sess.span_fatal(
 431                ast_ty.span, fmt!("unbound path %s",
 432                                  path_to_str(path, tcx.sess.intr()))),
 433            Some(&d) => d
 434          };
 435          // Kind bounds on path types are only supported for traits.
 436          match a_def {
 437              // But don't emit the error if the user meant to do a trait anyway.
 438              ast::DefTrait(*) => { },
 439              _ if bounds.is_some() =>
 440                  tcx.sess.span_err(ast_ty.span,
 441                      "kind bounds can only be used on trait types"),
 442              _ => { },
 443          }
 444          match a_def {
 445            ast::DefTrait(_) => {
 446                let path_str = path_to_str(path, tcx.sess.intr());
 447                tcx.sess.span_err(
 448                    ast_ty.span,
 449                    fmt!("reference to trait `%s` where a type is expected; \
 450                          try `@%s`, `~%s`, or `&%s`",
 451                         path_str, path_str, path_str, path_str));
 452                ty::mk_err()
 453            }
 454            ast::DefTy(did) | ast::DefStruct(did) => {
 455              ast_path_to_ty(this, rscope, did, path).ty
 456            }
 457            ast::DefPrimTy(nty) => {
 458              match nty {
 459                ast::ty_bool => {
 460                  check_path_args(tcx, path, NO_TPS | NO_REGIONS);
 461                  ty::mk_bool()
 462                }
 463                ast::ty_char => {
 464                  check_path_args(tcx, path, NO_TPS | NO_REGIONS);
 465                  ty::mk_char()
 466                }
 467                ast::ty_int(it) => {
 468                  check_path_args(tcx, path, NO_TPS | NO_REGIONS);
 469                  ty::mk_mach_int(it)
 470                }
 471                ast::ty_uint(uit) => {
 472                  check_path_args(tcx, path, NO_TPS | NO_REGIONS);
 473                  ty::mk_mach_uint(uit)
 474                }
 475                ast::ty_float(ft) => {
 476                  check_path_args(tcx, path, NO_TPS | NO_REGIONS);
 477                  ty::mk_mach_float(ft)
 478                }
 479                ast::ty_str => {
 480                  tcx.sess.span_err(ast_ty.span,
 481                                    "bare `str` is not a type");
 482                  // return /something/ so they can at least get more errors
 483                  ty::mk_estr(tcx, ty::vstore_uniq)
 484                }
 485              }
 486            }
 487            ast::DefTyParam(id, n) => {
 488              check_path_args(tcx, path, NO_TPS | NO_REGIONS);
 489              ty::mk_param(tcx, n, id)
 490            }
 491            ast::DefSelfTy(id) => {
 492              // n.b.: resolve guarantees that the this type only appears in a
 493              // trait, which we rely upon in various places when creating
 494              // substs
 495              check_path_args(tcx, path, NO_TPS | NO_REGIONS);
 496              let did = ast_util::local_def(id);
 497              ty::mk_self(tcx, did)
 498            }
 499            _ => {
 500              tcx.sess.span_fatal(ast_ty.span,
 501                                  fmt!("found value name used as a type: %?", a_def));
 502            }
 503          }
 504        }
 505        ast::ty_fixed_length_vec(ref a_mt, e) => {
 506          match const_eval::eval_const_expr_partial(&tcx, e) {
 507            Ok(ref r) => {
 508              match *r {
 509                const_eval::const_int(i) =>
 510                  ty::mk_evec(tcx, ast_mt_to_mt(this, rscope, a_mt),
 511                              ty::vstore_fixed(i as uint)),
 512                const_eval::const_uint(i) =>
 513                  ty::mk_evec(tcx, ast_mt_to_mt(this, rscope, a_mt),
 514                              ty::vstore_fixed(i as uint)),
 515                _ => {
 516                  tcx.sess.span_fatal(
 517                      ast_ty.span, "expected constant expr for vector length");
 518                }
 519              }
 520            }
 521            Err(ref r) => {
 522              tcx.sess.span_fatal(
 523                  ast_ty.span,
 524                  fmt!("expected constant expr for vector length: %s",
 525                       *r));
 526            }
 527          }
 528        }
 529        ast::ty_typeof(_e) => {
 530            tcx.sess.span_bug(ast_ty.span, "typeof is reserved but unimplemented");
 531        }
 532        ast::ty_infer => {
 533          // ty_infer should only appear as the type of arguments or return
 534          // values in a fn_expr, or as the type of local variables.  Both of
 535          // these cases are handled specially and should not descend into this
 536          // routine.
 537          this.tcx().sess.span_bug(
 538              ast_ty.span,
 539              "found `ty_infer` in unexpected place");
 540        }
 541        ast::ty_mac(_) => {
 542          tcx.sess.span_bug(ast_ty.span,
 543                            "found `ty_mac` in unexpected place");
 544        }
 545      };
 546  
 547      tcx.ast_ty_to_ty_cache.insert(ast_ty.id, ty::atttce_resolved(typ));
 548      return typ;
 549  }

 565  pub fn bound_lifetimes<AC:AstConv>(
 566      this: &AC,
 567      ast_lifetimes: &OptVec<ast::Lifetime>) -> OptVec<ast::Ident>
 568  {
 569      /*!
 570       *
 571       * Converts a list of lifetimes into a list of bound identifier
 572       * names.  Does not permit special names like 'static or 'this to
 573       * be bound.  Note that this function is for use in closures,
 574       * methods, and fn definitions.  It is legal to bind 'this in a
 575       * type.  Eventually this distinction should go away and the same
 576       * rules should apply everywhere ('this would not be a special name
 577       * at that point).
 578       */
 579  
 580      let special_idents = [special_idents::statik, special_idents::self_];
 581      let mut bound_lifetime_names = opt_vec::Empty;
 582      ast_lifetimes.map_to_vec(|ast_lifetime| {
 583          if special_idents.iter().any(|&i| i == ast_lifetime.ident) {
 584              this.tcx().sess.span_err(
 585                  ast_lifetime.span,
 586                  fmt!("illegal lifetime parameter name: `%s`",
 587                       lifetime_to_str(ast_lifetime, this.tcx().sess.intr())));
 588          } else {
 589              bound_lifetime_names.push(ast_lifetime.ident);
 590          }
 591      });
 592      bound_lifetime_names
 593  }

 631  fn ty_of_method_or_bare_fn<AC:AstConv,RS:RegionScope + Clone + 'static>(
 632      this: &AC,
 633      rscope: &RS,
 634      purity: ast::purity,
 635      abi: AbiSet,
 636      lifetimes: &OptVec<ast::Lifetime>,
 637      opt_self_info: Option<&SelfInfo>,
 638      decl: &ast::fn_decl) -> (Option<Option<ty::t>>, ty::BareFnTy)
 639  {
 640      debug!("ty_of_bare_fn");
 641  
 642      // new region names that appear inside of the fn decl are bound to
 643      // that function type
 644      let bound_lifetime_names = bound_lifetimes(this, lifetimes);
 645      let rb =
 646          in_binding_rscope(rscope,
 647                            RegionParamNames(bound_lifetime_names.clone()));
 648  
 649      let opt_transformed_self_ty = do opt_self_info.map_move |self_info| {
 650          transform_self_ty(this, &rb, self_info)
 651      };

 653      let input_tys = decl.inputs.map(|a| ty_of_arg(this, &rb, a, None));
 654  
 655      let output_ty = match decl.output.node {
 656          ast::ty_infer => this.ty_infer(decl.output.span),
 657          _ => ast_ty_to_ty(this, &rb, &decl.output)
 658      };
 659  
 660      return (opt_transformed_self_ty,
 661              ty::BareFnTy {
 662                  purity: purity,
 663                  abis: abi,
 664                  sig: ty::FnSig {bound_lifetime_names: bound_lifetime_names,
 665                                  inputs: input_tys,
 666                                  output: output_ty}
 667              });

 671          rscope: &RS,
 672          self_info: &SelfInfo) -> Option<ty::t>
 673      {
 674          match self_info.explicit_self.node {

 676              ast::sty_value => {
 677                  Some(self_info.untransformed_self_ty)
 678              }
 679              ast::sty_region(ref lifetime, mutability) => {
 680                  let region =
 681                      ast_region_to_region(this, rscope,

 685                                   ty::mt {ty: self_info.untransformed_self_ty,
 686                                           mutbl: mutability}))
 687              }

 689                  Some(ty::mk_box(this.tcx(),
 690                                  ty::mt {ty: self_info.untransformed_self_ty,

 702  pub fn ty_of_closure<AC:AstConv,RS:RegionScope + Clone + 'static>(
 703      this: &AC,
 704      rscope: &RS,
 705      sigil: ast::Sigil,
 706      purity: ast::purity,
 707      onceness: ast::Onceness,
 708      bounds: ty::BuiltinBounds,
 709      opt_lifetime: &Option<ast::Lifetime>,
 710      decl: &ast::fn_decl,
 711      expected_sig: Option<ty::FnSig>,
 712      lifetimes: &OptVec<ast::Lifetime>,
 713      span: Span)
 714      -> ty::ClosureTy
 715  {
 716      // The caller should not both provide explicit bound lifetime
 717      // names and expected types.  Either we infer the bound lifetime
 718      // names or they are provided, but not both.
 719      assert!(lifetimes.is_empty() || expected_sig.is_none());
 720  
 721      debug!("ty_of_fn_decl");
 722      let _i = indenter();
 723  
 724      // resolve the function bound region in the original region
 725      // scope `rscope`, not the scope of the function parameters
 726      let bound_region = match opt_lifetime {
 727          &Some(_) => {
 728              ast_region_to_region(this, rscope, span, opt_lifetime)
 729          }
 730          &None => {
 731              match sigil {
 732                  ast::OwnedSigil | ast::ManagedSigil => {
 733                      // @fn(), ~fn() default to static as the bound
 734                      // on their upvars:
 735                      ty::re_static
 736                  }
 737                  ast::BorrowedSigil => {
 738                      // &fn() defaults as normal for an omitted lifetime:
 739                      ast_region_to_region(this, rscope, span, opt_lifetime)
 740                  }
 741              }
 742          }
 743      };
 744  
 745      // new region names that appear inside of the fn decl are bound to
 746      // that function type
 747      let bound_lifetime_names = bound_lifetimes(this, lifetimes);
 748      let rb =
 749          in_binding_rscope(rscope,
 750                            RegionParamNames(bound_lifetime_names.clone()));
 751  
 752      let input_tys = do decl.inputs.iter().enumerate().map |(i, a)| {
 753          let expected_arg_ty = do expected_sig.and_then_ref |e| {
 754              // no guarantee that the correct number of expected args
 755              // were supplied
 756              if i < e.inputs.len() {Some(e.inputs[i])} else {None}
 757          };
 758          ty_of_arg(this, &rb, a, expected_arg_ty)
 759      }.collect();
 760  
 761      let expected_ret_ty = expected_sig.map(|e| e.output);
 762      let output_ty = match decl.output.node {
 763          ast::ty_infer if expected_ret_ty.is_some() => expected_ret_ty.unwrap(),
 764          ast::ty_infer => this.ty_infer(decl.output.span),
 765          _ => ast_ty_to_ty(this, &rb, &decl.output)
 766      };
 767  
 768      ty::ClosureTy {
 769          purity: purity,
 770          sigil: sigil,
 771          onceness: onceness,
 772          region: bound_region,
 773          bounds: bounds,
 774          sig: ty::FnSig {bound_lifetime_names: bound_lifetime_names,
 775                          inputs: input_tys,
 776                          output: output_ty}
 777      }
 778  }

 780  fn conv_builtin_bounds(tcx: ty::ctxt, ast_bounds: &Option<OptVec<ast::TyParamBound>>,
 781                         store: ty::TraitStore)
 782                         -> ty::BuiltinBounds {
 783      //! Converts a list of bounds from the AST into a `BuiltinBounds`
 784      //! struct. Reports an error if any of the bounds that appear
 785      //! in the AST refer to general traits and not the built-in traits
 786      //! like `Send`. Used to translate the bounds that
 787      //! appear in closure and trait types, where only builtin bounds are
 788      //! legal.
 789      //! If no bounds were specified, we choose a "default" bound based on
 790      //! the allocation type of the fn/trait, as per issue #7264. The user can
 791      //! override this with an empty bounds list, e.g. "~fn:()" or "~Trait:".
 792  
 793      match (ast_bounds, store) {
 794          (&Some(ref bound_vec), _) => {
 795              let mut builtin_bounds = ty::EmptyBuiltinBounds();
 796              for ast_bound in bound_vec.iter() {
 797                  match *ast_bound {
 798                      ast::TraitTyParamBound(ref b) => {
 799                          match lookup_def_tcx(tcx, b.path.span, b.ref_id) {
 800                              ast::DefTrait(trait_did) => {
 801                                  if ty::try_add_builtin_trait(tcx, trait_did,
 802                                                               &mut builtin_bounds) {
 803                                      loop; // success
 804                                  }
 805                              }
 806                              _ => { }
 807                          }
 808                          tcx.sess.span_fatal(
 809                              b.path.span,
 810                              fmt!("only the builtin traits can be used \
 811                                    as closure or object bounds"));
 812                      }
 813                      ast::RegionTyParamBound => {
 814                          builtin_bounds.add(ty::BoundStatic);
 815                      }
 816                  }
 817              }
 818              builtin_bounds
 819          },
 820          // ~Trait is sugar for ~Trait:Send.
 821          (&None, ty::UniqTraitStore) => {
 822              let mut set = ty::EmptyBuiltinBounds(); set.add(ty::BoundSend); set
 823          }
 824          // @Trait is sugar for @Trait:'static.
 825          // &'static Trait is sugar for &'static Trait:'static.
 826          (&None, ty::BoxTraitStore) |
 827          (&None, ty::RegionTraitStore(ty::re_static)) => {

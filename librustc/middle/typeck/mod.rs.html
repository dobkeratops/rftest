  11  /*
  12  
  13  typeck.rs, an introduction
  14  
  15  The type checker is responsible for:
  16  
  17  1. Determining the type of each expression
  18  2. Resolving methods and traits
  19  3. Guaranteeing that most type rules are met ("most?", you say, "why most?"
  20     Well, dear reader, read on)
  21  
  22  The main entry point is `check_crate()`.  Type checking operates in two major
  23  phases: collect and check.  The collect phase passes over all items and
  24  determines their type, without examining their "innards".  The check phase
  25  then checks function bodies and so forth.
  26  
  27  Within the check phase, we check each function body one at a time (bodies of
  28  function expressions are checked as part of the containing function).
  29  Inference is used to supply types wherever they are unknown. The actual
  30  checking of a function itself has several phases (check, regionck, writeback),
  31  as discussed in the documentation for the `check` module.
  32  
  33  The type checker is defined into various submodules which are documented
  34  independently:
  35  
  36  - astconv: converts the AST representation of types
  37    into the `ty` representation
  38  
  39  - collect: computes the types of each top-level item and enters them into
  40    the `cx.tcache` table for later use
  41  
  42  - check: walks over function bodies and type checks them, inferring types for
  43    local variables, type parameters, etc as necessary.
  44  
  45  - infer: finds the types to use for each type variable such that
  46    all subtyping and assignment constraints are met.  In essence, the check
  47    module specifies the constraints, and the infer module solves them.
  48  
  49  */

 157  #[deriving(Clone)]
 158  pub enum vtable_origin {
 159      /*
 160        Statically known vtable. def_id gives the class or impl item
 161        from whence comes the vtable, and tys are the type substs.
 162        vtable_res is the vtable itself
 163       */
 164      vtable_static(ast::DefId, ~[ty::t], vtable_res),
 165  
 166      /*
 167        Dynamic vtable, comes from a parameter that has a bound on it:
 168        fn foo<T:quux,baz,bar>(a: T) -- a's vtable would have a
 169        vtable_param origin
 170  
 171        The first argument is the param index (identifying T in the example),
 172        and the second is the bound number (identifying baz)
 173       */
 174      vtable_param(param_index, uint),
 175  }

 228  pub fn write_ty_to_tcx(tcx: ty::ctxt, node_id: ast::NodeId, ty: ty::t) {
 229      debug!("write_ty_to_tcx(%d, %s)", node_id, ppaux::ty_to_str(tcx, ty));

 233  pub fn write_substs_to_tcx(tcx: ty::ctxt,
 234                             node_id: ast::NodeId,
 235                             substs: ~[ty::t]) {
 236      if substs.len() > 0u {
 237          debug!("write_substs_to_tcx(%d, %?)", node_id,
 238                 substs.map(|t| ppaux::ty_to_str(tcx, *t)));
 239          assert!(substs.iter().all(|t| !ty::type_needs_infer(*t)));
 240          tcx.node_type_substs.insert(node_id, substs);
 241      }
 242  }

 274  pub fn require_same_types(
 275      tcx: ty::ctxt,
 276      maybe_infcx: Option<@mut infer::InferCtxt>,
 277      t1_is_expected: bool,
 278      span: Span,
 279      t1: ty::t,
 280      t2: ty::t,
 281      msg: &fn() -> ~str) -> bool {
 282  
 283      let l_tcx;
 284      let l_infcx;
 285      match maybe_infcx {
 286        None => {
 287          l_tcx = tcx;
 288          l_infcx = infer::new_infer_ctxt(tcx);
 289        }
 290        Some(i) => {
 291          l_tcx = i.tcx;
 292          l_infcx = i;
 293        }
 294      }
 295  
 296      match infer::mk_eqty(l_infcx, t1_is_expected, infer::Misc(span), t1, t2) {
 297          result::Ok(()) => true,
 298          result::Err(ref terr) => {
 299              l_tcx.sess.span_err(span, msg() + ": " +
 300                                  ty::type_err_to_str(l_tcx, terr));
 301              ty::note_and_explain_type_err(l_tcx, terr);
 302              false
 303          }
 304      }
 305  }

 307  // a list of mapping from in-scope-region-names ("isr") to the
 308  // corresponding ty::Region
 309  pub type isr_alist = @List<(ty::bound_region, ty::Region)>;
 310  
 311  trait get_and_find_region {
 312      fn get(&self, br: ty::bound_region) -> ty::Region;
 313      fn find(&self, br: ty::bound_region) -> Option<ty::Region>;
 314  }
 315  
 316  impl get_and_find_region for isr_alist {
 317      fn get(&self, br: ty::bound_region) -> ty::Region {
 318          self.find(br).unwrap()
 319      }
 320  
 321      fn find(&self, br: ty::bound_region) -> Option<ty::Region> {
 322          let mut ret = None;
 323          do list::each(*self) |isr| {
 324              let (isr_br, isr_r) = *isr;
 325              if isr_br == br { ret = Some(isr_r); false } else { true }
 326          };
 327          ret
 328      }
 329  }

 331  fn check_main_fn_ty(ccx: &CrateCtxt,
 332                      main_id: ast::NodeId,
 333                      main_span: Span) {
 334      let tcx = ccx.tcx;
 335      let main_t = ty::node_id_to_type(tcx, main_id);
 336      match ty::get(main_t).sty {
 337          ty::ty_bare_fn(*) => {
 338              match tcx.items.find(&main_id) {
 339                  Some(&ast_map::node_item(it,_)) => {
 340                      match it.node {
 341                          ast::item_fn(_, _, _, ref ps, _)
 342                          if ps.is_parameterized() => {
 343                              tcx.sess.span_err(
 344                                  main_span,
 345                                  "main function is not allowed to have type parameters");
 346                              return;
 347                          }
 348                          _ => ()
 349                      }
 350                  }
 351                  _ => ()
 352              }
 353              let se_ty = ty::mk_bare_fn(tcx, ty::BareFnTy {
 354                  purity: ast::impure_fn,
 355                  abis: abi::AbiSet::Rust(),
 356                  sig: ty::FnSig {
 357                      bound_lifetime_names: opt_vec::Empty,
 358                      inputs: ~[],
 359                      output: ty::mk_nil()
 360                  }
 361              });
 362  
 363              require_same_types(tcx, None, false, main_span, main_t, se_ty,
 364                  || fmt!("main function expects type: `%s`",
 365                          ppaux::ty_to_str(ccx.tcx, se_ty)));
 366          }
 367          _ => {
 368              tcx.sess.span_bug(main_span,
 369                                fmt!("main has a non-function type: found `%s`",
 370                                     ppaux::ty_to_str(tcx, main_t)));
 371          }
 372      }
 373  }

 375  fn check_start_fn_ty(ccx: &CrateCtxt,
 376                       start_id: ast::NodeId,
 377                       start_span: Span) {
 378      let tcx = ccx.tcx;
 379      let start_t = ty::node_id_to_type(tcx, start_id);
 380      match ty::get(start_t).sty {
 381          ty::ty_bare_fn(_) => {
 382              match tcx.items.find(&start_id) {
 383                  Some(&ast_map::node_item(it,_)) => {
 384                      match it.node {
 385                          ast::item_fn(_,_,_,ref ps,_)
 386                          if ps.is_parameterized() => {
 387                              tcx.sess.span_err(
 388                                  start_span,
 389                                  "start function is not allowed to have type parameters");
 390                              return;
 391                          }
 392                          _ => ()
 393                      }
 394                  }
 395                  _ => ()
 396              }
 397  
 398              let se_ty = ty::mk_bare_fn(tcx, ty::BareFnTy {
 399                  purity: ast::impure_fn,
 400                  abis: abi::AbiSet::Rust(),
 401                  sig: ty::FnSig {
 402                      bound_lifetime_names: opt_vec::Empty,
 403                      inputs: ~[
 404                          ty::mk_int(),
 405                          ty::mk_imm_ptr(tcx, ty::mk_imm_ptr(tcx, ty::mk_u8()))
 406                      ],
 407                      output: ty::mk_int()
 408                  }
 409              });
 410  
 411              require_same_types(tcx, None, false, start_span, start_t, se_ty,
 412                  || fmt!("start function expects type: `%s`", ppaux::ty_to_str(ccx.tcx, se_ty)));
 413  
 414          }
 415          _ => {
 416              tcx.sess.span_bug(start_span,
 417                                fmt!("start has a non-function type: found `%s`",
 418                                     ppaux::ty_to_str(tcx, start_t)));
 419          }
 420      }
 421  }

 423  fn check_for_entry_fn(ccx: &CrateCtxt) {
 424      let tcx = ccx.tcx;
 425      if !*tcx.sess.building_library {
 426          match *tcx.sess.entry_fn {
 427            Some((id, sp)) => match *tcx.sess.entry_type {
 428                Some(session::EntryMain) => check_main_fn_ty(ccx, id, sp),
 429                Some(session::EntryStart) => check_start_fn_ty(ccx, id, sp),
 430                Some(session::EntryNone) => {}
 431                None => tcx.sess.bug("entry function without a type")
 432            },
 433            None => {}
 434          }
 435      }
 436  }

 438  pub fn check_crate(tcx: ty::ctxt,
 439                     trait_map: resolve::TraitMap,
 440                     crate: &ast::Crate)
 441                  -> (method_map, vtable_map) {
 442      let time_passes = tcx.sess.time_passes();
 443      let ccx = @mut CrateCtxt {
 444          trait_map: trait_map,
 445          method_map: @mut HashMap::new(),
 446          vtable_map: @mut HashMap::new(),
 447          tcx: tcx
 448      };
 449  
 450      time(time_passes, ~"type collecting", ||

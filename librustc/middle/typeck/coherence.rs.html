  12  //
  13  // The job of the coherence phase of typechecking is to ensure that each trait
  14  // has at most one implementation for each type. Then we build a mapping from
  15  // each trait in the system to its implementations.

  59  fn get_base_type(inference_context: &InferCtxt,
  60                   span: Span,
  61                   original_type: t)
  62                   -> Option<t> {
  63      let resolved_type;
  64      match resolve_type(inference_context,
  65                         original_type,
  66                         resolve_ivar) {
  67          Ok(resulting_type) if !type_is_ty_var(resulting_type) => {
  68              resolved_type = resulting_type;
  69          }
  70          _ => {
  71              inference_context.tcx.sess.span_fatal(span,
  72                                                    "the type of this value must be known in order \
  73                                                     to determine the base type");
  74          }
  75      }
  76  
  77      match get(resolved_type).sty {
  78          ty_enum(..) | ty_trait(..) | ty_struct(..) => {
  79              debug!("(getting base type) found base type");
  80              Some(resolved_type)
  81          }
  82  
  83          ty_nil | ty_bot | ty_bool | ty_char | ty_int(..) | ty_uint(..) | ty_float(..) |
  84          ty_str(..) | ty_vec(..) | ty_bare_fn(..) | ty_closure(..) | ty_tup(..) |
  85          ty_infer(..) | ty_param(..) | ty_self(..) |
  86          ty_unboxed_vec(..) | ty_err | ty_box(_) |
  87          ty_uniq(_) | ty_ptr(_) | ty_rptr(_, _) => {
  88              debug!("(getting base type) no base type; found {:?}",
  89                     get(original_type).sty);
  90              None
  91          }
  92      }
  93  }

  95  fn type_is_defined_in_local_crate(original_type: t) -> bool {
  96      /*!
  97       *
  98       * For coherence, when we have `impl Trait for Type`, we need to
  99       * guarantee that `Type` is "local" to the
 100       * crate.  For our purposes, this means that it must contain
 101       * some nominal type defined in this crate.
 102       */
 103  
 104      let mut found_nominal = false;
 105      ty::walk_ty(original_type, |t| {
 106          match get(t).sty {
 107              ty_enum(def_id, _) |
 108              ty_trait(~ty::TyTrait { def_id, .. }) |
 109              ty_struct(def_id, _) => {
 110                  if def_id.krate == ast::LOCAL_CRATE {
 111                      found_nominal = true;
 112                  }
 113              }
 114  
 115              _ => { }
 116          }
 117      });
 118      return found_nominal;
 119  }

 121  // Returns the def ID of the base type, if there is one.
 122  fn get_base_type_def_id(inference_context: &InferCtxt,
 123                          span: Span,
 124                          original_type: t)
 125                          -> Option<DefId> {
 126      match get_base_type(inference_context, span, original_type) {
 127          None => {
 128              return None;
 129          }
 130          Some(base_type) => {
 131              match get(base_type).sty {
 132                  ty_enum(def_id, _) |
 133                  ty_struct(def_id, _) |
 134                  ty_trait(~ty::TyTrait { def_id, .. }) => {
 135                      return Some(def_id);
 136                  }
 137                  _ => {
 138                      fail!("get_base_type() returned a type that wasn't an \
 139                             enum, struct, or trait");
 140                  }
 141              }
 142          }
 143      }

 146  struct CoherenceChecker<'a> {
 147      crate_context: &'a CrateCtxt<'a>,
 148      inference_context: InferCtxt<'a>,
 149  }

 151  struct CoherenceCheckVisitor<'a> {
 152      cc: &'a CoherenceChecker<'a>
 153  }
 154  
 155  impl<'a> visit::Visitor<()> for CoherenceCheckVisitor<'a> {
 156      fn visit_item(&mut self, item: &Item, _: ()) {
 157  
 158          //debug!("(checking coherence) item '{}'", token::get_ident(item.ident));
 159  
 160          match item.node {
 161              ItemImpl(_, ref opt_trait, _, _) => {
 162                  match opt_trait.clone() {
 163                      Some(opt_trait) => {
 164                          self.cc.check_implementation(item, [opt_trait]);
 165                      }
 166                      None => self.cc.check_implementation(item, [])
 167                  }
 168              }
 169              _ => {
 170                  // Nothing to do.
 171              }
 172          };
 173  
 174          visit::walk_item(self, item, ());
 175      }
 176  }

 178  struct PrivilegedScopeVisitor<'a> { cc: &'a CoherenceChecker<'a> }
 179  
 180  impl<'a> visit::Visitor<()> for PrivilegedScopeVisitor<'a> {
 181      fn visit_item(&mut self, item: &Item, _: ()) {
 182  
 183          match item.node {
 184              ItemMod(ref module_) => {
 185                  // Then visit the module items.
 186                  visit::walk_mod(self, module_, ());
 187              }
 188              ItemImpl(_, None, ast_ty, _) => {
 189                  if !self.cc.ast_type_is_defined_in_local_crate(ast_ty) {
 190                      // This is an error.
 191                      let session = &self.cc.crate_context.tcx.sess;
 192                      session.span_err(item.span,
 193                                       "cannot associate methods with a type outside the \
 194                                       crate the type is defined in; define and implement \
 195                                       a trait or new type instead");
 196                  }
 197              }
 198              ItemImpl(_, Some(ref trait_ref), _, _) => {
 199                  // `for_ty` is `Type` in `impl Trait for Type`
 200                  let for_ty =
 201                      ty::node_id_to_type(self.cc.crate_context.tcx,
 202                                          item.id);
 203                  if !type_is_defined_in_local_crate(for_ty) {
 204                      // This implementation is not in scope of its base
 205                      // type. This still might be OK if the trait is
 206                      // defined in the same crate.
 207  
 208                      let trait_def_id =
 209                          self.cc.trait_ref_to_trait_def_id(trait_ref);
 210  
 211                      if trait_def_id.krate != LOCAL_CRATE {
 212                          let session = &self.cc.crate_context.tcx.sess;
 213                          session.span_err(item.span,
 214                                  "cannot provide an extension implementation \
 215                                  where both trait and type are not defined in this crate");
 216                      }
 217                  }
 218  
 219                  visit::walk_item(self, item, ());
 220              }
 221              _ => {
 222                  visit::walk_item(self, item, ());
 223              }
 224          }
 225      }
 226  }

 228  impl<'a> CoherenceChecker<'a> {
 229      fn check(&self, krate: &Crate) {
 230          // Check implementations and traits. This populates the tables
 231          // containing the inherent methods and extension methods. It also
 232          // builds up the trait inheritance table.
 233          let mut visitor = CoherenceCheckVisitor { cc: self };
 234          visit::walk_crate(&mut visitor, krate, ());
 235  
 236          // Check that there are no overlapping trait instances
 237          self.check_implementation_coherence();
 238  
 239          // Check whether traits with base types are in privileged scopes.
 240          self.check_privileged_scopes(krate);
 241  
 242          // Bring in external crates. It's fine for this to happen after the
 243          // coherence checks, because we ensure by construction that no errors
 244          // can happen at link time.
 245          self.add_external_crates();
 246  
 247          // Populate the table of destructors. It might seem a bit strange to
 248          // do this here, but it's actually the most convenient place, since
 249          // the coherence tables contain the trait -> type mappings.
 250          self.populate_destructor_table();
 251      }
 252  
 253      fn check_implementation(&self, item: &Item,
 254                              associated_traits: &[TraitRef]) {
 255          let tcx = self.crate_context.tcx;
 256          let self_type = ty::lookup_item_type(tcx, local_def(item.id));
 257  
 258          // If there are no traits, then this implementation must have a
 259          // base type.
 260  
 261          if associated_traits.len() == 0 {
 262              debug!("(checking implementation) no associated traits for item '{}'",
 263                     token::get_ident(item.ident));
 264  
 265              match get_base_type_def_id(&self.inference_context,
 266                                         item.span,
 267                                         self_type.ty) {
 268                  None => {
 269                      let session = &self.crate_context.tcx.sess;
 270                      session.span_err(item.span,
 271                                       "no base type found for inherent implementation; \
 272                                        implement a trait or new type instead");
 273                  }
 274                  Some(_) => {
 275                      // Nothing to do.
 276                  }
 277              }
 278          }
 279  
 280          let implementation = self.create_impl_from_item(item);
 281  
 282          for associated_trait in associated_traits.iter() {
 283              let trait_ref = ty::node_id_to_trait_ref(
 284                  self.crate_context.tcx, associated_trait.ref_id);
 285              debug!("(checking implementation) adding impl for trait '{}', item '{}'",
 286                     trait_ref.repr(self.crate_context.tcx),
 287                     token::get_ident(item.ident));
 288  
 289              self.add_trait_impl(trait_ref.def_id, implementation);
 290          }
 291  
 292          // Add the implementation to the mapping from implementation to base
 293          // type def ID, if there is a base type for this implementation and
 294          // the implementation does not have any associated traits.
 295          match get_base_type_def_id(&self.inference_context,
 296                                     item.span,
 297                                     self_type.ty) {
 298              None => {
 299                  // Nothing to do.
 300              }
 301              Some(base_type_def_id) => {
 302                  // FIXME: Gather up default methods?
 303                  if associated_traits.len() == 0 {
 304                      self.add_inherent_impl(base_type_def_id, implementation);
 305                  }
 306              }
 307          }
 308  
 309          tcx.impls.borrow_mut().insert(implementation.did, implementation);
 310      }
 311  
 312      // Creates default method IDs and performs type substitutions for an impl
 313      // and trait pair. Then, for each provided method in the trait, inserts a
 314      // `ProvidedMethodInfo` instance into the `provided_method_sources` map.
 315      fn instantiate_default_methods(&self, impl_id: ast::DefId,
 316                                     trait_ref: &ty::TraitRef,
 317                                     all_methods: &mut Vec<@Method> ) {
 318          let tcx = self.crate_context.tcx;
 319          debug!("instantiate_default_methods(impl_id={:?}, trait_ref={})",
 320                 impl_id, trait_ref.repr(tcx));
 321  
 322          let impl_poly_type = ty::lookup_item_type(tcx, impl_id);
 323  
 324          let prov = ty::provided_trait_methods(tcx, trait_ref.def_id);
 325          for trait_method in prov.iter() {
 326              // Synthesize an ID.
 327              let new_id = tcx.sess.next_node_id();
 328              let new_did = local_def(new_id);
 329  
 330              debug!("new_did={:?} trait_method={}", new_did, trait_method.repr(tcx));
 331  
 332              // Create substitutions for the various trait parameters.
 333              let new_method_ty =
 334                  @subst_receiver_types_in_method_ty(
 335                      tcx,
 336                      impl_id,
 337                      trait_ref,
 338                      new_did,
 339                      *trait_method,
 340                      Some(trait_method.def_id));
 341  
 342              debug!("new_method_ty={}", new_method_ty.repr(tcx));
 343              all_methods.push(new_method_ty);
 344  
 345              // construct the polytype for the method based on the method_ty
 346              let new_generics = ty::Generics {
 347                  type_param_defs:
 348                      Rc::new(vec::append(
 349                          Vec::from_slice(impl_poly_type.generics
 350                                                        .type_param_defs()),
 351                              new_method_ty.generics.type_param_defs())),
 352                  region_param_defs:
 353                      impl_poly_type.generics.region_param_defs.clone()
 354              };
 355              let new_polytype = ty::ty_param_bounds_and_ty {
 356                  generics: new_generics,
 357                  ty: ty::mk_bare_fn(tcx, new_method_ty.fty.clone())
 358              };
 359              debug!("new_polytype={}", new_polytype.repr(tcx));
 360  
 361              tcx.tcache.borrow_mut().insert(new_did, new_polytype);
 362              tcx.methods.borrow_mut().insert(new_did, new_method_ty);
 363  
 364              // Pair the new synthesized ID up with the
 365              // ID of the method.
 366              self.crate_context.tcx.provided_method_sources.borrow_mut()
 367                  .insert(new_did, trait_method.def_id);
 368          }
 369      }

 371      fn add_inherent_impl(&self, base_def_id: DefId,
 372                           implementation: @Impl) {
 373          let tcx = self.crate_context.tcx;
 374          let implementation_list;
 375          let mut inherent_impls = tcx.inherent_impls.borrow_mut();
 376          match inherent_impls.find(&base_def_id) {
 377              None => {
 378                  implementation_list = @RefCell::new(Vec::new());
 379                  inherent_impls.insert(base_def_id, implementation_list);
 380              }
 381              Some(&existing_implementation_list) => {
 382                  implementation_list = existing_implementation_list;
 383              }
 384          }
 385  
 386          implementation_list.borrow_mut().push(implementation);
 387      }

 407      fn check_implementation_coherence(&self) {
 408          for &trait_id in self.crate_context.tcx.trait_impls.borrow().keys() {
 409              self.check_implementation_coherence_of(trait_id);
 410          }
 411      }

 413      fn check_implementation_coherence_of(&self, trait_def_id: DefId) {
 414          // Unify pairs of polytypes.
 415          self.iter_impls_of_trait_local(trait_def_id, |a| {
 416              let implementation_a = a;
 417              let polytype_a =
 418                  self.get_self_type_for_implementation(implementation_a);
 419  
 420              // "We have an impl of trait <trait_def_id> for type <polytype_a>,
 421              // and that impl is <implementation_a>"
 422              self.iter_impls_of_trait(trait_def_id, |b| {
 423                  let implementation_b = b;
 424  
 425                  // An impl is coherent with itself
 426                  if a.did != b.did {
 427                      let polytype_b = self.get_self_type_for_implementation(
 428                              implementation_b);
 429  
 430                      if self.polytypes_unify(polytype_a.clone(), polytype_b) {

 735  pub fn make_substs_for_receiver_types(tcx: &ty::ctxt,
 736                                        impl_id: ast::DefId,
 737                                        trait_ref: &ty::TraitRef,
 738                                        method: &ty::Method)
 739                                        -> ty::substs {
 740      /*!
 741       * Substitutes the values for the receiver's type parameters
 742       * that are found in method, leaving the method's type parameters
 743       * intact.  This is in fact a mildly complex operation,
 744       * largely because of the hokey way that we concatenate the
 745       * receiver and method generics.
 746       */
 747

 749      let num_impl_type_parameters = {
 750          let impl_polytype = ty::lookup_item_type(tcx, impl_id);
 751          impl_polytype.generics.type_param_defs().len()
 752      };

 754      // determine how many type parameters appear on the trait
 755      let num_trait_type_parameters = trait_ref.substs.tps.len();
 756  
 757      // the current method type has the type parameters from the trait + method
 758      let num_method_type_parameters =
 759          num_trait_type_parameters + method.generics.type_param_defs().len();
 760  
 761      // the new method type will have the type parameters from the impl + method
 762      let combined_tps = Vec::from_fn(num_method_type_parameters, |i| {
 763          if i < num_trait_type_parameters {
 764              // replace type parameters that come from trait with new value
 765              *trait_ref.substs.tps.get(i)
 766          } else {
 767              // replace type parameters that belong to method with another
 768              // type parameter, this time with the index adjusted
 769              let method_index = i - num_trait_type_parameters;
 770              let type_param_def = &method.generics.type_param_defs()[method_index];
 771              let new_index = num_impl_type_parameters + method_index;
 772              ty::mk_param(tcx, new_index, type_param_def.def_id)
 773          }
 774      });
 775  
 776      return ty::substs {
 777          regions: trait_ref.substs.regions.clone(),
 778          self_ty: trait_ref.substs.self_ty,
 779          tps: combined_tps
 780      };
 781  }

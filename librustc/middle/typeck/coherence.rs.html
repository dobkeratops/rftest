  12  //
  13  // The job of the coherence phase of typechecking is to ensure that each trait
  14  // has at most one implementation for each type. Then we build a mapping from
  15  // each trait in the system to its implementations.

  65  pub fn get_base_type(inference_context: @mut InferCtxt,
  66                       span: span,
  67                       original_type: t)
  68                    -> Option<t> {
  69      let resolved_type;
  70      match resolve_type(inference_context,
  71                         original_type,
  72                         resolve_ivar) {
  73          Ok(resulting_type) if !type_is_ty_var(resulting_type) => {
  74              resolved_type = resulting_type;
  75          }
  76          _ => {
  77              inference_context.tcx.sess.span_fatal(span,
  78                                                    "the type of this value must be known in order \
  79                                                     to determine the base type");
  80          }
  81      }
  82  
  83      match get(resolved_type).sty {
  84          ty_enum(*) | ty_trait(*) | ty_struct(*) => {
  85              debug!("(getting base type) found base type");
  86              Some(resolved_type)
  87          }
  88  
  89          ty_nil | ty_bot | ty_bool | ty_int(*) | ty_uint(*) | ty_float(*) |
  90          ty_estr(*) | ty_evec(*) | ty_bare_fn(*) | ty_closure(*) | ty_tup(*) |
  91          ty_infer(*) | ty_param(*) | ty_self(*) | ty_type | ty_opaque_box |
  92          ty_opaque_closure_ptr(*) | ty_unboxed_vec(*) | ty_err | ty_box(_) |
  93          ty_uniq(_) | ty_ptr(_) | ty_rptr(_, _) => {
  94              debug!("(getting base type) no base type; found %?",
  95                     get(original_type).sty);
  96              None
  97          }
  98      }
  99  }

 101  pub fn type_is_defined_in_local_crate(original_type: t) -> bool {
 102      /*!
 103       *
 104       * For coherence, when we have `impl Trait for Type`, we need to
 105       * guarantee that `Type` is "local" to the
 106       * crate.  For our purposes, this means that it must contain
 107       * some nominal type defined in this crate.
 108       */
 109  
 110      let mut found_nominal = false;
 111      do ty::walk_ty(original_type) |t| {
 112          match get(t).sty {
 113              ty_enum(def_id, _) |
 114              ty_trait(def_id, _, _, _, _) |
 115              ty_struct(def_id, _) => {
 116                  if def_id.crate == ast::LOCAL_CRATE {
 117                      found_nominal = true;
 118                  }
 119              }
 120  
 121              _ => { }
 122          }
 123      }
 124      return found_nominal;
 125  }

 127  // Returns the def ID of the base type, if there is one.
 128  pub fn get_base_type_def_id(inference_context: @mut InferCtxt,
 129                              span: span,
 130                              original_type: t)
 131                           -> Option<def_id> {
 132      match get_base_type(inference_context, span, original_type) {
 133          None => {
 134              return None;
 135          }
 136          Some(base_type) => {
 137              match get(base_type).sty {
 138                  ty_enum(def_id, _) |
 139                  ty_struct(def_id, _) |
 140                  ty_trait(def_id, _, _, _, _) => {
 141                      return Some(def_id);
 142                  }
 143                  _ => {
 144                      fail!("get_base_type() returned a type that wasn't an \
 145                             enum, struct, or trait");
 146                  }
 147              }
 148          }
 149      }

 163      inference_context: @mut InferCtxt,
 164  
 165      // A mapping from implementations to the corresponding base type

 172      pub fn check_coherence(self, crate: &Crate) {
 173          // Check implementations and traits. This populates the tables
 174          // containing the inherent methods and extension methods. It also
 175          // builds up the trait inheritance table.
 176          visit_crate(crate, ((), mk_simple_visitor(@SimpleVisitor {
 177              visit_item: |item| {
 178  //                debug!("(checking coherence) item '%s'",
 179  //                       self.crate_context.tcx.sess.str_of(item.ident));
 180  
 181                  match item.node {
 182                      item_impl(_, ref opt_trait, _, _) => {
 183                          let opt_trait : ~[trait_ref] =
 184                              opt_trait.iter()
 185                                       .map(|x| (*x).clone())
 186                                       .collect();
 187                          self.check_implementation(item, opt_trait);
 188                      }
 189                      _ => {
 190                          // Nothing to do.
 191                      }
 192                  };
 193              },
 194              .. *default_simple_visitor()
 195          })));
 196  
 197          // Check that there are no overlapping trait instances
 198          self.check_implementation_coherence();
 199  
 200          // Check whether traits with base types are in privileged scopes.
 201          self.check_privileged_scopes(crate);
 202  
 203          // Bring in external crates. It's fine for this to happen after the
 204          // coherence checks, because we ensure by construction that no errors
 205          // can happen at link time.
 206          self.add_external_crates();
 207  
 208          // Populate the table of destructors. It might seem a bit strange to
 209          // do this here, but it's actually the most convenient place, since
 210          // the coherence tables contain the trait -> type mappings.
 211          self.populate_destructor_table();
 212      }

 214      pub fn check_implementation(&self,
 215                                  item: @item,
 216                                  associated_traits: &[trait_ref]) {
 217          let tcx = self.crate_context.tcx;
 218          let self_type = ty::lookup_item_type(tcx, local_def(item.id));
 219  
 220          // If there are no traits, then this implementation must have a
 221          // base type.
 222  
 223          if associated_traits.len() == 0 {
 224              debug!("(checking implementation) no associated traits for item \
 225                      '%s'",
 226                     self.crate_context.tcx.sess.str_of(item.ident));
 227  
 228              match get_base_type_def_id(self.inference_context,
 229                                         item.span,
 230                                         self_type.ty) {
 231                  None => {
 232                      let session = self.crate_context.tcx.sess;
 233                      session.span_err(item.span,
 234                                       "no base type found for inherent implementation; \
 235                                        implement a trait or new type instead");
 236                  }
 237                  Some(_) => {
 238                      // Nothing to do.
 239                  }
 240              }
 241          }
 242  
 243          let implementation = self.create_impl_from_item(item);
 244  
 245          for associated_trait in associated_traits.iter() {
 246              let trait_ref = ty::node_id_to_trait_ref(
 247                  self.crate_context.tcx, associated_trait.ref_id);
 248              debug!("(checking implementation) adding impl for trait '%s', item '%s'",
 249                     trait_ref.repr(self.crate_context.tcx),
 250                     self.crate_context.tcx.sess.str_of(item.ident));
 251  
 252              self.add_trait_impl(trait_ref.def_id, implementation);
 253          }
 254  
 255          // Add the implementation to the mapping from implementation to base
 256          // type def ID, if there is a base type for this implementation and
 257          // the implementation does not have any associated traits.
 258          match get_base_type_def_id(self.inference_context,
 259                                     item.span,
 260                                     self_type.ty) {
 261              None => {
 262                  // Nothing to do.
 263              }
 264              Some(base_type_def_id) => {
 265                  // XXX: Gather up default methods?
 266                  if associated_traits.len() == 0 {
 267                      self.add_inherent_impl(base_type_def_id, implementation);
 268                  }
 269  
 270                  self.base_type_def_ids.insert(local_def(item.id),
 271                                                base_type_def_id);
 272              }
 273          }
 274  
 275          tcx.impls.insert(implementation.did, implementation);
 276      }
 277  
 278      // Creates default method IDs and performs type substitutions for an impl
 279      // and trait pair. Then, for each provided method in the trait, inserts a
 280      // `ProvidedMethodInfo` instance into the `provided_method_sources` map.
 281      pub fn instantiate_default_methods(&self,
 282                                         impl_id: ast::def_id,
 283                                         trait_ref: &ty::TraitRef,
 284                                         all_methods: &mut ~[@Method]) {
 285          let tcx = self.crate_context.tcx;
 286          debug!("instantiate_default_methods(impl_id=%?, trait_ref=%s)",
 287                 impl_id, trait_ref.repr(tcx));
 288  
 289          let impl_poly_type = ty::lookup_item_type(tcx, impl_id);
 290  
 291          let provided = ty::provided_trait_methods(tcx, trait_ref.def_id);
 292          for trait_method in provided.iter() {
 293              // Synthesize an ID.
 294              let new_id = parse::next_node_id(tcx.sess.parse_sess);
 295              let new_did = local_def(new_id);
 296  
 297              debug!("new_did=%? trait_method=%s", new_did, trait_method.repr(tcx));
 298  
 299              // Create substitutions for the various trait parameters.
 300              let new_method_ty =
 301                  @subst_receiver_types_in_method_ty(
 302                      tcx,
 303                      impl_id,
 304                      trait_ref,
 305                      new_did,
 306                      *trait_method,
 307                      Some(trait_method.def_id));
 308  
 309              debug!("new_method_ty=%s", new_method_ty.repr(tcx));
 310              all_methods.push(new_method_ty);
 311  
 312              // construct the polytype for the method based on the method_ty
 313              let new_generics = ty::Generics {
 314                  type_param_defs:
 315                      @vec::append(
 316                          (*impl_poly_type.generics.type_param_defs).clone(),
 317                          *new_method_ty.generics.type_param_defs),
 318                  region_param:
 319                      impl_poly_type.generics.region_param
 320              };
 321              let new_polytype = ty::ty_param_bounds_and_ty {
 322                  generics: new_generics,
 323                  ty: ty::mk_bare_fn(tcx, new_method_ty.fty.clone())
 324              };
 325              debug!("new_polytype=%s", new_polytype.repr(tcx));
 326  
 327              tcx.tcache.insert(new_did, new_polytype);
 328              tcx.methods.insert(new_did, new_method_ty);
 329  
 330              // Pair the new synthesized ID up with the
 331              // ID of the method.
 332              self.crate_context.tcx.provided_method_sources
 333                  .insert(new_did, trait_method.def_id);
 334          }
 335      }

 337      pub fn add_inherent_impl(&self,
 338                               base_def_id: def_id,
 339                               implementation: @Impl) {
 340          let tcx = self.crate_context.tcx;
 341          let implementation_list;
 342          match tcx.inherent_impls.find(&base_def_id) {
 343              None => {
 344                  implementation_list = @mut ~[];
 345                  tcx.inherent_impls.insert(base_def_id, implementation_list);
 346              }
 347              Some(&existing_implementation_list) => {
 348                  implementation_list = existing_implementation_list;
 349              }
 350          }
 351  
 352          implementation_list.push(implementation);
 353      }

 380      pub fn check_implementation_coherence_of(&self, trait_def_id: def_id) {
 381          // Unify pairs of polytypes.
 382          do self.iter_impls_of_trait(trait_def_id) |a| {
 383              let implementation_a = a;
 384              let polytype_a =
 385                  self.get_self_type_for_implementation(implementation_a);
 386  
 387              // "We have an impl of trait <trait_def_id> for type <polytype_a>,
 388              // and that impl is <implementation_a>"
 389              do self.iter_impls_of_trait(trait_def_id) |b| {
 390                  let implementation_b = b;
 391  
 392                  // An impl is coherent with itself
 393                  if a.did != b.did {
 394                      let polytype_b = self.get_self_type_for_implementation(
 395                              implementation_b);
 396  
 397                      if self.polytypes_unify(polytype_a, polytype_b) {
 398                          let session = self.crate_context.tcx.sess;
 399                          session.span_err(
 400                              self.span_of_impl(implementation_b),
 401                              fmt!("conflicting implementations for trait `%s`",
 402                                   ty::item_path_str(self.crate_context.tcx,
 403                                                     trait_def_id)));
 404                          session.span_note(self.span_of_impl(implementation_a),
 405                                            "note conflicting implementation here");
 406                      }
 407                  }
 408              }
 409          }
 410      }

 412      pub fn iter_impls_of_trait(&self, trait_def_id: def_id, f: &fn(@Impl)) {
 413          match self.crate_context.tcx.trait_impls.find(&trait_def_id) {
 414              Some(impls) => {
 415                  for &im in impls.iter() {
 416                      f(im);
 417                  }
 418              }
 419              None => { /* no impls? */ }
 420          }
 421      }

 423      pub fn polytypes_unify(&self,
 424                             polytype_a: ty_param_bounds_and_ty,
 425                             polytype_b: ty_param_bounds_and_ty)
 426                             -> bool {
 427          let universally_quantified_a =
 428              self.universally_quantify_polytype(polytype_a);
 429          let universally_quantified_b =
 430              self.universally_quantify_polytype(polytype_b);
 431  
 432          return self.can_unify_universally_quantified(
 433              &universally_quantified_a, &universally_quantified_b) ||
 434              self.can_unify_universally_quantified(
 435              &universally_quantified_b, &universally_quantified_a);
 436      }

 438      // Converts a polytype to a monotype by replacing all parameters with
 439      // type variables. Returns the monotype and the type variables created.
 440      pub fn universally_quantify_polytype(&self,
 441                                           polytype: ty_param_bounds_and_ty)
 442                                           -> UniversalQuantificationResult {
 443          let regions = match polytype.generics.region_param {
 444              None => opt_vec::Empty,
 445              Some(_) => {
 446                  opt_vec::with(
 447                      self.inference_context.next_region_var(
 448                          infer::BoundRegionInCoherence))
 449              }
 450          };
 451  
 452          let bounds_count = polytype.generics.type_param_defs.len();
 453          let type_parameters = self.inference_context.next_ty_vars(bounds_count);
 454  
 455          let substitutions = substs {
 456              regions: ty::NonerasedRegions(regions),
 457              self_ty: None,
 458              tps: type_parameters
 459          };
 460          let monotype = subst(self.crate_context.tcx,
 461                               &substitutions,
 462                               polytype.ty);
 463  
 464          UniversalQuantificationResult {
 465              monotype: monotype,
 466              type_variables: substitutions.tps,
 467              type_param_defs: polytype.generics.type_param_defs
 468          }
 469      }

 487      // Privileged scope checking
 488      pub fn check_privileged_scopes(self, crate: &Crate) {
 489          visit_crate(crate, ((), mk_vt(@Visitor {
 490              visit_item: |item, (_context, visitor)| {
 491                  match item.node {
 492                      item_mod(ref module_) => {
 493                          // Then visit the module items.
 494                          visit_mod(module_, item.span, item.id, ((), visitor));
 495                      }
 496                      item_impl(_, None, ref ast_ty, _) => {
 497                          if !self.ast_type_is_defined_in_local_crate(ast_ty) {
 498                              // This is an error.
 499                              let session = self.crate_context.tcx.sess;
 500                              session.span_err(item.span,
 501                                               "cannot associate methods with a type outside the \
 502                                                crate the type is defined in; define and implement \
 503                                                a trait or new type instead");
 504                          }
 505                      }
 506                      item_impl(_, Some(ref trait_ref), _, _) => {
 507                          // `for_ty` is `Type` in `impl Trait for Type`
 508                          let for_ty =
 509                              ty::node_id_to_type(self.crate_context.tcx,
 510                                                  item.id);
 511                          if !type_is_defined_in_local_crate(for_ty) {
 512                              // This implementation is not in scope of its base
 513                              // type. This still might be OK if the trait is
 514                              // defined in the same crate.
 515  
 516                              let trait_def_id =
 517                                  self.trait_ref_to_trait_def_id(trait_ref);
 518  
 519                              if trait_def_id.crate != LOCAL_CRATE {
 520                                  let session = self.crate_context.tcx.sess;
 521                                  session.span_err(item.span,
 522                                                   "cannot provide an extension implementation \
 523                                                    for a trait not defined in this crate");
 524                              }
 525                          }
 526  
 527                          visit_item(item, ((), visitor));
 528                      }
 529                      _ => {
 530                          visit_item(item, ((), visitor));
 531                      }
 532                  }
 533              },
 534              .. *default_visitor()
 535          })));
 536      }

 546      // here for historical reasons
 547      pub fn check_trait_methods_are_implemented(
 548          &self,
 549          all_methods: &mut ~[@Method],
 550          trait_did: def_id,
 551          trait_ref_span: span) {
 552  
 553          let tcx = self.crate_context.tcx;
 554

 557          for elt in all_methods.iter() {

 559          }

 562          for method in r.iter() {

 572      /// For coherence, when we have `impl Type`, we need to guarantee that
 573      /// `Type` is "local" to the crate. For our purposes, this means that it
 574      /// must precisely name some nominal type defined in this crate.
 575      pub fn ast_type_is_defined_in_local_crate(&self, original_type: &ast::Ty)
 576                                                -> bool {
 577          match original_type.node {
 578              ty_path(_, _, path_id) => {
 579                  match self.crate_context.tcx.def_map.get_copy(&path_id) {
 580                      def_ty(def_id) | def_struct(def_id) => {
 581                          if def_id.crate != LOCAL_CRATE {
 582                              return false;
 583                          }
 584  
 585                          // Make sure that this type precisely names a nominal
 586                          // type.
 587                          match self.crate_context
 588                                    .tcx
 589                                    .items
 590                                    .find(&def_id.node) {
 591                              None => {
 592                                  self.crate_context.tcx.sess.span_bug(
 593                                      original_type.span,
 594                                      "resolve didn't resolve this type?!");
 595                              }
 596                              Some(&node_item(item, _)) => {
 597                                  match item.node {
 598                                      item_struct(*) | item_enum(*) => true,
 599                                      _ => false,
 600                                  }
 601                              }
 602                              Some(_) => false,
 603                          }
 604                      }
 605                      _ => false
 606                  }
 607              }
 608              _ => false
 609          }
 610      }

 613      pub fn create_impl_from_item(&self, item: @item) -> @Impl {

 641                      ident: item.ident,
 642                      methods: methods
 643                  };
 644              }
 645              _ => {
 646                  self.crate_context.tcx.sess.span_bug(item.span,

 665      // External crate handling
 666  
 667      pub fn add_external_impl(&self,
 668                               impls_seen: &mut HashSet<def_id>,
 669                               impl_def_id: def_id) {
 670          let tcx = self.crate_context.tcx;
 671          let implementation = @csearch::get_impl(tcx, impl_def_id);
 672  
 673          debug!("coherence: adding impl from external crate: %s",
 674                 ty::item_path_str(tcx, implementation.did));
 675  
 676          // Make sure we don't visit the same implementation multiple times.
 677          if !impls_seen.insert(implementation.did) {
 678              // Skip this one.
 679              return
 680          }
 681          // Good. Continue.
 682  
 683          let self_type = lookup_item_type(tcx, implementation.did);
 684          let associated_traits = get_impl_trait(tcx,
 685                                                 implementation.did);
 686  
 687          // Do a sanity check to make sure that inherent methods have base
 688          // types.
 689          if associated_traits.is_none() {
 690              match get_base_type_def_id(self.inference_context,
 691                                         dummy_sp(),
 692                                         self_type.ty) {
 693                  None => {
 694                      tcx.sess.bug(fmt!("no base type for external impl with no \
 695                                        trait: %s (type %s)!",
 696                                       tcx.sess.str_of(implementation.ident),
 697                                       ty_to_str(tcx, self_type.ty)));
 698                  }
 699                  Some(_) => {} // Nothing to do.
 700              }
 701          }
 702  
 703          // Record all the trait methods.
 704          for trait_ref in associated_traits.iter() {
 705                self.add_trait_impl(trait_ref.def_id, implementation);
 706          }
 707  
 708          // For any methods that use a default implementation, add them to
 709          // the map. This is a bit unfortunate.
 710          for method in implementation.methods.iter() {
 711              for source in method.provided_source.iter() {
 712                  tcx.provided_method_sources.insert(method.def_id, *source);
 713              }
 714          }
 715  
 716          // Add the implementation to the mapping from implementation to base
 717          // type def ID, if there is a base type for this implementation.
 718          match get_base_type_def_id(self.inference_context,
 719                                     dummy_sp(),
 720                                     self_type.ty) {
 721              None => {} // Nothing to do.
 722              Some(base_type_def_id) => {
 723                  // inherent methods apply to `impl Type` but not
 724                  // `impl Trait for Type`:
 725                  if associated_traits.is_none() {
 726                      self.add_inherent_impl(base_type_def_id,
 727                                             implementation);
 728                  }
 729  
 730                  self.base_type_def_ids.insert(implementation.did,
 731                                                base_type_def_id);
 732              }
 733          }
 734  
 735          tcx.impls.insert(implementation.did, implementation);
 736      }

 761      pub fn populate_destructor_table(&self) {
 762          let tcx = self.crate_context.tcx;
 763          let drop_trait = match tcx.lang_items.drop_trait() {
 764              Some(id) => id, None => { return }
 765          };

 767  
 768          let impls;
 769          match impls_opt {
 770              None => return, // No types with (new-style) destructors present.
 771              Some(found_impls) => impls = found_impls
 772          }
 773  
 774          for impl_info in impls.iter() {
 775              if impl_info.methods.len() < 1 {
 776                  // We'll error out later. For now, just don't ICE.
 777                  loop;
 778              }
 779              let method_def_id = impl_info.methods[0].def_id;
 780  
 781              let self_type = self.get_self_type_for_implementation(*impl_info);
 782              match ty::get(self_type.ty).sty {
 783                  ty::ty_struct(type_def_id, _) => {
 784                      tcx.destructor_for_type.insert(type_def_id,
 785                                                     method_def_id);
 786                      tcx.destructors.insert(method_def_id);
 787                  }
 788                  _ => {
 789                      // Destructors only work on nominal types.
 790                      if impl_info.did.crate == ast::LOCAL_CRATE {
 791                          match tcx.items.find(&impl_info.did.node) {
 792                              Some(&ast_map::node_item(@ref item, _)) => {
 793                                  tcx.sess.span_err((*item).span,
 794                                                    "the Drop trait may only be implemented on \
 795                                                     structures");
 796                              }
 797                              _ => {
 798                                  tcx.sess.bug("didn't find impl in ast map");
 799                              }
 800                          }
 801                      } else {
 802                          tcx.sess.bug("found external impl of Drop trait on \
 803                                        something other than a struct");
 804                      }
 805                  }
 806              }
 807          }
 808      }
 809  }

 811  pub fn make_substs_for_receiver_types(tcx: ty::ctxt,
 812                                        impl_id: ast::def_id,
 813                                        trait_ref: &ty::TraitRef,
 814                                        method: &ty::Method)
 815                                        -> ty::substs {
 816      /*!
 817       * Substitutes the values for the receiver's type parameters
 818       * that are found in method, leaving the method's type parameters
 819       * intact.  This is in fact a mildly complex operation,
 820       * largely because of the hokey way that we concatenate the
 821       * receiver and method generics.
 822       */
 823  
 824      // determine how many type parameters were declared on the impl
 825      let num_impl_type_parameters = {
 826          let impl_polytype = ty::lookup_item_type(tcx, impl_id);
 827          impl_polytype.generics.type_param_defs.len()
 828      };
 829  
 830      // determine how many type parameters appear on the trait
 831      let num_trait_type_parameters = trait_ref.substs.tps.len();
 832  
 833      // the current method type has the type parameters from the trait + method
 834      let num_method_type_parameters =
 835          num_trait_type_parameters + method.generics.type_param_defs.len();
 836  
 837      // the new method type will have the type parameters from the impl + method
 838      let combined_tps = vec::from_fn(num_method_type_parameters, |i| {
 839          if i < num_trait_type_parameters {
 840              // replace type parameters that come from trait with new value
 841              trait_ref.substs.tps[i]
 842          } else {
 843              // replace type parameters that belong to method with another
 844              // type parameter, this time with the index adjusted
 845              let method_index = i - num_trait_type_parameters;
 846              let type_param_def = &method.generics.type_param_defs[method_index];
 847              let new_index = num_impl_type_parameters + method_index;
 848              ty::mk_param(tcx, new_index, type_param_def.def_id)
 849          }
 850      });
 851  
 852      return ty::substs {
 853          regions: trait_ref.substs.regions.clone(),
 854          self_ty: trait_ref.substs.self_ty,
 855          tps: combined_tps
 856      };
 857  }

 859  fn subst_receiver_types_in_method_ty(tcx: ty::ctxt,
 860                                       impl_id: ast::def_id,
 861                                       trait_ref: &ty::TraitRef,
 862                                       new_def_id: ast::def_id,
 863                                       method: &ty::Method,
 864                                       provided_source: Option<ast::def_id>)
 865                                       -> ty::Method {
 866  
 867      let combined_substs = make_substs_for_receiver_types(
 868          tcx, impl_id, trait_ref, method);
 869  
 870      ty::Method::new(
 871          method.ident,
 872  
 873          // method types *can* appear in the generic bounds
 874          method.generics.subst(tcx, &combined_substs),
 875  
 876          // method tps cannot appear in the self_ty, so use `substs` from trait ref
 877          method.transformed_self_ty.subst(tcx, &trait_ref.substs),

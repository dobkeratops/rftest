  59  fn get_base_type(inference_context: &InferCtxt,
  60                   span: Span,
  61                   original_type: t)
  62                   -> Option<t> {
  63      let resolved_type;
  64      match resolve_type(inference_context,
  65                         original_type,
  66                         resolve_ivar) {
  67          Ok(resulting_type) if !type_is_ty_var(resulting_type) => {
  68              resolved_type = resulting_type;
  69          }
  70          _ => {
  71              inference_context.tcx.sess.span_fatal(span,
  72                                                    "the type of this value must be known in order \
  73                                                     to determine the base type");
  74          }
  75      }
  76  
  77      match get(resolved_type).sty {
  78          ty_enum(..) | ty_trait(..) | ty_struct(..) => {
  79              debug!("(getting base type) found base type");
  80              Some(resolved_type)
  81          }
  82  
  83          ty_nil | ty_bot | ty_bool | ty_char | ty_int(..) | ty_uint(..) | ty_float(..) |
  84          ty_str(..) | ty_vec(..) | ty_bare_fn(..) | ty_closure(..) | ty_tup(..) |
  85          ty_infer(..) | ty_param(..) | ty_self(..) | ty_err |
  86          ty_box(_) | ty_uniq(_) | ty_ptr(_) | ty_rptr(_, _) => {
  87              debug!("(getting base type) no base type; found {:?}",
  88                     get(original_type).sty);
  89              None
  90          }
  91      }
  92  }

  94  fn type_is_defined_in_local_crate(original_type: t) -> bool {
  95      /*!
  96       *
  97       * For coherence, when we have `impl Trait for Type`, we need to
  98       * guarantee that `Type` is "local" to the
  99       * crate.  For our purposes, this means that it must contain
 100       * some nominal type defined in this crate.
 101       */
 102  
 103      let mut found_nominal = false;
 104      ty::walk_ty(original_type, |t| {
 105          match get(t).sty {
 106              ty_enum(def_id, _) |
 107              ty_trait(~ty::TyTrait { def_id, .. }) |
 108              ty_struct(def_id, _) => {
 109                  if def_id.krate == ast::LOCAL_CRATE {
 110                      found_nominal = true;
 111                  }
 112              }
 113  
 114              _ => { }
 115          }
 116      });
 117      return found_nominal;
 118  }

 121  fn get_base_type_def_id(inference_context: &InferCtxt,
 122                          span: Span,
 123                          original_type: t)
 124                          -> Option<DefId> {

 126          None => {
 127              return None;
 128          }
 129          Some(base_type) => {
 130              match get(base_type).sty {

 135                  }

 137                      fail!("get_base_type() returned a type that wasn't an \
 138                             enum, struct, or trait");
 139                  }
 140              }
 141          }
 142      }

 150  struct CoherenceCheckVisitor<'a> {
 151      cc: &'a CoherenceChecker<'a>
 152  }
 153  
 154  impl<'a> visit::Visitor<()> for CoherenceCheckVisitor<'a> {
 155      fn visit_item(&mut self, item: &Item, _: ()) {
 156  
 157          //debug!("(checking coherence) item '{}'", token::get_ident(item.ident));
 158  
 159          match item.node {
 160              ItemImpl(_, ref opt_trait, _, _) => {
 161                  match opt_trait.clone() {
 162                      Some(opt_trait) => {
 163                          self.cc.check_implementation(item, [opt_trait]);
 164                      }
 165                      None => self.cc.check_implementation(item, [])
 166                  }
 167              }
 168              _ => {
 169                  // Nothing to do.
 170              }
 171          };
 172  
 173          visit::walk_item(self, item, ());
 174      }
 175  }

 177  struct PrivilegedScopeVisitor<'a> { cc: &'a CoherenceChecker<'a> }
 178  
 179  impl<'a> visit::Visitor<()> for PrivilegedScopeVisitor<'a> {
 180      fn visit_item(&mut self, item: &Item, _: ()) {
 181  
 182          match item.node {
 183              ItemMod(ref module_) => {
 184                  // Then visit the module items.
 185                  visit::walk_mod(self, module_, ());
 186              }
 187              ItemImpl(_, None, ast_ty, _) => {
 188                  if !self.cc.ast_type_is_defined_in_local_crate(ast_ty) {
 189                      // This is an error.
 190                      let session = &self.cc.crate_context.tcx.sess;
 191                      session.span_err(item.span,
 192                                       "cannot associate methods with a type outside the \
 193                                       crate the type is defined in; define and implement \
 194                                       a trait or new type instead");
 195                  }
 196              }
 197              ItemImpl(_, Some(ref trait_ref), _, _) => {
 198                  // `for_ty` is `Type` in `impl Trait for Type`
 199                  let for_ty =
 200                      ty::node_id_to_type(self.cc.crate_context.tcx,
 201                                          item.id);
 202                  if !type_is_defined_in_local_crate(for_ty) {
 203                      // This implementation is not in scope of its base
 204                      // type. This still might be OK if the trait is
 205                      // defined in the same crate.
 206  
 207                      let trait_def_id =
 208                          self.cc.trait_ref_to_trait_def_id(trait_ref);
 209  
 210                      if trait_def_id.krate != LOCAL_CRATE {
 211                          let session = &self.cc.crate_context.tcx.sess;
 212                          session.span_err(item.span,
 213                                  "cannot provide an extension implementation \
 214                                  where both trait and type are not defined in this crate");
 215                      }
 216                  }
 217  
 218                  visit::walk_item(self, item, ());
 219              }
 220              _ => {
 221                  visit::walk_item(self, item, ());
 222              }
 223          }
 224      }
 225  }

 227  impl<'a> CoherenceChecker<'a> {
 228      fn check(&self, krate: &Crate) {
 229          // Check implementations and traits. This populates the tables
 230          // containing the inherent methods and extension methods. It also
 231          // builds up the trait inheritance table.
 232          let mut visitor = CoherenceCheckVisitor { cc: self };
 233          visit::walk_crate(&mut visitor, krate, ());
 234  
 235          // Check that there are no overlapping trait instances
 236          self.check_implementation_coherence();
 237  
 238          // Check whether traits with base types are in privileged scopes.
 239          self.check_privileged_scopes(krate);
 240  
 241          // Bring in external crates. It's fine for this to happen after the
 242          // coherence checks, because we ensure by construction that no errors
 243          // can happen at link time.
 244          self.add_external_crates();
 245  
 246          // Populate the table of destructors. It might seem a bit strange to
 247          // do this here, but it's actually the most convenient place, since
 248          // the coherence tables contain the trait -> type mappings.
 249          self.populate_destructor_table();
 250      }
 251  
 252      fn check_implementation(&self, item: &Item,
 253                              associated_traits: &[TraitRef]) {
 254          let tcx = self.crate_context.tcx;
 255          let self_type = ty::lookup_item_type(tcx, local_def(item.id));
 256  
 257          // If there are no traits, then this implementation must have a
 258          // base type.
 259  
 260          if associated_traits.len() == 0 {
 261              debug!("(checking implementation) no associated traits for item '{}'",
 262                     token::get_ident(item.ident));
 263  
 264              match get_base_type_def_id(&self.inference_context,
 265                                         item.span,
 266                                         self_type.ty) {
 267                  None => {
 268                      let session = &self.crate_context.tcx.sess;
 269                      session.span_err(item.span,
 270                                       "no base type found for inherent implementation; \
 271                                        implement a trait or new type instead");
 272                  }
 273                  Some(_) => {
 274                      // Nothing to do.
 275                  }
 276              }
 277          }
 278  
 279          let implementation = self.create_impl_from_item(item);
 280  
 281          for associated_trait in associated_traits.iter() {
 282              let trait_ref = ty::node_id_to_trait_ref(
 283                  self.crate_context.tcx, associated_trait.ref_id);
 284              debug!("(checking implementation) adding impl for trait '{}', item '{}'",
 285                     trait_ref.repr(self.crate_context.tcx),
 286                     token::get_ident(item.ident));
 287  
 288              self.add_trait_impl(trait_ref.def_id, implementation);
 289          }
 290  
 291          // Add the implementation to the mapping from implementation to base
 292          // type def ID, if there is a base type for this implementation and
 293          // the implementation does not have any associated traits.
 294          match get_base_type_def_id(&self.inference_context,
 295                                     item.span,
 296                                     self_type.ty) {
 297              None => {
 298                  // Nothing to do.
 299              }
 300              Some(base_type_def_id) => {
 301                  // FIXME: Gather up default methods?
 302                  if associated_traits.len() == 0 {
 303                      self.add_inherent_impl(base_type_def_id, implementation);
 304                  }
 305              }
 306          }
 307  
 308          tcx.impls.borrow_mut().insert(implementation.did, implementation);
 309      }
 310  
 311      // Creates default method IDs and performs type substitutions for an impl
 312      // and trait pair. Then, for each provided method in the trait, inserts a
 313      // `ProvidedMethodInfo` instance into the `provided_method_sources` map.
 314      fn instantiate_default_methods(&self, impl_id: ast::DefId,
 315                                     trait_ref: &ty::TraitRef,
 316                                     all_methods: &mut Vec<@Method> ) {
 317          let tcx = self.crate_context.tcx;
 318          debug!("instantiate_default_methods(impl_id={:?}, trait_ref={})",
 319                 impl_id, trait_ref.repr(tcx));
 320  
 321          let impl_poly_type = ty::lookup_item_type(tcx, impl_id);
 322  
 323          let prov = ty::provided_trait_methods(tcx, trait_ref.def_id);
 324          for trait_method in prov.iter() {
 325              // Synthesize an ID.
 326              let new_id = tcx.sess.next_node_id();
 327              let new_did = local_def(new_id);
 328  
 329              debug!("new_did={:?} trait_method={}", new_did, trait_method.repr(tcx));
 330  
 331              // Create substitutions for the various trait parameters.
 332              let new_method_ty =
 333                  @subst_receiver_types_in_method_ty(
 334                      tcx,
 335                      impl_id,
 336                      trait_ref,
 337                      new_did,
 338                      *trait_method,
 339                      Some(trait_method.def_id));
 340  
 341              debug!("new_method_ty={}", new_method_ty.repr(tcx));
 342              all_methods.push(new_method_ty);
 343  
 344              // construct the polytype for the method based on the method_ty
 345              let new_generics = ty::Generics {
 346                  type_param_defs:
 347                      Rc::new(Vec::from_slice(impl_poly_type.generics.type_param_defs()).append(
 348                              new_method_ty.generics.type_param_defs())),
 349                  region_param_defs:
 350                      Rc::new(Vec::from_slice(impl_poly_type.generics.region_param_defs()).append(
 351                              new_method_ty.generics.region_param_defs()))
 352              };
 353              let new_polytype = ty::ty_param_bounds_and_ty {
 354                  generics: new_generics,
 355                  ty: ty::mk_bare_fn(tcx, new_method_ty.fty.clone())
 356              };
 357              debug!("new_polytype={}", new_polytype.repr(tcx));
 358  
 359              tcx.tcache.borrow_mut().insert(new_did, new_polytype);
 360              tcx.methods.borrow_mut().insert(new_did, new_method_ty);
 361  
 362              // Pair the new synthesized ID up with the
 363              // ID of the method.
 364              self.crate_context.tcx.provided_method_sources.borrow_mut()
 365                  .insert(new_did, trait_method.def_id);
 366          }
 367      }

 411      fn check_implementation_coherence_of(&self, trait_def_id: DefId) {
 412          // Unify pairs of polytypes.
 413          self.iter_impls_of_trait_local(trait_def_id, |a| {
 414              let implementation_a = a;
 415              let polytype_a =
 416                  self.get_self_type_for_implementation(implementation_a);
 417  
 418              // "We have an impl of trait <trait_def_id> for type <polytype_a>,
 419              // and that impl is <implementation_a>"
 420              self.iter_impls_of_trait(trait_def_id, |b| {
 421                  let implementation_b = b;
 422  
 423                  // An impl is coherent with itself
 424                  if a.did != b.did {
 425                      let polytype_b = self.get_self_type_for_implementation(

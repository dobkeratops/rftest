  59  pub fn get_base_type(inference_context: @mut InferCtxt,
  60                       span: Span,
  61                       original_type: t)
  62                    -> Option<t> {
  63      let resolved_type;
  64      match resolve_type(inference_context,
  65                         original_type,
  66                         resolve_ivar) {
  67          Ok(resulting_type) if !type_is_ty_var(resulting_type) => {
  68              resolved_type = resulting_type;
  69          }
  70          _ => {
  71              inference_context.tcx.sess.span_fatal(span,
  72                                                    "the type of this value must be known in order \
  73                                                     to determine the base type");
  74          }
  75      }
  76  
  77      match get(resolved_type).sty {
  78          ty_enum(*) | ty_trait(*) | ty_struct(*) => {
  79              debug!("(getting base type) found base type");
  80              Some(resolved_type)
  81          }
  82  
  83          ty_nil | ty_bot | ty_bool | ty_char | ty_int(*) | ty_uint(*) | ty_float(*) |
  84          ty_estr(*) | ty_evec(*) | ty_bare_fn(*) | ty_closure(*) | ty_tup(*) |
  85          ty_infer(*) | ty_param(*) | ty_self(*) | ty_type | ty_opaque_box |
  86          ty_opaque_closure_ptr(*) | ty_unboxed_vec(*) | ty_err | ty_box(_) |
  87          ty_uniq(_) | ty_ptr(_) | ty_rptr(_, _) => {
  88              debug!("(getting base type) no base type; found %?",
  89                     get(original_type).sty);
  90              None
  91          }
  92      }
  93  }

  95  pub fn type_is_defined_in_local_crate(original_type: t) -> bool {
  96      /*!
  97       *
  98       * For coherence, when we have `impl Trait for Type`, we need to
  99       * guarantee that `Type` is "local" to the
 100       * crate.  For our purposes, this means that it must contain
 101       * some nominal type defined in this crate.
 102       */
 103  
 104      let mut found_nominal = false;
 105      do ty::walk_ty(original_type) |t| {
 106          match get(t).sty {
 107              ty_enum(def_id, _) |
 108              ty_trait(def_id, _, _, _, _) |
 109              ty_struct(def_id, _) => {
 110                  if def_id.crate == ast::LOCAL_CRATE {
 111                      found_nominal = true;
 112                  }
 113              }
 114  
 115              _ => { }
 116          }
 117      }
 118      return found_nominal;
 119  }

 121  // Returns the def ID of the base type, if there is one.
 122  pub fn get_base_type_def_id(inference_context: @mut InferCtxt,
 123                              span: Span,
 124                              original_type: t)
 125                           -> Option<DefId> {
 126      match get_base_type(inference_context, span, original_type) {
 127          None => {
 128              return None;
 129          }
 130          Some(base_type) => {
 131              match get(base_type).sty {
 132                  ty_enum(def_id, _) |
 133                  ty_struct(def_id, _) |
 134                  ty_trait(def_id, _, _, _, _) => {
 135                      return Some(def_id);
 136                  }
 137                  _ => {
 138                      fail!("get_base_type() returned a type that wasn't an \
 139                             enum, struct, or trait");
 140                  }
 141              }
 142          }
 143      }

 158  struct CoherenceCheckVisitor { cc: CoherenceChecker }
 159  
 160  impl visit::Visitor<()> for CoherenceCheckVisitor {
 161      fn visit_item(&mut self, item:@item, _:()) {
 162  
 163  //                debug!("(checking coherence) item '%s'",
 164  //                       self.cc.crate_context.tcx.sess.str_of(item.ident));
 165  
 166                  match item.node {
 167                      item_impl(_, ref opt_trait, _, _) => {
 168                          let opt_trait : ~[trait_ref] =
 169                              opt_trait.iter()
 170                                       .map(|x| (*x).clone())
 171                                       .collect();
 172                          self.cc.check_implementation(item, opt_trait);
 173                      }
 174                      _ => {
 175                          // Nothing to do.
 176                      }
 177                  };
 178  
 179          visit::walk_item(self, item, ());
 180      }
 181  }

 183  struct PrivilegedScopeVisitor { cc: CoherenceChecker }
 184  
 185  impl visit::Visitor<()> for PrivilegedScopeVisitor {
 186      fn visit_item(&mut self, item:@item, _:()) {
 187  
 188                  match item.node {
 189                      item_mod(ref module_) => {
 190                          // Then visit the module items.
 191                          visit::walk_mod(self, module_, ());
 192                      }
 193                      item_impl(_, None, ref ast_ty, _) => {
 194                          if !self.cc.ast_type_is_defined_in_local_crate(ast_ty) {
 195                              // This is an error.
 196                              let session = self.cc.crate_context.tcx.sess;
 197                              session.span_err(item.span,
 198                                               "cannot associate methods with a type outside the \
 199                                                crate the type is defined in; define and implement \
 200                                                a trait or new type instead");
 201                          }
 202                      }
 203                      item_impl(_, Some(ref trait_ref), _, _) => {
 204                          // `for_ty` is `Type` in `impl Trait for Type`
 205                          let for_ty =
 206                              ty::node_id_to_type(self.cc.crate_context.tcx,
 207                                                  item.id);
 208                          if !type_is_defined_in_local_crate(for_ty) {
 209                              // This implementation is not in scope of its base
 210                              // type. This still might be OK if the trait is
 211                              // defined in the same crate.
 212  
 213                              let trait_def_id =
 214                                  self.cc.trait_ref_to_trait_def_id(trait_ref);
 215  
 216                              if trait_def_id.crate != LOCAL_CRATE {
 217                                  let session = self.cc.crate_context.tcx.sess;
 218                                  session.span_err(item.span,
 219                                                   "cannot provide an extension implementation \
 220                                                    for a trait not defined in this crate");
 221                              }
 222                          }
 223  
 224                          visit::walk_item(self, item, ());
 225                      }
 226                      _ => {
 227                          visit::walk_item(self, item, ());
 228                      }
 229                  }
 230      }
 231  }

 233  impl CoherenceChecker {
 234      pub fn check_coherence(self, crate: &Crate) {
 235          // Check implementations and traits. This populates the tables
 236          // containing the inherent methods and extension methods. It also
 237          // builds up the trait inheritance table.
 238  
 239          let mut visitor = CoherenceCheckVisitor { cc: self };
 240          visit::walk_crate(&mut visitor, crate, ());
 241  
 242          // Check that there are no overlapping trait instances
 243          self.check_implementation_coherence();
 244  
 245          // Check whether traits with base types are in privileged scopes.
 246          self.check_privileged_scopes(crate);
 247  
 248          // Bring in external crates. It's fine for this to happen after the
 249          // coherence checks, because we ensure by construction that no errors
 250          // can happen at link time.
 251          self.add_external_crates();
 252  
 253          // Populate the table of destructors. It might seem a bit strange to
 254          // do this here, but it's actually the most convenient place, since
 255          // the coherence tables contain the trait -> type mappings.
 256          self.populate_destructor_table();
 257      }

 259      pub fn check_implementation(&self,
 260                                  item: @item,
 261                                  associated_traits: &[trait_ref]) {
 262          let tcx = self.crate_context.tcx;
 263          let self_type = ty::lookup_item_type(tcx, local_def(item.id));
 264  
 265          // If there are no traits, then this implementation must have a
 266          // base type.
 267  
 268          if associated_traits.len() == 0 {
 269              debug!("(checking implementation) no associated traits for item \
 270                      '%s'",
 271                     self.crate_context.tcx.sess.str_of(item.ident));
 272  
 273              match get_base_type_def_id(self.inference_context,
 274                                         item.span,
 275                                         self_type.ty) {
 276                  None => {
 277                      let session = self.crate_context.tcx.sess;
 278                      session.span_err(item.span,
 279                                       "no base type found for inherent implementation; \
 280                                        implement a trait or new type instead");
 281                  }
 282                  Some(_) => {
 283                      // Nothing to do.
 284                  }
 285              }
 286          }
 287  
 288          let implementation = self.create_impl_from_item(item);
 289  
 290          for associated_trait in associated_traits.iter() {
 291              let trait_ref = ty::node_id_to_trait_ref(
 292                  self.crate_context.tcx, associated_trait.ref_id);
 293              debug!("(checking implementation) adding impl for trait '%s', item '%s'",
 294                     trait_ref.repr(self.crate_context.tcx),
 295                     self.crate_context.tcx.sess.str_of(item.ident));
 296  
 297              self.add_trait_impl(trait_ref.def_id, implementation);
 298          }
 299  
 300          // Add the implementation to the mapping from implementation to base
 301          // type def ID, if there is a base type for this implementation and
 302          // the implementation does not have any associated traits.
 303          match get_base_type_def_id(self.inference_context,
 304                                     item.span,
 305                                     self_type.ty) {
 306              None => {
 307                  // Nothing to do.
 308              }
 309              Some(base_type_def_id) => {
 310                  // XXX: Gather up default methods?
 311                  if associated_traits.len() == 0 {
 312                      self.add_inherent_impl(base_type_def_id, implementation);
 313                  }
 314              }
 315          }
 316  
 317          tcx.impls.insert(implementation.did, implementation);
 318      }
 319  
 320      // Creates default method IDs and performs type substitutions for an impl
 321      // and trait pair. Then, for each provided method in the trait, inserts a
 322      // `ProvidedMethodInfo` instance into the `provided_method_sources` map.
 323      pub fn instantiate_default_methods(&self,
 324                                         impl_id: ast::DefId,
 325                                         trait_ref: &ty::TraitRef,
 326                                         all_methods: &mut ~[@Method]) {
 327          let tcx = self.crate_context.tcx;
 328          debug!("instantiate_default_methods(impl_id=%?, trait_ref=%s)",
 329                 impl_id, trait_ref.repr(tcx));
 330  
 331          let impl_poly_type = ty::lookup_item_type(tcx, impl_id);
 332  
 333          let provided = ty::provided_trait_methods(tcx, trait_ref.def_id);
 334          for trait_method in provided.iter() {
 335              // Synthesize an ID.
 336              let new_id = tcx.sess.next_node_id();
 337              let new_did = local_def(new_id);
 338  
 339              debug!("new_did=%? trait_method=%s", new_did, trait_method.repr(tcx));
 340  
 341              // Create substitutions for the various trait parameters.
 342              let new_method_ty =
 343                  @subst_receiver_types_in_method_ty(
 344                      tcx,
 345                      impl_id,
 346                      trait_ref,
 347                      new_did,
 348                      *trait_method,
 349                      Some(trait_method.def_id));
 350  
 351              debug!("new_method_ty=%s", new_method_ty.repr(tcx));
 352              all_methods.push(new_method_ty);
 353  
 354              // construct the polytype for the method based on the method_ty
 355              let new_generics = ty::Generics {
 356                  type_param_defs:
 357                      @vec::append(
 358                          (*impl_poly_type.generics.type_param_defs).clone(),
 359                          *new_method_ty.generics.type_param_defs),
 360                  region_param:
 361                      impl_poly_type.generics.region_param
 362              };
 363              let new_polytype = ty::ty_param_bounds_and_ty {
 364                  generics: new_generics,
 365                  ty: ty::mk_bare_fn(tcx, new_method_ty.fty.clone())
 366              };
 367              debug!("new_polytype=%s", new_polytype.repr(tcx));
 368  
 369              tcx.tcache.insert(new_did, new_polytype);
 370              tcx.methods.insert(new_did, new_method_ty);
 371  
 372              // Pair the new synthesized ID up with the
 373              // ID of the method.
 374              self.crate_context.tcx.provided_method_sources
 375                  .insert(new_did, trait_method.def_id);
 376          }
 377      }

 415      pub fn check_implementation_coherence(&self) {
 416          do self.crate_context.tcx.trait_impls.each_key |&trait_id| {
 417              self.check_implementation_coherence_of(trait_id);
 418              true
 419          };

 422      pub fn check_implementation_coherence_of(&self, trait_def_id: DefId) {
 423          // Unify pairs of polytypes.
 424          do self.iter_impls_of_trait(trait_def_id) |a| {
 425              let implementation_a = a;
 426              let polytype_a =
 427                  self.get_self_type_for_implementation(implementation_a);
 428  
 429              // "We have an impl of trait <trait_def_id> for type <polytype_a>,
 430              // and that impl is <implementation_a>"
 431              do self.iter_impls_of_trait(trait_def_id) |b| {
 432                  let implementation_b = b;
 433  
 434                  // An impl is coherent with itself
 435                  if a.did != b.did {

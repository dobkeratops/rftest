  12 //
  13 // The job of the coherence phase of typechecking is to ensure that each trait
  14 // has at most one implementation for each type. Then we build a mapping from
  15 // each trait in the system to its implementations.

  65 pub fn get_base_type(inference_context: @mut InferCtxt,
  66                      span: span,
  67                      original_type: t)
  68                   -> Option<t> {
  69     let resolved_type;
  70     match resolve_type(inference_context,
  71                        original_type,
  72                        resolve_ivar) {
  73         Ok(resulting_type) if !type_is_ty_var(resulting_type) => {
  74             resolved_type = resulting_type;
  75         }
  76         _ => {
  77             inference_context.tcx.sess.span_fatal(span,
  78                                                   "the type of this value must be known in order \
  79                                                    to determine the base type");
  80         }
  81     }
  82 
  83     match get(resolved_type).sty {
  84         ty_enum(*) | ty_trait(*) | ty_struct(*) => {
  85             debug!("(getting base type) found base type");
  86             Some(resolved_type)
  87         }
  88 
  89         ty_nil | ty_bot | ty_bool | ty_int(*) | ty_uint(*) | ty_float(*) |
  90         ty_estr(*) | ty_evec(*) | ty_bare_fn(*) | ty_closure(*) | ty_tup(*) |
  91         ty_infer(*) | ty_param(*) | ty_self(*) | ty_type | ty_opaque_box |
  92         ty_opaque_closure_ptr(*) | ty_unboxed_vec(*) | ty_err | ty_box(_) |
  93         ty_uniq(_) | ty_ptr(_) | ty_rptr(_, _) => {
  94             debug!("(getting base type) no base type; found %?",
  95                    get(original_type).sty);
  96             None
  97         }
  98     }
  99 }

 101 pub fn type_is_defined_in_local_crate(original_type: t) -> bool {
 102     /*!
 103      *
 104      * For coherence, when we have `impl Trait for Type`, we need to
 105      * guarantee that `Type` is "local" to the
 106      * crate.  For our purposes, this means that it must contain
 107      * some nominal type defined in this crate.
 108      */
 109 
 110     let mut found_nominal = false;
 111     do ty::walk_ty(original_type) |t| {
 112         match get(t).sty {
 113             ty_enum(def_id, _) |
 114             ty_trait(def_id, _, _, _, _) |
 115             ty_struct(def_id, _) => {
 116                 if def_id.crate == ast::LOCAL_CRATE {
 117                     found_nominal = true;
 118                 }
 119             }
 120 
 121             _ => { }
 122         }
 123     }
 124     return found_nominal;
 125 }

 127 // Returns the def ID of the base type, if there is one.
 128 pub fn get_base_type_def_id(inference_context: @mut InferCtxt,
 129                             span: span,
 130                             original_type: t)
 131                          -> Option<def_id> {
 132     match get_base_type(inference_context, span, original_type) {
 133         None => {
 134             return None;
 135         }
 136         Some(base_type) => {
 137             match get(base_type).sty {
 138                 ty_enum(def_id, _) |
 139                 ty_struct(def_id, _) |
 140                 ty_trait(def_id, _, _, _, _) => {
 141                     return Some(def_id);
 142                 }
 143                 _ => {
 144                     fail!("get_base_type() returned a type that wasn't an \
 145                            enum, struct, or trait");
 146                 }
 147             }
 148         }
 149     }

 161 pub struct CoherenceChecker {
 162     crate_context: @mut CrateCtxt,
 163     inference_context: @mut InferCtxt,
 164 
 165     // A mapping from implementations to the corresponding base type
 166     // definition ID.
 167 
 168     base_type_def_ids: @mut HashMap<def_id,def_id>,
 169 }
 170 
 171 impl CoherenceChecker {
 172     pub fn check_coherence(self, crate: &Crate) {
 173         // Check implementations and traits. This populates the tables
 174         // containing the inherent methods and extension methods. It also
 175         // builds up the trait inheritance table.
 176         visit_crate(crate, ((), mk_simple_visitor(@SimpleVisitor {
 177             visit_item: |item| {
 178 //                debug!("(checking coherence) item '%s'",
 179 //                       self.crate_context.tcx.sess.str_of(item.ident));
 180 
 181                 match item.node {
 182                     item_impl(_, ref opt_trait, _, _) => {
 183                         let opt_trait : ~[trait_ref] =
 184                             opt_trait.iter()
 185                                      .transform(|x| (*x).clone())
 186                                      .collect();
 187                         self.check_implementation(item, opt_trait);
 188                     }
 189                     _ => {
 190                         // Nothing to do.
 191                     }
 192                 };
 193             },
 194             .. *default_simple_visitor()
 195         })));
 196 
 197         // Check that there are no overlapping trait instances
 198         self.check_implementation_coherence();
 199 
 200         // Check whether traits with base types are in privileged scopes.
 201         self.check_privileged_scopes(crate);
 202 
 203         // Bring in external crates. It's fine for this to happen after the
 204         // coherence checks, because we ensure by construction that no errors
 205         // can happen at link time.
 206         self.add_external_crates();
 207 
 208         // Populate the table of destructors. It might seem a bit strange to
 209         // do this here, but it's actually the most convenient place, since
 210         // the coherence tables contain the trait -> type mappings.
 211         self.populate_destructor_table();
 212     }

 214     pub fn check_implementation(&self,
 215                                 item: @item,
 216                                 associated_traits: &[trait_ref]) {
 217         let tcx = self.crate_context.tcx;
 218         let self_type = ty::lookup_item_type(tcx, local_def(item.id));
 219 
 220         // If there are no traits, then this implementation must have a
 221         // base type.
 222 
 223         if associated_traits.len() == 0 {
 224             debug!("(checking implementation) no associated traits for item \
 225                     '%s'",
 226                    self.crate_context.tcx.sess.str_of(item.ident));
 227 
 228             match get_base_type_def_id(self.inference_context,
 229                                        item.span,
 230                                        self_type.ty) {
 231                 None => {
 232                     let session = self.crate_context.tcx.sess;
 233                     session.span_err(item.span,
 234                                      "no base type found for inherent implementation; \
 235                                       implement a trait or new type instead");
 236                 }
 237                 Some(_) => {
 238                     // Nothing to do.
 239                 }
 240             }
 241         }
 242 
 243         let implementation = self.create_impl_from_item(item);
 244 
 245         for associated_trait in associated_traits.iter() {
 246             let trait_ref = ty::node_id_to_trait_ref(
 247                 self.crate_context.tcx, associated_trait.ref_id);
 248             debug!("(checking implementation) adding impl for trait '%s', item '%s'",
 249                    trait_ref.repr(self.crate_context.tcx),
 250                    self.crate_context.tcx.sess.str_of(item.ident));
 251 
 252             self.add_trait_impl(trait_ref.def_id, implementation);
 253         }
 254 
 255         // Add the implementation to the mapping from implementation to base
 256         // type def ID, if there is a base type for this implementation and
 257         // the implementation does not have any associated traits.
 258         match get_base_type_def_id(self.inference_context,
 259                                    item.span,
 260                                    self_type.ty) {
 261             None => {
 262                 // Nothing to do.
 263             }
 264             Some(base_type_def_id) => {
 265                 // XXX: Gather up default methods?
 266                 if associated_traits.len() == 0 {
 267                     self.add_inherent_impl(base_type_def_id, implementation);
 268                 }
 269 
 270                 self.base_type_def_ids.insert(local_def(item.id),
 271                                               base_type_def_id);
 272             }
 273         }
 274 
 275         tcx.impls.insert(implementation.did, implementation);
 276     }
 277 
 278     // Creates default method IDs and performs type substitutions for an impl
 279     // and trait pair. Then, for each provided method in the trait, inserts a
 280     // `ProvidedMethodInfo` instance into the `provided_method_sources` map.
 281     pub fn instantiate_default_methods(&self,
 282                                        impl_id: ast::def_id,
 283                                        trait_ref: &ty::TraitRef,
 284                                        all_methods: &mut ~[@Method]) {
 285         let tcx = self.crate_context.tcx;
 286         debug!("instantiate_default_methods(impl_id=%?, trait_ref=%s)",
 287                impl_id, trait_ref.repr(tcx));
 288 
 289         let impl_poly_type = ty::lookup_item_type(tcx, impl_id);
 290 
 291         let provided = ty::provided_trait_methods(tcx, trait_ref.def_id);
 292         for trait_method in provided.iter() {
 293             // Synthesize an ID.
 294             let new_id = parse::next_node_id(tcx.sess.parse_sess);
 295             let new_did = local_def(new_id);
 296 
 297             debug!("new_did=%? trait_method=%s", new_did, trait_method.repr(tcx));
 298 
 299             // Create substitutions for the various trait parameters.
 300             let new_method_ty =
 301                 @subst_receiver_types_in_method_ty(
 302                     tcx,
 303                     impl_id,
 304                     trait_ref,
 305                     new_did,
 306                     *trait_method,
 307                     Some(trait_method.def_id));
 308 
 309             debug!("new_method_ty=%s", new_method_ty.repr(tcx));
 310             all_methods.push(new_method_ty);
 311 
 312             // construct the polytype for the method based on the method_ty
 313             let new_generics = ty::Generics {
 314                 type_param_defs:
 315                     @vec::append(
 316                         (*impl_poly_type.generics.type_param_defs).clone(),
 317                         *new_method_ty.generics.type_param_defs),
 318                 region_param:
 319                     impl_poly_type.generics.region_param
 320             };
 321             let new_polytype = ty::ty_param_bounds_and_ty {
 322                 generics: new_generics,
 323                 ty: ty::mk_bare_fn(tcx, new_method_ty.fty.clone())
 324             };
 325             debug!("new_polytype=%s", new_polytype.repr(tcx));
 326 
 327             tcx.tcache.insert(new_did, new_polytype);
 328             tcx.methods.insert(new_did, new_method_ty);
 329 
 330             // Pair the new synthesized ID up with the
 331             // ID of the method.
 332             self.crate_context.tcx.provided_method_sources
 333                 .insert(new_did, trait_method.def_id);
 334         }
 335     }

 337     pub fn add_inherent_impl(&self,
 338                              base_def_id: def_id,
 339                              implementation: @Impl) {

 348                 implementation_list = existing_implementation_list;

 350         }
 351 
 352         implementation_list.push(implementation);
 353     }

 380     pub fn check_implementation_coherence_of(&self, trait_def_id: def_id) {
 381         // Unify pairs of polytypes.
 382         do self.iter_impls_of_trait(trait_def_id) |a| {
 383             let implementation_a = a;
 384             let polytype_a =
 385                 self.get_self_type_for_implementation(implementation_a);
 386 
 387             // "We have an impl of trait <trait_def_id> for type <polytype_a>,
 388             // and that impl is <implementation_a>"
 389             do self.iter_impls_of_trait(trait_def_id) |b| {
 390                 let implementation_b = b;
 391 
 392                 // An impl is coherent with itself
 393                 if a.did != b.did {
 394                     let polytype_b = self.get_self_type_for_implementation(
 395                             implementation_b);
 396 
 397                     if self.polytypes_unify(polytype_a, polytype_b) {
 398                         let session = self.crate_context.tcx.sess;
 399                         session.span_err(
 400                             self.span_of_impl(implementation_b),
 401                             fmt!("conflicting implementations for trait `%s`",
 402                                  ty::item_path_str(self.crate_context.tcx,
 403                                                    trait_def_id)));
 404                         session.span_note(self.span_of_impl(implementation_a),
 405                                           "note conflicting implementation here");
 406                     }
 407                 }
 408             }
 409         }
 410     }

 412     pub fn iter_impls_of_trait(&self, trait_def_id: def_id, f: &fn(@Impl)) {
 413         match self.crate_context.tcx.trait_impls.find(&trait_def_id) {
 414             Some(impls) => {
 415                 for &im in impls.iter() {
 416                     f(im);
 417                 }
 418             }
 419             None => { /* no impls? */ }
 420         }
 421     }

 423     pub fn polytypes_unify(&self,
 424                            polytype_a: ty_param_bounds_and_ty,
 425                            polytype_b: ty_param_bounds_and_ty)
 426                            -> bool {
 427         let universally_quantified_a =
 428             self.universally_quantify_polytype(polytype_a);
 429         let universally_quantified_b =
 430             self.universally_quantify_polytype(polytype_b);
 431 
 432         return self.can_unify_universally_quantified(
 433             &universally_quantified_a, &universally_quantified_b) ||
 434             self.can_unify_universally_quantified(
 435             &universally_quantified_b, &universally_quantified_a);
 436     }

 438     // Converts a polytype to a monotype by replacing all parameters with
 439     // type variables. Returns the monotype and the type variables created.
 440     pub fn universally_quantify_polytype(&self,
 441                                          polytype: ty_param_bounds_and_ty)
 442                                          -> UniversalQuantificationResult {
 443         let regions = match polytype.generics.region_param {
 444             None => opt_vec::Empty,
 445             Some(_) => {
 446                 opt_vec::with(
 447                     self.inference_context.next_region_var(
 448                         infer::BoundRegionInCoherence))
 449             }
 450         };
 451 
 452         let bounds_count = polytype.generics.type_param_defs.len();
 453         let type_parameters = self.inference_context.next_ty_vars(bounds_count);
 454

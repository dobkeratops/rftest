  12  //
  13  // The job of the coherence phase of typechecking is to ensure that each trait
  14  // has at most one implementation for each type. Then we build a mapping from
  15  // each trait in the system to its implementations.

  58  fn get_base_type(inference_context: &InferCtxt,
  59                   span: Span,
  60                   original_type: t)
  61                   -> Option<t> {
  62      let resolved_type;
  63      match resolve_type(inference_context,
  64                         original_type,
  65                         resolve_ivar) {
  66          Ok(resulting_type) if !type_is_ty_var(resulting_type) => {
  67              resolved_type = resulting_type;
  68          }
  69          _ => {
  70              inference_context.tcx.sess.span_fatal(span,
  71                                                    "the type of this value must be known in order \
  72                                                     to determine the base type");
  73          }
  74      }
  75  
  76      match get(resolved_type).sty {
  77          ty_enum(..) | ty_trait(..) | ty_struct(..) => {
  78              debug!("(getting base type) found base type");
  79              Some(resolved_type)
  80          }
  81  
  82          ty_nil | ty_bot | ty_bool | ty_char | ty_int(..) | ty_uint(..) | ty_float(..) |
  83          ty_str(..) | ty_vec(..) | ty_bare_fn(..) | ty_closure(..) | ty_tup(..) |
  84          ty_infer(..) | ty_param(..) | ty_self(..) | ty_err |
  85          ty_box(_) | ty_uniq(_) | ty_ptr(_) | ty_rptr(_, _) => {
  86              debug!("(getting base type) no base type; found {:?}",
  87                     get(original_type).sty);
  88              None
  89          }
  90      }
  91  }

  93  fn type_is_defined_in_local_crate(original_type: t) -> bool {
  94      /*!
  95       *
  96       * For coherence, when we have `impl Trait for Type`, we need to
  97       * guarantee that `Type` is "local" to the
  98       * crate.  For our purposes, this means that it must contain
  99       * some nominal type defined in this crate.
 100       */
 101  
 102      let mut found_nominal = false;
 103      ty::walk_ty(original_type, |t| {
 104          match get(t).sty {
 105              ty_enum(def_id, _) |
 106              ty_trait(~ty::TyTrait { def_id, .. }) |
 107              ty_struct(def_id, _) => {
 108                  if def_id.krate == ast::LOCAL_CRATE {
 109                      found_nominal = true;
 110                  }
 111              }
 112  
 113              _ => { }
 114          }
 115      });
 116      return found_nominal;
 117  }

 119  // Returns the def ID of the base type, if there is one.
 120  fn get_base_type_def_id(inference_context: &InferCtxt,
 121                          span: Span,
 122                          original_type: t)
 123                          -> Option<DefId> {
 124      match get_base_type(inference_context, span, original_type) {
 125          None => {
 126              return None;
 127          }
 128          Some(base_type) => {
 129              match get(base_type).sty {
 130                  ty_enum(def_id, _) |
 131                  ty_struct(def_id, _) |
 132                  ty_trait(~ty::TyTrait { def_id, .. }) => {
 133                      return Some(def_id);
 134                  }
 135                  _ => {
 136                      fail!("get_base_type() returned a type that wasn't an \
 137                             enum, struct, or trait");
 138                  }
 139              }
 140          }
 141      }

 144  struct CoherenceChecker<'a> {

 147  }

 149  struct CoherenceCheckVisitor<'a> {
 150      cc: &'a CoherenceChecker<'a>
 151  }
 152  
 153  impl<'a> visit::Visitor<()> for CoherenceCheckVisitor<'a> {
 154      fn visit_item(&mut self, item: &Item, _: ()) {
 155  
 156          //debug!("(checking coherence) item '{}'", token::get_ident(item.ident));
 157  
 158          match item.node {
 159              ItemImpl(_, ref opt_trait, _, _) => {
 160                  match opt_trait.clone() {
 161                      Some(opt_trait) => {
 162                          self.cc.check_implementation(item, [opt_trait]);
 163                      }
 164                      None => self.cc.check_implementation(item, [])
 165                  }
 166              }
 167              _ => {
 168                  // Nothing to do.
 169              }
 170          };
 171  
 172          visit::walk_item(self, item, ());
 173      }
 174  }

 176  struct PrivilegedScopeVisitor<'a> { cc: &'a CoherenceChecker<'a> }
 177  
 178  impl<'a> visit::Visitor<()> for PrivilegedScopeVisitor<'a> {
 179      fn visit_item(&mut self, item: &Item, _: ()) {
 180  
 181          match item.node {
 182              ItemMod(ref module_) => {
 183                  // Then visit the module items.
 184                  visit::walk_mod(self, module_, ());
 185              }
 186              ItemImpl(_, None, ast_ty, _) => {
 187                  if !self.cc.ast_type_is_defined_in_local_crate(ast_ty) {
 188                      // This is an error.
 189                      let session = &self.cc.crate_context.tcx.sess;
 190                      session.span_err(item.span,
 191                                       "cannot associate methods with a type outside the \
 192                                       crate the type is defined in; define and implement \
 193                                       a trait or new type instead");
 194                  }
 195              }
 196              ItemImpl(_, Some(ref trait_ref), _, _) => {
 197                  // `for_ty` is `Type` in `impl Trait for Type`
 198                  let for_ty =
 199                      ty::node_id_to_type(self.cc.crate_context.tcx,
 200                                          item.id);
 201                  if !type_is_defined_in_local_crate(for_ty) {
 202                      // This implementation is not in scope of its base
 203                      // type. This still might be OK if the trait is
 204                      // defined in the same crate.
 205  
 206                      let trait_def_id =
 207                          self.cc.trait_ref_to_trait_def_id(trait_ref);
 208  
 209                      if trait_def_id.krate != LOCAL_CRATE {
 210                          let session = &self.cc.crate_context.tcx.sess;
 211                          session.span_err(item.span,
 212                                  "cannot provide an extension implementation \
 213                                  where both trait and type are not defined in this crate");
 214                      }
 215                  }
 216  
 217                  visit::walk_item(self, item, ());
 218              }
 219              _ => {
 220                  visit::walk_item(self, item, ());
 221              }
 222          }
 223      }
 224  }

 226  impl<'a> CoherenceChecker<'a> {
 227      fn check(&self, krate: &Crate) {
 228          // Check implementations and traits. This populates the tables
 229          // containing the inherent methods and extension methods. It also
 230          // builds up the trait inheritance table.
 231          let mut visitor = CoherenceCheckVisitor { cc: self };
 232          visit::walk_crate(&mut visitor, krate, ());
 233  
 234          // Check that there are no overlapping trait instances
 235          self.check_implementation_coherence();
 236  
 237          // Check whether traits with base types are in privileged scopes.
 238          self.check_privileged_scopes(krate);
 239  
 240          // Bring in external crates. It's fine for this to happen after the
 241          // coherence checks, because we ensure by construction that no errors
 242          // can happen at link time.
 243          self.add_external_crates();
 244  
 245          // Populate the table of destructors. It might seem a bit strange to
 246          // do this here, but it's actually the most convenient place, since
 247          // the coherence tables contain the trait -> type mappings.
 248          self.populate_destructor_table();
 249      }
 250  
 251      fn check_implementation(&self, item: &Item,
 252                              associated_traits: &[TraitRef]) {
 253          let tcx = self.crate_context.tcx;
 254          let self_type = ty::lookup_item_type(tcx, local_def(item.id));
 255  
 256          // If there are no traits, then this implementation must have a
 257          // base type.
 258  
 259          if associated_traits.len() == 0 {
 260              debug!("(checking implementation) no associated traits for item '{}'",
 261                     token::get_ident(item.ident));
 262  
 263              match get_base_type_def_id(&self.inference_context,
 264                                         item.span,
 265                                         self_type.ty) {
 266                  None => {
 267                      let session = &self.crate_context.tcx.sess;
 268                      session.span_err(item.span,
 269                                       "no base type found for inherent implementation; \
 270                                        implement a trait or new type instead");
 271                  }
 272                  Some(_) => {
 273                      // Nothing to do.
 274                  }
 275              }
 276          }
 277  
 278          let implementation = self.create_impl_from_item(item);
 279  
 280          for associated_trait in associated_traits.iter() {
 281              let trait_ref = ty::node_id_to_trait_ref(
 282                  self.crate_context.tcx, associated_trait.ref_id);
 283              debug!("(checking implementation) adding impl for trait '{}', item '{}'",
 284                     trait_ref.repr(self.crate_context.tcx),
 285                     token::get_ident(item.ident));
 286  
 287              self.add_trait_impl(trait_ref.def_id, implementation);
 288          }
 289  
 290          // Add the implementation to the mapping from implementation to base
 291          // type def ID, if there is a base type for this implementation and
 292          // the implementation does not have any associated traits.
 293          match get_base_type_def_id(&self.inference_context,
 294                                     item.span,
 295                                     self_type.ty) {
 296              None => {
 297                  // Nothing to do.
 298              }
 299              Some(base_type_def_id) => {
 300                  // FIXME: Gather up default methods?
 301                  if associated_traits.len() == 0 {
 302                      self.add_inherent_impl(base_type_def_id, implementation);
 303                  }
 304              }
 305          }
 306  
 307          tcx.impls.borrow_mut().insert(implementation.did, implementation);
 308      }
 309  
 310      // Creates default method IDs and performs type substitutions for an impl
 311      // and trait pair. Then, for each provided method in the trait, inserts a
 312      // `ProvidedMethodInfo` instance into the `provided_method_sources` map.
 313      fn instantiate_default_methods(&self, impl_id: ast::DefId,
 314                                     trait_ref: &ty::TraitRef,
 315                                     all_methods: &mut Vec<@Method> ) {
 316          let tcx = self.crate_context.tcx;
 317          debug!("instantiate_default_methods(impl_id={:?}, trait_ref={})",
 318                 impl_id, trait_ref.repr(tcx));
 319  
 320          let impl_poly_type = ty::lookup_item_type(tcx, impl_id);
 321  
 322          let prov = ty::provided_trait_methods(tcx, trait_ref.def_id);
 323          for trait_method in prov.iter() {
 324              // Synthesize an ID.
 325              let new_id = tcx.sess.next_node_id();
 326              let new_did = local_def(new_id);
 327  
 328              debug!("new_did={:?} trait_method={}", new_did, trait_method.repr(tcx));
 329  
 330              // Create substitutions for the various trait parameters.
 331              let new_method_ty =
 332                  @subst_receiver_types_in_method_ty(
 333                      tcx,
 334                      impl_id,
 335                      trait_ref,
 336                      new_did,
 337                      *trait_method,
 338                      Some(trait_method.def_id));
 339  
 340              debug!("new_method_ty={}", new_method_ty.repr(tcx));
 341              all_methods.push(new_method_ty);
 342  
 343              // construct the polytype for the method based on the method_ty
 344              let new_generics = ty::Generics {
 345                  type_param_defs:
 346                      Rc::new(Vec::from_slice(impl_poly_type.generics.type_param_defs()).append(
 347                              new_method_ty.generics.type_param_defs())),
 348                  region_param_defs:
 349                      impl_poly_type.generics.region_param_defs.clone()
 350              };
 351              let new_polytype = ty::ty_param_bounds_and_ty {
 352                  generics: new_generics,
 353                  ty: ty::mk_bare_fn(tcx, new_method_ty.fty.clone())
 354              };
 355              debug!("new_polytype={}", new_polytype.repr(tcx));
 356  
 357              tcx.tcache.borrow_mut().insert(new_did, new_polytype);
 358              tcx.methods.borrow_mut().insert(new_did, new_method_ty);
 359  
 360              // Pair the new synthesized ID up with the
 361              // ID of the method.
 362              self.crate_context.tcx.provided_method_sources.borrow_mut()
 363                  .insert(new_did, trait_method.def_id);
 364          }
 365      }

 367      fn add_inherent_impl(&self, base_def_id: DefId,
 368                           implementation: @Impl) {
 369          let tcx = self.crate_context.tcx;
 370          let implementation_list;
 371          let mut inherent_impls = tcx.inherent_impls.borrow_mut();
 372          match inherent_impls.find(&base_def_id) {
 373              None => {
 374                  implementation_list = @RefCell::new(Vec::new());
 375                  inherent_impls.insert(base_def_id, implementation_list);
 376              }
 377              Some(&existing_implementation_list) => {
 378                  implementation_list = existing_implementation_list;
 379              }
 380          }
 381  
 382          implementation_list.borrow_mut().push(implementation);
 383      }

 403      fn check_implementation_coherence(&self) {
 404          for &trait_id in self.crate_context.tcx.trait_impls.borrow().keys() {
 405              self.check_implementation_coherence_of(trait_id);
 406          }
 407      }

 409      fn check_implementation_coherence_of(&self, trait_def_id: DefId) {
 410          // Unify pairs of polytypes.
 411          self.iter_impls_of_trait_local(trait_def_id, |a| {
 412              let implementation_a = a;
 413              let polytype_a =
 414                  self.get_self_type_for_implementation(implementation_a);
 415  
 416              // "We have an impl of trait <trait_def_id> for type <polytype_a>,
 417              // and that impl is <implementation_a>"
 418              self.iter_impls_of_trait(trait_def_id, |b| {
 419                  let implementation_b = b;
 420  
 421                  // An impl is coherent with itself
 422                  if a.did != b.did {
 423                      let polytype_b = self.get_self_type_for_implementation(
 424                              implementation_b);
 425  
 426                      if self.polytypes_unify(polytype_a.clone(), polytype_b) {
 427                          let session = &self.crate_context.tcx.sess;
 428                          session.span_err(
 429                              self.span_of_impl(implementation_a),

 731  pub fn make_substs_for_receiver_types(tcx: &ty::ctxt,
 732                                        impl_id: ast::DefId,
 733                                        trait_ref: &ty::TraitRef,
 734                                        method: &ty::Method)
 735                                        -> ty::substs {
 736      /*!
 737       * Substitutes the values for the receiver's type parameters
 738       * that are found in method, leaving the method's type parameters
 739       * intact.  This is in fact a mildly complex operation,
 740       * largely because of the hokey way that we concatenate the
 741       * receiver and method generics.
 742       */
 743

 745      let num_impl_type_parameters = {
 746          let impl_polytype = ty::lookup_item_type(tcx, impl_id);
 747          impl_polytype.generics.type_param_defs().len()
 748      };
 749  
 750      // determine how many type parameters appear on the trait
 751      let num_trait_type_parameters = trait_ref.substs.tps.len();
 752  
 753      // the current method type has the type parameters from the trait + method
 754      let num_method_type_parameters =
 755          num_trait_type_parameters + method.generics.type_param_defs().len();
 756  
 757      // the new method type will have the type parameters from the impl + method
 758      let combined_tps = Vec::from_fn(num_method_type_parameters, |i| {
 759          if i < num_trait_type_parameters {
 760              // replace type parameters that come from trait with new value
 761              *trait_ref.substs.tps.get(i)
 762          } else {
 763              // replace type parameters that belong to method with another
 764              // type parameter, this time with the index adjusted
 765              let method_index = i - num_trait_type_parameters;
 766              let type_param_def = &method.generics.type_param_defs()[method_index];
 767              let new_index = num_impl_type_parameters + method_index;
 768              ty::mk_param(tcx, new_index, type_param_def.def_id)
 769          }
 770      });
 771  
 772      return ty::substs {
 773          regions: trait_ref.substs.regions.clone(),
 774          self_ty: trait_ref.substs.self_ty,
 775          tps: combined_tps
 776      };
 777  }

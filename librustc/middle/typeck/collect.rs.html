   11  /*
   12  
   13  # Collect phase
   14  
   15  The collect phase of type check has the job of visiting all items,
   16  determining their type, and writing that type into the `tcx.tcache`
   17  table.  Despite its name, this table does not really operate as a
   18  *cache*, at least not for the types of items defined within the
   19  current crate: we assume that after the collect phase, the types of
   20  all local items will be present in the table.
   21  
   22  Unlike most of the types that are present in Rust, the types computed
   23  for each item are in fact polytypes.  In "layman's terms", this means
   24  that they are generic types that may have type parameters (more
   25  mathematically phrased, they are universally quantified over a set of
   26  type parameters).  Polytypes are represented by an instance of
   27  `ty::ty_param_bounds_and_ty`.  This combines the core type along with
   28  a list of the bounds for each parameter.  Type parameters themselves
   29  are represented as `ty_param()` instances.
   30  
   31  */

  140  pub fn get_enum_variant_types(ccx: &CrateCtxt,
  141                                enum_ty: ty::t,
  142                                variants: &[ast::P<ast::Variant>],
  143                                generics: &ast::Generics) {
  144      let tcx = ccx.tcx;
  145  
  146      // Create a set of parameter types shared among all the variants.
  147      for variant in variants.iter() {
  148          // Nullary enum constructors get turned into constants; n-ary enum
  149          // constructors get turned into functions.
  150          let scope = variant.node.id;
  151          let result_ty = match variant.node.kind {
  152              ast::TupleVariantKind(ref args) if args.len() > 0 => {
  153                  let rs = ExplicitRscope;
  154                  let input_tys: Vec<_> = args.iter().map(|va| ccx.to_ty(&rs, va.ty)).collect();
  155                  ty::mk_ctor_fn(tcx, scope, input_tys.as_slice(), enum_ty)
  156              }
  157  
  158              ast::TupleVariantKind(_) => {
  159                  enum_ty
  160              }
  161  
  162              ast::StructVariantKind(struct_def) => {
  163                  let tpt = ty_param_bounds_and_ty {
  164                      generics: ty_generics_for_type(ccx, generics),
  165                      ty: enum_ty
  166                  };
  167  
  168                  convert_struct(ccx, struct_def, tpt, variant.node.id);
  169  
  170                  let input_tys: Vec<_> = struct_def.fields.iter().map(
  171                      |f| ty::node_id_to_type(ccx.tcx, f.node.id)).collect();
  172                  ty::mk_ctor_fn(tcx, scope, input_tys.as_slice(), enum_ty)
  173              }
  174          };
  175  
  176          let tpt = ty_param_bounds_and_ty {
  177              generics: ty_generics_for_type(ccx, generics),
  178              ty: result_ty
  179          };
  180  
  181          tcx.tcache.borrow_mut().insert(local_def(variant.node.id), tpt);
  182  
  183          write_ty_to_tcx(tcx, variant.node.id, result_ty);
  184      }
  185  }

  187  pub fn ensure_trait_methods(ccx: &CrateCtxt, trait_id: ast::NodeId) {
  188      let tcx = ccx.tcx;
  189      match tcx.map.get(trait_id) {
  190          ast_map::NodeItem(item) => {
  191              match item.node {
  192                  ast::ItemTrait(ref generics, _, ref ms) => {
  193                      let trait_ty_generics = ty_generics_for_type(ccx, generics);
  194  
  195                      // For each method, construct a suitable ty::Method and
  196                      // store it into the `tcx.methods` table:
  197                      for m in ms.iter() {
  198                          let ty_method = @match m {
  199                              &ast::Required(ref m) => {
  200                                  ty_method_of_trait_method(
  201                                      ccx, trait_id, &trait_ty_generics,
  202                                      &m.id, &m.ident, &m.explicit_self,
  203                                      &m.generics, &m.fn_style, m.decl)
  204                              }
  205  
  206                              &ast::Provided(ref m) => {
  207                                  ty_method_of_trait_method(
  208                                      ccx, trait_id, &trait_ty_generics,
  209                                      &m.id, &m.ident, &m.explicit_self,
  210                                      &m.generics, &m.fn_style, m.decl)
  211                              }
  212                          };
  213  
  214                          if ty_method.explicit_self == ast::SelfStatic {
  215                              make_static_method_ty(ccx, trait_id, ty_method,
  216                                                    &trait_ty_generics);
  217                          }
  218  
  219                          tcx.methods.borrow_mut().insert(ty_method.def_id,
  220                                                          ty_method);
  221                      }
  222  
  223                      // Add an entry mapping
  224                      let method_def_ids = @ms.iter().map(|m| {
  225                          match m {
  226                              &ast::Required(ref ty_method) => {
  227                                  local_def(ty_method.id)
  228                              }
  229                              &ast::Provided(ref method) => {
  230                                  local_def(method.id)
  231                              }
  232                          }
  233                      }).collect();
  234  
  235                      let trait_def_id = local_def(trait_id);
  236                      tcx.trait_method_def_ids.borrow_mut()
  237                          .insert(trait_def_id, method_def_ids);
  238                  }
  239                  _ => {} // Ignore things that aren't traits.
  240              }
  241          }
  242          _ => { /* Ignore things that aren't traits */ }
  243      }

  245      fn make_static_method_ty(ccx: &CrateCtxt,
  246                               trait_id: ast::NodeId,
  247                               m: &ty::Method,
  248                               trait_ty_generics: &ty::Generics) {
  249          // If declaration is
  250          //
  251          //     trait Trait<'a,'b,'c,a,b,c> {
  252          //        fn foo<'d,'e,'f,d,e,f>(...) -> Self;
  253          //     }
  254          //
  255          // and we will create a function like
  256          //
  257          //     fn foo<'a,'b,'c,   // First the lifetime params from trait
  258          //            'd,'e,'f,   // Then lifetime params from `foo()`
  259          //            a,b,c,      // Then type params from trait
  260          //            D:Trait<'a,'b,'c,a,b,c>, // Then this sucker
  261          //            E,F,G       // Then type params from `foo()`, offset by 1
  262          //           >(...) -> D' {}
  263          //
  264          // Note that `Self` is replaced with an explicit type
  265          // parameter D that is sandwiched in between the trait params
  266          // and the method params, and thus the indices of the method
  267          // type parameters are offset by 1 (that is, the method
  268          // parameters are mapped from d, e, f to E, F, and G).  The
  269          // choice of this ordering is somewhat arbitrary.
  270          //
  271          // Note also that the bound for `D` is `Trait<'a,'b,'c,a,b,c>`.
  272          // This implies that the lifetime parameters that were inherited
  273          // from the trait (i.e., `'a`, `'b`, and `'c`) all must be early
  274          // bound, since they appear in a trait bound.
  275          //
  276          // Also, this system is rather a hack that should be replaced
  277          // with a more uniform treatment of Self (which is partly
  278          // underway).
  279  
  280          // build up a subst that shifts all of the parameters over
  281          // by one and substitute in a new type param for self
  282  
  283          let tcx = ccx.tcx;
  284  
  285          let dummy_defid = ast::DefId {krate: 0, node: 0};
  286  
  287          // Represents [A',B',C']
  288          let num_trait_bounds = trait_ty_generics.type_param_defs().len();
  289          let non_shifted_trait_tps = Vec::from_fn(num_trait_bounds, |i| {
  290              ty::mk_param(tcx, i, trait_ty_generics.type_param_defs()[i].def_id)
  291          });
  292  
  293          // Represents [D']
  294          let self_param = ty::mk_param(tcx, num_trait_bounds,
  295                                        dummy_defid);
  296  
  297          // Represents [E',F',G']
  298          let num_method_bounds = m.generics.type_param_defs().len();
  299          let shifted_method_tps = Vec::from_fn(num_method_bounds, |i| {
  300              ty::mk_param(tcx, i + num_trait_bounds + 1,
  301                           m.generics.type_param_defs()[i].def_id)
  302          });
  303  
  304          // Convert the regions 'a, 'b, 'c defined on the trait into
  305          // bound regions on the fn. Note that because these appear in the
  306          // bound for `Self` they must be early bound.
  307          let new_early_region_param_defs = trait_ty_generics.region_param_defs.clone();
  308          let rps_from_trait =
  309              trait_ty_generics.region_param_defs().iter().
  310              enumerate().
  311              map(|(index,d)| ty::ReEarlyBound(d.def_id.node, index, d.name)).
  312              collect();
  313  
  314          // build up the substitution from
  315          //     'a,'b,'c => 'a,'b,'c
  316          //     A,B,C => A',B',C'
  317          //     Self => D'
  318          //     D,E,F => E',F',G'
  319          let substs = substs {
  320              regions: ty::NonerasedRegions(rps_from_trait),
  321              self_ty: Some(self_param),
  322              tps: non_shifted_trait_tps.append(shifted_method_tps.as_slice())
  323          };
  324  
  325          // create the type of `foo`, applying the substitution above
  326          let ty = ty::subst(tcx,
  327                             &substs,
  328                             ty::mk_bare_fn(tcx, m.fty.clone()));
  329  
  330          // create the type parameter definitions for `foo`, applying
  331          // the substitution to any traits that appear in their bounds.
  332  
  333          // add in the type parameters from the trait
  334          let mut new_type_param_defs = Vec::new();
  335          let substd_type_param_defs =
  336              trait_ty_generics.type_param_defs.subst(tcx, &substs);
  337          new_type_param_defs.push_all(substd_type_param_defs.as_slice());
  338  
  339          // add in the "self" type parameter
  340          let self_trait_def = get_trait_def(ccx, local_def(trait_id));
  341          let self_trait_ref = self_trait_def.trait_ref.subst(tcx, &substs);
  342          new_type_param_defs.push(ty::TypeParameterDef {
  343              ident: special_idents::self_,
  344              def_id: dummy_defid,
  345              bounds: @ty::ParamBounds {
  346                  builtin_bounds: ty::EmptyBuiltinBounds(),
  347                  trait_bounds: vec!(self_trait_ref)
  348              },
  349              default: None
  350          });
  351  
  352          // add in the type parameters from the method
  353          let substd_type_param_defs = m.generics.type_param_defs.subst(tcx, &substs);
  354          new_type_param_defs.push_all(substd_type_param_defs.as_slice());
  355  
  356          debug!("static method {} type_param_defs={} ty={}, substs={}",
  357                 m.def_id.repr(tcx),
  358                 new_type_param_defs.repr(tcx),
  359                 ty.repr(tcx),
  360                 substs.repr(tcx));
  361  
  362          tcx.tcache.borrow_mut().insert(m.def_id,
  363                            ty_param_bounds_and_ty {
  364                                generics: ty::Generics {
  365                                    type_param_defs: Rc::new(new_type_param_defs),
  366                                    region_param_defs: new_early_region_param_defs
  367                                },
  368                                ty: ty
  369                            });
  370      }

  372      fn ty_method_of_trait_method(this: &CrateCtxt,
  373                                   trait_id: ast::NodeId,
  374                                   trait_generics: &ty::Generics,
  375                                   m_id: &ast::NodeId,
  376                                   m_ident: &ast::Ident,
  377                                   m_explicit_self: &ast::ExplicitSelf,
  378                                   m_generics: &ast::Generics,
  379                                   m_fn_style: &ast::FnStyle,
  380                                   m_decl: &ast::FnDecl) -> ty::Method
  381      {
  382          let trait_self_ty = ty::mk_self(this.tcx, local_def(trait_id));
  383          let fty = astconv::ty_of_method(this, *m_id, *m_fn_style, trait_self_ty,

  402  pub fn ensure_supertraits(ccx: &CrateCtxt,

  404                            sp: codemap::Span,

  454  fn convert_methods(ccx: &CrateCtxt,
  455                     container: MethodContainer,
  456                     ms: &[@ast::Method],
  457                     untransformed_rcvr_ty: ty::t,
  458                     rcvr_ty_generics: &ty::Generics,
  459                     rcvr_ast_generics: &ast::Generics,
  460                     rcvr_visibility: ast::Visibility)
  461  {
  462      let tcx = ccx.tcx;
  463      let mut seen_methods = HashSet::new();
  464      for m in ms.iter() {
  465          if !seen_methods.insert(m.ident.repr(ccx.tcx)) {
  466              tcx.sess.span_err(m.span, "duplicate method in trait impl");
  467          }

  469          let num_rcvr_ty_params = rcvr_ty_generics.type_param_defs().len();
  470          let m_ty_generics = ty_generics_for_fn_or_method(ccx, &m.generics,
  471                                                           num_rcvr_ty_params);

  473                                  container,
  474                                  *m,
  475                                  untransformed_rcvr_ty,
  476                                  rcvr_ast_generics,
  477                                  rcvr_visibility);
  478          let fty = ty::mk_bare_fn(tcx, mty.fty.clone());
  479          debug!("method {} (id {}) has type {}",
  480                  m.ident.repr(ccx.tcx),
  481                  m.id,
  482                  fty.repr(ccx.tcx));
  483          tcx.tcache.borrow_mut().insert(
  484              local_def(m.id),
  485  
  486              // n.b.: the type of a method is parameterized by both
  487              // the parameters on the receiver and those on the method
  488              // itself
  489              ty_param_bounds_and_ty {
  490                  generics: ty::Generics {
  491                      type_param_defs: Rc::new(Vec::from_slice(rcvr_ty_generics.type_param_defs())
  492                                               .append(m_ty_generics.type_param_defs())),
  493                      region_param_defs: Rc::new(Vec::from_slice(rcvr_ty_generics.region_param_defs())
  494                                                 .append(m_ty_generics.region_param_defs())),
  495                  },
  496                  ty: fty
  497              });
  498  
  499          write_ty_to_tcx(tcx, m.id, fty);
  500  
  501          tcx.methods.borrow_mut().insert(mty.def_id, mty);
  502      }

  562  pub fn convert(ccx: &CrateCtxt, it: &ast::Item) {
  563      let tcx = ccx.tcx;
  564      debug!("convert: item {} with id {}", token::get_ident(it.ident), it.id);
  565      match it.node {
  566          // These don't define types.
  567          ast::ItemForeignMod(_) | ast::ItemMod(_) | ast::ItemMac(_) => {}
  568          ast::ItemEnum(ref enum_definition, ref generics) => {
  569              ensure_no_ty_param_bounds(ccx, it.span, generics, "enumeration");
  570              let tpt = ty_of_item(ccx, it);
  571              write_ty_to_tcx(tcx, it.id, tpt.ty);
  572              get_enum_variant_types(ccx,
  573                                     tpt.ty,
  574                                     enum_definition.variants.as_slice(),
  575                                     generics);
  576          },
  577          ast::ItemImpl(ref generics, ref opt_trait_ref, selfty, ref ms) => {
  578              let ty_generics = ty_generics_for_type(ccx, generics);
  579              let selfty = ccx.to_ty(&ExplicitRscope, selfty);
  580              write_ty_to_tcx(tcx, it.id, selfty);
  581  
  582              tcx.tcache.borrow_mut().insert(local_def(it.id),
  583                                  ty_param_bounds_and_ty {
  584                                      generics: ty_generics.clone(),
  585                                      ty: selfty});
  586  
  587              // If there is a trait reference, treat the methods as always public.
  588              // This is to work around some incorrect behavior in privacy checking:
  589              // when the method belongs to a trait, it should acquire the privacy
  590              // from the trait, not the impl. Forcing the visibility to be public
  591              // makes things sorta work.
  592              let parent_visibility = if opt_trait_ref.is_some() {
  593                  ast::Public
  594              } else {
  595                  it.vis
  596              };
  597  
  598              convert_methods(ccx,
  599                              ImplContainer(local_def(it.id)),
  600                              ms.as_slice(),
  601                              selfty,
  602                              &ty_generics,
  603                              generics,
  604                              parent_visibility);
  605  
  606              for trait_ref in opt_trait_ref.iter() {
  607                  let trait_ref = instantiate_trait_ref(ccx, trait_ref, selfty);
  608  
  609                  // Prevent the builtin kind traits from being manually implemented.
  610                  if tcx.lang_items.to_builtin_kind(trait_ref.def_id).is_some() {
  611                      tcx.sess.span_err(it.span,
  612                          "cannot provide an explicit implementation \
  613                           for a builtin kind");
  614                  }
  615              }
  616          },
  617          ast::ItemTrait(ref generics, _, ref trait_methods) => {
  618              let trait_def = trait_def_of_item(ccx, it);
  619  
  620              // Run convert_methods on the provided methods.
  621              let (_, provided_methods) =
  622                  split_trait_methods(trait_methods.as_slice());
  623              let untransformed_rcvr_ty = ty::mk_self(tcx, local_def(it.id));
  624              convert_methods(ccx,
  625                              TraitContainer(local_def(it.id)),
  626                              provided_methods.as_slice(),
  627                              untransformed_rcvr_ty,
  628                              &trait_def.generics,
  629                              generics,
  630                              it.vis);
  631  
  632              // We need to do this *after* converting methods, since
  633              // convert_methods produces a tcache entry that is wrong for
  634              // static trait methods. This is somewhat unfortunate.
  635              ensure_trait_methods(ccx, it.id);
  636          },
  637          ast::ItemStruct(struct_def, ref generics) => {
  638              ensure_no_ty_param_bounds(ccx, it.span, generics, "structure");
  639  
  640              // Write the class type
  641              let tpt = ty_of_item(ccx, it);
  642              write_ty_to_tcx(tcx, it.id, tpt.ty);
  643  
  644              tcx.tcache.borrow_mut().insert(local_def(it.id), tpt.clone());
  645  
  646              convert_struct(ccx, struct_def, tpt, it.id);
  647          },
  648          ast::ItemTy(_, ref generics) => {
  649              ensure_no_ty_param_bounds(ccx, it.span, generics, "type");
  650              let tpt = ty_of_item(ccx, it);
  651              write_ty_to_tcx(tcx, it.id, tpt.ty);
  652          },
  653          ast::ItemFn(_, _, abi, ref generics, _) => {
  654              ensure_generics_abi(ccx, it.span, abi, generics);
  655              let tpt = ty_of_item(ccx, it);
  656              write_ty_to_tcx(tcx, it.id, tpt.ty);
  657          },
  658          _ => {
  659              // This call populates the type cache with the converted type
  660              // of the item in passing. All we have to do here is to write
  661              // it into the node type table.
  662              let tpt = ty_of_item(ccx, it);
  663              write_ty_to_tcx(tcx, it.id, tpt.ty);
  664          },
  665      }
  666  }

  668  pub fn convert_struct(ccx: &CrateCtxt,
  669                        struct_def: &ast::StructDef,
  670                        tpt: ty::ty_param_bounds_and_ty,
  671                        id: ast::NodeId) {
  672      let tcx = ccx.tcx;
  673  
  674      // Write the type of each of the members
  675      for f in struct_def.fields.iter() {
  676         convert_field(ccx, &tpt.generics, f);
  677      }
  678      let substs = mk_item_substs(ccx, &tpt.generics, None);
  679      let selfty = ty::mk_struct(tcx, local_def(id), substs);
  680  
  681      // If this struct is enum-like or tuple-like, create the type of its
  682      // constructor.
  683      match struct_def.ctor_id {
  684          None => {}
  685          Some(ctor_id) => {
  686              if struct_def.fields.len() == 0 {
  687                  // Enum-like.
  688                  write_ty_to_tcx(tcx, ctor_id, selfty);
  689  
  690                  tcx.tcache.borrow_mut().insert(local_def(ctor_id), tpt);
  691              } else if struct_def.fields.get(0).node.kind.is_unnamed() {
  692                  // Tuple-like.
  693                  let inputs: Vec<_> = struct_def.fields.iter().map(
  694                          |field| tcx.tcache.borrow().get(
  695                              &local_def(field.node.id)).ty).collect();
  696                  let ctor_fn_ty = ty::mk_ctor_fn(tcx,
  697                                                  ctor_id,
  698                                                  inputs.as_slice(),
  699                                                  selfty);
  700                  write_ty_to_tcx(tcx, ctor_id, ctor_fn_ty);
  701                  tcx.tcache.borrow_mut().insert(local_def(ctor_id),
  702                                    ty_param_bounds_and_ty {
  703                      generics: tpt.generics,
  704                      ty: ctor_fn_ty
  705                  });
  706              }
  707          }
  708      }
  709  }

  728  pub fn instantiate_trait_ref(ccx: &CrateCtxt,
  729                               ast_trait_ref: &ast::TraitRef,
  730                               self_ty: ty::t) -> @ty::TraitRef
  731  {
  732      /*!
  733       * Instantiates the path for the given trait reference, assuming that
  734       * it's bound to a valid trait type. Returns the def_id for the defining
  735       * trait. Fails if the type is a type other than a trait type.
  736       */

  739      let rscope = ExplicitRscope;

  742          ast::DefTrait(trait_did) => {

  744                  astconv::ast_path_to_trait_ref(
  745                      ccx, &rscope, trait_did, Some(self_ty), &ast_trait_ref.path);
  746  
  747              ccx.tcx.trait_refs.borrow_mut().insert(ast_trait_ref.ref_id,
  748                                                     trait_ref);
  749              return trait_ref;
  750          }
  751          _ => {
  752              ccx.tcx.sess.span_fatal(
  753                  ast_trait_ref.path.span,
  754                  format!("`{}` is not a trait",
  755                      path_to_str(&ast_trait_ref.path)));
  756          }
  757      }

  760  fn get_trait_def(ccx: &CrateCtxt, trait_id: ast::DefId) -> @ty::TraitDef {
  761      if trait_id.krate != ast::LOCAL_CRATE {
  762          return ty::lookup_trait_def(ccx.tcx, trait_id)
  763      }
  764  
  765      match ccx.tcx.map.get(trait_id.node) {

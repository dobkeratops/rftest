   11 /*
   12 
   13 # Collect phase
   14 
   15 The collect phase of type check has the job of visiting all items,
   16 determining their type, and writing that type into the `tcx.tcache`
   17 table.  Despite its name, this table does not really operate as a
   18 *cache*, at least not for the types of items defined within the
   19 current crate: we assume that after the collect phase, the types of
   20 all local items will be present in the table.
   21 
   22 Unlike most of the types that are present in Rust, the types computed
   23 for each item are in fact polytypes.  In "layman's terms", this means
   24 that they are generic types that may have type parameters (more
   25 mathematically phrased, they are universally quantified over a set of
   26 type parameters).  Polytypes are represented by an instance of
   27 `ty::ty_param_bounds_and_ty`.  This combines the core type along with
   28 a list of the bounds for each parameter.  Type parameters themselves
   29 are represented as `ty_param()` instances.
   30 
   31 */

  138 pub fn get_enum_variant_types(ccx: &CrateCtxt,
  139                               enum_ty: ty::t,
  140                               variants: &[ast::variant],
  141                               generics: &ast::Generics,
  142                               rp: Option<ty::region_variance>) {
  143     let tcx = ccx.tcx;
  144 
  145     // Create a set of parameter types shared among all the variants.
  146     for variant in variants.iter() {
  147         let region_parameterization =
  148             RegionParameterization::from_variance_and_generics(rp, generics);
  149 
  150         // Nullary enum constructors get turned into constants; n-ary enum
  151         // constructors get turned into functions.
  152         let result_ty;
  153         match variant.node.kind {
  154             ast::tuple_variant_kind(ref args) if args.len() > 0 => {
  155                 let rs = type_rscope(region_parameterization);
  156                 let input_tys = args.map(|va| ccx.to_ty(&rs, &va.ty));
  157                 result_ty = Some(ty::mk_ctor_fn(tcx, input_tys, enum_ty));
  158             }
  159 
  160             ast::tuple_variant_kind(_) => {
  161                 result_ty = Some(enum_ty);
  162             }
  163 
  164             ast::struct_variant_kind(struct_def) => {
  165                 let tpt = ty_param_bounds_and_ty {
  166                     generics: ty_generics(ccx, rp, generics, 0),
  167                     ty: enum_ty
  168                 };
  169 
  170                 convert_struct(ccx,
  171                                rp,
  172                                struct_def,
  173                                generics,
  174                                tpt,
  175                                variant.node.id);
  176 
  177                 let input_tys = struct_def.fields.map(
  178                     |f| ty::node_id_to_type(ccx.tcx, f.node.id));
  179                 result_ty = Some(ty::mk_ctor_fn(tcx, input_tys, enum_ty));
  180             }
  181         };
  182 
  183         match result_ty {
  184             None => {}
  185             Some(result_ty) => {
  186                 let tpt = ty_param_bounds_and_ty {
  187                     generics: ty_generics(ccx, rp, generics, 0),
  188                     ty: result_ty
  189                 };
  190                 tcx.tcache.insert(local_def(variant.node.id), tpt);
  191                 write_ty_to_tcx(tcx, variant.node.id, result_ty);
  192             }
  193         }
  194     }
  195 }

  197 pub fn ensure_trait_methods(ccx: &CrateCtxt,
  198                             trait_id: ast::NodeId)
  199 {
  200     let tcx = ccx.tcx;
  201     let region_paramd = tcx.region_paramd_items.find(&trait_id).map_move(|x| *x);
  202     match tcx.items.get_copy(&trait_id) {
  203         ast_map::node_item(@ast::item {
  204             node: ast::item_trait(ref generics, _, ref ms),
  205             _
  206         }, _) => {
  207             let trait_ty_generics = ty_generics(ccx, region_paramd, generics, 0);
  208 
  209             // For each method, construct a suitable ty::Method and
  210             // store it into the `tcx.methods` table:
  211             for m in ms.iter() {
  212                 let ty_method = @match m {
  213                     &ast::required(ref m) => {
  214                         ty_method_of_trait_method(
  215                             ccx, trait_id, region_paramd, generics,
  216                             &m.id, &m.ident, &m.explicit_self,
  217                             &m.generics, &m.purity, &m.decl)
  218                     }
  219 
  220                     &ast::provided(ref m) => {
  221                         ty_method_of_trait_method(
  222                             ccx, trait_id, region_paramd, generics,
  223                             &m.id, &m.ident, &m.explicit_self,
  224                             &m.generics, &m.purity, &m.decl)
  225                     }
  226                 };
  227 
  228                 if ty_method.explicit_self == ast::sty_static {
  229                     make_static_method_ty(ccx, trait_id, ty_method,
  230                                           &trait_ty_generics);
  231                 }
  232 
  233                 tcx.methods.insert(ty_method.def_id, ty_method);
  234             }
  235 
  236             // Add an entry mapping
  237             let method_def_ids = @ms.map(|m| {
  238                 match m {
  239                     &ast::required(ref ty_method) => local_def(ty_method.id),
  240                     &ast::provided(ref method) => local_def(method.id)
  241                 }
  242             });
  243 
  244             let trait_def_id = local_def(trait_id);
  245             tcx.trait_method_def_ids.insert(trait_def_id, method_def_ids);
  246         }
  247         _ => { /* Ignore things that aren't traits */ }
  248     }

  250     fn make_static_method_ty(ccx: &CrateCtxt,
  251                              trait_id: ast::NodeId,
  252                              m: &ty::Method,
  253                              trait_ty_generics: &ty::Generics) {
  254         // If declaration is
  255         //
  256         //     trait<A,B,C> {
  257         //        fn foo<D,E,F>(...) -> Self;
  258         //     }
  259         //
  260         // and we will create a function like
  261         //
  262         //     fn foo<A',B',C',D',E',F',G'>(...) -> D' {}
  263         //
  264         // Note that `Self` is replaced with an explicit type
  265         // parameter D' that is sandwiched in between the trait params
  266         // and the method params, and thus the indices of the method
  267         // type parameters are offset by 1 (that is, the method
  268         // parameters are mapped from D, E, F to E', F', and G').  The
  269         // choice of this ordering is somewhat arbitrary.
  270         //
  271         // Also, this system is rather a hack that should be replaced
  272         // with a more uniform treatment of Self (which is partly
  273         // underway).
  274 
  275         // build up a subst that shifts all of the parameters over
  276         // by one and substitute in a new type param for self
  277 
  278         let tcx = ccx.tcx;
  279 
  280         let dummy_defid = ast::def_id {crate: 0, node: 0};
  281 
  282         // Represents [A',B',C']
  283         let num_trait_bounds = trait_ty_generics.type_param_defs.len();
  284         let non_shifted_trait_tps = do vec::from_fn(num_trait_bounds) |i| {
  285             ty::mk_param(tcx, i, trait_ty_generics.type_param_defs[i].def_id)
  286         };
  287 
  288         // Represents [D']
  289         let self_param = ty::mk_param(tcx, num_trait_bounds,
  290                                       dummy_defid);
  291 
  292         // Represents [E',F',G']
  293         let num_method_bounds = m.generics.type_param_defs.len();
  294         let shifted_method_tps = do vec::from_fn(num_method_bounds) |i| {
  295             ty::mk_param(tcx, i + num_trait_bounds + 1,
  296                          m.generics.type_param_defs[i].def_id)
  297         };
  298 
  299         // build up the substitution from
  300         //     A,B,C => A',B',C'
  301         //     Self => D'
  302         //     D,E,F => E',F',G'
  303         let substs = substs {
  304             regions: ty::NonerasedRegions(opt_vec::Empty),
  305             self_ty: Some(self_param),
  306             tps: non_shifted_trait_tps + shifted_method_tps
  307         };
  308 
  309         // create the type of `foo`, applying the substitution above
  310         let ty = ty::subst(tcx,
  311                            &substs,
  312                            ty::mk_bare_fn(tcx, m.fty.clone()));
  313 
  314         // create the type parameter definitions for `foo`, applying
  315         // the substitution to any traits that appear in their bounds.
  316 
  317         // add in the type parameters from the trait
  318         let mut new_type_param_defs = ~[];
  319         let substd_type_param_defs =
  320             trait_ty_generics.type_param_defs.subst(tcx, &substs);
  321         new_type_param_defs.push_all(*substd_type_param_defs);
  322 
  323         // add in the "self" type parameter
  324         let self_trait_def = get_trait_def(ccx, local_def(trait_id));
  325         let self_trait_ref = self_trait_def.trait_ref.subst(tcx, &substs);
  326         new_type_param_defs.push(ty::TypeParameterDef {
  327             ident: special_idents::self_,
  328             def_id: dummy_defid,
  329             bounds: @ty::ParamBounds {
  330                 builtin_bounds: ty::EmptyBuiltinBounds(),
  331                 trait_bounds: ~[self_trait_ref]
  332             }
  333         });
  334 
  335         // add in the type parameters from the method
  336         let substd_type_param_defs = m.generics.type_param_defs.subst(tcx, &substs);
  337         new_type_param_defs.push_all(*substd_type_param_defs);
  338 
  339         debug!("static method %s type_param_defs=%s substs=%s",
  340                m.def_id.repr(tcx),
  341                new_type_param_defs.repr(tcx),
  342                substs.repr(tcx));
  343 
  344         tcx.tcache.insert(m.def_id,
  345                           ty_param_bounds_and_ty {
  346                               generics: ty::Generics {
  347                                   type_param_defs: @new_type_param_defs,
  348                                   region_param: trait_ty_generics.region_param
  349                               },
  350                               ty: ty
  351                           });
  352     }

  354     fn ty_method_of_trait_method(this: &CrateCtxt,
  355                                  trait_id: ast::NodeId,
  356                                  trait_rp: Option<ty::region_variance>,
  357                                  trait_generics: &ast::Generics,

  378             ast::public,

  380             local_def(trait_id),
  381             None
  382         )

  386 pub fn ensure_supertraits(ccx: &CrateCtxt,
  387                           id: ast::NodeId,
  388                           sp: codemap::span,
  389                           rp: Option<ty::region_variance>,

  391                           generics: &ast::Generics)

  394     if tcx.supertraits.contains_key(&local_def(id)) { return; }

  397     let mut ty_trait_refs: ~[@ty::TraitRef] = ~[];
  398     for ast_trait_ref in ast_trait_refs.iter() {
  399         let trait_ref = instantiate_trait_ref(ccx, ast_trait_ref, rp,

  403         if ty_trait_refs.iter().any(|other_trait| other_trait.def_id == trait_ref.def_id) {

  421  * - impl_tps: the type params declared on the impl itself (not the method!)
  422  * - cm: info about the method we are checking
  423  * - trait_m: the method in the trait
  424  * - trait_substs: the substitutions used on the type of the trait
  425  * - self_ty: the self type of the impl
  426  */
  427 pub fn compare_impl_method(tcx: ty::ctxt,
  428                            impl_tps: uint,
  429                            cm: &ConvertedMethod,
  430                            trait_m: &ty::Method,
  431                            trait_substs: &ty::substs,
  432                            self_ty: ty::t) {
  433     debug!("compare_impl_method()");
  434     let infcx = infer::new_infer_ctxt(tcx);
  435 
  436     let impl_m = &cm.mty;
  437 
  438     // Try to give more informative error messages about self typing
  439     // mismatches.  Note that any mismatch will also be detected
  440     // below, where we construct a canonical function type that
  441     // includes the self parameter as a normal parameter.  It's just
  442     // that the error messages you get out of this code are a bit more
  443     // inscrutable, particularly for cases where one method has no
  444     // self.
  445     match (&trait_m.explicit_self, &impl_m.explicit_self) {
  446         (&ast::sty_static, &ast::sty_static) => {}
  447         (&ast::sty_static, _) => {

  449                 cm.span,
  450                 fmt!("method `%s` has a `%s` declaration in the impl, \
  451                       but not in the trait",
  452                      tcx.sess.str_of(trait_m.ident),
  453                      explicit_self_to_str(&impl_m.explicit_self, tcx.sess.intr())));
  454             return;
  455         }
  456         (_, &ast::sty_static) => {
  457             tcx.sess.span_err(
  458                 cm.span,
  459                 fmt!("method `%s` has a `%s` declaration in the trait, \
  460                       but not in the impl",
  461                      tcx.sess.str_of(trait_m.ident),
  462                      explicit_self_to_str(&trait_m.explicit_self, tcx.sess.intr())));
  463             return;
  464         }
  465         _ => {
  466             // Let the type checker catch other errors below
  467         }
  468     }
  469 
  470     let num_impl_m_type_params = impl_m.generics.type_param_defs.len();

  472     if num_impl_m_type_params != num_trait_m_type_params {
  473         tcx.sess.span_err(
  474             cm.span,
  475             fmt!("method `%s` has %u type %s, but its trait \
  476                   declaration has %u type %s",
  477                  tcx.sess.str_of(trait_m.ident),
  478                  num_impl_m_type_params,
  479                  pluralize(num_impl_m_type_params, ~"parameter"),
  480                  num_trait_m_type_params,
  481                  pluralize(num_trait_m_type_params, ~"parameter")));
  482         return;
  483     }
  484 
  485     if impl_m.fty.sig.inputs.len() != trait_m.fty.sig.inputs.len() {
  486         tcx.sess.span_err(
  487             cm.span,
  488             fmt!("method `%s` has %u parameter%s \
  489                   but the trait has %u",
  490                  tcx.sess.str_of(trait_m.ident),
  491                  impl_m.fty.sig.inputs.len(),
  492                  if impl_m.fty.sig.inputs.len() == 1 { "" } else { "s" },
  493                  trait_m.fty.sig.inputs.len()));
  494         return;
  495     }
  496 
  497     for (i, trait_param_def) in trait_m.generics.type_param_defs.iter().enumerate() {
  498         // For each of the corresponding impl ty param's bounds...
  499         let impl_param_def = &impl_m.generics.type_param_defs[i];
  500 
  501         // Check that the impl does not require any builtin-bounds
  502         // that the trait does not guarantee:
  503         let extra_bounds =

  505             trait_param_def.bounds.builtin_bounds;

  508                cm.span,
  509                fmt!("in method `%s`, \

  513                      in the trait declaration",

  516                     extra_bounds.user_string(tcx)));

  518         }

  528                 fmt!("in method `%s`, \

  535                                ~"bound"),
  536                      trait_param_def.bounds.trait_bounds.len(),
  537                      pluralize(trait_param_def.bounds.trait_bounds.len(),
  538                                ~"bound")));
  539             return;
  540         }
  541     }
  542 
  543     // Replace any references to the self region in the self type with
  544     // a free region.  So, for example, if the impl type is
  545     // "&'self str", then this would replace the self type with a free
  546     // region `self`.
  547     let dummy_self_r = ty::re_free(ty::FreeRegion {scope_id: cm.body_id,
  548                                                    bound_region: ty::br_self});

  551     // We are going to create a synthetic fn type that includes
  552     // both the method's self argument and its normal arguments.

  554     // into a function `fn(self: &T, a: uint)`.
  555     let mut trait_fn_args = ~[];

  559     // represent the self argument (unless this is a static method).
  560     // This argument will have the *transformed* self type.
  561     for &t in trait_m.transformed_self_ty.iter() {
  562         trait_fn_args.push(t);
  563     }
  564     for &t in impl_m.transformed_self_ty.iter() {
  565         impl_fn_args.push(t);
  566     }
  567 
  568     // Add in the normal arguments.
  569     trait_fn_args.push_all(trait_m.fty.sig.inputs);
  570     impl_fn_args.push_all(impl_m.fty.sig.inputs);
  571 
  572     // Create a bare fn type for trait/impl that includes self argument
  573     let trait_fty =
  574         ty::mk_bare_fn(tcx,
  575                        ty::BareFnTy {
  576                             purity: trait_m.fty.purity,
  577                             abis: trait_m.fty.abis,
  578                             sig: ty::FnSig {
  579                                 bound_lifetime_names:
  580                                     trait_m.fty
  581                                            .sig
  582                                            .bound_lifetime_names

  584                                 inputs: trait_fn_args,
  585                                 output: trait_m.fty.sig.output
  586                             }
  587                         });
  588     let impl_fty =
  589         ty::mk_bare_fn(tcx,
  590                        ty::BareFnTy {
  591                             purity: impl_m.fty.purity,
  592                             abis: impl_m.fty.abis,
  593                             sig: ty::FnSig {
  594                                 bound_lifetime_names:
  595                                     impl_m.fty
  596                                           .sig
  597                                           .bound_lifetime_names
  598                                           .clone(),
  599                                     inputs: impl_fn_args,
  600                                     output: impl_m.fty.sig.output
  601                             }
  602                         });
  603 
  604     // Perform substitutions so that the trait/impl methods are expressed
  605     // in terms of the same set of type/region parameters:
  606     // - replace trait type parameters with those from `trait_substs`,
  607     //   except with any reference to bound self replaced with `dummy_self_r`
  608     // - replace method parameters on the trait with fresh, dummy parameters
  609     //   that correspond to the parameters we will find on the impl
  610     // - replace self region with a fresh, dummy region
  611     let impl_fty = {
  612         debug!("impl_fty (pre-subst): %s", ppaux::ty_to_str(tcx, impl_fty));
  613         replace_bound_self(tcx, impl_fty, dummy_self_r)
  614     };
  615     debug!("impl_fty (post-subst): %s", ppaux::ty_to_str(tcx, impl_fty));
  616     let trait_fty = {
  617         let num_trait_m_type_params = trait_m.generics.type_param_defs.len();
  618         let dummy_tps = do vec::from_fn(num_trait_m_type_params) |i| {
  619             ty::mk_param(tcx, i + impl_tps,
  620                          impl_m.generics.type_param_defs[i].def_id)
  621         };
  622         let trait_tps = trait_substs.tps.map(
  623             |t| replace_bound_self(tcx, *t, dummy_self_r));
  624         let substs = substs {
  625             regions: ty::NonerasedRegions(opt_vec::with(dummy_self_r)),
  626             self_ty: Some(self_ty),
  627             tps: vec::append(trait_tps, dummy_tps)
  628         };
  629         debug!("trait_fty (pre-subst): %s substs=%s",
  630                trait_fty.repr(tcx), substs.repr(tcx));
  631         ty::subst(tcx, &substs, trait_fty)
  632     };
  633     debug!("trait_fty (post-subst): %s", trait_fty.repr(tcx));
  634 
  635     match infer::mk_subty(infcx, false, infer::MethodCompatCheck(cm.span),
  636                           impl_fty, trait_fty) {

  638         result::Err(ref terr) => {

  640                 cm.span,
  641                 fmt!("method `%s` has an incompatible type: %s",
  642                      tcx.sess.str_of(trait_m.ident),
  643                      ty::type_err_to_str(tcx, terr)));
  644             ty::note_and_explain_type_err(tcx, terr);
  645         }
  646     }
  647     return;

  649     // Replaces bound references to the self region with `with_r`.
  650     fn replace_bound_self(tcx: ty::ctxt, ty: ty::t,
  651                           with_r: ty::Region) -> ty::t {

  653             if r == ty::re_bound(ty::br_self) {with_r} else {r}
  654         }
  655     }

 1186     fn compute_bounds(

 1198          * traits, or one of the two built-in traits (formerly known

 1203             builtin_bounds: ty::EmptyBuiltinBounds(),

 1210                     let trait_ref = instantiate_trait_ref(ccx, b, rp, generics, ty);
 1211                     if !astconv::try_add_builtin_trait(

   11  /*
   12  
   13  # Collect phase
   14  
   15  The collect phase of type check has the job of visiting all items,
   16  determining their type, and writing that type into the `tcx.tcache`
   17  table.  Despite its name, this table does not really operate as a
   18  *cache*, at least not for the types of items defined within the
   19  current crate: we assume that after the collect phase, the types of
   20  all local items will be present in the table.
   21  
   22  Unlike most of the types that are present in Rust, the types computed
   23  for each item are in fact polytypes.  In "layman's terms", this means
   24  that they are generic types that may have type parameters (more
   25  mathematically phrased, they are universally quantified over a set of
   26  type parameters).  Polytypes are represented by an instance of
   27  `ty::ty_param_bounds_and_ty`.  This combines the core type along with
   28  a list of the bounds for each parameter.  Type parameters themselves
   29  are represented as `ty_param()` instances.
   30  
   31  */

  149  pub fn get_enum_variant_types(ccx: &CrateCtxt,
  150                                enum_ty: ty::t,
  151                                variants: &[ast::variant],
  152                                generics: &ast::Generics,
  153                                rp: Option<ty::region_variance>) {
  154      let tcx = ccx.tcx;
  155  
  156      // Create a set of parameter types shared among all the variants.
  157      for variant in variants.iter() {
  158          let region_parameterization =
  159              RegionParameterization::from_variance_and_generics(rp, generics);
  160  
  161          // Nullary enum constructors get turned into constants; n-ary enum
  162          // constructors get turned into functions.
  163          let result_ty;
  164          match variant.node.kind {
  165              ast::tuple_variant_kind(ref args) if args.len() > 0 => {
  166                  let rs = TypeRscope(region_parameterization);
  167                  let input_tys = args.map(|va| ccx.to_ty(&rs, &va.ty));
  168                  result_ty = Some(ty::mk_ctor_fn(tcx, input_tys, enum_ty));
  169              }
  170  
  171              ast::tuple_variant_kind(_) => {
  172                  result_ty = Some(enum_ty);
  173              }
  174  
  175              ast::struct_variant_kind(struct_def) => {
  176                  let tpt = ty_param_bounds_and_ty {
  177                      generics: ty_generics(ccx, rp, generics, 0),
  178                      ty: enum_ty
  179                  };
  180  
  181                  convert_struct(ccx,
  182                                 rp,
  183                                 struct_def,
  184                                 generics,
  185                                 tpt,
  186                                 variant.node.id);
  187  
  188                  let input_tys = struct_def.fields.map(
  189                      |f| ty::node_id_to_type(ccx.tcx, f.node.id));
  190                  result_ty = Some(ty::mk_ctor_fn(tcx, input_tys, enum_ty));
  191              }
  192          };
  193  
  194          match result_ty {
  195              None => {}
  196              Some(result_ty) => {
  197                  let tpt = ty_param_bounds_and_ty {
  198                      generics: ty_generics(ccx, rp, generics, 0),
  199                      ty: result_ty
  200                  };
  201                  tcx.tcache.insert(local_def(variant.node.id), tpt);
  202                  write_ty_to_tcx(tcx, variant.node.id, result_ty);
  203              }
  204          }
  205      }
  206  }

  208  pub fn ensure_trait_methods(ccx: &CrateCtxt,
  209                              trait_id: ast::NodeId)
  210  {
  211      let tcx = ccx.tcx;
  212      let region_paramd = tcx.region_paramd_items.find(&trait_id).map_move(|x| *x);
  213      match tcx.items.get_copy(&trait_id) {
  214          ast_map::node_item(@ast::item {
  215              node: ast::item_trait(ref generics, _, ref ms),
  216              _
  217          }, _) => {
  218              let trait_ty_generics = ty_generics(ccx, region_paramd, generics, 0);
  219  
  220              // For each method, construct a suitable ty::Method and
  221              // store it into the `tcx.methods` table:
  222              for m in ms.iter() {
  223                  let ty_method = @match m {
  224                      &ast::required(ref m) => {
  225                          ty_method_of_trait_method(
  226                              ccx, trait_id, region_paramd, generics,
  227                              &m.id, &m.ident, &m.explicit_self,
  228                              &m.generics, &m.purity, &m.decl)
  229                      }
  230  
  231                      &ast::provided(ref m) => {
  232                          ty_method_of_trait_method(
  233                              ccx, trait_id, region_paramd, generics,
  234                              &m.id, &m.ident, &m.explicit_self,
  235                              &m.generics, &m.purity, &m.decl)
  236                      }
  237                  };
  238  
  239                  if ty_method.explicit_self == ast::sty_static {
  240                      make_static_method_ty(ccx, trait_id, ty_method,
  241                                            &trait_ty_generics);
  242                  }
  243  
  244                  tcx.methods.insert(ty_method.def_id, ty_method);
  245              }
  246  
  247              // Add an entry mapping
  248              let method_def_ids = @ms.map(|m| {
  249                  match m {
  250                      &ast::required(ref ty_method) => local_def(ty_method.id),
  251                      &ast::provided(ref method) => local_def(method.id)
  252                  }
  253              });
  254  
  255              let trait_def_id = local_def(trait_id);
  256              tcx.trait_method_def_ids.insert(trait_def_id, method_def_ids);
  257          }
  258          _ => { /* Ignore things that aren't traits */ }
  259      }

  261      fn make_static_method_ty(ccx: &CrateCtxt,
  262                               trait_id: ast::NodeId,
  263                               m: &ty::Method,
  264                               trait_ty_generics: &ty::Generics) {
  265          // If declaration is
  266          //
  267          //     trait<A,B,C> {
  268          //        fn foo<D,E,F>(...) -> Self;
  269          //     }
  270          //
  271          // and we will create a function like
  272          //
  273          //     fn foo<A',B',C',D',E',F',G'>(...) -> D' {}
  274          //
  275          // Note that `Self` is replaced with an explicit type
  276          // parameter D' that is sandwiched in between the trait params
  277          // and the method params, and thus the indices of the method
  278          // type parameters are offset by 1 (that is, the method
  279          // parameters are mapped from D, E, F to E', F', and G').  The
  280          // choice of this ordering is somewhat arbitrary.
  281          //
  282          // Also, this system is rather a hack that should be replaced
  283          // with a more uniform treatment of Self (which is partly
  284          // underway).
  285  
  286          // build up a subst that shifts all of the parameters over
  287          // by one and substitute in a new type param for self
  288  
  289          let tcx = ccx.tcx;
  290  
  291          let dummy_defid = ast::DefId {crate: 0, node: 0};
  292  
  293          // Represents [A',B',C']
  294          let num_trait_bounds = trait_ty_generics.type_param_defs.len();
  295          let non_shifted_trait_tps = do vec::from_fn(num_trait_bounds) |i| {
  296              ty::mk_param(tcx, i, trait_ty_generics.type_param_defs[i].def_id)
  297          };
  298  
  299          // Represents [D']
  300          let self_param = ty::mk_param(tcx, num_trait_bounds,
  301                                        dummy_defid);
  302  
  303          // Represents [E',F',G']
  304          let num_method_bounds = m.generics.type_param_defs.len();
  305          let shifted_method_tps = do vec::from_fn(num_method_bounds) |i| {
  306              ty::mk_param(tcx, i + num_trait_bounds + 1,
  307                           m.generics.type_param_defs[i].def_id)
  308          };
  309  
  310          // build up the substitution from
  311          //     A,B,C => A',B',C'
  312          //     Self => D'
  313          //     D,E,F => E',F',G'
  314          let substs = substs {
  315              regions: ty::NonerasedRegions(opt_vec::Empty),
  316              self_ty: Some(self_param),
  317              tps: non_shifted_trait_tps + shifted_method_tps
  318          };
  319  
  320          // create the type of `foo`, applying the substitution above
  321          let ty = ty::subst(tcx,
  322                             &substs,
  323                             ty::mk_bare_fn(tcx, m.fty.clone()));
  324  
  325          // create the type parameter definitions for `foo`, applying
  326          // the substitution to any traits that appear in their bounds.
  327  
  328          // add in the type parameters from the trait
  329          let mut new_type_param_defs = ~[];
  330          let substd_type_param_defs =
  331              trait_ty_generics.type_param_defs.subst(tcx, &substs);
  332          new_type_param_defs.push_all(*substd_type_param_defs);
  333  
  334          // add in the "self" type parameter
  335          let self_trait_def = get_trait_def(ccx, local_def(trait_id));
  336          let self_trait_ref = self_trait_def.trait_ref.subst(tcx, &substs);
  337          new_type_param_defs.push(ty::TypeParameterDef {
  338              ident: special_idents::self_,
  339              def_id: dummy_defid,
  340              bounds: @ty::ParamBounds {
  341                  builtin_bounds: ty::EmptyBuiltinBounds(),
  342                  trait_bounds: ~[self_trait_ref]
  343              }
  344          });
  345  
  346          // add in the type parameters from the method
  347          let substd_type_param_defs = m.generics.type_param_defs.subst(tcx, &substs);
  348          new_type_param_defs.push_all(*substd_type_param_defs);
  349  
  350          debug!("static method %s type_param_defs=%s ty=%s, substs=%s",
  351                 m.def_id.repr(tcx),
  352                 new_type_param_defs.repr(tcx),
  353                 ty.repr(tcx),
  354                 substs.repr(tcx));
  355  
  356          tcx.tcache.insert(m.def_id,
  357                            ty_param_bounds_and_ty {
  358                                generics: ty::Generics {
  359                                    type_param_defs: @new_type_param_defs,
  360                                    region_param: trait_ty_generics.region_param
  361                                },
  362                                ty: ty
  363                            });
  364      }

  366      fn ty_method_of_trait_method(this: &CrateCtxt,
  367                                   trait_id: ast::NodeId,
  368                                   trait_rp: Option<ty::region_variance>,
  369                                   trait_generics: &ast::Generics,
  370                                   m_id: &ast::NodeId,
  371                                   m_ident: &ast::Ident,
  372                                   m_explicit_self: &ast::explicit_self,
  373                                   m_generics: &ast::Generics,
  374                                   m_purity: &ast::purity,
  375                                   m_decl: &ast::fn_decl) -> ty::Method
  376      {
  377          let trait_self_ty = ty::mk_self(this.tcx, local_def(trait_id));
  378          let rscope = MethodRscope::new(m_explicit_self.node, trait_rp, trait_generics);
  379          let (transformed_self_ty, fty) =
  380              astconv::ty_of_method(this, &rscope, *m_purity, &m_generics.lifetimes,
  381                                    trait_self_ty, *m_explicit_self, m_decl);
  382          let num_trait_type_params = trait_generics.ty_params.len();
  383          ty::Method::new(
  384              *m_ident,
  385              ty_generics(this, None, m_generics, num_trait_type_params),
  386              transformed_self_ty,
  387              fty,
  388              m_explicit_self.node,
  389              // assume public, because this is only invoked on trait methods
  390              ast::public,
  391              local_def(*m_id),
  392              TraitContainer(local_def(trait_id)),
  393              None
  394          )
  395      }

  406  
  407      // Called only the first time trait_def_of_item is called.
  408      // Supertraits are ensured at the same time.
  409      assert!(!tcx.supertraits.contains_key(&local_def(id)));
  410  
  411      let self_ty = ty::mk_self(ccx.tcx, local_def(id));
  412      let mut ty_trait_refs: ~[@ty::TraitRef] = ~[];
  413      let mut bounds = ty::EmptyBuiltinBounds();
  414      for ast_trait_ref in ast_trait_refs.iter() {
  415          let trait_def_id = ty::trait_ref_to_def_id(ccx.tcx, ast_trait_ref);
  416          // FIXME(#8559): Need to instantiate the trait_ref whether or not it's a
  417          // builtin trait, so that the trait's node id appears in the tcx trait_ref
  418          // map. This is only needed for metadata; see the similar fixme in encoder.rs.
  419          let trait_ref = instantiate_trait_ref(ccx, ast_trait_ref, rp,
  420                                                generics, self_ty);
  421          if !ty::try_add_builtin_trait(ccx.tcx, trait_def_id, &mut bounds) {
  422  
  423              // FIXME(#5527) Could have same trait multiple times
  424              if ty_trait_refs.iter().any(|other_trait| other_trait.def_id == trait_ref.def_id) {
  425                  // This means a trait inherited from the same supertrait more
  426                  // than once.
  427                  tcx.sess.span_err(sp, "Duplicate supertrait in trait declaration");
  428                  break;
  429              } else {
  430                  ty_trait_refs.push(trait_ref);
  431              }
  432          }
  433      }
  434      tcx.supertraits.insert(local_def(id), @ty_trait_refs);
  435      bounds
  436  }
  437  
  438  /**
  439   * Checks that a method from an impl/class conforms to the signature of
  440   * the same method as declared in the trait.
  441   *
  442   * # Parameters
  443   *
  444   * - impl_tps: the type params declared on the impl itself (not the method!)
  445   * - cm: info about the method we are checking
  446   * - trait_m: the method in the trait
  447   * - trait_substs: the substitutions used on the type of the trait
  448   * - self_ty: the self type of the impl
  449   */
  450  pub fn compare_impl_method(tcx: ty::ctxt,
  451                             impl_tps: uint,
  452                             cm: &ConvertedMethod,
  453                             trait_m: &ty::Method,
  454                             trait_substs: &ty::substs,
  455                             self_ty: ty::t) {
  456      debug!("compare_impl_method()");
  457      let infcx = infer::new_infer_ctxt(tcx);
  458  
  459      let impl_m = &cm.mty;
  460  
  461      // Try to give more informative error messages about self typing
  462      // mismatches.  Note that any mismatch will also be detected
  463      // below, where we construct a canonical function type that
  464      // includes the self parameter as a normal parameter.  It's just
  465      // that the error messages you get out of this code are a bit more
  466      // inscrutable, particularly for cases where one method has no
  467      // self.
  468      match (&trait_m.explicit_self, &impl_m.explicit_self) {
  469          (&ast::sty_static, &ast::sty_static) => {}
  470          (&ast::sty_static, _) => {
  471              tcx.sess.span_err(
  472                  cm.span,
  473                  fmt!("method `%s` has a `%s` declaration in the impl, \
  474                        but not in the trait",
  475                       tcx.sess.str_of(trait_m.ident),
  476                       explicit_self_to_str(&impl_m.explicit_self, tcx.sess.intr())));
  477              return;
  478          }
  479          (_, &ast::sty_static) => {
  480              tcx.sess.span_err(
  481                  cm.span,
  482                  fmt!("method `%s` has a `%s` declaration in the trait, \
  483                        but not in the impl",
  484                       tcx.sess.str_of(trait_m.ident),
  485                       explicit_self_to_str(&trait_m.explicit_self, tcx.sess.intr())));
  486              return;
  487          }
  488          _ => {
  489              // Let the type checker catch other errors below
  490          }
  491      }
  492  
  493      let num_impl_m_type_params = impl_m.generics.type_param_defs.len();
  494      let num_trait_m_type_params = trait_m.generics.type_param_defs.len();
  495      if num_impl_m_type_params != num_trait_m_type_params {
  496          tcx.sess.span_err(
  497              cm.span,
  498              fmt!("method `%s` has %u type %s, but its trait \
  499                    declaration has %u type %s",
  500                   tcx.sess.str_of(trait_m.ident),
  501                   num_impl_m_type_params,
  502                   pluralize(num_impl_m_type_params, ~"parameter"),
  503                   num_trait_m_type_params,
  504                   pluralize(num_trait_m_type_params, ~"parameter")));
  505          return;
  506      }
  507  
  508      if impl_m.fty.sig.inputs.len() != trait_m.fty.sig.inputs.len() {
  509          tcx.sess.span_err(
  510              cm.span,
  511              fmt!("method `%s` has %u parameter%s \
  512                    but the trait has %u",
  513                   tcx.sess.str_of(trait_m.ident),
  514                   impl_m.fty.sig.inputs.len(),
  515                   if impl_m.fty.sig.inputs.len() == 1 { "" } else { "s" },
  516                   trait_m.fty.sig.inputs.len()));
  517          return;
  518      }
  519  
  520      for (i, trait_param_def) in trait_m.generics.type_param_defs.iter().enumerate() {
  521          // For each of the corresponding impl ty param's bounds...
  522          let impl_param_def = &impl_m.generics.type_param_defs[i];
  523  
  524          // Check that the impl does not require any builtin-bounds
  525          // that the trait does not guarantee:
  526          let extra_bounds =
  527              impl_param_def.bounds.builtin_bounds -
  528              trait_param_def.bounds.builtin_bounds;
  529          if !extra_bounds.is_empty() {
  530             tcx.sess.span_err(
  531                 cm.span,
  532                 fmt!("in method `%s`, \
  533                       type parameter %u requires `%s`, \
  534                       which is not required by \
  535                       the corresponding type parameter \
  536                       in the trait declaration",
  537                      tcx.sess.str_of(trait_m.ident),
  538                      i,
  539                      extra_bounds.user_string(tcx)));
  540             return;
  541          }
  542  
  543          // FIXME(#2687)---we should be checking that the bounds of the
  544          // trait imply the bounds of the subtype, but it appears we
  545          // are...not checking this.
  546          if impl_param_def.bounds.trait_bounds.len() !=
  547              trait_param_def.bounds.trait_bounds.len()
  548          {
  549              tcx.sess.span_err(
  550                  cm.span,
  551                  fmt!("in method `%s`, \
  552                        type parameter %u has %u trait %s, but the \
  553                        corresponding type parameter in \
  554                        the trait declaration has %u trait %s",
  555                       tcx.sess.str_of(trait_m.ident),
  556                       i, impl_param_def.bounds.trait_bounds.len(),
  557                       pluralize(impl_param_def.bounds.trait_bounds.len(),
  558                                 ~"bound"),
  559                       trait_param_def.bounds.trait_bounds.len(),
  560                       pluralize(trait_param_def.bounds.trait_bounds.len(),
  561                                 ~"bound")));
  562              return;
  563          }
  564      }
  565  
  566      // Replace any references to the self region in the self type with
  567      // a free region.  So, for example, if the impl type is
  568      // "&'self str", then this would replace the self type with a free
  569      // region `self`.
  570      let dummy_self_r = ty::re_free(ty::FreeRegion {scope_id: cm.body_id,
  571                                                     bound_region: ty::br_self});
  572      let self_ty = replace_bound_self(tcx, self_ty, dummy_self_r);
  573  
  574      // We are going to create a synthetic fn type that includes
  575      // both the method's self argument and its normal arguments.
  576      // So a method like `fn(&self, a: uint)` would be converted
  577      // into a function `fn(self: &T, a: uint)`.
  578      let mut trait_fn_args = ~[];
  579      let mut impl_fn_args = ~[];
  580  
  581      // For both the trait and the impl, create an argument to
  582      // represent the self argument (unless this is a static method).
  583      // This argument will have the *transformed* self type.
  584      for &t in trait_m.transformed_self_ty.iter() {
  585          trait_fn_args.push(t);
  586      }
  587      for &t in impl_m.transformed_self_ty.iter() {
  588          impl_fn_args.push(t);
  589      }
  590  
  591      // Add in the normal arguments.
  592      trait_fn_args.push_all(trait_m.fty.sig.inputs);
  593      impl_fn_args.push_all(impl_m.fty.sig.inputs);
  594  
  595      // Create a bare fn type for trait/impl that includes self argument
  596      let trait_fty =
  597          ty::mk_bare_fn(tcx,
  598                         ty::BareFnTy {
  599                              purity: trait_m.fty.purity,
  600                              abis: trait_m.fty.abis,
  601                              sig: ty::FnSig {
  602                                  bound_lifetime_names:
  603                                      trait_m.fty
  604                                             .sig
  605                                             .bound_lifetime_names
  606                                             .clone(),
  607                                  inputs: trait_fn_args,
  608                                  output: trait_m.fty.sig.output
  609                              }
  610                          });
  611      let impl_fty =
  612          ty::mk_bare_fn(tcx,
  613                         ty::BareFnTy {
  614                              purity: impl_m.fty.purity,
  615                              abis: impl_m.fty.abis,
  616                              sig: ty::FnSig {
  617                                  bound_lifetime_names:
  618                                      impl_m.fty
  619                                            .sig
  620                                            .bound_lifetime_names
  621                                            .clone(),
  622                                      inputs: impl_fn_args,
  623                                      output: impl_m.fty.sig.output
  624                              }
  625                          });
  626  
  627      // Perform substitutions so that the trait/impl methods are expressed
  628      // in terms of the same set of type/region parameters:
  629      // - replace trait type parameters with those from `trait_substs`,
  630      //   except with any reference to bound self replaced with `dummy_self_r`
  631      // - replace method parameters on the trait with fresh, dummy parameters
  632      //   that correspond to the parameters we will find on the impl
  633      // - replace self region with a fresh, dummy region
  634      let impl_fty = {
  635          debug!("impl_fty (pre-subst): %s", ppaux::ty_to_str(tcx, impl_fty));
  636          replace_bound_self(tcx, impl_fty, dummy_self_r)
  637      };
  638      debug!("impl_fty (post-subst): %s", ppaux::ty_to_str(tcx, impl_fty));
  639      let trait_fty = {
  640          let num_trait_m_type_params = trait_m.generics.type_param_defs.len();
  641          let dummy_tps = do vec::from_fn(num_trait_m_type_params) |i| {
  642              ty::mk_param(tcx, i + impl_tps,
  643                           impl_m.generics.type_param_defs[i].def_id)
  644          };
  645          let trait_tps = trait_substs.tps.map(
  646              |t| replace_bound_self(tcx, *t, dummy_self_r));
  647          let substs = substs {
  648              regions: ty::NonerasedRegions(opt_vec::with(dummy_self_r)),
  649              self_ty: Some(self_ty),
  650              tps: vec::append(trait_tps, dummy_tps)
  651          };
  652          debug!("trait_fty (pre-subst): %s substs=%s",
  653                 trait_fty.repr(tcx), substs.repr(tcx));
  654          ty::subst(tcx, &substs, trait_fty)
  655      };
  656      debug!("trait_fty (post-subst): %s", trait_fty.repr(tcx));
  657  
  658      match infer::mk_subty(infcx, false, infer::MethodCompatCheck(cm.span),
  659                            impl_fty, trait_fty) {
  660          result::Ok(()) => {}
  661          result::Err(ref terr) => {
  662              tcx.sess.span_err(
  663                  cm.span,
  664                  fmt!("method `%s` has an incompatible type: %s",
  665                       tcx.sess.str_of(trait_m.ident),
  666                       ty::type_err_to_str(tcx, terr)));
  667              ty::note_and_explain_type_err(tcx, terr);
  668          }
  669      }
  670      return;
  671  
  672      // Replaces bound references to the self region with `with_r`.
  673      fn replace_bound_self(tcx: ty::ctxt, ty: ty::t,
  674                            with_r: ty::Region) -> ty::t {
  675          do ty::fold_regions(tcx, ty) |r, _in_fn| {
  676              if r == ty::re_bound(ty::br_self) {with_r} else {r}
  677          }
  678      }
  679  }

  681  pub fn check_methods_against_trait(ccx: &CrateCtxt,
  682                                     generics: &ast::Generics,
  683                                     rp: Option<ty::region_variance>,
  684                                     selfty: ty::t,
  685                                     a_trait_ty: &ast::trait_ref,
  686                                     impl_ms: &[ConvertedMethod])
  687  {
  688      let tcx = ccx.tcx;
  689      let trait_ref = instantiate_trait_ref(ccx, a_trait_ty, rp,
  690                                            generics, selfty);
  691  
  692      if trait_ref.def_id.crate == ast::LOCAL_CRATE {
  693          ensure_trait_methods(ccx, trait_ref.def_id.node);
  694      }
  695  
  696      // Check that each method we impl is a method on the trait
  697      // Trait methods we don't implement must be default methods, but if not
  698      // we'll catch it in coherence
  699      let trait_ms = ty::trait_methods(tcx, trait_ref.def_id);
  700      for impl_m in impl_ms.iter() {
  701          match trait_ms.iter().find(|trait_m| trait_m.ident.name == impl_m.mty.ident.name) {
  702              Some(trait_m) => {
  703                  let num_impl_tps = generics.ty_params.len();
  704                  compare_impl_method(
  705                      ccx.tcx, num_impl_tps, impl_m, *trait_m,
  706                      &trait_ref.substs, selfty);
  707              }
  708              None => {
  709                  // This method is not part of the trait
  710                  tcx.sess.span_err(
  711                      impl_m.span,
  712                      fmt!("method `%s` is not a member of trait `%s`",
  713                           tcx.sess.str_of(impl_m.mty.ident),
  714                           path_to_str(&a_trait_ty.path, tcx.sess.intr())));
  715              }
  716          }
  717      }

  747  pub fn convert_methods(ccx: &CrateCtxt,
  748                         container: MethodContainer,
  749                         ms: &[@ast::method],
  750                         untransformed_rcvr_ty: ty::t,
  751                         rcvr_ty_generics: &ty::Generics,
  752                         rcvr_ast_generics: &ast::Generics,
  753                         rcvr_visibility: ast::visibility)
  754                      -> ~[ConvertedMethod]
  755  {
  756      let tcx = ccx.tcx;
  757      return ms.iter().map(|m| {
  758          let num_rcvr_ty_params = rcvr_ty_generics.type_param_defs.len();
  759          let m_ty_generics =
  760              ty_generics(ccx, rcvr_ty_generics.region_param, &m.generics,
  761                          num_rcvr_ty_params);
  762          let mty = @ty_of_method(ccx,
  763                                  container,
  764                                  *m,
  765                                  rcvr_ty_generics.region_param,
  766                                  untransformed_rcvr_ty,
  767                                  rcvr_ast_generics,
  768                                  rcvr_visibility,
  769                                  &m.generics);
  770          let fty = ty::mk_bare_fn(tcx, mty.fty.clone());
  771          tcx.tcache.insert(
  772              local_def(m.id),
  773  
  774              // n.b.: the type of a method is parameterized by both
  775              // the tps on the receiver and those on the method itself
  776              ty_param_bounds_and_ty {
  777                  generics: ty::Generics {
  778                      type_param_defs: @vec::append(
  779                          (*rcvr_ty_generics.type_param_defs).clone(),
  780                          *m_ty_generics.type_param_defs),
  781                      region_param: rcvr_ty_generics.region_param
  782                  },

  951                        rp: Option<ty::region_variance>,

  977                      struct_def.fields.map(

  983                      generics: tpt.generics,
  984                      ty: ctor_fn_ty
  985                  });
  986              }

  988      }

   11  /*!
   12  
   13  The region check is a final pass that runs over the AST after we have
   14  inferred the type constraints but before we have actually finalized
   15  the types.  Its purpose is to embed some final region constraints.
   16  The reason that this is not done earlier is that sometimes we don't
   17  know whether a given type will be a region pointer or not until this
   18  phase.
   19  
   20  In particular, we ensure that, if the type of an expression or
   21  variable is `&'r T`, then the expression or variable must occur within
   22  the region scope `r`.  Note that in some cases `r` may still be a
   23  region variable, so this gives us a chance to influence the value for
   24  `r` that we infer to ensure we choose a value large enough to enclose
   25  all uses.  There is a lengthy comment in visit_node() that explains
   26  this point a bit better.
   27  
   28  */

   57  fn encl_region_of_def(fcx: @mut FnCtxt, def: ast::Def) -> ty::Region {
   58      let tcx = fcx.tcx();
   59      match def {
   60          DefLocal(node_id, _) | DefArg(node_id, _) |
   61          DefSelf(node_id) | DefBinding(node_id, _) => {
   62              tcx.region_maps.encl_region(node_id)
   63          }
   64          DefUpvar(_, subdef, closure_id, body_id) => {
   65              match ty::ty_closure_sigil(fcx.node_ty(closure_id)) {
   66                  BorrowedSigil => encl_region_of_def(fcx, *subdef),
   67                  ManagedSigil | OwnedSigil => re_scope(body_id)
   68              }
   69          }
   70          _ => {
   71              tcx.sess.bug(fmt!("unexpected def in encl_region_of_def: %?",
   72                                def))
   73          }
   74      }
   75  }

   88      pub fn resolve_type(&mut self, unresolved_ty: ty::t) -> ty::t {
   89          /*!
   90           * Try to resolve the type for the given node, returning
   91           * t_err if an error results.  Note that we never care
   92           * about the details of the error, the same error will be
   93           * detected and reported in the writeback phase.
   94           *
   95           * Note one important point: we do not attempt to resolve
   96           * *region variables* here.  This is because regionck is
   97           * essentially adding constraints to those region variables
   98           * and so may yet influence how they are resolved.
   99           *
  100           * Consider this silly example:
  101           *
  102           *     fn borrow(x: &int) -> &int {x}
  103           *     fn foo(x: @int) -> int {  // block: B
  104           *         let b = borrow(x);    // region: <R0>
  105           *         *b
  106           *     }
  107           *
  108           * Here, the region of `b` will be `<R0>`.  `<R0>` is
  109           * constrainted to be some subregion of the block B and some
  110           * superregion of the call.  If we forced it now, we'd choose
  111           * the smaller region (the call).  But that would make the *b
  112           * illegal.  Since we don't resolve, the type of b will be
  113           * `&<R0>.int` and then `*b` will require that `<R0>` be
  114           * bigger than the let and the `*b` expression, so we will
  115           * effectively resolve `<R0>` to be the block B.
  116           */
  117          match resolve_type(self.fcx.infcx(), unresolved_ty,
  118                             resolve_and_force_all_but_regions) {
  119              Ok(t) => t,
  120              Err(_) => ty::mk_err()
  121          }
  122      }

  126          let t = self.fcx.node_ty(id);
  127          self.resolve_type(t)
  128      }

  130      /// Try to resolve the type for the given node.
  131      pub fn resolve_expr_type_adjusted(&mut self, expr: @ast::Expr) -> ty::t {
  132          let ty_unadjusted = self.resolve_node_type(expr.id);
  133          if ty::type_is_error(ty_unadjusted) || ty::type_is_bot(ty_unadjusted) {
  134              ty_unadjusted
  135          } else {
  136              let tcx = self.fcx.tcx();
  137              let adjustments = self.fcx.inh.adjustments;
  138              ty::adjust_ty(tcx, expr.span, ty_unadjusted,
  139                            adjustments.find_copy(&expr.id))
  140          }
  141      }
  142  }

  144  pub fn regionck_expr(fcx: @mut FnCtxt, e: @ast::Expr) {
  145      let mut rcx = Rcx { fcx: fcx, errors_reported: 0,
  146                           repeating_scope: e.id };
  147      let rcx = &mut rcx;
  148      if fcx.err_count_since_creation() == 0 {
  149          // regionck assumes typeck succeeded
  150          rcx.visit_expr(e, ());
  151      }
  152      fcx.infcx().resolve_regions();
  153  }

  212  fn constrain_bindings_in_pat(pat: @ast::Pat, rcx: &mut Rcx) {
  213      let tcx = rcx.fcx.tcx();
  214      debug!("regionck::visit_pat(pat=%s)", pat.repr(tcx));
  215      do pat_util::pat_bindings(tcx.def_map, pat) |_, id, span, _| {
  216          // If we have a variable that contains region'd data, that
  217          // data will be accessible from anywhere that the variable is
  218          // accessed. We must be wary of loops like this:
  219          //
  220          //     // from src/test/compile-fail/borrowck-lend-flow.rs
  221          //     let mut v = ~3, w = ~4;
  222          //     let mut x = &mut w;
  223          //     loop {
  224          //         **x += 1;   // (2)
  225          //         borrow(v);  //~ ERROR cannot borrow
  226          //         x = &mut v; // (1)
  227          //     }
  228          //
  229          // Typically, we try to determine the region of a borrow from
  230          // those points where it is dereferenced. In this case, one
  231          // might imagine that the lifetime of `x` need only be the
  232          // body of the loop. But of course this is incorrect because
  233          // the pointer that is created at point (1) is consumed at
  234          // point (2), meaning that it must be live across the loop
  235          // iteration. The easiest way to guarantee this is to require
  236          // that the lifetime of any regions that appear in a
  237          // variable's type enclose at least the variable's scope.
  238  
  239          let encl_region = tcx.region_maps.encl_region(id);
  240          constrain_regions_in_type_of_node(
  241              rcx, id, encl_region,
  242              infer::BindingTypeIsNotValidAtDecl(span));
  243      }
  244  }

  246  fn visit_expr(rcx: &mut Rcx, expr: @ast::Expr) {
  247      debug!("regionck::visit_expr(e=%s, repeating_scope=%?)",
  248             expr.repr(rcx.fcx.tcx()), rcx.repeating_scope);
  249  
  250      let has_method_map = rcx.fcx.inh.method_map.contains_key(&expr.id);
  251  
  252      // Record cleanup scopes, which are used by borrowck to decide the
  253      // maximum lifetime of a temporary rvalue.  These were derived by
  254      // examining where trans creates block scopes, not because this
  255      // reflects some principled decision around temporary lifetimes.
  256      // Ordinarily this would seem like something that should be setup
  257      // in region, but we need to know which uses of operators are
  258      // overloaded.  See #3511.
  259      let tcx = rcx.fcx.tcx();
  260      match expr.node {
  261          // You'd think that x += y where `+=` is overloaded would be a
  262          // cleanup scope. You'd be... kind of right. In fact the
  263          // handling of `+=` and friends in trans for overloaded
  264          // operators is a hopeless mess and I can't figure out how to
  265          // represent it. - ndm
  266          //
  267          // ast::expr_assign_op(*) |
  268  
  269          ast::ExprIndex(*) |
  270          ast::ExprBinary(*) |
  271          ast::ExprUnary(*) if has_method_map => {
  272              tcx.region_maps.record_cleanup_scope(expr.id);
  273          }
  274          ast::ExprBinary(_, ast::BiAnd, lhs, rhs) |
  275          ast::ExprBinary(_, ast::BiOr, lhs, rhs) => {
  276              tcx.region_maps.record_cleanup_scope(lhs.id);
  277              tcx.region_maps.record_cleanup_scope(rhs.id);
  278          }
  279          ast::ExprCall(*) |
  280          ast::ExprMethodCall(*) => {
  281              tcx.region_maps.record_cleanup_scope(expr.id);
  282          }
  283          ast::ExprMatch(_, ref arms) => {
  284              tcx.region_maps.record_cleanup_scope(expr.id);
  285              for arm in arms.iter() {
  286                  for guard in arm.guard.iter() {
  287                      tcx.region_maps.record_cleanup_scope(guard.id);
  288                  }
  289              }
  290          }
  291          ast::ExprLoop(ref body, _) => {
  292              tcx.region_maps.record_cleanup_scope(body.id);
  293          }
  294          ast::ExprWhile(cond, ref body) => {
  295              tcx.region_maps.record_cleanup_scope(cond.id);
  296              tcx.region_maps.record_cleanup_scope(body.id);
  297          }
  298          _ => {}
  299      }
  300  
  301      // Check any autoderefs or autorefs that appear.
  302      {
  303          let r = rcx.fcx.inh.adjustments.find(&expr.id);
  304          for &adjustment in r.iter() {
  305              debug!("adjustment=%?", adjustment);
  306              match *adjustment {
  307                  @ty::AutoDerefRef(
  308                      ty::AutoDerefRef {autoderefs: autoderefs, autoref: opt_autoref}) =>
  309                  {
  310                      let expr_ty = rcx.resolve_node_type(expr.id);
  311                      constrain_derefs(rcx, expr, autoderefs, expr_ty);
  312                      for autoref in opt_autoref.iter() {
  313                          guarantor::for_autoref(rcx, expr, autoderefs, autoref);
  314  
  315                          // Require that the resulting region encompasses
  316                          // the current node.
  317                          //
  318                          // FIXME(#6268) remove to support nested method calls
  319                          constrain_regions_in_type_of_node(
  320                              rcx, expr.id, ty::re_scope(expr.id),
  321                              infer::AutoBorrow(expr.span));
  322                      }
  323                  }
  324                  _ => {}
  325              }
  326          }
  327      }
  328  
  329      match expr.node {
  330          ast::ExprCall(callee, ref args, _) => {
  331              constrain_callee(rcx, callee.id, expr, callee);
  332              constrain_call(rcx, callee.id, expr, None, *args, false);
  333  
  334              visit::walk_expr(rcx, expr, ());
  335          }
  336  
  337          ast::ExprMethodCall(callee_id, arg0, _, _, ref args, _) => {

  339  
  340              visit::walk_expr(rcx, expr, ());
  341          }
  342  
  343          ast::ExprIndex(callee_id, lhs, rhs) |
  344          ast::ExprAssignOp(callee_id, _, lhs, rhs) |
  345          ast::ExprBinary(callee_id, _, lhs, rhs) if has_method_map => {
  346              // As `expr_method_call`, but the call is via an
  347              // overloaded op.  Note that we (sadly) currently use an
  348              // implicit "by ref" sort of passing style here.  This
  349              // should be converted to an adjustment!

  351

  354  
  355          ast::ExprUnary(callee_id, _, lhs) if has_method_map => {
  356              // As above.
  357              constrain_call(rcx, callee_id, expr, Some(lhs), [], true);
  358  
  359              visit::walk_expr(rcx, expr, ());
  360          }
  361  
  362          ast::ExprUnary(_, ast::UnDeref, base) => {
  363              // For *a, the lifetime of a must enclose the deref
  364              let base_ty = rcx.resolve_node_type(base.id);
  365              constrain_derefs(rcx, expr, 1, base_ty);
  366  
  367              visit::walk_expr(rcx, expr, ());
  368          }
  369  
  370          ast::ExprIndex(_, vec_expr, _) => {
  371              // For a[b], the lifetime of a must enclose the deref
  372              let vec_type = rcx.resolve_expr_type_adjusted(vec_expr);
  373              constrain_index(rcx, expr, vec_type);
  374  
  375              visit::walk_expr(rcx, expr, ());
  376          }
  377  
  378          ast::ExprCast(source, _) => {
  379              // Determine if we are casting `source` to an trait

  456  fn check_expr_fn_block(rcx: &mut Rcx,
  457                         expr: @ast::Expr) {
  458      let tcx = rcx.fcx.tcx();

  460          ast::ExprFnBlock(_, ref body) => {
  461              let function_type = rcx.resolve_node_type(expr.id);
  462              match ty::get(function_type).sty {

  464                      ty::ClosureTy {
  465                          sigil: ast::BorrowedSigil, region: region, _}) => {

  481                  }

  486              visit::walk_expr(rcx, expr, ());
  487              rcx.set_repeating_scope(repeating_scope);
  488          }
  489  
  490          _ => {
  491              tcx.sess.span_bug(
  492                  expr.span,
  493                  "Expected expr_fn_block");
  494          }
  495      }
  496  }

  523  fn constrain_call(rcx: &mut Rcx,
  524                    // might be expr_call, expr_method_call, or an overloaded
  525                    // operator
  526                    callee_id: ast::NodeId,
  527                    call_expr: @ast::Expr,
  528                    receiver: Option<@ast::Expr>,
  529                    arg_exprs: &[@ast::Expr],
  530                    implicitly_ref_args: bool)
  531  {
  532      //! Invoked on every call site (i.e., normal calls, method calls,
  533      //! and overloaded operators). Constrains the regions which appear
  534      //! in the type of the function. Also constrains the regions that
  535      //! appear in the arguments appropriately.
  536  
  537      let tcx = rcx.fcx.tcx();
  538      debug!("constrain_call(call_expr=%s, implicitly_ref_args=%?)",
  539             call_expr.repr(tcx), implicitly_ref_args);
  540      let callee_ty = rcx.resolve_node_type(callee_id);
  541      if ty::type_is_error(callee_ty) {
  542          // Bail, as function type is unknown
  543          return;
  544      }
  545      let fn_sig = ty::ty_fn_sig(callee_ty);
  546  
  547      // `callee_region` is the scope representing the time in which the
  548      // call occurs.
  549      //
  550      // FIXME(#6268) to support nested method calls, should be callee_id
  551      let callee_scope = call_expr.id;
  552      let callee_region = ty::re_scope(callee_scope);
  553  
  554      for &arg_expr in arg_exprs.iter() {
  555          // ensure that any regions appearing in the argument type are
  556          // valid for at least the lifetime of the function:
  557          constrain_regions_in_type_of_node(
  558              rcx, arg_expr.id, callee_region,
  559              infer::CallArg(arg_expr.span));
  560  
  561          // unfortunately, there are two means of taking implicit
  562          // references, and we need to propagate constraints as a
  563          // result. modes are going away and the "DerefArgs" code
  564          // should be ported to use adjustments
  565          if implicitly_ref_args {
  566              guarantor::for_by_ref(rcx, arg_expr, callee_scope);
  567          }
  568      }
  569  
  570      // as loop above, but for receiver
  571      for &r in receiver.iter() {
  572          constrain_regions_in_type_of_node(
  573              rcx, r.id, callee_region, infer::CallRcvr(r.span));
  574          if implicitly_ref_args {
  575              guarantor::for_by_ref(rcx, r, callee_scope);
  576          }
  577      }
  578  
  579      // constrain regions that may appear in the return type to be
  580      // valid for the function call:
  581      constrain_regions_in_type(
  582          rcx, callee_region, infer::CallReturn(call_expr.span),
  583          fn_sig.output);
  584  }

  702  fn constrain_regions_in_type(
  703      rcx: &mut Rcx,
  704      minimum_lifetime: ty::Region,
  705      origin: infer::SubregionOrigin,
  706      ty: ty::t) -> bool
  707  {
  708      /*!
  709       * Requires that any regions which appear in `ty` must be
  710       * superregions of `minimum_lifetime`.  Also enforces the constraint
  711       * that given a pointer type `&'r T`, T must not contain regions
  712       * that outlive 'r, as well as analogous constraints for other
  713       * lifetime'd types.
  714       *
  715       * This check prevents regions from being used outside of the block in
  716       * which they are valid.  Recall that regions represent blocks of
  717       * code or expressions: this requirement basically says "any place
  718       * that uses or may use a region R must be within the block of
  719       * code that R corresponds to."
  720       */
  721  
  722      let e = rcx.errors_reported;
  723      let tcx = rcx.fcx.ccx.tcx;
  724  
  725      debug!("constrain_regions_in_type(minimum_lifetime=%s, ty=%s)",
  726             region_to_str(tcx, "", false, minimum_lifetime),
  727             ty_to_str(tcx, ty));
  728  
  729      do relate_nested_regions(tcx, Some(minimum_lifetime), ty) |r_sub, r_sup| {
  730          debug!("relate(r_sub=%s, r_sup=%s)",
  731                 region_to_str(tcx, "", false, r_sub),
  732                 region_to_str(tcx, "", false, r_sup));
  733  
  734          if r_sup.is_bound() || r_sub.is_bound() {
  735              // a bound region is one which appears inside an fn type.
  736              // (e.g., the `&` in `fn(&T)`).  Such regions need not be
  737              // constrained by `minimum_lifetime` as they are placeholders
  738              // for regions that are as-yet-unknown.
  739          } else if r_sub == minimum_lifetime {
  740              rcx.fcx.mk_subr(
  741                  true, origin,
  742                  r_sub, r_sup);
  743          } else {
  744              rcx.fcx.mk_subr(
  745                  true, infer::ReferenceOutlivesReferent(ty, origin.span()),
  746                  r_sub, r_sup);
  747          }
  748      }
  749  
  750      return (e == rcx.errors_reported);
  751  }
  752  
  753  pub mod guarantor {
  754      /*!
  755       * The routines in this module are aiming to deal with the case
  756       * where a the contents of a borrowed pointer are re-borrowed.
  757       * Imagine you have a borrowed pointer `b` with lifetime L1 and
  758       * you have an expression `&*b`.  The result of this borrow will
  759       * be another borrowed pointer with lifetime L2 (which is an
  760       * inference variable).  The borrow checker is going to enforce
  761       * the constraint that L2 < L1, because otherwise you are
  762       * re-borrowing data for a lifetime larger than the original loan.
  763       * However, without the routines in this module, the region
  764       * inferencer would not know of this dependency and thus it might
  765       * infer the lifetime of L2 to be greater than L1 (issue #3148).
  766       *
  767       * There are a number of troublesome scenarios in the tests
  768       * `region-dependent-*.rs`, but here is one example:
  769       *
  770       *     struct Foo { i: int }
  771       *     struct Bar { foo: Foo  }
  772       *     fn get_i(x: &'a Bar) -> &'a int {
  773       *        let foo = &x.foo; // Lifetime L1
  774       *        &foo.i            // Lifetime L2
  775       *     }
  776       *
  777       * Note that this comes up either with `&` expressions, `ref`
  778       * bindings, and `autorefs`, which are the three ways to introduce
  779       * a borrow.
  780       *
  781       * The key point here is that when you are borrowing a value that
  782       * is "guaranteed" by a borrowed pointer, you must link the
  783       * lifetime of that borrowed pointer (L1, here) to the lifetime of
  784       * the borrow itself (L2).  What do I mean by "guaranteed" by a
  785       * borrowed pointer? I mean any data that is reached by first
  786       * dereferencing a borrowed pointer and then either traversing
  787       * interior offsets or owned pointers.  We say that the guarantor
  788       * of such data it the region of the borrowed pointer that was
  789       * traversed.  This is essentially the same as the ownership
  790       * relation, except that a borrowed pointer never owns its
  791       * contents.
  792       *
  793       * NB: I really wanted to use the `mem_categorization` code here
  794       * but I cannot because final type resolution hasn't happened yet,
  795       * and `mem_categorization` requires that all types be known.
  796       * So this is very similar logic to what you would find there,
  797       * but more special purpose.
  798       */
  799

  913      fn link(
  914          rcx: &mut Rcx,
  915          span: Span,
  916          id: ast::NodeId,
  917          guarantor: Option<ty::Region>) {
  918          /*!
  919           *
  920           * Links the lifetime of the borrowed pointer resulting from a borrow
  921           * to the lifetime of its guarantor (if any).
  922           */
  923  
  924          debug!("link(id=%?, guarantor=%?)", id, guarantor);
  925  
  926          let bound = match guarantor {
  927              None => {
  928                  // If guarantor is None, then the value being borrowed
  929                  // is not guaranteed by a region pointer, so there are
  930                  // no lifetimes to link.
  931                  return;
  932              }
  933              Some(r) => { r }
  934          };
  935  
  936          // this routine is used for the result of ref bindings and &
  937          // expressions, both of which always yield a region variable, so
  938          // mk_subr should never fail.
  939          let rptr_ty = rcx.resolve_node_type(id);
  940          if !ty::type_is_bot(rptr_ty) {
  941              let tcx = rcx.fcx.ccx.tcx;
  942              debug!("rptr_ty=%s", ty_to_str(tcx, rptr_ty));
  943              let r = ty::ty_region(tcx, span, rptr_ty);
  944              rcx.fcx.mk_subr(true, infer::Reborrow(span), r, bound);
  945          }
  946      }

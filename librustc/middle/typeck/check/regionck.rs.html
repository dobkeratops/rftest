   11 /*!
   12 
   13 The region check is a final pass that runs over the AST after we have
   14 inferred the type constraints but before we have actually finalized
   15 the types.  Its purpose is to embed some final region constraints.
   16 The reason that this is not done earlier is that sometimes we don't
   17 know whether a given type will be a region pointer or not until this
   18 phase.
   19 
   20 In particular, we ensure that, if the type of an expression or
   21 variable is `&'r T`, then the expression or variable must occur within
   22 the region scope `r`.  Note that in some cases `r` may still be a
   23 region variable, so this gives us a chance to influence the value for
   24 `r` that we infer to ensure we choose a value large enough to enclose
   25 all uses.  There is a lengthy comment in visit_node() that explains
   26 this point a bit better.
   27 
   28 */

   58 fn encl_region_of_def(fcx: @mut FnCtxt, def: ast::def) -> ty::Region {
   59     let tcx = fcx.tcx();
   60     match def {
   61         def_local(node_id, _) | def_arg(node_id, _) |
   62         def_self(node_id, _) | def_binding(node_id, _) => {
   63             tcx.region_maps.encl_region(node_id)
   64         }
   65         def_upvar(_, subdef, closure_id, body_id) => {
   66             match ty::ty_closure_sigil(fcx.node_ty(closure_id)) {
   67                 BorrowedSigil => encl_region_of_def(fcx, *subdef),
   68                 ManagedSigil | OwnedSigil => re_scope(body_id)
   69             }
   70         }
   71         _ => {
   72             tcx.sess.bug(fmt!("unexpected def in encl_region_of_def: %?",
   73                               def))
   74         }
   75     }
   76 }

   89     pub fn resolve_type(&mut self, unresolved_ty: ty::t) -> ty::t {
   90         /*!
   91          * Try to resolve the type for the given node, returning
   92          * t_err if an error results.  Note that we never care
   93          * about the details of the error, the same error will be
   94          * detected and reported in the writeback phase.
   95          *
   96          * Note one important point: we do not attempt to resolve
   97          * *region variables* here.  This is because regionck is
   98          * essentially adding constraints to those region variables
   99          * and so may yet influence how they are resolved.
  100          *
  101          * Consider this silly example:
  102          *
  103          *     fn borrow(x: &int) -> &int {x}
  104          *     fn foo(x: @int) -> int {  // block: B
  105          *         let b = borrow(x);    // region: <R0>
  106          *         *b
  107          *     }
  108          *
  109          * Here, the region of `b` will be `<R0>`.  `<R0>` is
  110          * constrainted to be some subregion of the block B and some
  111          * superregion of the call.  If we forced it now, we'd choose
  112          * the smaller region (the call).  But that would make the *b
  113          * illegal.  Since we don't resolve, the type of b will be
  114          * `&<R0>.int` and then `*b` will require that `<R0>` be
  115          * bigger than the let and the `*b` expression, so we will
  116          * effectively resolve `<R0>` to be the block B.
  117          */
  118         match resolve_type(self.fcx.infcx(), unresolved_ty,
  119                            resolve_and_force_all_but_regions) {
  120             Ok(t) => t,
  121             Err(_) => ty::mk_err()
  122         }
  123     }

  130     /// Try to resolve the type for the given node.
  131     pub fn resolve_expr_type_adjusted(@mut self, expr: @ast::expr) -> ty::t {
  132         let ty_unadjusted = self.resolve_node_type(expr.id);
  133         if ty::type_is_error(ty_unadjusted) || ty::type_is_bot(ty_unadjusted) {
  134             ty_unadjusted
  135         } else {

  139                           adjustments.find_copy(&expr.id))

  141     }

  211 fn constrain_bindings_in_pat(pat: @ast::pat, rcx: @mut Rcx) {
  212     let tcx = rcx.fcx.tcx();
  213     debug!("regionck::visit_pat(pat=%s)", pat.repr(tcx));
  214     do pat_util::pat_bindings(tcx.def_map, pat) |_, id, span, _| {
  215         // If we have a variable that contains region'd data, that
  216         // data will be accessible from anywhere that the variable is
  217         // accessed. We must be wary of loops like this:
  218         //
  219         //     // from src/test/compile-fail/borrowck-lend-flow.rs
  220         //     let mut v = ~3, w = ~4;
  221         //     let mut x = &mut w;
  222         //     loop {
  223         //         **x += 1;   // (2)
  224         //         borrow(v);  //~ ERROR cannot borrow
  225         //         x = &mut v; // (1)
  226         //     }
  227         //
  228         // Typically, we try to determine the region of a borrow from
  229         // those points where it is dereferenced. In this case, one
  230         // might imagine that the lifetime of `x` need only be the
  231         // body of the loop. But of course this is incorrect because
  232         // the pointer that is created at point (1) is consumed at
  233         // point (2), meaning that it must be live across the loop
  234         // iteration. The easiest way to guarantee this is to require
  235         // that the lifetime of any regions that appear in a
  236         // variable's type enclose at least the variable's scope.
  237 
  238         let encl_region = tcx.region_maps.encl_region(id);
  239         constrain_regions_in_type_of_node(
  240             rcx, id, encl_region,
  241             infer::BindingTypeIsNotValidAtDecl(span));
  242     }
  243 }

  245 fn visit_expr(expr: @ast::expr, (rcx, v): (@mut Rcx, rvt)) {
  246     debug!("regionck::visit_expr(e=%s, repeating_scope=%?)",
  247            expr.repr(rcx.fcx.tcx()), rcx.repeating_scope);
  248 
  249     let has_method_map = rcx.fcx.inh.method_map.contains_key(&expr.id);
  250 
  251     // Record cleanup scopes, which are used by borrowck to decide the
  252     // maximum lifetime of a temporary rvalue.  These were derived by
  253     // examining where trans creates block scopes, not because this
  254     // reflects some principled decision around temporary lifetimes.
  255     // Ordinarily this would seem like something that should be setup
  256     // in region, but we need to know which uses of operators are
  257     // overloaded.  See #3511.
  258     let tcx = rcx.fcx.tcx();
  259     match expr.node {
  260         // You'd think that x += y where `+=` is overloaded would be a
  261         // cleanup scope. You'd be... kind of right. In fact the
  262         // handling of `+=` and friends in trans for overloaded
  263         // operators is a hopeless mess and I can't figure out how to
  264         // represent it. - ndm
  265         //
  266         // ast::expr_assign_op(*) |
  267 
  268         ast::expr_index(*) |
  269         ast::expr_binary(*) |
  270         ast::expr_unary(*) if has_method_map => {
  271             tcx.region_maps.record_cleanup_scope(expr.id);
  272         }
  273         ast::expr_binary(_, ast::and, lhs, rhs) |
  274         ast::expr_binary(_, ast::or, lhs, rhs) => {
  275             tcx.region_maps.record_cleanup_scope(lhs.id);
  276             tcx.region_maps.record_cleanup_scope(rhs.id);
  277         }
  278         ast::expr_call(*) |
  279         ast::expr_method_call(*) => {
  280             tcx.region_maps.record_cleanup_scope(expr.id);
  281         }
  282         ast::expr_match(_, ref arms) => {
  283             tcx.region_maps.record_cleanup_scope(expr.id);
  284             for arm in arms.iter() {
  285                 for guard in arm.guard.iter() {
  286                     tcx.region_maps.record_cleanup_scope(guard.id);
  287                 }
  288             }
  289         }
  290         ast::expr_loop(ref body, _) => {
  291             tcx.region_maps.record_cleanup_scope(body.id);
  292         }
  293         ast::expr_while(cond, ref body) => {
  294             tcx.region_maps.record_cleanup_scope(cond.id);
  295             tcx.region_maps.record_cleanup_scope(body.id);
  296         }
  297         _ => {}
  298     }
  299 
  300     // Check any autoderefs or autorefs that appear.
  301     {
  302         let r = rcx.fcx.inh.adjustments.find(&expr.id);
  303         for &adjustment in r.iter() {
  304             debug!("adjustment=%?", adjustment);
  305             match *adjustment {
  306                 @ty::AutoDerefRef(
  307                     ty::AutoDerefRef {autoderefs: autoderefs, autoref: opt_autoref}) =>
  308                 {
  309                     let expr_ty = rcx.resolve_node_type(expr.id);
  310                     constrain_derefs(rcx, expr, autoderefs, expr_ty);
  311                     for autoref in opt_autoref.iter() {
  312                         guarantor::for_autoref(rcx, expr, autoderefs, autoref);
  313 
  314                         // Require that the resulting region encompasses
  315                         // the current node.
  316                         //
  317                         // FIXME(#6268) remove to support nested method calls
  318                         constrain_regions_in_type_of_node(
  319                             rcx, expr.id, ty::re_scope(expr.id),
  320                             infer::AutoBorrow(expr.span));
  321                     }
  322                 }
  323                 _ => {}
  324             }
  325         }
  326     }
  327 
  328     match expr.node {
  329         ast::expr_call(callee, ref args, _) => {
  330             constrain_callee(rcx, callee.id, expr, callee);
  331             constrain_call(rcx, callee.id, expr, None, *args, false);
  332 
  333             oldvisit::visit_expr(expr, (rcx, v));
  334         }

  341

  377         ast::expr_cast(source, _) => {
  378             // Determine if we are casting `source` to an trait
  379             // instance.  If so, we have to be sure that the type of
  380             // the source obeys the trait's region bound.
  381             //
  382             // Note: there is a subtle point here concerning type
  383             // parameters.  It is possible that the type of `source`
  384             // contains type parameters, which in turn may contain
  385             // regions that are not visible to us (only the caller
  386             // knows about them).  The kind checker is ultimately
  387             // responsible for guaranteeing region safety in that
  388             // particular case.  There is an extensive comment on the

  703 fn constrain_regions_in_type(
  704     rcx: @mut Rcx,
  705     minimum_lifetime: ty::Region,
  706     origin: infer::SubregionOrigin,
  707     ty: ty::t) -> bool
  708 {
  709     /*!
  710      * Requires that any regions which appear in `ty` must be
  711      * superregions of `minimum_lifetime`.  Also enforces the constraint
  712      * that given a pointer type `&'r T`, T must not contain regions
  713      * that outlive 'r, as well as analogous constraints for other
  714      * lifetime'd types.
  715      *
  716      * This check prevents regions from being used outside of the block in
  717      * which they are valid.  Recall that regions represent blocks of
  718      * code or expressions: this requirement basically says "any place
  719      * that uses or may use a region R must be within the block of
  720      * code that R corresponds to."
  721      */
  722 
  723     let e = rcx.errors_reported;
  724     let tcx = rcx.fcx.ccx.tcx;
  725 
  726     debug!("constrain_regions_in_type(minimum_lifetime=%s, ty=%s)",
  727            region_to_str(tcx, "", false, minimum_lifetime),
  728            ty_to_str(tcx, ty));
  729 
  730     do relate_nested_regions(tcx, Some(minimum_lifetime), ty) |r_sub, r_sup| {
  731         debug!("relate(r_sub=%s, r_sup=%s)",
  732                region_to_str(tcx, "", false, r_sub),
  733                region_to_str(tcx, "", false, r_sup));
  734 
  735         if r_sup.is_bound() || r_sub.is_bound() {
  736             // a bound region is one which appears inside an fn type.
  737             // (e.g., the `&` in `fn(&T)`).  Such regions need not be
  738             // constrained by `minimum_lifetime` as they are placeholders
  739             // for regions that are as-yet-unknown.
  740         } else if r_sub == minimum_lifetime {
  741             rcx.fcx.mk_subr(
  742                 true, origin,
  743                 r_sub, r_sup);
  744         } else {
  745             rcx.fcx.mk_subr(
  746                 true, infer::ReferenceOutlivesReferent(ty, origin.span()),
  747                 r_sub, r_sup);
  748         }
  749     }
  750 
  751     return (e == rcx.errors_reported);
  752 }

  754 pub mod guarantor {
  755     /*!
  756      * The routines in this module are aiming to deal with the case
  757      * where a the contents of a borrowed pointer are re-borrowed.
  758      * Imagine you have a borrowed pointer `b` with lifetime L1 and
  759      * you have an expression `&*b`.  The result of this borrow will
  760      * be another borrowed pointer with lifetime L2 (which is an
  761      * inference variable).  The borrow checker is going to enforce
  762      * the constraint that L2 < L1, because otherwise you are
  763      * re-borrowing data for a lifetime larger than the original loan.
  764      * However, without the routines in this module, the region
  765      * inferencer would not know of this dependency and thus it might
  766      * infer the lifetime of L2 to be greater than L1 (issue #3148).
  767      *
  768      * There are a number of troublesome scenarios in the tests
  769      * `region-dependent-*.rs`, but here is one example:
  770      *
  771      *     struct Foo { i: int }
  772      *     struct Bar { foo: Foo  }
  773      *     fn get_i(x: &'a Bar) -> &'a int {
  774      *        let foo = &x.foo; // Lifetime L1
  775      *        &foo.i            // Lifetime L2
  776      *     }
  777      *
  778      * Note that this comes up either with `&` expressions, `ref`
  779      * bindings, and `autorefs`, which are the three ways to introduce
  780      * a borrow.
  781      *
  782      * The key point here is that when you are borrowing a value that
  783      * is "guaranteed" by a borrowed pointer, you must link the
  784      * lifetime of that borrowed pointer (L1, here) to the lifetime of
  785      * the borrow itself (L2).  What do I mean by "guaranteed" by a
  786      * borrowed pointer? I mean any data that is reached by first
  787      * dereferencing a borrowed pointer and then either traversing
  788      * interior offsets or owned pointers.  We say that the guarantor
  789      * of such data it the region of the borrowed pointer that was
  790      * traversed.  This is essentially the same as the ownership
  791      * relation, except that a borrowed pointer never owns its
  792      * contents.
  793      *

  913     fn link(
  914         rcx: @mut Rcx,
  915         span: span,
  916         id: ast::NodeId,
  917         guarantor: Option<ty::Region>) {
  918         /*!
  919          *
  920          * Links the lifetime of the borrowed pointer resulting from a borrow
  921          * to the lifetime of its guarantor (if any).
  922          */

  924         debug!("link(id=%?, guarantor=%?)", id, guarantor);
  925 
  926         let bound = match guarantor {
  927             None => {
  928                 // If guarantor is None, then the value being borrowed
  929                 // is not guaranteed by a region pointer, so there are
  930                 // no lifetimes to link.
  931                 return;
  932             }
  933             Some(r) => { r }
  934         };
  935 
  936         // this routine is used for the result of ref bindings and &
  937         // expressions, both of which always yield a region variable, so
  938         // mk_subr should never fail.
  939         let rptr_ty = rcx.resolve_node_type(id);
  940         if !ty::type_is_bot(rptr_ty) {
  941             let tcx = rcx.fcx.ccx.tcx;
  942             debug!("rptr_ty=%s", ty_to_str(tcx, rptr_ty));
  943             let r = ty::ty_region(tcx, span, rptr_ty);
  944             rcx.fcx.mk_subr(true, infer::Reborrow(span), r, bound);
  945         }
  946     }

  996             }

 1001                 // not guaranteed by a region pointer.

 1003             }

 1042     fn categorize(rcx: @mut Rcx, expr: @ast::expr) -> ExprCategorization {
 1043         debug!("categorize()");
 1044 
 1045         let mut expr_ct = categorize_unadjusted(rcx, expr);
 1046         debug!("before adjustments, cat=%?", expr_ct.cat);
 1047 
 1048         match rcx.fcx.inh.adjustments.find(&expr.id) {
 1049             Some(&@ty::AutoAddEnv(*)) => {
 1050                 // This is basically an rvalue, not a pointer, no regions
 1051                 // involved.
 1052                 expr_ct.cat = ExprCategorization {
 1053                     guarantor: None,
 1054                     pointer: NotPointer
 1055                 };
 1056             }
 1057 
 1058             Some(&@ty::AutoDerefRef(ref adjustment)) => {
 1059                 debug!("adjustment=%?", adjustment);
 1060 
 1061                 expr_ct = apply_autoderefs(
 1062                     rcx, expr, adjustment.autoderefs, expr_ct);
 1063 
 1064                 match adjustment.autoref {
 1065                     None => {
 1066                     }
 1067                     Some(ty::AutoUnsafe(_)) => {
 1068                         expr_ct.cat.guarantor = None;
 1069                         expr_ct.cat.pointer = OtherPointer;
 1070                         debug!("autoref, cat=%?", expr_ct.cat);
 1071                     }
 1072                     Some(ty::AutoPtr(r, _)) |
 1073                     Some(ty::AutoBorrowVec(r, _)) |
 1074                     Some(ty::AutoBorrowVecRef(r, _)) |
 1075                     Some(ty::AutoBorrowFn(r)) => {
 1076                         // If there is an autoref, then the result of this
 1077                         // expression will be some sort of borrowed pointer.
 1078                         expr_ct.cat.guarantor = None;
 1079                         expr_ct.cat.pointer = BorrowedPointer(r);
 1080                         debug!("autoref, cat=%?", expr_ct.cat);
 1081                     }
 1082                 }
 1083             }
 1084 
 1085             None => {}
 1086         }
 1087 
 1088         debug!("result=%?", expr_ct.cat);
 1089         return expr_ct.cat;
 1090     }

 1115     fn apply_autoderefs(

 1190     fn link_ref_bindings_in_pat(
 1191         rcx: @mut Rcx,
 1192         pat: @ast::pat,
 1193         guarantor: Option<ty::Region>) {
 1194         /*!
 1195          *
 1196          * Descends through the pattern, tracking the guarantor
 1197          * of the value being matched.  When a ref binding is encountered,
 1198          * links the lifetime of that ref binding to the lifetime of
 1199          * the guarantor.  We begin with the guarantor of the
 1200          * discriminant but of course as we go we may pass through
 1201          * other pointers.
 1202          */
 1203 
 1204         debug!("link_ref_bindings_in_pat(pat=%s, guarantor=%?)",
 1205                rcx.fcx.pat_to_str(pat), guarantor);
 1206 
 1207         match pat.node {
 1208             ast::pat_wild => {}
 1209             ast::pat_ident(ast::bind_by_ref(_), _, opt_p) => {
 1210                 link(rcx, pat.span, pat.id, guarantor);
 1211 
 1212                 for p in opt_p.iter() {
 1213                     link_ref_bindings_in_pat(rcx, *p, guarantor);
 1214                 }
 1215             }
 1216             ast::pat_ident(_, _, opt_p) => {
 1217                 for p in opt_p.iter() {
 1218                     link_ref_bindings_in_pat(rcx, *p, guarantor);
 1219                 }
 1220             }
 1221             ast::pat_enum(_, None) => {}
 1222             ast::pat_enum(_, Some(ref pats)) => {
 1223                 link_ref_bindings_in_pats(rcx, pats, guarantor);
 1224             }
 1225             ast::pat_struct(_, ref fpats, _) => {
 1226                 for fpat in fpats.iter() {
 1227                     link_ref_bindings_in_pat(rcx, fpat.pat, guarantor);
 1228                 }
 1229             }
 1230             ast::pat_tup(ref ps) => {
 1231                 link_ref_bindings_in_pats(rcx, ps, guarantor)
 1232             }
 1233             ast::pat_box(p) => {
 1234                 link_ref_bindings_in_pat(rcx, p, None)
 1235             }
 1236             ast::pat_uniq(p) => {
 1237                 link_ref_bindings_in_pat(rcx, p, guarantor)
 1238             }
 1239             ast::pat_region(p) => {

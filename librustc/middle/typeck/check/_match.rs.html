  28  pub fn check_match(fcx: @mut FnCtxt,
  29                     expr: @ast::Expr,
  30                     discrim: @ast::Expr,
  31                     arms: &[ast::Arm]) {
  32      let tcx = fcx.ccx.tcx;
  33  
  34      let discrim_ty = fcx.infcx().next_ty_var();
  35      check_expr_has_type(fcx, discrim, discrim_ty);
  36  
  37      // Typecheck the patterns first, so that we get types for all the
  38      // bindings.
  39      for arm in arms.iter() {
  40          let mut pcx = pat_ctxt {
  41              fcx: fcx,
  42              map: pat_id_map(tcx.def_map, arm.pats[0]),
  43          };
  44  
  45          for p in arm.pats.iter() { check_pat(&mut pcx, *p, discrim_ty);}
  46      }
  47  
  48      // The result of the match is the common supertype of all the
  49      // arms. Start out the value as bottom, since it's the, well,
  50      // bottom the type lattice, and we'll be moving up the lattice as
  51      // we process each arm. (Note that any match with 0 arms is matching
  52      // on any empty type and is therefore unreachable; should the flow
  53      // of execution reach it, we will fail, so bottom is an appropriate
  54      // type in that case)
  55      let mut result_ty = ty::mk_bot();
  56  
  57      // Now typecheck the blocks.
  58      let mut saw_err = ty::type_is_error(discrim_ty);
  59      for arm in arms.iter() {
  60          let mut guard_err = false;
  61          let mut guard_bot = false;
  62          match arm.guard {
  63            Some(e) => {
  64                check_expr_has_type(fcx, e, ty::mk_bool());
  65                let e_ty = fcx.expr_ty(e);
  66                if ty::type_is_error(e_ty) {
  67                    guard_err = true;
  68                }
  69                else if ty::type_is_bot(e_ty) {
  70                    guard_bot = true;
  71                }
  72            },
  73            None => ()
  74          }
  75          check_block(fcx, &arm.body);
  76          let bty = fcx.node_ty(arm.body.id);
  77          saw_err = saw_err || ty::type_is_error(bty);
  78          if guard_err {
  79              fcx.write_error(arm.body.id);
  80              saw_err = true;
  81          }
  82          else if guard_bot {
  83              fcx.write_bot(arm.body.id);
  84          }
  85  
  86          result_ty =
  87              infer::common_supertype(
  88                  fcx.infcx(),
  89                  infer::MatchExpression(expr.span),
  90                  true, // result_ty is "expected" here
  91                  result_ty,
  92                  bty);
  93      }
  94  
  95      if saw_err {
  96          result_ty = ty::mk_err();
  97      } else if ty::type_is_bot(discrim_ty) {
  98          result_ty = ty::mk_bot();
  99      }
 100  
 101      fcx.write_ty(expr.id, result_ty);
 102  }

 109  pub fn check_pat_variant(pcx: &pat_ctxt, pat: @ast::Pat, path: &ast::Path,
 110                           subpats: &Option<~[@ast::Pat]>, expected: ty::t) {
 111  
 112      // Typecheck the path.
 113      let fcx = pcx.fcx;
 114      let tcx = pcx.fcx.ccx.tcx;
 115  
 116      let arg_types;
 117      let kind_name;
 118  
 119      // structure_of requires type variables to be resolved.
 120      // So when we pass in <expected>, it's an error if it
 121      // contains type variables.
 122  
 123      // Check to see whether this is an enum or a struct.
 124      match *structure_of(pcx.fcx, pat.span, expected) {
 125          ty::ty_enum(_, ref expected_substs) => {
 126              // Lookup the enum and variant def ids:
 127              let v_def = lookup_def(pcx.fcx, pat.span, pat.id);
 128              match ast_util::variant_def_ids(v_def) {
 129                  Some((enm, var)) => {
 130                      // Assign the pattern the type of the *enum*, not the variant.
 131                      let enum_tpt = ty::lookup_item_type(tcx, enm);
 132                      instantiate_path(pcx.fcx,
 133                                       path,
 134                                       enum_tpt,
 135                                       v_def,
 136                                       pat.span,
 137                                       pat.id);
 138  
 139                      // check that the type of the value being matched is a subtype
 140                      // of the type of the pattern:
 141                      let pat_ty = fcx.node_ty(pat.id);
 142                      demand::subtype(fcx, pat.span, expected, pat_ty);
 143  
 144                      // Get the expected types of the arguments.
 145                      arg_types = {
 146                          let vinfo =
 147                              ty::enum_variant_with_id(tcx, enm, var);
 148                          let var_tpt = ty::lookup_item_type(tcx, var);
 149                          vinfo.args.map(|t| {
 150                              if var_tpt.generics.type_param_defs.len() ==
 151                                  expected_substs.tps.len()
 152                              {
 153                                  ty::subst(tcx, expected_substs, *t)
 154                              }
 155                              else {
 156                                  *t // In this case, an error was already signaled
 157                                      // anyway
 158                              }
 159                          })
 160                      };
 161  
 162                      kind_name = "variant";
 163                  }
 164                  None => {
 165                      // See [Note-Type-error-reporting] in middle/typeck/infer/mod.rs
 166                      fcx.infcx().type_error_message_str_with_expected(pat.span,
 167                                                         |expected, actual| {
 168                                                         expected.map_move_default(~"", |e| {
 169                          fmt!("mismatched types: expected `%s` but found %s",
 170                               e, actual)})},
 171                               Some(expected), ~"a structure pattern",
 172                               None);
 173                      fcx.write_error(pat.id);
 174                      kind_name = "[error]";
 175                      arg_types = (*subpats).clone()
 176                                            .unwrap_or_default()
 177                                            .map(|_| ty::mk_err());
 178                  }
 179              }
 180          }
 181          ty::ty_struct(struct_def_id, ref expected_substs) => {
 182              // Lookup the struct ctor def id
 183              let s_def = lookup_def(pcx.fcx, pat.span, pat.id);
 184              let s_def_id = ast_util::def_id_of_def(s_def);
 185  
 186              // Assign the pattern the type of the struct.
 187              let ctor_tpt = ty::lookup_item_type(tcx, s_def_id);
 188              let struct_tpt = if ty::is_fn_ty(ctor_tpt.ty) {
 189                  ty::ty_param_bounds_and_ty {ty: ty::ty_fn_ret(ctor_tpt.ty),
 190                                          ..ctor_tpt}
 191              } else {
 192                  ctor_tpt
 193              };
 194              instantiate_path(pcx.fcx,
 195                               path,
 196                               struct_tpt,
 197                               s_def,
 198                               pat.span,
 199                               pat.id);
 200  
 201              // Check that the type of the value being matched is a subtype of
 202              // the type of the pattern.
 203              let pat_ty = fcx.node_ty(pat.id);
 204              demand::subtype(fcx, pat.span, expected, pat_ty);
 205  
 206              // Get the expected types of the arguments.
 207              let class_fields = ty::struct_fields(
 208                  tcx, struct_def_id, expected_substs);
 209              arg_types = class_fields.map(|field| field.mt.ty);
 210  
 211              kind_name = "structure";
 212          }
 213          _ => {
 214              // See [Note-Type-error-reporting] in middle/typeck/infer/mod.rs
 215              fcx.infcx().type_error_message_str_with_expected(pat.span,
 216                                                 |expected, actual| {
 217                                                 expected.map_move_default(~"", |e| {
 218                      fmt!("mismatched types: expected `%s` but found %s",
 219                           e, actual)})},
 220                      Some(expected), ~"an enum or structure pattern",
 221                      None);
 222              fcx.write_error(pat.id);
 223              kind_name = "[error]";
 224              arg_types = (*subpats).clone()
 225                                    .unwrap_or_default()
 226                                    .map(|_| ty::mk_err());
 227          }
 228      }
 229  
 230      let arg_len = arg_types.len();
 231  
 232      // Count the number of subpatterns.
 233      let subpats_len;
 234      match *subpats {
 235          None => subpats_len = arg_len,
 236          Some(ref subpats) => subpats_len = subpats.len()
 237      }
 238  
 239      let mut error_happened = false;
 240  
 241      if arg_len > 0 {
 242          // N-ary variant.
 243          if arg_len != subpats_len {
 244              let s = fmt!("this pattern has %u field%s, but the corresponding %s has %u field%s",
 245                           subpats_len,
 246                           if subpats_len == 1u { ~"" } else { ~"s" },
 247                           kind_name,
 248                           arg_len,
 249                           if arg_len == 1u { ~"" } else { ~"s" });
 250              tcx.sess.span_err(pat.span, s);
 251              error_happened = true;
 252          }
 253  
 254          if !error_happened {
 255              for pats in subpats.iter() {
 256                  for (subpat, arg_ty) in pats.iter().zip(arg_types.iter()) {
 257                      check_pat(pcx, *subpat, *arg_ty);
 258                  }
 259              }
 260          }
 261      } else if subpats_len > 0 {
 262          tcx.sess.span_err(pat.span,
 263                            fmt!("this pattern has %u field%s, but the corresponding %s has no \
 264                                  fields",
 265                                 subpats_len,
 266                                 if subpats_len == 1u { "" } else { "s" },
 267                                 kind_name));
 268          error_happened = true;
 269      }
 270  
 271      if error_happened {
 272          for pats in subpats.iter() {
 273              for pat in pats.iter() {
 274                  check_pat(pcx, *pat, ty::mk_err());
 275              }
 276          }
 277      }
 278  }

 280  /// `path` is the AST path item naming the type of this struct.
 281  /// `fields` is the field patterns of the struct pattern.
 282  /// `class_fields` describes the type of each field of the struct.
 283  /// `class_id` is the ID of the struct.
 284  /// `substitutions` are the type substitutions applied to this struct type
 285  /// (e.g. K,V in HashMap<K,V>).
 286  /// `etc` is true if the pattern said '...' and false otherwise.
 287  pub fn check_struct_pat_fields(pcx: &pat_ctxt,
 288                                 span: Span,
 289                                 path: &ast::Path,
 290                                 fields: &[ast::FieldPat],
 291                                 class_fields: ~[ty::field_ty],
 292                                 class_id: ast::DefId,
 293                                 substitutions: &ty::substs,
 294                                 etc: bool) {
 295      let tcx = pcx.fcx.ccx.tcx;
 296  
 297      // Index the class fields.
 298      let mut field_map = HashMap::new();
 299      for (i, class_field) in class_fields.iter().enumerate() {
 300          field_map.insert(class_field.name, i);
 301      }
 302  
 303      // Typecheck each field.
 304      let mut found_fields = HashSet::new();
 305      for field in fields.iter() {
 306          match field_map.find(&field.ident.name) {
 307              Some(&index) => {
 308                  let class_field = class_fields[index];
 309                  let field_type = ty::lookup_field_type(tcx,
 310                                                         class_id,
 311                                                         class_field.id,
 312                                                         substitutions);
 313                  check_pat(pcx, field.pat, field_type);
 314                  found_fields.insert(index);
 315              }
 316              None => {
 317                  let name = pprust::path_to_str(path, tcx.sess.intr());
 318                  // Check the pattern anyway, so that attempts to look
 319                  // up its type won't fail
 320                  check_pat(pcx, field.pat, ty::mk_err());
 321                  tcx.sess.span_err(span,
 322                      fmt!("struct `%s` does not have a field named `%s`",
 323                           name,
 324                           tcx.sess.str_of(field.ident)));
 325              }
 326          }
 327      }
 328  
 329      // Report an error if not all the fields were specified.
 330      if !etc {
 331          for (i, field) in class_fields.iter().enumerate() {
 332              if found_fields.contains(&i) {
 333                  loop;
 334              }
 335              tcx.sess.span_err(span,
 336                                fmt!("pattern does not mention field `%s`",
 337                                     token::interner_get(field.name)));
 338          }
 339      }
 340  }

 342  pub fn check_struct_pat(pcx: &pat_ctxt, pat_id: ast::NodeId, span: Span,
 343                          expected: ty::t, path: &ast::Path,
 344                          fields: &[ast::FieldPat], etc: bool,
 345                          struct_id: ast::DefId,
 346                          substitutions: &ty::substs) {
 347      let fcx = pcx.fcx;
 348      let tcx = pcx.fcx.ccx.tcx;
 349  
 350      let class_fields = ty::lookup_struct_fields(tcx, struct_id);
 351  
 352      // Check to ensure that the struct is the one specified.
 353      match tcx.def_map.find(&pat_id) {
 354          Some(&ast::DefStruct(supplied_def_id))
 355                  if supplied_def_id == struct_id => {
 356              // OK.
 357          }
 358          Some(&ast::DefStruct(*)) | Some(&ast::DefVariant(*)) => {
 359              let name = pprust::path_to_str(path, tcx.sess.intr());
 360              tcx.sess.span_err(span,
 361                                fmt!("mismatched types: expected `%s` but found `%s`",
 362                                     fcx.infcx().ty_to_str(expected),
 363                                     name));
 364          }
 365          _ => {
 366              tcx.sess.span_bug(span, "resolve didn't write in struct ID");
 367          }
 368      }
 369  
 370      check_struct_pat_fields(pcx, span, path, fields, class_fields, struct_id,
 371                              substitutions, etc);
 372  }

 374  pub fn check_struct_like_enum_variant_pat(pcx: &pat_ctxt,
 375                                            pat_id: ast::NodeId,
 376                                            span: Span,
 377                                            expected: ty::t,
 378                                            path: &ast::Path,
 379                                            fields: &[ast::FieldPat],
 380                                            etc: bool,
 381                                            enum_id: ast::DefId,
 382                                            substitutions: &ty::substs) {
 383      let fcx = pcx.fcx;
 384      let tcx = pcx.fcx.ccx.tcx;
 385  
 386      // Find the variant that was specified.
 387      match tcx.def_map.find(&pat_id) {
 388          Some(&ast::DefVariant(found_enum_id, variant_id, _))
 389                  if found_enum_id == enum_id => {
 390              // Get the struct fields from this struct-like enum variant.
 391              let class_fields = ty::lookup_struct_fields(tcx, variant_id);
 392  
 393              check_struct_pat_fields(pcx, span, path, fields, class_fields,
 394                                      variant_id, substitutions, etc);
 395          }
 396          Some(&ast::DefStruct(*)) | Some(&ast::DefVariant(*)) => {
 397              let name = pprust::path_to_str(path, tcx.sess.intr());
 398              tcx.sess.span_err(span,
 399                                fmt!("mismatched types: expected `%s` but \
 400                                      found `%s`",
 401                                     fcx.infcx().ty_to_str(expected),
 402                                     name));
 403          }
 404          _ => {
 405              tcx.sess.span_bug(span, "resolve didn't write in variant");
 406          }
 407      }
 408  }

 410  // Pattern checking is top-down rather than bottom-up so that bindings get
 411  // their types immediately.
 412  pub fn check_pat(pcx: &pat_ctxt, pat: @ast::Pat, expected: ty::t) {
 413      let fcx = pcx.fcx;
 414      let tcx = pcx.fcx.ccx.tcx;
 415  
 416      match pat.node {
 417        ast::PatWild => {
 418          fcx.write_ty(pat.id, expected);
 419        }
 420        ast::PatLit(lt) => {
 421          check_expr_has_type(fcx, lt, expected);
 422          fcx.write_ty(pat.id, fcx.expr_ty(lt));
 423        }
 424        ast::PatRange(begin, end) => {
 425          check_expr_has_type(fcx, begin, expected);
 426          check_expr_has_type(fcx, end, expected);
 427          let b_ty =
 428              fcx.infcx().resolve_type_vars_if_possible(fcx.expr_ty(begin));
 429          let e_ty =
 430              fcx.infcx().resolve_type_vars_if_possible(fcx.expr_ty(end));

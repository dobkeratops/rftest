  27 pub fn check_match(fcx: @mut FnCtxt,
  28                    expr: @ast::expr,
  29                    discrim: @ast::expr,
  30                    arms: &[ast::arm]) {
  31     let tcx = fcx.ccx.tcx;
  32 
  33     let discrim_ty = fcx.infcx().next_ty_var();
  34     check_expr_has_type(fcx, discrim, discrim_ty);
  35 
  36     // Typecheck the patterns first, so that we get types for all the
  37     // bindings.
  38     for arm in arms.iter() {
  39         let pcx = pat_ctxt {
  40             fcx: fcx,
  41             map: pat_id_map(tcx.def_map, arm.pats[0]),
  42         };
  43 
  44         for p in arm.pats.iter() { check_pat(&pcx, *p, discrim_ty);}
  45     }
  46 
  47     // The result of the match is the common supertype of all the
  48     // arms. Start out the value as bottom, since it's the, well,
  49     // bottom the type lattice, and we'll be moving up the lattice as
  50     // we process each arm. (Note that any match with 0 arms is matching
  51     // on any empty type and is therefore unreachable; should the flow
  52     // of execution reach it, we will fail, so bottom is an appropriate
  53     // type in that case)
  54     let mut result_ty = ty::mk_bot();
  55 
  56     // Now typecheck the blocks.
  57     let mut saw_err = ty::type_is_error(discrim_ty);
  58     for arm in arms.iter() {
  59         let mut guard_err = false;
  60         let mut guard_bot = false;
  61         match arm.guard {
  62           Some(e) => {
  63               check_expr_has_type(fcx, e, ty::mk_bool());
  64               let e_ty = fcx.expr_ty(e);
  65               if ty::type_is_error(e_ty) {
  66                   guard_err = true;
  67               }
  68               else if ty::type_is_bot(e_ty) {
  69                   guard_bot = true;
  70               }
  71           },
  72           None => ()
  73         }
  74         check_block(fcx, &arm.body);
  75         let bty = fcx.node_ty(arm.body.id);
  76         saw_err = saw_err || ty::type_is_error(bty);
  77         if guard_err {
  78             fcx.write_error(arm.body.id);
  79             saw_err = true;
  80         }
  81         else if guard_bot {
  82             fcx.write_bot(arm.body.id);
  83         }
  84 
  85         result_ty =
  86             infer::common_supertype(
  87                 fcx.infcx(),
  88                 infer::MatchExpression(expr.span),
  89                 true, // result_ty is "expected" here
  90                 result_ty,
  91                 bty);
  92     }
  93 
  94     if saw_err {
  95         result_ty = ty::mk_err();
  96     } else if ty::type_is_bot(discrim_ty) {
  97         result_ty = ty::mk_bot();
  98     }
  99 
 100     fcx.write_ty(expr.id, result_ty);
 101 }

 108 pub fn check_pat_variant(pcx: &pat_ctxt, pat: @ast::pat, path: &ast::Path,
 109                          subpats: &Option<~[@ast::pat]>, expected: ty::t) {
 110 
 111     // Typecheck the path.
 112     let fcx = pcx.fcx;
 113     let tcx = pcx.fcx.ccx.tcx;
 114 
 115     let arg_types;
 116     let kind_name;
 117 
 118     // structure_of requires type variables to be resolved.
 119     // So when we pass in <expected>, it's an error if it
 120     // contains type variables.
 121 
 122     // Check to see whether this is an enum or a struct.
 123     match *structure_of(pcx.fcx, pat.span, expected) {
 124         ty::ty_enum(_, ref expected_substs) => {
 125             // Lookup the enum and variant def ids:
 126             let v_def = lookup_def(pcx.fcx, pat.span, pat.id);
 127             match ast_util::variant_def_ids(v_def) {
 128                 Some((enm, var)) => {
 129                     // Assign the pattern the type of the *enum*, not the variant.
 130                     let enum_tpt = ty::lookup_item_type(tcx, enm);
 131                     instantiate_path(pcx.fcx, path, enum_tpt, pat.span, pat.id);
 132 
 133                     // check that the type of the value being matched is a subtype
 134                     // of the type of the pattern:
 135                     let pat_ty = fcx.node_ty(pat.id);
 136                     demand::subtype(fcx, pat.span, expected, pat_ty);
 137 
 138                     // Get the expected types of the arguments.
 139                     arg_types = {
 140                         let vinfo =
 141                             ty::enum_variant_with_id(tcx, enm, var);
 142                         let var_tpt = ty::lookup_item_type(tcx, var);
 143                         vinfo.args.map(|t| {
 144                             if var_tpt.generics.type_param_defs.len() ==
 145                                 expected_substs.tps.len()
 146                             {
 147                                 ty::subst(tcx, expected_substs, *t)
 148                             }
 149                             else {
 150                                 *t // In this case, an error was already signaled
 151                                     // anyway
 152                             }
 153                         })
 154                     };
 155 
 156                     kind_name = "variant";
 157                 }
 158                 None => {
 159                     fcx.infcx().type_error_message_str_with_expected(pat.span,
 160                                                        |expected, actual| {
 161                                                        expected.map_default(~"", |e| {
 162                         fmt!("mismatched types: expected `%s` but found %s",
 163                              *e, actual)})},
 164                              Some(expected), ~"a structure pattern",
 165                              None);
 166                     fcx.write_error(pat.id);
 167                     kind_name = "[error]";
 168                     arg_types = (*subpats).clone()
 169                                           .get_or_default(~[])
 170                                           .map(|_| ty::mk_err());
 171                 }
 172             }
 173         }
 174         ty::ty_struct(struct_def_id, ref expected_substs) => {
 175             // Lookup the struct ctor def id
 176             let s_def = lookup_def(pcx.fcx, pat.span, pat.id);
 177             let s_def_id = ast_util::def_id_of_def(s_def);
 178 
 179             // Assign the pattern the type of the struct.
 180             let ctor_tpt = ty::lookup_item_type(tcx, s_def_id);
 181             let struct_tpt = if ty::is_fn_ty(ctor_tpt.ty) {
 182                 ty::ty_param_bounds_and_ty {ty: ty::ty_fn_ret(ctor_tpt.ty),
 183                                         ..ctor_tpt}
 184             } else {
 185                 ctor_tpt
 186             };
 187             instantiate_path(pcx.fcx, path, struct_tpt, pat.span, pat.id);
 188 
 189             // Check that the type of the value being matched is a subtype of
 190             // the type of the pattern.
 191             let pat_ty = fcx.node_ty(pat.id);
 192             demand::subtype(fcx, pat.span, expected, pat_ty);
 193 
 194             // Get the expected types of the arguments.
 195             let class_fields = ty::struct_fields(
 196                 tcx, struct_def_id, expected_substs);
 197             arg_types = class_fields.map(|field| field.mt.ty);
 198 
 199             kind_name = "structure";
 200         }
 201         _ => {
 202             fcx.infcx().type_error_message_str_with_expected(pat.span,
 203                                                |expected, actual| {
 204                                                expected.map_default(~"", |e| {
 205                     fmt!("mismatched types: expected `%s` but found %s",
 206                          *e, actual)})},
 207                     Some(expected), ~"an enum or structure pattern",
 208                     None);
 209             fcx.write_error(pat.id);
 210             kind_name = "[error]";
 211             arg_types = (*subpats).clone()
 212                                   .get_or_default(~[])
 213                                   .map(|_| ty::mk_err());
 214         }
 215     }
 216 
 217     let arg_len = arg_types.len();
 218 
 219     // Count the number of subpatterns.
 220     let subpats_len;
 221     match *subpats {
 222         None => subpats_len = arg_len,
 223         Some(ref subpats) => subpats_len = subpats.len()
 224     }
 225 
 226     let mut error_happened = false;
 227 
 228     if arg_len > 0 {
 229         // N-ary variant.
 230         if arg_len != subpats_len {
 231             let s = fmt!("this pattern has %u field%s, but the corresponding %s has %u field%s",
 232                          subpats_len,
 233                          if subpats_len == 1u { ~"" } else { ~"s" },
 234                          kind_name,
 235                          arg_len,
 236                          if arg_len == 1u { ~"" } else { ~"s" });
 237             tcx.sess.span_err(pat.span, s);
 238             error_happened = true;
 239         }
 240 
 241         if !error_happened {
 242             for pats in subpats.iter() {
 243                 for (subpat, arg_ty) in pats.iter().zip(arg_types.iter()) {
 244                     check_pat(pcx, *subpat, *arg_ty);
 245                 }
 246             }
 247         }
 248     } else if subpats_len > 0 {
 249         tcx.sess.span_err(pat.span,
 250                           fmt!("this pattern has %u field%s, but the corresponding %s has no \
 251                                 fields",
 252                                subpats_len,
 253                                if subpats_len == 1u { "" } else { "s" },
 254                                kind_name));
 255         error_happened = true;
 256     }
 257 
 258     if error_happened {
 259         for pats in subpats.iter() {
 260             for pat in pats.iter() {
 261                 check_pat(pcx, *pat, ty::mk_err());
 262             }
 263         }
 264     }
 265 }

 267 /// `path` is the AST path item naming the type of this struct.
 268 /// `fields` is the field patterns of the struct pattern.
 269 /// `class_fields` describes the type of each field of the struct.
 270 /// `class_id` is the ID of the struct.
 271 /// `substitutions` are the type substitutions applied to this struct type
 272 /// (e.g. K,V in HashMap<K,V>).
 273 /// `etc` is true if the pattern said '...' and false otherwise.
 274 pub fn check_struct_pat_fields(pcx: &pat_ctxt,
 275                                span: span,
 276                                path: &ast::Path,
 277                                fields: &[ast::field_pat],
 278                                class_fields: ~[ty::field_ty],
 279                                class_id: ast::def_id,
 280                                substitutions: &ty::substs,
 281                                etc: bool) {
 282     let tcx = pcx.fcx.ccx.tcx;
 283 
 284     // Index the class fields.
 285     let mut field_map = HashMap::new();
 286     for (i, class_field) in class_fields.iter().enumerate() {
 287         field_map.insert(class_field.ident, i);
 288     }
 289 
 290     // Typecheck each field.
 291     let mut found_fields = HashSet::new();
 292     for field in fields.iter() {
 293         match field_map.find(&field.ident) {
 294             Some(&index) => {
 295                 let class_field = class_fields[index];
 296                 let field_type = ty::lookup_field_type(tcx,
 297                                                        class_id,
 298                                                        class_field.id,
 299                                                        substitutions);
 300                 check_pat(pcx, field.pat, field_type);
 301                 found_fields.insert(index);
 302             }
 303             None => {
 304                 let name = pprust::path_to_str(path, tcx.sess.intr());
 305                 tcx.sess.span_err(span,
 306                                   fmt!("struct `%s` does not have a field
 307                                         named `%s`", name,
 308                                        tcx.sess.str_of(field.ident)));
 309             }
 310         }
 311     }
 312 
 313     // Report an error if not all the fields were specified.
 314     if !etc {
 315         for (i, field) in class_fields.iter().enumerate() {
 316             if found_fields.contains(&i) {
 317                 loop;
 318             }
 319             tcx.sess.span_err(span,
 320                               fmt!("pattern does not mention field `%s`",
 321                                    tcx.sess.str_of(field.ident)));
 322         }
 323     }
 324 }

 326 pub fn check_struct_pat(pcx: &pat_ctxt, pat_id: ast::NodeId, span: span,
 327                         expected: ty::t, path: &ast::Path,
 328                         fields: &[ast::field_pat], etc: bool,
 329                         class_id: ast::def_id, substitutions: &ty::substs) {
 330     let fcx = pcx.fcx;
 331     let tcx = pcx.fcx.ccx.tcx;
 332 
 333     let class_fields = ty::lookup_struct_fields(tcx, class_id);
 334 
 335     // Check to ensure that the struct is the one specified.
 336     match tcx.def_map.find(&pat_id) {
 337         Some(&ast::def_struct(supplied_def_id))
 338                 if supplied_def_id == class_id => {
 339             // OK.
 340         }
 341         Some(&ast::def_struct(*)) | Some(&ast::def_variant(*)) => {
 342             let name = pprust::path_to_str(path, tcx.sess.intr());
 343             tcx.sess.span_err(span,
 344                               fmt!("mismatched types: expected `%s` but found `%s`",
 345                                    fcx.infcx().ty_to_str(expected),
 346                                    name));
 347         }
 348         _ => {
 349             tcx.sess.span_bug(span, "resolve didn't write in class");
 350         }
 351     }
 352 
 353     check_struct_pat_fields(pcx, span, path, fields, class_fields, class_id,
 354                             substitutions, etc);
 355 }

 357 pub fn check_struct_like_enum_variant_pat(pcx: &pat_ctxt,
 358                                           pat_id: ast::NodeId,
 359                                           span: span,
 360                                           expected: ty::t,
 361                                           path: &ast::Path,
 362                                           fields: &[ast::field_pat],
 363                                           etc: bool,
 364                                           enum_id: ast::def_id,
 365                                           substitutions: &ty::substs) {
 366     let fcx = pcx.fcx;
 367     let tcx = pcx.fcx.ccx.tcx;
 368 
 369     // Find the variant that was specified.
 370     match tcx.def_map.find(&pat_id) {
 371         Some(&ast::def_variant(found_enum_id, variant_id))
 372                 if found_enum_id == enum_id => {
 373             // Get the struct fields from this struct-like enum variant.
 374             let class_fields = ty::lookup_struct_fields(tcx, variant_id);
 375 
 376             check_struct_pat_fields(pcx, span, path, fields, class_fields,
 377                                     variant_id, substitutions, etc);
 378         }
 379         Some(&ast::def_struct(*)) | Some(&ast::def_variant(*)) => {
 380             let name = pprust::path_to_str(path, tcx.sess.intr());
 381             tcx.sess.span_err(span,
 382                               fmt!("mismatched types: expected `%s` but \
 383                                     found `%s`",
 384                                    fcx.infcx().ty_to_str(expected),
 385                                    name));
 386         }
 387         _ => {
 388             tcx.sess.span_bug(span, "resolve didn't write in variant");
 389         }
 390     }
 391 }

 393 // Pattern checking is top-down rather than bottom-up so that bindings get
 394 // their types immediately.
 395 pub fn check_pat(pcx: &pat_ctxt, pat: @ast::pat, expected: ty::t) {
 396     let fcx = pcx.fcx;
 397     let tcx = pcx.fcx.ccx.tcx;
 398 
 399     match pat.node {
 400       ast::pat_wild => {
 401         fcx.write_ty(pat.id, expected);
 402       }
 403       ast::pat_lit(lt) => {
 404         check_expr_has_type(fcx, lt, expected);
 405         fcx.write_ty(pat.id, fcx.expr_ty(lt));
 406       }
 407       ast::pat_range(begin, end) => {
 408         check_expr_has_type(fcx, begin, expected);
 409         check_expr_has_type(fcx, end, expected);
 410         let b_ty =
 411             fcx.infcx().resolve_type_vars_if_possible(fcx.expr_ty(begin));
 412         let e_ty =
 413             fcx.infcx().resolve_type_vars_if_possible(fcx.expr_ty(end));
 414         debug!("pat_range beginning type: %?", b_ty);
 415         debug!("pat_range ending type: %?", e_ty);
 416         if !require_same_types(
 417             tcx, Some(fcx.infcx()), false, pat.span, b_ty, e_ty,
 418             || ~"mismatched types in range")
 419         {
 420             // no-op
 421         } else if !ty::type_is_numeric(b_ty) {
 422             tcx.sess.span_err(pat.span, "non-numeric type used in range");
 423         } else {
 424             match valid_range_bounds(fcx.ccx, begin, end) {
 425                 Some(false) => {
 426                     tcx.sess.span_err(begin.span,
 427                         "lower range bound must be less than upper");
 428                 },
 429                 None => {
 430                     tcx.sess.span_err(begin.span,
 431                         "mismatched types in range");
 432                 },
 433                 _ => { },
 434             }
 435         }
 436         fcx.write_ty(pat.id, b_ty);
 437       }
 438       ast::pat_enum(*) |
 439       ast::pat_ident(*) if pat_is_const(tcx.def_map, pat) => {
 440         let const_did = ast_util::def_id_of_def(tcx.def_map.get_copy(&pat.id));
 441         let const_tpt = ty::lookup_item_type(tcx, const_did);

  35  fn resolve_type_vars_in_type(fcx: @mut FnCtxt, sp: Span, typ: ty::t)
  36                            -> Option<ty::t> {
  37      if !ty::type_needs_infer(typ) { return Some(typ); }
  38      match resolve_type(fcx.infcx(), typ, resolve_all | force_all) {
  39          Ok(new_type) => return Some(new_type),
  40          Err(e) => {
  41              if !fcx.ccx.tcx.sess.has_errors() {
  42                  fcx.ccx.tcx.sess.span_err(
  43                      sp,
  44                      fmt!("cannot determine a type \
  45                            for this expression: %s",
  46                           infer::fixup_err_to_str(e)))
  47              }
  48              return None;
  49          }
  50      }
  51  }

  53  fn resolve_type_vars_in_types(fcx: @mut FnCtxt, sp: Span, tys: &[ty::t])
  54                            -> ~[ty::t] {
  55      tys.map(|t| {
  56          match resolve_type_vars_in_type(fcx, sp, *t) {
  57              Some(t1) => t1,
  58              None => ty::mk_err()
  59          }
  60      })
  61  }

  63  fn resolve_method_map_entry(fcx: @mut FnCtxt, sp: Span, id: ast::NodeId) {
  64      // Resolve any method map entry
  65      match fcx.inh.method_map.find(&id) {
  66          None => {}
  67          Some(mme) => {
  68              {
  69                  let r = resolve_type_vars_in_type(fcx, sp, mme.self_ty);
  70                  for t in r.iter() {
  71                      let method_map = fcx.ccx.method_map;
  72                      let new_entry = method_map_entry { self_ty: *t, ..*mme };
  73                      debug!("writeback::resolve_method_map_entry(id=%?, \
  74                              new_entry=%?)",
  75                             id, new_entry);
  76                      method_map.insert(id, new_entry);
  77                  }
  78              }
  79          }
  80      }
  81  }

  83  fn resolve_vtable_map_entry(fcx: @mut FnCtxt, sp: Span, id: ast::NodeId) {

 117  fn resolve_type_vars_for_node(wbcx: &mut WbCtxt, sp: Span, id: ast::NodeId)
 118                             -> Option<ty::t> {
 119      let fcx = wbcx.fcx;
 120      let tcx = fcx.ccx.tcx;
 121  
 122      // Resolve any borrowings for the node with id `id`
 123      match fcx.inh.adjustments.find(&id) {
 124          None => (),
 125  
 126          Some(&@ty::AutoAddEnv(r, s)) => {
 127              match resolve_region(fcx.infcx(), r, resolve_all | force_all) {
 128                  Err(e) => {
 129                      // This should not, I think, happen:
 130                      fcx.ccx.tcx.sess.span_err(
 131                          sp, fmt!("cannot resolve bound for closure: %s",
 132                                   infer::fixup_err_to_str(e)));
 133                  }
 134                  Ok(r1) => {
 135                      let resolved_adj = @ty::AutoAddEnv(r1, s);
 136                      debug!("Adjustments for node %d: %?", id, resolved_adj);
 137                      fcx.tcx().adjustments.insert(id, resolved_adj);
 138                  }
 139              }
 140          }
 141  
 142          Some(&@ty::AutoDerefRef(adj)) => {
 143              let fixup_region = |r| {
 144                  match resolve_region(fcx.infcx(), r, resolve_all | force_all) {
 145                      Ok(r1) => r1,
 146                      Err(e) => {
 147                          // This should not, I think, happen.
 148                          fcx.ccx.tcx.sess.span_err(
 149                              sp, fmt!("cannot resolve scope of borrow: %s",
 150                                       infer::fixup_err_to_str(e)));
 151                          r
 152                      }
 153                  }
 154              };
 155  
 156              let resolved_autoref = match adj.autoref {
 157                  None => None,
 158                  Some(ref r) => Some(r.map_region(fixup_region))
 159              };
 160  
 161              let resolved_adj = @ty::AutoDerefRef(ty::AutoDerefRef {
 162                  autoderefs: adj.autoderefs,
 163                  autoref: resolved_autoref,
 164              });
 165              debug!("Adjustments for node %d: %?", id, resolved_adj);
 166              fcx.tcx().adjustments.insert(id, resolved_adj);
 167          }
 168      }
 169  
 170      // Resolve the type of the node with id `id`
 171      let n_ty = fcx.node_ty(id);
 172      match resolve_type_vars_in_type(fcx, sp, n_ty) {
 173        None => {
 174          wbcx.success = false;
 175          return None;
 176        }
 177  
 178        Some(t) => {
 179          debug!("resolve_type_vars_for_node(id=%d, n_ty=%s, t=%s)",
 180                 id, ppaux::ty_to_str(tcx, n_ty), ppaux::ty_to_str(tcx, t));
 181          write_ty_to_tcx(tcx, id, t);
 182          let mut ret = Some(t);
 183          do fcx.opt_node_ty_substs(id) |substs| {
 184            let mut new_tps = ~[];
 185            for subst in substs.tps.iter() {
 186                match resolve_type_vars_in_type(fcx, sp, *subst) {
 187                  Some(t) => new_tps.push(t),
 188                  None => { wbcx.success = false; ret = None; break }
 189                }
 190            }
 191            write_substs_to_tcx(tcx, id, new_tps);
 192            ret.is_some()
 193          };
 194          ret
 195        }
 196      }
 197  }

 210  struct WbCtxt {
 211      fcx: @mut FnCtxt,
 212  
 213      // As soon as we hit an error we have to stop resolving

 216  }

 218  fn visit_stmt(s: @ast::Stmt, wbcx: &mut WbCtxt) {
 219      if !wbcx.success { return; }
 220      resolve_type_vars_for_node(wbcx, s.span, ty::stmt_node_id(s));
 221      visit::walk_stmt(wbcx, s, ());
 222  }

 224  fn visit_expr(e: @ast::Expr, wbcx: &mut WbCtxt) {
 225      if !wbcx.success {
 226          return;
 227      }
 228  
 229      resolve_type_vars_for_node(wbcx, e.span, e.id);
 230  
 231      resolve_method_map_entry(wbcx.fcx, e.span, e.id);
 232      {
 233          let r = e.get_callee_id();
 234          for callee_id in r.iter() {
 235              resolve_method_map_entry(wbcx.fcx, e.span, *callee_id);
 236          }
 237      }
 238  
 239      resolve_vtable_map_entry(wbcx.fcx, e.span, e.id);
 240      {
 241          let r = e.get_callee_id();
 242          for callee_id in r.iter() {
 243              resolve_vtable_map_entry(wbcx.fcx, e.span, *callee_id);
 244          }
 245      }
 246  
 247      match e.node {
 248          ast::ExprFnBlock(ref decl, _) => {
 249              for input in decl.inputs.iter() {
 250                  let _ = resolve_type_vars_for_node(wbcx, e.span, input.id);
 251              }
 252          }
 253  
 254          ast::ExprBinary(callee_id, _, _, _) |
 255          ast::ExprUnary(callee_id, _, _) |
 256          ast::ExprAssignOp(callee_id, _, _, _) |
 257          ast::ExprIndex(callee_id, _, _) => {
 258              maybe_resolve_type_vars_for_node(wbcx, e.span, callee_id);
 259          }
 260  
 261          ast::ExprMethodCall(callee_id, _, _, _, _, _) => {
 262              // We must always have written in a callee ID type for these.
 263              resolve_type_vars_for_node(wbcx, e.span, callee_id);
 264          }
 265  
 266          _ => ()
 267      }
 268  
 269      visit::walk_expr(wbcx, e, ());
 270  }

 295  fn visit_local(l: @ast::Local, wbcx: &mut WbCtxt) {
 296      if !wbcx.success { return; }

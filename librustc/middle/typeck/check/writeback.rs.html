  34  fn resolve_type_vars_in_type(fcx: @mut FnCtxt, sp: span, typ: ty::t)
  35                            -> Option<ty::t> {
  36      if !ty::type_needs_infer(typ) { return Some(typ); }
  37      match resolve_type(fcx.infcx(), typ, resolve_all | force_all) {
  38          Ok(new_type) => return Some(new_type),
  39          Err(e) => {
  40              if !fcx.ccx.tcx.sess.has_errors() {
  41                  fcx.ccx.tcx.sess.span_err(
  42                      sp,
  43                      fmt!("cannot determine a type \
  44                            for this expression: %s",
  45                           infer::fixup_err_to_str(e)))
  46              }
  47              return None;
  48          }
  49      }
  50  }

  52  fn resolve_type_vars_in_types(fcx: @mut FnCtxt, sp: span, tys: &[ty::t])
  53                            -> ~[ty::t] {

  59      })

  62  fn resolve_method_map_entry(fcx: @mut FnCtxt, sp: span, id: ast::NodeId) {
  63      // Resolve any method map entry
  64      match fcx.inh.method_map.find(&id) {
  65          None => {}
  66          Some(mme) => {
  67              {
  68                  let r = resolve_type_vars_in_type(fcx, sp, mme.self_ty);
  69                  for t in r.iter() {
  70                      let method_map = fcx.ccx.method_map;
  71                      let new_entry = method_map_entry { self_ty: *t, ..*mme };

  73                              new_entry=%?)",
  74                             id, new_entry);
  75                      method_map.insert(id, new_entry);
  76                  }
  77              }
  78          }
  79      }

 116  fn resolve_type_vars_for_node(wbcx: @mut WbCtxt, sp: span, id: ast::NodeId)
 117                             -> Option<ty::t> {
 118      let fcx = wbcx.fcx;
 119      let tcx = fcx.ccx.tcx;
 120  
 121      // Resolve any borrowings for the node with id `id`
 122      match fcx.inh.adjustments.find(&id) {
 123          None => (),
 124  
 125          Some(&@ty::AutoAddEnv(r, s)) => {
 126              match resolve_region(fcx.infcx(), r, resolve_all | force_all) {
 127                  Err(e) => {
 128                      // This should not, I think, happen:
 129                      fcx.ccx.tcx.sess.span_err(
 130                          sp, fmt!("cannot resolve bound for closure: %s",
 131                                   infer::fixup_err_to_str(e)));
 132                  }
 133                  Ok(r1) => {
 134                      let resolved_adj = @ty::AutoAddEnv(r1, s);
 135                      debug!("Adjustments for node %d: %?", id, resolved_adj);
 136                      fcx.tcx().adjustments.insert(id, resolved_adj);
 137                  }
 138              }
 139          }
 140  
 141          Some(&@ty::AutoDerefRef(adj)) => {
 142              let fixup_region = |r| {
 143                  match resolve_region(fcx.infcx(), r, resolve_all | force_all) {
 144                      Ok(r1) => r1,
 145                      Err(e) => {
 146                          // This should not, I think, happen.
 147                          fcx.ccx.tcx.sess.span_err(
 148                              sp, fmt!("cannot resolve scope of borrow: %s",
 149                                       infer::fixup_err_to_str(e)));
 150                          r
 151                      }
 152                  }
 153              };
 154  
 155              let resolved_autoref = match adj.autoref {
 156                  None => None,
 157                  Some(ref r) => Some(r.map_region(fixup_region))
 158              };
 159  
 160              let resolved_adj = @ty::AutoDerefRef(ty::AutoDerefRef {
 161                  autoderefs: adj.autoderefs,
 162                  autoref: resolved_autoref,
 163              });
 164              debug!("Adjustments for node %d: %?", id, resolved_adj);
 165              fcx.tcx().adjustments.insert(id, resolved_adj);
 166          }
 167      }
 168  
 169      // Resolve the type of the node with id `id`
 170      let n_ty = fcx.node_ty(id);
 171      match resolve_type_vars_in_type(fcx, sp, n_ty) {
 172        None => {
 173          wbcx.success = false;
 174          return None;
 175        }
 176  
 177        Some(t) => {
 178          debug!("resolve_type_vars_for_node(id=%d, n_ty=%s, t=%s)",
 179                 id, ppaux::ty_to_str(tcx, n_ty), ppaux::ty_to_str(tcx, t));
 180          write_ty_to_tcx(tcx, id, t);
 181          let mut ret = Some(t);
 182          do fcx.opt_node_ty_substs(id) |substs| {
 183            let mut new_tps = ~[];
 184            for subst in substs.tps.iter() {
 185                match resolve_type_vars_in_type(fcx, sp, *subst) {
 186                  Some(t) => new_tps.push(t),
 187                  None => { wbcx.success = false; ret = None; break }
 188                }
 189            }
 190            write_substs_to_tcx(tcx, id, new_tps);
 191            ret.is_some()
 192          };
 193          ret
 194        }
 195      }
 196  }

 219  fn visit_stmt(s: @ast::stmt, (wbcx, v): (@mut WbCtxt, wb_vt)) {
 220      if !wbcx.success { return; }
 221      resolve_type_vars_for_node(wbcx, s.span, ty::stmt_node_id(s));
 222      oldvisit::visit_stmt(s, (wbcx, v));
 223  }

 225  fn visit_expr(e: @ast::expr, (wbcx, v): (@mut WbCtxt, wb_vt)) {
 226      if !wbcx.success {
 227          return;
 228      }
 229  
 230      resolve_type_vars_for_node(wbcx, e.span, e.id);
 231  
 232      resolve_method_map_entry(wbcx.fcx, e.span, e.id);
 233      {
 234          let r = e.get_callee_id();
 235          for callee_id in r.iter() {
 236              resolve_method_map_entry(wbcx.fcx, e.span, *callee_id);
 237          }
 238      }
 239  
 240      resolve_vtable_map_entry(wbcx.fcx, e.span, e.id);
 241      {
 242          let r = e.get_callee_id();
 243          for callee_id in r.iter() {
 244              resolve_vtable_map_entry(wbcx.fcx, e.span, *callee_id);
 245          }
 246      }
 247  
 248      match e.node {
 249          ast::expr_fn_block(ref decl, _) => {
 250              for input in decl.inputs.iter() {
 251                  let _ = resolve_type_vars_for_node(wbcx, e.span, input.id);
 252              }
 253          }
 254  
 255          ast::expr_binary(callee_id, _, _, _) |
 256          ast::expr_unary(callee_id, _, _) |
 257          ast::expr_assign_op(callee_id, _, _, _) |
 258          ast::expr_index(callee_id, _, _) => {
 259              maybe_resolve_type_vars_for_node(wbcx, e.span, callee_id);
 260          }
 261  
 262          ast::expr_method_call(callee_id, _, _, _, _, _) => {
 263              // We must always have written in a callee ID type for these.
 264              resolve_type_vars_for_node(wbcx, e.span, callee_id);
 265          }
 266  
 267          _ => ()
 268      }
 269  
 270      oldvisit::visit_expr(e, (wbcx, v));
 271  }

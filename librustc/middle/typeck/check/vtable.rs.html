  34 // vtable resolution looks for places where trait bounds are
  35 // substituted in and figures out which vtable is used. There is some
  36 // extra complication thrown in to support early "opportunistic"
  37 // vtable resolution. This is a hacky mechanism that is invoked while
  38 // typechecking function calls (after typechecking non-closure
  39 // arguments and before typechecking closure arguments) in the hope of
  40 // solving for the trait parameters from the impl. (For example,
  41 // determining that if a parameter bounded by BaseIter<A> is
  42 // instantiated with Option<int>, that A = int.)
  43 //
  44 // In early resolution mode, no vtables are recorded, and a number of
  45 // errors are ignored. Early resolution only works if a type is
  46 // *fully* resolved. (We could be less restrictive than that, but it
  47 // would require much more care, and this seems to work decently in
  48 // practice.)
  49 //
  50 // While resolution on a single type requires the type to be fully
  51 // resolved, when resolving a substitution against a list of bounds,
  52 // we do not require all of the types to be resolved in advance.
  53 // Furthermore, we process substitutions in reverse order, which
  54 // allows resolution on later parameters to give information on
  55 // earlier params referenced by the typeclass bounds.
  56 // It may be better to do something more clever, like processing fully
  57 // resolved types first.
  58

  63 pub struct LocationInfo {
  64     span: span,
  65     id: ast::NodeId
  66 }

  79 fn has_trait_bounds(type_param_defs: &[ty::TypeParameterDef]) -> bool {
  80     type_param_defs.iter().any(
  81         |type_param_def| !type_param_def.bounds.trait_bounds.is_empty())
  82 }

  84 fn lookup_vtables(vcx: &VtableContext,
  85                   location_info: &LocationInfo,
  86                   type_param_defs: &[ty::TypeParameterDef],
  87                   substs: &ty::substs,
  88                   is_early: bool) -> vtable_res {
  89     debug!("lookup_vtables(location_info=%?, \
  90             type_param_defs=%s, \
  91             substs=%s",
  92            location_info,
  93            type_param_defs.repr(vcx.tcx()),
  94            substs.repr(vcx.tcx()));
  95     let _i = indenter();
  96 
  97 
  98     // We do this backwards for reasons discussed above.
  99     assert_eq!(substs.tps.len(), type_param_defs.len());
 100     let mut result =
 101         substs.tps.rev_iter()
 102         .zip(type_param_defs.rev_iter())
 103         .transform(|(ty, def)|
 104                    lookup_vtables_for_param(vcx, location_info, Some(substs),
 105                                             &*def.bounds, *ty, is_early))
 106         .to_owned_vec();
 107     result.reverse();
 108 
 109     assert_eq!(substs.tps.len(), result.len());
 110     debug!("lookup_vtables result(\
 111             location_info=%?, \
 112             type_param_defs=%s, \
 113             substs=%s, \
 114             result=%s)",
 115            location_info,
 116            type_param_defs.repr(vcx.tcx()),
 117            substs.repr(vcx.tcx()),
 118            result.repr(vcx.tcx()));
 119     @result
 120 }

 122 fn lookup_vtables_for_param(vcx: &VtableContext,
 123                             location_info: &LocationInfo,
 124                             // None for substs means the identity
 125                             substs: Option<&ty::substs>,
 126                             type_param_bounds: &ty::ParamBounds,
 127                             ty: ty::t,
 128                             is_early: bool) -> vtable_param_res {
 129     let tcx = vcx.tcx();
 130 
 131     // ty is the value supplied for the type parameter A...
 132     let mut param_result = ~[];
 133 
 134     do ty::each_bound_trait_and_supertraits(
 135         tcx, type_param_bounds.trait_bounds) |trait_ref|
 136     {
 137         // ...and here trait_ref is each bound that was declared on A,
 138         // expressed in terms of the type parameters.
 139 
 140         // Substitute the values of the type parameters that may
 141         // appear in the bound.
 142         let trait_ref = substs.map_default(trait_ref, |substs| {
 143             debug!("about to subst: %s, %s",
 144                    trait_ref.repr(tcx), substs.repr(tcx));
 145             trait_ref.subst(tcx, *substs)
 146         });
 147 
 148         debug!("after subst: %s", trait_ref.repr(tcx));
 149 
 150         match lookup_vtable(vcx, location_info, ty, trait_ref, is_early) {
 151             Some(vtable) => param_result.push(vtable),
 152             None => {
 153                 vcx.tcx().sess.span_fatal(
 154                     location_info.span,
 155                     fmt!("failed to find an implementation of \
 156                           trait %s for %s",
 157                          vcx.infcx.trait_ref_to_str(trait_ref),
 158                          vcx.infcx.ty_to_str(ty)));
 159             }
 160         }
 161         true
 162     };
 163 
 164     debug!("lookup_vtables_for_param result(\
 165             location_info=%?, \
 166             type_param_bounds=%s, \
 167             ty=%s, \
 168             result=%s)",
 169            location_info,
 170            type_param_bounds.repr(vcx.tcx()),
 171            ty.repr(vcx.tcx()),
 172            param_result.repr(vcx.tcx()));
 173 
 174     return @param_result;
 175 }

 177 fn relate_trait_refs(vcx: &VtableContext,
 178                      location_info: &LocationInfo,
 179                      act_trait_ref: @ty::TraitRef,
 180                      exp_trait_ref: @ty::TraitRef)
 181 {
 182     /*!
 183      *
 184      * Checks that an implementation of `act_trait_ref` is suitable
 185      * for use where `exp_trait_ref` is required and reports an
 186      * error otherwise.
 187      */
 188 
 189     match infer::mk_sub_trait_refs(vcx.infcx,
 190                                    false,
 191                                    infer::RelateTraitRefs(location_info.span),
 192                                    act_trait_ref,
 193                                    exp_trait_ref)
 194     {
 195         result::Ok(()) => {} // Ok.
 196         result::Err(ref err) => {
 197             // There is an error, but we need to do some work to make
 198             // the message good.
 199             // Resolve any type vars in the trait refs
 200             let r_act_trait_ref =
 201                 vcx.infcx.resolve_type_vars_in_trait_ref_if_possible(act_trait_ref);
 202             let r_exp_trait_ref =
 203                 vcx.infcx.resolve_type_vars_in_trait_ref_if_possible(exp_trait_ref);
 204             // Only print the message if there aren't any previous type errors
 205             // inside the types.
 206             if !ty::trait_ref_contains_error(&r_act_trait_ref) &&
 207                 !ty::trait_ref_contains_error(&r_exp_trait_ref)
 208             {
 209                 let tcx = vcx.tcx();
 210                 tcx.sess.span_err(
 211                     location_info.span,
 212                     fmt!("expected %s, but found %s (%s)",
 213                          ppaux::trait_ref_to_str(tcx, &r_exp_trait_ref),
 214                          ppaux::trait_ref_to_str(tcx, &r_act_trait_ref),
 215                          ty::type_err_to_str(tcx, err)));
 216             }
 217         }
 218     }
 219 }

 221 // Look up the vtable implementing the trait `trait_ref` at type `t`
 222 fn lookup_vtable(vcx: &VtableContext,
 223                  location_info: &LocationInfo,
 224                  ty: ty::t,
 225                  trait_ref: @ty::TraitRef,
 226                  is_early: bool)
 227     -> Option<vtable_origin>
 228 {
 229     debug!("lookup_vtable(ty=%s, trait_ref=%s)",
 230            vcx.infcx.ty_to_str(ty),
 231            vcx.infcx.trait_ref_to_str(trait_ref));
 232     let _i = indenter();
 233 
 234     let tcx = vcx.tcx();
 235 
 236     let ty = match fixup_ty(vcx, location_info, ty, is_early) {
 237         Some(ty) => ty,
 238         None => {
 239             // fixup_ty can only fail if this is early resolution
 240             assert!(is_early);
 241             // The type has unconstrained type variables in it, so we can't
 242             // do early resolution on it. Return some completely bogus vtable
 243             // information: we aren't storing it anyways.
 244             return Some(vtable_param(param_self, 0));
 245         }
 246     };
 247 
 248     // If the type is self or a param, we look at the trait/supertrait
 249     // bounds to see if they include the trait we are looking for.
 250     let vtable_opt = match ty::get(ty).sty {
 251         ty::ty_param(param_ty {idx: n, def_id: did}) => {
 252             let type_param_def = tcx.ty_param_defs.get(&did.node);
 253             lookup_vtable_from_bounds(vcx, location_info,
 254                                       type_param_def.bounds.trait_bounds,
 255                                       param_numbered(n),
 256                                       trait_ref)
 257         }
 258 
 259         ty::ty_self(trait_id) => {
 260             let self_trait_ref = ty::lookup_trait_def(tcx, trait_id).trait_ref;
 261             lookup_vtable_from_bounds(vcx, location_info,
 262                                       &[self_trait_ref],
 263                                       param_self,
 264                                       trait_ref)
 265         }
 266 
 267         // Default case just falls through
 268         _ => None
 269     };
 270 
 271     if vtable_opt.is_some() { return vtable_opt; }
 272 
 273     // If we aren't a self type or param, or it was, but we didn't find it,
 274     // do a search.
 275     return search_for_vtable(vcx, location_info,
 276                              ty, trait_ref, is_early)
 277 }
 278 
 279 // Given a list of bounds on a type, search those bounds to see if any
 280 // of them are the vtable we are looking for.
 281 fn lookup_vtable_from_bounds(vcx: &VtableContext,
 282                              location_info: &LocationInfo,
 283                              bounds: &[@ty::TraitRef],
 284                              param: param_index,
 285                              trait_ref: @ty::TraitRef)
 286     -> Option<vtable_origin> {
 287     let tcx = vcx.tcx();
 288 
 289     let mut n_bound = 0;
 290     let mut ret = None;
 291     do ty::each_bound_trait_and_supertraits(tcx, bounds) |bound_trait_ref| {
 292         debug!("checking bounds trait %s",
 293                bound_trait_ref.repr(vcx.tcx()));
 294 
 295         if bound_trait_ref.def_id == trait_ref.def_id {
 296             relate_trait_refs(vcx,
 297                               location_info,
 298                               bound_trait_ref,
 299                               trait_ref);
 300             let vtable = vtable_param(param, n_bound);
 301             debug!("found param vtable: %?",
 302                    vtable);
 303             ret = Some(vtable);
 304             false
 305         } else {
 306             n_bound += 1;
 307             true
 308         }
 309     };
 310     ret
 311 }

 313 fn search_for_vtable(vcx: &VtableContext,
 314                      location_info: &LocationInfo,
 315                      ty: ty::t,
 316                      trait_ref: @ty::TraitRef,
 317                      is_early: bool)
 318     -> Option<vtable_origin>
 319 {
 320     let tcx = vcx.tcx();
 321 
 322     let mut found = ~[];
 323     let mut impls_seen = HashSet::new();
 324 
 325     // XXX: this is a bad way to do this, since we do
 326     // pointless allocations.
 327     let impls = tcx.trait_impls.find(&trait_ref.def_id)
 328         .map_default(@mut ~[], |x| **x);
 329     // impls is the list of all impls in scope for trait_ref.
 330     for im in impls.iter() {
 331         // im is one specific impl of trait_ref.
 332 
 333         // First, ensure we haven't processed this impl yet.
 334         if impls_seen.contains(&im.did) {
 335             loop;
 336         }
 337         impls_seen.insert(im.did);
 338 
 339         // ty::impl_traits gives us the trait im implements.
 340         //
 341         // If foo implements a trait t, and if t is the same trait as
 342         // trait_ref, we need to unify it with trait_ref in order to
 343         // get all the ty vars sorted out.
 344         let r = ty::impl_trait_ref(tcx, im.did);
 345         let of_trait_ref = r.expect("trait_ref missing on trait impl");
 346         if of_trait_ref.def_id != trait_ref.def_id { loop; }
 347 
 348         // At this point, we know that of_trait_ref is the same trait
 349         // as trait_ref, but possibly applied to different substs.
 350         //
 351         // Next, we check whether the "for" ty in the impl is
 352         // compatible with the type that we're casting to a
 353         // trait. That is, if im is:
 354         //
 355         // impl<T> some_trait<T> for self_ty<T> { ... }
 356         //
 357         // we check whether self_ty<T> is the type of the thing that
 358         // we're trying to cast to some_trait.  If not, then we try
 359         // the next impl.
 360         //
 361         // XXX: document a bit more what this means
 362         //
 363         // FIXME(#5781) this should be mk_eqty not mk_subty
 364         let ty::ty_param_substs_and_ty {
 365             substs: substs,
 366             ty: for_ty
 367         } = impl_self_ty(vcx, location_info, im.did);
 368         match infer::mk_subty(vcx.infcx,
 369                               false,
 370                               infer::RelateSelfType(
 371                                   location_info.span),
 372                               ty,
 373                               for_ty) {
 374             result::Err(_) => loop,
 375             result::Ok(()) => ()
 376         }
 377 
 378         // Now, in the previous example, for_ty is bound to
 379         // the type self_ty, and substs is bound to [T].
 380         debug!("The self ty is %s and its substs are %s",
 381                vcx.infcx.ty_to_str(for_ty),
 382                vcx.infcx.tys_to_str(substs.tps));
 383 
 384         // Next, we unify trait_ref -- the type that we want to cast
 385         // to -- with of_trait_ref -- the trait that im implements. At
 386         // this point, we require that they be unifiable with each
 387         // other -- that's what relate_trait_refs does.
 388         //
 389         // For example, in the above example, of_trait_ref would be
 390         // some_trait<T>, so we would be unifying trait_ref<U> (for
 391         // some value of U) with some_trait<T>. This would fail if T
 392         // and U weren't compatible.
 393 
 394         debug!("(checking vtable) @2 relating trait \
 395                 ty %s to of_trait_ref %s",
 396                vcx.infcx.trait_ref_to_str(trait_ref),
 397                vcx.infcx.trait_ref_to_str(of_trait_ref));
 398 
 399         let of_trait_ref = of_trait_ref.subst(tcx, &substs);
 400         relate_trait_refs(vcx, location_info, of_trait_ref, trait_ref);
 401 
 402 
 403         // Recall that trait_ref -- the trait type we're casting to --
 404         // is the trait with id trait_ref.def_id applied to the substs
 405         // trait_ref.substs.
 406 
 407         // Resolve any sub bounds. Note that there still may be free
 408         // type variables in substs. This might still be OK: the
 409         // process of looking up bounds might constrain some of them.
 410         let im_generics =
 411             ty::lookup_item_type(tcx, im.did).generics;
 412         let subres = lookup_vtables(vcx, location_info,
 413                                     *im_generics.type_param_defs, &substs,
 414                                     is_early);
 415 
 416 
 417         // substs might contain type variables, so we call
 418         // fixup_substs to resolve them.
 419         let substs_f = match fixup_substs(vcx,
 420                                           location_info,
 421                                           trait_ref.def_id,
 422                                           substs,
 423                                           is_early) {
 424             Some(ref substs) => (*substs).clone(),
 425             None => {
 426                 assert!(is_early);
 427                 // Bail out with a bogus answer
 428                 return Some(vtable_param(param_self, 0));
 429             }
 430         };
 431 
 432         debug!("The fixed-up substs are %s - \

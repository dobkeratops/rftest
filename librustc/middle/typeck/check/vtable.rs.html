  34  // vtable resolution looks for places where trait bounds are
  35  // substituted in and figures out which vtable is used. There is some
  36  // extra complication thrown in to support early "opportunistic"
  37  // vtable resolution. This is a hacky mechanism that is invoked while
  38  // typechecking function calls (after typechecking non-closure
  39  // arguments and before typechecking closure arguments) in the hope of
  40  // solving for the trait parameters from the impl. (For example,
  41  // determining that if a parameter bounded by BaseIter<A> is
  42  // instantiated with Option<int>, that A = int.)
  43  //
  44  // In early resolution mode, no vtables are recorded, and a number of
  45  // errors are ignored. Early resolution only works if a type is
  46  // *fully* resolved. (We could be less restrictive than that, but it
  47  // would require much more care, and this seems to work decently in
  48  // practice.)
  49  //
  50  // While resolution on a single type requires the type to be fully
  51  // resolved, when resolving a substitution against a list of bounds,
  52  // we do not require all of the types to be resolved in advance.
  53  // Furthermore, we process substitutions in reverse order, which
  54  // allows resolution on later parameters to give information on
  55  // earlier params referenced by the typeclass bounds.
  56  // It may be better to do something more clever, like processing fully
  57  // resolved types first.
  58

  60  /// Location info records the span and ID of the expression or item that is
  61  /// responsible for this vtable instantiation. (This may not be an expression
  62  /// if the vtable instantiation is being performed as part of "deriving".)
  63  pub struct LocationInfo {
  64      span: span,
  65      id: ast::NodeId
  66  }

  79  fn has_trait_bounds(type_param_defs: &[ty::TypeParameterDef]) -> bool {
  80      type_param_defs.iter().any(
  81          |type_param_def| !type_param_def.bounds.trait_bounds.is_empty())
  82  }

  84  fn lookup_vtables(vcx: &VtableContext,
  85                    location_info: &LocationInfo,
  86                    type_param_defs: &[ty::TypeParameterDef],
  87                    substs: &ty::substs,
  88                    is_early: bool) -> vtable_res {
  89      debug!("lookup_vtables(location_info=%?, \
  90              type_param_defs=%s, \
  91              substs=%s",
  92             location_info,
  93             type_param_defs.repr(vcx.tcx()),
  94             substs.repr(vcx.tcx()));
  95      let _i = indenter();
  96  
  97  
  98      // We do this backwards for reasons discussed above.
  99      assert_eq!(substs.tps.len(), type_param_defs.len());
 100      let mut result =
 101          substs.tps.rev_iter()
 102          .zip(type_param_defs.rev_iter())
 103          .map(|(ty, def)|
 104                     lookup_vtables_for_param(vcx, location_info, Some(substs),
 105                                              &*def.bounds, *ty, is_early))
 106          .to_owned_vec();
 107      result.reverse();
 108  
 109      assert_eq!(substs.tps.len(), result.len());
 110      debug!("lookup_vtables result(\
 111              location_info=%?, \
 112              type_param_defs=%s, \
 113              substs=%s, \
 114              result=%s)",
 115             location_info,
 116             type_param_defs.repr(vcx.tcx()),
 117             substs.repr(vcx.tcx()),
 118             result.repr(vcx.tcx()));
 119      @result
 120  }

 122  fn lookup_vtables_for_param(vcx: &VtableContext,
 123                              location_info: &LocationInfo,
 124                              // None for substs means the identity
 125                              substs: Option<&ty::substs>,
 126                              type_param_bounds: &ty::ParamBounds,
 127                              ty: ty::t,
 128                              is_early: bool) -> vtable_param_res {
 129      let tcx = vcx.tcx();
 130  
 131      // ty is the value supplied for the type parameter A...
 132      let mut param_result = ~[];
 133  
 134      do ty::each_bound_trait_and_supertraits(tcx, type_param_bounds.trait_bounds) |trait_ref| {
 135          // ...and here trait_ref is each bound that was declared on A,
 136          // expressed in terms of the type parameters.
 137  
 138          // Substitute the values of the type parameters that may
 139          // appear in the bound.
 140          let trait_ref = substs.map_default(trait_ref, |substs| {
 141              debug!("about to subst: %s, %s",
 142                     trait_ref.repr(tcx), substs.repr(tcx));
 143              trait_ref.subst(tcx, *substs)
 144          });
 145  
 146          debug!("after subst: %s", trait_ref.repr(tcx));
 147  
 148          match lookup_vtable(vcx, location_info, ty, trait_ref, is_early) {
 149              Some(vtable) => param_result.push(vtable),
 150              None => {
 151                  vcx.tcx().sess.span_fatal(
 152                      location_info.span,
 153                      fmt!("failed to find an implementation of \
 154                            trait %s for %s",
 155                           vcx.infcx.trait_ref_to_str(trait_ref),
 156                           vcx.infcx.ty_to_str(ty)));
 157              }
 158          }
 159          true
 160      };
 161  
 162      debug!("lookup_vtables_for_param result(\
 163              location_info=%?, \
 164              type_param_bounds=%s, \
 165              ty=%s, \
 166              result=%s)",
 167             location_info,
 168             type_param_bounds.repr(vcx.tcx()),
 169             ty.repr(vcx.tcx()),
 170             param_result.repr(vcx.tcx()));
 171  
 172      return @param_result;
 173  }

 175  fn relate_trait_refs(vcx: &VtableContext,
 176                       location_info: &LocationInfo,
 177                       act_trait_ref: @ty::TraitRef,
 178                       exp_trait_ref: @ty::TraitRef)
 179  {
 180      /*!
 181       *
 182       * Checks that an implementation of `act_trait_ref` is suitable
 183       * for use where `exp_trait_ref` is required and reports an
 184       * error otherwise.
 185       */
 186  
 187      match infer::mk_sub_trait_refs(vcx.infcx,
 188                                     false,
 189                                     infer::RelateTraitRefs(location_info.span),
 190                                     act_trait_ref,
 191                                     exp_trait_ref)
 192      {
 193          result::Ok(()) => {} // Ok.
 194          result::Err(ref err) => {
 195              // There is an error, but we need to do some work to make
 196              // the message good.
 197              // Resolve any type vars in the trait refs
 198              let r_act_trait_ref =
 199                  vcx.infcx.resolve_type_vars_in_trait_ref_if_possible(act_trait_ref);
 200              let r_exp_trait_ref =
 201                  vcx.infcx.resolve_type_vars_in_trait_ref_if_possible(exp_trait_ref);
 202              // Only print the message if there aren't any previous type errors
 203              // inside the types.
 204              if !ty::trait_ref_contains_error(&r_act_trait_ref) &&
 205                  !ty::trait_ref_contains_error(&r_exp_trait_ref)
 206              {
 207                  let tcx = vcx.tcx();
 208                  tcx.sess.span_err(
 209                      location_info.span,
 210                      fmt!("expected %s, but found %s (%s)",
 211                           ppaux::trait_ref_to_str(tcx, &r_exp_trait_ref),
 212                           ppaux::trait_ref_to_str(tcx, &r_act_trait_ref),
 213                           ty::type_err_to_str(tcx, err)));
 214              }
 215          }
 216      }
 217  }

 219  // Look up the vtable implementing the trait `trait_ref` at type `t`
 220  fn lookup_vtable(vcx: &VtableContext,
 221                   location_info: &LocationInfo,
 222                   ty: ty::t,
 223                   trait_ref: @ty::TraitRef,
 224                   is_early: bool)
 225      -> Option<vtable_origin>
 226  {
 227      debug!("lookup_vtable(ty=%s, trait_ref=%s)",
 228             vcx.infcx.ty_to_str(ty),
 229             vcx.infcx.trait_ref_to_str(trait_ref));
 230      let _i = indenter();
 231  
 232      let tcx = vcx.tcx();
 233  
 234      let ty = match fixup_ty(vcx, location_info, ty, is_early) {
 235          Some(ty) => ty,
 236          None => {
 237              // fixup_ty can only fail if this is early resolution
 238              assert!(is_early);
 239              // The type has unconstrained type variables in it, so we can't
 240              // do early resolution on it. Return some completely bogus vtable
 241              // information: we aren't storing it anyways.
 242              return Some(vtable_param(param_self, 0));
 243          }
 244      };
 245  
 246      // If the type is self or a param, we look at the trait/supertrait
 247      // bounds to see if they include the trait we are looking for.
 248      let vtable_opt = match ty::get(ty).sty {
 249          ty::ty_param(param_ty {idx: n, def_id: did}) => {
 250              let type_param_def = tcx.ty_param_defs.get(&did.node);
 251              lookup_vtable_from_bounds(vcx, location_info,
 252                                        type_param_def.bounds.trait_bounds,
 253                                        param_numbered(n),
 254                                        trait_ref)
 255          }
 256  
 257          ty::ty_self(trait_id) => {
 258              let self_trait_ref = ty::lookup_trait_def(tcx, trait_id).trait_ref;
 259              lookup_vtable_from_bounds(vcx, location_info,
 260                                        &[self_trait_ref],
 261                                        param_self,
 262                                        trait_ref)
 263          }
 264  
 265          // Default case just falls through
 266          _ => None
 267      };
 268  
 269      if vtable_opt.is_some() { return vtable_opt; }
 270  
 271      // If we aren't a self type or param, or it was, but we didn't find it,
 272      // do a search.
 273      return search_for_vtable(vcx, location_info,
 274                               ty, trait_ref, is_early)
 275  }
 276  
 277  // Given a list of bounds on a type, search those bounds to see if any
 278  // of them are the vtable we are looking for.
 279  fn lookup_vtable_from_bounds(vcx: &VtableContext,
 280                               location_info: &LocationInfo,
 281                               bounds: &[@ty::TraitRef],
 282                               param: param_index,
 283                               trait_ref: @ty::TraitRef)
 284      -> Option<vtable_origin> {
 285      let tcx = vcx.tcx();
 286  
 287      let mut n_bound = 0;
 288      let mut ret = None;
 289      do ty::each_bound_trait_and_supertraits(tcx, bounds) |bound_trait_ref| {
 290          debug!("checking bounds trait %s",
 291                 bound_trait_ref.repr(vcx.tcx()));
 292  
 293          if bound_trait_ref.def_id == trait_ref.def_id {
 294              relate_trait_refs(vcx,
 295                                location_info,
 296                                bound_trait_ref,
 297                                trait_ref);
 298              let vtable = vtable_param(param, n_bound);
 299              debug!("found param vtable: %?",
 300                     vtable);
 301              ret = Some(vtable);
 302              false
 303          } else {
 304              n_bound += 1;
 305              true
 306          }
 307      };
 308      ret
 309  }

 311  fn search_for_vtable(vcx: &VtableContext,
 312                       location_info: &LocationInfo,
 313                       ty: ty::t,
 314                       trait_ref: @ty::TraitRef,
 315                       is_early: bool)
 316      -> Option<vtable_origin>
 317  {
 318      let tcx = vcx.tcx();
 319  
 320      let mut found = ~[];
 321      let mut impls_seen = HashSet::new();
 322  
 323      // XXX: this is a bad way to do this, since we do
 324      // pointless allocations.
 325      let impls = tcx.trait_impls.find(&trait_ref.def_id)
 326          .map_default(@mut ~[], |x| **x);
 327      // impls is the list of all impls in scope for trait_ref.
 328      for im in impls.iter() {
 329          // im is one specific impl of trait_ref.
 330  
 331          // First, ensure we haven't processed this impl yet.
 332          if impls_seen.contains(&im.did) {
 333              loop;
 334          }
 335          impls_seen.insert(im.did);
 336  
 337          // ty::impl_traits gives us the trait im implements.
 338          //
 339          // If foo implements a trait t, and if t is the same trait as
 340          // trait_ref, we need to unify it with trait_ref in order to
 341          // get all the ty vars sorted out.
 342          let r = ty::impl_trait_ref(tcx, im.did);
 343          let of_trait_ref = r.expect("trait_ref missing on trait impl");
 344          if of_trait_ref.def_id != trait_ref.def_id { loop; }
 345  
 346          // At this point, we know that of_trait_ref is the same trait
 347          // as trait_ref, but possibly applied to different substs.
 348          //
 349          // Next, we check whether the "for" ty in the impl is
 350          // compatible with the type that we're casting to a
 351          // trait. That is, if im is:
 352          //
 353          // impl<T> some_trait<T> for self_ty<T> { ... }
 354          //
 355          // we check whether self_ty<T> is the type of the thing that
 356          // we're trying to cast to some_trait.  If not, then we try
 357          // the next impl.
 358          //
 359          // XXX: document a bit more what this means
 360          //
 361          // FIXME(#5781) this should be mk_eqty not mk_subty
 362          let ty::ty_param_substs_and_ty {
 363              substs: substs,
 364              ty: for_ty
 365          } = impl_self_ty(vcx, location_info, im.did);
 366          match infer::mk_subty(vcx.infcx,
 367                                false,
 368                                infer::RelateSelfType(
 369                                    location_info.span),
 370                                ty,
 371                                for_ty) {
 372              result::Err(_) => loop,
 373              result::Ok(()) => ()
 374          }
 375  
 376          // Now, in the previous example, for_ty is bound to
 377          // the type self_ty, and substs is bound to [T].
 378          debug!("The self ty is %s and its substs are %s",
 379                 vcx.infcx.ty_to_str(for_ty),
 380                 vcx.infcx.tys_to_str(substs.tps));
 381  
 382          // Next, we unify trait_ref -- the type that we want to cast
 383          // to -- with of_trait_ref -- the trait that im implements. At
 384          // this point, we require that they be unifiable with each
 385          // other -- that's what relate_trait_refs does.
 386          //
 387          // For example, in the above example, of_trait_ref would be
 388          // some_trait<T>, so we would be unifying trait_ref<U> (for
 389          // some value of U) with some_trait<T>. This would fail if T
 390          // and U weren't compatible.
 391  
 392          debug!("(checking vtable) @2 relating trait \
 393                  ty %s to of_trait_ref %s",
 394                 vcx.infcx.trait_ref_to_str(trait_ref),
 395                 vcx.infcx.trait_ref_to_str(of_trait_ref));
 396  
 397          let of_trait_ref = of_trait_ref.subst(tcx, &substs);
 398          relate_trait_refs(vcx, location_info, of_trait_ref, trait_ref);
 399  
 400  
 401          // Recall that trait_ref -- the trait type we're casting to --
 402          // is the trait with id trait_ref.def_id applied to the substs
 403          // trait_ref.substs.
 404  
 405          // Resolve any sub bounds. Note that there still may be free
 406          // type variables in substs. This might still be OK: the
 407          // process of looking up bounds might constrain some of them.
 408          let im_generics =

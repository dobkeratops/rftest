  35  // vtable resolution looks for places where trait bounds are
  36  // substituted in and figures out which vtable is used. There is some
  37  // extra complication thrown in to support early "opportunistic"
  38  // vtable resolution. This is a hacky mechanism that is invoked while
  39  // typechecking function calls (after typechecking non-closure
  40  // arguments and before typechecking closure arguments) in the hope of
  41  // solving for the trait parameters from the impl. (For example,
  42  // determining that if a parameter bounded by BaseIter<A> is
  43  // instantiated with Option<int>, that A = int.)
  44  //
  45  // In early resolution mode, no vtables are recorded, and a number of
  46  // errors are ignored. Early resolution only works if a type is
  47  // *fully* resolved. (We could be less restrictive than that, but it
  48  // would require much more care, and this seems to work decently in
  49  // practice.)
  50  //
  51  // While resolution on a single type requires the type to be fully
  52  // resolved, when resolving a substitution against a list of bounds,
  53  // we do not require all of the types to be resolved in advance.
  54  // Furthermore, we process substitutions in reverse order, which
  55  // allows resolution on later parameters to give information on
  56  // earlier params referenced by the typeclass bounds.
  57  // It may be better to do something more clever, like processing fully
  58  // resolved types first.
  59

  61  /// Location info records the span and ID of the expression or item that is
  62  /// responsible for this vtable instantiation. (This may not be an expression
  63  /// if the vtable instantiation is being performed as part of "deriving".)
  64  pub struct LocationInfo {
  65      span: Span,
  66      id: ast::NodeId
  67  }

  80  fn has_trait_bounds(type_param_defs: &[ty::TypeParameterDef]) -> bool {
  81      type_param_defs.iter().any(
  82          |type_param_def| !type_param_def.bounds.trait_bounds.is_empty())
  83  }

  85  fn lookup_vtables(vcx: &VtableContext,
  86                    location_info: &LocationInfo,
  87                    type_param_defs: &[ty::TypeParameterDef],
  88                    substs: &ty::substs,
  89                    is_early: bool) -> vtable_res {
  90      debug!("lookup_vtables(location_info=%?, \
  91              type_param_defs=%s, \
  92              substs=%s",
  93             location_info,
  94             type_param_defs.repr(vcx.tcx()),
  95             substs.repr(vcx.tcx()));
  96      let _i = indenter();
  97  
  98  
  99      // We do this backwards for reasons discussed above.
 100      assert_eq!(substs.tps.len(), type_param_defs.len());
 101      let mut result =
 102          substs.tps.rev_iter()
 103          .zip(type_param_defs.rev_iter())
 104          .map(|(ty, def)|
 105                     lookup_vtables_for_param(vcx, location_info, Some(substs),
 106                                              &*def.bounds, *ty, is_early))
 107          .to_owned_vec();
 108      result.reverse();
 109  
 110      assert_eq!(substs.tps.len(), result.len());
 111      debug!("lookup_vtables result(\
 112              location_info=%?, \
 113              type_param_defs=%s, \
 114              substs=%s, \
 115              result=%s)",
 116             location_info,
 117             type_param_defs.repr(vcx.tcx()),
 118             substs.repr(vcx.tcx()),
 119             result.repr(vcx.tcx()));
 120      @result
 121  }

 123  fn lookup_vtables_for_param(vcx: &VtableContext,
 124                              location_info: &LocationInfo,
 125                              // None for substs means the identity
 126                              substs: Option<&ty::substs>,
 127                              type_param_bounds: &ty::ParamBounds,
 128                              ty: ty::t,
 129                              is_early: bool) -> vtable_param_res {
 130      let tcx = vcx.tcx();
 131  
 132      // ty is the value supplied for the type parameter A...
 133      let mut param_result = ~[];
 134  
 135      do ty::each_bound_trait_and_supertraits(tcx, type_param_bounds.trait_bounds) |trait_ref| {
 136          // ...and here trait_ref is each bound that was declared on A,
 137          // expressed in terms of the type parameters.
 138  
 139          ty::populate_implementations_for_trait_if_necessary(tcx,
 140                                                              trait_ref.def_id);
 141  
 142          // Substitute the values of the type parameters that may
 143          // appear in the bound.
 144          let trait_ref = substs.map_default(trait_ref, |substs| {
 145              debug!("about to subst: %s, %s",
 146                     trait_ref.repr(tcx), substs.repr(tcx));
 147              trait_ref.subst(tcx, *substs)
 148          });
 149  
 150          debug!("after subst: %s", trait_ref.repr(tcx));
 151  
 152          match lookup_vtable(vcx, location_info, ty, trait_ref, is_early) {
 153              Some(vtable) => param_result.push(vtable),
 154              None => {
 155                  vcx.tcx().sess.span_fatal(
 156                      location_info.span,
 157                      fmt!("failed to find an implementation of \
 158                            trait %s for %s",
 159                           vcx.infcx.trait_ref_to_str(trait_ref),
 160                           vcx.infcx.ty_to_str(ty)));
 161              }
 162          }
 163          true
 164      };
 165  
 166      debug!("lookup_vtables_for_param result(\
 167              location_info=%?, \
 168              type_param_bounds=%s, \
 169              ty=%s, \
 170              result=%s)",
 171             location_info,
 172             type_param_bounds.repr(vcx.tcx()),
 173             ty.repr(vcx.tcx()),
 174             param_result.repr(vcx.tcx()));
 175  
 176      return @param_result;
 177  }

 179  fn relate_trait_refs(vcx: &VtableContext,
 180                       location_info: &LocationInfo,
 181                       act_trait_ref: @ty::TraitRef,
 182                       exp_trait_ref: @ty::TraitRef)
 183  {
 184      /*!
 185       *
 186       * Checks that an implementation of `act_trait_ref` is suitable
 187       * for use where `exp_trait_ref` is required and reports an
 188       * error otherwise.
 189       */
 190  
 191      match infer::mk_sub_trait_refs(vcx.infcx,
 192                                     false,
 193                                     infer::RelateTraitRefs(location_info.span),
 194                                     act_trait_ref,
 195                                     exp_trait_ref)
 196      {
 197          result::Ok(()) => {} // Ok.
 198          result::Err(ref err) => {
 199              // There is an error, but we need to do some work to make
 200              // the message good.
 201              // Resolve any type vars in the trait refs
 202              let r_act_trait_ref =
 203                  vcx.infcx.resolve_type_vars_in_trait_ref_if_possible(act_trait_ref);
 204              let r_exp_trait_ref =
 205                  vcx.infcx.resolve_type_vars_in_trait_ref_if_possible(exp_trait_ref);
 206              // Only print the message if there aren't any previous type errors
 207              // inside the types.
 208              if !ty::trait_ref_contains_error(&r_act_trait_ref) &&
 209                  !ty::trait_ref_contains_error(&r_exp_trait_ref)
 210              {
 211                  let tcx = vcx.tcx();
 212                  tcx.sess.span_err(
 213                      location_info.span,
 214                      fmt!("expected %s, but found %s (%s)",
 215                           ppaux::trait_ref_to_str(tcx, &r_exp_trait_ref),
 216                           ppaux::trait_ref_to_str(tcx, &r_act_trait_ref),
 217                           ty::type_err_to_str(tcx, err)));
 218              }
 219          }
 220      }
 221  }

 223  // Look up the vtable implementing the trait `trait_ref` at type `t`
 224  fn lookup_vtable(vcx: &VtableContext,
 225                   location_info: &LocationInfo,
 226                   ty: ty::t,
 227                   trait_ref: @ty::TraitRef,
 228                   is_early: bool)
 229      -> Option<vtable_origin>
 230  {
 231      debug!("lookup_vtable(ty=%s, trait_ref=%s)",
 232             vcx.infcx.ty_to_str(ty),
 233             vcx.infcx.trait_ref_to_str(trait_ref));
 234      let _i = indenter();
 235  
 236      let tcx = vcx.tcx();
 237  
 238      let ty = match fixup_ty(vcx, location_info, ty, is_early) {
 239          Some(ty) => ty,
 240          None => {
 241              // fixup_ty can only fail if this is early resolution
 242              assert!(is_early);
 243              // The type has unconstrained type variables in it, so we can't
 244              // do early resolution on it. Return some completely bogus vtable
 245              // information: we aren't storing it anyways.
 246              return Some(vtable_param(param_self, 0));
 247          }
 248      };
 249  
 250      // If the type is self or a param, we look at the trait/supertrait
 251      // bounds to see if they include the trait we are looking for.
 252      let vtable_opt = match ty::get(ty).sty {
 253          ty::ty_param(param_ty {idx: n, def_id: did}) => {
 254              let type_param_def = tcx.ty_param_defs.get(&did.node);
 255              lookup_vtable_from_bounds(vcx, location_info,
 256                                        type_param_def.bounds.trait_bounds,
 257                                        param_numbered(n),
 258                                        trait_ref)
 259          }
 260  
 261          ty::ty_self(trait_id) => {
 262              let self_trait_ref = ty::lookup_trait_def(tcx, trait_id).trait_ref;
 263              lookup_vtable_from_bounds(vcx, location_info,
 264                                        &[self_trait_ref],
 265                                        param_self,
 266                                        trait_ref)
 267          }
 268  
 269          // Default case just falls through
 270          _ => None
 271      };
 272  
 273      if vtable_opt.is_some() { return vtable_opt; }
 274  
 275      // If we aren't a self type or param, or it was, but we didn't find it,
 276      // do a search.
 277      return search_for_vtable(vcx, location_info,
 278                               ty, trait_ref, is_early)
 279  }
 280  
 281  // Given a list of bounds on a type, search those bounds to see if any
 282  // of them are the vtable we are looking for.
 283  fn lookup_vtable_from_bounds(vcx: &VtableContext,
 284                               location_info: &LocationInfo,
 285                               bounds: &[@ty::TraitRef],
 286                               param: param_index,
 287                               trait_ref: @ty::TraitRef)
 288      -> Option<vtable_origin> {
 289      let tcx = vcx.tcx();
 290  
 291      let mut n_bound = 0;
 292      let mut ret = None;
 293      do ty::each_bound_trait_and_supertraits(tcx, bounds) |bound_trait_ref| {
 294          debug!("checking bounds trait %s",
 295                 bound_trait_ref.repr(vcx.tcx()));
 296  
 297          if bound_trait_ref.def_id == trait_ref.def_id {
 298              relate_trait_refs(vcx,
 299                                location_info,
 300                                bound_trait_ref,
 301                                trait_ref);
 302              let vtable = vtable_param(param, n_bound);
 303              debug!("found param vtable: %?",
 304                     vtable);
 305              ret = Some(vtable);
 306              false
 307          } else {
 308              n_bound += 1;
 309              true
 310          }
 311      };
 312      ret
 313  }

 315  fn search_for_vtable(vcx: &VtableContext,
 316                       location_info: &LocationInfo,
 317                       ty: ty::t,
 318                       trait_ref: @ty::TraitRef,
 319                       is_early: bool)
 320      -> Option<vtable_origin>
 321  {
 322      let tcx = vcx.tcx();
 323  
 324      let mut found = ~[];
 325      let mut impls_seen = HashSet::new();
 326  
 327      // Load the implementations from external metadata if necessary.
 328      ty::populate_implementations_for_trait_if_necessary(tcx,
 329                                                          trait_ref.def_id);
 330  
 331      // XXX: this is a bad way to do this, since we do
 332      // pointless allocations.
 333      let impls = tcx.trait_impls.find(&trait_ref.def_id)
 334          .map_default(@mut ~[], |x| **x);
 335      // impls is the list of all impls in scope for trait_ref.
 336      for im in impls.iter() {
 337          // im is one specific impl of trait_ref.
 338  
 339          // First, ensure we haven't processed this impl yet.
 340          if impls_seen.contains(&im.did) {
 341              loop;
 342          }
 343          impls_seen.insert(im.did);
 344  
 345          // ty::impl_traits gives us the trait im implements.
 346          //
 347          // If foo implements a trait t, and if t is the same trait as
 348          // trait_ref, we need to unify it with trait_ref in order to
 349          // get all the ty vars sorted out.
 350          let r = ty::impl_trait_ref(tcx, im.did);
 351          let of_trait_ref = r.expect("trait_ref missing on trait impl");
 352          if of_trait_ref.def_id != trait_ref.def_id { loop; }
 353  
 354          // At this point, we know that of_trait_ref is the same trait
 355          // as trait_ref, but possibly applied to different substs.
 356          //
 357          // Next, we check whether the "for" ty in the impl is
 358          // compatible with the type that we're casting to a
 359          // trait. That is, if im is:
 360          //
 361          // impl<T> some_trait<T> for self_ty<T> { ... }
 362          //
 363          // we check whether self_ty<T> is the type of the thing that
 364          // we're trying to cast to some_trait.  If not, then we try
 365          // the next impl.
 366          //
 367          // XXX: document a bit more what this means
 368          //
 369          // FIXME(#5781) this should be mk_eqty not mk_subty
 370          let ty::ty_param_substs_and_ty {
 371              substs: substs,
 372              ty: for_ty
 373          } = impl_self_ty(vcx, location_info, im.did);
 374          match infer::mk_subty(vcx.infcx,
 375                                false,
 376                                infer::RelateSelfType(
 377                                    location_info.span),
 378                                ty,
 379                                for_ty) {
 380              result::Err(_) => loop,
 381              result::Ok(()) => ()
 382          }
 383  
 384          // Now, in the previous example, for_ty is bound to
 385          // the type self_ty, and substs is bound to [T].
 386          debug!("The self ty is %s and its substs are %s",
 387                 vcx.infcx.ty_to_str(for_ty),
 388                 vcx.infcx.tys_to_str(substs.tps));
 389  
 390          // Next, we unify trait_ref -- the type that we want to cast
 391          // to -- with of_trait_ref -- the trait that im implements. At
 392          // this point, we require that they be unifiable with each
 393          // other -- that's what relate_trait_refs does.
 394          //
 395          // For example, in the above example, of_trait_ref would be
 396          // some_trait<T>, so we would be unifying trait_ref<U> (for
 397          // some value of U) with some_trait<T>. This would fail if T
 398          // and U weren't compatible.
 399  
 400          debug!("(checking vtable) @2 relating trait \
 401                  ty %s to of_trait_ref %s",
 402                 vcx.infcx.trait_ref_to_str(trait_ref),
 403                 vcx.infcx.trait_ref_to_str(of_trait_ref));
 404  
 405          let of_trait_ref = of_trait_ref.subst(tcx, &substs);
 406          relate_trait_refs(vcx, location_info, of_trait_ref, trait_ref);
 407  
 408  
 409          // Recall that trait_ref -- the trait type we're casting to --
 410          // is the trait with id trait_ref.def_id applied to the substs

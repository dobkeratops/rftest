   11  /*!
   12  
   13  # Method lookup
   14  
   15  Method lookup can be rather complex due to the interaction of a number
   16  of factors, such as self types, autoderef, trait lookup, etc.  The
   17  algorithm is divided into two parts: candidate collection and
   18  candidate selection.
   19  
   20  ## Candidate collection
   21  
   22  A `Candidate` is a method item that might plausibly be the method
   23  being invoked.  Candidates are grouped into two kinds, inherent and
   24  extension.  Inherent candidates are those that are derived from the
   25  type of the receiver itself.  So, if you have a receiver of some
   26  nominal type `Foo` (e.g., a struct), any methods defined within an
   27  impl like `impl Foo` are inherent methods.  Nothing needs to be
   28  imported to use an inherent method, they are associated with the type
   29  itself (note that inherent impls can only be defined in the same
   30  module as the type itself).
   31  
   32  Inherent candidates are not always derived from impls.  If you have a
   33  trait instance, such as a value of type `@ToStr`, then the trait
   34  methods (`to_str()`, in this case) are inherently associated with it.
   35  Another case is type parameters, in which case the methods of their
   36  bounds are inherent.
   37  
   38  Extension candidates are derived from imported traits.  If I have the
   39  trait `ToStr` imported, and I call `to_str()` on a value of type `T`,
   40  then we will go off to find out whether there is an impl of `ToStr`
   41  for `T`.  These kinds of method calls are called "extension methods".
   42  They can be defined in any module, not only the one that defined `T`.
   43  Furthermore, you must import the trait to call such a method.
   44  
   45  For better or worse, we currently give weight to inherent methods over
   46  extension methods during candidate selection (below).
   47  
   48  ## Candidate selection
   49  
   50  Once we know the set of candidates, we can go off and try to select
   51  which one is actually being called.  We do this by taking the type of
   52  the receiver, let's call it R, and checking whether it matches against
   53  the expected receiver type for each of the collected candidates.  We
   54  first check for inherent candidates and see whether we get exactly one
   55  match (zero means keep searching, more than one is an error).  If so,
   56  we return that as the candidate.  Otherwise we search the extension
   57  candidates in the same way.
   58  
   59  If find no matching candidate at all, we proceed to auto-deref the
   60  receiver type and search again.  We keep doing that until we cannot
   61  auto-deref any longer.  At each step, we also check for candidates
   62  based on "autoptr", which if the current type is `T`, checks for `&mut
   63  T`, `&const T`, and `&T` receivers.  Finally, at the very end, we will
   64  also try autoslice, which converts `~[]` to `&[]` (there is no point
   65  at trying autoslice earlier, because no autoderefable type is also
   66  sliceable).
   67  
   68  ## Why two phases?
   69  
   70  You might wonder why we first collect the candidates and then select.
   71  Both the inherent candidate collection and the candidate selection
   72  proceed by progressively deref'ing the receiver type, after all.  The
   73  answer is that two phases are needed to elegantly deal with explicit
   74  self.  After all, if there is an impl for the type `Foo`, it can
   75  define a method with the type `@self`, which means that it expects a
   76  receiver of type `@Foo`.  If we have a receiver of type `@Foo`, but we
   77  waited to search for that impl until we have deref'd the `@` away and
   78  obtained the type `Foo`, we would never match this method.
   79  
   80  */

  121  pub fn lookup(
  122          fcx: @mut FnCtxt,
  123  
  124          // In a call `a.b::<X, Y, ...>(...)`:
  125          expr: @ast::expr,                   // The expression `a.b(...)`.
  126          self_expr: @ast::expr,              // The expression `a`.
  127          callee_id: NodeId,                  /* Where to store `a.b`'s type,
  128                                               * also the scope of the call */
  129          m_name: ast::ident,                 // The ident `b`.
  130          self_ty: ty::t,                     // The type of `a`.
  131          supplied_tps: &[ty::t],             // The list of types X, Y, ... .
  132          deref_args: check::DerefArgs,       // Whether we autopointer first.
  133          check_traits: CheckTraitsFlag,      // Whether we check traits only.
  134          autoderef_receiver: AutoderefReceiverFlag)
  135       -> Option<method_map_entry> {
  136      let impl_dups = @mut HashSet::new();
  137      let lcx = LookupContext {
  138          fcx: fcx,
  139          expr: expr,
  140          self_expr: self_expr,
  141          callee_id: callee_id,
  142          m_name: m_name,
  143          supplied_tps: supplied_tps,
  144          impl_dups: impl_dups,
  145          inherent_candidates: @mut ~[],
  146          extension_candidates: @mut ~[],
  147          deref_args: deref_args,
  148          check_traits: check_traits,
  149          autoderef_receiver: autoderef_receiver,
  150      };
  151  
  152      let self_ty = structurally_resolved_type(fcx, self_expr.span, self_ty);
  153      debug!("method lookup(self_ty=%s, expr=%s, self_expr=%s)",
  154             self_ty.repr(fcx.tcx()), expr.repr(fcx.tcx()),
  155             self_expr.repr(fcx.tcx()));
  156  
  157      debug!("searching inherent candidates");
  158      lcx.push_inherent_candidates(self_ty);
  159      let mme = lcx.search(self_ty);
  160      if mme.is_some() {
  161          return mme;
  162      }
  163  
  164      debug!("searching extension candidates");
  165      lcx.reset_candidates();
  166      lcx.push_bound_candidates(self_ty);
  167      lcx.push_extension_candidates();
  168      return lcx.search(self_ty);
  169  }

  186  /**
  187   * A potential method that might be called, assuming the receiver
  188   * is of a suitable type.
  189   */
  190  #[deriving(Clone)]
  191  pub struct Candidate {
  192      rcvr_match_condition: RcvrMatchCondition,
  193      rcvr_substs: ty::substs,
  194      method_ty: @ty::Method,
  195      origin: method_origin,
  196  }

  198  /// This type represents the conditions under which the receiver is
  199  /// considered to "match" a given method candidate. Typically the test
  200  /// is whether the receiver is of a particular type. However, this
  201  /// type is the type of the receiver *after accounting for the
  202  /// method's self type* (e.g., if the method is an `@self` method, we
  203  /// have *already verified* that the receiver is of some type `@T` and
  204  /// now we must check that the type `T` is correct).  Unfortunately,
  205  /// because traits are not types, this is a pain to do.
  206  #[deriving(Clone)]
  207  enum RcvrMatchCondition {
  208      RcvrMatchesIfObject(ast::def_id),
  209      RcvrMatchesIfSubtype(ty::t)
  210  }

  212  impl<'self> LookupContext<'self> {
  213      fn search(&self, self_ty: ty::t) -> Option<method_map_entry> {
  214          let mut self_ty = self_ty;
  215          let mut autoderefs = 0;
  216          loop {
  217              debug!("loop: self_ty=%s autoderefs=%u",
  218                     self.ty_to_str(self_ty), autoderefs);
  219  
  220              match self.deref_args {
  221                  check::DontDerefArgs => {
  222                      match self.search_for_autoderefd_method(self_ty,
  223                                                              autoderefs) {
  224                          Some(mme) => { return Some(mme); }
  225                          None => {}
  226                      }
  227  
  228                      match self.search_for_autoptrd_method(self_ty,
  229                                                            autoderefs) {
  230                          Some(mme) => { return Some(mme); }
  231                          None => {}
  232                      }
  233                  }
  234                  check::DoDerefArgs => {
  235                      match self.search_for_autoptrd_method(self_ty,
  236                                                            autoderefs) {
  237                          Some(mme) => { return Some(mme); }
  238                          None => {}
  239                      }
  240  
  241                      match self.search_for_autoderefd_method(self_ty,
  242                                                              autoderefs) {
  243                          Some(mme) => { return Some(mme); }
  244                          None => {}
  245                      }
  246                  }
  247              }
  248  
  249              // Don't autoderef if we aren't supposed to.
  250              if self.autoderef_receiver == DontAutoderefReceiver {
  251                  break;
  252              }
  253  
  254              // Otherwise, perform autoderef.
  255              match self.deref(self_ty) {
  256                  None => { break; }
  257                  Some(ty) => {
  258                      self_ty = ty;
  259                      autoderefs += 1;
  260                  }
  261              }
  262          }
  263  
  264          self.search_for_autosliced_method(self_ty, autoderefs)
  265      }
  266  
  267      fn deref(&self, ty: ty::t)
  268                   -> Option<ty::t> {
  269          match ty::deref(self.tcx(), ty, false) {
  270              None => None,
  271              Some(t) => {
  272                  Some(structurally_resolved_type(self.fcx,
  273                                                  self.self_expr.span,
  274                                                  t.ty))
  275              }
  276          }
  277      }

  282      fn reset_candidates(&self) {
  283          *self.inherent_candidates = ~[];
  284          *self.extension_candidates = ~[];
  285      }
  286  
  287      fn push_inherent_candidates(&self, self_ty: ty::t) {
  288          /*!
  289           * Collect all inherent candidates into
  290           * `self.inherent_candidates`.  See comment at the start of
  291           * the file.  To find the inherent candidates, we repeatedly
  292           * deref the self-ty to find the "base-type".  So, for
  293           * example, if the receiver is @@C where `C` is a struct type,
  294           * we'll want to find the inherent impls for `C`.
  295           */
  296  
  297          let mut self_ty = self_ty;
  298          loop {
  299              match get(self_ty).sty {
  300                  ty_trait(did, ref substs, _, _, _) => {
  301                      self.push_inherent_candidates_from_trait(did, substs);
  302                      self.push_inherent_impl_candidates_for_type(did);
  303                  }
  304                  ty_enum(did, _) | ty_struct(did, _) => {
  305                      if self.check_traits == CheckTraitsAndInherentMethods {
  306                          self.push_inherent_impl_candidates_for_type(did);
  307                      }
  308                  }
  309                  _ => { /* No inherent methods in these types */ }
  310              }
  311  
  312              // n.b.: Generally speaking, we only loop if we hit the
  313              // fallthrough case in the match above.  The exception
  314              // would be newtype enums.
  315              self_ty = match self.deref(self_ty) {
  316                  None => { return; }
  317                  Some(ty) => { ty }
  318              }
  319          }
  320      }
  321  
  322      fn push_bound_candidates(&self, self_ty: ty::t) {
  323          let mut self_ty = self_ty;
  324          loop {
  325              match get(self_ty).sty {
  326                  ty_param(p) => {
  327                      self.push_inherent_candidates_from_param(self_ty, p);
  328                  }
  329                  ty_self(self_did) => {
  330                      // Call is of the form "self.foo()" and appears in one
  331                      // of a trait's default method implementations.
  332                      self.push_inherent_candidates_from_self(
  333                          self_ty, self_did);
  334                  }
  335                  _ => { /* No bound methods in these types */ }
  336              }
  337  
  338              self_ty = match self.deref(self_ty) {
  339                  None => { return; }
  340                  Some(ty) => { ty }
  341              }
  342          }
  343      }
  344  
  345      fn push_extension_candidates(&self) {
  346          // If the method being called is associated with a trait, then
  347          // find all the impls of that trait.  Each of those are
  348          // candidates.
  349          let trait_map: &mut resolve::TraitMap = &mut self.fcx.ccx.trait_map;
  350          let opt_applicable_traits = trait_map.find(&self.expr.id);
  351          for applicable_traits in opt_applicable_traits.iter() {
  352              for trait_did in applicable_traits.iter() {
  353                  // Look for explicit implementations.
  354                  let opt_impl_infos = self.tcx().trait_impls.find(trait_did);
  355                  for impl_infos in opt_impl_infos.iter() {
  356                      for impl_info in impl_infos.iter() {
  357                          self.push_candidates_from_impl(
  358                              self.extension_candidates, *impl_info);
  359  
  360                      }
  361                  }
  362              }
  363          }
  364      }

  366      fn push_inherent_candidates_from_trait(&self,
  367                                                 did: def_id,
  368                                                 substs: &ty::substs) {
  369          debug!("push_inherent_candidates_from_trait(did=%s, substs=%s)",
  370                 self.did_to_str(did),
  371                 substs_to_str(self.tcx(), substs));
  372          let _indenter = indenter();
  373  
  374          let tcx = self.tcx();
  375          let ms = ty::trait_methods(tcx, did);
  376          let index = match ms.iter().position(|m| m.ident == self.m_name) {
  377              Some(i) => i,
  378              None => { return; } // no method with the right name
  379          };
  380          let method = ms[index];
  381  
  382          match method.explicit_self {
  383              ast::sty_static => {
  384                  return; // not a method we can call with dot notation
  385              }
  386              _ => {}
  387          }
  388  
  389          // It is illegal to invoke a method on a trait instance that
  390          // refers to the `self` type. An error will be reported by
  391          // `enforce_object_limitations()` if the method refers
  392          // to the `Self` type. Substituting ty_err here allows
  393          // compiler to soldier on.
  394          //
  395          // NOTE: `confirm_candidate()` also relies upon this substitution
  396          // for Self. (fix)
  397          let rcvr_substs = substs {
  398              self_ty: Some(ty::mk_err()),
  399              ..(*substs).clone()
  400          };
  401

  561      fn consider_reborrow(&self,
  562                               self_ty: ty::t,
  563                               autoderefs: uint)
  564                               -> (ty::t, ty::AutoAdjustment) {
  565          /*!
  566           * In the event that we are invoking a method with a receiver
  567           * of a borrowed type like `&T`, `&mut T`, or `&mut [T]`,
  568           * we will "reborrow" the receiver implicitly.  For example, if
  569           * you have a call `r.inc()` and where `r` has type `&mut T`,
  570           * then we treat that like `(&mut *r).inc()`.  This avoids
  571           * consuming the original pointer.
  572           *
  573           * You might think that this would be a natural byproduct of
  574           * the auto-deref/auto-ref process.  This is true for `@mut T`
  575           * but not for an `&mut T` receiver.  With `@mut T`, we would
  576           * begin by testing for methods with a self type `@mut T`,
  577           * then autoderef to `T`, then autoref to `&mut T`.  But with
  578           * an `&mut T` receiver the process begins with `&mut T`, only
  579           * without any autoadjustments.
  580           */
  581  
  582          let tcx = self.tcx();
  583          return match ty::get(self_ty).sty {
  584              ty::ty_rptr(_, self_mt) if default_method_hack(self_mt) => {
  585                  (self_ty,
  586                   ty::AutoDerefRef(ty::AutoDerefRef {
  587                       autoderefs: autoderefs,
  588                       autoref: None}))
  589              }
  590              ty::ty_rptr(_, self_mt) => {
  591                  let region =
  592                      self.infcx().next_region_var(
  593                          infer::Autoref(self.expr.span));
  594                  (ty::mk_rptr(tcx, region, self_mt),
  595                   ty::AutoDerefRef(ty::AutoDerefRef {
  596                       autoderefs: autoderefs+1,
  597                       autoref: Some(ty::AutoPtr(region, self_mt.mutbl))}))
  598              }
  599              ty::ty_evec(self_mt, vstore_slice(_)) => {
  600                  let region =
  601                      self.infcx().next_region_var(
  602                          infer::Autoref(self.expr.span));
  603                  (ty::mk_evec(tcx, self_mt, vstore_slice(region)),
  604                   ty::AutoDerefRef(ty::AutoDerefRef {
  605                       autoderefs: autoderefs,
  606                       autoref: Some(ty::AutoBorrowVec(region, self_mt.mutbl))}))
  607              }
  608              ty_trait(did, ref substs, ty::RegionTraitStore(_), mutbl, bounds) => {
  609                  let region =
  610                      self.infcx().next_region_var(
  611                          infer::Autoref(self.expr.span));
  612                  (ty::mk_trait(tcx, did, substs.clone(),
  613                                ty::RegionTraitStore(region),
  614                                mutbl, bounds),
  615                   ty::AutoDerefRef(ty::AutoDerefRef {
  616                       autoderefs: autoderefs,
  617                       autoref: Some(ty::AutoBorrowObj(region, mutbl))}))
  618              }
  619              _ => {
  620                  (self_ty,
  621                   ty::AutoDerefRef(ty::AutoDerefRef {
  622                       autoderefs: autoderefs,
  623                       autoref: None}))
  624              }
  625          };

  628              // FIXME(#6129). Default methods can't deal with autoref.
  629              //
  630              // I am a horrible monster and I pray for death. Currently
  631              // the default method code fails when you try to reborrow
  632              // because it is not handling types correctly. In lieu of
  633              // fixing that, I am introducing this horrible hack. - ndm

  638      fn search_for_autosliced_method(&self,
  639                                          self_ty: ty::t,
  640                                          autoderefs: uint)
  641                                          -> Option<method_map_entry> {
  642          /*!
  643           *
  644           * Searches for a candidate by converting things like
  645           * `~[]` to `&[]`. */
  646  
  647          let tcx = self.tcx();
  648          let sty = ty::get(self_ty).sty.clone();
  649          match sty {
  650              ty_evec(mt, vstore_box) |
  651              ty_evec(mt, vstore_uniq) |
  652              ty_evec(mt, vstore_slice(_)) | // NDM(#3148)
  653              ty_evec(mt, vstore_fixed(_)) => {
  654                  // First try to borrow to a slice
  655                  let entry = self.search_for_some_kind_of_autorefd_method(
  656                      AutoBorrowVec, autoderefs, [m_const, m_imm, m_mutbl],
  657                      |m,r| ty::mk_evec(tcx,
  658                                        ty::mt {ty:mt.ty, mutbl:m},
  659                                        vstore_slice(r)));
  660  
  661                  if entry.is_some() { return entry; }
  662  
  663                  // Then try to borrow to a slice *and* borrow a pointer.
  664                  self.search_for_some_kind_of_autorefd_method(
  665                      AutoBorrowVecRef, autoderefs, [m_const, m_imm, m_mutbl],
  666                      |m,r| {
  667                          let slice_ty = ty::mk_evec(tcx,
  668                                                     ty::mt {ty:mt.ty, mutbl:m},
  669                                                     vstore_slice(r));
  670                          // NB: we do not try to autoref to a mutable
  671                          // pointer. That would be creating a pointer
  672                          // to a temporary pointer (the borrowed
  673                          // slice), so any update the callee makes to
  674                          // it can't be observed.
  675                          ty::mk_rptr(tcx, r, ty::mt {ty:slice_ty, mutbl:m_imm})
  676                      })
  677              }
  678  
  679              ty_estr(vstore_box) |
  680              ty_estr(vstore_uniq) |
  681              ty_estr(vstore_fixed(_)) => {
  682                  let entry = self.search_for_some_kind_of_autorefd_method(
  683                      AutoBorrowVec, autoderefs, [m_imm],
  684                      |_m,r| ty::mk_estr(tcx, vstore_slice(r)));
  685  
  686                  if entry.is_some() { return entry; }
  687  
  688                  self.search_for_some_kind_of_autorefd_method(
  689                      AutoBorrowVecRef, autoderefs, [m_imm],
  690                      |m,r| {
  691                          let slice_ty = ty::mk_estr(tcx, vstore_slice(r));
  692                          ty::mk_rptr(tcx, r, ty::mt {ty:slice_ty, mutbl:m})
  693                      })
  694              }
  695  
  696              ty_trait(trt_did, trt_substs, _, _, b) => {
  697                  // Coerce ~/@/&Trait instances to &Trait.
  698  
  699                  self.search_for_some_kind_of_autorefd_method(
  700                      AutoBorrowObj, autoderefs, [m_const, m_imm, m_mutbl],
  701                      |trt_mut, reg| {
  702                          ty::mk_trait(tcx, trt_did, trt_substs.clone(),
  703                                       RegionTraitStore(reg), trt_mut, b)
  704                      })
  705              }
  706  
  707              ty_closure(*) => {
  708                  // This case should probably be handled similarly to
  709                  // Trait instances.
  710                  None
  711              }
  712  
  713              _ => None
  714          }
  715      }

  806      fn consider_candidates(&self,
  807                                 rcvr_ty: ty::t,
  808                                 candidates: &mut ~[Candidate])
  809                                 -> Option<method_map_entry> {
  810          // XXX(pcwalton): Do we need to clone here?
  811          let relevant_candidates: ~[Candidate] =
  812              candidates.iter().map(|c| (*c).clone()).
  813                  filter(|c| self.is_relevant(rcvr_ty, c)).collect();
  814  
  815          let relevant_candidates = self.merge_candidates(relevant_candidates);
  816  
  817          if relevant_candidates.len() == 0 {
  818              return None;
  819          }
  820  
  821          if relevant_candidates.len() > 1 {
  822              self.tcx().sess.span_err(
  823                  self.expr.span,
  824                  "multiple applicable methods in scope");
  825              for (idx, candidate) in relevant_candidates.iter().enumerate() {
  826                  self.report_candidate(idx, &candidate.origin);
  827              }
  828          }
  829  
  830          Some(self.confirm_candidate(rcvr_ty, &relevant_candidates[0]))
  831      }

  833      fn merge_candidates(&self, candidates: &[Candidate]) -> ~[Candidate] {
  834          let mut merged = ~[];
  835          let mut i = 0;
  836          while i < candidates.len() {
  837              let candidate_a = &candidates[i];
  838  
  839              let mut skip = false;
  840  
  841              let mut j = i + 1;
  842              while j < candidates.len() {
  843                  let candidate_b = &candidates[j];
  844                  debug!("attempting to merge %? and %?",
  845                         candidate_a, candidate_b);
  846                  let candidates_same = match (&candidate_a.origin,
  847                                               &candidate_b.origin) {
  848                      (&method_param(ref p1), &method_param(ref p2)) => {
  849                          let same_trait = p1.trait_id == p2.trait_id;
  850                          let same_method = p1.method_num == p2.method_num;
  851                          let same_param = p1.param_num == p2.param_num;
  852                          // The bound number may be different because
  853                          // multiple bounds may lead to the same trait
  854                          // impl
  855                          same_trait && same_method && same_param
  856                      }
  857                      _ => false
  858                  };
  859                  if candidates_same {
  860                      skip = true;
  861                      break;
  862                  }
  863                  j += 1;
  864              }
  865  
  866              i += 1;
  867  
  868              if skip {
  869                  // There are more than one of these and we need only one
  870                  loop;
  871              } else {
  872                  merged.push(candidate_a.clone());
  873              }
  874          }
  875  
  876          return merged;
  877      }

  879      fn confirm_candidate(&self, rcvr_ty: ty::t, candidate: &Candidate)
  880                               -> method_map_entry {
  881          let tcx = self.tcx();
  882          let fty = self.fn_ty_from_origin(&candidate.origin);
  883  
  884          debug!("confirm_candidate(expr=%s, candidate=%s, fty=%s)",
  885                 self.expr.repr(tcx),
  886                 self.cand_to_str(candidate),
  887                 self.ty_to_str(fty));
  888  
  889          self.enforce_object_limitations(fty, candidate);
  890          self.enforce_drop_trait_limitations(candidate);
  891  
  892          // static methods should never have gotten this far:
  893          assert!(candidate.method_ty.explicit_self != sty_static);
  894  
  895          let transformed_self_ty = match candidate.origin {
  896              method_trait(trait_def_id, _) => {
  897                  self.construct_transformed_self_ty_for_object(
  898                      trait_def_id, candidate)
  899              }
  900              _ => {
  901                  let t = candidate.method_ty.transformed_self_ty.unwrap();
  902                  ty::subst(tcx, &candidate.rcvr_substs, t)
  903              }
  904          };
  905  
  906          // Determine the values for the type parameters of the method.
  907          // If they were not explicitly supplied, just construct fresh
  908          // type variables.
  909          let num_supplied_tps = self.supplied_tps.len();
  910          let num_method_tps = candidate.method_ty.generics.type_param_defs.len();
  911          let m_substs = {
  912              if num_supplied_tps == 0u {
  913                  self.fcx.infcx().next_ty_vars(num_method_tps)
  914              } else if num_method_tps == 0u {
  915                  tcx.sess.span_err(
  916                      self.expr.span,
  917                      "this method does not take type parameters");
  918                  self.fcx.infcx().next_ty_vars(num_method_tps)
  919              } else if num_supplied_tps != num_method_tps {
  920                  tcx.sess.span_err(
  921                      self.expr.span,
  922                      "incorrect number of type \
  923                       parameters given for this method");
  924                  self.fcx.infcx().next_ty_vars(num_method_tps)
  925              } else {
  926                  self.supplied_tps.to_owned()
  927              }
  928          };
  929  
  930          // Construct the full set of type parameters for the method,
  931          // which is equal to the class tps + the method tps.
  932          let all_substs = substs {
  933              tps: vec::append(candidate.rcvr_substs.tps.clone(), m_substs),
  934              regions: candidate.rcvr_substs.regions.clone(),
  935              self_ty: candidate.rcvr_substs.self_ty,
  936          };
  937  
  938          // Compute the method type with type parameters substituted
  939          debug!("fty=%s all_substs=%s",
  940                 self.ty_to_str(fty),
  941                 ty::substs_to_str(tcx, &all_substs));
  942          let fty = ty::subst(tcx, &all_substs, fty);
  943          debug!("after subst, fty=%s", self.ty_to_str(fty));
  944  
  945          // Replace any bound regions that appear in the function
  946          // signature with region variables
  947          let bare_fn_ty = match ty::get(fty).sty {
  948              ty::ty_bare_fn(ref f) => f,
  949              ref s => {
  950                  tcx.sess.span_bug(
  951                      self.expr.span,
  952                      fmt!("Invoking method with non-bare-fn ty: %?", s));
  953              }
  954          };
  955          let (_, opt_transformed_self_ty, fn_sig) =
  956              replace_bound_regions_in_fn_sig(
  957                  tcx, @Nil, Some(transformed_self_ty), &bare_fn_ty.sig,
  958                  |br| self.fcx.infcx().next_region_var(
  959                      infer::BoundRegionInFnCall(self.expr.span, br)));
  960          let transformed_self_ty = opt_transformed_self_ty.unwrap();
  961          let fty = ty::mk_bare_fn(tcx, ty::BareFnTy {
  962              sig: fn_sig,
  963              purity: bare_fn_ty.purity,
  964              abis: bare_fn_ty.abis.clone(),
  965          });
  966          debug!("after replacing bound regions, fty=%s", self.ty_to_str(fty));
  967  
  968          let self_mode = get_mode_from_explicit_self(candidate.method_ty.explicit_self);
  969  
  970          // before we only checked whether self_ty could be a subtype
  971          // of rcvr_ty; now we actually make it so (this may cause
  972          // variables to unify etc).  Since we checked beforehand, and
  973          // nothing has changed in the meantime, this unification
  974          // should never fail.
  975          match self.fcx.mk_subty(false, infer::Misc(self.self_expr.span),
  976                                  rcvr_ty, transformed_self_ty) {
  977              result::Ok(_) => (),
  978              result::Err(_) => {
  979                  self.bug(fmt!("%s was a subtype of %s but now is not?",
  980                                self.ty_to_str(rcvr_ty),
  981                                self.ty_to_str(transformed_self_ty)));
  982              }
  983          }
  984  
  985          self.fcx.write_ty(self.callee_id, fty);
  986          self.fcx.write_substs(self.callee_id, all_substs);
  987          method_map_entry {
  988              self_ty: transformed_self_ty,
  989              self_mode: self_mode,
  990              explicit_self: candidate.method_ty.explicit_self,
  991              origin: candidate.origin,
  992          }
  993      }

  995      fn construct_transformed_self_ty_for_object(&self,
  996                                                  trait_def_id: ast::def_id,
  997                                                  candidate: &Candidate) -> ty::t
  998      {
  999          /*!
 1000           * This is a bit tricky. We have a match against a trait method
 1001           * being invoked on an object, and we want to generate the
 1002           * self-type. As an example, consider a trait
 1003           *
 1004           *     trait Foo {
 1005           *         fn r_method<'a>(&'a self);
 1006           *         fn m_method(@mut self);
 1007           *     }
 1008           *
 1009           * Now, assuming that `r_method` is being called, we want the
 1010           * result to be `&'a Foo`. Assuming that `m_method` is being
 1011           * called, we want the result to be `@mut Foo`. Of course,
 1012           * this transformation has already been done as part of
 1013           * `candidate.method_ty.transformed_self_ty`, but there the
 1014           * type is expressed in terms of `Self` (i.e., `&'a Self`, `@mut Self`).
 1015           * Because objects are not standalone types, we can't just substitute
 1016           * `s/Self/Foo/`, so we must instead perform this kind of hokey
 1017           * match below.
 1018           */
 1019  
 1020          let substs = ty::substs {regions: candidate.rcvr_substs.regions.clone(),
 1021                                   self_ty: None,
 1022                                   tps: candidate.rcvr_substs.tps.clone()};
 1023          match candidate.method_ty.explicit_self {
 1024              ast::sty_static => {
 1025                  self.bug(~"static method for object type receiver");
 1026              }
 1027              ast::sty_value => {
 1028                  ty::mk_err() // error reported in `enforce_object_limitations()`
 1029              }
 1030              ast::sty_region(*) | ast::sty_box(*) | ast::sty_uniq(*) => {
 1031                  let transformed_self_ty =
 1032                      candidate.method_ty.transformed_self_ty.clone().unwrap();
 1033                  match ty::get(transformed_self_ty).sty {
 1034                      ty::ty_rptr(r, mt) => { // must be sty_region
 1035                          ty::mk_trait(self.tcx(), trait_def_id,
 1036                                       substs, RegionTraitStore(r), mt.mutbl,
 1037                                       ty::EmptyBuiltinBounds())
 1038                      }
 1039                      ty::ty_box(mt) => { // must be sty_box
 1040                          ty::mk_trait(self.tcx(), trait_def_id,
 1041                                       substs, BoxTraitStore, mt.mutbl,
 1042                                       ty::EmptyBuiltinBounds())
 1043                      }
 1044                      ty::ty_uniq(mt) => { // must be sty_uniq
 1045                          ty::mk_trait(self.tcx(), trait_def_id,
 1046                                       substs, UniqTraitStore, mt.mutbl,
 1047                                       ty::EmptyBuiltinBounds())
 1048                      }
 1049                      _ => {
 1050                          self.bug(
 1051                              fmt!("'impossible' transformed_self_ty: %s",
 1052                                   transformed_self_ty.repr(self.tcx())));
 1053                      }
 1054                  }
 1055              }
 1056          }
 1057      }

 1059      fn enforce_object_limitations(&self,
 1060                                    method_fty: ty::t,
 1061                                    candidate: &Candidate)
 1062      {
 1063          /*!
 1064           * There are some limitations to calling functions through an
 1065           * object, because (a) the self type is not known
 1066           * (that's the whole point of a trait instance, after all, to
 1067           * obscure the self type) and (b) the call must go through a
 1068           * vtable and hence cannot be monomorphized.
 1069           */
 1070  
 1071          match candidate.origin {
 1072              method_static(*) | method_param(*) => {
 1073                  return; // not a call to a trait instance
 1074              }
 1075              method_trait(*) => {}
 1076          }
 1077  
 1078          match candidate.method_ty.explicit_self {
 1079              ast::sty_static => { // reason (a) above
 1080                  self.tcx().sess.span_err(
 1081                      self.expr.span,
 1082                      "cannot call a method without a receiver \
 1083                       through an object");
 1084              }
 1085  
 1086              ast::sty_value => { // reason (a) above
 1087                  self.tcx().sess.span_err(
 1088                      self.expr.span,
 1089                      "cannot call a method with a by-value receiver \
 1090                       through an object");
 1091              }
 1092  
 1093              ast::sty_region(*) | ast::sty_box(*) | ast::sty_uniq(*) => {}
 1094          }
 1095  
 1096          if ty::type_has_self(method_fty) { // reason (a) above
 1097              self.tcx().sess.span_err(
 1098                  self.expr.span,
 1099                  "cannot call a method whose type contains a \
 1100                   self-type through an object");
 1101          }
 1102  
 1103          if candidate.method_ty.generics.has_type_params() { // reason (b) above
 1104              self.tcx().sess.span_err(
 1105                  self.expr.span,
 1106                  "cannot call a generic method through an object");
 1107          }
 1108      }

 1110      fn enforce_drop_trait_limitations(&self, candidate: &Candidate) {
 1111          // No code can call the finalize method explicitly.
 1112          let bad;
 1113          match candidate.origin {
 1114              method_static(method_id) => {
 1115                  bad = self.tcx().destructors.contains(&method_id);
 1116              }
 1117              // XXX: does this properly enforce this on everything now
 1118              // that self has been merged in? -sully
 1119              method_param(method_param { trait_id: trait_id, _ }) |
 1120              method_trait(trait_id, _) => {
 1121                  bad = self.tcx().destructor_for_type.contains_key(&trait_id);
 1122              }
 1123          }
 1124  
 1125          if bad {
 1126              self.tcx().sess.span_err(self.expr.span,
 1127                                       "explicit call to destructor");
 1128          }
 1129      }
 1130  
 1131      // `rcvr_ty` is the type of the expression. It may be a subtype of a
 1132      // candidate method's `self_ty`.
 1133      fn is_relevant(&self, rcvr_ty: ty::t, candidate: &Candidate) -> bool {
 1134          debug!("is_relevant(rcvr_ty=%s, candidate=%s)",
 1135                 self.ty_to_str(rcvr_ty), self.cand_to_str(candidate));
 1136  
 1137          return match candidate.method_ty.explicit_self {
 1138              sty_static => {
 1139                  debug!("(is relevant?) explicit self is static");
 1140                  false
 1141              }
 1142  
 1143              sty_value => {
 1144                  rcvr_matches_ty(self.fcx, rcvr_ty, candidate)
 1145              }
 1146  
 1147              sty_region(_, m) => {
 1148                  debug!("(is relevant?) explicit self is a region");
 1149                  match ty::get(rcvr_ty).sty {
 1150                      ty::ty_rptr(_, mt) => {
 1151                          mutability_matches(mt.mutbl, m) &&
 1152                          rcvr_matches_ty(self.fcx, mt.ty, candidate)
 1153                      }
 1154  
 1155                      ty::ty_trait(self_did, _, RegionTraitStore(_), self_m, _) => {
 1156                          mutability_matches(self_m, m) &&
 1157                          rcvr_matches_object(self_did, candidate)
 1158                      }
 1159  
 1160                      _ => false
 1161                  }
 1162              }
 1163  
 1164              sty_box(m) => {
 1165                  debug!("(is relevant?) explicit self is a box");
 1166                  match ty::get(rcvr_ty).sty {
 1167                      ty::ty_box(mt) => {
 1168                          mutability_matches(mt.mutbl, m) &&
 1169                          rcvr_matches_ty(self.fcx, mt.ty, candidate)
 1170                      }
 1171  
 1172                      ty::ty_trait(self_did, _, BoxTraitStore, self_m, _) => {
 1173                          mutability_matches(self_m, m) &&
 1174                          rcvr_matches_object(self_did, candidate)
 1175                      }
 1176  
 1177                      _ => false
 1178                  }
 1179              }
 1180  
 1181              sty_uniq => {
 1182                  debug!("(is relevant?) explicit self is a unique pointer");
 1183                  match ty::get(rcvr_ty).sty {
 1184                      ty::ty_uniq(mt) => {
 1185                          rcvr_matches_ty(self.fcx, mt.ty, candidate)
 1186                      }
 1187  
 1188                      ty::ty_trait(self_did, _, UniqTraitStore, _, _) => {
 1189                          rcvr_matches_object(self_did, candidate)
 1190                      }
 1191  
 1192                      _ => false
 1193                  }
 1194              }
 1195          };

 1197          fn rcvr_matches_object(self_did: ast::def_id,
 1198                                 candidate: &Candidate) -> bool {
 1199              match candidate.rcvr_match_condition {
 1200                  RcvrMatchesIfObject(desired_did) => {
 1201                      self_did == desired_did
 1202                  }
 1203                  RcvrMatchesIfSubtype(_) => {
 1204                      false
 1205                  }
 1206              }
 1207          }

 1211                             candidate: &Candidate) -> bool {
 1212              match candidate.rcvr_match_condition {
 1213                  RcvrMatchesIfObject(_) => {
 1214                      false
 1215                  }
 1216                  RcvrMatchesIfSubtype(of_type) => {
 1217                      fcx.can_mk_subty(rcvr_ty, of_type).is_ok()
 1218                  }
 1219              }
 1220          }

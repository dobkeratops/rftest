   11  /*!
   12  
   13  # Method lookup
   14  
   15  Method lookup can be rather complex due to the interaction of a number
   16  of factors, such as self types, autoderef, trait lookup, etc.  The
   17  algorithm is divided into two parts: candidate collection and
   18  candidate selection.
   19  
   20  ## Candidate collection
   21  
   22  A `Candidate` is a method item that might plausibly be the method
   23  being invoked.  Candidates are grouped into two kinds, inherent and
   24  extension.  Inherent candidates are those that are derived from the
   25  type of the receiver itself.  So, if you have a receiver of some
   26  nominal type `Foo` (e.g., a struct), any methods defined within an
   27  impl like `impl Foo` are inherent methods.  Nothing needs to be
   28  imported to use an inherent method, they are associated with the type
   29  itself (note that inherent impls can only be defined in the same
   30  module as the type itself).
   31  
   32  Inherent candidates are not always derived from impls.  If you have a
   33  trait instance, such as a value of type `@ToStr`, then the trait
   34  methods (`to_str()`, in this case) are inherently associated with it.
   35  Another case is type parameters, in which case the methods of their
   36  bounds are inherent.
   37  
   38  Extension candidates are derived from imported traits.  If I have the
   39  trait `ToStr` imported, and I call `to_str()` on a value of type `T`,
   40  then we will go off to find out whether there is an impl of `ToStr`
   41  for `T`.  These kinds of method calls are called "extension methods".
   42  They can be defined in any module, not only the one that defined `T`.
   43  Furthermore, you must import the trait to call such a method.
   44  
   45  For better or worse, we currently give weight to inherent methods over
   46  extension methods during candidate selection (below).
   47  
   48  ## Candidate selection
   49  
   50  Once we know the set of candidates, we can go off and try to select
   51  which one is actually being called.  We do this by taking the type of
   52  the receiver, let's call it R, and checking whether it matches against
   53  the expected receiver type for each of the collected candidates.  We
   54  first check for inherent candidates and see whether we get exactly one
   55  match (zero means keep searching, more than one is an error).  If so,
   56  we return that as the candidate.  Otherwise we search the extension
   57  candidates in the same way.
   58  
   59  If find no matching candidate at all, we proceed to auto-deref the
   60  receiver type and search again.  We keep doing that until we cannot
   61  auto-deref any longer.  At each step, we also check for candidates
   62  based on "autoptr", which if the current type is `T`, checks for `&mut
   63  T`, `&const T`, and `&T` receivers.  Finally, at the very end, we will
   64  also try autoslice, which converts `~[]` to `&[]` (there is no point
   65  at trying autoslice earlier, because no autoderefable type is also
   66  sliceable).
   67  
   68  ## Why two phases?
   69  
   70  You might wonder why we first collect the candidates and then select.
   71  Both the inherent candidate collection and the candidate selection
   72  proceed by progressively deref'ing the receiver type, after all.  The
   73  answer is that two phases are needed to elegantly deal with explicit
   74  self.  After all, if there is an impl for the type `Foo`, it can
   75  define a method with the type `@self`, which means that it expects a
   76  receiver of type `@Foo`.  If we have a receiver of type `@Foo`, but we
   77  waited to search for that impl until we have deref'd the `@` away and
   78  obtained the type `Foo`, we would never match this method.
   79  
   80  */

  120  pub fn lookup(
  121          fcx: @mut FnCtxt,
  122  
  123          // In a call `a.b::<X, Y, ...>(...)`:
  124          expr: @ast::expr,                   // The expression `a.b(...)`.
  125          self_expr: @ast::expr,              // The expression `a`.
  126          callee_id: NodeId,                  /* Where to store `a.b`'s type,
  127                                               * also the scope of the call */
  128          m_name: ast::ident,                 // The ident `b`.
  129          self_ty: ty::t,                     // The type of `a`.
  130          supplied_tps: &[ty::t],             // The list of types X, Y, ... .
  131          deref_args: check::DerefArgs,       // Whether we autopointer first.
  132          check_traits: CheckTraitsFlag,      // Whether we check traits only.
  133          autoderef_receiver: AutoderefReceiverFlag)
  134       -> Option<method_map_entry> {
  135      let impl_dups = @mut HashSet::new();
  136      let lcx = LookupContext {
  137          fcx: fcx,
  138          expr: expr,
  139          self_expr: self_expr,
  140          callee_id: callee_id,
  141          m_name: m_name,
  142          supplied_tps: supplied_tps,
  143          impl_dups: impl_dups,
  144          inherent_candidates: @mut ~[],
  145          extension_candidates: @mut ~[],
  146          deref_args: deref_args,
  147          check_traits: check_traits,
  148          autoderef_receiver: autoderef_receiver,
  149      };
  150      let mme = lcx.do_lookup(self_ty);
  151      debug!("method lookup for %s yielded %?", expr.repr(fcx.tcx()), mme);
  152      return mme;
  153  }

  182  impl<'self> LookupContext<'self> {
  183      pub fn do_lookup(&self, self_ty: ty::t) -> Option<method_map_entry> {
  184          let self_ty = structurally_resolved_type(self.fcx,
  185                                                       self.self_expr.span,
  186                                                       self_ty);
  187  
  188          debug!("do_lookup(self_ty=%s, expr=%s, self_expr=%s)",
  189                 self.ty_to_str(self_ty),
  190                 self.expr.repr(self.tcx()),
  191                 self.self_expr.repr(self.tcx()));
  192  
  193          // Prepare the list of candidates
  194          self.push_inherent_candidates(self_ty);
  195          self.push_extension_candidates();
  196  
  197          let mut self_ty = self_ty;
  198          let mut autoderefs = 0;
  199          loop {
  200              debug!("loop: self_ty=%s autoderefs=%u",
  201                     self.ty_to_str(self_ty), autoderefs);
  202  
  203              match self.deref_args {
  204                  check::DontDerefArgs => {
  205                      match self.search_for_autoderefd_method(self_ty,
  206                                                              autoderefs) {
  207                          Some(mme) => { return Some(mme); }
  208                          None => {}
  209                      }
  210  
  211                      match self.search_for_autoptrd_method(self_ty,
  212                                                            autoderefs) {
  213                          Some(mme) => { return Some(mme); }
  214                          None => {}
  215                      }
  216                  }
  217                  check::DoDerefArgs => {
  218                      match self.search_for_autoptrd_method(self_ty,
  219                                                            autoderefs) {
  220                          Some(mme) => { return Some(mme); }
  221                          None => {}
  222                      }
  223  
  224                      match self.search_for_autoderefd_method(self_ty,
  225                                                              autoderefs) {
  226                          Some(mme) => { return Some(mme); }
  227                          None => {}
  228                      }
  229                  }
  230              }
  231  
  232              // Don't autoderef if we aren't supposed to.
  233              if self.autoderef_receiver == DontAutoderefReceiver {
  234                  break;
  235              }
  236  
  237              // Otherwise, perform autoderef.
  238              match self.deref(self_ty) {
  239                  None => { break; }
  240                  Some(ty) => {
  241                      self_ty = ty;
  242                      autoderefs += 1;
  243                  }
  244              }
  245          }
  246  
  247          self.search_for_autosliced_method(self_ty, autoderefs)
  248      }

  250      pub fn deref(&self, ty: ty::t)
  251                   -> Option<ty::t> {
  252          match ty::deref(self.tcx(), ty, false) {
  253              None => None,
  254              Some(t) => {
  255                  Some(structurally_resolved_type(self.fcx,
  256                                                  self.self_expr.span,
  257                                                  t.ty))
  258              }
  259          }
  260      }

  263      // Candidate collection (see comment at start of file)
  264  
  265      pub fn push_inherent_candidates(&self, self_ty: ty::t) {
  266          /*!
  267           * Collect all inherent candidates into
  268           * `self.inherent_candidates`.  See comment at the start of
  269           * the file.  To find the inherent candidates, we repeatedly
  270           * deref the self-ty to find the "base-type".  So, for
  271           * example, if the receiver is @@C where `C` is a struct type,
  272           * we'll want to find the inherent impls for `C`.
  273           */
  274  
  275          let mut self_ty = self_ty;
  276          loop {
  277              match get(self_ty).sty {
  278                  ty_param(p) => {
  279                      self.push_inherent_candidates_from_param(self_ty, p);
  280                  }
  281                  ty_trait(did, ref substs, store, _, _) => {
  282                      self.push_inherent_candidates_from_trait(
  283                          self_ty, did, substs, store);
  284                      self.push_inherent_impl_candidates_for_type(did);
  285                  }
  286                  ty_self(self_did) => {
  287                      // Call is of the form "self.foo()" and appears in one
  288                      // of a trait's default method implementations.
  289                      self.push_inherent_candidates_from_self(
  290                          self_ty, self_did);
  291                  }
  292                  ty_enum(did, _) | ty_struct(did, _) => {
  293                      if self.check_traits == CheckTraitsAndInherentMethods {
  294                          self.push_inherent_impl_candidates_for_type(did);
  295                      }
  296                  }
  297                  _ => { /* No inherent methods in these types */ }
  298              }
  299  
  300              // n.b.: Generally speaking, we only loop if we hit the
  301              // fallthrough case in the match above.  The exception
  302              // would be newtype enums.
  303              self_ty = match self.deref(self_ty) {
  304                  None => { return; }
  305                  Some(ty) => { ty }
  306              }
  307          }
  308      }

  310      pub fn push_extension_candidates(&self) {
  311          // If the method being called is associated with a trait, then
  312          // find all the impls of that trait.  Each of those are
  313          // candidates.
  314          let trait_map: &mut resolve::TraitMap = &mut self.fcx.ccx.trait_map;
  315          let opt_applicable_traits = trait_map.find(&self.expr.id);
  316          for applicable_traits in opt_applicable_traits.iter() {
  317              for trait_did in applicable_traits.iter() {
  318                  // Look for explicit implementations.
  319                  let opt_impl_infos = self.tcx().trait_impls.find(trait_did);
  320                  for impl_infos in opt_impl_infos.iter() {
  321                      for impl_info in impl_infos.iter() {
  322                          self.push_candidates_from_impl(
  323                              self.extension_candidates, *impl_info);
  324  
  325                      }
  326                  }
  327              }
  328          }
  329      }

  331      pub fn push_inherent_candidates_from_trait(&self,
  332                                                 self_ty: ty::t,
  333                                                 did: def_id,
  334                                                 substs: &ty::substs,
  335                                                 store: ty::TraitStore) {
  336          debug!("push_inherent_candidates_from_trait(did=%s, substs=%s)",
  337                 self.did_to_str(did),
  338                 substs_to_str(self.tcx(), substs));
  339          let _indenter = indenter();
  340  
  341          let tcx = self.tcx();
  342          let ms = ty::trait_methods(tcx, did);
  343          let index = match ms.iter().position(|m| m.ident == self.m_name) {
  344              Some(i) => i,
  345              None => { return; } // no method with the right name
  346          };
  347          let method = ms[index];
  348  
  349          /* FIXME(#5762) we should transform the vstore in accordance
  350             with the self type
  351  
  352          match method.self_type {
  353              ast::sty_region(_) => {
  354                  return; // inapplicable
  355              }
  356              ast::sty_region(_) => vstore_slice(r)
  357              ast::sty_box(_) => vstore_box, // NDM mutability, as per #5762
  358              ast::sty_uniq(_) => vstore_uniq
  359          }
  360          */
  361  
  362          // It is illegal to invoke a method on a trait instance that
  363          // refers to the `self` type.  Nonetheless, we substitute
  364          // `trait_ty` for `self` here, because it allows the compiler
  365          // to soldier on.  An error will be reported should this
  366          // candidate be selected if the method refers to `self`.
  367          //
  368          // NB: `confirm_candidate()` also relies upon this substitution
  369          // for Self.
  370          let rcvr_substs = substs {
  371              self_ty: Some(self_ty),
  372              ..(*substs).clone()
  373          };
  374  
  375          self.inherent_candidates.push(Candidate {
  376              rcvr_ty: self_ty,
  377              rcvr_substs: rcvr_substs,
  378              method_ty: method,
  379              origin: method_trait(did, index, store)
  380          });
  381      }

  383      pub fn push_inherent_candidates_from_param(&self,
  384                                                 rcvr_ty: ty::t,
  385                                                 param_ty: param_ty) {
  386          debug!("push_inherent_candidates_from_param(param_ty=%?)",
  387                 param_ty);
  388          let _indenter = indenter();
  389  
  390          let tcx = self.tcx();
  391          let type_param_def = match tcx.ty_param_defs.find(&param_ty.def_id.node) {
  392              Some(t) => t,
  393              None => {
  394                  tcx.sess.span_bug(
  395                      self.expr.span,
  396                      fmt!("No param def for %?", param_ty));
  397              }
  398          };
  399  
  400          self.push_inherent_candidates_from_bounds(
  401              rcvr_ty, type_param_def.bounds.trait_bounds,
  402              param_numbered(param_ty.idx));
  403      }

  534      pub fn consider_reborrow(&self,
  535                               self_ty: ty::t,
  536                               autoderefs: uint)
  537                               -> (ty::t, ty::AutoAdjustment) {
  538          /*!
  539           *
  540           * In the event that we are invoking a method with a receiver
  541           * of a borrowed type like `&T`, `&mut T`, or `&mut [T]`,
  542           * we will "reborrow" the receiver implicitly.  For example, if
  543           * you have a call `r.inc()` and where `r` has type `&mut T`,
  544           * then we treat that like `(&mut *r).inc()`.  This avoids
  545           * consuming the original pointer.
  546           *
  547           * You might think that this would be a natural byproduct of
  548           * the auto-deref/auto-ref process.  This is true for `@mut T`
  549           * but not for an `&mut T` receiver.  With `@mut T`, we would
  550           * begin by testing for methods with a self type `@mut T`,
  551           * then autoderef to `T`, then autoref to `&mut T`.  But with
  552           * an `&mut T` receiver the process begins with `&mut T`, only
  553           * without any autoadjustments.
  554           */
  555  
  556          let tcx = self.tcx();
  557          return match ty::get(self_ty).sty {
  558              ty::ty_rptr(_, self_mt) if default_method_hack(self_mt) => {
  559                  (self_ty,
  560                   ty::AutoDerefRef(ty::AutoDerefRef {
  561                       autoderefs: autoderefs,
  562                       autoref: None}))
  563              }
  564              ty::ty_rptr(_, self_mt) => {
  565                  let region =
  566                      self.infcx().next_region_var(
  567                          infer::Autoref(self.expr.span));
  568                  (ty::mk_rptr(tcx, region, self_mt),
  569                   ty::AutoDerefRef(ty::AutoDerefRef {
  570                       autoderefs: autoderefs+1,
  571                       autoref: Some(ty::AutoPtr(region, self_mt.mutbl))}))
  572              }
  573              ty::ty_evec(self_mt, vstore_slice(_)) => {
  574                  let region =
  575                      self.infcx().next_region_var(
  576                          infer::Autoref(self.expr.span));
  577                  (ty::mk_evec(tcx, self_mt, vstore_slice(region)),
  578                   ty::AutoDerefRef(ty::AutoDerefRef {
  579                       autoderefs: autoderefs,
  580                       autoref: Some(ty::AutoBorrowVec(region, self_mt.mutbl))}))
  581              }
  582              _ => {
  583                  (self_ty,
  584                   ty::AutoDerefRef(ty::AutoDerefRef {
  585                       autoderefs: autoderefs,
  586                       autoref: None}))
  587              }
  588          };
  589  
  590          fn default_method_hack(self_mt: ty::mt) -> bool {
  591              // FIXME(#6129). Default methods can't deal with autoref.
  592              //
  593              // I am a horrible monster and I pray for death. Currently
  594              // the default method code fails when you try to reborrow
  595              // because it is not handling types correctly. In lieu of
  596              // fixing that, I am introducing this horrible hack. - ndm
  597              self_mt.mutbl == m_imm && ty::type_is_self(self_mt.ty)
  598          }
  599      }

  601      pub fn search_for_autosliced_method(&self,
  602                                          self_ty: ty::t,
  603                                          autoderefs: uint)
  604                                          -> Option<method_map_entry> {
  605          /*!
  606           *
  607           * Searches for a candidate by converting things like
  608           * `~[]` to `&[]`. */
  609  
  610          let tcx = self.tcx();
  611          match ty::get(self_ty).sty {
  612              ty_evec(mt, vstore_box) |
  613              ty_evec(mt, vstore_uniq) |
  614              ty_evec(mt, vstore_slice(_)) | // NDM(#3148)
  615              ty_evec(mt, vstore_fixed(_)) => {
  616                  // First try to borrow to a slice
  617                  let entry = self.search_for_some_kind_of_autorefd_method(
  618                      AutoBorrowVec, autoderefs, [m_const, m_imm, m_mutbl],
  619                      |m,r| ty::mk_evec(tcx,
  620                                        ty::mt {ty:mt.ty, mutbl:m},
  621                                        vstore_slice(r)));
  622  
  623                  if entry.is_some() { return entry; }
  624  
  625                  // Then try to borrow to a slice *and* borrow a pointer.
  626                  self.search_for_some_kind_of_autorefd_method(
  627                      AutoBorrowVecRef, autoderefs, [m_const, m_imm, m_mutbl],
  628                      |m,r| {
  629                          let slice_ty = ty::mk_evec(tcx,
  630                                                     ty::mt {ty:mt.ty, mutbl:m},
  631                                                     vstore_slice(r));
  632                          // NB: we do not try to autoref to a mutable
  633                          // pointer. That would be creating a pointer
  634                          // to a temporary pointer (the borrowed
  635                          // slice), so any update the callee makes to
  636                          // it can't be observed.
  637                          ty::mk_rptr(tcx, r, ty::mt {ty:slice_ty, mutbl:m_imm})
  638                      })
  639              }
  640  
  641              ty_estr(vstore_box) |
  642              ty_estr(vstore_uniq) |
  643              ty_estr(vstore_fixed(_)) => {
  644                  let entry = self.search_for_some_kind_of_autorefd_method(
  645                      AutoBorrowVec, autoderefs, [m_imm],
  646                      |_m,r| ty::mk_estr(tcx, vstore_slice(r)));
  647  
  648                  if entry.is_some() { return entry; }
  649  
  650                  self.search_for_some_kind_of_autorefd_method(
  651                      AutoBorrowVecRef, autoderefs, [m_imm],
  652                      |m,r| {
  653                          let slice_ty = ty::mk_estr(tcx, vstore_slice(r));
  654                          ty::mk_rptr(tcx, r, ty::mt {ty:slice_ty, mutbl:m})
  655                      })
  656              }
  657  
  658              ty_trait(*) | ty_closure(*) => {
  659                  // NDM---eventually these should be some variant of autoref
  660                  None
  661              }
  662  
  663              _ => None
  664          }
  665      }

  756      pub fn consider_candidates(&self,
  757                                 rcvr_ty: ty::t,
  758                                 candidates: &mut ~[Candidate])
  759                                 -> Option<method_map_entry> {
  760          // XXX(pcwalton): Do we need to clone here?
  761          let relevant_candidates: ~[Candidate] =
  762              candidates.iter().map(|c| (*c).clone()).
  763                  filter(|c| self.is_relevant(rcvr_ty, c)).collect();
  764  
  765          let relevant_candidates = self.merge_candidates(relevant_candidates);
  766  
  767          if relevant_candidates.len() == 0 {
  768              return None;
  769          }
  770  
  771          if relevant_candidates.len() > 1 {
  772              self.tcx().sess.span_err(
  773                  self.expr.span,
  774                  "multiple applicable methods in scope");
  775              for (idx, candidate) in relevant_candidates.iter().enumerate() {
  776                  self.report_candidate(idx, &candidate.origin);
  777              }
  778          }
  779  
  780          Some(self.confirm_candidate(rcvr_ty, &relevant_candidates[0]))
  781      }

  783      pub fn merge_candidates(&self, candidates: &[Candidate]) -> ~[Candidate] {
  784          let mut merged = ~[];
  785          let mut i = 0;
  786          while i < candidates.len() {
  787              let candidate_a = &candidates[i];
  788  
  789              let mut skip = false;
  790  
  791              let mut j = i + 1;
  792              while j < candidates.len() {
  793                  let candidate_b = &candidates[j];
  794                  debug!("attempting to merge %? and %?",
  795                         candidate_a, candidate_b);
  796                  let candidates_same = match (&candidate_a.origin,
  797                                               &candidate_b.origin) {
  798                      (&method_param(ref p1), &method_param(ref p2)) => {
  799                          let same_trait = p1.trait_id == p2.trait_id;
  800                          let same_method = p1.method_num == p2.method_num;
  801                          let same_param = p1.param_num == p2.param_num;
  802                          // The bound number may be different because
  803                          // multiple bounds may lead to the same trait
  804                          // impl
  805                          same_trait && same_method && same_param
  806                      }
  807                      _ => false
  808                  };
  809                  if candidates_same {
  810                      skip = true;

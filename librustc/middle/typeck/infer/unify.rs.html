  71           * union-find algorithm with path compression:
  72           * http://en.wikipedia.org/wiki/Disjoint-set_data_structure
  73           */

  80              tcx: ty::ctxt,

  84              let vid_u = vid.to_uint();
  85              let var_val = match vb.vals.find(&vid_u) {
  86                  Some(&ref var_val) => (*var_val).clone(),
  87                  None => {

  89                          "failed lookup of vid `%u`", vid_u));
  90                  }
  91              };

  99                      }

 101                  }

 128      fn unify<T:Clone + InferStr,
 129               V:Clone + Vid + ToStr + UnifyVid<T>>(
 130               &mut self,
 131               node_a: &Node<V, T>,
 132               node_b: &Node<V, T>)
 133               -> (V, uint) {
 134          // Rank optimization: if you don't know what it is, check
 135          // out <http://en.wikipedia.org/wiki/Disjoint-set_data_structure>
 136  
 137          debug!("unify(node_a(id=%?, rank=%?), \
 138                  node_b(id=%?, rank=%?))",
 139                 node_a.root, node_a.rank,
 140                 node_b.root, node_b.rank);
 141  
 142          if node_a.rank > node_b.rank {
 143              // a has greater rank, so a should become b's parent,
 144              // i.e., b should redirect to a.
 145              self.set(node_b.root.clone(), Redirect(node_a.root.clone()));
 146              (node_a.root.clone(), node_a.rank)
 147          } else if node_a.rank < node_b.rank {
 148              // b has greater rank, so a should redirect to b.
 149              self.set(node_a.root.clone(), Redirect(node_b.root.clone()));
 150              (node_b.root.clone(), node_b.rank)
 151          } else {
 152              // If equal, redirect one to the other and increment the
 153              // other's rank.
 154              assert_eq!(node_a.rank, node_b.rank);
 155              self.set(node_b.root.clone(), Redirect(node_a.root.clone()));
 156              (node_a.root.clone(), node_a.rank + 1)
 157          }
 158      }

 200      fn simple_vars<T:Clone + Eq + InferStr + SimplyUnifiable,
 201                     V:Clone + Eq + Vid + ToStr + UnifyVid<Option<T>>>(
 202                     &mut self,
 203                     a_is_expected: bool,
 204                     a_id: V,
 205                     b_id: V)
 206                     -> ures {
 207          /*!
 208           *
 209           * Unifies two simple variables.  Because simple variables do
 210           * not have any subtyping relationships, if both variables
 211           * have already been associated with a value, then those two
 212           * values must be the same. */
 213  
 214          let node_a = self.get(a_id);
 215          let node_b = self.get(b_id);
 216          let a_id = node_a.root.clone();
 217          let b_id = node_b.root.clone();
 218  
 219          if a_id == b_id { return uok(); }
 220  
 221          let combined = match (&node_a.possible_types, &node_b.possible_types)
 222          {
 223              (&None, &None) => None,
 224              (&Some(ref v), &None) | (&None, &Some(ref v)) => {
 225                  Some((*v).clone())
 226              }
 227              (&Some(ref v1), &Some(ref v2)) => {
 228                  if *v1 != *v2 {
 229                      return mk_err(a_is_expected, (*v1).clone(), (*v2).clone())
 230                  }
 231                  Some((*v1).clone())
 232              }
 233          };
 234  
 235          let (new_root, new_rank) = self.unify(&node_a, &node_b);
 236          self.set(new_root, Root(combined, new_rank));
 237          return uok();
 238      }

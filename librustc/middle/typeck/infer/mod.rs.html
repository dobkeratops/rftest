 277  pub fn common_supertype(cx: @mut InferCtxt,
 278                          origin: TypeOrigin,
 279                          a_is_expected: bool,
 280                          a: ty::t,
 281                          b: ty::t)
 282                          -> ty::t {
 283      /*!
 284       * Computes the least upper-bound of `a` and `b`. If this is
 285       * not possible, reports an error and returns ty::err.
 286       */

 288      debug!("common_supertype(%s, %s)", a.inf_str(cx), b.inf_str(cx));
 289  
 290      let trace = TypeTrace {
 291          origin: origin,
 292          values: Types(expected_found(a_is_expected, a, b))
 293      };
 294  
 295      let result = do cx.commit {
 296          cx.lub(a_is_expected, trace).tys(a, b)
 297      };
 298  
 299      match result {
 300          Ok(t) => t,
 301          Err(ref err) => {
 302              cx.report_and_explain_type_error(trace, err);
 303              ty::mk_err()
 304          }
 305      }
 306  }

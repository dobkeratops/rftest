 277  pub fn common_supertype(cx: @mut InferCtxt,
 278                          origin: TypeOrigin,
 279                          a_is_expected: bool,
 280                          a: ty::t,
 281                          b: ty::t)
 282                          -> ty::t {
 283      /*!
 284       * Computes the least upper-bound of `a` and `b`. If this is
 285       * not possible, reports an error and returns ty::err.
 286       */

 288      debug!("common_supertype(%s, %s)", a.inf_str(cx), b.inf_str(cx));
 289  
 290      let trace = TypeTrace {
 291          origin: origin,
 292          values: Types(expected_found(a_is_expected, a, b))
 293      };
 294  
 295      let result = do cx.commit {
 296          cx.lub(a_is_expected, trace).tys(a, b)
 297      };
 298  
 299      match result {
 300          Ok(t) => t,
 301          Err(ref err) => {
 302              cx.report_and_explain_type_error(trace, err);
 303              ty::mk_err()
 304          }
 305      }
 306  }

 701      // [Note-Type-error-reporting]
 702      // An invariant is that anytime the expected or actual type is ty_err (the special
 703      // error type, meaning that an error occurred when typechecking this expression),
 704      // this is a derived error. The error cascaded from another error (that was already
 705      // reported), so it's not useful to display it to the user.
 706      // The following four methods -- type_error_message_str, type_error_message_str_with_expected,
 707      // type_error_message, and report_mismatched_types -- implement this logic.
 708      // They check if either the actual or expected type is ty_err, and don't print the error
 709      // in this case. The typechecker should only ever report type errors involving mismatched
 710      // types using one of these four methods, and should not call span_err directly for such
 711      // errors.
 712      pub fn type_error_message_str(@mut self,
 713                                    sp: Span,
 714                                    mk_msg: &fn(Option<~str>, ~str) -> ~str,
 715                                    actual_ty: ~str,
 716                                    err: Option<&ty::type_err>) {
 717          self.type_error_message_str_with_expected(sp, mk_msg, None, actual_ty, err)
 718      }
 719  
 720      pub fn type_error_message_str_with_expected(@mut self,
 721                                                  sp: Span,
 722                                                  mk_msg:
 723                                                  &fn(Option<~str>, ~str) ->
 724                                                  ~str,
 725                                                  expected_ty: Option<ty::t>,
 726                                                  actual_ty: ~str,
 727                                                  err: Option<&ty::type_err>) {
 728          debug!("hi! expected_ty = %?, actual_ty = %s", expected_ty, actual_ty);
 729  
 730          let error_str = do err.map_move_default(~"") |t_err| {
 731              fmt!(" (%s)", ty::type_err_to_str(self.tcx, t_err))
 732          };
 733          let resolved_expected = do expected_ty.map_move |e_ty| {
 734              self.resolve_type_vars_if_possible(e_ty)
 735          };
 736          if !resolved_expected.map_move_default(false, |e| { ty::type_is_error(e) }) {
 737              match resolved_expected {
 738                  None => self.tcx.sess.span_err(sp,
 739                              fmt!("%s%s", mk_msg(None, actual_ty), error_str)),
 740                  Some(e) => {
 741                      self.tcx.sess.span_err(sp,
 742                          fmt!("%s%s", mk_msg(Some(self.ty_to_str(e)), actual_ty), error_str));
 743                  }
 744              }
 745              for err in err.iter() {
 746                  ty::note_and_explain_type_err(self.tcx, *err)
 747              }
 748          }
 749      }

  11  /*!
  12   *
  13   * # Lattice Variables
  14   *
  15   * This file contains generic code for operating on inference variables
  16   * that are characterized by an upper- and lower-bound.  The logic and
  17   * reasoning is explained in detail in the large comment in `infer.rs`.
  18   *
  19   * The code in here is defined quite generically so that it can be
  20   * applied both to type variables, which represent types being inferred,
  21   * and fn variables, which represent function types being inferred.
  22   * It may eventually be applied to ther types as well, who knows.
  23   * In some cases, the functions are also generic with respect to the
  24   * operation on the lattice (GLB vs LUB).
  25   *
  26   * Although all the functions are generic, we generally write the
  27   * comments in a way that is specific to type variables and the LUB
  28   * operation.  It's just easier that way.
  29   *
  30   * In general all of the functions are defined parametrically
  31   * over a `LatticeValue`, which is a value defined with respect to
  32   * a lattice.
  33   */

 113  impl CombineFieldsLatticeMethods for CombineFields {
 114      fn var_sub_var<T:Clone + InferStr + LatticeValue,
 115                     V:Clone + Eq + ToStr + Vid + UnifyVid<Bounds<T>>>(
 116                     &self,
 117                     a_id: V,
 118                     b_id: V)
 119                     -> ures {
 120          /*!
 121           *
 122           * Make one variable a subtype of another variable.  This is a
 123           * subtle and tricky process, as described in detail at the
 124           * top of infer.rs*/
 125  
 126          // Need to make sub_id a subtype of sup_id.
 127          let node_a = self.infcx.get(a_id);
 128          let node_b = self.infcx.get(b_id);
 129          let a_id = node_a.root.clone();
 130          let b_id = node_b.root.clone();
 131          let a_bounds = node_a.possible_types.clone();
 132          let b_bounds = node_b.possible_types.clone();
 133  
 134          debug!("vars(%s=%s <: %s=%s)",
 135                 a_id.to_str(), a_bounds.inf_str(self.infcx),
 136                 b_id.to_str(), b_bounds.inf_str(self.infcx));
 137  
 138          if a_id == b_id { return uok(); }
 139  
 140          // If both A's UB and B's LB have already been bound to types,
 141          // see if we can make those types subtypes.
 142          match (&a_bounds.ub, &b_bounds.lb) {
 143              (&Some(ref a_ub), &Some(ref b_lb)) => {
 144                  let r = self.infcx.try(
 145                      || LatticeValue::sub(self, a_ub, b_lb));
 146                  match r {
 147                      Ok(()) => {
 148                          return Ok(());
 149                      }
 150                      Err(_) => { /*fallthrough */ }
 151                  }
 152              }
 153              _ => { /*fallthrough*/ }
 154          }
 155  
 156          // Otherwise, we need to merge A and B so as to guarantee that
 157          // A remains a subtype of B.  Actually, there are other options,
 158          // but that's the route we choose to take.
 159  
 160          let (new_root, new_rank) = self.infcx.unify(&node_a, &node_b);
 161          self.set_var_to_merged_bounds(new_root,
 162                                        &a_bounds, &b_bounds,
 163                                        new_rank)
 164      }

 166      /// make variable a subtype of T
 167      fn var_sub_t<T:Clone + InferStr + LatticeValue,
 168                   V:Clone + Eq + ToStr + Vid + UnifyVid<Bounds<T>>>(
 169                   &self,
 170                   a_id: V,
 171                   b: T)
 172                   -> ures {
 173          /*!
 174           *
 175           * Make a variable (`a_id`) a subtype of the concrete type `b` */
 176  
 177          let node_a = self.infcx.get(a_id);
 178          let a_id = node_a.root.clone();
 179          let a_bounds = &node_a.possible_types;
 180          let b_bounds = &Bounds { lb: None, ub: Some(b.clone()) };
 181  
 182          debug!("var_sub_t(%s=%s <: %s)",
 183                 a_id.to_str(),
 184                 a_bounds.inf_str(self.infcx),
 185                 b.inf_str(self.infcx));
 186  
 187          self.set_var_to_merged_bounds(
 188              a_id, a_bounds, b_bounds, node_a.rank)
 189      }
 190  
 191      fn t_sub_var<T:Clone + InferStr + LatticeValue,
 192                   V:Clone + Eq + ToStr + Vid + UnifyVid<Bounds<T>>>(
 193                   &self,
 194                   a: T,
 195                   b_id: V)
 196                   -> ures {
 197          /*!
 198           *
 199           * Make a concrete type (`a`) a subtype of the variable `b_id` */
 200  
 201          let a_bounds = &Bounds { lb: Some(a.clone()), ub: None };
 202          let node_b = self.infcx.get(b_id);
 203          let b_id = node_b.root.clone();
 204          let b_bounds = &node_b.possible_types;
 205  
 206          debug!("t_sub_var(%s <: %s=%s)",
 207                 a.inf_str(self.infcx),
 208                 b_id.to_str(),
 209                 b_bounds.inf_str(self.infcx));
 210  
 211          self.set_var_to_merged_bounds(
 212              b_id, a_bounds, b_bounds, node_b.rank)
 213      }

 215      fn merge_bnd<T:Clone + InferStr + LatticeValue>(
 216                   &self,
 217                   a: &Bound<T>,
 218                   b: &Bound<T>,
 219                   lattice_op: LatticeOp<T>)
 220                   -> cres<Bound<T>> {
 221          /*!
 222           *
 223           * Combines two bounds into a more general bound. */
 224  
 225          debug!("merge_bnd(%s,%s)",
 226                 a.inf_str(self.infcx),
 227                 b.inf_str(self.infcx));
 228          let _r = indenter();
 229  
 230          match (a, b) {
 231              (&None,          &None) => Ok(None),
 232              (&Some(_),       &None) => Ok((*a).clone()),
 233              (&None,          &Some(_)) => Ok((*b).clone()),
 234              (&Some(ref v_a), &Some(ref v_b)) => {
 235                  do lattice_op(self, v_a, v_b).chain |v| {
 236                      Ok(Some(v))
 237                  }
 238              }
 239          }
 240      }

 242      fn set_var_to_merged_bounds<T:Clone + InferStr + LatticeValue,
 243                                  V:Clone+Eq+ToStr+Vid+UnifyVid<Bounds<T>>>(
 244                                  &self,
 245                                  v_id: V,
 246                                  a: &Bounds<T>,
 247                                  b: &Bounds<T>,
 248                                  rank: uint)
 249                                  -> ures {
 250          /*!
 251           *
 252           * Updates the bounds for the variable `v_id` to be the intersection
 253           * of `a` and `b`.  That is, the new bounds for `v_id` will be
 254           * a bounds c such that:
 255           *    c.ub <: a.ub
 256           *    c.ub <: b.ub
 257           *    a.lb <: c.lb
 258           *    b.lb <: c.lb
 259           * If this cannot be achieved, the result is failure. */
 260  
 261          // Think of the two diamonds, we want to find the
 262          // intersection.  There are basically four possibilities (you
 263          // can swap A/B in these pictures):
 264          //
 265          //       A         A
 266          //      / \       / \
 267          //     / B \     / B \
 268          //    / / \ \   / / \ \
 269          //   * *   * * * /   * *
 270          //    \ \ / /   \   / /
 271          //     \ B /   / \ / /
 272          //      \ /   *   \ /
 273          //       A     \ / A
 274          //              B
 275  
 276          debug!("merge(%s,%s,%s)",
 277                 v_id.to_str(),
 278                 a.inf_str(self.infcx),
 279                 b.inf_str(self.infcx));
 280          let _indent = indenter();
 281  
 282          // First, relate the lower/upper bounds of A and B.
 283          // Note that these relations *must* hold for us to
 284          // to be able to merge A and B at all, and relating
 285          // them explicitly gives the type inferencer more
 286          // information and helps to produce tighter bounds
 287          // when necessary.
 288          let () = if_ok!(self.bnds(&a.lb, &b.ub));
 289          let () = if_ok!(self.bnds(&b.lb, &a.ub));
 290          let ub = if_ok!(self.merge_bnd(&a.ub, &b.ub, LatticeValue::glb));
 291          let lb = if_ok!(self.merge_bnd(&a.lb, &b.lb, LatticeValue::lub));
 292          let bounds = Bounds { lb: lb, ub: ub };
 293          debug!("merge(%s): bounds=%s",
 294                 v_id.to_str(),
 295                 bounds.inf_str(self.infcx));
 296  
 297          // the new bounds must themselves
 298          // be relatable:
 299          let () = if_ok!(self.bnds(&bounds.lb, &bounds.ub));
 300          self.infcx.set(v_id, Root(bounds, rank));
 301          uok()
 302      }

 304      fn bnds<T:Clone + InferStr + LatticeValue>(&self,
 305                                                 a: &Bound<T>,
 306                                                 b: &Bound<T>)
 307                                                 -> ures {
 308          debug!("bnds(%s <: %s)", a.inf_str(self.infcx),
 309                 b.inf_str(self.infcx));
 310          let _r = indenter();
 311  
 312          match (a, b) {
 313              (&None, &None) |
 314              (&Some(_), &None) |
 315              (&None, &Some(_)) => {
 316                  uok()
 317              }
 318              (&Some(ref t_a), &Some(ref t_b)) => {
 319                  LatticeValue::sub(self, t_a, t_b)
 320              }
 321          }
 322      }
 323  }

 369  pub fn super_lattice_tys<L:LatticeDir + TyLatticeDir + Combine>(
 370      this: &L,
 371      a: ty::t,
 372      b: ty::t) -> cres<ty::t> {
 373      debug!("%s.lattice_tys(%s, %s)", this.tag(),
 374             a.inf_str(this.infcx()),
 375             b.inf_str(this.infcx()));
 376      let _r = indenter();
 377  
 378      if a == b {
 379          return Ok(a);
 380      }
 381  
 382      let tcx = this.infcx().tcx;
 383  
 384      match (&ty::get(a).sty, &ty::get(b).sty) {
 385          (&ty::ty_bot, _) => { return this.ty_bot(b); }
 386          (_, &ty::ty_bot) => { return this.ty_bot(a); }
 387  
 388          (&ty::ty_infer(TyVar(a_id)), &ty::ty_infer(TyVar(b_id))) => {
 389              let r = if_ok!(lattice_vars(this, a_id, b_id,
 390                                          |x, y| this.tys(*x, *y)));
 391              return match r {
 392                  VarResult(v) => Ok(ty::mk_var(tcx, v)),
 393                  ValueResult(t) => Ok(t)
 394              };
 395          }
 396  
 397          (&ty::ty_infer(TyVar(a_id)), _) => {
 398              return lattice_var_and_t(this, a_id, &b,
 399                                       |x, y| this.tys(*x, *y));
 400          }
 401  
 402          (_, &ty::ty_infer(TyVar(b_id))) => {
 403              return lattice_var_and_t(this, b_id, &a,
 404                                       |x, y| this.tys(*x, *y));
 405          }
 406  
 407          _ => {
 408              return super_tys(this, a, b);
 409          }
 410      }
 411  }

 147     fn fn_sigs(&self, a: &ty::FnSig, b: &ty::FnSig) -> cres<ty::FnSig> {
 148         debug!("fn_sigs(a=%s, b=%s)",
 149                a.inf_str(self.infcx), b.inf_str(self.infcx));
 150         let _indenter = indenter();
 151 
 152         // Rather than checking the subtype relationship between `a` and `b`
 153         // as-is, we need to do some extra work here in order to make sure
 154         // that function subtyping works correctly with respect to regions
 155         //
 156         // Note: this is a subtle algorithm.  For a full explanation,
 157         // please see the large comment in `region_inference.rs`.
 158 
 159         // Take a snapshot.  We'll never roll this back, but in later
 160         // phases we do want to be able to examine "all bindings that
 161         // were created as part of this type comparison", and making a
 162         // snapshot is a convenient way to do that.
 163         let snapshot = self.infcx.region_vars.start_snapshot();
 164 
 165         // First, we instantiate each bound region in the subtype with a fresh
 166         // region variable.
 167         let (a_sig, _) =
 168             self.infcx.replace_bound_regions_with_fresh_regions(
 169                 self.trace, a);
 170 
 171         // Second, we instantiate each bound region in the supertype with a
 172         // fresh concrete region.
 173         let (skol_isr, _, b_sig) = {
 174             do replace_bound_regions_in_fn_sig(self.infcx.tcx, @Nil,
 175                                               None, b) |br| {
 176                 let skol = self.infcx.region_vars.new_skolemized(br);
 177                 debug!("Bound region %s skolemized to %?",
 178                        bound_region_to_str(self.infcx.tcx, "", false, br),
 179                        skol);
 180                 skol
 181             }
 182         };
 183 
 184         debug!("a_sig=%s", a_sig.inf_str(self.infcx));
 185         debug!("b_sig=%s", b_sig.inf_str(self.infcx));
 186 
 187         // Compare types now that bound regions have been replaced.
 188         let sig = if_ok!(super_fn_sigs(self, &a_sig, &b_sig));
 189 
 190         // Presuming type comparison succeeds, we need to check
 191         // that the skolemized regions do not "leak".
 192         let new_vars =
 193             self.infcx.region_vars.vars_created_since_snapshot(snapshot);
 194 
 195         let mut ret = Ok(sig);
 196         do list::each(skol_isr) |pair| {
 197             let (skol_br, skol) = *pair;
 198             let tainted = self.infcx.region_vars.tainted(snapshot, skol);
 199             for tainted_region in tainted.iter() {
 200                 // Each skolemized should only be relatable to itself
 201                 // or new variables:
 202                 match *tainted_region {
 203                     ty::re_infer(ty::ReVar(ref vid)) => {
 204                         if new_vars.iter().any(|x| x == vid) { loop; }
 205                     }
 206                     _ => {
 207                         if *tainted_region == skol { loop; }
 208                     }
 209                 };
 210 
 211                 // A is not as polymorphic as B:
 212                 if self.a_is_expected {
 213                     ret = Err(ty::terr_regions_insufficiently_polymorphic(
 214                         skol_br, *tainted_region));
 215                     break
 216                 } else {
 217                     ret = Err(ty::terr_regions_overly_polymorphic(
 218                         skol_br, *tainted_region));
 219                     break
 220                 }
 221             }
 222             ret.is_ok()
 223         };
 224 
 225         ret
 226     }

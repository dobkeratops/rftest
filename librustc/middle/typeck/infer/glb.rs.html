 165     fn fn_sigs(&self, a: &ty::FnSig, b: &ty::FnSig) -> cres<ty::FnSig> {
 166         // Note: this is a subtle algorithm.  For a full explanation,
 167         // please see the large comment in `region_inference.rs`.
 168 
 169         debug!("%s.fn_sigs(%?, %?)",
 170                self.tag(), a.inf_str(self.infcx), b.inf_str(self.infcx));
 171         let _indenter = indenter();
 172 
 173         // Take a snapshot.  We'll never roll this back, but in later
 174         // phases we do want to be able to examine "all bindings that
 175         // were created as part of this type comparison", and making a
 176         // snapshot is a convenient way to do that.
 177         let snapshot = self.infcx.region_vars.start_snapshot();

 179         // Instantiate each bound region with a fresh region variable.
 180         let (a_with_fresh, a_isr) =
 181             self.infcx.replace_bound_regions_with_fresh_regions(
 182                 self.trace, a);

 184         let (b_with_fresh, b_isr) =
 185             self.infcx.replace_bound_regions_with_fresh_regions(
 186                 self.trace, b);
 187         let b_vars = var_ids(self, b_isr);

 190         let sig0 = if_ok!(super_fn_sigs(self, &a_with_fresh, &b_with_fresh));

 195             self.infcx.region_vars.vars_created_since_snapshot(snapshot);

 206         fn generalize_region(this: &Glb,
 207                              snapshot: uint,
 208                              new_vars: &[RegionVid],
 209                              a_isr: isr_alist,
 210                              a_vars: &[RegionVid],
 211                              b_vars: &[RegionVid],
 212                              r0: ty::Region) -> ty::Region {
 213             if !is_var_in_set(new_vars, r0) {
 214                 return r0;
 215             }
 216 
 217             let tainted = this.infcx.region_vars.tainted(snapshot, r0);
 218 
 219             let mut a_r = None;
 220             let mut b_r = None;
 221             let mut only_new_vars = true;
 222             for r in tainted.iter() {
 223                 if is_var_in_set(a_vars, *r) {
 224                     if a_r.is_some() {
 225                         return fresh_bound_variable(this);
 226                     } else {
 227                         a_r = Some(*r);
 228                     }
 229                 } else if is_var_in_set(b_vars, *r) {
 230                     if b_r.is_some() {
 231                         return fresh_bound_variable(this);
 232                     } else {
 233                         b_r = Some(*r);
 234                     }
 235                 } else if !is_var_in_set(new_vars, *r) {
 236                     only_new_vars = false;
 237                 }
 238             }
 239 
 240                 // NB---I do not believe this algorithm computes
 241                 // (necessarily) the GLB.  As written it can
 242                 // spuriously fail.  In particular, if there is a case
 243                 // like: &fn(fn(&a)) and fn(fn(&b)), where a and b are
 244                 // free, it will return fn(&c) where c = GLB(a,b).  If
 245                 // however this GLB is not defined, then the result is
 246                 // an error, even though something like
 247                 // "fn<X>(fn(&X))" where X is bound would be a
 248                 // subtype of both of those.
 249                 //
 250                 // The problem is that if we were to return a bound
 251                 // variable, we'd be computing a lower-bound, but not
 252                 // necessarily the *greatest* lower-bound.
 253 
 254             if a_r.is_some() && b_r.is_some() && only_new_vars {
 255                 // Related to exactly one bound variable from each fn:
 256                 return rev_lookup(this, a_isr, a_r.get());
 257             } else if a_r.is_none() && b_r.is_none() {
 258                 // Not related to bound variables from either fn:
 259                 return r0;
 260             } else {
 261                 // Other:
 262                 return fresh_bound_variable(this);
 263             }
 264         }

 273                 if a_r == r {

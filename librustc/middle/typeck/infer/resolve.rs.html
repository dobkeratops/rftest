  11  // Resolution is the process of removing type variables and replacing
  12  // them with their inferred values.  Unfortunately our inference has
  13  // become fairly complex and so there are a number of options to
  14  // control *just how much* you want to resolve and how you want to do
  15  // it.
  16  //
  17  // # Controlling the scope of resolution
  18  //
  19  // The options resolve_* determine what kinds of variables get
  20  // resolved.  Generally resolution starts with a top-level type
  21  // variable; we will always resolve this.  However, once we have
  22  // resolved that variable, we may end up with a type that still
  23  // contains type variables.  For example, if we resolve `<T0>` we may
  24  // end up with something like `[<T1>]`.  If the option
  25  // `resolve_nested_tvar` is passed, we will then go and recursively
  26  // resolve `<T1>`.
  27  //
  28  // The options `resolve_rvar` controls whether we resolve region
  29  // variables. The options `resolve_fvar` and `resolve_ivar` control
  30  // whether we resolve floating point and integral variables,
  31  // respectively.
  32  //
  33  // # What do if things are unconstrained
  34  //
  35  // Sometimes we will encounter a variable that has no constraints, and
  36  // therefore cannot sensibly be mapped to any particular result.  By
  37  // default, we will leave such variables as is (so you will get back a
  38  // variable in your result).  The options force_* will cause the
  39  // resolution to fail in this case intead, except for the case of
  40  // integral variables, which resolve to `int` if forced.
  41  //
  42  // # resolve_all and force_all
  43  //
  44  // The options are a bit set, so you can use the *_all to resolve or
  45  // force all kinds of variables (including those we may add in the
  46  // future).  If you want to resolve everything but one type, you are
  47  // probably better off writing `resolve_all - resolve_ivar`.
  48

 119      pub fn resolve_type_chk(&mut self, typ: ty::t) -> fres<ty::t> {
 120          self.err = None;
 121  
 122          debug!("Resolving {} (modes={:x})",
 123                 ty_to_str(self.infcx.tcx, typ),
 124                 self.modes);
 125  
 126          // n.b. This is a hokey mess because the current fold doesn't
 127          // allow us to pass back errors in any useful way.
 128  
 129          assert!(self.v_seen.is_empty());
 130          let rty = indent(|| self.resolve_type(typ) );
 131          assert!(self.v_seen.is_empty());
 132          match self.err {
 133            None => {
 134              debug!("Resolved to {} + {} (modes={:x})",
 135                     ty_to_str(self.infcx.tcx, rty),
 136                     ty_to_str(self.infcx.tcx, rty),
 137                     self.modes);
 138              return Ok(rty);
 139            }
 140            Some(e) => return Err(e)
 141          }
 142      }

 154      pub fn resolve_type(&mut self, typ: ty::t) -> ty::t {
 155          debug!("resolve_type({})", typ.inf_str(self.infcx));
 156          let _i = indenter();
 157  
 158          if !ty::type_needs_infer(typ) {
 159              return typ;
 160          }
 161  
 162          if self.type_depth > 0 && !self.should(resolve_nested_tvar) {
 163              return typ;
 164          }
 165  
 166          match ty::get(typ).sty {
 167              ty::ty_infer(TyVar(vid)) => {
 168                  self.resolve_ty_var(vid)
 169              }
 170              ty::ty_infer(IntVar(vid)) => {
 171                  self.resolve_int_var(vid)
 172              }
 173              ty::ty_infer(FloatVar(vid)) => {
 174                  self.resolve_float_var(vid)
 175              }
 176              _ => {
 177                  if self.modes & resolve_all == 0 {
 178                      // if we are only resolving top-level type
 179                      // variables, and this is not a top-level type
 180                      // variable, then shortcircuit for efficiency
 181                      typ
 182                  } else {
 183                      self.type_depth += 1;
 184                      let result = ty_fold::super_fold_ty(self, typ);
 185                      self.type_depth -= 1;
 186                      result
 187                  }
 188              }
 189          }
 190      }

 216      pub fn resolve_ty_var(&mut self, vid: TyVid) -> ty::t {
 217          if self.v_seen.contains(&vid) {
 218              self.err = Some(cyclic_ty(vid));
 219              return ty::mk_var(self.infcx.tcx, vid);
 220          } else {
 221              self.v_seen.push(vid);
 222              let tcx = self.infcx.tcx;
 223  
 224              // Nonobvious: prefer the most specific type
 225              // (i.e., the lower bound) to the more general
 226              // one.  More general types in Rust (e.g., fn())
 227              // tend to carry more restrictions or higher
 228              // perf. penalties, so it pays to know more.
 229  
 230              let nde = self.infcx.get(vid);
 231              let bounds = nde.possible_types;
 232  
 233              let t1 = match bounds {
 234                Bounds { ub:_, lb:Some(t) } if !type_is_bot(t)
 235                  => self.resolve_type(t),
 236                Bounds { ub:Some(t), lb:_ } => self.resolve_type(t),
 237                Bounds { ub:_, lb:Some(t) } => self.resolve_type(t),
 238                Bounds { ub:None, lb:None } => {
 239                  if self.should(force_tvar) {
 240                      self.err = Some(unresolved_ty(vid));
 241                  }
 242                  ty::mk_var(tcx, vid)
 243                }
 244              };
 245              self.v_seen.pop().unwrap();
 246              return t1;
 247          }
 248      }

 264                  ty
 265              } else {
 266                  ty::mk_int_var(self.infcx.tcx, vid)
 267              }
 268            }
 269          }
 270      }

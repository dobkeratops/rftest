  11  /*!
  12  
  13  # Type Coercion
  14  
  15  Under certain circumstances we will coerce from one type to another,
  16  for example by auto-borrowing.  This occurs in situations where the
  17  compiler has a firm 'expected type' that was supplied from the user,
  18  and where the actual type is similar to that expected type in purpose
  19  but not in representation (so actual subtyping is inappropriate).
  20  
  21  ## Reborrowing
  22  
  23  Note that if we are expecting a reference, we will *reborrow*
  24  even if the argument provided was already a reference.  This is
  25  useful for freezing mut/const things (that is, when the expected is &T
  26  but you have &const T or &mut T) and also for avoiding the linearity
  27  of mut things (when the expected is &mut T and you have &mut T).  See
  28  the various `src/test/run-pass/coerce-reborrow-*.rs` tests for
  29  examples of where this is useful.
  30  
  31  ## Subtle note
  32  
  33  When deciding what type coercions to consider, we do not attempt to
  34  resolve any type variables we may encounter.  This is because `b`
  35  represents the expected type "as the user wrote it", meaning that if
  36  the user defined a generic function like
  37  
  38     fn foo<A>(a: A, b: A) { ... }
  39  
  40  and then we wrote `foo(&1, @2)`, we will not auto-borrow
  41  either argument.  In older code we went to some lengths to
  42  resolve the `b` variable, which could mean that we'd
  43  auto-borrow later arguments but not earlier ones, which
  44  seems very confusing.
  45  
  46  ## Subtler note
  47  
  48  However, right now, if the user manually specifies the
  49  values for the type variables, as so:
  50  
  51     foo::<&int>(@1, @2)
  52  
  53  then we *will* auto-borrow, because we can't distinguish this from a
  54  function that declared `&int`.  This is inconsistent but it's easiest
  55  at the moment. The right thing to do, I think, is to consider the
  56  *unsubstituted* type when deciding whether to auto-borrow, but the
  57  *substituted* type when considering the bounds and so forth. But most
  58  of our methods don't give access to the unsubstituted type, and
  59  rightly so because they'd be error-prone.  So maybe the thing to do is
  60  to actually determine the kind of coercions that should occur
  61  separately and pass them in.  Or maybe it's ok as is.  Anyway, it's
  62  sort of a minor point so I've opted to leave it for later---after all
  63  we may want to adjust precisely when coercions occur.
  64  
  65  */

  89      pub fn get_ref<'a>(&'a self) -> &'a CombineFields<'f> {

  93      pub fn tys(&self, a: ty::t, b: ty::t) -> CoerceResult {
  94          debug!("Coerce.tys({} => {})",
  95                 a.inf_str(self.get_ref().infcx),
  96                 b.inf_str(self.get_ref().infcx));
  97          let _indent = indenter();
  98  
  99          // Examine the supertype and consider auto-borrowing.
 100          //
 101          // Note: does not attempt to resolve type variables we encounter.
 102          // See above for details.
 103          match ty::get(b).sty {
 104              ty::ty_rptr(_, mt_b) => {
 105                  return self.unpack_actual_value(a, |sty_a| {
 106                      self.coerce_borrowed_pointer(a, sty_a, b, mt_b)
 107                  });
 108              }
 109  
 110              ty::ty_str(vstore_slice(_)) => {
 111                  return self.unpack_actual_value(a, |sty_a| {
 112                      self.coerce_borrowed_string(a, sty_a, b)
 113                  });
 114              }
 115  
 116              ty::ty_vec(mt_b, vstore_slice(_)) => {
 117                  return self.unpack_actual_value(a, |sty_a| {
 118                      self.coerce_borrowed_vector(a, sty_a, b, mt_b)
 119                  });
 120              }
 121  
 122              ty::ty_closure(~ty::ClosureTy {sigil: ast::BorrowedSigil, ..}) => {
 123                  return self.unpack_actual_value(a, |sty_a| {
 124                      self.coerce_borrowed_fn(a, sty_a, b)
 125                  });
 126              }
 127  
 128              ty::ty_ptr(mt_b) => {
 129                  return self.unpack_actual_value(a, |sty_a| {
 130                      self.coerce_unsafe_ptr(a, sty_a, b, mt_b)
 131                  });
 132              }
 133  
 134              ty::ty_trait(~ty::TyTrait {
 135                  def_id, ref substs, store: ty::UniqTraitStore, mutability: m, bounds
 136              }) => {
 137                  let result = self.unpack_actual_value(a, |sty_a| {
 138                      match *sty_a {
 139                          ty::ty_uniq(..) => {
 140                              self.coerce_object(a, sty_a, b, def_id, substs,
 141                                                 ty::UniqTraitStore, m, bounds)
 142                          }
 143                          _ => Err(ty::terr_mismatch)
 144                      }
 145                  });
 146  
 147                  match result {
 148                      Ok(t) => return Ok(t),
 149                      Err(..) => {}
 150                  }
 151              }
 152  
 153              ty::ty_trait(~ty::TyTrait {
 154                  def_id, ref substs, store: ty::RegionTraitStore(region), mutability: m, bounds
 155              }) => {
 156                  let result = self.unpack_actual_value(a, |sty_a| {
 157                      match *sty_a {
 158                          ty::ty_rptr(..) => {
 159                              self.coerce_object(a, sty_a, b, def_id, substs,
 160                                                 ty::RegionTraitStore(region), m, bounds)
 161                          }
 162                          _ => self.coerce_borrowed_object(a, sty_a, b, m)
 163                      }
 164                  });
 165  
 166                  match result {
 167                      Ok(t) => return Ok(t),
 168                      Err(..) => {}
 169                  }
 170              }
 171  
 172              _ => {}
 173          }
 174  
 175          self.unpack_actual_value(a, |sty_a| {
 176              match *sty_a {
 177                  ty::ty_bare_fn(ref a_f) => {
 178                      // Bare functions are coercable to any closure type.
 179                      //
 180                      // FIXME(#3320) this should go away and be
 181                      // replaced with proper inference, got a patch
 182                      // underway - ndm
 183                      self.coerce_from_bare_fn(a, a_f, b)
 184                  }
 185                  _ => {
 186                      // Otherwise, just use subtyping rules.
 187                      self.subtype(a, b)
 188                  }
 189              }
 190          })
 191      }

 193      pub fn subtype(&self, a: ty::t, b: ty::t) -> CoerceResult {
 194          match Sub(*self.get_ref()).tys(a, b) {
 195              Ok(_) => Ok(None),         // No coercion required.
 196              Err(ref e) => Err(*e)
 197          }
 198      }

 200      pub fn unpack_actual_value(&self, a: ty::t, f: |&ty::sty| -> CoerceResult)
 201                                 -> CoerceResult {
 202          match resolve_type(self.get_ref().infcx, a, try_resolve_tvar_shallow) {
 203              Ok(t) => {
 204                  f(&ty::get(t).sty)
 205              }
 206              Err(e) => {
 207                  self.get_ref().infcx.tcx.sess.span_bug(
 208                      self.get_ref().trace.origin.span(),
 209                      format!("failed to resolve even without \
 210                            any force options: {:?}", e));
 211              }
 212          }
 213      }

 215      pub fn coerce_borrowed_pointer(&self,
 216                                     a: ty::t,
 217                                     sty_a: &ty::sty,
 218                                     b: ty::t,
 219                                     mt_b: ty::mt)
 220                                     -> CoerceResult {
 221          debug!("coerce_borrowed_pointer(a={}, sty_a={:?}, b={}, mt_b={:?})",
 222                 a.inf_str(self.get_ref().infcx), sty_a,
 223                 b.inf_str(self.get_ref().infcx), mt_b);
 224  
 225          // If we have a parameter of type `&M T_a` and the value
 226          // provided is `expr`, we will be adding an implicit borrow,
 227          // meaning that we convert `f(expr)` to `f(&M *expr)`.  Therefore,
 228          // to type check, we will construct the type that `&M*expr` would
 229          // yield.
 230  
 231          let sub = Sub(*self.get_ref());
 232          let r_borrow = self.get_ref().infcx.next_region_var(Coercion(self.get_ref().trace));
 233  
 234          let inner_ty = match *sty_a {
 235              ty::ty_box(typ) | ty::ty_uniq(typ) => typ,
 236              ty::ty_rptr(_, mt_a) => mt_a.ty,
 237              _ => {
 238                  return self.subtype(a, b);
 239              }
 240          };
 241  
 242          let a_borrowed = ty::mk_rptr(self.get_ref().infcx.tcx,
 243                                       r_borrow,
 244                                       mt {ty: inner_ty, mutbl: mt_b.mutbl});
 245          if_ok!(sub.tys(a_borrowed, b));
 246          Ok(Some(@AutoDerefRef(AutoDerefRef {
 247              autoderefs: 1,
 248              autoref: Some(AutoPtr(r_borrow, mt_b.mutbl))
 249          })))
 250      }

 252      pub fn coerce_borrowed_string(&self,
 253                                    a: ty::t,
 254                                    sty_a: &ty::sty,
 255                                    b: ty::t)
 256                                    -> CoerceResult {
 257          debug!("coerce_borrowed_string(a={}, sty_a={:?}, b={})",
 258                 a.inf_str(self.get_ref().infcx), sty_a,
 259                 b.inf_str(self.get_ref().infcx));
 260  
 261          match *sty_a {
 262              ty::ty_str(vstore_uniq) => {}
 263              _ => {
 264                  return self.subtype(a, b);
 265              }
 266          };
 267  
 268          let r_a = self.get_ref().infcx.next_region_var(Coercion(self.get_ref().trace));
 269          let a_borrowed = ty::mk_str(self.get_ref().infcx.tcx, vstore_slice(r_a));
 270          if_ok!(self.subtype(a_borrowed, b));
 271          Ok(Some(@AutoDerefRef(AutoDerefRef {
 272              autoderefs: 0,
 273              autoref: Some(AutoBorrowVec(r_a, MutImmutable))
 274          })))
 275      }

 336      pub fn coerce_borrowed_fn(&self,
 337                                a: ty::t,
 338                                sty_a: &ty::sty,
 339                                b: ty::t)
 340                                -> CoerceResult {
 341          debug!("coerce_borrowed_fn(a={}, sty_a={:?}, b={})",
 342                 a.inf_str(self.get_ref().infcx), sty_a,
 343                 b.inf_str(self.get_ref().infcx));
 344  
 345          let fn_ty = match *sty_a {
 346              ty::ty_closure(ref f) if f.sigil == ast::ManagedSigil ||
 347                                       f.sigil == ast::OwnedSigil => {
 348                  (*f).clone()
 349              }
 350              ty::ty_bare_fn(ref f) => {
 351                  return self.coerce_from_bare_fn(a, f, b);
 352              }
 353              _ => {
 354                  return self.subtype(a, b);
 355              }
 356          };
 357  
 358          let r_borrow = self.get_ref().infcx.next_region_var(Coercion(self.get_ref().trace));
 359          let a_borrowed = ty::mk_closure(
 360              self.get_ref().infcx.tcx,
 361              ty::ClosureTy {
 362                  sigil: ast::BorrowedSigil,
 363                  region: r_borrow,
 364                  .. *fn_ty
 365              });
 366  
 367          if_ok!(self.subtype(a_borrowed, b));
 368          Ok(Some(@AutoDerefRef(AutoDerefRef {
 369              autoderefs: 0,
 370              autoref: Some(AutoBorrowFn(r_borrow))
 371          })))
 372      }

 374      fn coerce_from_bare_fn(&self, a: ty::t, fn_ty_a: &ty::BareFnTy, b: ty::t)
 375                             -> CoerceResult {
 376          /*!
 377           *
 378           * Attempts to coerce from a bare Rust function (`extern
 379           * "Rust" fn`) into a closure or a `proc`.
 380           */
 381  
 382          self.unpack_actual_value(b, |sty_b| {
 383  
 384              debug!("coerce_from_bare_fn(a={}, b={})",
 385                     a.inf_str(self.get_ref().infcx), b.inf_str(self.get_ref().infcx));
 386  
 387              if !fn_ty_a.abis.is_rust() || fn_ty_a.purity != ast::ImpureFn {
 388                  return self.subtype(a, b);
 389              }
 390  
 391              let fn_ty_b = match *sty_b {
 392                  ty::ty_closure(ref f) => (*f).clone(),
 393                  _ => return self.subtype(a, b)
 394              };
 395  
 396              let adj = @ty::AutoAddEnv(fn_ty_b.region, fn_ty_b.sigil);
 397              let a_closure = ty::mk_closure(self.get_ref().infcx.tcx,
 398                                             ty::ClosureTy {
 399                                                  sig: fn_ty_a.sig.clone(),
 400                                                  .. *fn_ty_b
 401                                             });
 402              if_ok!(self.subtype(a_closure, b));
 403              Ok(Some(adj))
 404          })
 405      }

 411                               mt_b: ty::mt)
 412                               -> CoerceResult {
 413          debug!("coerce_unsafe_ptr(a={}, sty_a={:?}, b={})",
 414                 a.inf_str(self.get_ref().infcx), sty_a,
 415                 b.inf_str(self.get_ref().infcx));

 417          let mt_a = match *sty_a {
 418              ty::ty_rptr(_, mt) => mt,
 419              _ => {
 420                  return self.subtype(a, b);

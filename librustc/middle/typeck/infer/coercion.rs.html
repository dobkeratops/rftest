  11  /*!
  12  
  13  # Type Coercion
  14  
  15  Under certain circumstances we will coerce from one type to another,
  16  for example by auto-borrowing.  This occurs in situations where the
  17  compiler has a firm 'expected type' that was supplied from the user,
  18  and where the actual type is similar to that expected type in purpose
  19  but not in representation (so actual subtyping is inappropriate).
  20  
  21  ## Reborrowing
  22  
  23  Note that if we are expecting a reference, we will *reborrow*
  24  even if the argument provided was already a reference.  This is
  25  useful for freezing mut/const things (that is, when the expected is &T
  26  but you have &const T or &mut T) and also for avoiding the linearity
  27  of mut things (when the expected is &mut T and you have &mut T).  See
  28  the various `src/test/run-pass/coerce-reborrow-*.rs` tests for
  29  examples of where this is useful.
  30  
  31  ## Subtle note
  32  
  33  When deciding what type coercions to consider, we do not attempt to
  34  resolve any type variables we may encounter.  This is because `b`
  35  represents the expected type "as the user wrote it", meaning that if
  36  the user defined a generic function like
  37  
  38     fn foo<A>(a: A, b: A) { ... }
  39  
  40  and then we wrote `foo(&1, @2)`, we will not auto-borrow
  41  either argument.  In older code we went to some lengths to
  42  resolve the `b` variable, which could mean that we'd
  43  auto-borrow later arguments but not earlier ones, which
  44  seems very confusing.
  45  
  46  ## Subtler note
  47  
  48  However, right now, if the user manually specifies the
  49  values for the type variables, as so:
  50  
  51     foo::<&int>(@1, @2)
  52  
  53  then we *will* auto-borrow, because we can't distinguish this from a
  54  function that declared `&int`.  This is inconsistent but it's easiest
  55  at the moment. The right thing to do, I think, is to consider the
  56  *unsubstituted* type when deciding whether to auto-borrow, but the
  57  *substituted* type when considering the bounds and so forth. But most
  58  of our methods don't give access to the unsubstituted type, and
  59  rightly so because they'd be error-prone.  So maybe the thing to do is
  60  to actually determine the kind of coercions that should occur
  61  separately and pass them in.  Or maybe it's ok as is.  Anyway, it's
  62  sort of a minor point so I've opted to leave it for later---after all
  63  we may want to adjust precisely when coercions occur.
  64  
  65  */

  84  // Note: Coerce is not actually a combiner, in that it does not
  85  // conform to the same interface, though it performs a similar

  90      pub fn get_ref<'a>(&'a self) -> &'a CombineFields<'f> {
  91          let Coerce(ref v) = *self; v
  92      }

  94      pub fn tys(&self, a: ty::t, b: ty::t) -> CoerceResult {
  95          debug!("Coerce.tys({} => {})",
  96                 a.inf_str(self.get_ref().infcx),
  97                 b.inf_str(self.get_ref().infcx));
  98          let _indent = indenter();
  99  
 100          // Examine the supertype and consider auto-borrowing.
 101          //
 102          // Note: does not attempt to resolve type variables we encounter.
 103          // See above for details.
 104          match ty::get(b).sty {
 105              ty::ty_rptr(_, mt_b) => {
 106                  return self.unpack_actual_value(a, |sty_a| {
 107                      self.coerce_borrowed_pointer(a, sty_a, b, mt_b)
 108                  });
 109              }
 110  
 111              ty::ty_str(vstore_slice(_)) => {
 112                  return self.unpack_actual_value(a, |sty_a| {
 113                      self.coerce_borrowed_string(a, sty_a, b)
 114                  });
 115              }
 116  
 117              ty::ty_vec(mt_b, vstore_slice(_)) => {
 118                  return self.unpack_actual_value(a, |sty_a| {
 119                      self.coerce_borrowed_vector(a, sty_a, b, mt_b)
 120                  });
 121              }
 122  
 123              ty::ty_closure(~ty::ClosureTy {sigil: ast::BorrowedSigil, ..}) => {
 124                  return self.unpack_actual_value(a, |sty_a| {
 125                      self.coerce_borrowed_fn(a, sty_a, b)
 126                  });
 127              }
 128  
 129              ty::ty_ptr(mt_b) => {
 130                  return self.unpack_actual_value(a, |sty_a| {
 131                      self.coerce_unsafe_ptr(a, sty_a, b, mt_b)
 132                  });
 133              }
 134  
 135              ty::ty_trait(~ty::TyTrait {
 136                  def_id, ref substs, store: ty::UniqTraitStore, mutability: m, bounds
 137              }) => {
 138                  let result = self.unpack_actual_value(a, |sty_a| {
 139                      match *sty_a {
 140                          ty::ty_uniq(..) => {
 141                              self.coerce_object(a, sty_a, b, def_id, substs,
 142                                                 ty::UniqTraitStore, m, bounds)
 143                          }
 144                          _ => Err(ty::terr_mismatch)
 145                      }
 146                  });
 147  
 148                  match result {
 149                      Ok(t) => return Ok(t),
 150                      Err(..) => {}
 151                  }
 152              }
 153  
 154              ty::ty_trait(~ty::TyTrait {
 155                  def_id, ref substs, store: ty::RegionTraitStore(region), mutability: m, bounds
 156              }) => {
 157                  let result = self.unpack_actual_value(a, |sty_a| {
 158                      match *sty_a {
 159                          ty::ty_rptr(..) => {
 160                              self.coerce_object(a, sty_a, b, def_id, substs,
 161                                                 ty::RegionTraitStore(region), m, bounds)
 162                          }
 163                          _ => self.coerce_borrowed_object(a, sty_a, b, m)
 164                      }
 165                  });
 166  
 167                  match result {
 168                      Ok(t) => return Ok(t),
 169                      Err(..) => {}
 170                  }
 171              }
 172  
 173              _ => {}
 174          }
 175  
 176          self.unpack_actual_value(a, |sty_a| {
 177              match *sty_a {
 178                  ty::ty_bare_fn(ref a_f) => {
 179                      // Bare functions are coercable to any closure type.
 180                      //
 181                      // FIXME(#3320) this should go away and be
 182                      // replaced with proper inference, got a patch
 183                      // underway - ndm
 184                      self.coerce_from_bare_fn(a, a_f, b)
 185                  }
 186                  _ => {
 187                      // Otherwise, just use subtyping rules.
 188                      self.subtype(a, b)
 189                  }
 190              }
 191          })
 192      }

 194      pub fn subtype(&self, a: ty::t, b: ty::t) -> CoerceResult {
 195          match Sub(*self.get_ref()).tys(a, b) {
 196              Ok(_) => Ok(None),         // No coercion required.
 197              Err(ref e) => Err(*e)
 198          }
 199      }

 201      pub fn unpack_actual_value(&self, a: ty::t, f: |&ty::sty| -> CoerceResult)
 202                                 -> CoerceResult {
 203          match resolve_type(self.get_ref().infcx, a, try_resolve_tvar_shallow) {
 204              Ok(t) => {
 205                  f(&ty::get(t).sty)
 206              }
 207              Err(e) => {
 208                  self.get_ref().infcx.tcx.sess.span_bug(
 209                      self.get_ref().trace.origin.span(),
 210                      format!("failed to resolve even without \
 211                            any force options: {:?}", e));
 212              }
 213          }
 214      }

 216      pub fn coerce_borrowed_pointer(&self,
 217                                     a: ty::t,
 218                                     sty_a: &ty::sty,
 219                                     b: ty::t,
 220                                     mt_b: ty::mt)
 221                                     -> CoerceResult {
 222          debug!("coerce_borrowed_pointer(a={}, sty_a={:?}, b={}, mt_b={:?})",
 223                 a.inf_str(self.get_ref().infcx), sty_a,
 224                 b.inf_str(self.get_ref().infcx), mt_b);
 225  
 226          // If we have a parameter of type `&M T_a` and the value
 227          // provided is `expr`, we will be adding an implicit borrow,
 228          // meaning that we convert `f(expr)` to `f(&M *expr)`.  Therefore,
 229          // to type check, we will construct the type that `&M*expr` would
 230          // yield.
 231  
 232          let sub = Sub(*self.get_ref());
 233          let r_borrow = self.get_ref().infcx.next_region_var(Coercion(self.get_ref().trace));
 234  
 235          let inner_ty = match *sty_a {
 236              ty::ty_box(typ) | ty::ty_uniq(typ) => typ,
 237              ty::ty_rptr(_, mt_a) => mt_a.ty,
 238              _ => {
 239                  return self.subtype(a, b);
 240              }
 241          };
 242  
 243          let a_borrowed = ty::mk_rptr(self.get_ref().infcx.tcx,
 244                                       r_borrow,
 245                                       mt {ty: inner_ty, mutbl: mt_b.mutbl});
 246          if_ok!(sub.tys(a_borrowed, b));
 247          Ok(Some(@AutoDerefRef(AutoDerefRef {
 248              autoderefs: 1,
 249              autoref: Some(AutoPtr(r_borrow, mt_b.mutbl))
 250          })))
 251      }

 253      pub fn coerce_borrowed_string(&self,
 254                                    a: ty::t,
 255                                    sty_a: &ty::sty,
 256                                    b: ty::t)
 257                                    -> CoerceResult {
 258          debug!("coerce_borrowed_string(a={}, sty_a={:?}, b={})",
 259                 a.inf_str(self.get_ref().infcx), sty_a,
 260                 b.inf_str(self.get_ref().infcx));
 261  
 262          match *sty_a {
 263              ty::ty_str(vstore_uniq) => {}
 264              _ => {
 265                  return self.subtype(a, b);
 266              }
 267          };
 268  
 269          let r_a = self.get_ref().infcx.next_region_var(Coercion(self.get_ref().trace));
 270          let a_borrowed = ty::mk_str(self.get_ref().infcx.tcx, vstore_slice(r_a));
 271          if_ok!(self.subtype(a_borrowed, b));
 272          Ok(Some(@AutoDerefRef(AutoDerefRef {
 273              autoderefs: 0,
 274              autoref: Some(AutoBorrowVec(r_a, MutImmutable))
 275          })))
 276      }

 337      pub fn coerce_borrowed_fn(&self,
 338                                a: ty::t,
 339                                sty_a: &ty::sty,
 340                                b: ty::t)
 341                                -> CoerceResult {
 342          debug!("coerce_borrowed_fn(a={}, sty_a={:?}, b={})",
 343                 a.inf_str(self.get_ref().infcx), sty_a,
 344                 b.inf_str(self.get_ref().infcx));
 345  
 346          let fn_ty = match *sty_a {
 347              ty::ty_closure(ref f) if f.sigil == ast::ManagedSigil ||
 348                                       f.sigil == ast::OwnedSigil => {
 349                  (*f).clone()
 350              }
 351              ty::ty_bare_fn(ref f) => {
 352                  return self.coerce_from_bare_fn(a, f, b);
 353              }
 354              _ => {
 355                  return self.subtype(a, b);
 356              }
 357          };
 358  
 359          let r_borrow = self.get_ref().infcx.next_region_var(Coercion(self.get_ref().trace));
 360          let a_borrowed = ty::mk_closure(
 361              self.get_ref().infcx.tcx,
 362              ty::ClosureTy {
 363                  sigil: ast::BorrowedSigil,
 364                  region: r_borrow,
 365                  .. *fn_ty
 366              });
 367  
 368          if_ok!(self.subtype(a_borrowed, b));
 369          Ok(Some(@AutoDerefRef(AutoDerefRef {
 370              autoderefs: 0,
 371              autoref: Some(AutoBorrowFn(r_borrow))
 372          })))
 373      }

 375      fn coerce_from_bare_fn(&self, a: ty::t, fn_ty_a: &ty::BareFnTy, b: ty::t)
 376                             -> CoerceResult {
 377          /*!
 378           *
 379           * Attempts to coerce from a bare Rust function (`extern
 380           * "Rust" fn`) into a closure or a `proc`.
 381           */
 382  
 383          self.unpack_actual_value(b, |sty_b| {
 384  
 385              debug!("coerce_from_bare_fn(a={}, b={})",
 386                     a.inf_str(self.get_ref().infcx), b.inf_str(self.get_ref().infcx));
 387  
 388              if fn_ty_a.abi != abi::Rust || fn_ty_a.purity != ast::ImpureFn {
 389                  return self.subtype(a, b);
 390              }
 391  
 392              let fn_ty_b = match *sty_b {
 393                  ty::ty_closure(ref f) => (*f).clone(),
 394                  _ => return self.subtype(a, b)
 395              };
 396  
 397              let adj = @ty::AutoAddEnv(fn_ty_b.region, fn_ty_b.sigil);
 398              let a_closure = ty::mk_closure(self.get_ref().infcx.tcx,
 399                                             ty::ClosureTy {
 400                                                  sig: fn_ty_a.sig.clone(),
 401                                                  .. *fn_ty_b
 402                                             });
 403              if_ok!(self.subtype(a_closure, b));
 404              Ok(Some(adj))
 405          })
 406      }

 409                               a: ty::t,
 410                               sty_a: &ty::sty,
 411                               b: ty::t,
 412                               mt_b: ty::mt)
 413                               -> CoerceResult {
 414          debug!("coerce_unsafe_ptr(a={}, sty_a={:?}, b={})",
 415                 a.inf_str(self.get_ref().infcx), sty_a,
 416                 b.inf_str(self.get_ref().infcx));
 417  
 418          let mt_a = match *sty_a {
 419              ty::ty_rptr(_, mt) => mt,
 420              _ => {
 421                  return self.subtype(a, b);

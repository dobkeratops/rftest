  12  // Type combining
  13  //
  14  // There are three type combiners: sub, lub, and glb.  Each implements
  15  // the trait `combine` and contains methods for combining two
  16  // instances of various things and yielding a new instance.  These
  17  // combiner methods always yield a `result<T>`---failure is propagated
  18  // upward using `chain()` methods.
  19  //
  20  // There is a lot of common code for these operations, which is
  21  // abstracted out into functions named `super_X()` which take a combiner
  22  // instance as the first parameter.  This would be better implemented
  23  // using traits.  For this system to work properly, you should not
  24  // call the `super_X(foo, ...)` functions directly, but rather call
  25  // `foo.X(...)`.  The implementation of `X()` can then choose to delegate
  26  // to the `super` routine or to do other things.
  27  // (FIXME (#2794): revise this paragraph once default methods in traits
  28  // are working.)
  29  //
  30  // In reality, the sub operation is rather different from lub/glb, but
  31  // they are combined into one trait to avoid duplication (they used to
  32  // be separate but there were many bugs because there were two copies
  33  // of most routines).
  34  //
  35  // The differences are:
  36  //
  37  // - when making two things have a sub relationship, the order of the
  38  //   arguments is significant (a <: b) and the return value of the
  39  //   combine functions is largely irrelevant.  The important thing is
  40  //   whether the action succeeds or fails.  If it succeeds, then side
  41  //   effects have been committed into the type variables.
  42  //
  43  // - for GLB/LUB, the order of arguments is not significant (GLB(a,b) ==
  44  //   GLB(b,a)) and the return value is important (it is the GLB).  Of
  45  //   course GLB/LUB may also have side effects.
  46  //
  47  // Contravariance
  48  //
  49  // When you are relating two things which have a contravariant
  50  // relationship, you should use `contratys()` or `contraregions()`,
  51  // rather than inversing the order of arguments!  This is necessary
  52  // because the order of arguments is not relevant for LUB and GLB.  It
  53  // is also useful to track which value is the "expected" value in
  54  // terms of error reporting, although we do not do that properly right
  55  // now.

 127  pub fn expected_found<C:Combine,T>(
 128          this: &C, a: T, b: T) -> ty::expected_found<T> {
 129      if this.a_is_expected() {
 130          ty::expected_found {expected: a, found: b}
 131      } else {
 132          ty::expected_found {expected: b, found: a}
 133      }
 134  }

 136  pub fn eq_tys<C:Combine>(this: &C, a: ty::t, b: ty::t) -> ures {
 137      let suber = this.sub();
 138      do this.infcx().try {
 139          do suber.tys(a, b).chain |_ok| {
 140              suber.contratys(a, b)
 141          }.to_ures()
 142      }
 143  }

 145  pub fn eq_regions<C:Combine>(this: &C, a: ty::Region, b: ty::Region)
 146                            -> ures {
 147      debug!("eq_regions(%s, %s)",
 148             a.inf_str(this.infcx()),
 149             b.inf_str(this.infcx()));
 150      let sub = this.sub();
 151      do indent {
 152          this.infcx().try(|| {
 153              do sub.regions(a, b).chain |_r| {
 154                  sub.contraregions(a, b)
 155              }
 156          }).chain_err(|e| {
 157              // substitute a better error, but use the regions
 158              // found in the original error
 159              match e {
 160                ty::terr_regions_does_not_outlive(a1, b1) =>
 161                  Err(ty::terr_regions_not_same(a1, b1)),
 162                _ => Err(e)
 163              }
 164          }).to_ures()
 165      }
 166  }

 168  pub fn eq_opt_regions<C:Combine>(
 169      this: &C,
 170      a: Option<ty::Region>,
 171      b: Option<ty::Region>) -> cres<Option<ty::Region>> {
 172  
 173      match (a, b) {
 174        (None, None) => {
 175          Ok(None)
 176        }
 177        (Some(a), Some(b)) => {
 178          do eq_regions(this, a, b).then {
 179              Ok(Some(a))
 180          }
 181        }
 182        (_, _) => {
 183          // If these two substitutions are for the same type (and
 184          // they should be), then the type should either
 185          // consistently have a region parameter or not have a
 186          // region parameter.
 187          this.infcx().tcx.sess.bug(
 188              fmt!("substitution a had opt_region %s and \
 189                    b had opt_region %s",
 190                   a.inf_str(this.infcx()),
 191                   b.inf_str(this.infcx())));
 192        }
 193      }
 194  }

 196  pub fn super_substs<C:Combine>(
 197      this: &C, generics: &ty::Generics,
 198      a: &ty::substs, b: &ty::substs) -> cres<ty::substs> {
 199  
 200      fn relate_region_params<C:Combine>(
 201          this: &C,
 202          generics: &ty::Generics,
 203          a: &ty::RegionSubsts,
 204          b: &ty::RegionSubsts)
 205          -> cres<ty::RegionSubsts>
 206      {
 207          match (a, b) {
 208              (&ty::ErasedRegions, _) |
 209              (_, &ty::ErasedRegions) => {
 210                  Ok(ty::ErasedRegions)
 211              }
 212  
 213              (&ty::NonerasedRegions(ref a_rs),
 214               &ty::NonerasedRegions(ref b_rs)) => {
 215                  match generics.region_param {
 216                      None => {
 217                          assert!(a_rs.is_empty());
 218                          assert!(b_rs.is_empty());
 219                          Ok(ty::NonerasedRegions(opt_vec::Empty))
 220                      }
 221  
 222                      Some(variance) => {
 223                          assert_eq!(a_rs.len(), 1);
 224                          assert_eq!(b_rs.len(), 1);
 225                          let a_r = *a_rs.get(0);
 226                          let b_r = *b_rs.get(0);
 227  
 228                          match variance {
 229                              ty::rv_invariant => {
 230                                  do eq_regions(this, a_r, b_r).then {
 231                                      Ok(ty::NonerasedRegions(opt_vec::with(a_r)))
 232                                  }
 233                              }
 234  
 235                              ty::rv_covariant => {
 236                                  do this.regions(a_r, b_r).chain |r| {
 237                                      Ok(ty::NonerasedRegions(opt_vec::with(r)))
 238                                  }
 239                              }
 240  
 241                              ty::rv_contravariant => {
 242                                  do this.contraregions(a_r, b_r).chain |r| {
 243                                      Ok(ty::NonerasedRegions(opt_vec::with(r)))
 244                                  }
 245                              }
 246                          }
 247                      }
 248                  }
 249              }
 250          }
 251      }
 252  
 253      do this.tps(a.tps, b.tps).chain |tps| {
 254          do this.self_tys(a.self_ty, b.self_ty).chain |self_ty| {
 255              do relate_region_params(this,
 256                                      generics,
 257                                      &a.regions,
 258                                      &b.regions).chain |regions| {
 259                  Ok(substs {
 260                      regions: regions,
 261                      self_ty: self_ty,
 262                      tps: tps.clone()
 263                  })
 264              }
 265          }
 266      }
 267  }

 269  pub fn super_tps<C:Combine>(
 270      this: &C, as_: &[ty::t], bs: &[ty::t]) -> cres<~[ty::t]> {
 271  
 272      // Note: type parameters are always treated as *invariant*
 273      // (otherwise the type system would be unsound).  In the
 274      // future we could allow type parameters to declare a
 275      // variance.
 276  
 277      if vec::same_length(as_, bs) {
 278          iter_vec2(as_, bs, |a, b| {
 279              eq_tys(this, *a, *b)
 280          }).then(|| Ok(as_.to_owned()) )
 281      } else {
 282          Err(ty::terr_ty_param_size(
 283              expected_found(this, as_.len(), bs.len())))
 284      }
 285  }

 287  pub fn super_self_tys<C:Combine>(
 288      this: &C, a: Option<ty::t>, b: Option<ty::t>) -> cres<Option<ty::t>> {
 289  
 290      match (a, b) {
 291        (None, None) => {
 292          Ok(None)
 293        }
 294        (Some(a), Some(b)) => {
 295            // FIXME(#5781) this should be eq_tys
 296            // eq_tys(this, a, b).then(|| Ok(Some(a)) )
 297            this.contratys(a, b).chain(|t| Ok(Some(t)))
 298        }
 299        (None, Some(_)) |
 300        (Some(_), None) => {
 301          // I think it should never happen that we unify two substs and
 302          // one of them has a self_ty and one doesn't...? I could be
 303          // wrong about this.
 304            this.infcx().tcx.sess.bug(
 305                fmt!("substitution a had a self_ty and substitution b didn't, \
 306                      or vice versa"));
 307        }
 308      }
 309  }

 320  pub fn super_flds<C:Combine>(
 321      this: &C, a: ty::field, b: ty::field) -> cres<ty::field> {
 322  
 323      if a.ident == b.ident {
 324          this.mts(&a.mt, &b.mt)
 325              .chain(|mt| Ok(ty::field {ident: a.ident, mt: mt}) )
 326              .chain_err(|e| Err(ty::terr_in_field(@e, a.ident)) )
 327      } else {
 328          Err(ty::terr_record_fields(
 329              expected_found(this, a.ident, b.ident)))
 330      }
 331  }

 339  pub fn super_vstores<C:Combine>(this: &C,
 340                                  vk: ty::terr_vstore_kind,
 341                                  a: ty::vstore,
 342                                  b: ty::vstore)
 343                                  -> cres<ty::vstore> {
 344      debug!("%s.super_vstores(a=%?, b=%?)", this.tag(), a, b);
 345  
 346      match (a, b) {
 347        (ty::vstore_slice(a_r), ty::vstore_slice(b_r)) => {
 348          do this.contraregions(a_r, b_r).chain |r| {
 349              Ok(ty::vstore_slice(r))
 350          }
 351        }
 352  
 353        _ if a == b => {
 354          Ok(a)
 355        }
 356  
 357        _ => {
 358          Err(ty::terr_vstores_differ(vk, expected_found(this, a, b)))
 359        }
 360      }
 361  }

 387  pub fn super_closure_tys<C:Combine>(
 388      this: &C, a_f: &ty::ClosureTy, b_f: &ty::ClosureTy) -> cres<ty::ClosureTy>
 389  {
 390      let p = if_ok!(this.sigils(a_f.sigil, b_f.sigil));
 391      let r = if_ok!(this.contraregions(a_f.region, b_f.region));
 392      let purity = if_ok!(this.purities(a_f.purity, b_f.purity));
 393      let onceness = if_ok!(this.oncenesses(a_f.onceness, b_f.onceness));
 394      let bounds = if_ok!(this.bounds(a_f.bounds, b_f.bounds));
 395      let sig = if_ok!(this.fn_sigs(&a_f.sig, &b_f.sig));
 396      Ok(ty::ClosureTy {purity: purity,
 397                        sigil: p,
 398                        onceness: onceness,
 399                        region: r,
 400                        bounds: bounds,
 401                        sig: sig})
 402  }

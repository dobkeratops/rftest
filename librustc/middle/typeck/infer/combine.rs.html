  12  // Type combining
  13  //
  14  // There are three type combiners: sub, lub, and glb.  Each implements
  15  // the trait `Combine` and contains methods for combining two
  16  // instances of various things and yielding a new instance.  These
  17  // combiner methods always yield a `result<T>`---failure is propagated
  18  // upward using `and_then()` methods.  There is a lot of common code for
  19  // these operations, implemented as default methods on the `Combine`
  20  // trait.
  21  //
  22  // In reality, the sub operation is rather different from lub/glb, but
  23  // they are combined into one trait to avoid duplication (they used to
  24  // be separate but there were many bugs because there were two copies
  25  // of most routines).
  26  //
  27  // The differences are:
  28  //
  29  // - when making two things have a sub relationship, the order of the
  30  //   arguments is significant (a <: b) and the return value of the
  31  //   combine functions is largely irrelevant.  The important thing is
  32  //   whether the action succeeds or fails.  If it succeeds, then side
  33  //   effects have been committed into the type variables.
  34  //
  35  // - for GLB/LUB, the order of arguments is not significant (GLB(a,b) ==
  36  //   GLB(b,a)) and the return value is important (it is the GLB).  Of
  37  //   course GLB/LUB may also have side effects.
  38  //
  39  // Contravariance
  40  //
  41  // When you are relating two things which have a contravariant
  42  // relationship, you should use `contratys()` or `contraregions()`,
  43  // rather than inversing the order of arguments!  This is necessary
  44  // because the order of arguments is not relevant for LUB and GLB.  It
  45  // is also useful to track which value is the "expected" value in
  46  // terms of error reporting, although we do not do that properly right
  47  // now.

  83      fn tps(&self, as_: &[ty::t], bs: &[ty::t]) -> cres<~[ty::t]> {
  84  
  85          // Note: type parameters are always treated as *invariant*
  86          // (otherwise the type system would be unsound).  In the
  87          // future we could allow type parameters to declare a
  88          // variance.
  89  
  90          if as_.len() == bs.len() {
  91              result::fold_(as_.iter().zip(bs.iter())
  92                            .map(|(a, b)| eq_tys(self, *a, *b)))
  93                  .then(|| Ok(as_.to_owned()))
  94          } else {
  95              Err(ty::terr_ty_param_size(expected_found(self,
  96                                                        as_.len(),
  97                                                        bs.len())))
  98          }
  99      }

 101      fn self_tys(&self, a: Option<ty::t>, b: Option<ty::t>)
 102                 -> cres<Option<ty::t>> {
 103  
 104          match (a, b) {
 105              (None, None) => {
 106                  Ok(None)
 107              }
 108              (Some(a), Some(b)) => {
 109                  // FIXME(#5781) this should be eq_tys
 110                  // eq_tys(self, a, b).then(|| Ok(Some(a)) )
 111                  self.contratys(a, b).and_then(|t| Ok(Some(t)))
 112              }
 113              (None, Some(_)) |
 114                  (Some(_), None) => {
 115                  // I think it should never happen that we unify two
 116                  // substs and one of them has a self_ty and one
 117                  // doesn't...? I could be wrong about this.
 118                  self.infcx().tcx.sess.bug(
 119                                            fmt!("substitution a had a self_ty \
 120                                                 and substitution b didn't, \
 121                                                 or vice versa"));
 122              }
 123          }
 124      }

 126      fn substs(&self, generics: &ty::Generics, as_: &ty::substs,
 127                bs: &ty::substs) -> cres<ty::substs> {
 128  
 129          fn relate_region_params<C:Combine>(
 130                                             this: &C,
 131                                             generics: &ty::Generics,
 132                                             a: &ty::RegionSubsts,
 133                                             b: &ty::RegionSubsts)
 134              -> cres<ty::RegionSubsts>
 135              {
 136              match (a, b) {
 137                  (&ty::ErasedRegions, _) |
 138                      (_, &ty::ErasedRegions) => {
 139                      Ok(ty::ErasedRegions)
 140                  }
 141  
 142                  (&ty::NonerasedRegions(ref a_rs),
 143                   &ty::NonerasedRegions(ref b_rs)) => {
 144                      match generics.region_param {
 145                          None => {
 146                              assert!(a_rs.is_empty());
 147                              assert!(b_rs.is_empty());
 148                              Ok(ty::NonerasedRegions(opt_vec::Empty))
 149                          }
 150  
 151                          Some(variance) => {
 152                              assert_eq!(a_rs.len(), 1);
 153                              assert_eq!(b_rs.len(), 1);
 154                              let a_r = *a_rs.get(0);
 155                              let b_r = *b_rs.get(0);
 156  
 157                              match variance {
 158                                  ty::rv_invariant => {
 159                                      do eq_regions(this, a_r, b_r).then {
 160                                          Ok(ty::NonerasedRegions(opt_vec::with(a_r)))
 161                                      }
 162                                  }
 163  
 164                                  ty::rv_covariant => {
 165                                      do this.regions(a_r, b_r).and_then |r| {
 166                                          Ok(ty::NonerasedRegions(opt_vec::with(r)))
 167                                      }
 168                                  }
 169  
 170                                  ty::rv_contravariant => {
 171                                      do this.contraregions(a_r, b_r).and_then |r| {
 172                                          Ok(ty::NonerasedRegions(opt_vec::with(r)))
 173                                      }
 174                                  }
 175                              }
 176                          }
 177                      }
 178                  }
 179              }
 180          }
 181  
 182          do self.tps(as_.tps, bs.tps).and_then |tps| {
 183              do self.self_tys(as_.self_ty, bs.self_ty).and_then |self_ty| {
 184                  do relate_region_params(self,
 185                                          generics,
 186                                          &as_.regions,
 187                                          &bs.regions).and_then |regions| {
 188                      Ok(substs {
 189                              regions: regions,
 190                              self_ty: self_ty,
 191                              tps: tps.clone()
 192                          })
 193                  }
 194              }
 195          }
 196      }

 287                  Err(ty::terr_vstores_differ(vk, expected_found(self, a, b)))

 318      fn trait_refs(&self,
 319                    a: &ty::TraitRef,
 320                    b: &ty::TraitRef) -> cres<ty::TraitRef> {
 321          // Different traits cannot be related
 322  
 323          // - NOTE in the future, expand out subtraits!
 324  
 325          if a.def_id != b.def_id {
 326              Err(ty::terr_traits(
 327                                  expected_found(self, a.def_id, b.def_id)))

 329              let tcx = self.infcx().tcx;

 331              let substs = if_ok!(self.substs(&trait_def.generics,
 332                                              &a.substs,
 333                                              &b.substs));
 334              Ok(ty::TraitRef {
 335                      def_id: a.def_id,
 336                      substs: substs
 337                  })
 338          }
 339      }

 357  pub fn eq_tys<C:Combine>(this: &C, a: ty::t, b: ty::t) -> ures {
 358      let suber = this.sub();
 359      do this.infcx().try {
 360          do suber.tys(a, b).and_then |_ok| {
 361              suber.contratys(a, b)
 362          }.to_ures()
 363      }
 364  }

 366  pub fn eq_regions<C:Combine>(this: &C, a: ty::Region, b: ty::Region)
 367                            -> ures {
 368      debug!("eq_regions(%s, %s)",
 369             a.inf_str(this.infcx()),
 370             b.inf_str(this.infcx()));
 371      let sub = this.sub();
 372      do indent {
 373          this.infcx().try(|| {
 374              do sub.regions(a, b).and_then |_r| {
 375                  sub.contraregions(a, b)
 376              }
 377          }).or_else(|e| {
 378              // substitute a better error, but use the regions
 379              // found in the original error
 380              match e {
 381                ty::terr_regions_does_not_outlive(a1, b1) =>
 382                  Err(ty::terr_regions_not_same(a1, b1)),
 383                _ => Err(e)
 384              }
 385          }).to_ures()
 386      }
 387  }

 389  pub fn eq_opt_regions<C:Combine>(
 390      this: &C,
 391      a: Option<ty::Region>,
 392      b: Option<ty::Region>) -> cres<Option<ty::Region>> {
 393  
 394      match (a, b) {
 395        (None, None) => {
 396          Ok(None)
 397        }
 398        (Some(a), Some(b)) => {
 399          do eq_regions(this, a, b).then {
 400              Ok(Some(a))
 401          }
 402        }
 403        (_, _) => {
 404          // If these two substitutions are for the same type (and
 405          // they should be), then the type should either
 406          // consistently have a region parameter or not have a
 407          // region parameter.
 408          this.infcx().tcx.sess.bug(
 409              fmt!("substitution a had opt_region %s and \
 410                    b had opt_region %s",
 411                   a.inf_str(this.infcx()),
 412                   b.inf_str(this.infcx())));
 413        }
 414      }
 415  }

 417  pub fn super_fn_sigs<C:Combine>(
 418      this: &C, a: &ty::FnSig, b: &ty::FnSig) -> cres<ty::FnSig> {
 419  
 420      fn argvecs<C:Combine>(this: &C, a_args: &[ty::t], b_args: &[ty::t]) -> cres<~[ty::t]> {
 421          if a_args.len() == b_args.len() {
 422              result::collect(a_args.iter().zip(b_args.iter())
 423                              .map(|(a, b)| this.args(*a, *b)))
 424          } else {
 425              Err(ty::terr_arg_count)
 426          }
 427      }
 428  
 429      do argvecs(this, a.inputs, b.inputs)
 430              .and_then |inputs| {
 431          do this.tys(a.output, b.output).and_then |output| {

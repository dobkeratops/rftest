   61  pub enum RegionResolutionError {
   62      /// `ConcreteFailure(o, a, b)`:
   63      ///
   64      /// `o` requires that `a <= b`, but this does not hold
   65      ConcreteFailure(SubregionOrigin, Region, Region),
   66  
   67      /// `SubSupConflict(v, sub_origin, sub_r, sup_origin, sup_r)`:
   68      ///
   69      /// Could not infer a value for `v` because `sub_r <= v` (due to
   70      /// `sub_origin`) but `v <= sup_r` (due to `sup_origin`) and
   71      /// `sub_r <= sup_r` does not hold.
   72      SubSupConflict(RegionVariableOrigin,
   73                     SubregionOrigin, Region,
   74                     SubregionOrigin, Region),
   75  
   76      /// `SupSupConflict(v, origin1, r1, origin2, r2)`:
   77      ///
   78      /// Could not infer a value for `v` because `v <= r1` (due to
   79      /// `origin1`) and `v <= r2` (due to `origin2`) and
   80      /// `r1` and `r2` have no intersection.
   81      SupSupConflict(RegionVariableOrigin,
   82                     SubregionOrigin, Region,
   83                     SubregionOrigin, Region),
   84  }

   88  pub struct RegionVarBindings {
   89      tcx: ty::ctxt,
   90      var_origins: ~[RegionVariableOrigin],
   91      constraints: HashMap<Constraint, SubregionOrigin>,
   92      lubs: CombineMap,
   93      glbs: CombineMap,
   94      skolemization_count: uint,
   95      bound_count: uint,
   96  
   97      // The undo log records actions that might later be undone.
   98      //
   99      // Note: when the undo_log is empty, we are not actively
  100      // snapshotting.  When the `start_snapshot()` method is called, we
  101      // push a Snapshot entry onto the list to indicate that we are now
  102      // actively snapshotting.  The reason for this is that otherwise
  103      // we end up adding entries for things like the lower bound on
  104      // a variable and so forth, which can never be rolled back.
  105      undo_log: ~[UndoLogEntry],
  106  
  107      // This contains the results of inference.  It begins as an empty
  108      // cell and only acquires a value after inference is complete.
  109      // We use a cell vs a mutable option to circumvent borrowck errors.
  110      values: Cell<~[VarValue]>,
  111  }

  127  impl RegionVarBindings {
  128      pub fn in_snapshot(&self) -> bool {
  129          self.undo_log.len() > 0
  130      }
  131  
  132      pub fn start_snapshot(&mut self) -> uint {
  133          debug!("RegionVarBindings: snapshot()=%u", self.undo_log.len());
  134          if self.in_snapshot() {
  135              self.undo_log.len()
  136          } else {
  137              self.undo_log.push(Snapshot);
  138              0
  139          }
  140      }
  141  
  142      pub fn commit(&mut self) {
  143          debug!("RegionVarBindings: commit()");
  144          while self.undo_log.len() > 0 {
  145              self.undo_log.pop();
  146          }
  147      }
  148  
  149      pub fn rollback_to(&mut self, snapshot: uint) {
  150          debug!("RegionVarBindings: rollback_to(%u)", snapshot);
  151          while self.undo_log.len() > snapshot {
  152              let undo_item = self.undo_log.pop();
  153              debug!("undo_item=%?", undo_item);
  154              match undo_item {
  155                Snapshot => {}
  156                AddVar(vid) => {
  157                  assert_eq!(self.var_origins.len(), vid.to_uint() + 1);
  158                  self.var_origins.pop();
  159                }
  160                AddConstraint(ref constraint) => {
  161                  self.constraints.remove(constraint);
  162                }
  163                AddCombination(Glb, ref regions) => {
  164                  self.glbs.remove(regions);
  165                }
  166                AddCombination(Lub, ref regions) => {
  167                  self.lubs.remove(regions);
  168                }
  169              }
  170          }
  171      }

  177      pub fn new_region_var(&mut self, origin: RegionVariableOrigin) -> RegionVid {
  178          let id = self.num_vars();
  179          self.var_origins.push(origin);
  180          let vid = RegionVid { id: id };
  181          if self.in_snapshot() {
  182              self.undo_log.push(AddVar(vid));
  183          }
  184          debug!("created new region variable %? with origin %?",
  185                 vid, origin.repr(self.tcx));
  186          return vid;
  187      }

  195      pub fn new_bound(&mut self) -> Region {
  196          // Creates a fresh bound variable for use in GLB computations.
  197          // See discussion of GLB computation in the large comment at
  198          // the top of this file for more details.
  199          //
  200          // This computation is mildly wrong in the face of rollover.
  201          // It's conceivable, if unlikely, that one might wind up with
  202          // accidental capture for nested functions in that case, if
  203          // the outer function had bound regions created a very long
  204          // time before and the inner function somehow wound up rolling
  205          // over such that supposedly fresh identifiers were in fact
  206          // shadowed.  We should convert our bound_region
  207          // representation to use deBruijn indices or something like
  208          // that to eliminate that possibility.
  209  
  210          let sc = self.bound_count;
  211          self.bound_count += 1;
  212          re_bound(br_fresh(sc))
  213      }

  215      pub fn add_constraint(&mut self,
  216                            constraint: Constraint,
  217                            origin: SubregionOrigin) {
  218          // cannot add constraints once regions are resolved
  219          assert!(self.values.is_empty());
  220  
  221          debug!("RegionVarBindings: add_constraint(%?)", constraint);
  222  
  223          if self.constraints.insert(constraint, origin) {
  224              if self.in_snapshot() {
  225                  self.undo_log.push(AddConstraint(constraint));
  226              }
  227          }
  228      }

  230      pub fn make_subregion(&mut self,
  231                            origin: SubregionOrigin,
  232                            sub: Region,
  233                            sup: Region) {
  234          // cannot add constraints once regions are resolved
  235          assert!(self.values.is_empty());
  236  
  237          debug!("RegionVarBindings: make_subregion(%?, %?)", sub, sup);
  238          match (sub, sup) {
  239            (re_infer(ReVar(sub_id)), re_infer(ReVar(sup_id))) => {
  240              self.add_constraint(ConstrainVarSubVar(sub_id, sup_id), origin);
  241            }
  242            (r, re_infer(ReVar(sup_id))) => {
  243              self.add_constraint(ConstrainRegSubVar(r, sup_id), origin);
  244            }
  245            (re_infer(ReVar(sub_id)), r) => {
  246              self.add_constraint(ConstrainVarSubReg(sub_id, r), origin);
  247            }
  248            (re_bound(br), _) => {
  249              self.tcx.sess.span_bug(
  250                  origin.span(),
  251                  fmt!("Cannot relate bound region as subregion: %?", br));
  252            }
  253            (_, re_bound(br)) => {
  254              self.tcx.sess.span_bug(
  255                  origin.span(),
  256                  fmt!("Cannot relate bound region as superregion: %?", br));
  257            }
  258            _ => {
  259              self.add_constraint(ConstrainRegSubReg(sub, sup), origin);
  260            }
  261          }
  262      }

  311      pub fn resolve_var(&mut self, rid: RegionVid) -> ty::Region {
  312          if self.values.is_empty() {
  313              self.tcx.sess.span_bug(
  314                  self.var_origins[rid.to_uint()].span(),
  315                  fmt!("Attempt to resolve region variable before values have \
  316                        been computed!"));
  317          }
  318  
  319          let v = self.values.with_ref(|values| values[rid.to_uint()]);
  320          debug!("RegionVarBindings: resolve_var(%?=%u)=%?",
  321                 rid, rid.to_uint(), v);
  322          match v {
  323              Value(r) => r,
  324  
  325              NoValue => {
  326                  // No constraints, return ty::re_empty
  327                  re_empty
  328              }
  329  
  330              ErrorValue => {
  331                  // An error that has previously been reported.
  332                  re_static
  333              }
  334          }
  335      }

  347      pub fn combine_vars(&mut self,
  348                          t: CombineMapType,
  349                          a: Region,
  350                          b: Region,
  351                          origin: SubregionOrigin,
  352                          relate: &fn(this: &mut RegionVarBindings,
  353                                      old_r: Region,
  354                                      new_r: Region))
  355                          -> Region {
  356          let vars = TwoRegions { a: a, b: b };
  357          match self.combine_map(t).find(&vars) {
  358              Some(&c) => {
  359                  return re_infer(ReVar(c));
  360              }
  361              None => {}
  362          }
  363          let c = self.new_region_var(infer::MiscVariable(origin.span()));
  364          self.combine_map(t).insert(vars, c);
  365          if self.in_snapshot() {
  366              self.undo_log.push(AddCombination(t, vars));
  367          }
  368          relate(self, a, re_infer(ReVar(c)));
  369          relate(self, b, re_infer(ReVar(c)));
  370          debug!("combine_vars() c=%?", c);
  371          re_infer(ReVar(c))
  372      }

  384      pub fn tainted(&mut self, snapshot: uint, r0: Region) -> ~[Region] {
  385          /*!
  386           * Computes all regions that have been related to `r0` in any
  387           * way since the snapshot `snapshot` was taken---`r0` itself
  388           * will be the first entry. This is used when checking whether
  389           * skolemized regions are being improperly related to other
  390           * regions.
  391           */
  392  
  393          debug!("tainted(snapshot=%u, r0=%?)", snapshot, r0);
  394          let _indenter = indenter();
  395  
  396          let undo_len = self.undo_log.len();
  397  
  398          // `result_set` acts as a worklist: we explore all outgoing
  399          // edges and add any new regions we find to result_set.  This
  400          // is not a terribly efficient implementation.
  401          let mut result_set = ~[r0];
  402          let mut result_index = 0;
  403          while result_index < result_set.len() {
  404              // nb: can't use uint::range() here because result_set grows
  405              let r = result_set[result_index];
  406  
  407              debug!("result_index=%u, r=%?", result_index, r);
  408  
  409              let mut undo_index = snapshot;
  410              while undo_index < undo_len {
  411                  // nb: can't use uint::range() here as we move result_set
  412                  let regs = match self.undo_log[undo_index] {
  413                      AddConstraint(ConstrainVarSubVar(ref a, ref b)) => {
  414                          Some((re_infer(ReVar(*a)),
  415                                re_infer(ReVar(*b))))
  416                      }
  417                      AddConstraint(ConstrainRegSubVar(ref a, ref b)) => {
  418                          Some((*a, re_infer(ReVar(*b))))
  419                      }
  420                      AddConstraint(ConstrainVarSubReg(ref a, ref b)) => {
  421                          Some((re_infer(ReVar(*a)), *b))
  422                      }
  423                      AddConstraint(ConstrainRegSubReg(a, b)) => {
  424                          Some((a, b))
  425                      }
  426                      _ => {
  427                          None
  428                      }
  429                  };
  430  
  431                  match regs {
  432                      None => {}
  433                      Some((r1, r2)) => {
  434                          result_set =
  435                              consider_adding_edge(result_set, r, r1, r2);
  436                          result_set =
  437                              consider_adding_edge(result_set, r, r2, r1);
  438                      }
  439                  }
  440  
  441                  undo_index += 1;
  442              }
  443  
  444              result_index += 1;
  445          }
  446  
  447          return result_set;
  448  
  449          fn consider_adding_edge(result_set: ~[Region],
  450                                  r: Region,
  451                                  r1: Region,
  452                                  r2: Region) -> ~[Region]
  453          {
  454              let mut result_set = result_set;
  455              if r == r1 { // Clearly, this is potentially inefficient.
  456                  if !result_set.iter().any(|x| *x == r2) {
  457                      result_set.push(r2);
  458                  }
  459              }
  460              return result_set;
  461          }
  462      }
  463  
  464      /**
  465      This function performs the actual region resolution.  It must be
  466      called after all constraints have been added.  It performs a
  467      fixed-point iteration to find region values which satisfy all
  468      constraints, assuming such values can be found; if they cannot,
  469      errors are reported.
  470      */
  471      pub fn resolve_regions(&mut self) -> OptVec<RegionResolutionError> {
  472          debug!("RegionVarBindings: resolve_regions()");
  473          let mut errors = opt_vec::Empty;
  474          let v = self.infer_variable_values(&mut errors);
  475          self.values.put_back(v);
  476          errors
  477      }
  478  }

  484      }

  486      fn lub_concrete_regions(&self, a: Region, b: Region) -> Region {
  487          match (a, b) {
  488            (re_static, _) | (_, re_static) => {
  489              re_static // nothing lives longer than static
  490            }
  491  
  492            (re_empty, r) | (r, re_empty) => {
  493              r // everything lives longer than empty
  494            }
  495  
  496            (re_infer(ReVar(v_id)), _) | (_, re_infer(ReVar(v_id))) => {
  497              self.tcx.sess.span_bug(
  498                  self.var_origins[v_id.to_uint()].span(),
  499                  fmt!("lub_concrete_regions invoked with \
  500                        non-concrete regions: %?, %?", a, b));
  501            }
  502  
  503            (f @ re_free(ref fr), re_scope(s_id)) |
  504            (re_scope(s_id), f @ re_free(ref fr)) => {
  505              // A "free" region can be interpreted as "some region
  506              // at least as big as the block fr.scope_id".  So, we can
  507              // reasonably compare free regions and scopes:
  508              let rm = self.tcx.region_maps;
  509              match rm.nearest_common_ancestor(fr.scope_id, s_id) {
  510                // if the free region's scope `fr.scope_id` is bigger than
  511                // the scope region `s_id`, then the LUB is the free
  512                // region itself:
  513                Some(r_id) if r_id == fr.scope_id => f,
  514  
  515                // otherwise, we don't know what the free region is,
  516                // so we must conservatively say the LUB is static:
  517                _ => re_static
  518              }
  519            }
  520  
  521            (re_scope(a_id), re_scope(b_id)) => {
  522              // The region corresponding to an outer block is a
  523              // subtype of the region corresponding to an inner
  524              // block.
  525              let rm = self.tcx.region_maps;
  526              match rm.nearest_common_ancestor(a_id, b_id) {
  527                Some(r_id) => re_scope(r_id),
  528                _ => re_static
  529              }
  530            }
  531  
  532            (re_free(ref a_fr), re_free(ref b_fr)) => {
  533               self.lub_free_regions(a_fr, b_fr)
  534            }
  535  
  536            // For these types, we cannot define any additional
  537            // relationship:
  538            (re_infer(ReSkolemized(*)), _) |
  539            (_, re_infer(ReSkolemized(*))) |
  540            (re_bound(_), re_bound(_)) |
  541            (re_bound(_), re_free(_)) |
  542            (re_bound(_), re_scope(_)) |
  543            (re_free(_), re_bound(_)) |
  544            (re_scope(_), re_bound(_)) => {
  545              if a == b {a} else {re_static}
  546            }
  547          }
  548      }
  549  
  550      fn lub_free_regions(&self,
  551                          a: &FreeRegion,
  552                          b: &FreeRegion) -> ty::Region
  553      {
  554          /*!
  555           * Computes a region that encloses both free region arguments.
  556           * Guarantee that if the same two regions are given as argument,
  557           * in any order, a consistent result is returned.
  558           */
  559  
  560          return match a.cmp(b) {
  561              Less => helper(self, a, b),
  562              Greater => helper(self, b, a),
  563              Equal => ty::re_free(*a)
  564          };
  565  
  566          fn helper(this: &RegionVarBindings,
  567                    a: &FreeRegion,
  568                    b: &FreeRegion) -> ty::Region
  569          {
  570              let rm = this.tcx.region_maps;
  571              if rm.sub_free_region(*a, *b) {
  572                  ty::re_free(*b)
  573              } else if rm.sub_free_region(*b, *a) {
  574                  ty::re_free(*a)
  575              } else {
  576                  ty::re_static
  577              }
  578          }
  579      }

  581      fn glb_concrete_regions(&self,
  582                              a: Region,
  583                              b: Region)
  584                           -> cres<Region> {
  585          debug!("glb_concrete_regions(%?, %?)", a, b);
  586          match (a, b) {
  587              (re_static, r) | (r, re_static) => {
  588                  // static lives longer than everything else
  589                  Ok(r)
  590              }
  591  
  592              (re_empty, _) | (_, re_empty) => {
  593                  // nothing lives shorter than everything else
  594                  Ok(re_empty)
  595              }
  596  
  597              (re_infer(ReVar(v_id)), _) |
  598              (_, re_infer(ReVar(v_id))) => {
  599                  self.tcx.sess.span_bug(
  600                      self.var_origins[v_id.to_uint()].span(),
  601                      fmt!("glb_concrete_regions invoked with \
  602                            non-concrete regions: %?, %?", a, b));
  603              }
  604  
  605              (re_free(ref fr), s @ re_scope(s_id)) |
  606              (s @ re_scope(s_id), re_free(ref fr)) => {
  607                  // Free region is something "at least as big as
  608                  // `fr.scope_id`."  If we find that the scope `fr.scope_id` is bigger
  609                  // than the scope `s_id`, then we can say that the GLB
  610                  // is the scope `s_id`.  Otherwise, as we do not know
  611                  // big the free region is precisely, the GLB is undefined.
  612                  let rm = self.tcx.region_maps;
  613                  match rm.nearest_common_ancestor(fr.scope_id, s_id) {
  614                      Some(r_id) if r_id == fr.scope_id => Ok(s),
  615                      _ => Err(ty::terr_regions_no_overlap(b, a))
  616                  }
  617              }
  618  
  619              (re_scope(a_id), re_scope(b_id)) => {
  620                  self.intersect_scopes(a, b, a_id, b_id)
  621              }
  622  
  623              (re_free(ref a_fr), re_free(ref b_fr)) => {
  624                  self.glb_free_regions(a_fr, b_fr)
  625              }
  626  
  627              // For these types, we cannot define any additional
  628              // relationship:
  629              (re_infer(ReSkolemized(*)), _) |
  630              (_, re_infer(ReSkolemized(*))) |
  631              (re_bound(_), re_bound(_)) |
  632              (re_bound(_), re_free(_)) |
  633              (re_bound(_), re_scope(_)) |
  634              (re_free(_), re_bound(_)) |
  635              (re_scope(_), re_bound(_)) => {
  636                  if a == b {
  637                      Ok(a)
  638                  } else {
  639                      Err(ty::terr_regions_no_overlap(b, a))
  640                  }
  641              }
  642          }
  643      }

  645      fn glb_free_regions(&self,
  646                          a: &FreeRegion,
  647                          b: &FreeRegion) -> cres<ty::Region>
  648      {
  649          /*!
  650           * Computes a region that is enclosed by both free region arguments,
  651           * if any. Guarantees that if the same two regions are given as argument,
  652           * in any order, a consistent result is returned.
  653           */
  654  
  655          return match a.cmp(b) {
  656              Less => helper(self, a, b),
  657              Greater => helper(self, b, a),
  658              Equal => Ok(ty::re_free(*a))
  659          };
  660  
  661          fn helper(this: &RegionVarBindings,
  662                    a: &FreeRegion,
  663                    b: &FreeRegion) -> cres<ty::Region>
  664          {
  665              let rm = this.tcx.region_maps;
  666              if rm.sub_free_region(*a, *b) {
  667                  Ok(ty::re_free(*a))
  668              } else if rm.sub_free_region(*b, *a) {
  669                  Ok(ty::re_free(*b))
  670              } else {
  671                  this.intersect_scopes(ty::re_free(*a), ty::re_free(*b),
  672                                        a.scope_id, b.scope_id)
  673              }
  674          }
  675      }

  678                           origin: SubregionOrigin,
  679                           terr: &ty::type_err) {
  680          let terr_str = ty::type_err_to_str(self.tcx, terr);
  681          self.tcx.sess.span_err(origin.span(), terr_str);
  682      }

  684      fn intersect_scopes(&self,
  685                          region_a: ty::Region,
  686                          region_b: ty::Region,
  687                          scope_a: ast::NodeId,
  688                          scope_b: ast::NodeId) -> cres<Region>
  689      {
  690          // We want to generate the intersection of two
  691          // scopes or two free regions.  So, if one of
  692          // these scopes is a subscope of the other, return
  693          // it.  Otherwise fail.
  694          debug!("intersect_scopes(scope_a=%?, scope_b=%?, region_a=%?, region_b=%?)",
  695                 scope_a, scope_b, region_a, region_b);
  696          let rm = self.tcx.region_maps;
  697          match rm.nearest_common_ancestor(scope_a, scope_b) {
  698              Some(r_id) if scope_a == r_id => Ok(re_scope(scope_b)),
  699              Some(r_id) if scope_b == r_id => Ok(re_scope(scope_a)),
  700              _ => Err(ty::terr_regions_no_overlap(region_a, region_b))
  701          }
  702      }
  703  }

  724  impl RegionVarBindings {
  725      fn infer_variable_values(&self,
  726                               errors: &mut OptVec<RegionResolutionError>)
  727                               -> ~[VarValue] {
  728          let mut var_data = self.construct_var_data();
  729          self.expansion(var_data);
  730          self.contraction(var_data);
  731          self.collect_concrete_region_errors(errors);
  732          self.extract_values_and_collect_conflicts(var_data, errors)
  733      }
  734  
  735      fn construct_var_data(&self) -> ~[VarData] {
  736          vec::from_fn(self.num_vars(), |_| {
  737              VarData {
  738                  // All nodes are initially classified as contracting; during
  739                  // the expansion phase, we will shift the classification for
  740                  // those nodes that have a concrete region predecessor to
  741                  // Expanding.
  742                  classification: Contracting,
  743                  value: NoValue,
  744              }
  745          })
  746      }
  747  
  748      fn expansion(&self, var_data: &mut [VarData]) {
  749          do self.iterate_until_fixed_point("Expansion") |constraint| {
  750              match *constraint {
  751                ConstrainRegSubVar(a_region, b_vid) => {
  752                  let b_data = &mut var_data[b_vid.to_uint()];
  753                  self.expand_node(a_region, b_vid, b_data)
  754                }
  755                ConstrainVarSubVar(a_vid, b_vid) => {
  756                  match var_data[a_vid.to_uint()].value {
  757                    NoValue | ErrorValue => false,
  758                    Value(a_region) => {
  759                      let b_node = &mut var_data[b_vid.to_uint()];
  760                      self.expand_node(a_region, b_vid, b_node)
  761                    }
  762                  }
  763                }
  764                ConstrainVarSubReg(*) => {
  765                  // This is a contraction constraint.  Ignore it.
  766                  false
  767                }
  768                ConstrainRegSubReg(*) => {
  769                  // No region variables involved. Ignore.
  770                  false
  771                }
  772              }
  773          }
  774      }
  775  
  776      fn expand_node(&self,
  777                     a_region: Region,
  778                     b_vid: RegionVid,
  779                     b_data: &mut VarData)
  780                     -> bool {
  781          debug!("expand_node(%?, %? == %?)",
  782                 a_region, b_vid, b_data.value);
  783  
  784          b_data.classification = Expanding;
  785          match b_data.value {
  786            NoValue => {
  787              debug!("Setting initial value of %? to %?", b_vid, a_region);
  788  
  789              b_data.value = Value(a_region);
  790              return true;
  791            }
  792  
  793            Value(cur_region) => {
  794              let lub = self.lub_concrete_regions(a_region, cur_region);
  795              if lub == cur_region {
  796                  return false;
  797              }
  798  
  799              debug!("Expanding value of %? from %? to %?",
  800                     b_vid, cur_region, lub);
  801  
  802              b_data.value = Value(lub);
  803              return true;
  804            }
  805  
  806            ErrorValue => {
  807              return false;
  808            }
  809          }
  810      }

  812      fn contraction(&self,
  813                     var_data: &mut [VarData]) {
  814          do self.iterate_until_fixed_point("Contraction") |constraint| {
  815              match *constraint {
  816                ConstrainRegSubVar(*) => {
  817                  // This is an expansion constraint.  Ignore.
  818                  false
  819                }
  820                ConstrainVarSubVar(a_vid, b_vid) => {
  821                  match var_data[b_vid.to_uint()].value {
  822                    NoValue | ErrorValue => false,
  823                    Value(b_region) => {
  824                      let a_data = &mut var_data[a_vid.to_uint()];
  825                      self.contract_node(a_vid, a_data, b_region)
  826                    }
  827                  }
  828                }
  829                ConstrainVarSubReg(a_vid, b_region) => {
  830                  let a_data = &mut var_data[a_vid.to_uint()];
  831                  self.contract_node(a_vid, a_data, b_region)
  832                }
  833                ConstrainRegSubReg(*) => {
  834                  // No region variables involved. Ignore.
  835                  false
  836                }
  837              }
  838          }
  839      }
  840  
  841      fn contract_node(&self,
  842                       a_vid: RegionVid,
  843                       a_data: &mut VarData,
  844                       b_region: Region)
  845                       -> bool {
  846          debug!("contract_node(%? == %?/%?, %?)",
  847                 a_vid, a_data.value, a_data.classification, b_region);
  848  
  849          return match a_data.value {
  850              NoValue => {
  851                  assert_eq!(a_data.classification, Contracting);
  852                  a_data.value = Value(b_region);
  853                  true // changed
  854              }
  855  
  856              ErrorValue => {
  857                  false // no change
  858              }
  859  
  860              Value(a_region) => {
  861                  match a_data.classification {
  862                      Expanding => {
  863                          check_node(self, a_vid, a_data, a_region, b_region)
  864                      }
  865                      Contracting => {
  866                          adjust_node(self, a_vid, a_data, a_region, b_region)
  867                      }
  868                  }
  869              }
  870          };

  872          fn check_node(this: &RegionVarBindings,
  873                        a_vid: RegionVid,
  874                        a_data: &mut VarData,
  875                        a_region: Region,
  876                        b_region: Region)
  877                     -> bool {
  878              if !this.is_subregion_of(a_region, b_region) {
  879                  debug!("Setting %? to ErrorValue: %? not subregion of %?",
  880                         a_vid, a_region, b_region);
  881                  a_data.value = ErrorValue;
  882              }
  883              false
  884          }

  886          fn adjust_node(this: &RegionVarBindings,
  887                         a_vid: RegionVid,
  888                         a_data: &mut VarData,

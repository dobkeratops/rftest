   11  /*!
   12  
   13  Error Reporting Code for the inference engine
   14  
   15  Because of the way inference, and in particular region inference,
   16  works, it often happens that errors are not detected until far after
   17  the relevant line of code has been type-checked. Therefore, there is
   18  an elaborate system to track why a particular constraint in the
   19  inference graph arose so that we can explain to the user what gave
   20  rise to a patricular error.
   21  
   22  The basis of the system are the "origin" types. An "origin" is the
   23  reason that a constraint or inference variable arose. There are
   24  different "origin" enums for different kinds of constraints/variables
   25  (e.g., `TypeOrigin`, `RegionVariableOrigin`). An origin always has
   26  a span, but also more information so that we can generate a meaningful
   27  error message.
   28  
   29  Having a catalogue of all the different reasons an error can arise is
   30  also useful for other reasons, like cross-referencing FAQs etc, though
   31  we are not really taking advantage of this yet.
   32  
   33  # Region Inference
   34  
   35  Region inference is particularly tricky because it always succeeds "in
   36  the moment" and simply registers a constraint. Then, at the end, we
   37  can compute the full graph and report errors, so we need to be able to
   38  store and later report what gave rise to the conflicting constraints.
   39  
   40  # Subtype Trace
   41  
   42  Determing whether `T1 <: T2` often involves a number of subtypes and
   43  subconstraints along the way. A "TypeTrace" is an extended version
   44  of an origin that traces the types and other values that were being
   45  compared. It is not necessarily comprehensive (in fact, at the time of
   46  this writing it only tracks the root values being compared) but I'd
   47  like to extend it to include significant "waypoints". For example, if
   48  you are comparing `(T1, T2) <: (T3, T4)`, and the problem is that `T2
   49  <: T4` fails, I'd like the trace to include enough information to say
   50  "in the 2nd element of the tuple". Similarly, failures when comparing
   51  arguments or return types in fn types should be able to cite the
   52  specific position, etc.
   53  
   54  # Reality vs plan
   55  
   56  Of course, there is still a LOT of code in typeck that has yet to be
   57  ported to this system, and which relies on string concatenation at the
   58  time of error detection.
   59  
   60  */

  156      fn report_region_errors(&self,
  157                              errors: &Vec<RegionResolutionError>) {
  158          let p_errors = self.process_errors(errors);
  159          let errors = if p_errors.is_empty() { errors } else { &p_errors };
  160          for error in errors.iter() {
  161              match *error {
  162                  ConcreteFailure(origin, sub, sup) => {
  163                      self.report_concrete_failure(origin, sub, sup);
  164                  }
  165  
  166                  SubSupConflict(var_origin,
  167                                 sub_origin, sub_r,
  168                                 sup_origin, sup_r) => {
  169                      self.report_sub_sup_conflict(var_origin,
  170                                                   sub_origin, sub_r,
  171                                                   sup_origin, sup_r);
  172                  }
  173  
  174                  SupSupConflict(var_origin,
  175                                 origin1, r1,
  176                                 origin2, r2) => {
  177                      self.report_sup_sup_conflict(var_origin,
  178                                                   origin1, r1,
  179                                                   origin2, r2);
  180                  }
  181  
  182                  ProcessedErrors(ref var_origins,
  183                                  ref trace_origins,
  184                                  ref same_regions) => {
  185                      if !same_regions.is_empty() {
  186                          self.report_processed_errors(var_origins.as_slice(),
  187                                                       trace_origins.as_slice(),
  188                                                       same_regions.as_slice());
  189                      }
  190                  }
  191              }
  192          }
  193      }
  194  
  195      // This method goes through all the errors and try to group certain types
  196      // of error together, for the purpose of suggesting explicit lifetime
  197      // parameters to the user. This is done so that we can have a more
  198      // complete view of what lifetimes should be the same.
  199      // If the return value is an empty vector, it means that processing
  200      // failed (so the return value of this method should not be used)
  201      fn process_errors(&self, errors: &Vec<RegionResolutionError>)
  202                        -> Vec<RegionResolutionError> {
  203          debug!("process_errors()");
  204          let mut var_origins = Vec::new();
  205          let mut trace_origins = Vec::new();
  206          let mut same_regions = Vec::new();
  207          let mut processed_errors = Vec::new();
  208          for error in errors.iter() {
  209              match *error {
  210                  ConcreteFailure(origin, sub, sup) => {
  211                      debug!("processing ConcreteFailure")
  212                      let trace = match origin {
  213                          infer::Subtype(trace) => Some(trace),
  214                          _ => None,
  215                      };
  216                      match free_regions_from_same_fn(self.tcx, sub, sup) {
  217                          Some(ref same_frs) if trace.is_some() => {
  218                              let trace = trace.unwrap();
  219                              let terr = ty::terr_regions_does_not_outlive(sup,
  220                                                                           sub);
  221                              trace_origins.push((trace, terr));
  222                              append_to_same_regions(&mut same_regions, same_frs);
  223                          }
  224                          _ => processed_errors.push((*error).clone()),
  225                      }
  226                  }
  227                  SubSupConflict(var_origin, _, sub_r, _, sup_r) => {
  228                      debug!("processing SubSupConflict")
  229                      match free_regions_from_same_fn(self.tcx, sub_r, sup_r) {
  230                          Some(ref same_frs) => {
  231                              var_origins.push(var_origin);
  232                              append_to_same_regions(&mut same_regions, same_frs);
  233                          }
  234                          None => processed_errors.push((*error).clone()),
  235                      }
  236                  }
  237                  SupSupConflict(..) => processed_errors.push((*error).clone()),
  238                  _ => ()  // This shouldn't happen
  239              }
  240          }
  241          if !same_regions.is_empty() {
  242              let common_scope_id = same_regions.get(0).scope_id;
  243              for sr in same_regions.iter() {
  244                  // Since ProcessedErrors is used to reconstruct the function
  245                  // declaration, we want to make sure that they are, in fact,
  246                  // from the same scope
  247                  if sr.scope_id != common_scope_id {
  248                      debug!("returning empty result from process_errors because
  249                              {} != {}", sr.scope_id, common_scope_id);
  250                      return vec!();
  251                  }
  252              }
  253              let pe = ProcessedErrors(var_origins, trace_origins, same_regions);
  254              debug!("errors processed: {:?}", pe);
  255              processed_errors.push(pe);
  256          }
  257          return processed_errors;

  279          fn free_regions_from_same_fn(tcx: &ty::ctxt,
  280                                       sub: Region,
  281                                       sup: Region)
  282                                       -> Option<FreeRegionsFromSameFn> {
  283              debug!("free_regions_from_same_fn(sub={:?}, sup={:?})", sub, sup);
  284              let (scope_id, fr1, fr2) = match (sub, sup) {
  285                  (ReFree(fr1), ReFree(fr2)) => {
  286                      if fr1.scope_id != fr2.scope_id {
  287                          return None
  288                      }
  289                      assert!(fr1.scope_id == fr2.scope_id);
  290                      (fr1.scope_id, fr1, fr2)
  291                  },
  292                  _ => return None
  293              };
  294              let parent = tcx.map.get_parent(scope_id);
  295              let parent_node = tcx.map.find(parent);
  296              match parent_node {
  297                  Some(node) => match node {
  298                      ast_map::NodeItem(item) => match item.node {
  299                          ast::ItemFn(..) => {
  300                              Some(FreeRegionsFromSameFn::new(fr1, fr2, scope_id))
  301                          },
  302                          _ => None
  303                      },
  304                      ast_map::NodeMethod(..) => {
  305                          Some(FreeRegionsFromSameFn::new(fr1, fr2, scope_id))
  306                      },
  307                      _ => None
  308                  },
  309                  None => {
  310                      debug!("no parent node of scope_id {}", scope_id)
  311                      None
  312                  }
  313              }
  314          }

  316          fn append_to_same_regions(same_regions: &mut Vec<SameRegions>,
  317                                    same_frs: &FreeRegionsFromSameFn) {
  318              let scope_id = same_frs.scope_id;

  322                     && scope_id == sr.scope_id {
  323                      sr.push(sub_fr.bound_region);
  324                      return
  325                  }
  326              }
  327              same_regions.push(SameRegions {
  328                  scope_id: scope_id,
  329                  regions: vec!(sub_fr.bound_region, sup_fr.bound_region)
  330              })
  331          }
  332      }

  750      fn rebuild(&self)
  751                 -> (ast::FnDecl, Option<ast::ExplicitSelf_>, ast::Generics) {
  752          let mut expl_self_opt = self.expl_self_opt;
  753          let mut inputs = self.fn_decl.inputs.clone();
  754          let mut output = self.fn_decl.output;
  755          let mut ty_params = self.generics.ty_params.clone();
  756          let mut kept_lifetimes = HashSet::new();
  757          for sr in self.same_regions.iter() {
  758              self.cur_anon.set(0);
  759              self.offset_cur_anon();
  760              let (anon_nums, region_names) =
  761                                  self.extract_anon_nums_and_names(sr);
  762              let (lifetime, fresh_or_kept) = self.pick_lifetime(&region_names);
  763              match fresh_or_kept {
  764                  Kept => { kept_lifetimes.insert(lifetime.name); }
  765                  _ => ()
  766              }
  767              expl_self_opt = self.rebuild_expl_self(expl_self_opt, lifetime,
  768                                                     &anon_nums, &region_names);
  769              inputs = self.rebuild_args_ty(inputs.as_slice(), lifetime,
  770                                            &anon_nums, &region_names);
  771              output = self.rebuild_arg_ty_or_output(output, lifetime,
  772                                                     &anon_nums, &region_names);
  773              ty_params = self.rebuild_ty_params(ty_params, lifetime,
  774                                                 &region_names);
  775          }
  776          let fresh_lifetimes = self.life_giver.get_generated_lifetimes();
  777          let all_region_names = self.extract_all_region_names();
  778          let generics = self.rebuild_generics(self.generics,
  779                                               &fresh_lifetimes,
  780                                               &kept_lifetimes,
  781                                               &all_region_names,
  782                                               ty_params);
  783          let new_fn_decl = ast::FnDecl {
  784              inputs: inputs,
  785              output: output,
  786              cf: self.fn_decl.cf,
  787              variadic: self.fn_decl.variadic
  788          };
  789          (new_fn_decl, expl_self_opt, generics)
  790      }

  792      fn pick_lifetime(&self,
  793                       region_names: &HashSet<ast::Name>)
  794                       -> (ast::Lifetime, FreshOrKept) {
  795          if region_names.len() > 0 {
  796              // It's not necessary to convert the set of region names to a
  797              // vector of string and then sort them. However, it makes the
  798              // choice of lifetime name deterministic and thus easier to test.
  799              let mut names = Vec::new();
  800              for rn in region_names.iter() {
  801                  let lt_name = token::get_name(*rn).get().to_owned();
  802                  names.push(lt_name);
  803              }
  804              names.sort();
  805              let name = token::str_to_ident(names.get(0).as_slice()).name;
  806              return (name_to_dummy_lifetime(name), Kept);
  807          }
  808          return (self.life_giver.give_lifetime(), Fresh);
  809      }

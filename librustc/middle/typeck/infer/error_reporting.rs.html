   11  /*!
   12  
   13  Error Reporting Code for the inference engine
   14  
   15  Because of the way inference, and in particular region inference,
   16  works, it often happens that errors are not detected until far after
   17  the relevant line of code has been type-checked. Therefore, there is
   18  an elaborate system to track why a particular constraint in the
   19  inference graph arose so that we can explain to the user what gave
   20  rise to a patricular error.
   21  
   22  The basis of the system are the "origin" types. An "origin" is the
   23  reason that a constraint or inference variable arose. There are
   24  different "origin" enums for different kinds of constraints/variables
   25  (e.g., `TypeOrigin`, `RegionVariableOrigin`). An origin always has
   26  a span, but also more information so that we can generate a meaningful
   27  error message.
   28  
   29  Having a catalogue of all the different reasons an error can arise is
   30  also useful for other reasons, like cross-referencing FAQs etc, though
   31  we are not really taking advantage of this yet.
   32  
   33  # Region Inference
   34  
   35  Region inference is particularly tricky because it always succeeds "in
   36  the moment" and simply registers a constraint. Then, at the end, we
   37  can compute the full graph and report errors, so we need to be able to
   38  store and later report what gave rise to the conflicting constraints.
   39  
   40  # Subtype Trace
   41  
   42  Determing whether `T1 <: T2` often involves a number of subtypes and
   43  subconstraints along the way. A "TypeTrace" is an extended version
   44  of an origin that traces the types and other values that were being
   45  compared. It is not necessarily comprehensive (in fact, at the time of
   46  this writing it only tracks the root values being compared) but I'd
   47  like to extend it to include significant "waypoints". For example, if
   48  you are comparing `(T1, T2) <: (T3, T4)`, and the problem is that `T2
   49  <: T4` fails, I'd like the trace to include enough information to say
   50  "in the 2nd element of the tuple". Similarly, failures when comparing
   51  arguments or return types in fn types should be able to cite the
   52  specific position, etc.
   53  
   54  # Reality vs plan
   55  
   56  Of course, there is still a LOT of code in typeck that has yet to be
   57  ported to this system, and which relies on string concatenation at the
   58  time of error detection.
   59  
   60  */

  150  impl<'a> ErrorReporting for InferCtxt<'a> {
  151      fn report_region_errors(&self,
  152                              errors: &Vec<RegionResolutionError>) {
  153          let p_errors = self.process_errors(errors);
  154          let errors = if p_errors.is_empty() { errors } else { &p_errors };
  155          for error in errors.iter() {
  156              match *error {
  157                  ConcreteFailure(origin, sub, sup) => {
  158                      self.report_concrete_failure(origin, sub, sup);
  159                  }
  160  
  161                  SubSupConflict(var_origin,
  162                                 sub_origin, sub_r,
  163                                 sup_origin, sup_r) => {
  164                      self.report_sub_sup_conflict(var_origin,
  165                                                   sub_origin, sub_r,
  166                                                   sup_origin, sup_r);
  167                  }
  168  
  169                  SupSupConflict(var_origin,
  170                                 origin1, r1,
  171                                 origin2, r2) => {
  172                      self.report_sup_sup_conflict(var_origin,
  173                                                   origin1, r1,
  174                                                   origin2, r2);
  175                  }
  176  
  177                  ProcessedErrors(ref var_origins,
  178                                  ref trace_origins,
  179                                  ref same_regions) => {
  180                      if !same_regions.is_empty() {
  181                          self.report_processed_errors(var_origins.as_slice(),
  182                                                       trace_origins.as_slice(),
  183                                                       same_regions.as_slice());
  184                      }
  185                  }
  186              }
  187          }
  188      }
  189  
  190      // This method goes through all the errors and try to group certain types
  191      // of error together, for the purpose of suggesting explicit lifetime
  192      // parameters to the user. This is done so that we can have a more
  193      // complete view of what lifetimes should be the same.
  194      // If the return value is an empty vector, it means that processing
  195      // failed (so the return value of this method should not be used)
  196      fn process_errors(&self, errors: &Vec<RegionResolutionError>)
  197                        -> Vec<RegionResolutionError> {
  198          let mut var_origins = Vec::new();
  199          let mut trace_origins = Vec::new();
  200          let mut same_regions = Vec::new();
  201          let mut processed_errors = Vec::new();
  202          for error in errors.iter() {
  203              match *error {
  204                  ConcreteFailure(origin, sub, sup) => {
  205                      let trace = match origin {
  206                          infer::Subtype(trace) => Some(trace),
  207                          _ => None,
  208                      };
  209                      match free_regions_from_same_fn(self.tcx, sub, sup) {
  210                          Some(ref same_frs) if trace.is_some() => {
  211                              let trace = trace.unwrap();
  212                              let terr = ty::terr_regions_does_not_outlive(sup,
  213                                                                           sub);
  214                              trace_origins.push((trace, terr));
  215                              append_to_same_regions(&mut same_regions, same_frs);
  216                          }
  217                          _ => processed_errors.push((*error).clone()),
  218                      }
  219                  }
  220                  SubSupConflict(var_origin, _, sub_r, _, sup_r) => {
  221                      match free_regions_from_same_fn(self.tcx, sub_r, sup_r) {
  222                          Some(ref same_frs) => {
  223                              var_origins.push(var_origin);
  224                              append_to_same_regions(&mut same_regions, same_frs);
  225                          }
  226                          None => processed_errors.push((*error).clone()),
  227                      }
  228                  }
  229                  SupSupConflict(..) => processed_errors.push((*error).clone()),
  230                  _ => ()  // This shouldn't happen
  231              }
  232          }
  233          if !same_regions.is_empty() {
  234              let common_scope_id = same_regions.get(0).scope_id;
  235              for sr in same_regions.iter() {
  236                  // Since ProcessedErrors is used to reconstruct the function
  237                  // declaration, we want to make sure that they are, in fact,
  238                  // from the same scope
  239                  if sr.scope_id != common_scope_id {
  240                      return vec!();
  241                  }
  242              }
  243              let pe = ProcessedErrors(var_origins, trace_origins, same_regions);
  244              processed_errors.push(pe);
  245          }
  246          return processed_errors;
  247

  249          struct FreeRegionsFromSameFn {
  250              sub_fr: ty::FreeRegion,
  251              sup_fr: ty::FreeRegion,
  252              scope_id: ast::NodeId
  253          }
  254  
  255          fn free_regions_from_same_fn(tcx: &ty::ctxt,
  256                                       sub: Region,
  257                                       sup: Region)
  258                                       -> Option<FreeRegionsFromSameFn> {
  259              let (scope_id, fr1, fr2) = match (sub, sup) {
  260                  (ReFree(fr1), ReFree(fr2)) => {
  261                      if fr1.scope_id != fr2.scope_id {
  262                          return None
  263                      }
  264                      assert!(fr1.scope_id == fr2.scope_id);
  265                      (fr1.scope_id, fr1, fr2)
  266                  },
  267                  _ => return None
  268              };
  269              let parent = tcx.map.get_parent(scope_id);
  270              let parent_node = tcx.map.find(parent);
  271              match parent_node {
  272                  Some(node) => match node {
  273                      ast_map::NodeItem(item) => match item.node {
  274                          // FIXME: handle method
  275                          ast::ItemFn(..) => {
  276                              let fr_from_same_fn = FreeRegionsFromSameFn {
  277                                  sub_fr: fr1,
  278                                  sup_fr: fr2,
  279                                  scope_id: scope_id
  280                              };
  281                              Some(fr_from_same_fn)
  282                          },
  283                          _ => None
  284                      },
  285                      _ => None
  286                  },
  287                  None => None
  288              }
  289          }
  290  
  291          fn append_to_same_regions(same_regions: &mut Vec<SameRegions>,
  292                                    same_frs: &FreeRegionsFromSameFn) {
  293              let scope_id = same_frs.scope_id;
  294              let (sub_fr, sup_fr) = (same_frs.sub_fr, same_frs.sup_fr);
  295              for sr in same_regions.mut_iter() {
  296                  if sr.contains(&sup_fr.bound_region)
  297                     && scope_id == sr.scope_id {
  298                      sr.push(sub_fr.bound_region);
  299                      return
  300                  }
  301              }
  302              same_regions.push(SameRegions {
  303                  scope_id: scope_id,
  304                  regions: vec!(sub_fr.bound_region, sup_fr.bound_region)
  305              })
  306          }
  307      }

  309      fn report_type_error(&self, trace: TypeTrace, terr: &ty::type_err) {
  310          let expected_found_str = match self.values_str(&trace.values) {
  311              Some(v) => v,
  312              None => {
  313                  return; /* derived error */
  314              }
  315          };
  316  
  317          let message_root_str = match trace.origin {
  318              infer::Misc(_) => "mismatched types",
  319              infer::MethodCompatCheck(_) => "method not compatible with trait",
  320              infer::ExprAssignable(_) => "mismatched types",
  321              infer::RelateTraitRefs(_) => "mismatched traits",
  322              infer::RelateSelfType(_) => "mismatched types",
  323              infer::MatchExpression(_) => "match arms have incompatible types",
  324              infer::IfExpression(_) => "if and else have incompatible types",
  325          };
  326  
  327          self.tcx.sess.span_err(
  328              trace.origin.span(),
  329              format!("{}: {} ({})",
  330                   message_root_str,
  331                   expected_found_str,
  332                   ty::type_err_to_str(self.tcx, terr)));
  333      }

  342      fn values_str(&self, values: &ValuePairs) -> Option<~str> {
  343          /*!
  344           * Returns a string of the form "expected `{}` but found `{}`",
  345           * or None if this is a derived error.
  346           */
  347          match *values {
  348              infer::Types(ref exp_found) => {
  349                  self.expected_found_str(exp_found)
  350              }
  351              infer::TraitRefs(ref exp_found) => {
  352                  self.expected_found_str(exp_found)
  353              }
  354          }
  355      }

  357      fn expected_found_str<T:UserString+Resolvable>(
  358          &self,
  359          exp_found: &ty::expected_found<T>)
  360          -> Option<~str>
  361      {
  362          let expected = exp_found.expected.resolve(self);
  363          if expected.contains_error() {
  364              return None;
  365          }
  366  
  367          let found = exp_found.found.resolve(self);
  368          if found.contains_error() {
  369              return None;
  370          }
  371  
  372          Some(format!("expected `{}` but found `{}`",
  373                    expected.user_string(self.tcx),
  374                    found.user_string(self.tcx)))
  375      }

  377      fn report_concrete_failure(&self,
  378                                 origin: SubregionOrigin,
  379                                 sub: Region,
  380                                 sup: Region) {

  382              infer::Subtype(trace) => {
  383                  let terr = ty::terr_regions_does_not_outlive(sup, sub);
  384                  self.report_and_explain_type_error(trace, &terr);
  385              }
  386              infer::Reborrow(span) => {
  387                  self.tcx.sess.span_err(
  388                      span,
  389                      "lifetime of reference outlines \
  390                       lifetime of borrowed content...");
  391                  note_and_explain_region(
  392                      self.tcx,
  393                      "...the reference is valid for ",
  394                      sub,
  395                      "...");
  396                  note_and_explain_region(
  397                      self.tcx,
  398                      "...but the borrowed content is only valid for ",
  399                      sup,
  400                      "");
  401              }
  402              infer::ReborrowUpvar(span, ref upvar_id) => {
  403                  self.tcx.sess.span_err(
  404                      span,
  405                      format!("lifetime of borrowed pointer outlives \
  406                              lifetime of captured variable `{}`...",
  407                              ty::local_var_name_str(self.tcx, upvar_id.var_id).get().to_str()));
  408                  note_and_explain_region(
  409                      self.tcx,
  410                      "...the borrowed pointer is valid for ",
  411                      sub,
  412                      "...");
  413                  note_and_explain_region(
  414                      self.tcx,
  415                      format!("...but `{}` is only valid for ",
  416                              ty::local_var_name_str(self.tcx, upvar_id.var_id).get().to_str()),
  417                      sup,
  418                      "");
  419              }
  420              infer::InfStackClosure(span) => {
  421                  self.tcx.sess.span_err(
  422                      span,
  423                      "closure outlives stack frame");
  424                  note_and_explain_region(
  425                      self.tcx,
  426                      "...the closure must be valid for ",
  427                      sub,
  428                      "...");
  429                  note_and_explain_region(
  430                      self.tcx,
  431                      "...but the closure's stack frame is only valid for ",

  673      fn_decl: ast::P<ast::FnDecl>,
  674      generics: &'a ast::Generics,
  675      same_regions: &'a [SameRegions],
  676      life_giver: LifeGiver,
  677      cur_anon: Cell<uint>,
  678      inserted_anons: RefCell<HashSet<uint>>,
  679  }

  681  impl<'a> Rebuilder<'a> {
  682      fn new(tcx: &'a ty::ctxt,
  683             fn_decl: ast::P<ast::FnDecl>,
  684             generics: &'a ast::Generics,
  685             same_regions: &'a [SameRegions])
  686             -> Rebuilder<'a> {
  687          Rebuilder {
  688              tcx: tcx,
  689              fn_decl: fn_decl,
  690              generics: generics,
  691              same_regions: same_regions,
  692              life_giver: LifeGiver::with_taken(generics.lifetimes.as_slice()),
  693              cur_anon: Cell::new(0),
  694              inserted_anons: RefCell::new(HashSet::new()),
  695          }
  696      }

  698      fn rebuild(&self) -> (Vec<ast::Arg>, ast::P<ast::Ty>, ast::Generics) {
  699          let mut inputs = self.fn_decl.inputs.clone();
  700          let mut output = self.fn_decl.output;
  701          for sr in self.same_regions.iter() {
  702              self.cur_anon.set(0);
  703              self.offset_cur_anon();
  704              let (anon_nums, region_names) =
  705                                  self.extract_anon_nums_and_names(sr);
  706              let lifetime = self.life_giver.give_lifetime();
  707              inputs = self.rebuild_args_ty(inputs.as_slice(), lifetime,
  708                                            &anon_nums, &region_names);
  709              output = self.rebuild_arg_ty_or_output(output, lifetime,
  710                                                     &anon_nums, &region_names);
  711          }
  712          let generated_lifetimes = self.life_giver.get_generated_lifetimes();
  713          let all_region_names = self.extract_all_region_names();
  714          let generics = self.rebuild_generics(self.generics,
  715                                               generated_lifetimes,
  716                                               &all_region_names);
  717          (inputs, output, generics)
  718      }

  720      fn extract_anon_nums_and_names(&self, same_regions: &SameRegions)
  721                                     -> (HashSet<uint>, HashSet<ast::Name>) {
  722          let mut anon_nums = HashSet::new();
  723          let mut region_names = HashSet::new();

  725              match *br {
  726                  ty::BrAnon(i) => {
  727                      anon_nums.insert(i);
  728                  }
  729                  ty::BrNamed(_, name) => {
  730                      region_names.insert(name);
  731                  }
  732                  _ => ()
  733              }
  734          }
  735          (anon_nums, region_names)
  736      }

  795      fn rebuild_args_ty(&self,
  796                         inputs: &[ast::Arg],

  798                         anon_nums: &HashSet<uint>,
  799                         region_names: &HashSet<ast::Name>)
  800                         -> Vec<ast::Arg> {
  801          let mut new_inputs = Vec::new();
  802          for arg in inputs.iter() {
  803              let new_ty = self.rebuild_arg_ty_or_output(arg.ty, lifetime,
  804                                                         anon_nums, region_names);
  805              let possibly_new_arg = ast::Arg {
  806                  ty: new_ty,
  807                  pat: arg.pat,
  808                  id: arg.id

  811          }

  815      fn rebuild_arg_ty_or_output(&self,
  816                                  ty: ast::P<ast::Ty>,
  817                                  lifetime: ast::Lifetime,
  818                                  anon_nums: &HashSet<uint>,
  819                                  region_names: &HashSet<ast::Name>)
  820                                  -> ast::P<ast::Ty> {
  821          let mut new_ty = ty;
  822          let mut ty_queue = vec!(ty);
  823          let mut cur_ty;
  824          while !ty_queue.is_empty() {
  825              cur_ty = ty_queue.shift().unwrap();
  826              match cur_ty.node {
  827                  ast::TyRptr(lt_opt, mut_ty) => {
  828                      match lt_opt {
  829                          Some(lt) => if region_names.contains(&lt.name) {
  830                              new_ty = self.rebuild_ty(new_ty, cur_ty,
  831                                                       lifetime, None);
  832                          },
  833                          None => {
  834                              let anon = self.cur_anon.get();
  835                              if anon_nums.contains(&anon) {
  836                                  new_ty = self.rebuild_ty(new_ty, cur_ty,
  837                                                           lifetime, None);
  838                                  self.track_anon(anon);
  839                              }
  840                              self.inc_and_offset_cur_anon(1);
  841                          }
  842                      }
  843                      ty_queue.push(mut_ty.ty);
  844                  }
  845                  ast::TyPath(ref path, _, id) => {
  846                      let a_def = match self.tcx.def_map.borrow().find(&id) {
  847                          None => self.tcx.sess.fatal(format!("unbound path {}",
  848                                                      pprust::path_to_str(path))),
  849                          Some(&d) => d
  850                      };
  851                      match a_def {
  852                          ast::DefTy(did) | ast::DefStruct(did) => {
  853                              let ty::ty_param_bounds_and_ty {
  854                                  generics: generics,
  855                                  ty: _
  856                              } = ty::lookup_item_type(self.tcx, did);
  857  
  858                              let expected = generics.region_param_defs().len();
  859                              let lifetimes = &path.segments.last()
  860                                                   .unwrap().lifetimes;
  861                              let mut insert = Vec::new();
  862                              if lifetimes.len() == 0 {
  863                                  let anon = self.cur_anon.get();
  864                                  for (i, a) in range(anon,
  865                                                      anon+expected).enumerate() {
  866                                      if anon_nums.contains(&a) {
  867                                          insert.push(i);
  868                                      }
  869                                      self.track_anon(a);
  870                                  }
  871                                  self.inc_and_offset_cur_anon(expected);
  872                              } else {
  873                                  for (i, lt) in lifetimes.iter().enumerate() {
  874                                      if region_names.contains(&lt.name) {
  875                                          insert.push(i);
  876                                      }
  877                                  }
  878                              }
  879                              for i in insert.iter() {
  880                                  new_ty = self.rebuild_ty(new_ty, cur_ty,
  881                                                           lifetime,
  882                                                           Some((*i, expected)));
  883                              }
  884                          }
  885                          _ => ()
  886                      }
  887  
  888                  }
  889                  _ => ty_queue.push_all_move(ast_util::get_inner_tys(cur_ty))
  890              }
  891          }
  892          new_ty
  893      }

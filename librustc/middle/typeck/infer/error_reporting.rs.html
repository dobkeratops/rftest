   11  /*!
   12  
   13  Error Reporting Code for the inference engine
   14  
   15  Because of the way inference, and in particular region inference,
   16  works, it often happens that errors are not detected until far after
   17  the relevant line of code has been type-checked. Therefore, there is
   18  an elaborate system to track why a particular constraint in the
   19  inference graph arose so that we can explain to the user what gave
   20  rise to a patricular error.
   21  
   22  The basis of the system are the "origin" types. An "origin" is the
   23  reason that a constraint or inference variable arose. There are
   24  different "origin" enums for different kinds of constraints/variables
   25  (e.g., `TypeOrigin`, `RegionVariableOrigin`). An origin always has
   26  a span, but also more information so that we can generate a meaningful
   27  error message.
   28  
   29  Having a catalogue of all the different reasons an error can arise is
   30  also useful for other reasons, like cross-referencing FAQs etc, though
   31  we are not really taking advantage of this yet.
   32  
   33  # Region Inference
   34  
   35  Region inference is particularly tricky because it always succeeds "in
   36  the moment" and simply registers a constraint. Then, at the end, we
   37  can compute the full graph and report errors, so we need to be able to
   38  store and later report what gave rise to the conflicting constraints.
   39  
   40  # Subtype Trace
   41  
   42  Determing whether `T1 <: T2` often involves a number of subtypes and
   43  subconstraints along the way. A "TypeTrace" is an extended version
   44  of an origin that traces the types and other values that were being
   45  compared. It is not necessarily comprehensive (in fact, at the time of
   46  this writing it only tracks the root values being compared) but I'd
   47  like to extend it to include significant "waypoints". For example, if
   48  you are comparing `(T1, T2) <: (T3, T4)`, and the problem is that `T2
   49  <: T4` fails, I'd like the trace to include enough information to say
   50  "in the 2nd element of the tuple". Similarly, failures when comparing
   51  arguments or return types in fn types should be able to cite the
   52  specific position, etc.
   53  
   54  # Reality vs plan
   55  
   56  Of course, there is still a LOT of code in typeck that has yet to be
   57  ported to this system, and which relies on string concatenation at the
   58  time of error detection.
   59  
   60  */

  151  impl<'a> ErrorReporting for InferCtxt<'a> {
  152      fn report_region_errors(&self,
  153                              errors: &Vec<RegionResolutionError>) {
  154          let p_errors = self.process_errors(errors);
  155          let errors = if p_errors.is_empty() { errors } else { &p_errors };
  156          for error in errors.iter() {
  157              match *error {
  158                  ConcreteFailure(origin, sub, sup) => {
  159                      self.report_concrete_failure(origin, sub, sup);
  160                  }
  161  
  162                  SubSupConflict(var_origin,
  163                                 sub_origin, sub_r,
  164                                 sup_origin, sup_r) => {
  165                      self.report_sub_sup_conflict(var_origin,
  166                                                   sub_origin, sub_r,
  167                                                   sup_origin, sup_r);
  168                  }
  169  
  170                  SupSupConflict(var_origin,
  171                                 origin1, r1,
  172                                 origin2, r2) => {
  173                      self.report_sup_sup_conflict(var_origin,
  174                                                   origin1, r1,
  175                                                   origin2, r2);
  176                  }
  177  
  178                  ProcessedErrors(ref var_origins,
  179                                  ref trace_origins,
  180                                  ref same_regions) => {
  181                      if !same_regions.is_empty() {
  182                          self.report_processed_errors(var_origins.as_slice(),
  183                                                       trace_origins.as_slice(),
  184                                                       same_regions.as_slice());
  185                      }
  186                  }
  187              }
  188          }
  189      }
  190  
  191      // This method goes through all the errors and try to group certain types
  192      // of error together, for the purpose of suggesting explicit lifetime
  193      // parameters to the user. This is done so that we can have a more
  194      // complete view of what lifetimes should be the same.
  195      // If the return value is an empty vector, it means that processing
  196      // failed (so the return value of this method should not be used)
  197      fn process_errors(&self, errors: &Vec<RegionResolutionError>)
  198                        -> Vec<RegionResolutionError> {
  199          debug!("process_errors()");
  200          let mut var_origins = Vec::new();
  201          let mut trace_origins = Vec::new();
  202          let mut same_regions = Vec::new();
  203          let mut processed_errors = Vec::new();
  204          for error in errors.iter() {
  205              match *error {
  206                  ConcreteFailure(origin, sub, sup) => {
  207                      debug!("processing ConcreteFailure")
  208                      let trace = match origin {
  209                          infer::Subtype(trace) => Some(trace),
  210                          _ => None,
  211                      };
  212                      match free_regions_from_same_fn(self.tcx, sub, sup) {
  213                          Some(ref same_frs) if trace.is_some() => {
  214                              let trace = trace.unwrap();
  215                              let terr = ty::terr_regions_does_not_outlive(sup,
  216                                                                           sub);
  217                              trace_origins.push((trace, terr));
  218                              append_to_same_regions(&mut same_regions, same_frs);
  219                          }
  220                          _ => processed_errors.push((*error).clone()),
  221                      }
  222                  }
  223                  SubSupConflict(var_origin, _, sub_r, _, sup_r) => {
  224                      debug!("processing SubSupConflict")
  225                      match free_regions_from_same_fn(self.tcx, sub_r, sup_r) {
  226                          Some(ref same_frs) => {
  227                              var_origins.push(var_origin);
  228                              append_to_same_regions(&mut same_regions, same_frs);
  229                          }
  230                          None => processed_errors.push((*error).clone()),
  231                      }
  232                  }
  233                  SupSupConflict(..) => processed_errors.push((*error).clone()),
  234                  _ => ()  // This shouldn't happen
  235              }
  236          }
  237          if !same_regions.is_empty() {
  238              let common_scope_id = same_regions.get(0).scope_id;
  239              for sr in same_regions.iter() {
  240                  // Since ProcessedErrors is used to reconstruct the function
  241                  // declaration, we want to make sure that they are, in fact,
  242                  // from the same scope
  243                  if sr.scope_id != common_scope_id {
  244                      debug!("returning empty result from process_errors because
  245                              {} != {}", sr.scope_id, common_scope_id);
  246                      return vec!();
  247                  }
  248              }
  249              let pe = ProcessedErrors(var_origins, trace_origins, same_regions);
  250              debug!("errors processed: {:?}", pe);
  251              processed_errors.push(pe);
  252          }
  253          return processed_errors;
  254

  256          struct FreeRegionsFromSameFn {
  257              sub_fr: ty::FreeRegion,
  258              sup_fr: ty::FreeRegion,
  259              scope_id: ast::NodeId
  260          }
  261  
  262          fn free_regions_from_same_fn(tcx: &ty::ctxt,
  263                                       sub: Region,
  264                                       sup: Region)
  265                                       -> Option<FreeRegionsFromSameFn> {
  266              debug!("free_regions_from_same_fn(sub={:?}, sup={:?})", sub, sup);
  267              let (scope_id, fr1, fr2) = match (sub, sup) {
  268                  (ReFree(fr1), ReFree(fr2)) => {
  269                      if fr1.scope_id != fr2.scope_id {
  270                          return None
  271                      }
  272                      assert!(fr1.scope_id == fr2.scope_id);
  273                      (fr1.scope_id, fr1, fr2)
  274                  },
  275                  _ => return None
  276              };
  277              let parent = tcx.map.get_parent(scope_id);
  278              let parent_node = tcx.map.find(parent);
  279              match parent_node {
  280                  Some(node) => match node {
  281                      ast_map::NodeItem(item) => match item.node {
  282                          // FIXME: handle method
  283                          ast::ItemFn(..) => {
  284                              let fr_from_same_fn = FreeRegionsFromSameFn {
  285                                  sub_fr: fr1,
  286                                  sup_fr: fr2,
  287                                  scope_id: scope_id
  288                              };
  289                              Some(fr_from_same_fn)
  290                          },
  291                          _ => None
  292                      },
  293                      _ => None
  294                  },
  295                  None => {
  296                      debug!("no parent node of scope_id {}", scope_id)
  297                      None
  298                  }
  299              }
  300          }
  301  
  302          fn append_to_same_regions(same_regions: &mut Vec<SameRegions>,
  303                                    same_frs: &FreeRegionsFromSameFn) {
  304              let scope_id = same_frs.scope_id;
  305              let (sub_fr, sup_fr) = (same_frs.sub_fr, same_frs.sup_fr);
  306              for sr in same_regions.mut_iter() {
  307                  if sr.contains(&sup_fr.bound_region)
  308                     && scope_id == sr.scope_id {
  309                      sr.push(sub_fr.bound_region);
  310                      return
  311                  }
  312              }
  313              same_regions.push(SameRegions {
  314                  scope_id: scope_id,
  315                  regions: vec!(sub_fr.bound_region, sup_fr.bound_region)
  316              })
  317          }
  318      }

  320      fn report_type_error(&self, trace: TypeTrace, terr: &ty::type_err) {
  321          let expected_found_str = match self.values_str(&trace.values) {
  322              Some(v) => v,
  323              None => {
  324                  return; /* derived error */
  325              }
  326          };
  327  
  328          let message_root_str = match trace.origin {
  329              infer::Misc(_) => "mismatched types",
  330              infer::MethodCompatCheck(_) => "method not compatible with trait",
  331              infer::ExprAssignable(_) => "mismatched types",
  332              infer::RelateTraitRefs(_) => "mismatched traits",
  333              infer::RelateSelfType(_) => "mismatched types",
  334              infer::MatchExpression(_) => "match arms have incompatible types",
  335              infer::IfExpression(_) => "if and else have incompatible types",
  336          };
  337  
  338          self.tcx.sess.span_err(
  339              trace.origin.span(),
  340              format!("{}: {} ({})",
  341                   message_root_str,
  342                   expected_found_str,
  343                   ty::type_err_to_str(self.tcx, terr)));
  344      }

  353      fn values_str(&self, values: &ValuePairs) -> Option<~str> {
  354          /*!
  355           * Returns a string of the form "expected `{}` but found `{}`",
  356           * or None if this is a derived error.
  357           */
  358          match *values {
  359              infer::Types(ref exp_found) => {
  360                  self.expected_found_str(exp_found)
  361              }
  362              infer::TraitRefs(ref exp_found) => {
  363                  self.expected_found_str(exp_found)
  364              }
  365          }
  366      }
  367  
  368      fn expected_found_str<T:UserString+Resolvable>(
  369          &self,
  370          exp_found: &ty::expected_found<T>)
  371          -> Option<~str>
  372      {
  373          let expected = exp_found.expected.resolve(self);
  374          if expected.contains_error() {
  375              return None;
  376          }
  377  
  378          let found = exp_found.found.resolve(self);
  379          if found.contains_error() {
  380              return None;
  381          }
  382  
  383          Some(format!("expected `{}` but found `{}`",
  384                    expected.user_string(self.tcx),
  385                    found.user_string(self.tcx)))
  386      }

  388      fn report_concrete_failure(&self,
  389                                 origin: SubregionOrigin,
  390                                 sub: Region,
  391                                 sup: Region) {
  392          match origin {
  393              infer::Subtype(trace) => {
  394                  let terr = ty::terr_regions_does_not_outlive(sup, sub);
  395                  self.report_and_explain_type_error(trace, &terr);
  396              }
  397              infer::Reborrow(span) => {
  398                  self.tcx.sess.span_err(
  399                      span,
  400                      "lifetime of reference outlines \
  401                       lifetime of borrowed content...");
  402                  note_and_explain_region(
  403                      self.tcx,
  404                      "...the reference is valid for ",
  405                      sub,
  406                      "...");
  407                  note_and_explain_region(
  408                      self.tcx,
  409                      "...but the borrowed content is only valid for ",
  410                      sup,
  411                      "");
  412              }
  413              infer::ReborrowUpvar(span, ref upvar_id) => {
  414                  self.tcx.sess.span_err(
  415                      span,
  416                      format!("lifetime of borrowed pointer outlives \
  417                              lifetime of captured variable `{}`...",
  418                              ty::local_var_name_str(self.tcx, upvar_id.var_id).get().to_str()));
  419                  note_and_explain_region(
  420                      self.tcx,
  421                      "...the borrowed pointer is valid for ",
  422                      sub,
  423                      "...");
  424                  note_and_explain_region(
  425                      self.tcx,
  426                      format!("...but `{}` is only valid for ",
  427                              ty::local_var_name_str(self.tcx, upvar_id.var_id).get().to_str()),
  428                      sup,

  657      fn give_suggestion(&self, same_regions: &[SameRegions]) {
  658          let scope_id = same_regions[0].scope_id;

  662              Some(node) => match node {
  663                  ast_map::NodeItem(item) => match item.node {
  664                      // FIXME: handling method
  665                      ast::ItemFn(ref fn_decl, _, _, ref gen, _) => {
  666                          Some((item, fn_decl, gen))
  667                      },
  668                      _ => None
  669                  },
  670                  _ => None
  671              },
  672              None => None
  673          };
  674          let (item, fn_decl, generics) = node_inner.expect("expect item fn");
  675          let rebuilder = Rebuilder::new(self.tcx, *fn_decl,
  676                                         generics, same_regions);

  693  struct Rebuilder<'a> {
  694      tcx: &'a ty::ctxt,
  695      fn_decl: ast::P<ast::FnDecl>,

  697      same_regions: &'a [SameRegions],
  698      life_giver: LifeGiver,
  699      cur_anon: Cell<uint>,
  700      inserted_anons: RefCell<HashSet<uint>>,
  701  }

  703  impl<'a> Rebuilder<'a> {
  704      fn new(tcx: &'a ty::ctxt,
  705             fn_decl: ast::P<ast::FnDecl>,
  706             generics: &'a ast::Generics,
  707             same_regions: &'a [SameRegions])
  708             -> Rebuilder<'a> {
  709          Rebuilder {
  710              tcx: tcx,
  711              fn_decl: fn_decl,
  712              generics: generics,
  713              same_regions: same_regions,
  714              life_giver: LifeGiver::with_taken(generics.lifetimes.as_slice()),
  715              cur_anon: Cell::new(0),
  716              inserted_anons: RefCell::new(HashSet::new()),
  717          }
  718      }

  720      fn rebuild(&self) -> (Vec<ast::Arg>, ast::P<ast::Ty>, ast::Generics) {
  721          let mut inputs = self.fn_decl.inputs.clone();
  722          let mut output = self.fn_decl.output;
  723          let mut ty_params = self.generics.ty_params.clone();
  724          for sr in self.same_regions.iter() {
  725              self.cur_anon.set(0);
  726              self.offset_cur_anon();
  727              let (anon_nums, region_names) =
  728                                  self.extract_anon_nums_and_names(sr);
  729              let lifetime = self.life_giver.give_lifetime();
  730              inputs = self.rebuild_args_ty(inputs.as_slice(), lifetime,
  731                                            &anon_nums, &region_names);
  732              output = self.rebuild_arg_ty_or_output(output, lifetime,
  733                                                     &anon_nums, &region_names);
  734              ty_params = self.rebuild_ty_params(ty_params, lifetime,
  735                                                 &region_names);
  736          }
  737          let generated_lifetimes = self.life_giver.get_generated_lifetimes();
  738          let all_region_names = self.extract_all_region_names();
  739          let generics = self.rebuild_generics(self.generics,
  740                                               generated_lifetimes,
  741                                               &all_region_names, ty_params);
  742          (inputs, output, generics)
  743      }

  745      fn extract_anon_nums_and_names(&self, same_regions: &SameRegions)
  746                                     -> (HashSet<uint>, HashSet<ast::Name>) {
  747          let mut anon_nums = HashSet::new();
  748          let mut region_names = HashSet::new();
  749          for br in same_regions.regions.iter() {
  750              match *br {
  751                  ty::BrAnon(i) => {
  752                      anon_nums.insert(i);
  753                  }
  754                  ty::BrNamed(_, name) => {
  755                      region_names.insert(name);
  756                  }
  757                  _ => ()
  758              }
  759          }
  760          (anon_nums, region_names)
  761      }

  819                                 ty_param_bounds: OwnedSlice<ast::TyParamBound>,

  821                                 region_names: &HashSet<ast::Name>)
  822                                 -> OwnedSlice<ast::TyParamBound> {

  826                  &ast::TraitTyParamBound(ref tr) => {

  829                      for (i, lt) in last_seg.lifetimes.iter().enumerate() {

  835                          path: &tr.path,

  838                          anon_nums: &HashSet::new(),
  839                          region_names: region_names
  840                      };
  841                      let new_path = self.rebuild_path(rebuild_info, lifetime);
  842                      ast::TraitTyParamBound(ast::TraitRef {
  843                          path: new_path,
  844                          ref_id: tr.ref_id,
  845                      })
  846                  }
  847              }

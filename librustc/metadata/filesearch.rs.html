 119          let mut rslt = FileDoesntMatch;

 177  /// Returns the value of RUST_PATH, as a list
 178  /// of Paths. Includes default entries for, if they exist:
 179  /// $HOME/.rust

 182  pub fn rust_path() -> ~[Path] {
 183      let mut env_rust_path: ~[Path] = match get_rust_path() {
 184          Some(env_path) => {

 186                  env_path.split_str_iter(PATH_ENTRY_SEPARATOR).collect();
 187              env_path_components.map(|&s| Path(s))
 188          }
 189          None => ~[]
 190      };
 191      let cwd = os::getcwd();
 192      // now add in default entries
 193      let cwd_dot_rust = cwd.push(".rust");
 194      if !env_rust_path.contains(&cwd_dot_rust) {
 195          env_rust_path.push(cwd_dot_rust);
 196      }
 197      if !env_rust_path.contains(&cwd) {
 198          env_rust_path.push(cwd.clone());
 199      }

 201          if !env_rust_path.contains(&p.push(".rust")) {

 203          }

  35      pub fn for_each_lib_search_path(&self, f: |&Path| -> FileMatch) {
  36          let mut visited_dirs = HashSet::new();
  37          let mut found = false;
  38  
  39          debug!("filesearch: searching additional lib search paths [{:?}]",
  40                 self.addl_lib_search_paths.borrow().len());
  41          for path in self.addl_lib_search_paths.borrow().iter() {
  42              match f(path) {
  43                  FileMatches => found = true,
  44                  FileDoesntMatch => ()
  45              }
  46              visited_dirs.insert(path.as_vec().to_owned());
  47          }
  48  
  49          debug!("filesearch: searching target lib path");
  50          let tlib_path = make_target_lib_path(self.sysroot,
  51                                      self.target_triple);

  53              match f(&tlib_path) {
  54                  FileMatches => found = true,
  55                  FileDoesntMatch => ()
  56              }
  57          }
  58          visited_dirs.insert(tlib_path.as_vec().to_owned());
  59          // Try RUST_PATH
  60          if !found {
  61              let rustpath = rust_path();
  62              for path in rustpath.iter() {
  63                  let tlib_path = make_rustpkg_target_lib_path(
  64                      self.sysroot, path, self.target_triple);
  65                  debug!("is {} in visited_dirs? {:?}", tlib_path.display(),
  66                          visited_dirs.contains_equiv(&tlib_path.as_vec().to_owned()));
  67  
  68                  if !visited_dirs.contains_equiv(&tlib_path.as_vec()) {
  69                      visited_dirs.insert(tlib_path.as_vec().to_owned());
  70                      // Don't keep searching the RUST_PATH if one match turns up --
  71                      // if we did, we'd get a "multiple matching crates" error

  87      pub fn search(&self, pick: pick) {
  88          self.for_each_lib_search_path(|lib_search_path| {
  89              debug!("searching {}", lib_search_path.display());
  90              match fs::readdir(lib_search_path) {
  91                  Ok(files) => {
  92                      let mut rslt = FileDoesntMatch;
  93                      let is_rlib = |p: & &Path| {
  94                          p.extension_str() == Some("rlib")
  95                      };
  96                      // Reading metadata out of rlibs is faster, and if we find both
  97                      // an rlib and a dylib we only read one of the files of
  98                      // metadata, so in the name of speed, bring all rlib files to
  99                      // the front of the search list.
 100                      let files1 = files.iter().filter(|p| is_rlib(p));
 101                      let files2 = files.iter().filter(|p| !is_rlib(p));
 102                      for path in files1.chain(files2) {
 103                          debug!("testing {}", path.display());
 104                          let maybe_picked = pick(path);
 105                          match maybe_picked {
 106                              FileMatches => {
 107                                  debug!("picked {}", path.display());
 108                                  rslt = FileMatches;
 109                              }
 110                              FileDoesntMatch => {
 111                                  debug!("rejected {}", path.display());
 112                              }
 113                          }
 114                      }
 115                      rslt
 116                  }
 117                  Err(..) => FileDoesntMatch,
 118              }
 119          });
 120      }

 189  /// Returns the value of RUST_PATH, as a list
 190  /// of Paths. Includes default entries for, if they exist:
 191  /// $HOME/.rust

 193  /// or an ancestor of it
 194  pub fn rust_path() -> Vec<Path> {
 195      let mut env_rust_path: Vec<Path> = match get_rust_path() {
 196          Some(env_path) => {
 197              let env_path_components =
 198                  env_path.split_str(PATH_ENTRY_SEPARATOR);
 199              env_path_components.map(|s| Path::new(s)).collect()
 200          }
 201          None => Vec::new()
 202      };
 203      let mut cwd = os::getcwd();
 204      // now add in default entries
 205      let cwd_dot_rust = cwd.join(".rust");
 206      if !env_rust_path.contains(&cwd_dot_rust) {
 207          env_rust_path.push(cwd_dot_rust);
 208      }
 209      if !env_rust_path.contains(&cwd) {
 210          env_rust_path.push(cwd.clone());
 211      }
 212      loop {
 213          if { let f = cwd.filename(); f.is_none() || f.unwrap() == bytes!("..") } {
 214              break
 215          }
 216          cwd.set_filename(".rust");
 217          if !env_rust_path.contains(&cwd) && cwd.exists() {
 218              env_rust_path.push(cwd.clone());
 219          }
 220          cwd.pop();
 221      }
 222      let h = os::homedir();
 223      for h in h.iter() {
 224          let p = h.join(".rust");
 225          if !env_rust_path.contains(&p) && p.exists() {
 226              env_rust_path.push(p);
 227          }
 228      }
 229      env_rust_path
 230  }

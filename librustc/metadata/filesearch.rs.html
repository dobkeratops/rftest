 156 /// Returns the value of RUST_PATH, as a list
 157 /// of Paths. Includes default entries for, if they exist:
 158 /// $HOME/.rust
 159 /// DIR/.rust for any DIR that's the current working directory
 160 /// or an ancestor of it
 161 pub fn rust_path() -> ~[Path] {
 162     let mut env_rust_path: ~[Path] = match get_rust_path() {
 163         Some(env_path) => {
 164             let env_path_components: ~[&str] =
 165                 env_path.split_str_iter(PATH_ENTRY_SEPARATOR).collect();
 166             env_path_components.map(|&s| Path(s))
 167         }
 168         None => ~[]
 169     };
 170     let cwd = os::getcwd();
 171     // now add in default entries
 172     let cwd_dot_rust = cwd.push(".rust");
 173     if !env_rust_path.contains(&cwd_dot_rust) {
 174         env_rust_path.push(cwd_dot_rust);
 175     }
 176     if !env_rust_path.contains(&cwd) {
 177         env_rust_path.push(cwd.clone());
 178     }
 179     do cwd.each_parent() |p| {
 180         if !env_rust_path.contains(&p.push(".rust")) {
 181             push_if_exists(&mut env_rust_path, p);
 182         }
 183     }
 184     let h = os::homedir();
 185     for h in h.iter() {
 186         if !env_rust_path.contains(&h.push(".rust")) {
 187             push_if_exists(&mut env_rust_path, h);
 188         }
 189     }
 190     env_rust_path
 191 }

 149 pub fn get_rustpkg_root_nearest() -> Result<Path, ~str> {

 154             !os::path_is_dir(&cwd_rustpkg) && cwd_rustpkg != p;

 156         let mut rslt = result::Ok(cwd_rustpkg);
 157

 160                 if os::path_is_dir(&par_rustpkg) {

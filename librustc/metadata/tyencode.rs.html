  57  pub fn enc_ty(w: @io::Writer, cx: @ctxt, t: ty::t) {
  58      match cx.abbrevs {
  59        ac_no_abbrevs => {
  60          let result_str = match cx.tcx.short_names_cache.find(&t) {
  61              Some(&s) => s,
  62              None => {
  63                  let s = do io::with_str_writer |wr| {
  64                      enc_sty(wr, cx, &ty::get(t).sty);
  65                  }.to_managed();
  66                  cx.tcx.short_names_cache.insert(t, s);
  67                  s
  68            }
  69          };
  70          w.write_str(result_str);
  71        }
  72        ac_use_abbrevs(abbrevs) => {
  73            match abbrevs.find(&t) {
  74                Some(a) => { w.write_str(a.s); return; }
  75                None => {}
  76            }
  77            let pos = w.tell();
  78            enc_sty(w, cx, &ty::get(t).sty);
  79            let end = w.tell();
  80            let len = end - pos;
  81            fn estimate_sz(u: uint) -> uint {
  82                let mut n = u;
  83                let mut len = 0u;
  84                while n != 0u { len += 1u; n = n >> 4u; }
  85                return len;
  86            }
  87            let abbrev_len = 3u + estimate_sz(pos) + estimate_sz(len);
  88            if abbrev_len < len {
  89                // I.e. it's actually an abbreviation.
  90                let s = fmt!("#%x:%x#", pos, len).to_managed();
  91                let a = ty_abbrev { pos: pos, len: len, s: s };
  92                abbrevs.insert(t, a);
  93            }
  94            return;
  95        }
  96      }
  97  }

  56  pub fn enc_ty(w: @io::Writer, cx: @ctxt, t: ty::t) {
  57      match cx.abbrevs {
  58        ac_no_abbrevs => {
  59          let result_str = match cx.tcx.short_names_cache.find(&t) {
  60              Some(&s) => s,
  61              None => {
  62                  let s = do io::with_str_writer |wr| {
  63                      enc_sty(wr, cx, &ty::get(t).sty);
  64                  }.to_managed();
  65                  cx.tcx.short_names_cache.insert(t, s);
  66                  s
  67            }
  68          };
  69          w.write_str(result_str);
  70        }
  71        ac_use_abbrevs(abbrevs) => {
  72            match abbrevs.find(&t) {
  73                Some(a) => { w.write_str(a.s); return; }
  74                None => {}
  75            }
  76            let pos = w.tell();
  77            enc_sty(w, cx, &ty::get(t).sty);
  78            let end = w.tell();
  79            let len = end - pos;
  80            fn estimate_sz(u: uint) -> uint {
  81                let mut n = u;
  82                let mut len = 0u;
  83                while n != 0u { len += 1u; n = n >> 4u; }
  84                return len;
  85            }
  86            let abbrev_len = 3u + estimate_sz(pos) + estimate_sz(len);
  87            if abbrev_len < len {
  88                // I.e. it's actually an abbreviation.
  89                let s = fmt!("#%x:%x#", pos, len).to_managed();
  90                let a = ty_abbrev { pos: pos, len: len, s: s };
  91                abbrevs.insert(t, a);
  92            }
  93            return;
  94        }
  95      }
  96  }

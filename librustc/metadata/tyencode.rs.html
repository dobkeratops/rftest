  63  pub fn enc_ty(w: &mut MemWriter, cx: &ctxt, t: ty::t) {
  64      match cx.abbrevs {
  65        ac_no_abbrevs => {
  66            let result_str_opt = cx.tcx.short_names_cache.borrow()
  67                                              .find(&t)
  68                                              .map(|result| {
  69                                                  (*result).clone()
  70                                              });
  71            let result_str = match result_str_opt {
  72              Some(s) => s,
  73              None => {
  74                  let wr = &mut MemWriter::new();
  75                  enc_sty(wr, cx, &ty::get(t).sty);
  76                  let s = str::from_utf8(wr.get_ref()).unwrap();
  77                  cx.tcx.short_names_cache.borrow_mut().insert(t, s.to_str());
  78                  s.to_str()
  79              }
  80            };
  81            w.write(result_str.as_bytes());
  82        }
  83        ac_use_abbrevs(abbrevs) => {
  84            match abbrevs.borrow_mut().find(&t) {
  85                Some(a) => { w.write(a.s.as_bytes()); return; }
  86                None => {}
  87            }
  88            let pos = w.tell().unwrap();
  89            enc_sty(w, cx, &ty::get(t).sty);
  90            let end = w.tell().unwrap();
  91            let len = end - pos;
  92            fn estimate_sz(u: u64) -> u64 {
  93                let mut n = u;
  94                let mut len = 0;
  95                while n != 0 { len += 1; n = n >> 4; }
  96                return len;
  97            }
  98            let abbrev_len = 3 + estimate_sz(pos) + estimate_sz(len);
  99            if abbrev_len < len {
 100                // I.e. it's actually an abbreviation.
 101                let s = format!("\\#{:x}:{:x}\\#", pos, len);
 102                let a = ty_abbrev { pos: pos as uint,
 103                                    len: len as uint,
 104                                    s: s };
 105                abbrevs.borrow_mut().insert(t, a);
 106            }
 107            return;
 108        }
 109      }
 110  }

 157  fn enc_region(w: &mut MemWriter, cx: &ctxt, r: ty::Region) {

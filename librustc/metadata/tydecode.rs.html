  29  // Compact string representation for ty::t values. API ty_str &
  30  // parse_from_str. Extra parameters are for converting to/from def_ids in the
  31  // data buffer. Whatever format you choose should not contain pipe characters.
  32  
  33  // Def id conversion: when we encounter def-ids, they have to be translated.
  34  // For example, the crate number must be converted from the crate number used
  35  // in the library we are reading from into the local crate numbers in use
  36  // here.  To perform this translation, the type decoder is supplied with a
  37  // conversion function of type `conv_did`.
  38  //
  39  // Sometimes, particularly when inlining, the correct translation of the
  40  // def-id will depend on where it originated from.  Therefore, the conversion
  41  // function is given an indicator of the source of the def-id.  See
  42  // astencode.rs for more information.
  43  pub enum DefIdSource {
  44      // Identifies a struct, trait, enum, etc.
  45      NominalType,
  46  
  47      // Identifies a type alias (`type X = ...`).
  48      TypeWithId,
  49  
  50      // Identifies a type parameter (`fn foo<X>() { ... }`).
  51      TypeParameter
  52  }
  53  type conv_did<'self> =
  54      &'self fn(source: DefIdSource, ast::def_id) -> ast::def_id;

  56  pub struct PState<'self> {
  57      data: &'self [u8],

  59      pos: uint,
  60      tcx: ty::ctxt
  61  }

  63  fn peek(st: &PState) -> char {
  64      st.data[st.pos] as char
  65  }

  67  fn next(st: &mut PState) -> char {
  68      let ch = st.data[st.pos] as char;
  69      st.pos = st.pos + 1u;
  70      return ch;
  71  }

  73  fn next_byte(st: &mut PState) -> u8 {
  74      let b = st.data[st.pos];
  75      st.pos = st.pos + 1u;
  76      return b;
  77  }

  79  fn scan<R>(st: &mut PState, is_last: &fn(char) -> bool,
  80             op: &fn(&[u8]) -> R) -> R
  81  {
  82      let start_pos = st.pos;
  83      debug!("scan: '%c' (start)", st.data[st.pos] as char);
  84      while !is_last(st.data[st.pos] as char) {
  85          st.pos += 1;
  86          debug!("scan: '%c'", st.data[st.pos] as char);
  87      }
  88      let end_pos = st.pos;
  89      st.pos += 1;
  90      return op(st.data.slice(start_pos, end_pos));
  91  }

  93  pub fn parse_ident(st: &mut PState, last: char) -> ast::ident {
  94      fn is_last(b: char, c: char) -> bool { return c == b; }
  95      return parse_ident_(st, |a| is_last(last, a) );
  96  }
  97  
  98  fn parse_ident_(st: &mut PState, is_last: @fn(char) -> bool) ->
  99     ast::ident {
 100      let rslt = scan(st, is_last, str::from_bytes);
 101      return st.tcx.sess.ident_of(rslt);
 102  }

 132  fn parse_path(st: &mut PState) -> @ast::Path {
 133      let mut idents: ~[ast::ident] = ~[];
 134      fn is_last(c: char) -> bool { return c == '(' || c == ':'; }
 135      idents.push(parse_ident_(st, is_last));
 136      loop {
 137          match peek(st) {
 138            ':' => { next(st); next(st); }
 139            c => {
 140              if c == '(' {
 141                  return @ast::Path { span: dummy_sp(),
 142                                      global: false,
 143                                      idents: idents,
 144                                      rp: None,
 145                                      types: ~[] };
 146              } else { idents.push(parse_ident_(st, is_last)); }
 147            }
 148          }
 149      };
 150  }

 161  fn parse_vstore(st: &mut PState) -> ty::vstore {
 162      assert_eq!(next(st), '/');
 163  
 164      let c = peek(st);
 165      if '0' <= c && c <= '9' {
 166          let n = parse_uint(st);
 167          assert_eq!(next(st), '|');
 168          return ty::vstore_fixed(n);
 169      }
 170  
 171      match next(st) {
 172        '~' => ty::vstore_uniq,
 173        '@' => ty::vstore_box,
 174        '&' => ty::vstore_slice(parse_region(st)),
 175        c => st.tcx.sess.bug(fmt!("parse_vstore(): bad input '%c'", c))
 176      }
 177  }

 188  fn parse_substs(st: &mut PState, conv: conv_did) -> ty::substs {
 189      let regions = parse_region_substs(st);
 190  
 191      let self_ty = parse_opt(st, |st| parse_ty(st, |x,y| conv(x,y)) );
 192  
 193      assert_eq!(next(st), '[');
 194      let mut params: ~[ty::t] = ~[];
 195      while peek(st) != ']' { params.push(parse_ty(st, |x,y| conv(x,y))); }
 196      st.pos = st.pos + 1u;
 197  
 198      return ty::substs {
 199          regions: regions,
 200          self_ty: self_ty,
 201          tps: params
 202      };
 203  }
 204  
 205  fn parse_region_substs(st: &mut PState) -> ty::RegionSubsts {
 206      match next(st) {
 207          'e' => ty::ErasedRegions,
 208          'n' => {
 209              let mut regions = opt_vec::Empty;
 210              while peek(st) != '.' {
 211                  let r = parse_region(st);
 212                  regions.push(r);
 213              }
 214              assert_eq!(next(st), '.');
 215              ty::NonerasedRegions(regions)
 216          }
 217          _ => fail!("parse_bound_region: bad input")
 218      }
 219  }
 220  
 221  fn parse_bound_region(st: &mut PState) -> ty::bound_region {
 222      match next(st) {
 223        's' => ty::br_self,
 224        'a' => {
 225          let id = parse_uint(st);
 226          assert_eq!(next(st), '|');
 227          ty::br_anon(id)
 228        }
 229        '[' => ty::br_named(st.tcx.sess.ident_of(parse_str(st, ']'))),
 230        'c' => {
 231          let id = parse_uint(st) as int;
 232          assert_eq!(next(st), '|');
 233          ty::br_cap_avoid(id, @parse_bound_region(st))
 234        },
 235        _ => fail!("parse_bound_region: bad input")
 236      }
 237  }
 238  
 239  fn parse_region(st: &mut PState) -> ty::Region {
 240      match next(st) {
 241        'b' => {
 242          ty::re_bound(parse_bound_region(st))
 243        }
 244        'f' => {
 245          assert_eq!(next(st), '[');
 246          let id = parse_uint(st) as int;
 247          assert_eq!(next(st), '|');
 248          let br = parse_bound_region(st);
 249          assert_eq!(next(st), ']');
 250          ty::re_free(ty::FreeRegion {scope_id: id,
 251                                      bound_region: br})
 252        }
 253        's' => {
 254          let id = parse_uint(st) as int;
 255          assert_eq!(next(st), '|');
 256          ty::re_scope(id)
 257        }
 258        't' => {
 259          ty::re_static
 260        }
 261        'e' => {
 262          ty::re_static
 263        }
 264        _ => fail!("parse_region: bad input")
 265      }
 266  }

 268  fn parse_opt<T>(st: &mut PState, f: &fn(&mut PState) -> T) -> Option<T> {
 269      match next(st) {
 270        'n' => None,
 271        's' => Some(f(st)),
 272        _ => fail!("parse_opt: bad input")
 273      }
 274  }

 276  fn parse_str(st: &mut PState, term: char) -> ~str {
 277      let mut result = ~"";
 278      while peek(st) != term {
 279          unsafe {
 280              str::raw::push_byte(&mut result, next_byte(st));
 281          }
 282      }
 283      next(st);
 284      return result;
 285  }

 287  fn parse_trait_ref(st: &mut PState, conv: conv_did) -> ty::TraitRef {
 288      let def = parse_def(st, NominalType, |x,y| conv(x,y));
 289      let substs = parse_substs(st, |x,y| conv(x,y));
 290      ty::TraitRef {def_id: def, substs: substs}
 291  }

 293  fn parse_ty(st: &mut PState, conv: conv_did) -> ty::t {
 294      match next(st) {
 295        'n' => return ty::mk_nil(),
 296        'z' => return ty::mk_bot(),
 297        'b' => return ty::mk_bool(),
 298        'i' => return ty::mk_int(),
 299        'u' => return ty::mk_uint(),
 300        'l' => return ty::mk_float(),
 301        'M' => {
 302          match next(st) {
 303            'b' => return ty::mk_mach_uint(ast::ty_u8),
 304            'w' => return ty::mk_mach_uint(ast::ty_u16),
 305            'l' => return ty::mk_mach_uint(ast::ty_u32),
 306            'd' => return ty::mk_mach_uint(ast::ty_u64),
 307            'B' => return ty::mk_mach_int(ast::ty_i8),
 308            'W' => return ty::mk_mach_int(ast::ty_i16),
 309            'L' => return ty::mk_mach_int(ast::ty_i32),
 310            'D' => return ty::mk_mach_int(ast::ty_i64),
 311            'f' => return ty::mk_mach_float(ast::ty_f32),
 312            'F' => return ty::mk_mach_float(ast::ty_f64),
 313            _ => fail!("parse_ty: bad numeric type")
 314          }
 315        }
 316        'c' => return ty::mk_char(),
 317        't' => {
 318          assert_eq!(next(st), '[');
 319          let def = parse_def(st, NominalType, |x,y| conv(x,y));
 320          let substs = parse_substs(st, |x,y| conv(x,y));
 321          assert_eq!(next(st), ']');
 322          return ty::mk_enum(st.tcx, def, substs);
 323        }
 324        'x' => {
 325          assert_eq!(next(st), '[');
 326          let def = parse_def(st, NominalType, |x,y| conv(x,y));
 327          let substs = parse_substs(st, |x,y| conv(x,y));
 328          let store = parse_trait_store(st);
 329          let mt = parse_mutability(st);
 330          let bounds = parse_bounds(st, |x,y| conv(x,y));
 331          assert_eq!(next(st), ']');
 332          return ty::mk_trait(st.tcx, def, substs, store, mt, bounds.builtin_bounds);
 333        }
 334        'p' => {
 335          let did = parse_def(st, TypeParameter, conv);
 336          debug!("parsed ty_param: did=%?", did);
 337          return ty::mk_param(st.tcx, parse_uint(st), did);
 338        }
 339        's' => {
 340          let did = parse_def(st, TypeParameter, conv);
 341          return ty::mk_self(st.tcx, did);
 342        }
 343        '@' => return ty::mk_box(st.tcx, parse_mt(st, conv)),
 344        '~' => return ty::mk_uniq(st.tcx, parse_mt(st, conv)),
 345        '*' => return ty::mk_ptr(st.tcx, parse_mt(st, conv)),
 346        '&' => {
 347          let r = parse_region(st);
 348          let mt = parse_mt(st, conv);
 349          return ty::mk_rptr(st.tcx, r, mt);
 350        }
 351        'U' => return ty::mk_unboxed_vec(st.tcx, parse_mt(st, conv)),
 352        'V' => {
 353          let mt = parse_mt(st, conv);
 354          let v = parse_vstore(st);
 355          return ty::mk_evec(st.tcx, mt, v);
 356        }
 357        'v' => {
 358          let v = parse_vstore(st);
 359          return ty::mk_estr(st.tcx, v);
 360        }
 361        'T' => {
 362          assert_eq!(next(st), '[');
 363          let mut params = ~[];
 364          while peek(st) != ']' { params.push(parse_ty(st, |x,y| conv(x,y))); }
 365          st.pos = st.pos + 1u;
 366          return ty::mk_tup(st.tcx, params);
 367        }
 368        'f' => {
 369          return ty::mk_closure(st.tcx, parse_closure_ty(st, conv));
 370        }
 371        'F' => {
 372          return ty::mk_bare_fn(st.tcx, parse_bare_fn_ty(st, conv));
 373        }
 374        'Y' => return ty::mk_type(st.tcx),
 375        'C' => {
 376          let sigil = parse_sigil(st);
 377          return ty::mk_opaque_closure_ptr(st.tcx, sigil);
 378        }
 379        '#' => {
 380          let pos = parse_hex(st);
 381          assert_eq!(next(st), ':');
 382          let len = parse_hex(st);
 383          assert_eq!(next(st), '#');
 384          let key = ty::creader_cache_key {cnum: st.crate,
 385                                           pos: pos,
 386                                           len: len };
 387          match st.tcx.rcache.find(&key) {
 388            Some(&tt) => return tt,
 389            None => {
 390              let mut ps = PState {
 391                  pos: pos,
 392                  .. *st
 393              };
 394              let tt = parse_ty(&mut ps, conv);
 395              st.tcx.rcache.insert(key, tt);
 396              return tt;
 397            }
 398          }
 399        }
 400        '"' => {
 401          let _ = parse_def(st, TypeWithId, |x,y| conv(x,y));
 402          let inner = parse_ty(st, |x,y| conv(x,y));
 403          inner
 404        }
 405        'B' => ty::mk_opaque_box(st.tcx),
 406        'a' => {
 407            assert_eq!(next(st), '[');
 408            let did = parse_def(st, NominalType, |x,y| conv(x,y));
 409            let substs = parse_substs(st, |x,y| conv(x,y));
 410            assert_eq!(next(st), ']');
 411            return ty::mk_struct(st.tcx, did, substs);
 412        }
 413        c => { error!("unexpected char in type string: %c", c); fail!();}
 414      }
 415  }

 417  fn parse_mutability(st: &mut PState) -> ast::mutability {
 418      match peek(st) {
 419        'm' => { next(st); ast::m_mutbl }
 420        '?' => { next(st); ast::m_const }
 421        _ => { ast::m_imm }
 422      }
 423  }

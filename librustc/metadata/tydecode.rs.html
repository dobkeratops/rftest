  30  // Compact string representation for ty::t values. API ty_str &
  31  // parse_from_str. Extra parameters are for converting to/from def_ids in the
  32  // data buffer. Whatever format you choose should not contain pipe characters.

  34  // Def id conversion: when we encounter def-ids, they have to be translated.
  35  // For example, the crate number must be converted from the crate number used
  36  // in the library we are reading from into the local crate numbers in use
  37  // here.  To perform this translation, the type decoder is supplied with a
  38  // conversion function of type `conv_did`.
  39  //
  40  // Sometimes, particularly when inlining, the correct translation of the
  41  // def-id will depend on where it originated from.  Therefore, the conversion
  42  // function is given an indicator of the source of the def-id.  See
  43  // astencode.rs for more information.
  44  pub enum DefIdSource {
  45      // Identifies a struct, trait, enum, etc.
  46      NominalType,
  47  
  48      // Identifies a type alias (`type X = ...`).
  49      TypeWithId,
  50  
  51      // Identifies a type parameter (`fn foo<X>() { ... }`).
  52      TypeParameter,
  53  
  54      // Identifies a region parameter (`fn foo<'X>() { ... }`).
  55      RegionParameter,
  56  }
  57  pub type conv_did<'a> =
  58      |source: DefIdSource, ast::DefId|: 'a -> ast::DefId;

  67  fn peek(st: &PState) -> char {
  68      st.data[st.pos] as char
  69  }

  71  fn next(st: &mut PState) -> char {
  72      let ch = st.data[st.pos] as char;
  73      st.pos = st.pos + 1u;
  74      return ch;
  75  }

  77  fn next_byte(st: &mut PState) -> u8 {
  78      let b = st.data[st.pos];
  79      st.pos = st.pos + 1u;
  80      return b;
  81  }

  83  fn scan<R>(st: &mut PState, is_last: |char| -> bool, op: |&[u8]| -> R) -> R {
  84      let start_pos = st.pos;
  85      debug!("scan: '{}' (start)", st.data[st.pos] as char);
  86      while !is_last(st.data[st.pos] as char) {
  87          st.pos += 1;
  88          debug!("scan: '{}'", st.data[st.pos] as char);
  89      }
  90      let end_pos = st.pos;
  91      st.pos += 1;
  92      return op(st.data.slice(start_pos, end_pos));
  93  }

 140  fn parse_vstore<M>(st: &mut PState, conv: conv_did,
 141                     parse_mut: |&mut PState| -> M) -> ty::Vstore<M> {
 142      assert_eq!(next(st), '/');
 143  
 144      let c = peek(st);
 145      if '0' <= c && c <= '9' {
 146          let n = parse_uint(st);
 147          assert_eq!(next(st), '|');
 148          return ty::VstoreFixed(n);
 149      }
 150  
 151      match next(st) {
 152          '~' => ty::VstoreUniq,
 153          '&' => ty::VstoreSlice(parse_region(st, conv), parse_mut(st)),
 154          c => st.tcx.sess.bug(format!("parse_vstore(): bad input '{}'", c))
 155      }
 156  }

 159      match next(st) {

 162          c => st.tcx.sess.bug(format!("parse_trait_store(): bad input '{}'", c))
 163      }
 164  }

 166  fn parse_substs(st: &mut PState, conv: conv_did) -> ty::substs {
 167      let regions = parse_region_substs(st, |x,y| conv(x,y));
 168  
 169      let self_ty = parse_opt(st, |st| parse_ty(st, |x,y| conv(x,y)) );
 170  
 171      assert_eq!(next(st), '[');
 172      let mut params: Vec<ty::t> = Vec::new();
 173      while peek(st) != ']' { params.push(parse_ty(st, |x,y| conv(x,y))); }
 174      st.pos = st.pos + 1u;
 175  
 176      return ty::substs {
 177          regions: regions,
 178          self_ty: self_ty,
 179          tps: params
 180      };
 181  }

 183  fn parse_region_substs(st: &mut PState, conv: conv_did) -> ty::RegionSubsts {
 184      match next(st) {
 185          'e' => ty::ErasedRegions,
 186          'n' => {
 187              let mut regions = vec!();
 188              while peek(st) != '.' {
 189                  let r = parse_region(st, |x,y| conv(x,y));
 190                  regions.push(r);
 191              }
 192              assert_eq!(next(st), '.');
 193              ty::NonerasedRegions(OwnedSlice::from_vec(regions))
 194          }
 195          _ => fail!("parse_bound_region: bad input")
 196      }
 197  }

 199  fn parse_bound_region(st: &mut PState, conv: conv_did) -> ty::BoundRegion {
 200      match next(st) {
 201          'a' => {
 202              let id = parse_uint(st);
 203              assert_eq!(next(st), '|');
 204              ty::BrAnon(id)
 205          }
 206          '[' => {
 207              let def = parse_def(st, RegionParameter, |x,y| conv(x,y));
 208              let ident = token::str_to_ident(parse_str(st, ']'));
 209              ty::BrNamed(def, ident.name)
 210          }
 211          'f' => {
 212              let id = parse_uint(st);
 213              assert_eq!(next(st), '|');
 214              ty::BrFresh(id)
 215          }
 216          _ => fail!("parse_bound_region: bad input")
 217      }
 218  }

 220  fn parse_region(st: &mut PState, conv: conv_did) -> ty::Region {
 221      match next(st) {
 222        'b' => {
 223          assert_eq!(next(st), '[');
 224          let id = parse_uint(st) as ast::NodeId;
 225          assert_eq!(next(st), '|');
 226          let br = parse_bound_region(st, |x,y| conv(x,y));
 227          assert_eq!(next(st), ']');
 228          ty::ReLateBound(id, br)
 229        }
 230        'B' => {
 231          assert_eq!(next(st), '[');
 232          let node_id = parse_uint(st) as ast::NodeId;
 233          assert_eq!(next(st), '|');
 234          let index = parse_uint(st);
 235          assert_eq!(next(st), '|');
 236          let nm = token::str_to_ident(parse_str(st, ']'));
 237          ty::ReEarlyBound(node_id, index, nm.name)
 238        }
 239        'f' => {
 240          assert_eq!(next(st), '[');
 241          let id = parse_uint(st) as ast::NodeId;
 242          assert_eq!(next(st), '|');
 243          let br = parse_bound_region(st, |x,y| conv(x,y));
 244          assert_eq!(next(st), ']');
 245          ty::ReFree(ty::FreeRegion {scope_id: id,
 246                                      bound_region: br})
 247        }
 248        's' => {
 249          let id = parse_uint(st) as ast::NodeId;
 250          assert_eq!(next(st), '|');
 251          ty::ReScope(id)
 252        }
 253        't' => {
 254          ty::ReStatic
 255        }
 256        'e' => {
 257          ty::ReStatic
 258        }
 259        _ => fail!("parse_region: bad input")
 260      }
 261  }

 268      }
 269  }

 271  fn parse_str(st: &mut PState, term: char) -> ~str {
 272      let mut result = StrBuf::new();
 273      while peek(st) != term {
 274          unsafe {
 275              result.push_bytes([next_byte(st)])
 276          }
 277      }
 278      next(st);
 279      return result.into_owned();
 280  }

 282  fn parse_trait_ref(st: &mut PState, conv: conv_did) -> ty::TraitRef {
 283      let def = parse_def(st, NominalType, |x,y| conv(x,y));
 284      let substs = parse_substs(st, |x,y| conv(x,y));
 285      ty::TraitRef {def_id: def, substs: substs}
 286  }

 288  fn parse_ty(st: &mut PState, conv: conv_did) -> ty::t {
 289      match next(st) {
 290        'n' => return ty::mk_nil(),
 291        'z' => return ty::mk_bot(),
 292        'b' => return ty::mk_bool(),
 293        'i' => return ty::mk_int(),
 294        'u' => return ty::mk_uint(),
 295        'M' => {
 296          match next(st) {
 297            'b' => return ty::mk_mach_uint(ast::TyU8),
 298            'w' => return ty::mk_mach_uint(ast::TyU16),
 299            'l' => return ty::mk_mach_uint(ast::TyU32),
 300            'd' => return ty::mk_mach_uint(ast::TyU64),
 301            'B' => return ty::mk_mach_int(ast::TyI8),
 302            'W' => return ty::mk_mach_int(ast::TyI16),
 303            'L' => return ty::mk_mach_int(ast::TyI32),
 304            'D' => return ty::mk_mach_int(ast::TyI64),
 305            'f' => return ty::mk_mach_float(ast::TyF32),
 306            'F' => return ty::mk_mach_float(ast::TyF64),
 307            _ => fail!("parse_ty: bad numeric type")
 308          }
 309        }
 310        'c' => return ty::mk_char(),
 311        't' => {
 312          assert_eq!(next(st), '[');
 313          let def = parse_def(st, NominalType, |x,y| conv(x,y));
 314          let substs = parse_substs(st, |x,y| conv(x,y));
 315          assert_eq!(next(st), ']');
 316          return ty::mk_enum(st.tcx, def, substs);
 317        }
 318        'x' => {
 319          assert_eq!(next(st), '[');
 320          let def = parse_def(st, NominalType, |x,y| conv(x,y));
 321          let substs = parse_substs(st, |x,y| conv(x,y));
 322          let store = parse_trait_store(st, |x,y| conv(x,y));
 323          let bounds = parse_bounds(st, |x,y| conv(x,y));
 324          assert_eq!(next(st), ']');
 325          return ty::mk_trait(st.tcx, def, substs, store, bounds.builtin_bounds);
 326        }
 327        'p' => {
 328          let did = parse_def(st, TypeParameter, |x,y| conv(x,y));
 329          debug!("parsed ty_param: did={:?}", did);
 330          return ty::mk_param(st.tcx, parse_uint(st), did);
 331        }
 332        's' => {
 333          let did = parse_def(st, TypeParameter, |x,y| conv(x,y));
 334          return ty::mk_self(st.tcx, did);
 335        }
 336        '@' => return ty::mk_box(st.tcx, parse_ty(st, |x,y| conv(x,y))),
 337        '~' => return ty::mk_uniq(st.tcx, parse_ty(st, |x,y| conv(x,y))),
 338        '*' => return ty::mk_ptr(st.tcx, parse_mt(st, |x,y| conv(x,y))),
 339        '&' => {
 340          let r = parse_region(st, |x,y| conv(x,y));
 341          let mt = parse_mt(st, |x,y| conv(x,y));
 342          return ty::mk_rptr(st.tcx, r, mt);
 343        }
 344        'V' => {
 345          let ty = parse_ty(st, |x,y| conv(x,y));
 346          let v = parse_vstore(st, |x,y| conv(x,y), parse_mutability);
 347          return ty::mk_vec(st.tcx, ty, v);
 348        }
 349        'v' => {
 350          let v = parse_vstore(st, |x,y| conv(x,y), |_| ());
 351          return ty::mk_str(st.tcx, v);
 352        }
 353        'T' => {
 354          assert_eq!(next(st), '[');
 355          let mut params = Vec::new();
 356          while peek(st) != ']' { params.push(parse_ty(st, |x,y| conv(x,y))); }
 357          st.pos = st.pos + 1u;
 358          return ty::mk_tup(st.tcx, params);
 359        }
 360        'f' => {
 361          return ty::mk_closure(st.tcx, parse_closure_ty(st, |x,y| conv(x,y)));
 362        }
 363        'F' => {
 364          return ty::mk_bare_fn(st.tcx, parse_bare_fn_ty(st, |x,y| conv(x,y)));
 365        }
 366        '#' => {
 367          let pos = parse_hex(st);
 368          assert_eq!(next(st), ':');
 369          let len = parse_hex(st);
 370          assert_eq!(next(st), '#');
 371          let key = ty::creader_cache_key {cnum: st.krate,
 372                                           pos: pos,
 373                                           len: len };
 374  
 375          match st.tcx.rcache.borrow().find_copy(&key) {
 376            Some(tt) => return tt,
 377            None => {}
 378          }
 379          let mut ps = PState {
 380              pos: pos,
 381              .. *st
 382          };
 383          let tt = parse_ty(&mut ps, |x,y| conv(x,y));
 384          st.tcx.rcache.borrow_mut().insert(key, tt);
 385          return tt;
 386        }
 387        '"' => {
 388          let _ = parse_def(st, TypeWithId, |x,y| conv(x,y));
 389          let inner = parse_ty(st, |x,y| conv(x,y));
 390          inner
 391        }
 392        'a' => {
 393            assert_eq!(next(st), '[');
 394            let did = parse_def(st, NominalType, |x,y| conv(x,y));
 395            let substs = parse_substs(st, |x,y| conv(x,y));
 396            assert_eq!(next(st), ']');
 397            return ty::mk_struct(st.tcx, did, substs);
 398        }
 399        c => { error!("unexpected char in type string: {}", c); fail!();}
 400      }

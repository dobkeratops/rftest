  29  // Compact string representation for ty::t values. API ty_str &
  30  // parse_from_str. Extra parameters are for converting to/from def_ids in the
  31  // data buffer. Whatever format you choose should not contain pipe characters.
  32  
  33  // Def id conversion: when we encounter def-ids, they have to be translated.
  34  // For example, the crate number must be converted from the crate number used
  35  // in the library we are reading from into the local crate numbers in use
  36  // here.  To perform this translation, the type decoder is supplied with a
  37  // conversion function of type `conv_did`.
  38  //
  39  // Sometimes, particularly when inlining, the correct translation of the
  40  // def-id will depend on where it originated from.  Therefore, the conversion
  41  // function is given an indicator of the source of the def-id.  See
  42  // astencode.rs for more information.
  43  pub enum DefIdSource {
  44      // Identifies a struct, trait, enum, etc.
  45      NominalType,
  46  
  47      // Identifies a type alias (`type X = ...`).
  48      TypeWithId,
  49  
  50      // Identifies a type parameter (`fn foo<X>() { ... }`).
  51      TypeParameter,
  52  
  53      // Identifies a region parameter (`fn foo<'X>() { ... }`).
  54      RegionParameter,
  55  }
  56  pub type conv_did<'a> =
  57      |source: DefIdSource, ast::DefId|: 'a -> ast::DefId;

  59  pub struct PState<'a> {
  60      data: &'a [u8],
  61      krate: ast::CrateNum,
  62      pos: uint,
  63      tcx: &'a ty::ctxt
  64  }
  65  
  66  fn peek(st: &PState) -> char {
  67      st.data[st.pos] as char
  68  }
  69  
  70  fn next(st: &mut PState) -> char {
  71      let ch = st.data[st.pos] as char;
  72      st.pos = st.pos + 1u;
  73      return ch;
  74  }
  75  
  76  fn next_byte(st: &mut PState) -> u8 {
  77      let b = st.data[st.pos];
  78      st.pos = st.pos + 1u;
  79      return b;
  80  }

  82  fn scan<R>(st: &mut PState, is_last: |char| -> bool, op: |&[u8]| -> R) -> R {
  83      let start_pos = st.pos;
  84      debug!("scan: '{}' (start)", st.data[st.pos] as char);
  85      while !is_last(st.data[st.pos] as char) {
  86          st.pos += 1;
  87          debug!("scan: '{}'", st.data[st.pos] as char);
  88      }
  89      let end_pos = st.pos;
  90      st.pos += 1;
  91      return op(st.data.slice(start_pos, end_pos));
  92  }

  94  pub fn parse_ident(st: &mut PState, last: char) -> ast::Ident {
  95      fn is_last(b: char, c: char) -> bool { return c == b; }
  96      return parse_ident_(st, |a| is_last(last, a) );
  97  }

  99  fn parse_ident_(st: &mut PState, is_last: |char| -> bool) -> ast::Ident {
 100      scan(st, is_last, |bytes| {
 101          token::str_to_ident(str::from_utf8(bytes).unwrap())
 102      })
 103  }

 139  fn parse_sigil(st: &mut PState) -> ast::Sigil {
 140      match next(st) {
 141          '@' => ast::ManagedSigil,
 142          '~' => ast::OwnedSigil,
 143          '&' => ast::BorrowedSigil,
 144          c => st.tcx.sess.bug(format!("parse_sigil(): bad input '{}'", c))
 145      }
 146  }

 148  fn parse_vstore(st: &mut PState, conv: conv_did) -> ty::vstore {
 149      assert_eq!(next(st), '/');
 150  
 151      let c = peek(st);
 152      if '0' <= c && c <= '9' {
 153          let n = parse_uint(st);
 154          assert_eq!(next(st), '|');
 155          return ty::vstore_fixed(n);
 156      }
 157  
 158      match next(st) {
 159        '~' => ty::vstore_uniq,
 160        '&' => ty::vstore_slice(parse_region(st, conv)),
 161        c => st.tcx.sess.bug(format!("parse_vstore(): bad input '{}'", c))
 162      }
 163  }

 165  fn parse_trait_store(st: &mut PState, conv: conv_did) -> ty::TraitStore {
 166      match next(st) {
 167          '~' => ty::UniqTraitStore,
 168          '&' => ty::RegionTraitStore(parse_region(st, conv)),
 169          c => st.tcx.sess.bug(format!("parse_trait_store(): bad input '{}'", c))
 170      }
 171  }

 173  fn parse_substs(st: &mut PState, conv: conv_did) -> ty::substs {
 174      let regions = parse_region_substs(st, |x,y| conv(x,y));
 175  
 176      let self_ty = parse_opt(st, |st| parse_ty(st, |x,y| conv(x,y)) );
 177  
 178      assert_eq!(next(st), '[');
 179      let mut params: Vec<ty::t> = Vec::new();
 180      while peek(st) != ']' { params.push(parse_ty(st, |x,y| conv(x,y))); }
 181      st.pos = st.pos + 1u;
 182  
 183      return ty::substs {
 184          regions: regions,
 185          self_ty: self_ty,
 186          tps: params
 187      };
 188  }
 189  
 190  fn parse_region_substs(st: &mut PState, conv: conv_did) -> ty::RegionSubsts {
 191      match next(st) {
 192          'e' => ty::ErasedRegions,
 193          'n' => {
 194              let mut regions = vec!();
 195              while peek(st) != '.' {
 196                  let r = parse_region(st, |x,y| conv(x,y));
 197                  regions.push(r);
 198              }
 199              assert_eq!(next(st), '.');
 200              ty::NonerasedRegions(OwnedSlice::from_vec(regions))
 201          }
 202          _ => fail!("parse_bound_region: bad input")
 203      }
 204  }
 205  
 206  fn parse_bound_region(st: &mut PState, conv: conv_did) -> ty::BoundRegion {
 207      match next(st) {
 208          'a' => {
 209              let id = parse_uint(st);
 210              assert_eq!(next(st), '|');
 211              ty::BrAnon(id)
 212          }
 213          '[' => {
 214              let def = parse_def(st, RegionParameter, |x,y| conv(x,y));
 215              let ident = token::str_to_ident(parse_str(st, ']'));
 216              ty::BrNamed(def, ident.name)
 217          }
 218          'f' => {
 219              let id = parse_uint(st);
 220              assert_eq!(next(st), '|');
 221              ty::BrFresh(id)
 222          }
 223          _ => fail!("parse_bound_region: bad input")
 224      }
 225  }
 226  
 227  fn parse_region(st: &mut PState, conv: conv_did) -> ty::Region {
 228      match next(st) {
 229        'b' => {
 230          assert_eq!(next(st), '[');
 231          let id = parse_uint(st) as ast::NodeId;
 232          assert_eq!(next(st), '|');
 233          let br = parse_bound_region(st, |x,y| conv(x,y));
 234          assert_eq!(next(st), ']');
 235          ty::ReLateBound(id, br)
 236        }
 237        'B' => {
 238          assert_eq!(next(st), '[');
 239          let node_id = parse_uint(st) as ast::NodeId;
 240          assert_eq!(next(st), '|');
 241          let index = parse_uint(st);
 242          assert_eq!(next(st), '|');
 243          let nm = token::str_to_ident(parse_str(st, ']'));
 244          ty::ReEarlyBound(node_id, index, nm.name)
 245        }
 246        'f' => {
 247          assert_eq!(next(st), '[');
 248          let id = parse_uint(st) as ast::NodeId;
 249          assert_eq!(next(st), '|');
 250          let br = parse_bound_region(st, |x,y| conv(x,y));
 251          assert_eq!(next(st), ']');
 252          ty::ReFree(ty::FreeRegion {scope_id: id,
 253                                      bound_region: br})
 254        }
 255        's' => {
 256          let id = parse_uint(st) as ast::NodeId;
 257          assert_eq!(next(st), '|');
 258          ty::ReScope(id)
 259        }
 260        't' => {
 261          ty::ReStatic
 262        }
 263        'e' => {
 264          ty::ReStatic
 265        }
 266        _ => fail!("parse_region: bad input")
 267      }
 268  }

 270  fn parse_opt<T>(st: &mut PState, f: |&mut PState| -> T) -> Option<T> {
 271      match next(st) {
 272        'n' => None,
 273        's' => Some(f(st)),
 274        _ => fail!("parse_opt: bad input")
 275      }
 276  }

 278  fn parse_str(st: &mut PState, term: char) -> ~str {
 279      let mut result = ~"";
 280      while peek(st) != term {
 281          unsafe {
 282              str::raw::push_byte(&mut result, next_byte(st));
 283          }
 284      }
 285      next(st);
 286      return result;
 287  }

 289  fn parse_trait_ref(st: &mut PState, conv: conv_did) -> ty::TraitRef {
 290      let def = parse_def(st, NominalType, |x,y| conv(x,y));
 291      let substs = parse_substs(st, |x,y| conv(x,y));
 292      ty::TraitRef {def_id: def, substs: substs}
 293  }

 295  fn parse_ty(st: &mut PState, conv: conv_did) -> ty::t {
 296      match next(st) {
 297        'n' => return ty::mk_nil(),
 298        'z' => return ty::mk_bot(),
 299        'b' => return ty::mk_bool(),
 300        'i' => return ty::mk_int(),
 301        'u' => return ty::mk_uint(),
 302        'M' => {
 303          match next(st) {
 304            'b' => return ty::mk_mach_uint(ast::TyU8),
 305            'w' => return ty::mk_mach_uint(ast::TyU16),
 306            'l' => return ty::mk_mach_uint(ast::TyU32),
 307            'd' => return ty::mk_mach_uint(ast::TyU64),
 308            'B' => return ty::mk_mach_int(ast::TyI8),
 309            'W' => return ty::mk_mach_int(ast::TyI16),
 310            'L' => return ty::mk_mach_int(ast::TyI32),
 311            'D' => return ty::mk_mach_int(ast::TyI64),
 312            'f' => return ty::mk_mach_float(ast::TyF32),
 313            'F' => return ty::mk_mach_float(ast::TyF64),
 314            _ => fail!("parse_ty: bad numeric type")
 315          }
 316        }
 317        'c' => return ty::mk_char(),
 318        't' => {
 319          assert_eq!(next(st), '[');
 320          let def = parse_def(st, NominalType, |x,y| conv(x,y));
 321          let substs = parse_substs(st, |x,y| conv(x,y));
 322          assert_eq!(next(st), ']');
 323          return ty::mk_enum(st.tcx, def, substs);
 324        }
 325        'x' => {
 326          assert_eq!(next(st), '[');
 327          let def = parse_def(st, NominalType, |x,y| conv(x,y));
 328          let substs = parse_substs(st, |x,y| conv(x,y));
 329          let store = parse_trait_store(st, |x,y| conv(x,y));
 330          let mt = parse_mutability(st);
 331          let bounds = parse_bounds(st, |x,y| conv(x,y));
 332          assert_eq!(next(st), ']');
 333          return ty::mk_trait(st.tcx, def, substs, store, mt, bounds.builtin_bounds);
 334        }
 335        'p' => {
 336          let did = parse_def(st, TypeParameter, |x,y| conv(x,y));
 337          debug!("parsed ty_param: did={:?}", did);
 338          return ty::mk_param(st.tcx, parse_uint(st), did);
 339        }
 340        's' => {
 341          let did = parse_def(st, TypeParameter, |x,y| conv(x,y));
 342          return ty::mk_self(st.tcx, did);
 343        }
 344        '@' => return ty::mk_box(st.tcx, parse_ty(st, |x,y| conv(x,y))),
 345        '~' => return ty::mk_uniq(st.tcx, parse_ty(st, |x,y| conv(x,y))),
 346        '*' => return ty::mk_ptr(st.tcx, parse_mt(st, |x,y| conv(x,y))),
 347        '&' => {
 348          let r = parse_region(st, |x,y| conv(x,y));
 349          let mt = parse_mt(st, |x,y| conv(x,y));
 350          return ty::mk_rptr(st.tcx, r, mt);
 351        }
 352        'V' => {
 353          let mt = parse_mt(st, |x,y| conv(x,y));
 354          let v = parse_vstore(st, |x,y| conv(x,y));
 355          return ty::mk_vec(st.tcx, mt, v);
 356        }
 357        'v' => {
 358          let v = parse_vstore(st, |x,y| conv(x,y));
 359          return ty::mk_str(st.tcx, v);
 360        }
 361        'T' => {
 362          assert_eq!(next(st), '[');
 363          let mut params = Vec::new();
 364          while peek(st) != ']' { params.push(parse_ty(st, |x,y| conv(x,y))); }
 365          st.pos = st.pos + 1u;
 366          return ty::mk_tup(st.tcx, params);
 367        }
 368        'f' => {
 369          return ty::mk_closure(st.tcx, parse_closure_ty(st, |x,y| conv(x,y)));
 370        }
 371        'F' => {
 372          return ty::mk_bare_fn(st.tcx, parse_bare_fn_ty(st, |x,y| conv(x,y)));
 373        }
 374        '#' => {
 375          let pos = parse_hex(st);
 376          assert_eq!(next(st), ':');
 377          let len = parse_hex(st);
 378          assert_eq!(next(st), '#');
 379          let key = ty::creader_cache_key {cnum: st.krate,
 380                                           pos: pos,
 381                                           len: len };
 382  
 383          match st.tcx.rcache.borrow().find_copy(&key) {
 384            Some(tt) => return tt,
 385            None => {}
 386          }
 387          let mut ps = PState {
 388              pos: pos,
 389              .. *st
 390          };
 391          let tt = parse_ty(&mut ps, |x,y| conv(x,y));
 392          st.tcx.rcache.borrow_mut().insert(key, tt);
 393          return tt;
 394        }
 395        '"' => {
 396          let _ = parse_def(st, TypeWithId, |x,y| conv(x,y));
 397          let inner = parse_ty(st, |x,y| conv(x,y));
 398          inner
 399        }
 400        'a' => {
 401            assert_eq!(next(st), '[');
 402            let did = parse_def(st, NominalType, |x,y| conv(x,y));
 403            let substs = parse_substs(st, |x,y| conv(x,y));
 404            assert_eq!(next(st), ']');
 405            return ty::mk_struct(st.tcx, did, substs);
 406        }

  86  fn find_library_crate_aux(
  87      cx: &Context,
  88      (prefix, suffix): (~str, ~str),
  89      filesearch: @filesearch::FileSearch
  90  ) -> Option<(~str, @~[u8])> {
  91      let crate_name = crate_name_from_metas(cx.metas);
  92      // want: crate_name.dir_part() + prefix + crate_name.file_part + "-"
  93      let prefix = fmt!("%s%s-", prefix, crate_name);
  94      let mut matches = ~[];
  95      filesearch::search(filesearch, |path| -> Option<()> {
  96        let path_str = path.filename();
  97        match path_str {
  98            None => None,
  99            Some(path_str) =>
 100                if path_str.starts_with(prefix) && path_str.ends_with(suffix) {
 101                    debug!("%s is a candidate", path.to_str());
 102                    match get_metadata_section(cx.os, path) {
 103                        Some(cvec) =>
 104                            if !crate_matches(cvec, cx.metas, cx.hash) {
 105                                debug!("skipping %s, metadata doesn't match",
 106                                    path.to_str());
 107                                None
 108                            } else {
 109                                debug!("found %s with matching metadata", path.to_str());
 110                                matches.push((path.to_str(), cvec));
 111                                None
 112                            },
 113                        _ => {
 114                            debug!("could not load metadata for %s", path.to_str());
 115                            None
 116                        }
 117                    }
 118                 }
 119                 else {
 120                     None
 121                 }
 122        }
 123      });
 124  
 125      match matches.len() {
 126          0 => None,
 127          1 => Some(matches[0]),
 128          _ => {
 129              cx.diag.span_err(
 130                      cx.span, fmt!("multiple matching crates for `%s`", crate_name));
 131                  cx.diag.handler().note("candidates:");

 133                      let ident = pair.first();
 134                      let data = pair.second();
 135                      cx.diag.handler().note(fmt!("path: %s", ident));
 136                      let attrs = decoder::get_crate_attributes(data);
 137                      note_linkage_attrs(cx.intr, cx.diag, attrs);
 138                  }
 139                  cx.diag.handler().abort_if_errors();
 140                  None
 141          }
 142      }
 143  }

 186  pub fn metadata_matches(extern_metas: &[@ast::MetaItem],
 187                          local_metas: &[@ast::MetaItem]) -> bool {
 188  
 189  // extern_metas: metas we read from the crate

 195          attr::contains(extern_metas, *needed)

 199  fn get_metadata_section(os: os,
 200                          filename: &Path) -> Option<@~[u8]> {
 201      unsafe {
 202          let mb = do filename.to_c_str().with_ref |buf| {
 203              llvm::LLVMRustCreateMemoryBufferWithContentsOfFile(buf)
 204          };
 205          if mb as int == 0 { return option::None::<@~[u8]>; }
 206          let of = match mk_object_file(mb) {
 207              option::Some(of) => of,
 208              _ => return option::None::<@~[u8]>
 209          };
 210          let si = mk_section_iter(of.llof);
 211          while llvm::LLVMIsSectionIteratorAtEnd(of.llof, si.llsi) == False {
 212              let name_buf = llvm::LLVMGetSectionName(si.llsi);
 213              let name = str::raw::from_c_str(name_buf);
 214              debug!("get_metadata_section: name %s", name);
 215              if name == read_meta_section_name(os) {
 216                  let cbuf = llvm::LLVMGetSectionContents(si.llsi);
 217                  let csz = llvm::LLVMGetSectionSize(si.llsi) as uint;
 218                  let mut found = None;
 219                  let cvbuf: *u8 = cast::transmute(cbuf);
 220                  let vlen = encoder::metadata_encoding_version.len();
 221                  debug!("checking %u bytes of metadata-version stamp",
 222                         vlen);
 223                  let minsz = num::min(vlen, csz);
 224                  let mut version_ok = false;
 225                  do vec::raw::buf_as_slice(cvbuf, minsz) |buf0| {
 226                      version_ok = (buf0 ==
 227                                    encoder::metadata_encoding_version);
 228                  }
 229                  if !version_ok { return None; }
 230  
 231                  let cvbuf1 = ptr::offset(cvbuf, vlen as int);
 232                  debug!("inflating %u bytes of compressed metadata",
 233                         csz - vlen);
 234                  do vec::raw::buf_as_slice(cvbuf1, csz-vlen) |bytes| {
 235                      let inflated = flate::inflate_bytes(bytes);
 236                      found = Some(@(inflated));
 237                  }
 238                  if found != None {
 239                      return found;
 240                  }
 241              }
 242              llvm::LLVMMoveToNextSection(si.llsi);
 243          }
 244          return option::None::<@~[u8]>;
 245      }
 246  }

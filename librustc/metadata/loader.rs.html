  86 fn find_library_crate_aux(
  87     cx: &Context,
  88     (prefix, suffix): (~str, ~str),
  89     filesearch: @filesearch::FileSearch
  90 ) -> Option<(~str, @~[u8])> {
  91     let crate_name = crate_name_from_metas(cx.metas);
  92     let prefix = prefix + crate_name + "-";
  93 
  94     let mut matches = ~[];
  95     filesearch::search(filesearch, |path| -> Option<()> {
  96         debug!("inspecting file %s", path.to_str());
  97         match path.filename() {
  98             Some(ref f) if f.starts_with(prefix) && f.ends_with(suffix) => {
  99                 debug!("%s is a candidate", path.to_str());
 100                 match get_metadata_section(cx.os, path) {
 101                     Some(cvec) =>
 102                         if !crate_matches(cvec, cx.metas, cx.hash) {
 103                             debug!("skipping %s, metadata doesn't match",
 104                                    path.to_str());
 105                             None
 106                         } else {
 107                             debug!("found %s with matching metadata", path.to_str());
 108                             matches.push((path.to_str(), cvec));
 109                             None
 110                         },
 111                     _ => {
 112                         debug!("could not load metadata for %s", path.to_str());
 113                         None
 114                     }
 115                 }
 116             }
 117             _ => {
 118                 debug!("skipping %s, doesn't look like %s*%s", path.to_str(),
 119                        prefix, suffix);
 120                 None
 121             }
 122         }});
 123 
 124     match matches.len() {
 125         0 => None,
 126         1 => Some(matches[0]),
 127         _ => {
 128             cx.diag.span_err(
 129                     cx.span, fmt!("multiple matching crates for `%s`", crate_name));
 130                 cx.diag.handler().note("candidates:");
 131                 for pair in matches.iter() {
 132                     let ident = pair.first();
 133                     let data = pair.second();
 134                     cx.diag.handler().note(fmt!("path: %s", ident));
 135                     let attrs = decoder::get_crate_attributes(data);
 136                     note_linkage_attrs(cx.intr, cx.diag, attrs);
 137                 }
 138                 cx.diag.handler().abort_if_errors();
 139                 None
 140             }
 141         }
 142 }

 186 fn get_metadata_section(os: os,
 187                         filename: &Path) -> Option<@~[u8]> {
 188     unsafe {
 189         let mb = do filename.to_str().as_c_str |buf| {
 190             llvm::LLVMRustCreateMemoryBufferWithContentsOfFile(buf)
 191         };
 192         if mb as int == 0 { return option::None::<@~[u8]>; }
 193         let of = match mk_object_file(mb) {
 194             option::Some(of) => of,
 195             _ => return option::None::<@~[u8]>
 196         };
 197         let si = mk_section_iter(of.llof);
 198         while llvm::LLVMIsSectionIteratorAtEnd(of.llof, si.llsi) == False {
 199             let name_buf = llvm::LLVMGetSectionName(si.llsi);
 200             let name = str::raw::from_c_str(name_buf);
 201             debug!("get_metadata_section: name %s", name);
 202             if name == read_meta_section_name(os) {
 203                 let cbuf = llvm::LLVMGetSectionContents(si.llsi);
 204                 let csz = llvm::LLVMGetSectionSize(si.llsi) as uint;
 205                 let mut found = None;
 206                 let cvbuf: *u8 = cast::transmute(cbuf);
 207                 let vlen = encoder::metadata_encoding_version.len();
 208                 debug!("checking %u bytes of metadata-version stamp",
 209                        vlen);
 210                 let minsz = num::min(vlen, csz);
 211                 let mut version_ok = false;
 212                 do vec::raw::buf_as_slice(cvbuf, minsz) |buf0| {
 213                     version_ok = (buf0 ==
 214                                   encoder::metadata_encoding_version);
 215                 }
 216                 if !version_ok { return None; }
 217 
 218                 let cvbuf1 = ptr::offset(cvbuf, vlen as int);
 219                 debug!("inflating %u bytes of compressed metadata",
 220                        csz - vlen);
 221                 do vec::raw::buf_as_slice(cvbuf1, csz-vlen) |bytes| {
 222                     let inflated = flate::inflate_bytes(bytes);
 223                     found = Some(@(inflated));
 224                 }
 225                 if found != None {
 226                     return found;
 227                 }
 228             }
 229             llvm::LLVMMoveToNextSection(si.llsi);
 230         }
 231         return option::None::<@~[u8]>;
 232     }
 233 }

 125 fn visit_crate(e: &Env, c: &ast::Crate) {
 126     let cstore = e.cstore;
 127 
 128     for a in c.attrs.iter().filter(|m| "link_args" == m.name()) {
 129         match a.value_str() {
 130           Some(ref linkarg) => {
 131             cstore::add_used_link_args(cstore, *linkarg);
 132           }
 133           None => {/* fallthrough */ }
 134         }
 135     }
 136 }

 154 fn visit_item(e: &Env, i: @ast::item) {
 155     match i.node {
 156       ast::item_foreign_mod(ref fm) => {
 157         if fm.abis.is_rust() || fm.abis.is_intrinsic() {
 158             return;
 159         }
 160 
 161         let cstore = e.cstore;
 162         let mut already_added = false;
 163         let link_args = i.attrs.iter()
 164             .filter_map(|at| if "link_args" == at.name() {Some(at)} else {None})
 165             .collect::<~[&ast::Attribute]>();
 166 
 167         match fm.sort {
 168             ast::named => {
 169                 let link_name = i.attrs.iter()
 170                     .find_(|at| "link_name" == at.name())
 171                     .chain(|at| at.value_str());
 172 
 173                 let foreign_name = match link_name {
 174                         Some(nn) => {
 175                             if nn.is_empty() {
 176                                 e.diag.span_fatal(
 177                                     i.span,
 178                                     "empty #[link_name] not allowed; use \
 179                                      #[nolink].");
 180                             }
 181                             nn
 182                         }
 183                         None => token::ident_to_str(&i.ident)
 184                     };
 185                 if !attr::contains_name(i.attrs, "nolink") {
 186                     already_added =
 187                         !cstore::add_used_library(cstore, foreign_name);
 188                 }
 189                 if !link_args.is_empty() && already_added {
 190                     e.diag.span_fatal(i.span, ~"library '" + foreign_name +
 191                                "' already added: can't specify link_args.");
 192                 }
 193             }
 194             ast::anonymous => { /* do nothing */ }
 195         }
 196 
 197         for m in link_args.iter() {
 198             match m.value_str() {
 199                 Some(linkarg) => {
 200                     cstore::add_used_link_args(cstore, linkarg);
 201                 }
 202                 None => { /* fallthrough */ }
 203             }
 204         }
 205       }
 206       _ => { }
 207     }
 208 }

 224 fn existing_match(e: &Env, metas: &[@ast::MetaItem], hash: &str)
 225                -> Option<int> {
 226     for c in e.crate_cache.iter() {
 227         if loader::metadata_matches(*c.metas, metas)
 228             && (hash.is_empty() || c.hash.as_slice() == hash) {
 229             return Some(c.cnum);
 230         }
 231     }
 232     return None;
 233 }

 235 fn resolve_crate(e: @mut Env,
 236                  ident: ast::ident,
 237                  metas: ~[@ast::MetaItem],
 238                  hash: @str,
 239                  span: span)
 240               -> ast::CrateNum {
 241     let metas = metas_with_ident(token::ident_to_str(&ident), metas);
 242 
 243     match existing_match(e, metas, hash) {
 244       None => {
 245         let load_ctxt = loader::Context {
 246             diag: e.diag,
 247             filesearch: e.filesearch,
 248             span: span,
 249             ident: ident,
 250             metas: metas,
 251             hash: hash,
 252             os: e.os,
 253             is_static: e.statik,
 254             intr: e.intr
 255         };
 256         let (lident, ldata) = loader::load_library_crate(&load_ctxt);
 257 
 258         let cfilename = Path(lident);
 259         let cdata = ldata;
 260 
 261         let attrs = decoder::get_crate_attributes(cdata);
 262         let linkage_metas = attr::find_linkage_metas(attrs);
 263         let hash = decoder::get_crate_hash(cdata);
 264 
 265         // Claim this crate number and cache it
 266         let cnum = e.next_crate_num;
 267         e.crate_cache.push(cache_entry {
 268             cnum: cnum,
 269             span: span,
 270             hash: hash,
 271             metas: @linkage_metas
 272         });
 273         e.next_crate_num += 1;
 274 
 275         // Now resolve the crates referenced by this crate
 276         let cnum_map = resolve_crate_deps(e, cdata);
 277 
 278         let cname =
 279             match attr::last_meta_item_value_str_by_name(load_ctxt.metas,
 280                                                          "name") {
 281                 Some(v) => v,
 282                 None => token::ident_to_str(&ident),
 283             };
 284         let cmeta = @cstore::crate_metadata {
 285             name: cname,
 286             data: cdata,
 287             cnum_map: cnum_map,
 288             cnum: cnum
 289         };
 290 
 291         let cstore = e.cstore;
 292         cstore::set_crate_data(cstore, cnum, cmeta);
 293         cstore::add_used_crate_file(cstore, &cfilename);
 294         return cnum;
 295       }
 296       Some(cnum) => {
 297         return cnum;
 298       }
 299     }
 300 }

 302 // Go through the crate metadata and load any crates that it references
 303 fn resolve_crate_deps(e: @mut Env, cdata: @~[u8]) -> cstore::cnum_map {
 304     debug!("resolving deps of external crate");
 305     // The map from crate numbers in the crate we're resolving to local crate
 306     // numbers

 309     for dep in r.iter() {
 310         let extrn_cnum = dep.cnum;

 312         let cname_str = token::ident_to_str(&dep.name);
 313         let cmetas = metas_with(dep.vers, @"vers", ~[]);
 314         debug!("resolving dep crate %s ver: %s hash: %s",
 315                cname_str, dep.vers, dep.hash);
 316         match existing_match(e,
 317                              metas_with_ident(cname_str, cmetas.clone()),
 318                              dep.hash) {
 319           Some(local_cnum) => {
 320             debug!("already have it");
 321             // We've already seen this crate
 322             cnum_map.insert(extrn_cnum, local_cnum);
 323           }
 324           None => {
 325             debug!("need to load it");
 326             // This is a new one so we've got to load it
 327             // FIXME (#2404): Need better error reporting than just a bogus
 328             // span.
 329             let fake_span = dummy_sp();
 330             let local_cnum = resolve_crate(e, cname, cmetas, dep.hash,
 331                                            fake_span);
 332             cnum_map.insert(extrn_cnum, local_cnum);
 333           }
 334         }
 335     }
 336     return @mut cnum_map;
 337 }

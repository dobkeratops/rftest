 178  fn visit_item(e: &Env, i: &ast::Item) {
 179      match i.node {
 180          ast::ItemForeignMod(ref fm) => {
 181              if fm.abi == abi::Rust || fm.abi == abi::RustIntrinsic {
 182                  return;
 183              }
 184  
 185              // First, add all of the custom link_args attributes
 186              let link_args = i.attrs.iter()
 187                  .filter_map(|at| if at.name().equiv(&("link_args")) {
 188                      Some(at)
 189                  } else {
 190                      None
 191                  })
 192                  .collect::<Vec<&ast::Attribute>>();
 193              for m in link_args.iter() {
 194                  match m.value_str() {
 195                      Some(linkarg) => e.sess.cstore.add_used_link_args(linkarg.get()),
 196                      None => { /* fallthrough */ }
 197                  }
 198              }
 199  
 200              // Next, process all of the #[link(..)]-style arguments
 201              let link_args = i.attrs.iter()
 202                  .filter_map(|at| if at.name().equiv(&("link")) {
 203                      Some(at)
 204                  } else {
 205                      None
 206                  })
 207                  .collect::<Vec<&ast::Attribute>>();
 208              for m in link_args.iter() {
 209                  match m.meta_item_list() {
 210                      Some(items) => {
 211                          let kind = items.iter().find(|k| {
 212                              k.name().equiv(&("kind"))
 213                          }).and_then(|a| a.value_str());
 214                          let kind = match kind {
 215                              Some(k) => {
 216                                  if k.equiv(&("static")) {
 217                                      cstore::NativeStatic
 218                                  } else if e.sess.targ_cfg.os == abi::OsMacos &&
 219                                            k.equiv(&("framework")) {
 220                                      cstore::NativeFramework
 221                                  } else if k.equiv(&("framework")) {
 222                                      e.sess.span_err(m.span,
 223                                          "native frameworks are only available \
 224                                           on OSX targets");
 225                                      cstore::NativeUnknown
 226                                  } else {
 227                                      e.sess.span_err(m.span,
 228                                          format!("unknown kind: `{}`", k));
 229                                      cstore::NativeUnknown
 230                                  }
 231                              }
 232                              None => cstore::NativeUnknown
 233                          };
 234                          let n = items.iter().find(|n| {
 235                              n.name().equiv(&("name"))
 236                          }).and_then(|a| a.value_str());
 237                          let n = match n {
 238                              Some(n) => n,
 239                              None => {
 240                                  e.sess.span_err(m.span,
 241                                      "#[link(...)] specified without \
 242                                       `name = \"foo\"`");
 243                                  InternedString::new("foo")
 244                              }
 245                          };
 246                          if n.get().is_empty() {
 247                              e.sess.span_err(m.span, "#[link(name = \"\")] given with empty name");
 248                          } else {
 249                              e.sess.cstore.add_used_library(n.get().to_owned(), kind);
 250                          }
 251                      }
 252                      None => {}
 253                  }
 254              }
 255          }
 256          _ => { }
 257      }
 258  }

 339                  span: span,

 342              let source = cstore::CrateSource {

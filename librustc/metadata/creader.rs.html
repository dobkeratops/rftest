 179  fn visit_item(e: &Env, i: &ast::Item) {
 180      match i.node {
 181          ast::ItemForeignMod(ref fm) => {
 182              if fm.abi == abi::Rust || fm.abi == abi::RustIntrinsic {
 183                  return;
 184              }

 186              // First, add all of the custom link_args attributes
 187              let link_args = i.attrs.iter()

 189                      Some(at)

 195                  match m.value_str() {
 196                      Some(linkarg) => e.sess.cstore.add_used_link_args(linkarg.get()),
 197                      None => { /* fallthrough */ }

 201              // Next, process all of the #[link(..)]-style arguments
 202              let link_args = i.attrs.iter()
 203                  .filter_map(|at| if at.name().equiv(&("link")) {
 204                      Some(at)
 205                  } else {
 206                      None
 207                  })
 208                  .collect::<~[&ast::Attribute]>();
 209              for m in link_args.iter() {
 210                  match m.meta_item_list() {
 211                      Some(items) => {
 212                          let kind = items.iter().find(|k| {
 213                              k.name().equiv(&("kind"))
 214                          }).and_then(|a| a.value_str());

 216                              Some(k) => {

 218                                      cstore::NativeStatic

 220                                            k.equiv(&("framework")) {
 221                                      cstore::NativeFramework
 222                                  } else if k.equiv(&("framework")) {
 223                                      e.sess.span_err(m.span,
 224                                          "native frameworks are only available \
 225                                           on OSX targets");
 226                                      cstore::NativeUnknown
 227                                  } else {
 228                                      e.sess.span_err(m.span,
 229                                          format!("unknown kind: `{}`", k));
 230                                      cstore::NativeUnknown
 231                                  }
 232                              }
 233                              None => cstore::NativeUnknown
 234                          };
 235                          let n = items.iter().find(|n| {
 236                              n.name().equiv(&("name"))
 237                          }).and_then(|a| a.value_str());

 242                                      "#[link(...)] specified without \

 245                              }
 246                          };

 249                          } else {
 250                              e.sess.cstore.add_used_library(n.get().to_owned(), kind);
 251                          }
 252                      }
 253                      None => {}
 254                  }
 255              }
 256          }
 257          _ => { }
 258      }

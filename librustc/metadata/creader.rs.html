 142            let ident = token::ident_to_str(&ident);

 170  fn visit_item(e: &Env, i: @ast::item) {
 171      match i.node {
 172        ast::item_foreign_mod(ref fm) => {
 173          if fm.abis.is_rust() || fm.abis.is_intrinsic() {
 174              return;
 175          }
 176  
 177          let cstore = e.cstore;
 178          let mut already_added = false;
 179          let link_args = i.attrs.iter()
 180              .filter_map(|at| if "link_args" == at.name() {Some(at)} else {None})
 181              .collect::<~[&ast::Attribute]>();
 182  
 183          match fm.sort {
 184              ast::named => {
 185                  let link_name = i.attrs.iter()
 186                      .find(|at| "link_name" == at.name())
 187                      .chain(|at| at.value_str());
 188  
 189                  let foreign_name = match link_name {
 190                          Some(nn) => {
 191                              if nn.is_empty() {
 192                                  e.diag.span_fatal(
 193                                      i.span,
 194                                      "empty #[link_name] not allowed; use \
 195                                       #[nolink].");
 196                              }
 197                              nn
 198                          }
 199                          None => token::ident_to_str(&i.ident)
 200                      };
 201                  if !attr::contains_name(i.attrs, "nolink") {
 202                      already_added =
 203                          !cstore::add_used_library(cstore, foreign_name);
 204                  }
 205                  if !link_args.is_empty() && already_added {
 206                      e.diag.span_fatal(i.span, ~"library '" + foreign_name +
 207                                 "' already added: can't specify link_args.");
 208                  }
 209              }
 210              ast::anonymous => { /* do nothing */ }
 211          }
 212  
 213          for m in link_args.iter() {
 214              match m.value_str() {
 215                  Some(linkarg) => {
 216                      cstore::add_used_link_args(cstore, linkarg);
 217                  }
 218                  None => { /* fallthrough */ }
 219              }
 220          }
 221        }
 222        _ => { }
 223      }
 224  }

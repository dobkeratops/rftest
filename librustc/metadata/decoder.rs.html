   56  fn lookup_hash<'a>(d: ebml::Doc<'a>, eq_fn: |&[u8]| -> bool,
   57                     hash: u64) -> Option<ebml::Doc<'a>> {
   58      let index = reader::get_doc(d, tag_index);
   59      let table = reader::get_doc(index, tag_index_table);
   60      let hash_pos = table.start + (hash % 256 * 4) as uint;
   61      let pos = u64_from_be_bytes(d.data, hash_pos, 4) as uint;
   62      let tagged_doc = reader::doc_at(d.data, pos).unwrap();
   63  
   64      let belt = tag_index_buckets_bucket_elt;
   65  
   66      let mut ret = None;
   67      reader::tagged_docs(tagged_doc.doc, belt, |elt| {
   68          let pos = u64_from_be_bytes(elt.data, elt.start, 4) as uint;
   69          if eq_fn(elt.data.slice(elt.start + 4, elt.end)) {
   70              ret = Some(reader::doc_at(d.data, pos).unwrap().doc);
   71              false
   72          } else {
   73              true
   74          }
   75      });
   76      ret
   77  }

  322  fn item_to_def_like(item: ebml::Doc, did: ast::DefId, cnum: ast::CrateNum)
  323      -> DefLike {
  324      let fam = item_family(item);
  325      match fam {
  326          ImmStatic => DlDef(ast::DefStatic(did, false)),
  327          MutStatic => DlDef(ast::DefStatic(did, true)),
  328          Struct    => DlDef(ast::DefStruct(did)),
  329          UnsafeFn  => DlDef(ast::DefFn(did, ast::UnsafeFn)),
  330          Fn        => DlDef(ast::DefFn(did, ast::NormalFn)),
  331          ForeignFn => DlDef(ast::DefFn(did, ast::ExternFn)),
  332          StaticMethod | UnsafeStaticMethod => {
  333              let fn_style = if fam == UnsafeStaticMethod { ast::UnsafeFn } else
  334                  { ast::NormalFn };
  335              // def_static_method carries an optional field of its enclosing
  336              // trait or enclosing impl (if this is an inherent static method).
  337              // So we need to detect whether this is in a trait or not, which
  338              // we do through the mildly hacky way of checking whether there is
  339              // a trait_method_sort.
  340              let provenance = if reader::maybe_get_doc(
  341                    item, tag_item_trait_method_sort).is_some() {
  342                  ast::FromTrait(item_reqd_and_translated_parent_item(cnum,
  343                                                                      item))
  344              } else {
  345                  ast::FromImpl(item_reqd_and_translated_parent_item(cnum,
  346                                                                     item))
  347              };
  348              DlDef(ast::DefStaticMethod(did, provenance, fn_style))
  349          }
  350          Type | ForeignType => DlDef(ast::DefTy(did)),
  351          Mod => DlDef(ast::DefMod(did)),
  352          ForeignMod => DlDef(ast::DefForeignMod(did)),
  353          StructVariant => {
  354              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  355              DlDef(ast::DefVariant(enum_did, did, true))
  356          }
  357          TupleVariant => {
  358              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  359              DlDef(ast::DefVariant(enum_did, did, false))
  360          }
  361          Trait => DlDef(ast::DefTrait(did)),
  362          Enum => DlDef(ast::DefTy(did)),
  363          Impl => DlImpl(did),
  364          PublicField | InheritedField => DlField,
  365      }
  366  }

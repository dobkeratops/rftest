   54  fn lookup_hash(d: ebml::Doc, eq_fn: &fn(x:&[u8]) -> bool, hash: u64) ->
   55     Option<ebml::Doc> {
   56      let index = reader::get_doc(d, tag_index);
   57      let table = reader::get_doc(index, tag_index_table);
   58      let hash_pos = table.start + (hash % 256 * 4) as uint;
   59      let pos = io::u64_from_be_bytes(*d.data, hash_pos, 4) as uint;
   60      let tagged_doc = reader::doc_at(d.data, pos);
   61  
   62      let belt = tag_index_buckets_bucket_elt;
   63  
   64      let mut ret = None;
   65      do reader::tagged_docs(tagged_doc.doc, belt) |elt| {
   66          let pos = io::u64_from_be_bytes(*elt.data, elt.start, 4) as uint;
   67          if eq_fn(elt.data.slice(elt.start + 4, elt.end)) {
   68              ret = Some(reader::doc_at(d.data, pos).doc);
   69              false
   70          } else {
   71              true
   72          }
   73      };
   74      ret
   75  }

   79  pub fn maybe_find_item(item_id: int, items: ebml::Doc) -> Option<ebml::Doc> {

   86                  |a| eq_item(a, item_id),

  296

  335  fn item_to_def_like(item: ebml::Doc, did: ast::DefId, cnum: ast::CrateNum)
  336      -> DefLike {
  337      let fam = item_family(item);
  338      match fam {
  339          ImmStatic => DlDef(ast::DefStatic(did, false)),
  340          MutStatic => DlDef(ast::DefStatic(did, true)),
  341          Struct    => DlDef(ast::DefStruct(did)),
  342          UnsafeFn  => DlDef(ast::DefFn(did, ast::unsafe_fn)),
  343          Fn        => DlDef(ast::DefFn(did, ast::impure_fn)),
  344          ForeignFn => DlDef(ast::DefFn(did, ast::extern_fn)),
  345          StaticMethod | UnsafeStaticMethod => {
  346              let purity = if fam == UnsafeStaticMethod { ast::unsafe_fn } else
  347                  { ast::impure_fn };
  348              // def_static_method carries an optional field of its enclosing
  349              // trait or enclosing impl (if this is an inherent static method).
  350              // So we need to detect whether this is in a trait or not, which
  351              // we do through the mildly hacky way of checking whether there is
  352              // a trait_method_sort.
  353              let provenance = if reader::maybe_get_doc(
  354                    item, tag_item_trait_method_sort).is_some() {
  355                  ast::FromTrait(item_reqd_and_translated_parent_item(cnum,
  356                                                                      item))
  357              } else {
  358                  ast::FromImpl(item_reqd_and_translated_parent_item(cnum,
  359                                                                     item))
  360              };
  361              DlDef(ast::DefStaticMethod(did, provenance, purity))
  362          }
  363          Type | ForeignType => DlDef(ast::DefTy(did)),
  364          Mod => DlDef(ast::DefMod(did)),
  365          ForeignMod => DlDef(ast::DefForeignMod(did)),
  366          StructVariant => {
  367              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  368              DlDef(ast::DefVariant(enum_did, did, true))
  369          }
  370          TupleVariant => {
  371              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  372              DlDef(ast::DefVariant(enum_did, did, false))
  373          }
  374          Trait => DlDef(ast::DefTrait(did)),
  375          Enum => DlDef(ast::DefTy(did)),
  376          Impl => DlImpl(did),
  377          PublicField | PrivateField | InheritedField => DlField,
  378      }
  379  }

  560                         *self.path_builder,

  586              }

  590              self.pop_name(old_len)
  591          }
  592  
  593          continue
  594      }

   56  fn lookup_hash<'a>(d: ebml::Doc<'a>, eq_fn: |&[u8]| -> bool,
   57                     hash: u64) -> Option<ebml::Doc<'a>> {
   58      let index = reader::get_doc(d, tag_index);
   59      let table = reader::get_doc(index, tag_index_table);
   60      let hash_pos = table.start + (hash % 256 * 4) as uint;
   61      let pos = u64_from_be_bytes(d.data, hash_pos, 4) as uint;
   62      let tagged_doc = reader::doc_at(d.data, pos);
   63  
   64      let belt = tag_index_buckets_bucket_elt;
   65  
   66      let mut ret = None;
   67      reader::tagged_docs(tagged_doc.doc, belt, |elt| {
   68          let pos = u64_from_be_bytes(elt.data, elt.start, 4) as uint;
   69          if eq_fn(elt.data.slice(elt.start + 4, elt.end)) {
   70              ret = Some(reader::doc_at(d.data, pos).doc);
   71              false
   72          } else {
   73              true
   74          }
   75      });
   76      ret
   77  }

   81  pub fn maybe_find_item<'a>(item_id: ast::NodeId,
   82                             items: ebml::Doc<'a>) -> Option<ebml::Doc<'a>> {
   83      fn eq_item(bytes: &[u8], item_id: ast::NodeId) -> bool {
   84          return u64_from_be_bytes(
   85              bytes.slice(0u, 4u), 0u, 4u) as ast::NodeId
   86              == item_id;
   87      }
   88      lookup_hash(items,
   89                  |a| eq_item(a, item_id),
   90                  hash::hash(&(item_id as i64)))
   91  }

  332  fn item_to_def_like(item: ebml::Doc, did: ast::DefId, cnum: ast::CrateNum)
  333      -> DefLike {
  334      let fam = item_family(item);
  335      match fam {
  336          ImmStatic => DlDef(ast::DefStatic(did, false)),
  337          MutStatic => DlDef(ast::DefStatic(did, true)),
  338          Struct    => DlDef(ast::DefStruct(did)),
  339          UnsafeFn  => DlDef(ast::DefFn(did, ast::UnsafeFn)),
  340          Fn        => DlDef(ast::DefFn(did, ast::ImpureFn)),
  341          ForeignFn => DlDef(ast::DefFn(did, ast::ExternFn)),
  342          StaticMethod | UnsafeStaticMethod => {
  343              let purity = if fam == UnsafeStaticMethod { ast::UnsafeFn } else
  344                  { ast::ImpureFn };
  345              // def_static_method carries an optional field of its enclosing
  346              // trait or enclosing impl (if this is an inherent static method).
  347              // So we need to detect whether this is in a trait or not, which
  348              // we do through the mildly hacky way of checking whether there is
  349              // a trait_method_sort.
  350              let provenance = if reader::maybe_get_doc(
  351                    item, tag_item_trait_method_sort).is_some() {
  352                  ast::FromTrait(item_reqd_and_translated_parent_item(cnum,
  353                                                                      item))
  354              } else {
  355                  ast::FromImpl(item_reqd_and_translated_parent_item(cnum,
  356                                                                     item))
  357              };
  358              DlDef(ast::DefStaticMethod(did, provenance, purity))
  359          }
  360          Type | ForeignType => DlDef(ast::DefTy(did)),
  361          Mod => DlDef(ast::DefMod(did)),
  362          ForeignMod => DlDef(ast::DefForeignMod(did)),
  363          StructVariant => {
  364              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  365              DlDef(ast::DefVariant(enum_did, did, true))
  366          }
  367          TupleVariant => {
  368              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  369              DlDef(ast::DefVariant(enum_did, did, false))
  370          }
  371          Trait => DlDef(ast::DefTrait(did)),
  372          Enum => DlDef(ast::DefTy(did)),
  373          Impl => DlImpl(did),
  374          PublicField | PrivateField | InheritedField => DlField,
  375      }
  376  }

  380                       tcx: &ty::ctxt) -> ty::TraitDef

  384                                       tag_items_data_item_ty_param_bounds);

  386      let mut bounds = ty::EmptyBuiltinBounds();
  387      // Collect the builtin bounds from the encoded supertraits.
  388      // FIXME(#8559): They should be encoded directly.
  389      reader::tagged_docs(item_doc, tag_item_super_trait_ref, |trait_doc| {

  392          tcx.lang_items.to_builtin_kind(trait_ref.def_id).map(|bound| {
  393              bounds.add(bound);

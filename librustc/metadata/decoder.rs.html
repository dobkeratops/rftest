   54  fn lookup_hash(d: ebml::Doc, eq_fn: &fn(x:&[u8]) -> bool, hash: u64) ->
   55     Option<ebml::Doc> {
   56      let index = reader::get_doc(d, tag_index);
   57      let table = reader::get_doc(index, tag_index_table);
   58      let hash_pos = table.start + (hash % 256 * 4) as uint;
   59      let pos = io::u64_from_be_bytes(*d.data, hash_pos, 4) as uint;
   60      let tagged_doc = reader::doc_at(d.data, pos);
   61  
   62      let belt = tag_index_buckets_bucket_elt;
   63  
   64      let mut ret = None;
   65      do reader::tagged_docs(tagged_doc.doc, belt) |elt| {
   66          let pos = io::u64_from_be_bytes(*elt.data, elt.start, 4) as uint;
   67          if eq_fn(elt.data.slice(elt.start + 4, elt.end)) {
   68              ret = Some(reader::doc_at(d.data, pos).doc);
   69              false
   70          } else {
   71              true
   72          }
   73      };
   74      ret
   75  }

   79  pub fn maybe_find_item(item_id: int, items: ebml::Doc) -> Option<ebml::Doc> {

   86                  |a| eq_item(a, item_id),

  324  fn item_to_def_like(item: ebml::Doc, did: ast::def_id, cnum: ast::CrateNum)
  325      -> def_like {
  326      let fam = item_family(item);
  327      match fam {
  328          ImmStatic => dl_def(ast::def_static(did, false)),
  329          MutStatic => dl_def(ast::def_static(did, true)),
  330          Struct    => dl_def(ast::def_struct(did)),
  331          UnsafeFn  => dl_def(ast::def_fn(did, ast::unsafe_fn)),
  332          Fn        => dl_def(ast::def_fn(did, ast::impure_fn)),
  333          ForeignFn => dl_def(ast::def_fn(did, ast::extern_fn)),
  334          StaticMethod | UnsafeStaticMethod => {
  335              let purity = if fam == UnsafeStaticMethod { ast::unsafe_fn } else
  336                  { ast::impure_fn };
  337              // def_static_method carries an optional field of its enclosing
  338              // *trait*, but not an inclosing Impl (if this is an inherent
  339              // static method). So we need to detect whether this is in
  340              // a trait or not, which we do through the mildly hacky
  341              // way of checking whether there is a trait_method_sort.
  342              let trait_did_opt = if reader::maybe_get_doc(
  343                    item, tag_item_trait_method_sort).is_some() {
  344                  Some(item_reqd_and_translated_parent_item(cnum, item))
  345              } else { None };
  346              dl_def(ast::def_static_method(did, trait_did_opt, purity))
  347          }
  348          Type | ForeignType => dl_def(ast::def_ty(did)),
  349          Mod => dl_def(ast::def_mod(did)),
  350          ForeignMod => dl_def(ast::def_foreign_mod(did)),
  351          Variant => {
  352              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  353              dl_def(ast::def_variant(enum_did, did))
  354          }
  355          Trait => dl_def(ast::def_trait(did)),
  356          Enum => dl_def(ast::def_ty(did)),
  357          Impl => dl_impl(did),
  358          PublicField | PrivateField | InheritedField => dl_field,
  359      }
  360  }

  385  pub fn get_type(cdata: cmd, id: ast::NodeId, tcx: ty::ctxt)
  386      -> ty::ty_param_bounds_and_ty {
  387  
  388      let item = lookup_item(id, cdata.data);
  389      let t = item_type(ast::def_id { crate: cdata.cnum, node: id }, item, tcx,
  390                        cdata);
  391      let tp_defs = if family_has_type_params(item_family(item)) {
  392          item_ty_param_defs(item, tcx, cdata, tag_items_data_item_ty_param_bounds)

  494  impl<'self> EachItemContext<'self> {
  495      // Pushes the given name and returns the old length.
  496      fn push_name(&mut self, string: &str) -> uint {
  497          let path_len = self.path_builder.len();
  498          if path_len != 0 {
  499              self.path_builder.push_str("::")
  500          }
  501          self.path_builder.push_str(string);
  502          path_len
  503      }
  504  
  505      // Pops the given name.
  506      fn pop_name(&mut self, old_len: uint) {
  507          // XXX(pcwalton): There's no safe function to do this. :(
  508          unsafe {
  509              str::raw::set_len(self.path_builder, old_len)
  510          }
  511      }
  512  
  513      fn process_item_and_pop_name(&mut self,
  514                                   doc: ebml::Doc,
  515                                   def_id: ast::def_id,
  516                                   old_len: uint)
  517                                   -> bool {
  518          let def_like = item_to_def_like(doc, def_id, self.cdata.cnum);
  519          match def_like {
  520              dl_def(def) => {
  521                  debug!("(iterating over each item of a module) processing \
  522                          `%s` (def %?)",
  523                         *self.path_builder,
  524                         def);
  525              }
  526              _ => {
  527                  debug!("(iterating over each item of a module) processing \
  528                          `%s` (%d:%d)",
  529                         *self.path_builder,
  530                         def_id.crate,
  531                         def_id.node);
  532              }
  533          }
  534  
  535          let vis = item_visibility(doc);
  536  
  537          let mut continue = (self.callback)(*self.path_builder, def_like, vis);
  538  
  539          let family = item_family(doc);
  540          if family == ForeignMod {
  541              // These are unnamed; pop the name now.
  542              self.pop_name(old_len)
  543          }
  544  
  545          if continue {
  546              // Recurse if necessary.
  547              match family {
  548                  Mod | ForeignMod | Trait | Impl => {
  549                      continue = self.each_item_of_module(def_id);
  550                  }
  551                  ImmStatic | MutStatic | Struct | UnsafeFn | Fn | ForeignFn |
  552                  UnsafeStaticMethod | StaticMethod | Type | ForeignType |
  553                  Variant | Enum | PublicField | PrivateField |
  554                  InheritedField => {}
  555              }
  556          }
  557  
  558          if family != ForeignMod {
  559              self.pop_name(old_len)
  560          }
  561  
  562          continue
  563      }

  565      fn each_item_of_module(&mut self, def_id: ast::def_id) -> bool {
  566          // This item might not be in this crate. If it's not, look it up.
  567          let items = if def_id.crate == self.cdata.cnum {
  568              reader::get_doc(reader::Doc(self.cdata.data), tag_items)
  569          } else {
  570              let crate_data = (self.get_crate_data)(def_id.crate);
  571              let root = reader::Doc(crate_data.data);
  572              reader::get_doc(root, tag_items)
  573          };
  574  
  575          // Look up the item.
  576          let item_doc = match maybe_find_item(def_id.node, items) {
  577              None => return false,
  578              Some(item_doc) => item_doc,
  579          };
  580  
  581          self.each_child_of_module_or_crate(item_doc)
  582      }
  583  
  584      fn each_child_of_module_or_crate(&mut self, item_doc: ebml::Doc) -> bool {
  585          let mut continue = true;
  586  
  587          // Iterate over all children.
  588          do reader::tagged_docs(item_doc, tag_mod_child) |child_info_doc| {
  589              let child_def_id = reader::with_doc_data(child_info_doc,
  590                                                       parse_def_id);
  591              let child_def_id = translate_def_id(self.cdata, child_def_id);
  592  
  593              // This item may be in yet another crate, if it was the child of
  594              // a reexport.
  595              let other_crates_items = if child_def_id.crate ==
  596                      self.cdata.cnum {
  597                  reader::get_doc(reader::Doc(self.cdata.data), tag_items)
  598              } else {
  599                  let crate_data = (self.get_crate_data)(child_def_id.crate);
  600                  let root = reader::Doc(crate_data.data);
  601                  reader::get_doc(root, tag_items)
  602              };
  603  
  604              debug!("(iterating over each item of a module) looking up item \
  605                      %d:%d in `%s`, crate %d",
  606                     child_def_id.crate,
  607                     child_def_id.node,
  608                     *self.path_builder,
  609                     self.cdata.cnum);
  610  
  611              // Get the item.
  612              match maybe_find_item(child_def_id.node, other_crates_items) {
  613                  None => {}
  614                  Some(child_item_doc) => {
  615                      // Push the name.
  616                      let child_name = item_name(self.intr, child_item_doc);
  617                      debug!("(iterating over each item of a module) pushing \
  618                              name `%s` onto `%s`",
  619                             token::ident_to_str(&child_name),
  620                             *self.path_builder);
  621                      let old_len =
  622                          self.push_name(token::ident_to_str(&child_name));
  623  
  624                      // Process this item.
  625                      continue = self.process_item_and_pop_name(child_item_doc,
  626                                                                child_def_id,
  627                                                                old_len);
  628                  }
  629              }
  630              continue
  631          };
  632  
  633          if !continue {
  634              return false
  635          }
  636  
  637          // Iterate over reexports.
  638          do each_reexport(item_doc) |reexport_doc| {
  639              let def_id_doc = reader::get_doc(
  640                  reexport_doc,
  641                  tag_items_data_item_reexport_def_id);
  642              let orig_def_id = reader::with_doc_data(def_id_doc, parse_def_id);
  643  
  644              // NB: was "cdata"
  645              let def_id = translate_def_id(self.cdata, orig_def_id);
  646  
  647              let name_doc = reader::get_doc(reexport_doc,
  648                                             tag_items_data_item_reexport_name);
  649              let name = name_doc.as_str_slice();
  650  
  651              // Push the name.
  652              debug!("(iterating over each item of a module) pushing \
  653                      reexported name `%s` onto `%s` (crate %d, orig %d, \
  654                      in crate %d)",
  655                     name,
  656                     *self.path_builder,
  657                     def_id.crate,
  658                     orig_def_id.crate,
  659                     self.cdata.cnum);
  660              let old_len = self.push_name(name);
  661  
  662              // This reexport may be in yet another crate.
  663              let other_crates_items = if def_id.crate == self.cdata.cnum {
  664                  reader::get_doc(reader::Doc(self.cdata.data), tag_items)
  665              } else {
  666                  let crate_data = (self.get_crate_data)(def_id.crate);
  667                  let root = reader::Doc(crate_data.data);
  668                  reader::get_doc(root, tag_items)
  669              };
  670  
  671              // Get the item.
  672              match maybe_find_item(def_id.node, other_crates_items) {
  673                  None => {}
  674                  Some(reexported_item_doc) => {
  675                      continue = self.process_item_and_pop_name(
  676                          reexported_item_doc,
  677                          def_id,
  678                          old_len);
  679                  }
  680              }
  681  
  682              continue
  683          };
  684  
  685          continue
  686      }
  687  }

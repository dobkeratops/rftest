   56  fn lookup_hash<'a>(d: ebml::Doc<'a>, eq_fn: |&[u8]| -> bool,
   57                     hash: u64) -> Option<ebml::Doc<'a>> {
   58      let index = reader::get_doc(d, tag_index);
   59      let table = reader::get_doc(index, tag_index_table);
   60      let hash_pos = table.start + (hash % 256 * 4) as uint;
   61      let pos = u64_from_be_bytes(d.data, hash_pos, 4) as uint;
   62      let tagged_doc = reader::doc_at(d.data, pos).unwrap();
   63  
   64      let belt = tag_index_buckets_bucket_elt;
   65  
   66      let mut ret = None;
   67      reader::tagged_docs(tagged_doc.doc, belt, |elt| {
   68          let pos = u64_from_be_bytes(elt.data, elt.start, 4) as uint;
   69          if eq_fn(elt.data.slice(elt.start + 4, elt.end)) {
   70              ret = Some(reader::doc_at(d.data, pos).unwrap().doc);
   71              false
   72          } else {
   73              true
   74          }
   75      });
   76      ret
   77  }

   81  pub fn maybe_find_item<'a>(item_id: ast::NodeId,
   82                             items: ebml::Doc<'a>) -> Option<ebml::Doc<'a>> {
   83      fn eq_item(bytes: &[u8], item_id: ast::NodeId) -> bool {
   84          return u64_from_be_bytes(
   85              bytes.slice(0u, 4u), 0u, 4u) as ast::NodeId
   86              == item_id;
   87      }
   88      lookup_hash(items,
   89                  |a| eq_item(a, item_id),
   90                  hash::hash(&(item_id as i64)))
   91  }

  325  fn item_to_def_like(item: ebml::Doc, did: ast::DefId, cnum: ast::CrateNum)
  326      -> DefLike {
  327      let fam = item_family(item);
  328      match fam {
  329          ImmStatic => DlDef(ast::DefStatic(did, false)),
  330          MutStatic => DlDef(ast::DefStatic(did, true)),
  331          Struct    => DlDef(ast::DefStruct(did)),
  332          UnsafeFn  => DlDef(ast::DefFn(did, ast::UnsafeFn)),
  333          Fn        => DlDef(ast::DefFn(did, ast::ImpureFn)),
  334          ForeignFn => DlDef(ast::DefFn(did, ast::ExternFn)),
  335          StaticMethod | UnsafeStaticMethod => {
  336              let purity = if fam == UnsafeStaticMethod { ast::UnsafeFn } else
  337                  { ast::ImpureFn };
  338              // def_static_method carries an optional field of its enclosing
  339              // trait or enclosing impl (if this is an inherent static method).
  340              // So we need to detect whether this is in a trait or not, which
  341              // we do through the mildly hacky way of checking whether there is
  342              // a trait_method_sort.
  343              let provenance = if reader::maybe_get_doc(
  344                    item, tag_item_trait_method_sort).is_some() {
  345                  ast::FromTrait(item_reqd_and_translated_parent_item(cnum,
  346                                                                      item))
  347              } else {
  348                  ast::FromImpl(item_reqd_and_translated_parent_item(cnum,
  349                                                                     item))
  350              };
  351              DlDef(ast::DefStaticMethod(did, provenance, purity))
  352          }
  353          Type | ForeignType => DlDef(ast::DefTy(did)),
  354          Mod => DlDef(ast::DefMod(did)),
  355          ForeignMod => DlDef(ast::DefForeignMod(did)),
  356          StructVariant => {
  357              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  358              DlDef(ast::DefVariant(enum_did, did, true))
  359          }
  360          TupleVariant => {
  361              let enum_did = item_reqd_and_translated_parent_item(cnum, item);
  362              DlDef(ast::DefVariant(enum_did, did, false))
  363          }
  364          Trait => DlDef(ast::DefTrait(did)),
  365          Enum => DlDef(ast::DefTy(did)),
  366          Impl => DlImpl(did),
  367          PublicField | PrivateField | InheritedField => DlField,
  368      }
  369  }

  371  pub fn get_trait_def(cdata: Cmd,
  372                       item_id: ast::NodeId,
  373                       tcx: &ty::ctxt) -> ty::TraitDef
  374  {
  375      let item_doc = lookup_item(item_id, cdata.data());
  376      let tp_defs = item_ty_param_defs(item_doc, tcx, cdata,
  377                                       tag_items_data_item_ty_param_bounds);
  378      let rp_defs = item_region_param_defs(item_doc, cdata);
  379      let mut bounds = ty::EmptyBuiltinBounds();
  380      // Collect the builtin bounds from the encoded supertraits.
  381      // FIXME(#8559): They should be encoded directly.
  382      reader::tagged_docs(item_doc, tag_item_super_trait_ref, |trait_doc| {
  383          // NB. Bypasses real supertraits. See get_supertraits() if you wanted them.
  384          let trait_ref = doc_trait_ref(trait_doc, tcx, cdata);
  385          tcx.lang_items.to_builtin_kind(trait_ref.def_id).map(|bound| {
  386              bounds.add(bound);
  387          });
  388          true
  389      });
  390      ty::TraitDef {
  391          generics: ty::Generics {type_param_defs: tp_defs,
  392                                  region_param_defs: rp_defs},

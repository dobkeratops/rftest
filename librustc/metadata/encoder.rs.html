  322  fn encode_enum_variant_info(ecx: &EncodeContext,
  323                              ebml_w: &mut writer::Encoder,
  324                              id: NodeId,
  325                              variants: &[P<Variant>],
  326                              index: @RefCell<Vec<entry<i64>> >,
  327                              generics: &ast::Generics) {
  328      debug!("encode_enum_variant_info(id={:?})", id);
  329  
  330      let mut disr_val = 0;
  331      let mut i = 0;
  332      let vi = ty::enum_variants(ecx.tcx,
  333                                 ast::DefId { krate: LOCAL_CRATE, node: id });
  334      for variant in variants.iter() {
  335          let def_id = local_def(variant.node.id);
  336          index.borrow_mut().push(entry {
  337              val: variant.node.id as i64,
  338              pos: ebml_w.writer.tell().unwrap(),
  339          });
  340          ebml_w.start_tag(tag_items_data_item);
  341          encode_def_id(ebml_w, def_id);
  342          match variant.node.kind {
  343              ast::TupleVariantKind(_) => encode_family(ebml_w, 'v'),
  344              ast::StructVariantKind(_) => encode_family(ebml_w, 'V')
  345          }
  346          encode_name(ebml_w, variant.node.name.name);
  347          encode_parent_item(ebml_w, local_def(id));
  348          encode_visibility(ebml_w, variant.node.vis);
  349          encode_attributes(ebml_w, variant.node.attrs.as_slice());
  350          match variant.node.kind {
  351              ast::TupleVariantKind(ref args)
  352                      if args.len() > 0 && generics.ty_params.len() == 0 => {
  353                  encode_symbol(ecx, ebml_w, variant.node.id);
  354              }
  355              ast::TupleVariantKind(_) => {},
  356              ast::StructVariantKind(def) => {
  357                  let idx = encode_info_for_struct(ecx,
  358                                                   ebml_w,
  359                                                   def.fields.as_slice(),
  360                                                   index);
  361                  encode_struct_fields(ebml_w, def);
  362                  let bkts = create_index(idx);
  363                  encode_index(ebml_w, bkts, write_i64);
  364              }
  365          }
  366          if vi.get(i).disr_val != disr_val {
  367              encode_disr_val(ecx, ebml_w, vi.get(i).disr_val);
  368              disr_val = vi.get(i).disr_val;
  369          }
  370          encode_bounds_and_type(ebml_w, ecx,
  371                                 &lookup_item_type(ecx.tcx, def_id));
  372  
  373          ecx.tcx.map.with_path(variant.node.id, |path| encode_path(ebml_w, path));
  374          ebml_w.end_tag();
  375          disr_val += 1;
  376          i += 1;
  377      }
  378  }

  392  }

  519                  }
  520                  _ => {}
  521              }
  522          }
  523          _ => {}
  524      }

  527  }

  878                       index: @RefCell<Vec<entry<i64>> >) {

  880              val: item.id as i64,
  881              pos: ebml_w.writer.tell().unwrap(),

  883      }

  895              encode_family(ebml_w, 'b');
  896          } else {

  898          }

  900          encode_symbol(ecx, ebml_w, item.id);

  902          encode_path(ebml_w, path);
  903

  905

  907              (ecx.encode_inlined_item)(ecx, ebml_w, IIItemRef(item));
  908          }

  997        ItemStruct(struct_def, _) => {

 1037              Some(ctor_id) => {

 1055          encode_name(ebml_w, item.ident.name);
 1056          encode_attributes(ebml_w, item.attrs.as_slice());
 1057          match ty.node {
 1058              ast::TyPath(ref path, ref bounds, _) if path.segments
 1059                                                          .len() == 1 => {

 1065          for method in imp.methods.iter() {
 1066              ebml_w.start_tag(tag_item_impl_method);

 1068              ebml_w.writer.write(s.as_bytes());
 1069              ebml_w.end_tag();
 1070          }
 1071          for ast_trait_ref in opt_trait.iter() {
 1072              let trait_ref = ty::node_id_to_trait_ref(
 1073                  tcx, ast_trait_ref.ref_id);

 1075              let impl_vtables = ty::lookup_impl_vtables(tcx, def_id);
 1076              encode_impl_vtables(ebml_w, ecx, &impl_vtables);
 1077          }
 1078          encode_path(ebml_w, path.clone());
 1079          ebml_w.end_tag();
 1080  
 1081          // Iterate down the methods, emitting them. We rely on the
 1082          // assumption that all of the actually implemented methods
 1083          // appear first in the impl structure, in the same order they do
 1084          // in the ast. This is a little sketchy.
 1085          let num_implemented_methods = ast_methods.len();
 1086          for (i, m) in imp.methods.iter().enumerate() {
 1087              let ast_method = if i < num_implemented_methods {
 1088                  Some(*ast_methods.get(i))
 1089              } else { None };
 1090  
 1091              index.borrow_mut().push(entry {
 1092                  val: m.def_id.node as i64,
 1093                  pos: ebml_w.writer.tell().unwrap(),
 1094              });
 1095              encode_info_for_method(ecx,
 1096                                     ebml_w,
 1097                                     *m,
 1098                                     path.clone(),
 1099                                     false,
 1100                                     item.id,
 1101                                     ast_method)
 1102          }

 1104        ItemTrait(_, ref super_traits, ref ms) => {

 1123  
 1124              ebml_w.start_tag(tag_mod_child);

 1126              ebml_w.end_tag();
 1127          }
 1128          encode_path(ebml_w, path.clone());
 1129          // FIXME(#8559): This should use the tcx's supertrait cache instead of
 1130          // reading the AST's list, because the former has already filtered out

 1137          // Encode the implementations of this trait.
 1138          encode_extension_implementations(ecx, ebml_w, def_id);
 1139  
 1140          ebml_w.end_tag();

 1143          let r = ty::trait_method_def_ids(tcx, def_id);
 1144          for (i, &method_def_id) in r.iter().enumerate() {
 1145              assert_eq!(method_def_id.krate, ast::LOCAL_CRATE);
 1146

 1150                  val: method_def_id.node as i64,
 1151                  pos: ebml_w.writer.tell().unwrap(),
 1152              });

 1172  
 1173                  _ => {
 1174                      encode_family(ebml_w,
 1175                                    purity_fn_family(
 1176                                        method_ty.fty.purity));
 1177                  }
 1178              }
 1179  
 1180              match ms.get(i) {
 1181                  &Required(ref tm) => {

 1188                      // If this is a static method, we've already encoded
 1189                      // this.

 1191                          // FIXME: I feel like there is something funny going on.
 1192                          let tpt = ty::lookup_item_type(tcx, method_def_id);
 1193                          encode_bounds_and_type(ebml_w, ecx, &tpt);

 1195                      encode_method_sort(ebml_w, 'p');

 1197                          ecx, ebml_w, IIMethodRef(def_id, true, m));
 1198                  }
 1199              }
 1200  
 1201              ebml_w.end_tag();
 1202          }
 1203  
 1204          // Encode inherent implementations for this trait.
 1205          encode_inherent_implementations(ecx, ebml_w, def_id);

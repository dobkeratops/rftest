  317 fn encode_enum_variant_info(ecx: &EncodeContext,
  318                             ebml_w: &mut writer::Encoder,
  319                             id: NodeId,
  320                             variants: &[variant],
  321                             path: &[ast_map::path_elt],
  322                             index: @mut ~[entry<int>],
  323                             generics: &ast::Generics) {
  324     debug!("encode_enum_variant_info(id=%?)", id);
  325 
  326     let mut disr_val = 0;
  327     let mut i = 0;
  328     let vi = ty::enum_variants(ecx.tcx,
  329                                ast::def_id { crate: LOCAL_CRATE, node: id });
  330     for variant in variants.iter() {
  331         let def_id = local_def(variant.node.id);
  332         index.push(entry {val: variant.node.id, pos: ebml_w.writer.tell()});
  333         ebml_w.start_tag(tag_items_data_item);
  334         encode_def_id(ebml_w, def_id);
  335         encode_family(ebml_w, 'v');
  336         encode_name(ecx, ebml_w, variant.node.name);
  337         encode_parent_item(ebml_w, local_def(id));
  338         encode_visibility(ebml_w, variant.node.vis);
  339         match variant.node.kind {
  340             ast::tuple_variant_kind(ref args)
  341                     if args.len() > 0 && generics.ty_params.len() == 0 => {
  342                 encode_symbol(ecx, ebml_w, variant.node.id);
  343             }
  344             ast::tuple_variant_kind(_) | ast::struct_variant_kind(_) => {}
  345         }
  346         encode_discriminant(ecx, ebml_w, variant.node.id);
  347         if vi[i].disr_val != disr_val {
  348             encode_disr_val(ecx, ebml_w, vi[i].disr_val);
  349             disr_val = vi[i].disr_val;
  350         }
  351         encode_bounds_and_type(ebml_w, ecx,
  352                                &lookup_item_type(ecx.tcx, def_id));
  353         encode_path(ecx, ebml_w, path,
  354                     ast_map::path_name(variant.node.name));
  355         ebml_w.end_tag();
  356         disr_val += 1;
  357         i += 1;
  358     }
  359 }

  478 /// Iterates through "auxiliary node IDs", which are node IDs that describe
  479 /// top-level items that are sub-items of the given item. Specifically:
  480 ///
  481 /// * For enums, iterates through the node IDs of the variants.
  482 ///
  483 /// * For newtype structs, iterates through the node ID of the constructor.
  484 fn each_auxiliary_node_id(item: @item, callback: &fn(NodeId) -> bool)
  485                           -> bool {
  486     let mut continue = true;
  487     match item.node {
  488         item_enum(ref enum_def, _) => {
  489             for variant in enum_def.variants.iter() {
  490                 continue = callback(variant.node.id);
  491                 if !continue {
  492                     break
  493                 }
  494             }
  495         }
  496         item_struct(struct_def, _) => {
  497             // If this is a newtype struct, return the constructor.
  498             match struct_def.ctor_id {
  499                 Some(ctor_id) if struct_def.fields.len() > 0 &&
  500                         struct_def.fields[0].node.kind ==
  501                         ast::unnamed_field => {
  502                     continue = callback(ctor_id);
  503                 }
  504                 _ => {}
  505             }
  506         }
  507         _ => {}
  508     }
  509 
  510     continue
  511 }

 1256 fn create_index<T:Clone + Hash + IterBytes + 'static>(
 1257                 index: ~[entry<T>])
 1258                 -> ~[@~[entry<T>]] {
 1259     let mut buckets: ~[@mut ~[entry<T>]] = ~[];
 1260     for _ in range(0u, 256u) { buckets.push(@mut ~[]); };
 1261     for elt in index.iter() {
 1262         let h = elt.val.hash() as uint;
 1263         buckets[h % 256].push((*elt).clone());
 1264     }
 1265 
 1266     let mut buckets_frozen = ~[];
 1267     for bucket in buckets.iter() {
 1268         buckets_frozen.push(@/*bad*/(**bucket).clone());
 1269     }
 1270     return buckets_frozen;
 1271 }

 1273 fn encode_index<T:'static>(
 1274                 ebml_w: &mut writer::Encoder,
 1275                 buckets: ~[@~[entry<T>]],
 1276                 write_fn: &fn(@io::Writer, &T)) {

 1285             ebml_w.start_tag(tag_index_buckets_bucket_elt);
 1286             assert!(elt.pos < 0xffff_ffff);

 1292     }

 1296         assert!(*pos < 0xffff_ffff);

  317  fn encode_enum_variant_info(ecx: &EncodeContext,
  318                              ebml_w: &mut writer::Encoder,
  319                              id: NodeId,
  320                              variants: &[variant],
  321                              path: &[ast_map::path_elt],
  322                              index: @mut ~[entry<i64>],
  323                              generics: &ast::Generics) {
  324      debug!("encode_enum_variant_info(id=%?)", id);
  325  
  326      let mut disr_val = 0;
  327      let mut i = 0;
  328      let vi = ty::enum_variants(ecx.tcx,
  329                                 ast::def_id { crate: LOCAL_CRATE, node: id });
  330      for variant in variants.iter() {
  331          let def_id = local_def(variant.node.id);
  332          index.push(entry {val: variant.node.id as i64,
  333                            pos: ebml_w.writer.tell()});
  334          ebml_w.start_tag(tag_items_data_item);
  335          encode_def_id(ebml_w, def_id);
  336          encode_family(ebml_w, 'v');
  337          encode_name(ecx, ebml_w, variant.node.name);
  338          encode_parent_item(ebml_w, local_def(id));
  339          encode_visibility(ebml_w, variant.node.vis);
  340          match variant.node.kind {
  341              ast::tuple_variant_kind(ref args)
  342                      if args.len() > 0 && generics.ty_params.len() == 0 => {
  343                  encode_symbol(ecx, ebml_w, variant.node.id);
  344              }
  345              ast::tuple_variant_kind(_) | ast::struct_variant_kind(_) => {}
  346          }
  347          encode_discriminant(ecx, ebml_w, variant.node.id);
  348          if vi[i].disr_val != disr_val {
  349              encode_disr_val(ecx, ebml_w, vi[i].disr_val);
  350              disr_val = vi[i].disr_val;
  351          }
  352          encode_bounds_and_type(ebml_w, ecx,
  353                                 &lookup_item_type(ecx.tcx, def_id));
  354          encode_path(ecx, ebml_w, path,
  355                      ast_map::path_name(variant.node.name));
  356          ebml_w.end_tag();
  357          disr_val += 1;
  358          i += 1;
  359      }
  360  }

  479  /// Iterates through "auxiliary node IDs", which are node IDs that describe
  480  /// top-level items that are sub-items of the given item. Specifically:
  481  ///
  482  /// * For enums, iterates through the node IDs of the variants.
  483  ///
  484  /// * For newtype structs, iterates through the node ID of the constructor.
  485  fn each_auxiliary_node_id(item: @item, callback: &fn(NodeId) -> bool)
  486                            -> bool {
  487      let mut continue = true;
  488      match item.node {
  489          item_enum(ref enum_def, _) => {
  490              for variant in enum_def.variants.iter() {
  491                  continue = callback(variant.node.id);
  492                  if !continue {
  493                      break
  494                  }
  495              }
  496          }
  497          item_struct(struct_def, _) => {
  498              // If this is a newtype struct, return the constructor.
  499              match struct_def.ctor_id {
  500                  Some(ctor_id) if struct_def.fields.len() > 0 &&
  501                          struct_def.fields[0].node.kind ==
  502                          ast::unnamed_field => {
  503                      continue = callback(ctor_id);
  504                  }
  505                  _ => {}
  506              }
  507          }
  508          _ => {}
  509      }
  510  
  511      continue
  512  }

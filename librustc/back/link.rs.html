  129     pub fn exec(sess: Session,
  130                 c: ContextRef,
  131                 m: ModuleRef,
  132                 stacks: bool) {
  133         unsafe {
  134             let manager = llvm::LLVMRustPrepareJIT(intrinsics::morestack_addr());
  135 
  136             // We need to tell JIT where to resolve all linked
  137             // symbols from. The equivalent of -lstd, -lcore, etc.
  138             // By default the JIT will resolve symbols from the extra and
  139             // core linked into rustc. We don't want that,
  140             // incase the user wants to use an older extra library.
  141 
  142             let cstore = sess.cstore;
  143             let r = cstore::get_used_crate_files(cstore);
  144             for cratepath in r.iter() {
  145                 let path = cratepath.to_str();
  146 
  147                 debug!("linking: %s", path);
  148 
  149                 do path.as_c_str |buf_t| {
  150                     if !llvm::LLVMRustLoadCrate(manager, buf_t) {
  151                         llvm_err(sess, ~"Could not link");
  152                     }
  153                     debug!("linked: %s", path);
  154                 }
  155             }
  156 
  157             // We custom-build a JIT execution engine via some rust wrappers
  158             // first. This wrappers takes ownership of the module passed in.
  159             let ee = llvm::LLVMRustBuildJIT(manager, m, stacks);
  160             if ee.is_null() {
  161                 llvm::LLVMContextDispose(c);
  162                 llvm_err(sess, ~"Could not create the JIT");
  163             }
  164 
  165             // Next, we need to get a handle on the _rust_main function by
  166             // looking up it's corresponding ValueRef and then requesting that
  167             // the execution engine compiles the function.
  168             let fun = do "_rust_main".as_c_str |entry| {
  169                 llvm::LLVMGetNamedFunction(m, entry)
  170             };
  171             if fun.is_null() {
  172                 llvm::LLVMDisposeExecutionEngine(ee);
  173                 llvm::LLVMContextDispose(c);
  174                 llvm_err(sess, ~"Could not find _rust_main in the JIT");
  175             }
  176 
  177             // Finally, once we have the pointer to the code, we can do some
  178             // closure magic here to turn it straight into a callable rust
  179             // closure
  180             let code = llvm::LLVMGetPointerToGlobal(ee, fun);
  181             assert!(!code.is_null());
  182             let func: extern "Rust" fn() = cast::transmute(code);
  183             func();
  184 
  185             // Currently there is no method of re-using the executing engine
  186             // from LLVM in another call to the JIT. While this kinda defeats
  187             // the purpose of having a JIT in the first place, there isn't
  188             // actually much code currently which would re-use data between
  189             // different invocations of this. Additionally, the compilation
  190             // model currently isn't designed to support this scenario.
  191             //
  192             // We can't destroy the engine/context immediately here, however,
  193             // because of annihilation. The JIT code contains drop glue for any
  194             // types defined in the crate we just ran, and if any of those boxes
  195             // are going to be dropped during annihilation, the drop glue must
  196             // be run. Hence, we need to transfer ownership of this jit engine
  197             // to the caller of this function. To be convenient for now, we
  198             // shove it into TLS and have someone else remove it later on.
  199             let data = ~LLVMJITData { ee: ee, llcx: c };
  200             set_engine(data as ~Engine);
  201         }
  202     }
  203 
  204     // The stage1 compiler won't work, but that doesn't really matter. TLS
  205     // changed only very recently to allow storage of owned values.
  206     static engine_key: local_data::Key<~Engine> = &local_data::Key;
  207 
  208     fn set_engine(engine: ~Engine) {
  209         local_data::set(engine_key, engine)
  210     }
  211 
  212     pub fn consume_engine() -> Option<~Engine> {
  213         local_data::pop(engine_key)
  214     }
  215 }

  245     pub fn run_passes(sess: Session,
  246                       llcx: ContextRef,
  247                       llmod: ModuleRef,
  248                       output_type: output_type,
  249                       output: &Path) {
  250         unsafe {
  251             llvm::LLVMInitializePasses();
  252 
  253             let opts = sess.opts;
  254             if sess.time_llvm_passes() { llvm::LLVMRustEnableTimePasses(); }
  255             let td = mk_target_data(sess.targ_cfg.target_strs.data_layout);
  256             let pm = mk_pass_manager();
  257             llvm::LLVMAddTargetData(td.lltd, pm.llpm);
  258 
  259             // Generate a pre-optimization intermediate file if -save-temps
  260             // was specified.
  261             if opts.save_temps {
  262                 match output_type {
  263                   output_type_bitcode => {
  264                     if opts.optimize != session::No {
  265                         let filename = output.with_filetype("no-opt.bc");
  266                         do filename.to_str().as_c_str |buf| {
  267                             llvm::LLVMWriteBitcodeToFile(llmod, buf);
  268                         }
  269                     }
  270                   }
  271                   _ => {
  272                     let filename = output.with_filetype("bc");
  273                     do filename.to_str().as_c_str |buf| {
  274                         llvm::LLVMWriteBitcodeToFile(llmod, buf);
  275                     }
  276                   }
  277                 }
  278             }
  279 
  280             let mut mpm = passes::PassManager::new(td.lltd);
  281 
  282             if !sess.no_verify() {
  283                 mpm.add_pass_from_name("verify");
  284             }
  285 
  286             let passes = if sess.opts.custom_passes.len() > 0 {
  287                 sess.opts.custom_passes.clone()
  288             } else {
  289                 if sess.lint_llvm() {
  290                     mpm.add_pass_from_name("lint");
  291                 }
  292                 passes::create_standard_passes(opts.optimize)
  293             };
  294 
  295 
  296             debug!("Passes: %?", passes);
  297             passes::populate_pass_manager(sess, &mut mpm, passes);
  298 
  299             debug!("Running Module Optimization Pass");
  300             mpm.run(llmod);
  301 
  302             if opts.jit {
  303                 // If we are using JIT, go ahead and create and execute the
  304                 // engine now.  JIT execution takes ownership of the module and
  305                 // context, so don't dispose and return.
  306                 jit::exec(sess, llcx, llmod, true);
  307 
  308                 if sess.time_llvm_passes() {
  309                     llvm::LLVMRustPrintPassTimings();
  310                 }
  311                 return;
  312             } else if is_object_or_assembly_or_exe(output_type) {
  313                 let LLVMOptNone       = 0 as c_int; // -O0
  314                 let LLVMOptLess       = 1 as c_int; // -O1
  315                 let LLVMOptDefault    = 2 as c_int; // -O2, -Os
  316                 let LLVMOptAggressive = 3 as c_int; // -O3
  317 
  318                 let CodeGenOptLevel = match opts.optimize {
  319                   session::No => LLVMOptNone,
  320                   session::Less => LLVMOptLess,
  321                   session::Default => LLVMOptDefault,
  322                   session::Aggressive => LLVMOptAggressive
  323                 };
  324 
  325                 let FileType = match output_type {
  326                     output_type_object | output_type_exe => lib::llvm::ObjectFile,
  327                     _ => lib::llvm::AssemblyFile
  328                 };
  329 
  330                 // Write optimized bitcode if --save-temps was on.
  331 
  332                 if opts.save_temps {
  333                     // Always output the bitcode file with --save-temps
  334 
  335                     let filename = output.with_filetype("opt.bc");
  336                     do filename.to_str().as_c_str |buf| {
  337                         llvm::LLVMWriteBitcodeToFile(llmod, buf)
  338                     };
  339                     // Save the assembly file if -S is used
  340                     if output_type == output_type_assembly {
  341                         WriteOutputFile(
  342                             sess,
  343                             pm.llpm,
  344                             llmod,
  345                             sess.targ_cfg.target_strs.target_triple,
  346                             opts.target_feature,
  347                             output.to_str(),
  348                             lib::llvm::AssemblyFile as c_uint,
  349                             CodeGenOptLevel,
  350                             true);
  351                     }
  352 
  353                     // Save the object file for -c or --save-temps alone
  354                     // This .o is needed when an exe is built
  355                     if output_type == output_type_object ||
  356                            output_type == output_type_exe {
  357                         WriteOutputFile(
  358                             sess,
  359                             pm.llpm,
  360                             llmod,
  361                             sess.targ_cfg.target_strs.target_triple,
  362                             opts.target_feature,
  363                             output.to_str(),
  364                             lib::llvm::ObjectFile as c_uint,
  365                             CodeGenOptLevel,
  366                             true);
  367                     }
  368                 } else {
  369                     // If we aren't saving temps then just output the file
  370                     // type corresponding to the '-c' or '-S' flag used
  371                     WriteOutputFile(
  372                         sess,
  373                         pm.llpm,
  374                         llmod,
  375                         sess.targ_cfg.target_strs.target_triple,
  376                         opts.target_feature,
  377                         output.to_str(),
  378                         FileType as c_uint,
  379                         CodeGenOptLevel,
  380                         true);
  381                 }
  382                 // Clean up and return
  383 
  384                 llvm::LLVMDisposeModule(llmod);
  385                 llvm::LLVMContextDispose(llcx);
  386                 if sess.time_llvm_passes() {
  387                     llvm::LLVMRustPrintPassTimings();
  388                 }
  389                 return;
  390             }
  391 
  392             if output_type == output_type_llvm_assembly {
  393                 // Given options "-S --emit-llvm": output LLVM assembly
  394                 do output.to_str().as_c_str |buf_o| {
  395                     llvm::LLVMRustAddPrintModulePass(pm.llpm, llmod, buf_o);
  396                 }
  397             } else {
  398                 // If only a bitcode file is asked for by using the
  399                 // '--emit-llvm' flag, then output it here
  400                 do output.to_str().as_c_str |buf| {
  401                     llvm::LLVMWriteBitcodeToFile(llmod, buf);
  402                 }
  403             }
  404 
  405             llvm::LLVMDisposeModule(llmod);
  406             llvm::LLVMContextDispose(llcx);
  407             if sess.time_llvm_passes() { llvm::LLVMRustPrintPassTimings(); }
  408         }
  409     }

  411     pub fn run_ndk(sess: Session, assembly: &Path, object: &Path) {
  412         let cc_prog: ~str = match &sess.opts.android_cross_path {
  413             &Some(ref path) => {
  414                 fmt!("%s/bin/arm-linux-androideabi-gcc", *path)
  415             }
  416             &None => {
  417                 sess.fatal("need Android NDK path for building \
  418                             (--android-cross-path)")
  419             }
  420         };
  421 
  422         let cc_args = ~[
  423             ~"-c",
  424             ~"-o", object.to_str(),
  425             assembly.to_str()];
  426 
  427         let prog = run::process_output(cc_prog, cc_args);
  428 
  429         if prog.status != 0 {
  430             sess.err(fmt!("building with `%s` failed with code %d",
  431                         cc_prog, prog.status));
  432             sess.note(fmt!("%s arguments: %s",
  433                         cc_prog, cc_args.connect(" ")));
  434             sess.note(str::from_bytes(prog.error + prog.output));
  435             sess.abort_if_errors();
  436         }
  437     }

  441 /*
  442  * Name mangling and its relationship to metadata. This is complex. Read
  443  * carefully.
  444  *
  445  * The semantic model of Rust linkage is, broadly, that "there's no global
  446  * namespace" between crates. Our aim is to preserve the illusion of this
  447  * model despite the fact that it's not *quite* possible to implement on
  448  * modern linkers. We initially didn't use system linkers at all, but have
  449  * been convinced of their utility.
  450  *
  451  * There are a few issues to handle:
  452  *
  453  *  - Linkers operate on a flat namespace, so we have to flatten names.
  454  *    We do this using the C++ namespace-mangling technique. Foo::bar
  455  *    symbols and such.
  456  *
  457  *  - Symbols with the same name but different types need to get different
  458  *    linkage-names. We do this by hashing a string-encoding of the type into
  459  *    a fixed-size (currently 16-byte hex) cryptographic hash function (CHF:
  460  *    we use SHA1) to "prevent collisions". This is not airtight but 16 hex
  461  *    digits on uniform probability means you're going to need 2**32 same-name
  462  *    symbols in the same process before you're even hitting birthday-paradox
  463  *    collision probability.
  464  *
  465  *  - Symbols in different crates but with same names "within" the crate need
  466  *    to get different linkage-names.
  467  *
  468  * So here is what we do:
  469  *
  470  *  - Separate the meta tags into two sets: exported and local. Only work with
  471  *    the exported ones when considering linkage.
  472  *
  473  *  - Consider two exported tags as special (and mandatory): name and vers.
  474  *    Every crate gets them; if it doesn't name them explicitly we infer them
  475  *    as basename(crate) and "0.1", respectively. Call these CNAME, CVERS.
  476  *
  477  *  - Define CMETA as all the non-name, non-vers exported meta tags in the
  478  *    crate (in sorted order).
  479  *
  480  *  - Define CMH as hash(CMETA + hashes of dependent crates).
  481  *
  482  *  - Compile our crate to lib CNAME-CMH-CVERS.so
  483  *
  484  *  - Define STH(sym) as hash(CNAME, CMH, type_str(sym))
  485  *
  486  *  - Suffix a mangled sym with ::STH@CVERS, so that it is unique in the
  487  *    name, non-name metadata, and type sense, and versioned in the way
  488  *    system linkers understand.
  489  *
  490  */
  491 
  492 pub fn build_link_meta(sess: Session,
  493                        c: &ast::Crate,
  494                        output: &Path,
  495                        symbol_hasher: &mut hash::State)
  496                        -> LinkMeta {
  497     struct ProvidedMetas {
  498         name: Option<@str>,
  499         vers: Option<@str>,
  500         cmh_items: ~[@ast::MetaItem]
  501     }
  502 
  503     fn provided_link_metas(sess: Session, c: &ast::Crate) ->
  504        ProvidedMetas {
  505         let mut name = None;
  506         let mut vers = None;
  507         let mut cmh_items = ~[];
  508         let linkage_metas = attr::find_linkage_metas(c.attrs);
  509         attr::require_unique_names(sess.diagnostic(), linkage_metas);
  510         for meta in linkage_metas.iter() {
  511             match meta.name_str_pair() {
  512                 Some((n, value)) if "name" == n => name = Some(value),
  513                 Some((n, value)) if "vers" == n => vers = Some(value),
  514                 _ => cmh_items.push(*meta)
  515             }
  516         }
  517 
  518         ProvidedMetas {
  519             name: name,
  520             vers: vers,
  521             cmh_items: cmh_items
  522         }
  523     }
  524 
  525     // This calculates CMH as defined above
  526     fn crate_meta_extras_hash(symbol_hasher: &mut hash::State,
  527                               cmh_items: ~[@ast::MetaItem],
  528                               dep_hashes: ~[@str]) -> @str {
  529         fn len_and_str(s: &str) -> ~str {
  530             fmt!("%u_%s", s.len(), s)
  531         }
  532 
  533         fn len_and_str_lit(l: ast::lit) -> ~str {
  534             len_and_str(pprust::lit_to_str(@l))
  535         }
  536 
  537         let cmh_items = attr::sort_meta_items(cmh_items);
  538 
  539         fn hash(symbol_hasher: &mut hash::State, m: &@ast::MetaItem) {
  540             match m.node {
  541               ast::MetaNameValue(key, value) => {
  542                 write_string(symbol_hasher, len_and_str(key));
  543                 write_string(symbol_hasher, len_and_str_lit(value));
  544               }
  545               ast::MetaWord(name) => {
  546                 write_string(symbol_hasher, len_and_str(name));
  547               }
  548               ast::MetaList(name, ref mis) => {
  549                 write_string(symbol_hasher, len_and_str(name));
  550                 for m_ in mis.iter() {
  551                     hash(symbol_hasher, m_);
  552                 }
  553               }
  554             }
  555         }
  556 
  557         symbol_hasher.reset();
  558         for m in cmh_items.iter() {
  559             hash(symbol_hasher, m);
  560         }
  561 
  562         for dh in dep_hashes.iter() {
  563             write_string(symbol_hasher, len_and_str(*dh));
  564         }
  565 
  566     // tjc: allocation is unfortunate; need to change std::hash
  567         return truncated_hash_result(symbol_hasher).to_managed();
  568     }

  583                 let name = session::expect(sess,

  605     let ProvidedMetas {

  622 }

  629 // This calculates STH for a symbol, as defined above
  630 pub fn symbol_hash(tcx: ty::ctxt,
  631                    symbol_hasher: &mut hash::State,
  632                    t: ty::t,
  633                    link_meta: LinkMeta)
  634                    -> @str {
  635     // NB: do *not* use abbrevs here as we want the symbol names
  636     // to be independent of one another in the crate.
  637 
  638     symbol_hasher.reset();
  639     write_string(symbol_hasher, link_meta.name);
  640     write_string(symbol_hasher, "-");
  641     write_string(symbol_hasher, link_meta.extras_hash);
  642     write_string(symbol_hasher, "-");
  643     write_string(symbol_hasher, encoder::encoded_ty(tcx, t));
  644     let mut hash = truncated_hash_result(symbol_hasher);
  645     // Prefix with _ so that it never blends into adjacent digits
  646     hash.unshift_char('_');
  647     // tjc: allocation is unfortunate; need to change std::hash
  648     hash.to_managed()
  649 }

  651 pub fn get_symbol_hash(ccx: &mut CrateContext, t: ty::t) -> @str {
  652     match ccx.type_hashcodes.find(&t) {
  653       Some(&h) => h,
  654       None => {

  658       }
  659     }
  660 }

  663 // Name sanitation. LLVM will happily accept identifiers with weird names, but
  664 // gas doesn't!
  665 // gas accepts the following characters in symbols: a-z, A-Z, 0-9, ., _, $
  666 pub fn sanitize(s: &str) -> ~str {
  667     let mut result = ~"";
  668     for c in s.iter() {
  669         match c {
  670             // Escape these with $ sequences
  671             '@' => result.push_str("$SP$"),
  672             '~' => result.push_str("$UP$"),
  673             '*' => result.push_str("$RP$"),
  674             '&' => result.push_str("$BP$"),
  675             '<' => result.push_str("$LT$"),
  676             '>' => result.push_str("$GT$"),
  677             '(' => result.push_str("$LP$"),
  678             ')' => result.push_str("$RP$"),
  679             ',' => result.push_str("$C$"),
  680 
  681             // '.' doesn't occur in types and functions, so reuse it
  682             // for ':'
  683             ':' => result.push_char('.'),
  684 
  685             // These are legal symbols
  686             'a' .. 'z'
  687             | 'A' .. 'Z'
  688             | '0' .. '9'
  689             | '_' => result.push_char(c),
  690 
  691             _ => {
  692                 let mut tstr = ~"";
  693                 do char::escape_unicode(c) |c| { tstr.push_char(c); }
  694                 result.push_char('$');
  695                 result.push_str(tstr.slice_from(1));
  696             }
  697         }
  698     }
  699 
  700     // Underscore-qualify anything that didn't start as an ident.
  701     if result.len() > 0u &&
  702         result[0] != '_' as u8 &&
  703         ! char::is_XID_start(result[0] as char) {
  704         return ~"_" + result;
  705     }
  706 
  707     return result;
  708 }

  710 pub fn mangle(sess: Session, ss: path) -> ~str {
  711     // Follow C++ namespace-mangling style
  712 
  713     let mut n = ~"_ZN"; // Begin name-sequence.
  714 
  715     for s in ss.iter() {
  716         match *s {
  717             path_name(s) | path_mod(s) => {
  718                 let sani = sanitize(sess.str_of(s));
  719                 n.push_str(fmt!("%u%s", sani.len(), sani));
  720             }
  721         }
  722     }
  723     n.push_char('E'); // End name-sequence.
  724     n
  725 }

  784 pub fn output_dll_filename(os: session::os, lm: LinkMeta) -> ~str {
  785     let (dll_prefix, dll_suffix) = match os {

  788         session::os_linux => (linux::DLL_PREFIX, linux::DLL_SUFFIX),
  789         session::os_android => (android::DLL_PREFIX, android::DLL_SUFFIX),
  790         session::os_freebsd => (freebsd::DLL_PREFIX, freebsd::DLL_SUFFIX),

  795 // If the user wants an exe generated we need to invoke
  796 // cc to link the object file with some libs
  797 pub fn link_binary(sess: Session,
  798                    obj_filename: &Path,
  799                    out_filename: &Path,
  800                    lm: LinkMeta) {
  801     // In the future, FreeBSD will use clang as default compiler.
  802     // It would be flexible to use cc (system's default C compiler)
  803     // instead of hard-coded gcc.
  804     // For win32, there is no cc command,
  805     // so we add a condition to make it use gcc.
  806     let cc_prog: ~str = match sess.opts.linker {
  807         Some(ref linker) => linker.to_str(),
  808         None => match sess.targ_cfg.os {
  809             session::os_android =>
  810                 match &sess.opts.android_cross_path {
  811                     &Some(ref path) => {
  812                         fmt!("%s/bin/arm-linux-androideabi-gcc", *path)
  813                     }
  814                     &None => {
  815                         sess.fatal("need Android NDK path for linking \
  816                                     (--android-cross-path)")
  817                     }
  818                 },
  819             session::os_win32 => ~"gcc",
  820             _ => ~"cc"
  821         }
  822     };
  823     // The invocations of cc share some flags across platforms
  824 
  825 
  826     let output = if *sess.building_library {
  827         let long_libname = output_dll_filename(sess.targ_cfg.os, lm);
  828         debug!("link_meta.name:  %s", lm.name);
  829         debug!("long_libname: %s", long_libname);
  830         debug!("out_filename: %s", out_filename.to_str());
  831         debug!("dirname(out_filename): %s", out_filename.dir_path().to_str());
  832 
  833         out_filename.dir_path().push(long_libname)
  834     } else {
  835         out_filename.clone()
  836     };
  837 
  838     debug!("output: %s", output.to_str());
  839     let cc_args = link_args(sess, obj_filename, out_filename, lm);
  840     debug!("%s link args: %s", cc_prog, cc_args.connect(" "));
  841     // We run 'cc' here
  842     let prog = run::process_output(cc_prog, cc_args);
  843     if 0 != prog.status {
  844         sess.err(fmt!("linking with `%s` failed with code %d",
  845                       cc_prog, prog.status));
  846         sess.note(fmt!("%s arguments: %s",
  847                        cc_prog, cc_args.connect(" ")));
  848         sess.note(str::from_bytes(prog.error + prog.output));
  849         sess.abort_if_errors();
  850     }
  851 
  852     // Clean up on Darwin
  853     if sess.targ_cfg.os == session::os_macos {
  854         run::process_status("dsymutil", [output.to_str()]);
  855     }
  856 
  857     // Remove the temporary object file if we aren't saving temps
  858     if !sess.opts.save_temps {
  859         if ! os::remove_file(obj_filename) {
  860             sess.warn(fmt!("failed to delete object file `%s`",
  861                            obj_filename.to_str()));
  862         }
  863     }
  864 }

  866 pub fn link_args(sess: Session,
  867                  obj_filename: &Path,
  868                  out_filename: &Path,
  869                  lm:LinkMeta) -> ~[~str] {
  870 
  871     // Converts a library file-stem into a cc -l argument
  872     fn unlib(config: @session::config, stem: ~str) -> ~str {
  873         if stem.starts_with("lib") &&
  874             config.os != session::os_win32 {
  875             stem.slice(3, stem.len()).to_owned()
  876         } else {
  877             stem
  878         }
  879     }
  880 
  881 
  882     let output = if *sess.building_library {
  883         let long_libname = output_dll_filename(sess.targ_cfg.os, lm);
  884         out_filename.dir_path().push(long_libname)
  885     } else {
  886         out_filename.clone()
  887     };

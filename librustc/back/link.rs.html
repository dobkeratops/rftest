  116      pub fn exec(sess: Session,
  117                  c: ContextRef,
  118                  m: ModuleRef,
  119                  stacks: bool) {
  120          unsafe {
  121              let manager = llvm::LLVMRustPrepareJIT(intrinsics::morestack_addr());
  122  
  123              // We need to tell JIT where to resolve all linked
  124              // symbols from. The equivalent of -lstd, -lcore, etc.
  125              // By default the JIT will resolve symbols from the extra and
  126              // core linked into rustc. We don't want that,
  127              // incase the user wants to use an older extra library.
  128  
  129              let cstore = sess.cstore;
  130              let r = cstore::get_used_crate_files(cstore);
  131              for cratepath in r.iter() {
  132                  let path = cratepath.to_str();
  133  
  134                  debug!("linking: %s", path);
  135  
  136                  do path.with_c_str |buf_t| {
  137                      if !llvm::LLVMRustLoadCrate(manager, buf_t) {
  138                          llvm_err(sess, ~"Could not link");
  139                      }
  140                      debug!("linked: %s", path);
  141                  }
  142              }
  143  
  144              // We custom-build a JIT execution engine via some rust wrappers
  145              // first. This wrappers takes ownership of the module passed in.
  146              let ee = llvm::LLVMRustBuildJIT(manager, m, stacks);
  147              if ee.is_null() {
  148                  llvm::LLVMContextDispose(c);
  149                  llvm_err(sess, ~"Could not create the JIT");
  150              }
  151  
  152              // Next, we need to get a handle on the _rust_main function by
  153              // looking up it's corresponding ValueRef and then requesting that
  154              // the execution engine compiles the function.
  155              let fun = do "_rust_main".with_c_str |entry| {
  156                  llvm::LLVMGetNamedFunction(m, entry)
  157              };
  158              if fun.is_null() {
  159                  llvm::LLVMDisposeExecutionEngine(ee);
  160                  llvm::LLVMContextDispose(c);
  161                  llvm_err(sess, ~"Could not find _rust_main in the JIT");
  162              }
  163  
  164              // Finally, once we have the pointer to the code, we can do some
  165              // closure magic here to turn it straight into a callable rust
  166              // closure
  167              let code = llvm::LLVMGetPointerToGlobal(ee, fun);
  168              assert!(!code.is_null());
  169              let func: extern "Rust" fn() = cast::transmute(code);
  170              func();
  171  
  172              // Currently there is no method of re-using the executing engine
  173              // from LLVM in another call to the JIT. While this kinda defeats
  174              // the purpose of having a JIT in the first place, there isn't
  175              // actually much code currently which would re-use data between
  176              // different invocations of this. Additionally, the compilation
  177              // model currently isn't designed to support this scenario.
  178              //
  179              // We can't destroy the engine/context immediately here, however,
  180              // because of annihilation. The JIT code contains drop glue for any
  181              // types defined in the crate we just ran, and if any of those boxes
  182              // are going to be dropped during annihilation, the drop glue must
  183              // be run. Hence, we need to transfer ownership of this jit engine
  184              // to the caller of this function. To be convenient for now, we
  185              // shove it into TLS and have someone else remove it later on.
  186              let data = ~LLVMJITData { ee: ee, llcx: c };
  187              set_engine(data as ~Engine);
  188          }
  189      }
  190  
  191      // The stage1 compiler won't work, but that doesn't really matter. TLS
  192      // changed only very recently to allow storage of owned values.
  193      local_data_key!(engine_key: ~Engine)
  194  
  195      fn set_engine(engine: ~Engine) {
  196          local_data::set(engine_key, engine)
  197      }
  198  
  199      pub fn consume_engine() -> Option<~Engine> {
  200          local_data::pop(engine_key)
  201      }
  202  }

  223      pub fn run_passes(sess: Session,
  224                        llcx: ContextRef,
  225                        llmod: ModuleRef,
  226                        output_type: output_type,
  227                        output: &Path) {
  228          unsafe {
  229              llvm::LLVMInitializePasses();
  230  
  231              // Only initialize the platforms supported by Rust here, because
  232              // using --llvm-root will have multiple platforms that rustllvm
  233              // doesn't actually link to and it's pointless to put target info
  234              // into the registry that Rust can not generate machine code for.
  235              llvm::LLVMInitializeX86TargetInfo();
  236              llvm::LLVMInitializeX86Target();
  237              llvm::LLVMInitializeX86TargetMC();
  238              llvm::LLVMInitializeX86AsmPrinter();
  239              llvm::LLVMInitializeX86AsmParser();
  240  
  241              llvm::LLVMInitializeARMTargetInfo();
  242              llvm::LLVMInitializeARMTarget();
  243              llvm::LLVMInitializeARMTargetMC();
  244              llvm::LLVMInitializeARMAsmPrinter();
  245              llvm::LLVMInitializeARMAsmParser();
  246  
  247              llvm::LLVMInitializeMipsTargetInfo();
  248              llvm::LLVMInitializeMipsTarget();
  249              llvm::LLVMInitializeMipsTargetMC();
  250              llvm::LLVMInitializeMipsAsmPrinter();
  251              llvm::LLVMInitializeMipsAsmParser();
  252  
  253              if sess.opts.save_temps {
  254                  do output.with_filetype("no-opt.bc").with_c_str |buf| {
  255                      llvm::LLVMWriteBitcodeToFile(llmod, buf);
  256                  }
  257              }
  258  
  259              configure_llvm(sess);
  260  
  261              let OptLevel = match sess.opts.optimize {
  262                session::No => lib::llvm::CodeGenLevelNone,
  263                session::Less => lib::llvm::CodeGenLevelLess,
  264                session::Default => lib::llvm::CodeGenLevelDefault,
  265                session::Aggressive => lib::llvm::CodeGenLevelAggressive,
  266              };
  267  
  268              let tm = do sess.targ_cfg.target_strs.target_triple.with_c_str |T| {
  269                  do sess.opts.target_cpu.with_c_str |CPU| {
  270                      do sess.opts.target_feature.with_c_str |Features| {
  271                          llvm::LLVMRustCreateTargetMachine(
  272                              T, CPU, Features,
  273                              lib::llvm::CodeModelDefault,
  274                              lib::llvm::RelocPIC,
  275                              OptLevel,
  276                              true
  277                          )
  278                      }
  279                  }
  280              };
  281  
  282              // Create the two optimizing pass managers. These mirror what clang
  283              // does, and are by populated by LLVM's default PassManagerBuilder.
  284              // Each manager has a different set of passes, but they also share
  285              // some common passes.
  286              let fpm = llvm::LLVMCreateFunctionPassManagerForModule(llmod);
  287              let mpm = llvm::LLVMCreatePassManager();
  288  
  289              // If we're verifying or linting, add them to the function pass
  290              // manager.
  291              let addpass = |pass: &str| {
  292                  do pass.with_c_str |s| { llvm::LLVMRustAddPass(fpm, s) }
  293              };
  294              if !sess.no_verify() { assert!(addpass("verify")); }
  295              if sess.lint_llvm()  { assert!(addpass("lint"));   }
  296  
  297              if !sess.no_prepopulate_passes() {
  298                  llvm::LLVMRustAddAnalysisPasses(tm, fpm, llmod);
  299                  llvm::LLVMRustAddAnalysisPasses(tm, mpm, llmod);
  300                  populate_llvm_passess(fpm, mpm, llmod, OptLevel);
  301              }
  302  
  303              for pass in sess.opts.custom_passes.iter() {
  304                  do pass.with_c_str |s| {
  305                      if !llvm::LLVMRustAddPass(mpm, s) {
  306                          sess.warn(fmt!("Unknown pass %s, ignoring", *pass));
  307                      }
  308                  }
  309              }
  310  
  311              // Finally, run the actual optimization passes
  312              llvm::LLVMRustRunFunctionPassManager(fpm, llmod);
  313              llvm::LLVMRunPassManager(mpm, llmod);
  314  
  315              // Deallocate managers that we're now done with
  316              llvm::LLVMDisposePassManager(fpm);
  317              llvm::LLVMDisposePassManager(mpm);
  318  
  319              if sess.opts.save_temps {
  320                  do output.with_filetype("bc").with_c_str |buf| {
  321                      llvm::LLVMWriteBitcodeToFile(llmod, buf);
  322                  }
  323              }
  324  
  325              if sess.opts.jit {
  326                  // If we are using JIT, go ahead and create and execute the
  327                  // engine now. JIT execution takes ownership of the module and
  328                  // context, so don't dispose
  329                  jit::exec(sess, llcx, llmod, true);
  330              } else {
  331                  // Create a codegen-specific pass manager to emit the actual
  332                  // assembly or object files. This may not end up getting used,
  333                  // but we make it anyway for good measure.
  334                  let cpm = llvm::LLVMCreatePassManager();
  335                  llvm::LLVMRustAddAnalysisPasses(tm, cpm, llmod);
  336                  llvm::LLVMRustAddLibraryInfo(cpm, llmod);
  337  
  338                  match output_type {
  339                      output_type_none => {}
  340                      output_type_bitcode => {
  341                          do output.with_c_str |buf| {
  342                              llvm::LLVMWriteBitcodeToFile(llmod, buf);
  343                          }
  344                      }
  345                      output_type_llvm_assembly => {
  346                          do output.with_c_str |output| {
  347                              llvm::LLVMRustPrintModule(cpm, llmod, output)
  348                          }
  349                      }
  350                      output_type_assembly => {
  351                          WriteOutputFile(sess, tm, cpm, llmod, output.to_str(),
  352                                          lib::llvm::AssemblyFile);
  353                      }
  354                      output_type_exe | output_type_object => {
  355                          WriteOutputFile(sess, tm, cpm, llmod, output.to_str(),
  356                                          lib::llvm::ObjectFile);
  357                      }
  358                  }
  359  
  360                  llvm::LLVMDisposePassManager(cpm);
  361              }
  362  
  363              llvm::LLVMRustDisposeTargetMachine(tm);
  364              // the jit takes ownership of these two items
  365              if !sess.opts.jit {
  366                  llvm::LLVMDisposeModule(llmod);
  367                  llvm::LLVMContextDispose(llcx);
  368              }
  369              if sess.time_llvm_passes() { llvm::LLVMRustPrintPassTimings(); }
  370          }
  371      }

  373      pub fn run_assembler(sess: Session, assembly: &Path, object: &Path) {
  374          let cc_prog = super::get_cc_prog(sess);
  375  
  376          let cc_args = ~[
  377              ~"-c",
  378              ~"-o", object.to_str(),
  379              assembly.to_str()];
  380  
  381          let prog = run::process_output(cc_prog, cc_args);
  382  
  383          if prog.status != 0 {
  384              sess.err(fmt!("building with `%s` failed with code %d",
  385                          cc_prog, prog.status));
  386              sess.note(fmt!("%s arguments: %s",
  387                          cc_prog, cc_args.connect(" ")));
  388              sess.note(str::from_utf8(prog.error + prog.output));
  389              sess.abort_if_errors();
  390          }
  391      }

  393      unsafe fn configure_llvm(sess: Session) {
  394          // Copy what clan does by turning on loop vectorization at O2 and
  395          // slp vectorization at O3
  396          let vectorize_loop = !sess.no_vectorize_loops() &&
  397                               (sess.opts.optimize == session::Default ||
  398                                sess.opts.optimize == session::Aggressive);
  399          let vectorize_slp = !sess.no_vectorize_slp() &&
  400                              sess.opts.optimize == session::Aggressive;
  401  
  402          let mut llvm_c_strs = ~[];
  403          let mut llvm_args = ~[];
  404          let add = |arg: &str| {
  405              let s = arg.to_c_str();
  406              llvm_args.push(s.with_ref(|p| p));
  407              llvm_c_strs.push(s);
  408          };
  409          add("rustc"); // fake program name
  410          add("-arm-enable-ehabi");
  411          add("-arm-enable-ehabi-descriptors");
  412          if vectorize_loop { add("-vectorize-loops"); }
  413          if vectorize_slp  { add("-vectorize-slp");   }
  414          if sess.time_llvm_passes() { add("-time-passes"); }
  415          if sess.print_llvm_passes() { add("-debug-pass=Structure"); }
  416  
  417          for arg in sess.opts.llvm_args.iter() {
  418              add(*arg);
  419          }
  420  
  421          do llvm_args.as_imm_buf |p, len| {
  422              llvm::LLVMRustSetLLVMOptions(len as c_int, p);
  423          }
  424      }

  462  
  463  /*
  464   * Name mangling and its relationship to metadata. This is complex. Read
  465   * carefully.
  466   *
  467   * The semantic model of Rust linkage is, broadly, that "there's no global
  468   * namespace" between crates. Our aim is to preserve the illusion of this
  469   * model despite the fact that it's not *quite* possible to implement on
  470   * modern linkers. We initially didn't use system linkers at all, but have
  471   * been convinced of their utility.
  472   *
  473   * There are a few issues to handle:
  474   *
  475   *  - Linkers operate on a flat namespace, so we have to flatten names.
  476   *    We do this using the C++ namespace-mangling technique. Foo::bar
  477   *    symbols and such.
  478   *
  479   *  - Symbols with the same name but different types need to get different
  480   *    linkage-names. We do this by hashing a string-encoding of the type into
  481   *    a fixed-size (currently 16-byte hex) cryptographic hash function (CHF:
  482   *    we use SHA1) to "prevent collisions". This is not airtight but 16 hex
  483   *    digits on uniform probability means you're going to need 2**32 same-name
  484   *    symbols in the same process before you're even hitting birthday-paradox
  485   *    collision probability.
  486   *
  487   *  - Symbols in different crates but with same names "within" the crate need
  488   *    to get different linkage-names.
  489   *
  490   * So here is what we do:
  491   *
  492   *  - Separate the meta tags into two sets: exported and local. Only work with
  493   *    the exported ones when considering linkage.
  494   *
  495   *  - Consider two exported tags as special (and mandatory): name and vers.
  496   *    Every crate gets them; if it doesn't name them explicitly we infer them
  497   *    as basename(crate) and "0.1", respectively. Call these CNAME, CVERS.
  498   *
  499   *  - Define CMETA as all the non-name, non-vers exported meta tags in the
  500   *    crate (in sorted order).
  501   *
  502   *  - Define CMH as hash(CMETA + hashes of dependent crates).
  503   *
  504   *  - Compile our crate to lib CNAME-CMH-CVERS.so
  505   *
  506   *  - Define STH(sym) as hash(CNAME, CMH, type_str(sym))
  507   *
  508   *  - Suffix a mangled sym with ::STH@CVERS, so that it is unique in the
  509   *    name, non-name metadata, and type sense, and versioned in the way
  510   *    system linkers understand.
  511   *
  512   */
  513  
  514  pub fn build_link_meta(sess: Session,
  515                         c: &ast::Crate,
  516                         output: &Path,
  517                         symbol_hasher: &mut hash::State)
  518                         -> LinkMeta {
  519      struct ProvidedMetas {
  520          name: Option<@str>,
  521          vers: Option<@str>,
  522          pkg_id: Option<@str>,
  523          cmh_items: ~[@ast::MetaItem]
  524      }
  525  
  526      fn provided_link_metas(sess: Session, c: &ast::Crate) ->
  527         ProvidedMetas {
  528          let mut name = None;
  529          let mut vers = None;
  530          let mut pkg_id = None;
  531          let mut cmh_items = ~[];
  532          let linkage_metas = attr::find_linkage_metas(c.attrs);
  533          attr::require_unique_names(sess.diagnostic(), linkage_metas);
  534          for meta in linkage_metas.iter() {
  535              match meta.name_str_pair() {
  536                  Some((n, value)) if "name" == n => name = Some(value),
  537                  Some((n, value)) if "vers" == n => vers = Some(value),
  538                  Some((n, value)) if "package_id" == n => pkg_id = Some(value),
  539                  _ => cmh_items.push(*meta)
  540              }
  541          }
  542  
  543          ProvidedMetas {
  544              name: name,
  545              vers: vers,
  546              pkg_id: pkg_id,
  547              cmh_items: cmh_items
  548          }
  549      }
  550  
  551      // This calculates CMH as defined above
  552      fn crate_meta_extras_hash(symbol_hasher: &mut hash::State,
  553                                cmh_items: ~[@ast::MetaItem],
  554                                dep_hashes: ~[@str],
  555                                pkg_id: Option<@str>) -> @str {
  556          fn len_and_str(s: &str) -> ~str {
  557              fmt!("%u_%s", s.len(), s)
  558          }
  559  
  560          fn len_and_str_lit(l: ast::lit) -> ~str {
  561              len_and_str(pprust::lit_to_str(@l))
  562          }
  563  
  564          let cmh_items = attr::sort_meta_items(cmh_items);
  565  
  566          fn hash(symbol_hasher: &mut hash::State, m: &@ast::MetaItem) {
  567              match m.node {
  568                ast::MetaNameValue(key, value) => {
  569                  write_string(symbol_hasher, len_and_str(key));
  570                  write_string(symbol_hasher, len_and_str_lit(value));
  571                }
  572                ast::MetaWord(name) => {
  573                  write_string(symbol_hasher, len_and_str(name));
  574                }
  575                ast::MetaList(name, ref mis) => {
  576                  write_string(symbol_hasher, len_and_str(name));
  577                  for m_ in mis.iter() {
  578                      hash(symbol_hasher, m_);
  579                  }
  580                }
  581              }
  582          }
  583  
  584          symbol_hasher.reset();
  585          for m in cmh_items.iter() {
  586              hash(symbol_hasher, m);
  587          }
  588  
  589          for dh in dep_hashes.iter() {
  590              write_string(symbol_hasher, len_and_str(*dh));
  591          }
  592  
  593          for p in pkg_id.iter() {
  594              write_string(symbol_hasher, len_and_str(*p));
  595          }
  596  
  597          return truncated_hash_result(symbol_hasher).to_managed();
  598      }

  668                     t: ty::t,

  698  // Name sanitation. LLVM will happily accept identifiers with weird names, but
  699  // gas doesn't!
  700  // gas accepts the following characters in symbols: a-z, A-Z, 0-9, ., _, $
  701  pub fn sanitize(s: &str) -> ~str {
  702      let mut result = ~"";
  703      for c in s.iter() {
  704          match c {
  705              // Escape these with $ sequences
  706              '@' => result.push_str("$SP$"),
  707              '~' => result.push_str("$UP$"),
  708              '*' => result.push_str("$RP$"),
  709              '&' => result.push_str("$BP$"),
  710              '<' => result.push_str("$LT$"),
  711              '>' => result.push_str("$GT$"),
  712              '(' => result.push_str("$LP$"),
  713              ')' => result.push_str("$RP$"),
  714              ',' => result.push_str("$C$"),
  715  
  716              // '.' doesn't occur in types and functions, so reuse it
  717              // for ':'
  718              ':' => result.push_char('.'),
  719  
  720              // These are legal symbols
  721              'a' .. 'z'
  722              | 'A' .. 'Z'
  723              | '0' .. '9'
  724              | '_' | '.' | '$' => result.push_char(c),
  725  
  726              _ => {
  727                  let mut tstr = ~"";
  728                  do char::escape_unicode(c) |c| { tstr.push_char(c); }
  729                  result.push_char('$');
  730                  result.push_str(tstr.slice_from(1));
  731              }
  732          }
  733      }
  734  
  735      // Underscore-qualify anything that didn't start as an ident.
  736      if result.len() > 0u &&
  737          result[0] != '_' as u8 &&
  738          ! char::is_XID_start(result[0] as char) {
  739          return ~"_" + result;
  740      }
  741  
  742      return result;
  743  }

  745  pub fn mangle(sess: Session, ss: path,
  746                hash: Option<&str>, vers: Option<&str>) -> ~str {
  747      // Follow C++ namespace-mangling style, see
  748      // http://en.wikipedia.org/wiki/Name_mangling for more info.
  749      //
  750      // It turns out that on OSX you can actually have arbitrary symbols in
  751      // function names (at least when given to LLVM), but this is not possible
  752      // when using unix's linker. Perhaps one day when we just a linker from LLVM
  753      // we won't need to do this name mangling. The problem with name mangling is
  754      // that it seriously limits the available characters. For example we can't
  755      // have things like @T or ~[T] in symbol names when one would theoretically
  756      // want them for things like impls of traits on that type.
  757      //
  758      // To be able to work on all platforms and get *some* reasonable output, we
  759      // use C++ name-mangling.
  760  
  761      let mut n = ~"_ZN"; // _Z == Begin name-sequence, N == nested
  762  
  763      let push = |s: &str| {
  764          let sani = sanitize(s);
  765          n.push_str(fmt!("%u%s", sani.len(), sani));
  766      };
  767  
  768      // First, connect each component with <len, name> pairs.
  769      for s in ss.iter() {
  770          match *s {
  771              path_name(s) | path_mod(s) | path_pretty_name(s, _) => {
  772                  push(sess.str_of(s))
  773              }
  774          }
  775      }
  776  
  777      // next, if any identifiers are "pretty" and need extra information tacked
  778      // on, then use the hash to generate two unique characters. For now
  779      // hopefully 2 characters is enough to avoid collisions.
  780      static EXTRA_CHARS: &'static str =
  781          "abcdefghijklmnopqrstuvwxyz\
  782           ABCDEFGHIJKLMNOPQRSTUVWXYZ\
  783           0123456789";
  784      let mut hash = match hash { Some(s) => s.to_owned(), None => ~"" };
  785      for s in ss.iter() {
  786          match *s {
  787              path_pretty_name(_, extra) => {
  788                  let hi = (extra >> 32) as u32 as uint;
  789                  let lo = extra as u32 as uint;
  790                  hash.push_char(EXTRA_CHARS[hi % EXTRA_CHARS.len()] as char);
  791                  hash.push_char(EXTRA_CHARS[lo % EXTRA_CHARS.len()] as char);
  792              }
  793              _ => {}
  794          }
  795      }
  796      if hash.len() > 0 {
  797          push(hash);
  798      }
  799      match vers {
  800          Some(s) => push(s),
  801          None => {}
  802      }
  803  
  804      n.push_char('E'); // End name-sequence.
  805      n
  806  }

  808  pub fn exported_name(sess: Session,
  809                       path: path,
  810                       hash: &str,
  811                       vers: &str) -> ~str {
  812      // The version will get mangled to have a leading '_', but it makes more
  813      // sense to lead with a 'v' b/c this is a version...
  814      let vers = if vers.len() > 0 && !char::is_XID_start(vers.char_at(0)) {
  815          "v" + vers
  816      } else {
  817          vers.to_owned()
  818      };
  819  
  820      mangle(sess, path, Some(hash), Some(vers.as_slice()))
  821  }

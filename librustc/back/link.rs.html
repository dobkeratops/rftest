  135      pub fn exec(sess: Session,
  136                  c: ContextRef,
  137                  m: ModuleRef,
  138                  stacks: bool) {
  139          unsafe {
  140              let manager = llvm::LLVMRustPrepareJIT(intrinsics::morestack_addr());
  141  
  142              // We need to tell JIT where to resolve all linked
  143              // symbols from. The equivalent of -lstd, -lcore, etc.
  144              // By default the JIT will resolve symbols from the extra and
  145              // core linked into rustc. We don't want that,
  146              // incase the user wants to use an older extra library.
  147  
  148              let cstore = sess.cstore;
  149              let r = cstore::get_used_crate_files(cstore);
  150              for cratepath in r.iter() {
  151                  let path = cratepath.to_str();
  152  
  153                  debug!("linking: %s", path);
  154  
  155                  do path.to_c_str().with_ref |buf_t| {
  156                      if !llvm::LLVMRustLoadCrate(manager, buf_t) {
  157                          llvm_err(sess, ~"Could not link");
  158                      }
  159                      debug!("linked: %s", path);
  160                  }
  161              }
  162  
  163              // We custom-build a JIT execution engine via some rust wrappers
  164              // first. This wrappers takes ownership of the module passed in.
  165              let ee = llvm::LLVMRustBuildJIT(manager, m, stacks);
  166              if ee.is_null() {
  167                  llvm::LLVMContextDispose(c);
  168                  llvm_err(sess, ~"Could not create the JIT");
  169              }
  170  
  171              // Next, we need to get a handle on the _rust_main function by
  172              // looking up it's corresponding ValueRef and then requesting that
  173              // the execution engine compiles the function.
  174              let fun = do "_rust_main".to_c_str().with_ref |entry| {
  175                  llvm::LLVMGetNamedFunction(m, entry)
  176              };
  177              if fun.is_null() {
  178                  llvm::LLVMDisposeExecutionEngine(ee);
  179                  llvm::LLVMContextDispose(c);
  180                  llvm_err(sess, ~"Could not find _rust_main in the JIT");
  181              }
  182  
  183              // Finally, once we have the pointer to the code, we can do some
  184              // closure magic here to turn it straight into a callable rust
  185              // closure
  186              let code = llvm::LLVMGetPointerToGlobal(ee, fun);
  187              assert!(!code.is_null());
  188              let func: extern "Rust" fn() = cast::transmute(code);
  189              func();
  190  
  191              // Currently there is no method of re-using the executing engine
  192              // from LLVM in another call to the JIT. While this kinda defeats
  193              // the purpose of having a JIT in the first place, there isn't
  194              // actually much code currently which would re-use data between
  195              // different invocations of this. Additionally, the compilation
  196              // model currently isn't designed to support this scenario.
  197              //
  198              // We can't destroy the engine/context immediately here, however,
  199              // because of annihilation. The JIT code contains drop glue for any
  200              // types defined in the crate we just ran, and if any of those boxes
  201              // are going to be dropped during annihilation, the drop glue must
  202              // be run. Hence, we need to transfer ownership of this jit engine
  203              // to the caller of this function. To be convenient for now, we
  204              // shove it into TLS and have someone else remove it later on.
  205              let data = ~LLVMJITData { ee: ee, llcx: c };
  206              set_engine(data as ~Engine);
  207          }
  208      }
  209  
  210      // The stage1 compiler won't work, but that doesn't really matter. TLS
  211      // changed only very recently to allow storage of owned values.
  212      static engine_key: local_data::Key<~Engine> = &local_data::Key;
  213  
  214      fn set_engine(engine: ~Engine) {
  215          local_data::set(engine_key, engine)
  216      }
  217  
  218      pub fn consume_engine() -> Option<~Engine> {
  219          local_data::pop(engine_key)
  220      }
  221  }

  252      pub fn run_passes(sess: Session,
  253                        llcx: ContextRef,
  254                        llmod: ModuleRef,
  255                        output_type: output_type,
  256                        output: &Path) {
  257          unsafe {
  258              llvm::LLVMInitializePasses();
  259  
  260              let opts = sess.opts;
  261              if sess.time_llvm_passes() { llvm::LLVMRustEnableTimePasses(); }
  262              let td = mk_target_data(sess.targ_cfg.target_strs.data_layout);
  263              let pm = mk_pass_manager();
  264              llvm::LLVMAddTargetData(td.lltd, pm.llpm);
  265  
  266              // Generate a pre-optimization intermediate file if -save-temps
  267              // was specified.
  268              if opts.save_temps {
  269                  match output_type {
  270                    output_type_bitcode => {
  271                      if opts.optimize != session::No {
  272                          let filename = output.with_filetype("no-opt.bc");
  273                          do filename.to_c_str().with_ref |buf| {
  274                              llvm::LLVMWriteBitcodeToFile(llmod, buf);
  275                          }
  276                      }
  277                    }
  278                    _ => {
  279                      let filename = output.with_filetype("bc");
  280                      do filename.to_c_str().with_ref |buf| {
  281                          llvm::LLVMWriteBitcodeToFile(llmod, buf);
  282                      }
  283                    }
  284                  }
  285              }
  286  
  287              let mut mpm = passes::PassManager::new(td.lltd);
  288  
  289              if !sess.no_verify() {
  290                  mpm.add_pass_from_name("verify");
  291              }
  292  
  293              let passes = if sess.opts.custom_passes.len() > 0 {
  294                  sess.opts.custom_passes.clone()
  295              } else {
  296                  if sess.lint_llvm() {
  297                      mpm.add_pass_from_name("lint");
  298                  }
  299                  passes::create_standard_passes(opts.optimize)
  300              };
  301  
  302  
  303              debug!("Passes: %?", passes);
  304              passes::populate_pass_manager(sess, &mut mpm, passes);
  305  
  306              debug!("Running Module Optimization Pass");
  307              mpm.run(llmod);
  308  
  309              if opts.jit {
  310                  // If we are using JIT, go ahead and create and execute the
  311                  // engine now.  JIT execution takes ownership of the module and
  312                  // context, so don't dispose and return.
  313                  jit::exec(sess, llcx, llmod, true);
  314  
  315                  if sess.time_llvm_passes() {
  316                      llvm::LLVMRustPrintPassTimings();
  317                  }
  318                  return;
  319              } else if is_object_or_assembly_or_exe(output_type) {
  320                  let LLVMOptNone       = 0 as c_int; // -O0
  321                  let LLVMOptLess       = 1 as c_int; // -O1
  322                  let LLVMOptDefault    = 2 as c_int; // -O2, -Os
  323                  let LLVMOptAggressive = 3 as c_int; // -O3
  324  
  325                  let CodeGenOptLevel = match opts.optimize {
  326                    session::No => LLVMOptNone,
  327                    session::Less => LLVMOptLess,
  328                    session::Default => LLVMOptDefault,
  329                    session::Aggressive => LLVMOptAggressive
  330                  };
  331  
  332                  let FileType = match output_type {
  333                      output_type_object | output_type_exe => lib::llvm::ObjectFile,
  334                      _ => lib::llvm::AssemblyFile
  335                  };
  336  
  337                  // Write optimized bitcode if --save-temps was on.
  338  
  339                  if opts.save_temps {
  340                      // Always output the bitcode file with --save-temps
  341  
  342                      let filename = output.with_filetype("opt.bc");
  343                      do filename.to_c_str().with_ref |buf| {
  344                          llvm::LLVMWriteBitcodeToFile(llmod, buf)
  345                      };
  346                      // Save the assembly file if -S is used
  347                      if output_type == output_type_assembly {
  348                          WriteOutputFile(
  349                              sess,
  350                              pm.llpm,
  351                              llmod,
  352                              sess.targ_cfg.target_strs.target_triple,
  353                              opts.target_cpu,
  354                              opts.target_feature,
  355                              output.to_str(),
  356                              lib::llvm::AssemblyFile as c_uint,
  357                              CodeGenOptLevel,
  358                              true);
  359                      }
  360  
  361                      // Save the object file for -c or --save-temps alone
  362                      // This .o is needed when an exe is built
  363                      if output_type == output_type_object ||
  364                             output_type == output_type_exe {
  365                          WriteOutputFile(
  366                              sess,
  367                              pm.llpm,
  368                              llmod,
  369                              sess.targ_cfg.target_strs.target_triple,
  370                              opts.target_cpu,
  371                              opts.target_feature,
  372                              output.to_str(),
  373                              lib::llvm::ObjectFile as c_uint,
  374                              CodeGenOptLevel,
  375                              true);
  376                      }
  377                  } else {
  378                      // If we aren't saving temps then just output the file
  379                      // type corresponding to the '-c' or '-S' flag used
  380                      WriteOutputFile(
  381                          sess,
  382                          pm.llpm,
  383                          llmod,
  384                          sess.targ_cfg.target_strs.target_triple,
  385                          opts.target_cpu,
  386                          opts.target_feature,
  387                          output.to_str(),
  388                          FileType as c_uint,
  389                          CodeGenOptLevel,
  390                          true);
  391                  }
  392                  // Clean up and return
  393  
  394                  llvm::LLVMDisposeModule(llmod);
  395                  llvm::LLVMContextDispose(llcx);
  396                  if sess.time_llvm_passes() {
  397                      llvm::LLVMRustPrintPassTimings();
  398                  }
  399                  return;
  400              }
  401  
  402              if output_type == output_type_llvm_assembly {
  403                  // Given options "-S --emit-llvm": output LLVM assembly
  404                  do output.to_c_str().with_ref |buf_o| {
  405                      llvm::LLVMRustAddPrintModulePass(pm.llpm, llmod, buf_o);
  406                  }
  407              } else {
  408                  // If only a bitcode file is asked for by using the
  409                  // '--emit-llvm' flag, then output it here
  410                  do output.to_c_str().with_ref |buf| {
  411                      llvm::LLVMWriteBitcodeToFile(llmod, buf);
  412                  }
  413              }
  414  
  415              llvm::LLVMDisposeModule(llmod);
  416              llvm::LLVMContextDispose(llcx);
  417              if sess.time_llvm_passes() { llvm::LLVMRustPrintPassTimings(); }
  418          }
  419      }

  421      pub fn run_ndk(sess: Session, assembly: &Path, object: &Path) {
  422          let cc_prog: ~str = match &sess.opts.android_cross_path {
  423              &Some(ref path) => {
  424                  fmt!("%s/bin/arm-linux-androideabi-gcc", *path)
  425              }
  426              &None => {
  427                  sess.fatal("need Android NDK path for building \
  428                              (--android-cross-path)")
  429              }
  430          };
  431  
  432          let cc_args = ~[
  433              ~"-c",
  434              ~"-o", object.to_str(),
  435              assembly.to_str()];
  436  
  437          let prog = run::process_output(cc_prog, cc_args);
  438  
  439          if prog.status != 0 {
  440              sess.err(fmt!("building with `%s` failed with code %d",
  441                          cc_prog, prog.status));
  442              sess.note(fmt!("%s arguments: %s",
  443                          cc_prog, cc_args.connect(" ")));
  444              sess.note(str::from_bytes(prog.error + prog.output));
  445              sess.abort_if_errors();
  446          }
  447      }
  448  }
  449  
  450  
  451  /*
  452   * Name mangling and its relationship to metadata. This is complex. Read
  453   * carefully.
  454   *
  455   * The semantic model of Rust linkage is, broadly, that "there's no global
  456   * namespace" between crates. Our aim is to preserve the illusion of this
  457   * model despite the fact that it's not *quite* possible to implement on
  458   * modern linkers. We initially didn't use system linkers at all, but have
  459   * been convinced of their utility.
  460   *
  461   * There are a few issues to handle:
  462   *
  463   *  - Linkers operate on a flat namespace, so we have to flatten names.
  464   *    We do this using the C++ namespace-mangling technique. Foo::bar
  465   *    symbols and such.
  466   *
  467   *  - Symbols with the same name but different types need to get different
  468   *    linkage-names. We do this by hashing a string-encoding of the type into
  469   *    a fixed-size (currently 16-byte hex) cryptographic hash function (CHF:
  470   *    we use SHA1) to "prevent collisions". This is not airtight but 16 hex
  471   *    digits on uniform probability means you're going to need 2**32 same-name
  472   *    symbols in the same process before you're even hitting birthday-paradox
  473   *    collision probability.
  474   *
  475   *  - Symbols in different crates but with same names "within" the crate need
  476   *    to get different linkage-names.
  477   *
  478   * So here is what we do:
  479   *
  480   *  - Separate the meta tags into two sets: exported and local. Only work with
  481   *    the exported ones when considering linkage.
  482   *
  483   *  - Consider two exported tags as special (and mandatory): name and vers.
  484   *    Every crate gets them; if it doesn't name them explicitly we infer them
  485   *    as basename(crate) and "0.1", respectively. Call these CNAME, CVERS.
  486   *
  487   *  - Define CMETA as all the non-name, non-vers exported meta tags in the
  488   *    crate (in sorted order).
  489   *
  490   *  - Define CMH as hash(CMETA + hashes of dependent crates).
  491   *
  492   *  - Compile our crate to lib CNAME-CMH-CVERS.so
  493   *
  494   *  - Define STH(sym) as hash(CNAME, CMH, type_str(sym))
  495   *
  496   *  - Suffix a mangled sym with ::STH@CVERS, so that it is unique in the
  497   *    name, non-name metadata, and type sense, and versioned in the way
  498   *    system linkers understand.
  499   *
  500   */
  501  
  502  pub fn build_link_meta(sess: Session,
  503                         c: &ast::Crate,
  504                         output: &Path,
  505                         symbol_hasher: &mut hash::State)
  506                         -> LinkMeta {
  507      struct ProvidedMetas {
  508          name: Option<@str>,
  509          vers: Option<@str>,
  510          pkg_id: Option<@str>,
  511          cmh_items: ~[@ast::MetaItem]
  512      }
  513  
  514      fn provided_link_metas(sess: Session, c: &ast::Crate) ->
  515         ProvidedMetas {
  516          let mut name = None;
  517          let mut vers = None;
  518          let mut pkg_id = None;
  519          let mut cmh_items = ~[];
  520          let linkage_metas = attr::find_linkage_metas(c.attrs);
  521          attr::require_unique_names(sess.diagnostic(), linkage_metas);
  522          for meta in linkage_metas.iter() {
  523              match meta.name_str_pair() {
  524                  Some((n, value)) if "name" == n => name = Some(value),
  525                  Some((n, value)) if "vers" == n => vers = Some(value),
  526                  Some((n, value)) if "package_id" == n => pkg_id = Some(value),
  527                  _ => cmh_items.push(*meta)
  528              }
  529          }
  530  
  531          ProvidedMetas {
  532              name: name,
  533              vers: vers,
  534              pkg_id: pkg_id,
  535              cmh_items: cmh_items
  536          }
  537      }
  538  
  539      // This calculates CMH as defined above
  540      fn crate_meta_extras_hash(symbol_hasher: &mut hash::State,
  541                                cmh_items: ~[@ast::MetaItem],
  542                                dep_hashes: ~[@str],
  543                                pkg_id: Option<@str>) -> @str {
  544          fn len_and_str(s: &str) -> ~str {
  545              fmt!("%u_%s", s.len(), s)
  546          }
  547  
  548          fn len_and_str_lit(l: ast::lit) -> ~str {
  549              len_and_str(pprust::lit_to_str(@l))
  550          }
  551  
  552          let cmh_items = attr::sort_meta_items(cmh_items);
  553  
  554          fn hash(symbol_hasher: &mut hash::State, m: &@ast::MetaItem) {
  555              match m.node {
  556                ast::MetaNameValue(key, value) => {
  557                  write_string(symbol_hasher, len_and_str(key));
  558                  write_string(symbol_hasher, len_and_str_lit(value));
  559                }
  560                ast::MetaWord(name) => {
  561                  write_string(symbol_hasher, len_and_str(name));
  562                }
  563                ast::MetaList(name, ref mis) => {
  564                  write_string(symbol_hasher, len_and_str(name));
  565                  for m_ in mis.iter() {
  566                      hash(symbol_hasher, m_);
  567                  }
  568                }
  569              }
  570          }
  571  
  572          symbol_hasher.reset();
  573          for m in cmh_items.iter() {
  574              hash(symbol_hasher, m);
  575          }
  576  
  577          for dh in dep_hashes.iter() {
  578              write_string(symbol_hasher, len_and_str(*dh));
  579          }
  580  
  581          for p in pkg_id.iter() {
  582              write_string(symbol_hasher, len_and_str(*p));
  583          }
  584  
  585          return truncated_hash_result(symbol_hasher).to_managed();
  586      }

  650  pub fn symbol_hash(tcx: ty::ctxt,
  651                     symbol_hasher: &mut hash::State,
  652                     t: ty::t,
  653                     link_meta: LinkMeta)
  654                     -> @str {

  666      hash.unshift_char('_');

  679      }

  683  // Name sanitation. LLVM will happily accept identifiers with weird names, but
  684  // gas doesn't!
  685  // gas accepts the following characters in symbols: a-z, A-Z, 0-9, ., _, $
  686  pub fn sanitize(s: &str) -> ~str {
  687      let mut result = ~"";
  688      for c in s.iter() {
  689          match c {
  690              // Escape these with $ sequences
  691              '@' => result.push_str("$SP$"),
  692              '~' => result.push_str("$UP$"),
  693              '*' => result.push_str("$RP$"),
  694              '&' => result.push_str("$BP$"),
  695              '<' => result.push_str("$LT$"),
  696              '>' => result.push_str("$GT$"),
  697              '(' => result.push_str("$LP$"),
  698              ')' => result.push_str("$RP$"),
  699              ',' => result.push_str("$C$"),
  700  
  701              // '.' doesn't occur in types and functions, so reuse it
  702              // for ':'
  703              ':' => result.push_char('.'),
  704  
  705              // These are legal symbols
  706              'a' .. 'z'
  707              | 'A' .. 'Z'
  708              | '0' .. '9'
  709              | '_' => result.push_char(c),
  710  
  711              _ => {
  712                  let mut tstr = ~"";
  713                  do char::escape_unicode(c) |c| { tstr.push_char(c); }
  714                  result.push_char('$');
  715                  result.push_str(tstr.slice_from(1));
  716              }
  717          }
  718      }
  719  
  720      // Underscore-qualify anything that didn't start as an ident.
  721      if result.len() > 0u &&
  722          result[0] != '_' as u8 &&
  723          ! char::is_XID_start(result[0] as char) {
  724          return ~"_" + result;
  725      }
  726  
  727      return result;
  728  }

  730  pub fn mangle(sess: Session, ss: path) -> ~str {
  731      // Follow C++ namespace-mangling style
  732  
  733      let mut n = ~"_ZN"; // Begin name-sequence.
  734  
  735      for s in ss.iter() {
  736          match *s {
  737              path_name(s) | path_mod(s) => {
  738                  let sani = sanitize(sess.str_of(s));
  739                  n.push_str(fmt!("%u%s", sani.len(), sani));
  740              }
  741          }
  742      }
  743      n.push_char('E'); // End name-sequence.
  744      n
  745  }

  815  // If the user wants an exe generated we need to invoke
  816  // cc to link the object file with some libs
  817  pub fn link_binary(sess: Session,
  818                     obj_filename: &Path,
  819                     out_filename: &Path,
  820                     lm: LinkMeta) {

  822      // It would be flexible to use cc (system's default C compiler)
  823      // instead of hard-coded gcc.
  824      // For win32, there is no cc command,
  825      // so we add a condition to make it use gcc.
  826      let cc_prog: ~str = match sess.opts.linker {
  827          Some(ref linker) => linker.to_str(),
  828          None => match sess.targ_cfg.os {
  829              session::os_android =>
  830                  match &sess.opts.android_cross_path {
  831                      &Some(ref path) => {
  832                          fmt!("%s/bin/arm-linux-androideabi-gcc", *path)
  833                      }
  834                      &None => {
  835                          sess.fatal("need Android NDK path for linking \
  836                                      (--android-cross-path)")
  837                      }
  838                  },
  839              session::os_win32 => ~"gcc",
  840              _ => ~"cc"
  841          }
  842      };
  843      // The invocations of cc share some flags across platforms
  844  
  845  
  846      let output = if *sess.building_library {
  847          let long_libname = output_dll_filename(sess.targ_cfg.os, lm);
  848          debug!("link_meta.name:  %s", lm.name);
  849          debug!("long_libname: %s", long_libname);
  850          debug!("out_filename: %s", out_filename.to_str());

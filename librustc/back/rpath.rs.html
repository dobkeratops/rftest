  26 pub fn get_rpath_flags(sess: session::Session, out_filename: &Path)
  27                     -> ~[~str] {
  28     let os = sess.targ_cfg.os;
  29 
  30     // No rpath on windows
  31     if os == session::os_win32 {
  32         return ~[];
  33     }
  34 
  35     debug!("preparing the RPATH!");
  36 
  37     let sysroot = sess.filesearch.sysroot();
  38     let output = out_filename;
  39     let libs = cstore::get_used_crate_files(sess.cstore);
  40     // We don't currently rpath extern libraries, but we know
  41     // where rustrt is and we know every rust program needs it
  42     let libs = vec::append_one(libs, get_sysroot_absolute_rt_lib(sess));
  43 
  44     let rpaths = get_rpaths(os, sysroot, output, libs,
  45                             sess.opts.target_triple);
  46     rpaths_to_flags(rpaths)
  47 }

  58 fn get_rpaths(os: session::os,
  59               sysroot: &Path,
  60               output: &Path,
  61               libs: &[Path],
  62               target_triple: &str) -> ~[Path] {
  63     debug!("sysroot: %s", sysroot.to_str());
  64     debug!("output: %s", output.to_str());
  65     debug!("libs:");
  66     for libpath in libs.iter() {
  67         debug!("    %s", libpath.to_str());
  68     }
  69     debug!("target_triple: %s", target_triple);
  70 
  71     // Use relative paths to the libraries. Binaries can be moved
  72     // as long as they maintain the relative relationship to the
  73     // crates they depend on.
  74     let rel_rpaths = get_rpaths_relative_to_output(os, output, libs);
  75 
  76     // Make backup absolute paths to the libraries. Binaries can
  77     // be moved as long as the crates they link against don't move.
  78     let abs_rpaths = get_absolute_rpaths(libs);
  79 
  80     // And a final backup rpath to the global library location.
  81     let fallback_rpaths = ~[get_install_prefix_rpath(target_triple)];
  82 
  83     fn log_rpaths(desc: &str, rpaths: &[Path]) {
  84         debug!("%s rpaths:", desc);
  85         for rpath in rpaths.iter() {
  86             debug!("    %s", rpath.to_str());
  87         }
  88     }
  89 
  90     log_rpaths("relative", rel_rpaths);
  91     log_rpaths("absolute", abs_rpaths);
  92     log_rpaths("fallback", fallback_rpaths);
  93 
  94     let mut rpaths = rel_rpaths;
  95     rpaths.push_all(abs_rpaths);
  96     rpaths.push_all(fallback_rpaths);
  97 
  98     // Remove duplicates
  99     let rpaths = minimize_rpaths(rpaths);
 100     return rpaths;
 101 }

 109 pub fn get_rpath_relative_to_output(os: session::os,
 110                                     output: &Path,
 111                                     lib: &Path)
 112                                  -> Path {
 113     use std::os;
 114 
 115     assert!(not_win32(os));
 116 
 117     // Mac doesn't appear to support $ORIGIN
 118     let prefix = match os {
 119         session::os_android | session::os_linux | session::os_freebsd
 120                           => "$ORIGIN",
 121         session::os_macos => "@executable_path",
 122         session::os_win32 => util::unreachable()
 123     };
 124 
 125     Path(prefix).push_rel(&get_relative_to(&os::make_absolute(output),
 126                                            &os::make_absolute(lib)))
 127 }

 129 // Find the relative path from one file to another
 130 pub fn get_relative_to(abs1: &Path, abs2: &Path) -> Path {
 131     assert!(abs1.is_absolute);
 132     assert!(abs2.is_absolute);
 133     let abs1 = abs1.normalize();
 134     let abs2 = abs2.normalize();
 135     debug!("finding relative path from %s to %s",
 136            abs1.to_str(), abs2.to_str());
 137     let split1: &[~str] = abs1.components;
 138     let split2: &[~str] = abs2.components;
 139     let len1 = split1.len();
 140     let len2 = split2.len();
 141     assert!(len1 > 0);
 142     assert!(len2 > 0);
 143 
 144     let max_common_path = num::min(len1, len2) - 1;
 145     let mut start_idx = 0;
 146     while start_idx < max_common_path
 147         && split1[start_idx] == split2[start_idx] {
 148         start_idx += 1;
 149     }
 150 
 151     let mut path = ~[];
 152     for _ in range(start_idx, len1 - 1) { path.push(~".."); };
 153 
 154     path.push_all(split2.slice(start_idx, len2 - 1));
 155 
 156     return if !path.is_empty() {
 157         Path("").push_many(path)
 158     } else {
 159         Path(".")
 160     }
 161 }

 171 pub fn get_install_prefix_rpath(target_triple: &str) -> Path {
 172     let install_prefix = env!("CFG_PREFIX");
 173 
 174     if install_prefix == "" {
 175         fail!("rustc compiled without CFG_PREFIX environment variable");
 176     }
 177 
 178     let tlib = filesearch::relative_target_lib_path(target_triple);
 179     os::make_absolute(&Path(install_prefix).push_rel(&tlib))
 180 }

 182 pub fn minimize_rpaths(rpaths: &[Path]) -> ~[Path] {
 183     let mut set = HashSet::new();
 184     let mut minimized = ~[];
 185     for rpath in rpaths.iter() {
 186         if set.insert(rpath.to_str()) {
 187             minimized.push(rpath.clone());
 188         }
 189     }
 190     minimized
 191 }

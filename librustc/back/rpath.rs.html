  24  pub fn get_rpath_flags(sess: &Session, out_filename: &Path) -> Vec<~str> {
  25      let os = sess.targ_cfg.os;
  26  
  27      // No rpath on windows
  28      if os == abi::OsWin32 {
  29          return Vec::new();
  30      }
  31  
  32      let mut flags = Vec::new();
  33  
  34      if sess.targ_cfg.os == abi::OsFreebsd {
  35          flags.push_all([~"-Wl,-rpath,/usr/local/lib/gcc46",
  36                          ~"-Wl,-rpath,/usr/local/lib/gcc44",
  37                          ~"-Wl,-z,origin"]);
  38      }
  39  
  40      debug!("preparing the RPATH!");
  41  
  42      let sysroot = sess.filesearch().sysroot;
  43      let output = out_filename;
  44      let libs = sess.cstore.get_used_crates(cstore::RequireDynamic);
  45      let libs = libs.move_iter().filter_map(|(_, l)| l.map(|p| p.clone())).collect();
  46      // We don't currently rpath extern libraries, but we know
  47      // where rustrt is and we know every rust program needs it

  49

  51                              sess.opts.target_triple);
  52      flags.push_all(rpaths_to_flags(rpaths.as_slice()).as_slice());
  53      flags
  54  }

  72  fn get_rpaths(os: abi::Os,
  73                sysroot: &Path,
  74                output: &Path,
  75                libs: &[Path],
  76                target_triple: &str) -> Vec<~str> {
  77      debug!("sysroot: {}", sysroot.display());
  78      debug!("output: {}", output.display());
  79      debug!("libs:");
  80      for libpath in libs.iter() {
  81          debug!("    {}", libpath.display());
  82      }
  83      debug!("target_triple: {}", target_triple);
  84  
  85      // Use relative paths to the libraries. Binaries can be moved
  86      // as long as they maintain the relative relationship to the
  87      // crates they depend on.
  88      let rel_rpaths = get_rpaths_relative_to_output(os, output, libs);
  89  
  90      // Make backup absolute paths to the libraries. Binaries can
  91      // be moved as long as the crates they link against don't move.
  92      let abs_rpaths = get_absolute_rpaths(libs);
  93  
  94      // And a final backup rpath to the global library location.
  95      let fallback_rpaths = vec!(get_install_prefix_rpath(sysroot, target_triple));
  96  
  97      fn log_rpaths(desc: &str, rpaths: &[~str]) {
  98          debug!("{} rpaths:", desc);
  99          for rpath in rpaths.iter() {
 100              debug!("    {}", *rpath);
 101          }
 102      }
 103  
 104      log_rpaths("relative", rel_rpaths.as_slice());
 105      log_rpaths("absolute", abs_rpaths.as_slice());
 106      log_rpaths("fallback", fallback_rpaths.as_slice());
 107  
 108      let mut rpaths = rel_rpaths;
 109      rpaths.push_all(abs_rpaths.as_slice());
 110      rpaths.push_all(fallback_rpaths.as_slice());
 111  
 112      // Remove duplicates
 113      let rpaths = minimize_rpaths(rpaths.as_slice());
 114      return rpaths;
 115  }

 123  pub fn get_rpath_relative_to_output(os: abi::Os,
 124                                      output: &Path,
 125                                      lib: &Path)
 126                                   -> ~str {
 127      use std::os;
 128  
 129      assert!(not_win32(os));
 130  
 131      // Mac doesn't appear to support $ORIGIN
 132      let prefix = match os {
 133          abi::OsAndroid | abi::OsLinux | abi::OsFreebsd
 134                            => "$ORIGIN",
 135          abi::OsMacos => "@loader_path",
 136          abi::OsWin32 => unreachable!()
 137      };
 138  
 139      let mut lib = os::make_absolute(lib);
 140      lib.pop();
 141      let mut output = os::make_absolute(output);
 142      output.pop();
 143      let relative = lib.path_relative_from(&output);
 144      let relative = relative.expect("could not create rpath relative to output");
 145      // FIXME (#9639): This needs to handle non-utf8 paths
 146      prefix+"/"+relative.as_str().expect("non-utf8 component in path")
 147  }

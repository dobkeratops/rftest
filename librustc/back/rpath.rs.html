  23  pub fn get_rpath_flags(sess: session::Session, out_filename: &Path)
  24                      -> ~[~str] {
  25      let os = sess.targ_cfg.os;
  26  
  27      // No rpath on windows
  28      if os == session::os_win32 {
  29          return ~[];
  30      }
  31  
  32      debug!("preparing the RPATH!");
  33  
  34      let sysroot = sess.filesearch.sysroot();
  35      let output = out_filename;
  36      let libs = cstore::get_used_crate_files(sess.cstore);
  37      // We don't currently rpath extern libraries, but we know
  38      // where rustrt is and we know every rust program needs it
  39      let libs = vec::append_one(libs, get_sysroot_absolute_rt_lib(sess));
  40  
  41      let rpaths = get_rpaths(os, sysroot, output, libs,
  42                              sess.opts.target_triple);
  43      rpaths_to_flags(rpaths)
  44  }

  55  fn get_rpaths(os: session::os,
  56                sysroot: &Path,
  57                output: &Path,
  58                libs: &[Path],
  59                target_triple: &str) -> ~[Path] {
  60      debug!("sysroot: %s", sysroot.to_str());
  61      debug!("output: %s", output.to_str());
  62      debug!("libs:");
  63      for libpath in libs.iter() {
  64          debug!("    %s", libpath.to_str());
  65      }
  66      debug!("target_triple: %s", target_triple);
  67  
  68      // Use relative paths to the libraries. Binaries can be moved
  69      // as long as they maintain the relative relationship to the
  70      // crates they depend on.
  71      let rel_rpaths = get_rpaths_relative_to_output(os, output, libs);
  72  
  73      // Make backup absolute paths to the libraries. Binaries can
  74      // be moved as long as the crates they link against don't move.
  75      let abs_rpaths = get_absolute_rpaths(libs);
  76  
  77      // And a final backup rpath to the global library location.
  78      let fallback_rpaths = ~[get_install_prefix_rpath(target_triple)];
  79  
  80      fn log_rpaths(desc: &str, rpaths: &[Path]) {
  81          debug!("%s rpaths:", desc);
  82          for rpath in rpaths.iter() {
  83              debug!("    %s", rpath.to_str());
  84          }
  85      }
  86  
  87      log_rpaths("relative", rel_rpaths);
  88      log_rpaths("absolute", abs_rpaths);
  89      log_rpaths("fallback", fallback_rpaths);
  90  
  91      let mut rpaths = rel_rpaths;
  92      rpaths.push_all(abs_rpaths);
  93      rpaths.push_all(fallback_rpaths);
  94  
  95      // Remove duplicates
  96      let rpaths = minimize_rpaths(rpaths);
  97      return rpaths;
  98  }

 106  pub fn get_rpath_relative_to_output(os: session::os,
 107                                      output: &Path,
 108                                      lib: &Path)
 109                                   -> Path {
 110      use std::os;
 111  
 112      assert!(not_win32(os));
 113  
 114      // Mac doesn't appear to support $ORIGIN
 115      let prefix = match os {
 116          session::os_android | session::os_linux | session::os_freebsd
 117                            => "$ORIGIN",
 118          session::os_macos => "@executable_path",
 119          session::os_win32 => util::unreachable()
 120      };
 121  
 122      Path(prefix).push_rel(&os::make_absolute(output).get_relative_to(&os::make_absolute(lib)))
 123  }

 153  pub fn minimize_rpaths(rpaths: &[Path]) -> ~[Path] {
 154      let mut set = HashSet::new();
 155      let mut minimized = ~[];
 156      for rpath in rpaths.iter() {
 157          if set.insert(rpath.to_str()) {
 158              minimized.push(rpath.clone());
 159          }
 160      }
 161      minimized
 162  }

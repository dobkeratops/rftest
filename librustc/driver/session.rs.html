 357  /// Declare a macro that will define all CodegenOptions fields and parsers all
 358  /// at once. The goal of this macro is to define an interface that can be
 359  /// programmatically used by the option parser in order to initialize the struct
 360  /// without hardcoding field names all over the place.
 361  ///
 362  /// The goal is to invoke this macro once with the correct fields, and then this
 363  /// macro generates all necessary code. The main gotcha of this macro is the
 364  /// cgsetters module which is a bunch of generated code to parse an option into
 365  /// its respective field in the struct. There are a few hand-written parsers for
 366  /// parsing specific types of values in this module.
 367  macro_rules! cgoptions(
 368      ($($opt:ident : $t:ty = ($init:expr, $parse:ident, $desc:expr)),* ,) =>
 369  (
 370      #[deriving(Clone)]
 371      pub struct CodegenOptions { $(pub $opt: $t),* }
 372  
 373      pub fn basic_codegen_options() -> CodegenOptions {
 374          CodegenOptions { $($opt: $init),* }
 375      }

 379                                        &'static str)] =
 380          &[ $( (stringify!($opt), cgsetters::$opt, $desc) ),* ];

 388              }

 391          fn parse_bool(slot: &mut bool, v: Option<&str>) -> bool {
 392              match v {
 393                  Some(..) => false,
 394                  None => { *slot = true; true }
 395              }
 396          }

 398          fn parse_opt_string(slot: &mut Option<~str>, v: Option<&str>) -> bool {
 399              match v {
 400                  Some(s) => { *slot = Some(s.to_owned()); true },
 401                  None => false,
 402              }
 403          }
 404  
 405          fn parse_string(slot: &mut ~str, v: Option<&str>) -> bool {
 406              match v {
 407                  Some(s) => { *slot = s.to_owned(); true },
 408                  None => false,
 409              }
 410          }
 411  
 412          fn parse_list(slot: &mut Vec<~str>, v: Option<&str>)
 413                        -> bool {
 414              match v {
 415                  Some(s) => {
 416                      for s in s.words() {
 417                          slot.push(s.to_owned());
 418                      }
 419                      true
 420                  },
 421                  None => false,
 422              }
 423          }
 424  
 425      }
 426  ) )

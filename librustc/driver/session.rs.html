 359  /// Declare a macro that will define all CodegenOptions fields and parsers all
 360  /// at once. The goal of this macro is to define an interface that can be
 361  /// programmatically used by the option parser in order to initialize the struct
 362  /// without hardcoding field names all over the place.
 363  ///
 364  /// The goal is to invoke this macro once with the correct fields, and then this
 365  /// macro generates all necessary code. The main gotcha of this macro is the
 366  /// cgsetters module which is a bunch of generated code to parse an option into
 367  /// its respective field in the struct. There are a few hand-written parsers for
 368  /// parsing specific types of values in this module.
 369  macro_rules! cgoptions(
 370      ($($opt:ident : $t:ty = ($init:expr, $parse:ident, $desc:expr)),* ,) =>
 371  (
 372      #[deriving(Clone)]
 373      pub struct CodegenOptions { $(pub $opt: $t),* }
 374  
 375      pub fn basic_codegen_options() -> CodegenOptions {
 376          CodegenOptions { $($opt: $init),* }
 377      }

 403  pub fn building_library(req_crate_type: crate_type,
 404                          crate: &ast::Crate,
 405                          testing: bool) -> bool {
 406      match req_crate_type {
 407        bin_crate => false,
 408        lib_crate => true,
 409        unknown_crate => {
 410          if testing {
 411              false
 412          } else {
 413              match syntax::attr::first_attr_value_str_by_name(

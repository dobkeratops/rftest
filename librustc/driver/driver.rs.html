   66  pub fn default_configuration(sess: Session) ->
   67     ast::CrateConfig {
   68      let tos = match sess.targ_cfg.os {
   69          session::OsWin32 =>   @"win32",
   70          session::OsMacos =>   @"macos",
   71          session::OsLinux =>   @"linux",
   72          session::OsAndroid => @"android",
   73          session::OsFreebsd => @"freebsd"
   74      };
   75  
   76      // ARM is bi-endian, however using NDK seems to default
   77      // to little-endian unless a flag is provided.
   78      let (end,arch,wordsz) = match sess.targ_cfg.arch {
   79          abi::X86 =>    (@"little", @"x86",    @"32"),
   80          abi::X86_64 => (@"little", @"x86_64", @"64"),
   81          abi::Arm =>    (@"little", @"arm",    @"32"),
   82          abi::Mips =>   (@"big",    @"mips",   @"32")
   83      };
   84  
   85      let mk = attr::mk_name_value_item_str;
   86      return ~[ // Target bindings.
   87           attr::mk_word_item(os::FAMILY.to_managed()),
   88           mk(@"target_os", tos),
   89           mk(@"target_family", os::FAMILY.to_managed()),
   90           mk(@"target_arch", arch),
   91           mk(@"target_endian", end),
   92           mk(@"target_word_size", wordsz),
   93      ];
   94  }

  102  pub fn build_configuration(sess: Session) ->
  103     ast::CrateConfig {
  104      // Combine the configuration requested by the session (command line) with
  105      // some default and generated configuration items
  106      let default_cfg = default_configuration(sess);
  107      let mut user_cfg = sess.opts.cfg.clone();
  108      // If the user wants a test runner, then add the test cfg
  109      if sess.opts.test { append_configuration(&mut user_cfg, @"test") }
  110      // If the user requested GC, then add the GC cfg
  111      append_configuration(&mut user_cfg, if sess.opts.gc { @"gc" } else { @"nogc" });
  112      return vec::append(user_cfg, default_cfg);
  113  }

  132  pub fn phase_1_parse_input(sess: Session, cfg: ast::CrateConfig, input: &input)
  133      -> @ast::Crate {
  134      time(sess.time_passes(), ~"parsing", || {
  135          match *input {
  136              file_input(ref file) => {
  137                  parse::parse_crate_from_file(&(*file), cfg.clone(), sess.parse_sess)
  138              }
  139              str_input(src) => {
  140                  parse::parse_crate_from_source_str(
  141                      anon_src(), src, cfg.clone(), sess.parse_sess)
  142              }
  143          }
  144      })
  145  }

  147  // For continuing compilation after a parsed crate has been
  148  // modified
  149  
  150  /// Run the "early phases" of the compiler: initial `cfg` processing,
  151  /// syntax expansion, secondary `cfg` expansion, synthesis of a test
  152  /// harness if one is to be provided and injection of a dependency on the
  153  /// standard library and prelude.
  154  pub fn phase_2_configure_and_expand(sess: Session,
  155                                      cfg: ast::CrateConfig,
  156                                      mut crate: @ast::Crate) -> @ast::Crate {
  157      let time_passes = sess.time_passes();
  158  
  159      *sess.building_library = session::building_library(sess.opts.crate_type,
  160                                                         crate, sess.opts.test);
  161  
  162  
  163      // strip before expansion to allow macros to depend on
  164      // configuration variables e.g/ in
  165      //
  166      //   #[macro_escape] #[cfg(foo)]
  167      //   mod bar { macro_rules! baz!(() => {{}}) }
  168      //
  169      // baz! should not use this definition unless foo is enabled.
  170      crate = time(time_passes, ~"std macros injection", ||
  171                   syntax::ext::expand::inject_std_macros(sess.parse_sess,
  172                                                          cfg.clone(),
  173                                                          crate));
  174  
  175      crate = time(time_passes, ~"configuration 1", ||
  176                   front::config::strip_unconfigured_items(crate));
  177  
  178      crate = time(time_passes, ~"expansion", ||
  179                   syntax::ext::expand::expand_crate(sess.parse_sess, cfg.clone(),
  180                                                     crate));
  181  
  182      // strip again, in case expansion added anything with a #[cfg].
  183      crate = time(time_passes, ~"configuration 2", ||
  184                   front::config::strip_unconfigured_items(crate));
  185  
  186      crate = time(time_passes, ~"maybe building test harness", ||
  187                   front::test::modify_for_testing(sess, crate));
  188  
  189      crate = time(time_passes, ~"std injection", ||
  190                   front::std_inject::maybe_inject_libstd_ref(sess, crate));
  191  
  192      crate = time(time_passes, ~"assigning node ids", ||
  193                   front::assign_node_ids::assign_node_ids(sess, crate));
  194  
  195      return crate;
  196  }

  205  /// Run the resolution, typechecking, region checking and other
  206  /// miscellaneous analysis passes on the crate. Return various
  207  /// structures carrying the results of the analysis.
  208  pub fn phase_3_run_analysis_passes(sess: Session,
  209                                     crate: @ast::Crate) -> CrateAnalysis {
  210  
  211      let time_passes = sess.time_passes();
  212  
  213      let ast_map = time(time_passes, ~"ast indexing", ||
  214                         syntax::ast_map::map_crate(sess.diagnostic(), crate));
  215  
  216      time(time_passes, ~"external crate/lib resolution", ||
  217           creader::read_crates(sess.diagnostic(), crate, sess.cstore,
  218                                sess.filesearch,
  219                                session::sess_os_to_meta_os(sess.targ_cfg.os),
  220                                sess.opts.is_static,
  221                                token::get_ident_interner()));
  222  
  223      let lang_items = time(time_passes, ~"language item collection", ||
  224                            middle::lang_items::collect_language_items(crate, sess));
  225  
  226      let middle::resolve::CrateMap {
  227          def_map: def_map,
  228          exp_map2: exp_map2,
  229          trait_map: trait_map
  230      } =
  231          time(time_passes, ~"resolution", ||
  232               middle::resolve::resolve_crate(sess, lang_items, crate));
  233  
  234      time(time_passes, ~"looking for entry point",
  235           || middle::entry::find_entry_point(sess, crate, ast_map));
  236  
  237      let freevars = time(time_passes, ~"freevar finding", ||
  238                          freevars::annotate_freevars(def_map, crate));
  239  
  240      let region_map = time(time_passes, ~"region resolution", ||
  241                            middle::region::resolve_crate(sess, def_map, crate));
  242  
  243      let rp_set = time(time_passes, ~"region parameterization inference", ||
  244                        middle::region::determine_rp_in_crate(sess, ast_map, def_map, crate));
  245  
  246      let ty_cx = ty::mk_ctxt(sess, def_map, ast_map, freevars,
  247                              region_map, rp_set, lang_items);
  248  
  249      // passes are timed inside typeck
  250      let (method_map, vtable_map) = typeck::check_crate(
  251          ty_cx, trait_map, crate);
  252  
  253      // These next two const passes can probably be merged
  254      time(time_passes, ~"const marking", ||
  255           middle::const_eval::process_crate(crate, ty_cx));
  256  
  257      time(time_passes, ~"const checking", ||
  258           middle::check_const::check_crate(sess, crate, ast_map, def_map,
  259                                            method_map, ty_cx));
  260  
  261      time(time_passes, ~"privacy checking", ||
  262           middle::privacy::check_crate(ty_cx, &method_map, crate));
  263  
  264      time(time_passes, ~"effect checking", ||
  265           middle::effect::check_crate(ty_cx, method_map, crate));
  266  
  267      time(time_passes, ~"loop checking", ||
  268           middle::check_loop::check_crate(ty_cx, crate));
  269  
  270      time(time_passes, ~"stack checking", ||
  271           middle::stack_check::stack_check_crate(ty_cx, crate));
  272  
  273      let middle::moves::MoveMaps {moves_map, moved_variables_set,
  274                                   capture_map} =
  275          time(time_passes, ~"compute moves", ||
  276               middle::moves::compute_moves(ty_cx, method_map, crate));
  277  
  278      time(time_passes, ~"match checking", ||
  279           middle::check_match::check_crate(ty_cx, method_map,
  280                                            moves_map, crate));
  281  
  282      time(time_passes, ~"liveness checking", ||
  283           middle::liveness::check_crate(ty_cx, method_map,
  284                                         capture_map, crate));
  285  
  286      let (root_map, write_guard_map) =
  287          time(time_passes, ~"borrow checking", ||
  288               middle::borrowck::check_crate(ty_cx, method_map,
  289                                             moves_map, moved_variables_set,
  290                                             capture_map, crate));
  291  
  292      time(time_passes, ~"kind checking", ||
  293           kind::check_crate(ty_cx, method_map, crate));
  294  
  295      let reachable_map =
  296          time(time_passes, ~"reachability checking", ||
  297               reachable::find_reachable(ty_cx, method_map, crate));
  298  
  299      time(time_passes, ~"lint checking", ||
  300           lint::check_crate(ty_cx, crate));
  301  
  302      CrateAnalysis {
  303          exp_map2: exp_map2,
  304          ty_cx: ty_cx,
  305          maps: astencode::Maps {
  306              root_map: root_map,
  307              method_map: method_map,
  308              vtable_map: vtable_map,
  309              write_guard_map: write_guard_map,
  310              capture_map: capture_map
  311          },
  312          reachable: reachable_map
  313      }
  314  }

  333  /// Run LLVM itself, producing a bitcode file, assembly file or object file
  334  /// as a side effect.
  335  pub fn phase_5_run_llvm_passes(sess: Session,
  336                                 trans: &CrateTranslation,
  337                                 outputs: &OutputFilenames) {
  338  
  339      // On Windows, LLVM integrated assembler emits bad stack unwind tables when
  340      // segmented stacks are enabled.  However, unwind info directives in assembly
  341      // output are OK, so we generate assembly first and then run it through
  342      // an external assembler.
  343      if sess.targ_cfg.os == session::OsWin32 &&
  344          (sess.opts.output_type == link::output_type_object ||
  345           sess.opts.output_type == link::output_type_exe) {
  346          let output_type = link::output_type_assembly;
  347          let asm_filename = outputs.obj_filename.with_filetype("s");
  348  
  349          time(sess.time_passes(), ~"LLVM passes", ||
  350              link::write::run_passes(sess,
  351                                      trans.context,
  352                                      trans.module,
  353                                      output_type,
  354                                      &asm_filename));
  355  
  356          link::write::run_assembler(sess, &asm_filename, &outputs.obj_filename);
  357  
  358          // Remove assembly source unless --save-temps was specified
  359          if !sess.opts.save_temps {
  360              os::remove_file(&asm_filename);
  361          }
  362      } else {
  363          time(sess.time_passes(), ~"LLVM passes", ||
  364              link::write::run_passes(sess,
  365                                      trans.context,
  366                                      trans.module,
  367                                      sess.opts.output_type,
  368                                      &outputs.obj_filename));
  369      }
  370  }
  371  
  372  /// Run the linker on any artifacts that resulted from the LLVM run.

  374  pub fn phase_6_link_output(sess: Session,

  380                             &outputs.out_filename,
  381                             trans.link));
  382  }

  418  #[fixed_stack_segment]
  419  pub fn compile_input(sess: Session, cfg: ast::CrateConfig, input: &input,
  420                       outdir: &Option<Path>, output: &Option<Path>) {
  421      // We need nested scopes here, because the intermediate results can keep
  422      // large chunks of memory alive and we want to free them as soon as
  423      // possible to keep the peak memory usage low
  424      let (outputs, trans) = {
  425          let expanded_crate = {
  426              let crate = phase_1_parse_input(sess, cfg.clone(), input);
  427              if stop_after_phase_1(sess) { return; }
  428              phase_2_configure_and_expand(sess, cfg, crate)
  429          };
  430          let analysis = phase_3_run_analysis_passes(sess, expanded_crate);
  431          if stop_after_phase_3(sess) { return; }
  432          let outputs = build_output_filenames(input, outdir, output, [], sess);
  433          let trans = phase_4_translate_to_llvm(sess, expanded_crate,
  434                                                &analysis, outputs);
  435          (outputs, trans)
  436      };
  437      phase_5_run_llvm_passes(sess, &trans, outputs);
  438      if stop_after_phase_5(sess) { return; }
  439      phase_6_link_output(sess, &trans, outputs);
  440  }

  627  pub fn host_triple() -> ~str {
  628      // Get the host triple out of the build environment. This ensures that our
  629      // idea of the host triple is the same as for the set of libraries we've
  630      // actually built.  We can't just take LLVM's host triple because they
  631      // normalize all ix86 architectures to i386.
  632      //
  633      // Instead of grabbing the host triple (for the current host), we grab (at
  634      // compile time) the target triple that this rustc is built with and
  635      // calling that (at runtime) the host triple.
  636      (env!("CFG_COMPILER_TRIPLE")).to_owned()
  637  }

  639  pub fn build_session_options(binary: @str,
  640                               matches: &getopts::Matches,
  641                               demitter: @diagnostic::Emitter)
  642                               -> @session::options {
  643      let crate_type = if matches.opt_present("lib") {
  644          session::lib_crate
  645      } else if matches.opt_present("bin") {
  646          session::bin_crate
  647      } else {
  648          session::unknown_crate
  649      };
  650      let parse_only = matches.opt_present("parse-only");
  651      let no_trans = matches.opt_present("no-trans");
  652  
  653      let lint_levels = [lint::allow, lint::warn,
  654                         lint::deny, lint::forbid];
  655      let mut lint_opts = ~[];
  656      let lint_dict = lint::get_lint_dict();
  657      for level in lint_levels.iter() {
  658          let level_name = lint::level_to_str(*level);
  659  
  660          // FIXME: #4318 Instead of to_ascii and to_str_ascii, could use
  661          // to_ascii_move and to_str_move to not do a unnecessary copy.
  662          let level_short = level_name.slice_chars(0, 1);
  663          let level_short = level_short.to_ascii().to_upper().to_str_ascii();
  664          let flags = vec::append(matches.opt_strs(level_short),
  665                                  matches.opt_strs(level_name));
  666          for lint_name in flags.iter() {
  667              let lint_name = lint_name.replace("-", "_");
  668              match lint_dict.find_equiv(&lint_name) {
  669                None => {
  670                  early_error(demitter, fmt!("unknown %s flag: %s",
  671                                             level_name, lint_name));
  672                }
  673                Some(lint) => {
  674                  lint_opts.push((lint.lint, *level));
  675                }
  676              }
  677          }
  678      }
  679  
  680      let mut debugging_opts = 0u;
  681      let debug_flags = matches.opt_strs("Z");
  682      let debug_map = session::debugging_opts_map();
  683      for debug_flag in debug_flags.iter() {
  684          let mut this_bit = 0u;
  685          for tuple in debug_map.iter() {
  686              let (name, bit) = match *tuple { (ref a, _, b) => (a, b) };
  687              if name == debug_flag { this_bit = bit; break; }
  688          }
  689          if this_bit == 0u {
  690              early_error(demitter, fmt!("unknown debug flag: %s", *debug_flag))
  691          }
  692          debugging_opts |= this_bit;
  693      }
  694  
  695      if debugging_opts & session::debug_llvm != 0 {
  696          set_llvm_debug();
  697  
  698          fn set_llvm_debug() {
  699              #[fixed_stack_segment]; #[inline(never)];
  700              unsafe { llvm::LLVMSetDebug(1); }
  701          }
  702      }
  703  
  704      let output_type =
  705          if parse_only || no_trans {
  706              link::output_type_none
  707          } else if matches.opt_present("S") &&
  708                    matches.opt_present("emit-llvm") {
  709              link::output_type_llvm_assembly
  710          } else if matches.opt_present("S") {
  711              link::output_type_assembly
  712          } else if matches.opt_present("c") {
  713              link::output_type_object
  714          } else if matches.opt_present("emit-llvm") {
  715              link::output_type_bitcode
  716          } else { link::output_type_exe };
  717      let sysroot_opt = matches.opt_str("sysroot").map_move(|m| @Path(m));
  718      let target = matches.opt_str("target").unwrap_or(host_triple());
  719      let target_cpu = matches.opt_str("target-cpu").unwrap_or(~"generic");
  720      let target_feature = matches.opt_str("target-feature").unwrap_or(~"");
  721      let save_temps = matches.opt_present("save-temps");
  722      let opt_level = {
  723          if (debugging_opts & session::no_opt) != 0 {
  724              No
  725          } else if matches.opt_present("O") {
  726              if matches.opt_present("opt-level") {
  727                  early_error(demitter, ~"-O and --opt-level both provided");
  728              }
  729              Default
  730          } else if matches.opt_present("opt-level") {
  731              match matches.opt_str("opt-level").unwrap() {
  732                ~"0" => No,
  733                ~"1" => Less,
  734                ~"2" => Default,
  735                ~"3" => Aggressive,
  736                _ => {
  737                  early_error(demitter, ~"optimization level needs to be between 0-3")
  738                }
  739              }
  740          } else { No }
  741      };
  742      let gc = debugging_opts & session::gc != 0;
  743      let jit = debugging_opts & session::jit != 0;
  744      let extra_debuginfo = debugging_opts & session::extra_debug_info != 0;
  745      let debuginfo = debugging_opts & session::debug_info != 0 ||
  746          extra_debuginfo;
  747  
  748      // If debugging info is generated, do not collapse monomorphized function instances.
  749      // Functions with equivalent llvm code still need separate debugging descriptions because names
  750      // might differ.
  751      if debuginfo {
  752          debugging_opts |= session::no_monomorphic_collapse;
  753      }
  754  
  755      let statik = debugging_opts & session::statik != 0;
  756  
  757      let addl_lib_search_paths = matches.opt_strs("L").map(|s| Path(*s));
  758      let linker = matches.opt_str("linker");
  759      let linker_args = matches.opt_strs("link-args").flat_map( |a| {
  760          a.split_iter(' ').map(|arg| arg.to_owned()).collect()
  761      });
  762  
  763      let cfg = parse_cfgspecs(matches.opt_strs("cfg"), demitter);
  764      let test = matches.opt_present("test");
  765      let android_cross_path = matches.opt_str("android-cross-path");
  766  
  767      let custom_passes = match matches.opt_str("passes") {
  768          None => ~[],
  769          Some(s) => {
  770              s.split_iter(|c: char| c == ' ' || c == ',').map(|s| {
  771                  s.trim().to_owned()
  772              }).collect()
  773          }
  774      };
  775      let llvm_args = match matches.opt_str("llvm-args") {
  776          None => ~[],
  777          Some(s) => {
  778              s.split_iter(|c: char| c == ' ' || c == ',').map(|s| {
  779                  s.trim().to_owned()
  780              }).collect()
  781          }
  782      };
  783  
  784      let sopts = @session::options {
  785          crate_type: crate_type,
  786          is_static: statik,
  787          gc: gc,
  788          optimize: opt_level,
  789          custom_passes: custom_passes,
  790          llvm_args: llvm_args,
  791          debuginfo: debuginfo,
  792          extra_debuginfo: extra_debuginfo,
  793          lint_opts: lint_opts,
  794          save_temps: save_temps,
  795          jit: jit,
  796          output_type: output_type,
  797          addl_lib_search_paths: @mut addl_lib_search_paths,
  798          linker: linker,
  799          linker_args: linker_args,
  800          maybe_sysroot: sysroot_opt,
  801          target_triple: target,
  802          target_cpu: target_cpu,
  803          target_feature: target_feature,
  804          cfg: cfg,
  805          binary: binary,
  806          test: test,
  807          parse_only: parse_only,
  808          no_trans: no_trans,
  809          debugging_opts: debugging_opts,
  810          android_cross_path: android_cross_path
  811      };
  812      return sopts;
  813  }

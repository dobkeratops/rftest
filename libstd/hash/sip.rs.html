  11  /*!
  12   * Implementation of SipHash 2-4
  13   *
  14   * See: http://131002.net/siphash/
  15   *
  16   * Consider this as a main "general-purpose" hash for all hashtables: it
  17   * runs at good speed (competitive with spooky and city) and permits
  18   * strong _keyed_ hashing. Key your hashtables from a strong RNG,
  19   * such as `rand::Rng`.
  20   *
  21   * Although the SipHash algorithm is considered to be cryptographically
  22   * strong, this implementation has not been reviewed for such purposes.
  23   * As such, all cryptographic uses of this implementation are strongly
  24   * discouraged.
  25   */

  37  /// `SipState` computes a SipHash 2-4 hash over a stream of bytes.
  38  pub struct SipState {
  39      k0: u64,
  40      k1: u64,
  41      length: uint, // how many bytes we've processed
  42      v0: u64,      // hash state
  43      v1: u64,
  44      v2: u64,
  45      v3: u64,
  46      tail: [u8, ..8], // unprocessed bytes
  47      ntail: uint,  // how many bytes in tail are valid
  48  }
  49  
  50  // sadly, these macro definitions can't appear later,
  51  // because they're needed in the following defs;
  52  // this design could be improved.
  53  
  54  macro_rules! u8to64_le (
  55      ($buf:expr, $i:expr) =>
  56      ($buf[0+$i] as u64 |
  57       $buf[1+$i] as u64 << 8 |
  58       $buf[2+$i] as u64 << 16 |
  59       $buf[3+$i] as u64 << 24 |
  60       $buf[4+$i] as u64 << 32 |
  61       $buf[5+$i] as u64 << 40 |
  62       $buf[6+$i] as u64 << 48 |
  63       $buf[7+$i] as u64 << 56)
  64  )
  65  
  66  macro_rules! rotl (
  67      ($x:expr, $b:expr) =>
  68      (($x << $b) | ($x >> (64 - $b)))
  69  )
  70  
  71  macro_rules! compress (
  72      ($v0:expr, $v1:expr, $v2:expr, $v3:expr) =>
  73      ({
  74          $v0 += $v1; $v1 = rotl!($v1, 13); $v1 ^= $v0;
  75          $v0 = rotl!($v0, 32);
  76          $v2 += $v3; $v3 = rotl!($v3, 16); $v3 ^= $v2;
  77          $v0 += $v3; $v3 = rotl!($v3, 21); $v3 ^= $v0;
  78          $v2 += $v1; $v1 = rotl!($v1, 17); $v1 ^= $v2;
  79          $v2 = rotl!($v2, 32);
  80      })
  81  )

  83  impl SipState {
  84      /// Create a `SipState` that is keyed off the provided keys.
  85      #[inline]
  86      pub fn new() -> SipState {
  87          SipState::new_with_keys(0, 0)
  88      }
  89  
  90      /// Create a `SipState` that is keyed off the provided keys.
  91      #[inline]
  92      pub fn new_with_keys(key0: u64, key1: u64) -> SipState {
  93          let mut state = SipState {
  94              k0: key0,
  95              k1: key1,
  96              length: 0,
  97              v0: 0,
  98              v1: 0,
  99              v2: 0,
 100              v3: 0,
 101              tail: [ 0, 0, 0, 0, 0, 0, 0, 0 ],
 102              ntail: 0,
 103          };
 104          state.reset();
 105          state
 106      }
 107  
 108      /// Reset the state back to it's initial state.
 109      #[inline]
 110      pub fn reset(&mut self) {
 111          self.length = 0;
 112          self.v0 = self.k0 ^ 0x736f6d6570736575;
 113          self.v1 = self.k1 ^ 0x646f72616e646f6d;
 114          self.v2 = self.k0 ^ 0x6c7967656e657261;
 115          self.v3 = self.k1 ^ 0x7465646279746573;
 116          self.ntail = 0;
 117      }
 118  
 119      /// Return the computed hash.
 120      #[inline]
 121      pub fn result(&self) -> u64 {
 122          let mut v0 = self.v0;
 123          let mut v1 = self.v1;
 124          let mut v2 = self.v2;
 125          let mut v3 = self.v3;
 126  
 127          let mut b : u64 = (self.length as u64 & 0xff) << 56;
 128  
 129          if self.ntail > 0 { b |= self.tail[0] as u64 <<  0; }
 130          if self.ntail > 1 { b |= self.tail[1] as u64 <<  8; }
 131          if self.ntail > 2 { b |= self.tail[2] as u64 << 16; }
 132          if self.ntail > 3 { b |= self.tail[3] as u64 << 24; }
 133          if self.ntail > 4 { b |= self.tail[4] as u64 << 32; }
 134          if self.ntail > 5 { b |= self.tail[5] as u64 << 40; }
 135          if self.ntail > 6 { b |= self.tail[6] as u64 << 48; }
 136  
 137          v3 ^= b;
 138          compress!(v0, v1, v2, v3);
 139          compress!(v0, v1, v2, v3);
 140          v0 ^= b;
 141  
 142          v2 ^= 0xff;
 143          compress!(v0, v1, v2, v3);
 144          compress!(v0, v1, v2, v3);
 145          compress!(v0, v1, v2, v3);
 146          compress!(v0, v1, v2, v3);
 147  
 148          v0 ^ v1 ^ v2 ^ v3
 149      }
 150  }

 152  impl Writer for SipState {
 153      #[inline]
 154      fn write(&mut self, msg: &[u8]) -> IoResult<()> {
 155          let length = msg.len();
 156          self.length += length;
 157  
 158          let mut needed = 0u;
 159  
 160          if self.ntail != 0 {
 161              needed = 8 - self.ntail;
 162  
 163              if length < needed {
 164                  let mut t = 0;
 165                  while t < length {
 166                      self.tail[self.ntail+t] = msg[t];
 167                      t += 1;
 168                  }
 169                  self.ntail += length;
 170                  return Ok(());
 171              }
 172  
 173              let mut t = 0;
 174              while t < needed {
 175                  self.tail[self.ntail+t] = msg[t];
 176                  t += 1;
 177              }
 178  
 179              let m = u8to64_le!(self.tail, 0);
 180  
 181              self.v3 ^= m;
 182              compress!(self.v0, self.v1, self.v2, self.v3);
 183              compress!(self.v0, self.v1, self.v2, self.v3);
 184              self.v0 ^= m;
 185  
 186              self.ntail = 0;
 187          }
 188  
 189          // Buffered tail is now flushed, process new input.
 190          let len = length - needed;
 191          let end = len & (!0x7);
 192          let left = len & 0x7;
 193  
 194          let mut i = needed;
 195          while i < end {
 196              let mi = u8to64_le!(msg, i);
 197  
 198              self.v3 ^= mi;
 199              compress!(self.v0, self.v1, self.v2, self.v3);
 200              compress!(self.v0, self.v1, self.v2, self.v3);
 201              self.v0 ^= mi;
 202  
 203              i += 8;
 204          }
 205  
 206          let mut t = 0u;
 207          while t < left {
 208              self.tail[t] = msg[i+t];
 209              t += 1
 210          }
 211          self.ntail = left;
 212  
 213          Ok(())
 214      }
 215  }

 231  /// `SipHasher` computes the SipHash algorithm from a stream of bytes.
 232  #[deriving(Clone)]
 233  pub struct SipHasher {
 234      k0: u64,
 235      k1: u64,
 236  }
 237  
 238  impl SipHasher {
 239      /// Create a `Sip`.
 240      #[inline]
 241      pub fn new() -> SipHasher {
 242          SipHasher::new_with_keys(0, 0)
 243      }
 244  
 245      /// Create a `Sip` that is keyed off the provided keys.
 246      #[inline]
 247      pub fn new_with_keys(key0: u64, key1: u64) -> SipHasher {
 248          SipHasher {
 249              k0: key0,
 250              k1: key1,
 251          }
 252      }
 253  }
 254  
 255  impl Hasher<SipState> for SipHasher {
 256      #[inline]
 257      fn hash<T: Hash<SipState>>(&self, value: &T) -> u64 {
 258          let mut state = SipState::new_with_keys(self.k0, self.k1);
 259          value.hash(&mut state);
 260          state.result()
 261      }
 262  }

 267          SipHasher::new()
 268      }
 269  }

 271  /// Hash a value using the SipHash algorithm.
 272  #[inline]
 273  pub fn hash<T: Hash<SipState>>(value: &T) -> u64 {
 274      let mut state = SipState::new();
 275      value.hash(&mut state);
 276      state.result()
 277  }

 279  /// Hash a value with the SipHash algorithm with the provided keys.

 282      let mut state = SipState::new_with_keys(k0, k1);
 283      value.hash(&mut state);
 284      state.result()
 285  }

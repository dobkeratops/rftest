  24 /**
  25  * Trait for visitor that wishes to reflect on data. To use this, create a
  26  * struct that encapsulates the set of pointers you wish to walk through a
  27  * data structure, and implement both `MovePtr` for it as well as `TyVisitor`;
  28  * then build a MovePtrAdaptor wrapped around your struct.
  29  */
  30 pub trait MovePtr {
  31     fn move_ptr(&self, adjustment: &fn(*c_void) -> *c_void);
  32     fn push_ptr(&self);
  33     fn pop_ptr(&self);
  34 }

  36 /// Helper function for alignment calculation.
  37 #[inline]
  38 pub fn align(size: uint, align: uint) -> uint {
  39     ((size + align) - 1u) & !(align - 1u)
  40 }

  42 /// Adaptor to wrap around visitors implementing MovePtr.

  44     inner: V

  46 pub fn MovePtrAdaptor<V:TyVisitor + MovePtr>(v: V) -> MovePtrAdaptor<V> {
  47     MovePtrAdaptor { inner: v }
  48 }

  50 impl<V:TyVisitor + MovePtr> MovePtrAdaptor<V> {
  51     #[inline]
  52     pub fn bump(&self, sz: uint) {
  53         do self.inner.move_ptr() |p| {
  54             ((p as uint) + sz) as *c_void
  55         };
  56     }
  57 
  58     #[inline]
  59     pub fn align(&self, a: uint) {
  60         do self.inner.move_ptr() |p| {
  61             align(p as uint, a) as *c_void
  62         };
  63     }

  65     #[inline]
  66     pub fn align_to<T>(&self) {

  70     #[inline]
  71     pub fn bump_past<T>(&self) {

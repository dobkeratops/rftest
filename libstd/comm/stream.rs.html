  11  /// Stream channels
  12  ///
  13  /// This is the flavor of channels which are optimized for one sender and one
  14  /// receiver. The sender will be upgraded to a shared channel if the channel is
  15  /// cloned.
  16  ///
  17  /// High level implementation details can be found in the comment of the parent
  18  /// module.

  40  pub struct Packet<T> {

  89      pub fn send(&mut self, t: T) -> Result<(), T> {
  90          // If the other port has deterministically gone away, then definitely
  91          // must return the data back up the stack. Otherwise, the data is
  92          // considered as being sent.
  93          if self.port_dropped.load(atomics::SeqCst) { return Err(t) }
  94  
  95          match self.do_send(Data(t)) {
  96              UpSuccess | UpDisconnected => {},
  97              UpWoke(task) => { task.wake().map(|t| t.reawaken()); }
  98          }
  99          Ok(())
 100      }
 101      pub fn upgrade(&mut self, up: Receiver<T>) -> UpgradeResult {
 102          // If the port has gone away, then there's no need to proceed any
 103          // further.
 104          if self.port_dropped.load(atomics::SeqCst) { return UpDisconnected }
 105  
 106          self.do_send(GoUp(up))
 107      }
 108  
 109      fn do_send(&mut self, t: Message<T>) -> UpgradeResult {
 110          self.queue.push(t);
 111          match self.cnt.fetch_add(1, atomics::SeqCst) {
 112              // As described in the mod's doc comment, -1 == wakeup
 113              -1 => UpWoke(self.take_to_wake()),
 114              // As as described before, SPSC queues must be >= -2
 115              -2 => UpSuccess,
 116  
 117              // Be sure to preserve the disconnected state, and the return value
 118              // in this case is going to be whether our data was received or not.
 119              // This manifests itself on whether we have an empty queue or not.
 120              //
 121              // Primarily, are required to drain the queue here because the port
 122              // will never remove this data. We can only have at most one item to
 123              // drain (the port drains the rest).
 124              DISCONNECTED => {
 125                  self.cnt.store(DISCONNECTED, atomics::SeqCst);
 126                  let first = self.queue.pop();
 127                  let second = self.queue.pop();
 128                  assert!(second.is_none());
 129  
 130                  match first {
 131                      Some(..) => UpSuccess,  // we failed to send the data
 132                      None => UpDisconnected, // we successfully sent data
 133                  }
 134              }
 135  
 136              // Otherwise we just sent some data on a non-waiting queue, so just
 137              // make sure the world is sane and carry on!
 138              n => { assert!(n >= 0); UpSuccess }
 139          }
 140      }
 141  
 142      // Consumes ownership of the 'to_wake' field.
 143      fn take_to_wake(&mut self) -> BlockedTask {
 144          let task = self.to_wake.load(atomics::SeqCst);
 145          self.to_wake.store(0, atomics::SeqCst);
 146          assert!(task != 0);
 147          unsafe { BlockedTask::cast_from_uint(task) }
 148      }
 149  
 150      // Decrements the count on the channel for a sleeper, returning the sleeper
 151      // back if it shouldn't sleep. Note that this is the location where we take
 152      // steals into account.
 153      fn decrement(&mut self, task: BlockedTask) -> Result<(), BlockedTask> {
 154          assert_eq!(self.to_wake.load(atomics::SeqCst), 0);
 155          let n = unsafe { task.cast_to_uint() };
 156          self.to_wake.store(n, atomics::SeqCst);
 157  
 158          let steals = self.steals;
 159          self.steals = 0;
 160  
 161          match self.cnt.fetch_sub(1 + steals, atomics::SeqCst) {
 162              DISCONNECTED => { self.cnt.store(DISCONNECTED, atomics::SeqCst); }
 163              // If we factor in our steals and notice that the channel has no
 164              // data, we successfully sleep
 165              n => {
 166                  assert!(n >= 0);
 167                  if n - steals <= 0 { return Ok(()) }
 168              }
 169          }
 170  
 171          self.to_wake.store(0, atomics::SeqCst);
 172          Err(unsafe { BlockedTask::cast_from_uint(n) })
 173      }

 175      pub fn recv(&mut self) -> Result<T, Failure<T>> {
 176          // Optimistic preflight check (scheduling is expensive).
 177          match self.try_recv() {
 178              Err(Empty) => {}
 179              data => return data,
 180          }
 181  
 182          // Welp, our channel has no data. Deschedule the current task and
 183          // initiate the blocking protocol.
 184          let task: ~Task = Local::take();
 185          task.deschedule(1, |task| {
 186              self.decrement(task)
 187          });
 188  
 189          match self.try_recv() {
 190              // Messages which actually popped from the queue shouldn't count as
 191              // a steal, so offset the decrement here (we already have our
 192              // "steal" factored into the channel count above).
 193              data @ Ok(..) |
 194              data @ Err(Upgraded(..)) => {
 195                  self.steals -= 1;
 196                  data
 197              }
 198  
 199              data => data,
 200          }
 201      }

 203      pub fn try_recv(&mut self) -> Result<T, Failure<T>> {
 204          match self.queue.pop() {
 205              // If we stole some data, record to that effect (this will be
 206              // factored into cnt later on).
 207              //
 208              // Note that we don't allow steals to grow without bound in order to
 209              // prevent eventual overflow of either steals or cnt as an overflow
 210              // would have catastrophic results. Sometimes, steals > cnt, but
 211              // other times cnt > steals, so we don't know the relation between
 212              // steals and cnt. This code path is executed only rarely, so we do
 213              // a pretty slow operation, of swapping 0 into cnt, taking steals
 214              // down as much as possible (without going negative), and then
 215              // adding back in whatever we couldn't factor into steals.
 216              Some(data) => {
 217                  if self.steals > MAX_STEALS {
 218                      match self.cnt.swap(0, atomics::SeqCst) {
 219                          DISCONNECTED => {
 220                              self.cnt.store(DISCONNECTED, atomics::SeqCst);
 221                          }
 222                          n => {
 223                              let m = cmp::min(n, self.steals);
 224                              self.steals -= m;
 225                              self.bump(n - m);
 226                          }
 227                      }
 228                      assert!(self.steals >= 0);
 229                  }
 230                  self.steals += 1;
 231                  match data {
 232                      Data(t) => Ok(t),
 233                      GoUp(up) => Err(Upgraded(up)),
 234                  }
 235              }
 236  
 237              None => {
 238                  match self.cnt.load(atomics::SeqCst) {
 239                      n if n != DISCONNECTED => Err(Empty),
 240  
 241                      // This is a little bit of a tricky case. We failed to pop
 242                      // data above, and then we have viewed that the channel is
 243                      // disconnected. In this window more data could have been
 244                      // sent on the channel. It doesn't really make sense to
 245                      // return that the channel is disconnected when there's
 246                      // actually data on it, so be extra sure there's no data by
 247                      // popping one more time.
 248                      //
 249                      // We can ignore steals because the other end is
 250                      // disconnected and we'll never need to really factor in our
 251                      // steals again.
 252                      _ => {
 253                          match self.queue.pop() {
 254                              Some(Data(t)) => Ok(t),
 255                              Some(GoUp(up)) => Err(Upgraded(up)),
 256                              None => Err(Disconnected),
 257                          }
 258                      }
 259                  }
 260              }
 261          }
 262      }

 264      pub fn drop_chan(&mut self) {
 265          // Dropping a channel is pretty simple, we just flag it as disconnected
 266          // and then wakeup a blocker if there is one.
 267          match self.cnt.swap(DISCONNECTED, atomics::SeqCst) {
 268              -1 => { self.take_to_wake().wake().map(|t| t.reawaken()); }
 269              DISCONNECTED => {}
 270              n => { assert!(n >= 0); }
 271          }
 272      }

 274      pub fn drop_port(&mut self) {
 275          // Dropping a port seems like a fairly trivial thing. In theory all we
 276          // need to do is flag that we're disconnected and then everything else
 277          // can take over (we don't have anyone to wake up).
 278          //
 279          // The catch for Ports is that we want to drop the entire contents of
 280          // the queue. There are multiple reasons for having this property, the
 281          // largest of which is that if another chan is waiting in this channel
 282          // (but not received yet), then waiting on that port will cause a
 283          // deadlock.
 284          //
 285          // So if we accept that we must now destroy the entire contents of the
 286          // queue, this code may make a bit more sense. The tricky part is that
 287          // we can't let any in-flight sends go un-dropped, we have to make sure
 288          // *everything* is dropped and nothing new will come onto the channel.
 289  
 290          // The first thing we do is set a flag saying that we're done for. All
 291          // sends are gated on this flag, so we're immediately guaranteed that
 292          // there are a bounded number of active sends that we'll have to deal
 293          // with.
 294          self.port_dropped.store(true, atomics::SeqCst);
 295  
 296          // Now that we're guaranteed to deal with a bounded number of senders,
 297          // we need to drain the queue. This draining process happens atomically
 298          // with respect to the "count" of the channel. If the count is nonzero
 299          // (with steals taken into account), then there must be data on the
 300          // channel. In this case we drain everything and then try again. We will
 301          // continue to fail while active senders send data while we're dropping
 302          // data, but eventually we're guaranteed to break out of this loop
 303          // (because there is a bounded number of senders).
 304          let mut steals = self.steals;
 305          while {
 306              let cnt = self.cnt.compare_and_swap(
 307                              steals, DISCONNECTED, atomics::SeqCst);
 308              cnt != DISCONNECTED && cnt != steals
 309          } {
 310              loop {
 311                  match self.queue.pop() {
 312                      Some(..) => { steals += 1; }
 313                      None => break
 314                  }
 315              }
 316          }
 317  
 318          // At this point in time, we have gated all future senders from sending,
 319          // and we have flagged the channel as being disconnected. The senders
 320          // still have some responsibility, however, because some sends may not
 321          // complete until after we flag the disconnection. There are more
 322          // details in the sending methods that see DISCONNECTED
 323      }

 325      ////////////////////////////////////////////////////////////////////////////
 326      // select implementation
 327      ////////////////////////////////////////////////////////////////////////////
 328  
 329      // Tests to see whether this port can receive without blocking. If Ok is
 330      // returned, then that's the answer. If Err is returned, then the returned
 331      // port needs to be queried instead (an upgrade happened)
 332      pub fn can_recv(&mut self) -> Result<bool, Receiver<T>> {
 333          // We peek at the queue to see if there's anything on it, and we use
 334          // this return value to determine if we should pop from the queue and
 335          // upgrade this channel immediately. If it looks like we've got an
 336          // upgrade pending, then go through the whole recv rigamarole to update
 337          // the internal state.
 338          match self.queue.peek() {
 339              Some(&GoUp(..)) => {
 340                  match self.recv() {
 341                      Err(Upgraded(port)) => Err(port),
 342                      _ => unreachable!(),
 343                  }
 344              }
 345              Some(..) => Ok(true),
 346              None => Ok(false)
 347          }
 348      }
 349  
 350      // increment the count on the channel (used for selection)
 351      fn bump(&mut self, amt: int) -> int {
 352          match self.cnt.fetch_add(amt, atomics::SeqCst) {
 353              DISCONNECTED => {
 354                  self.cnt.store(DISCONNECTED, atomics::SeqCst);
 355                  DISCONNECTED
 356              }
 357              n => n
 358          }
 359      }

 361      // Attempts to start selecting on this port. Like a oneshot, this can fail
 362      // immediately because of an upgrade.
 363      pub fn start_selection(&mut self, task: BlockedTask) -> SelectionResult<T> {
 364          match self.decrement(task) {
 365              Ok(()) => SelSuccess,
 366              Err(task) => {
 367                  let ret = match self.queue.peek() {
 368                      Some(&GoUp(..)) => {
 369                          match self.queue.pop() {
 370                              Some(GoUp(port)) => SelUpgraded(task, port),
 371                              _ => unreachable!(),
 372                          }
 373                      }
 374                      Some(..) => SelCanceled(task),
 375                      None => SelCanceled(task),
 376                  };
 377                  // Undo our decrement above, and we should be guaranteed that the
 378                  // previous value is positive because we're not going to sleep
 379                  let prev = self.bump(1);
 380                  assert!(prev == DISCONNECTED || prev >= 0);
 381                  return ret;
 382              }
 383          }
 384      }

 386      // Removes a previous task from being blocked in this port
 387      pub fn abort_selection(&mut self,
 388                             was_upgrade: bool) -> Result<bool, Receiver<T>> {
 389          // If we're aborting selection after upgrading from a oneshot, then
 390          // we're guarantee that no one is waiting. The only way that we could
 391          // have seen the upgrade is if data was actually sent on the channel
 392          // half again. For us, this means that there is guaranteed to be data on
 393          // this channel. Furthermore, we're guaranteed that there was no
 394          // start_selection previously, so there's no need to modify `self.cnt`
 395          // at all.
 396          //
 397          // Hence, because of these invariants, we immediately return `Ok(true)`.
 398          // Note that the data may not actually be sent on the channel just yet.

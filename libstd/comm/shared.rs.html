  11  /// Shared channels
  12  ///
  13  /// This is the flavor of channels which are not necessarily optimized for any
  14  /// particular use case, but are the most general in how they are used. Shared
  15  /// channels are cloneable allowing for multiple senders.
  16  ///
  17  /// High level implementation details can be found in the comment of the parent
  18  /// module. You'll also note that the implementation of the shared and stream
  19  /// channels are quite similar, and this is no coincidence!

  68      // Creation of a packet *must* be followed by a call to inherit_blocker
  69      pub fn new() -> Packet<T> {
  70          let p = Packet {
  71              queue: mpsc::Queue::new(),
  72              cnt: atomics::AtomicInt::new(0),
  73              steals: 0,
  74              to_wake: atomics::AtomicUint::new(0),
  75              channels: atomics::AtomicInt::new(2),
  76              port_dropped: atomics::AtomicBool::new(false),
  77              sender_drain: atomics::AtomicInt::new(0),
  78              select_lock: unsafe { NativeMutex::new() },
  79          };
  80          // see comments in inherit_blocker about why we grab this lock
  81          unsafe { p.select_lock.lock_noguard() }
  82          return p;
  83      }

  85      // This function is used at the creation of a shared packet to inherit a
  86      // previously blocked task. This is done to prevent spurious wakeups of
  87      // tasks in select().
  88      //
  89      // This can only be called at channel-creation time
  90      pub fn inherit_blocker(&mut self, task: Option<BlockedTask>) {
  91          match task {
  92              Some(task) => {
  93                  assert_eq!(self.cnt.load(atomics::SeqCst), 0);
  94                  assert_eq!(self.to_wake.load(atomics::SeqCst), 0);
  95                  self.to_wake.store(unsafe { task.cast_to_uint() },
  96                                     atomics::SeqCst);
  97                  self.cnt.store(-1, atomics::SeqCst);
  98  
  99                  // This store is a little sketchy. What's happening here is
 100                  // that we're transferring a blocker from a oneshot or stream
 101                  // channel to this shared channel. In doing so, we never
 102                  // spuriously wake them up and rather only wake them up at the
 103                  // appropriate time. This implementation of shared channels
 104                  // assumes that any blocking recv() will undo the increment of
 105                  // steals performed in try_recv() once the recv is complete.
 106                  // This thread that we're inheriting, however, is not in the
 107                  // middle of recv. Hence, the first time we wake them up,
 108                  // they're going to wake up from their old port, move on to the
 109                  // upgraded port, and then call the block recv() function.
 110                  //
 111                  // When calling this function, they'll find there's data
 112                  // immediately available, counting it as a steal. This in fact
 113                  // wasn't a steal because we appropriately blocked them waiting
 114                  // for data.
 115                  //
 116                  // To offset this bad increment, we initially set the steal
 117                  // count to -1. You'll find some special code in
 118                  // abort_selection() as well to ensure that this -1 steal count
 119                  // doesn't escape too far.
 120                  self.steals = -1;
 121              }
 122              None => {}
 123          }
 124  
 125          // When the shared packet is constructed, we grabbed this lock. The
 126          // purpose of this lock is to ensure that abort_selection() doesn't
 127          // interfere with this method. After we unlock this lock, we're
 128          // signifying that we're done modifying self.cnt and self.to_wake and
 129          // the port is ready for the world to continue using it.
 130          unsafe { self.select_lock.unlock_noguard() }
 131      }

 133      pub fn send(&mut self, t: T) -> Result<(), T> {
 134          // See Port::drop for what's going on
 135          if self.port_dropped.load(atomics::SeqCst) { return Err(t) }
 136  
 137          // Note that the multiple sender case is a little tricker
 138          // semantically than the single sender case. The logic for
 139          // incrementing is "add and if disconnected store disconnected".
 140          // This could end up leading some senders to believe that there
 141          // wasn't a disconnect if in fact there was a disconnect. This means
 142          // that while one thread is attempting to re-store the disconnected
 143          // states, other threads could walk through merrily incrementing
 144          // this very-negative disconnected count. To prevent senders from
 145          // spuriously attempting to send when the channels is actually
 146          // disconnected, the count has a ranged check here.
 147          //
 148          // This is also done for another reason. Remember that the return
 149          // value of this function is:
 150          //
 151          //  `true` == the data *may* be received, this essentially has no
 152          //            meaning
 153          //  `false` == the data will *never* be received, this has a lot of
 154          //             meaning
 155          //
 156          // In the SPSC case, we have a check of 'queue.is_empty()' to see
 157          // whether the data was actually received, but this same condition
 158          // means nothing in a multi-producer context. As a result, this
 159          // preflight check serves as the definitive "this will never be
 160          // received". Once we get beyond this check, we have permanently
 161          // entered the realm of "this may be received"
 162          if self.cnt.load(atomics::SeqCst) < DISCONNECTED + FUDGE {
 163              return Err(t)
 164          }
 165  
 166          self.queue.push(t);
 167          match self.cnt.fetch_add(1, atomics::SeqCst) {
 168              -1 => {
 169                  self.take_to_wake().wake().map(|t| t.reawaken());
 170              }
 171  
 172              // In this case, we have possibly failed to send our data, and
 173              // we need to consider re-popping the data in order to fully
 174              // destroy it. We must arbitrate among the multiple senders,
 175              // however, because the queues that we're using are
 176              // single-consumer queues. In order to do this, all exiting
 177              // pushers will use an atomic count in order to count those
 178              // flowing through. Pushers who see 0 are required to drain as
 179              // much as possible, and then can only exit when they are the
 180              // only pusher (otherwise they must try again).
 181              n if n < DISCONNECTED + FUDGE => {
 182                  // see the comment in 'try' for a shared channel for why this
 183                  // window of "not disconnected" is ok.
 184                  self.cnt.store(DISCONNECTED, atomics::SeqCst);
 185  
 186                  if self.sender_drain.fetch_add(1, atomics::SeqCst) == 0 {
 187                      loop {
 188                          // drain the queue, for info on the thread yield see the
 189                          // discussion in try_recv
 190                          loop {
 191                              match self.queue.pop() {
 192                                  mpsc::Data(..) => {}
 193                                  mpsc::Empty => break,
 194                                  mpsc::Inconsistent => Thread::yield_now(),
 195                              }
 196                          }
 197                          // maybe we're done, if we're not the last ones
 198                          // here, then we need to go try again.
 199                          if self.sender_drain.fetch_sub(1, atomics::SeqCst) == 1 {
 200                              break
 201                          }
 202                      }
 203  
 204                      // At this point, there may still be data on the queue,
 205                      // but only if the count hasn't been incremented and
 206                      // some other sender hasn't finished pushing data just
 207                      // yet. That sender in question will drain its own data.
 208                  }
 209              }
 210  
 211              // Can't make any assumptions about this case like in the SPSC case.
 212              _ => {}
 213          }
 214  
 215          Ok(())
 216      }

 218      pub fn recv(&mut self) -> Result<T, Failure> {
 219          // This code is essentially the exact same as that found in the stream
 220          // case (see stream.rs)
 221          match self.try_recv() {
 222              Err(Empty) => {}
 223              data => return data,
 224          }
 225  
 226          let task: ~Task = Local::take();
 227          task.deschedule(1, |task| {
 228              self.decrement(task)
 229          });
 230  
 231          match self.try_recv() {
 232              data @ Ok(..) => { self.steals -= 1; data }
 233              data => data,
 234          }
 235      }
 236  
 237      // Essentially the exact same thing as the stream decrement function.
 238      fn decrement(&mut self, task: BlockedTask) -> Result<(), BlockedTask> {
 239          assert_eq!(self.to_wake.load(atomics::SeqCst), 0);
 240          let n = unsafe { task.cast_to_uint() };
 241          self.to_wake.store(n, atomics::SeqCst);
 242  
 243          let steals = self.steals;
 244          self.steals = 0;
 245  
 246          match self.cnt.fetch_sub(1 + steals, atomics::SeqCst) {
 247              DISCONNECTED => { self.cnt.store(DISCONNECTED, atomics::SeqCst); }
 248              // If we factor in our steals and notice that the channel has no
 249              // data, we successfully sleep
 250              n => {
 251                  assert!(n >= 0);
 252                  if n - steals <= 0 { return Ok(()) }
 253              }
 254          }
 255  
 256          self.to_wake.store(0, atomics::SeqCst);
 257          Err(unsafe { BlockedTask::cast_from_uint(n) })
 258      }

 260      pub fn try_recv(&mut self) -> Result<T, Failure> {
 261          let ret = match self.queue.pop() {
 262              mpsc::Data(t) => Some(t),
 263              mpsc::Empty => None,
 264  
 265              // This is a bit of an interesting case. The channel is
 266              // reported as having data available, but our pop() has
 267              // failed due to the queue being in an inconsistent state.
 268              // This means that there is some pusher somewhere which has
 269              // yet to complete, but we are guaranteed that a pop will
 270              // eventually succeed. In this case, we spin in a yield loop
 271              // because the remote sender should finish their enqueue
 272              // operation "very quickly".
 273              //
 274              // Note that this yield loop does *not* attempt to do a green
 275              // yield (regardless of the context), but *always* performs an
 276              // OS-thread yield. The reasoning for this is that the pusher in
 277              // question which is causing the inconsistent state is
 278              // guaranteed to *not* be a blocked task (green tasks can't get
 279              // pre-empted), so it must be on a different OS thread. Also,
 280              // `try_recv` is normally a "guaranteed no rescheduling" context
 281              // in a green-thread situation. By yielding control of the
 282              // thread, we will hopefully allow time for the remote task on
 283              // the other OS thread to make progress.
 284              //
 285              // Avoiding this yield loop would require a different queue
 286              // abstraction which provides the guarantee that after M
 287              // pushes have succeeded, at least M pops will succeed. The
 288              // current queues guarantee that if there are N active
 289              // pushes, you can pop N times once all N have finished.
 290              mpsc::Inconsistent => {
 291                  let data;
 292                  loop {
 293                      Thread::yield_now();
 294                      match self.queue.pop() {
 295                          mpsc::Data(t) => { data = t; break }
 296                          mpsc::Empty => fail!("inconsistent => empty"),
 297                          mpsc::Inconsistent => {}
 298                      }
 299                  }
 300                  Some(data)
 301              }
 302          };
 303          match ret {
 304              // See the discussion in the stream implementation for why we
 305              // might decrement steals.
 306              Some(data) => {
 307                  if self.steals > MAX_STEALS {
 308                      match self.cnt.swap(0, atomics::SeqCst) {
 309                          DISCONNECTED => {
 310                              self.cnt.store(DISCONNECTED, atomics::SeqCst);
 311                          }
 312                          n => {
 313                              let m = cmp::min(n, self.steals);
 314                              self.steals -= m;
 315                              self.bump(n - m);
 316                          }
 317                      }
 318                      assert!(self.steals >= 0);
 319                  }
 320                  self.steals += 1;
 321                  Ok(data)
 322              }
 323  
 324              // See the discussion in the stream implementation for why we try
 325              // again.
 326              None => {
 327                  match self.cnt.load(atomics::SeqCst) {
 328                      n if n != DISCONNECTED => Err(Empty),
 329                      _ => {
 330                          match self.queue.pop() {
 331                              mpsc::Data(t) => Ok(t),
 332                              mpsc::Empty => Err(Disconnected),
 333                              // with no senders, an inconsistency is impossible.
 334                              mpsc::Inconsistent => unreachable!(),
 335                          }
 336                      }
 337                  }
 338              }
 339          }
 340      }

 348      // Decrement the reference count on a channel. This is called whenever a
 349      // Chan is dropped and may end up waking up a receiver. It's the receiver's
 350      // responsibility on the other end to figure out that we've disconnected.
 351      pub fn drop_chan(&mut self) {
 352          match self.channels.fetch_sub(1, atomics::SeqCst) {
 353              1 => {}
 354              n if n > 1 => return,
 355              n => fail!("bad number of channels left {}", n),
 356          }
 357  
 358          match self.cnt.swap(DISCONNECTED, atomics::SeqCst) {
 359              -1 => { self.take_to_wake().wake().map(|t| t.reawaken()); }
 360              DISCONNECTED => {}
 361              n => { assert!(n >= 0); }
 362          }
 363      }

 365      // See the long discussion inside of stream.rs for why the queue is drained,
 366      // and why it is done in this fashion.
 367      pub fn drop_port(&mut self) {
 368          self.port_dropped.store(true, atomics::SeqCst);
 369          let mut steals = self.steals;
 370          while {
 371              let cnt = self.cnt.compare_and_swap(
 372                              steals, DISCONNECTED, atomics::SeqCst);
 373              cnt != DISCONNECTED && cnt != steals
 374          } {
 375              // See the discussion in 'try_recv' for why we yield
 376              // control of this thread.
 377              loop {
 378                  match self.queue.pop() {
 379                      mpsc::Data(..) => { steals += 1; }
 380                      mpsc::Empty | mpsc::Inconsistent => break,
 381                  }
 382              }
 383          }
 384      }

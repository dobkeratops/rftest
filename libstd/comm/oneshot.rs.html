  11  /// Oneshot channels/ports
  12  ///
  13  /// This is the initial flavor of channels/ports used for comm module. This is
  14  /// an optimization for the one-use case of a channel. The major optimization of
  15  /// this type is to have one and exactly one allocation when the chan/port pair
  16  /// is created.
  17  ///
  18  /// Another possible optimization would be to not use an UnsafeArc box because
  19  /// in theory we know when the shared packet can be deallocated (no real need
  20  /// for the atomic reference counting), but I was having trouble how to destroy
  21  /// the data early in a drop of a Port.
  22  ///
  23  /// # Implementation
  24  ///
  25  /// Oneshots are implemented around one atomic uint variable. This variable
  26  /// indicates both the state of the port/chan but also contains any tasks
  27  /// blocked on the port. All atomic operations happen on this one word.
  28  ///
  29  /// In order to upgrade a oneshot channel, an upgrade is considered a disconnect
  30  /// on behalf of the channel side of things (it can be mentally thought of as
  31  /// consuming the port). This upgrade is then also stored in the shared packet.
  32  /// The one caveat to consider is that when a port sees a disconnected channel
  33  /// it must check for data because there is no "data plus upgrade" state.
  34

  50  pub struct Packet<T> {

  54      data: Option<T>,

  56      // this contains the slot for the upgrade
  57      upgrade: MyUpgrade<T>,
  58  }

  85      pub fn new() -> Packet<T> {
  86          Packet {
  87              data: None,
  88              upgrade: NothingSent,
  89              state: atomics::AtomicUint::new(EMPTY),
  90          }
  91      }

  93      pub fn send(&mut self, t: T) -> bool {
  94          // Sanity check
  95          match self.upgrade {
  96              NothingSent => {}
  97              _ => fail!("sending on a oneshot that's already sent on "),
  98          }
  99          assert!(self.data.is_none());
 100          self.data = Some(t);
 101          self.upgrade = SendUsed;
 102  
 103          match self.state.swap(DATA, atomics::SeqCst) {
 104              // Sent the data, no one was waiting
 105              EMPTY => true,
 106  
 107              // Couldn't send the data, the port hung up first. We need to be
 108              // sure to deallocate the sent data (to not leave it stuck in the
 109              // queue)
 110              DISCONNECTED => {
 111                  self.data.take_unwrap();
 112                  false
 113              }
 114  
 115              // Not possible, these are one-use channels
 116              DATA => unreachable!(),
 117  
 118              // Anything else means that there was a task waiting on the other
 119              // end. We leave the 'DATA' state inside so it'll pick it up on the
 120              // other end.
 121              n => unsafe {
 122                  let t = BlockedTask::cast_from_uint(n);
 123                  t.wake().map(|t| t.reawaken());
 124                  true
 125              }
 126          }
 127      }
 128  
 129      // Just tests whether this channel has been sent on or not, this is only
 130      // safe to use from the sender.
 131      pub fn sent(&self) -> bool {
 132          match self.upgrade {
 133              NothingSent => false,
 134              _ => true,
 135          }
 136      }

 138      pub fn recv(&mut self) -> Result<T, Failure<T>> {
 139          // Attempt to not block the task (it's a little expensive). If it looks
 140          // like we're not empty, then immediately go through to `try_recv`.
 141          if self.state.load(atomics::SeqCst) == EMPTY {
 142              let t: ~Task = Local::take();
 143              t.deschedule(1, |task| {
 144                  let n = unsafe { task.cast_to_uint() };
 145                  match self.state.compare_and_swap(EMPTY, n, atomics::SeqCst) {
 146                      // Nothing on the channel, we legitimately block
 147                      EMPTY => Ok(()),
 148  
 149                      // If there's data or it's a disconnected channel, then we
 150                      // failed the cmpxchg, so we just wake ourselves back up
 151                      DATA | DISCONNECTED => {
 152                          unsafe { Err(BlockedTask::cast_from_uint(n)) }
 153                      }
 154  
 155                      // Only one thread is allowed to sleep on this port
 156                      _ => unreachable!()
 157                  }
 158              });
 159          }
 160  
 161          self.try_recv()
 162      }

 164      pub fn try_recv(&mut self) -> Result<T, Failure<T>> {
 165          match self.state.load(atomics::SeqCst) {
 166              EMPTY => Err(Empty),
 167  
 168              // We saw some data on the channel, but the channel can be used
 169              // again to send us an upgrade. As a result, we need to re-insert
 170              // into the channel that there's no data available (otherwise we'll
 171              // just see DATA next time). This is done as a cmpxchg because if
 172              // the state changes under our feet we'd rather just see that state
 173              // change.
 174              DATA => {
 175                  self.state.compare_and_swap(DATA, EMPTY, atomics::SeqCst);
 176                  match self.data.take() {
 177                      Some(data) => Ok(data),
 178                      None => unreachable!(),
 179                  }
 180              }
 181  
 182              // There's no guarantee that we receive before an upgrade happens,
 183              // and an upgrade flags the channel as disconnected, so when we see
 184              // this we first need to check if there's data available and *then*
 185              // we go through and process the upgrade.
 186              DISCONNECTED => {
 187                  match self.data.take() {
 188                      Some(data) => Ok(data),
 189                      None => {
 190                          match mem::replace(&mut self.upgrade, SendUsed) {
 191                              SendUsed | NothingSent => Err(Disconnected),
 192                              GoUp(upgrade) => Err(Upgraded(upgrade))
 193                          }
 194                      }
 195                  }
 196              }
 197              _ => unreachable!()
 198          }
 199      }
 200  
 201      // Returns whether the upgrade was completed. If the upgrade wasn't
 202      // completed, then the port couldn't get sent to the other half (it will
 203      // never receive it).
 204      pub fn upgrade(&mut self, up: Receiver<T>) -> UpgradeResult {
 205          let prev = match self.upgrade {
 206              NothingSent => NothingSent,
 207              SendUsed => SendUsed,
 208              _ => fail!("upgrading again"),
 209          };
 210          self.upgrade = GoUp(up);
 211  
 212          match self.state.swap(DISCONNECTED, atomics::SeqCst) {
 213              // If the channel is empty or has data on it, then we're good to go.
 214              // Senders will check the data before the upgrade (in case we
 215              // plastered over the DATA state).
 216              DATA | EMPTY => UpSuccess,
 217  
 218              // If the other end is already disconnected, then we failed the
 219              // upgrade. Be sure to trash the port we were given.
 220              DISCONNECTED => { self.upgrade = prev; UpDisconnected }
 221  
 222              // If someone's waiting, we gotta wake them up
 223              n => UpWoke(unsafe { BlockedTask::cast_from_uint(n) })
 224          }
 225      }

 227      pub fn drop_chan(&mut self) {
 228          match self.state.swap(DISCONNECTED, atomics::SeqCst) {
 229              DATA | DISCONNECTED | EMPTY => {}
 230  
 231              // If someone's waiting, we gotta wake them up
 232              n => unsafe {
 233                  let t = BlockedTask::cast_from_uint(n);
 234                  t.wake().map(|t| t.reawaken());
 235              }
 236          }
 237      }

 239      pub fn drop_port(&mut self) {
 240          match self.state.swap(DISCONNECTED, atomics::SeqCst) {
 241              // An empty channel has nothing to do, and a remotely disconnected
 242              // channel also has nothing to do b/c we're about to run the drop
 243              // glue
 244              DISCONNECTED | EMPTY => {}
 245  
 246              // There's data on the channel, so make sure we destroy it promptly.
 247              // This is why not using an arc is a little difficult (need the box
 248              // to stay valid while we take the data).
 249              DATA => { self.data.take_unwrap(); }
 250  
 251              // We're the only ones that can block on this port
 252              _ => unreachable!()
 253          }
 254      }
 255  
 256      ////////////////////////////////////////////////////////////////////////////
 257      // select implementation
 258      ////////////////////////////////////////////////////////////////////////////
 259  
 260      // If Ok, the value is whether this port has data, if Err, then the upgraded
 261      // port needs to be checked instead of this one.
 262      pub fn can_recv(&mut self) -> Result<bool, Receiver<T>> {
 263          match self.state.load(atomics::SeqCst) {
 264              EMPTY => Ok(false), // Welp, we tried
 265              DATA => Ok(true),   // we have some un-acquired data
 266              DISCONNECTED if self.data.is_some() => Ok(true), // we have data
 267              DISCONNECTED => {
 268                  match mem::replace(&mut self.upgrade, SendUsed) {
 269                      // The other end sent us an upgrade, so we need to
 270                      // propagate upwards whether the upgrade can receive
 271                      // data
 272                      GoUp(upgrade) => Err(upgrade),
 273  
 274                      // If the other end disconnected without sending an
 275                      // upgrade, then we have data to receive (the channel is
 276                      // disconnected).
 277                      up => { self.upgrade = up; Ok(true) }
 278                  }
 279              }
 280              _ => unreachable!(), // we're the "one blocker"
 281          }
 282      }

 284      // Attempts to start selection on this port. This can either succeed, fail
 285      // because there is data, or fail because there is an upgrade pending.
 286      pub fn start_selection(&mut self, task: BlockedTask) -> SelectionResult<T> {
 287          let n = unsafe { task.cast_to_uint() };
 288          match self.state.compare_and_swap(EMPTY, n, atomics::SeqCst) {
 289              EMPTY => SelSuccess,
 290              DATA => SelCanceled(unsafe { BlockedTask::cast_from_uint(n) }),
 291              DISCONNECTED if self.data.is_some() => {
 292                  SelCanceled(unsafe { BlockedTask::cast_from_uint(n) })
 293              }
 294              DISCONNECTED => {
 295                  match mem::replace(&mut self.upgrade, SendUsed) {
 296                      // The other end sent us an upgrade, so we need to
 297                      // propagate upwards whether the upgrade can receive
 298                      // data
 299                      GoUp(upgrade) => {
 300                          SelUpgraded(unsafe { BlockedTask::cast_from_uint(n) },
 301                                      upgrade)
 302                      }
 303  
 304                      // If the other end disconnected without sending an
 305                      // upgrade, then we have data to receive (the channel is
 306                      // disconnected).
 307                      up => {
 308                          self.upgrade = up;
 309                          SelCanceled(unsafe { BlockedTask::cast_from_uint(n) })
 310                      }
 311                  }
 312              }
 313              _ => unreachable!(), // we're the "one blocker"
 314          }
 315      }
 316  
 317      // Remove a previous selecting task from this port. This ensures that the
 318      // blocked task will no longer be visible to any other threads.
 319      //
 320      // The return value indicates whether there's data on this port.
 321      pub fn abort_selection(&mut self) -> Result<bool, Receiver<T>> {
 322          let state = match self.state.load(atomics::SeqCst) {
 323              // Each of these states means that no further activity will happen
 324              // with regard to abortion selection
 325              s @ EMPTY |
 326              s @ DATA |
 327              s @ DISCONNECTED => s,
 328  
 329              // If we've got a blocked task, then use an atomic to gain ownership
 330              // of it (may fail)
 331              n => self.state.compare_and_swap(n, EMPTY, atomics::SeqCst)
 332          };
 333  
 334          // Now that we've got ownership of our state, figure out what to do
 335          // about it.
 336          match state {
 337              EMPTY => unreachable!(),
 338              // our task used for select was stolen
 339              DATA => Ok(true),
 340  
 341              // If the other end has hung up, then we have complete ownership
 342              // of the port. First, check if there was data waiting for us. This
 343              // is possible if the other end sent something and then hung up.
 344              //
 345              // We then need to check to see if there was an upgrade requested,
 346              // and if so, the upgraded port needs to have its selection aborted.
 347              DISCONNECTED => {
 348                  if self.data.is_some() {
 349                      Ok(true)
 350                  } else {
 351                      match mem::replace(&mut self.upgrade, SendUsed) {
 352                          GoUp(port) => Err(port),
 353                          _ => Ok(true),
 354                      }
 355                  }
 356              }
 357  
 358              // We woke ourselves up from select. Assert that the task should be
 359              // trashed and returne that we don't have any data.
 360              n => {
 361                  let t = unsafe { BlockedTask::cast_from_uint(n) };
 362                  t.trash();
 363                  Ok(false)
 364              }
 365          }
 366      }
 367  }

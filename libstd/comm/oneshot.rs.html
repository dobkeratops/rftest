  11  /// Oneshot channels/ports
  12  ///
  13  /// This is the initial flavor of channels/ports used for comm module. This is
  14  /// an optimization for the one-use case of a channel. The major optimization of
  15  /// this type is to have one and exactly one allocation when the chan/port pair
  16  /// is created.
  17  ///
  18  /// Another possible optimization would be to not use an UnsafeArc box because
  19  /// in theory we know when the shared packet can be deallocated (no real need
  20  /// for the atomic reference counting), but I was having trouble how to destroy
  21  /// the data early in a drop of a Port.
  22  ///
  23  /// # Implementation
  24  ///
  25  /// Oneshots are implemented around one atomic uint variable. This variable
  26  /// indicates both the state of the port/chan but also contains any tasks
  27  /// blocked on the port. All atomic operations happen on this one word.
  28  ///
  29  /// In order to upgrade a oneshot channel, an upgrade is considered a disconnect
  30  /// on behalf of the channel side of things (it can be mentally thought of as
  31  /// consuming the port). This upgrade is then also stored in the shared packet.
  32  /// The one caveat to consider is that when a port sees a disconnected channel
  33  /// it must check for data because there is no "data plus upgrade" state.
  34

  93      pub fn send(&mut self, t: T) -> Result<(), T> {
  94          // Sanity check
  95          match self.upgrade {
  96              NothingSent => {}
  97              _ => fail!("sending on a oneshot that's already sent on "),
  98          }
  99          assert!(self.data.is_none());
 100          self.data = Some(t);
 101          self.upgrade = SendUsed;
 102  
 103          match self.state.swap(DATA, atomics::SeqCst) {
 104              // Sent the data, no one was waiting
 105              EMPTY => Ok(()),
 106  
 107              // Couldn't send the data, the port hung up first. Return the data
 108              // back up the stack.
 109              DISCONNECTED => {
 110                  Err(self.data.take_unwrap())
 111              }
 112  
 113              // Not possible, these are one-use channels
 114              DATA => unreachable!(),
 115  
 116              // Anything else means that there was a task waiting on the other
 117              // end. We leave the 'DATA' state inside so it'll pick it up on the
 118              // other end.
 119              n => unsafe {
 120                  let t = BlockedTask::cast_from_uint(n);
 121                  t.wake().map(|t| t.reawaken());
 122                  Ok(())
 123              }
 124          }
 125      }

 128      // safe to use from the sender.

 131              NothingSent => false,
 132              _ => true,
 133          }
 134      }

 136      pub fn recv(&mut self) -> Result<T, Failure<T>> {
 137          // Attempt to not block the task (it's a little expensive). If it looks
 138          // like we're not empty, then immediately go through to `try_recv`.
 139          if self.state.load(atomics::SeqCst) == EMPTY {
 140              let t: ~Task = Local::take();
 141              t.deschedule(1, |task| {
 142                  let n = unsafe { task.cast_to_uint() };
 143                  match self.state.compare_and_swap(EMPTY, n, atomics::SeqCst) {
 144                      // Nothing on the channel, we legitimately block
 145                      EMPTY => Ok(()),
 146  
 147                      // If there's data or it's a disconnected channel, then we
 148                      // failed the cmpxchg, so we just wake ourselves back up
 149                      DATA | DISCONNECTED => {
 150                          unsafe { Err(BlockedTask::cast_from_uint(n)) }
 151                      }
 152  
 153                      // Only one thread is allowed to sleep on this port
 154                      _ => unreachable!()
 155                  }
 156              });
 157          }
 158  
 159          self.try_recv()
 160      }

 162      pub fn try_recv(&mut self) -> Result<T, Failure<T>> {
 163          match self.state.load(atomics::SeqCst) {
 164              EMPTY => Err(Empty),
 165  
 166              // We saw some data on the channel, but the channel can be used
 167              // again to send us an upgrade. As a result, we need to re-insert
 168              // into the channel that there's no data available (otherwise we'll
 169              // just see DATA next time). This is done as a cmpxchg because if
 170              // the state changes under our feet we'd rather just see that state
 171              // change.
 172              DATA => {
 173                  self.state.compare_and_swap(DATA, EMPTY, atomics::SeqCst);
 174                  match self.data.take() {
 175                      Some(data) => Ok(data),
 176                      None => unreachable!(),
 177                  }
 178              }
 179  
 180              // There's no guarantee that we receive before an upgrade happens,
 181              // and an upgrade flags the channel as disconnected, so when we see
 182              // this we first need to check if there's data available and *then*
 183              // we go through and process the upgrade.
 184              DISCONNECTED => {
 185                  match self.data.take() {
 186                      Some(data) => Ok(data),
 187                      None => {
 188                          match mem::replace(&mut self.upgrade, SendUsed) {
 189                              SendUsed | NothingSent => Err(Disconnected),
 190                              GoUp(upgrade) => Err(Upgraded(upgrade))
 191                          }
 192                      }
 193                  }
 194              }
 195              _ => unreachable!()
 196          }
 197      }
 198  
 199      // Returns whether the upgrade was completed. If the upgrade wasn't
 200      // completed, then the port couldn't get sent to the other half (it will
 201      // never receive it).
 202      pub fn upgrade(&mut self, up: Receiver<T>) -> UpgradeResult {
 203          let prev = match self.upgrade {
 204              NothingSent => NothingSent,
 205              SendUsed => SendUsed,
 206              _ => fail!("upgrading again"),
 207          };
 208          self.upgrade = GoUp(up);
 209  
 210          match self.state.swap(DISCONNECTED, atomics::SeqCst) {
 211              // If the channel is empty or has data on it, then we're good to go.
 212              // Senders will check the data before the upgrade (in case we
 213              // plastered over the DATA state).
 214              DATA | EMPTY => UpSuccess,
 215  
 216              // If the other end is already disconnected, then we failed the
 217              // upgrade. Be sure to trash the port we were given.
 218              DISCONNECTED => { self.upgrade = prev; UpDisconnected }
 219  
 220              // If someone's waiting, we gotta wake them up
 221              n => UpWoke(unsafe { BlockedTask::cast_from_uint(n) })
 222          }
 223      }

 225      pub fn drop_chan(&mut self) {
 226          match self.state.swap(DISCONNECTED, atomics::SeqCst) {
 227              DATA | DISCONNECTED | EMPTY => {}
 228  
 229              // If someone's waiting, we gotta wake them up
 230              n => unsafe {
 231                  let t = BlockedTask::cast_from_uint(n);
 232                  t.wake().map(|t| t.reawaken());
 233              }
 234          }
 235      }

 237      pub fn drop_port(&mut self) {
 238          match self.state.swap(DISCONNECTED, atomics::SeqCst) {
 239              // An empty channel has nothing to do, and a remotely disconnected
 240              // channel also has nothing to do b/c we're about to run the drop
 241              // glue
 242              DISCONNECTED | EMPTY => {}
 243  
 244              // There's data on the channel, so make sure we destroy it promptly.
 245              // This is why not using an arc is a little difficult (need the box
 246              // to stay valid while we take the data).
 247              DATA => { self.data.take_unwrap(); }
 248  
 249              // We're the only ones that can block on this port
 250              _ => unreachable!()
 251          }
 252      }

 254      ////////////////////////////////////////////////////////////////////////////
 255      // select implementation
 256      ////////////////////////////////////////////////////////////////////////////
 257  
 258      // If Ok, the value is whether this port has data, if Err, then the upgraded
 259      // port needs to be checked instead of this one.
 260      pub fn can_recv(&mut self) -> Result<bool, Receiver<T>> {
 261          match self.state.load(atomics::SeqCst) {
 262              EMPTY => Ok(false), // Welp, we tried
 263              DATA => Ok(true),   // we have some un-acquired data
 264              DISCONNECTED if self.data.is_some() => Ok(true), // we have data
 265              DISCONNECTED => {
 266                  match mem::replace(&mut self.upgrade, SendUsed) {
 267                      // The other end sent us an upgrade, so we need to
 268                      // propagate upwards whether the upgrade can receive
 269                      // data
 270                      GoUp(upgrade) => Err(upgrade),
 271  
 272                      // If the other end disconnected without sending an
 273                      // upgrade, then we have data to receive (the channel is
 274                      // disconnected).
 275                      up => { self.upgrade = up; Ok(true) }
 276                  }
 277              }
 278              _ => unreachable!(), // we're the "one blocker"
 279          }
 280      }

 282      // Attempts to start selection on this port. This can either succeed, fail
 283      // because there is data, or fail because there is an upgrade pending.
 284      pub fn start_selection(&mut self, task: BlockedTask) -> SelectionResult<T> {
 285          let n = unsafe { task.cast_to_uint() };
 286          match self.state.compare_and_swap(EMPTY, n, atomics::SeqCst) {
 287              EMPTY => SelSuccess,
 288              DATA => SelCanceled(unsafe { BlockedTask::cast_from_uint(n) }),
 289              DISCONNECTED if self.data.is_some() => {
 290                  SelCanceled(unsafe { BlockedTask::cast_from_uint(n) })
 291              }
 292              DISCONNECTED => {
 293                  match mem::replace(&mut self.upgrade, SendUsed) {
 294                      // The other end sent us an upgrade, so we need to
 295                      // propagate upwards whether the upgrade can receive
 296                      // data
 297                      GoUp(upgrade) => {
 298                          SelUpgraded(unsafe { BlockedTask::cast_from_uint(n) },
 299                                      upgrade)
 300                      }
 301  
 302                      // If the other end disconnected without sending an
 303                      // upgrade, then we have data to receive (the channel is
 304                      // disconnected).
 305                      up => {
 306                          self.upgrade = up;
 307                          SelCanceled(unsafe { BlockedTask::cast_from_uint(n) })
 308                      }
 309                  }
 310              }
 311              _ => unreachable!(), // we're the "one blocker"
 312          }
 313      }

 315      // Remove a previous selecting task from this port. This ensures that the
 316      // blocked task will no longer be visible to any other threads.
 317      //
 318      // The return value indicates whether there's data on this port.
 319      pub fn abort_selection(&mut self) -> Result<bool, Receiver<T>> {
 320          let state = match self.state.load(atomics::SeqCst) {
 321              // Each of these states means that no further activity will happen
 322              // with regard to abortion selection
 323              s @ EMPTY |
 324              s @ DATA |
 325              s @ DISCONNECTED => s,
 326  
 327              // If we've got a blocked task, then use an atomic to gain ownership
 328              // of it (may fail)
 329              n => self.state.compare_and_swap(n, EMPTY, atomics::SeqCst)
 330          };
 331  
 332          // Now that we've got ownership of our state, figure out what to do
 333          // about it.
 334          match state {
 335              EMPTY => unreachable!(),
 336              // our task used for select was stolen
 337              DATA => Ok(true),
 338  
 339              // If the other end has hung up, then we have complete ownership
 340              // of the port. First, check if there was data waiting for us. This
 341              // is possible if the other end sent something and then hung up.
 342              //
 343              // We then need to check to see if there was an upgrade requested,
 344              // and if so, the upgraded port needs to have its selection aborted.
 345              DISCONNECTED => {
 346                  if self.data.is_some() {
 347                      Ok(true)
 348                  } else {
 349                      match mem::replace(&mut self.upgrade, SendUsed) {
 350                          GoUp(port) => Err(port),
 351                          _ => Ok(true),
 352                      }
 353                  }
 354              }
 355  
 356              // We woke ourselves up from select. Assert that the task should be
 357              // trashed and returne that we don't have any data.
 358              n => {
 359                  let t = unsafe { BlockedTask::cast_from_uint(n) };
 360                  t.trash();
 361                  Ok(false)
 362              }
 363          }
 364      }
 365  }

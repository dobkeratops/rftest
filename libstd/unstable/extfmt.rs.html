  11 //! Support for fmt! expressions.
  12 //!
  13 //! The syntax is close to that of Posix format strings:
  14 //!
  15 //! ~~~~~~
  16 //! Format := '%' Parameter? Flag* Width? Precision? Type
  17 //! Parameter := [0-9]+ '$'
  18 //! Flag := [ 0#+-]
  19 //! Width := Parameter | [0-9]+
  20 //! Precision := '.' [0-9]+
  21 //! Type := [bcdfiostuxX?]
  22 //! ~~~~~~
  23 //!
  24 //! * Parameter is the 1-based argument to apply the format to. Currently not
  25 //! implemented.
  26 //! * Flag 0 causes leading zeros to be used for padding when converting
  27 //! numbers.
  28 //! * Flag # causes the conversion to be done in an *alternative* manner.
  29 //! Currently not implemented.
  30 //! * Flag + causes signed numbers to always be prepended with a sign
  31 //! character.
  32 //! * Flag - left justifies the result
  33 //! * Width specifies the minimum field width of the result. By default
  34 //! leading spaces are added.
  35 //! * Precision specifies the minimum number of digits for integral types
  36 //! and the minimum number
  37 //! of decimal places for float.
  38 //!
  39 //! The types currently supported are:
  40 //!
  41 //! * b - bool
  42 //! * c - char
  43 //! * d - int
  44 //! * f - float
  45 //! * i - int (same as d)
  46 //! * o - uint as octal
  47 //! * t - uint as binary
  48 //! * u - uint
  49 //! * x - uint as lower-case hexadecimal
  50 //! * X - uint as upper-case hexadecimal
  51 //! * s - str (any flavor)
  52 //! * ? - arbitrary type (does not use the to_str trait)
  53 
  54 /*
  55 Syntax Extension: fmt
  56 
  57 Format a string
  58 
  59 The 'fmt' extension is modeled on the posix printf system.
  60 
  61 A posix conversion ostensibly looks like this
  62 
  63 > %~[parameter]~[flags]~[width]~[.precision]~[length]type
  64 
  65 Given the different numeric type bestiary we have, we omit the 'length'
  66 parameter and support slightly different conversions for 'type'
  67 
  68 > %~[parameter]~[flags]~[width]~[.precision]type
  69 
  70 we also only support translating-to-rust a tiny subset of the possible
  71 combinations at the moment.
  72 
  73 Example:
  74 
  75 debug!("hello, %s!", "world");
  76 
  77 */
  78

  81 /*
  82  * We have a 'ct' (compile-time) module that parses format strings into a
  83  * sequence of conversions. From those conversions AST fragments are built
  84  * that call into properly-typed functions in the 'rt' (run-time) module.
  85  * Each of those run-time conversion functions accepts another conversion
  86  * description that specifies how to format its output.
  87  *
  88  * The building of the AST is currently done in a module inside the compiler,
  89  * but should migrate over here as the plugin interface is defined.
  90  */

 165     pub fn parse_fmt_string(s: &str, err: ErrorFn) -> ~[Piece] {
 166         fn push_slice(ps: &mut ~[Piece], s: &str, from: uint, to: uint) {
 167             if to > from {
 168                 ps.push(PieceString(s.slice(from, to).to_owned()));
 169             }
 170         }
 171 
 172         let lim = s.len();
 173         let mut h = 0;
 174         let mut i = 0;
 175         let mut pieces = ~[];
 176 
 177         while i < lim {
 178             if s[i] == '%' as u8 {
 179                 i += 1;
 180 
 181                 if i >= lim {
 182                     err("unterminated conversion at end of string");
 183                 } else if s[i] == '%' as u8 {
 184                     push_slice(&mut pieces, s, h, i);
 185                     i += 1;
 186                 } else {
 187                     push_slice(&mut pieces, s, h, i - 1);
 188                     let Parsed {val, next} = parse_conversion(s, i, lim, err);
 189                     pieces.push(val);
 190                     i = next;
 191                 }
 192 
 193                 h = i;
 194             } else {
 195                 i += str::utf8_char_width(s[i]);
 196             }
 197         }
 198 
 199         push_slice(&mut pieces, s, h, i);
 200         pieces
 201     }

 203     pub fn peek_num(s: &str, i: uint, lim: uint) -> Option<Parsed<uint>> {
 204         let mut i = i;
 205         let mut accum = 0;
 206         let mut found = false;
 207 
 208         while i < lim {
 209             match char::to_digit(s[i] as char, 10) {
 210                 Some(x) => {
 211                     found = true;
 212                     accum *= 10;
 213                     accum += x;
 214                     i += 1;
 215                 }
 216                 None => break
 217             }
 218         }
 219 
 220         if found {
 221             Some(Parsed::new(accum, i))
 222         } else {
 223             None
 224         }
 225     }

 227     pub fn parse_conversion(s: &str, i: uint, lim: uint, err: ErrorFn) ->
 228         Parsed<Piece> {
 229         let param = parse_parameter(s, i, lim);
 230         // avoid copying ~[Flag] by destructuring
 231         let Parsed {val: flags_val, next: flags_next} = parse_flags(s,
 232             param.next, lim);
 233         let width = parse_count(s, flags_next, lim);
 234         let prec = parse_precision(s, width.next, lim);
 235         let ty = parse_type(s, prec.next, lim, err);
 236 
 237         Parsed::new(PieceConv(Conv {
 238             param: param.val,
 239             flags: flags_val,
 240             width: width.val,
 241             precision: prec.val,
 242             ty: ty.val}), ty.next)
 243     }
 244 
 245     pub fn parse_parameter(s: &str, i: uint, lim: uint) ->
 246         Parsed<Option<uint>> {
 247         if i >= lim { return Parsed::new(None, i); }
 248 
 249         match peek_num(s, i, lim) {
 250             Some(num) if num.next < lim && s[num.next] == '$' as u8 =>
 251                 Parsed::new(Some(num.val), num.next + 1),
 252             _ => Parsed::new(None, i)
 253         }
 254     }
 255 
 256     pub fn parse_flags(s: &str, i: uint, lim: uint) -> Parsed<~[Flag]> {
 257         let mut i = i;
 258         let mut flags = ~[];
 259 
 260         while i < lim {
 261             let f = match s[i] as char {
 262                 '-' => FlagLeftJustify,
 263                 '0' => FlagLeftZeroPad,
 264                 ' ' => FlagSpaceForSign,
 265                 '+' => FlagSignAlways,
 266                 '#' => FlagAlternate,
 267                 _ => break
 268             };
 269 
 270             flags.push(f);
 271             i += 1;
 272         }
 273 
 274         Parsed::new(flags, i)
 275     }
 276 
 277     pub fn parse_count(s: &str, i: uint, lim: uint) -> Parsed<Count> {
 278         if i >= lim {
 279             Parsed::new(CountImplied, i)
 280         } else if s[i] == '*' as u8 {
 281             let param = parse_parameter(s, i + 1, lim);
 282             let j = param.next;
 283 
 284             match param.val {
 285                 None => Parsed::new(CountIsNextParam, j),
 286                 Some(n) => Parsed::new(CountIsParam(n), j)
 287             }
 288         } else {
 289             match peek_num(s, i, lim) {
 290                 None => Parsed::new(CountImplied, i),
 291                 Some(num) => Parsed::new(CountIs(num.val), num.next)
 292             }
 293         }
 294     }
 295 
 296     pub fn parse_precision(s: &str, i: uint, lim: uint) -> Parsed<Count> {
 297         if i < lim && s[i] == '.' as u8 {
 298             let count = parse_count(s, i + 1, lim);
 299 
 300             // If there were no digits specified, i.e. the precision
 301             // was ".", then the precision is 0
 302             match count.val {
 303                 CountImplied => Parsed::new(CountIs(0), count.next),
 304                 _ => count
 305             }
 306         } else {
 307             Parsed::new(CountImplied, i)
 308         }
 309     }

 311     pub fn parse_type(s: &str, i: uint, lim: uint, err: ErrorFn) ->
 312         Parsed<Ty> {
 313         if i >= lim { err("missing type in conversion"); }
 314 
 315         // FIXME (#2249): Do we really want two signed types here?
 316         // How important is it to be printf compatible?
 317         let t = match s[i] as char {
 318             'b' => TyBool,
 319             's' => TyStr,
 320             'c' => TyChar,
 321             'd' | 'i' => TyInt(Signed),
 322             'u' => TyInt(Unsigned),
 323             'x' => TyHex(CaseLower),
 324             'X' => TyHex(CaseUpper),
 325             't' => TyBits,
 326             'o' => TyOctal,
 327             'f' => TyFloat,
 328             'p' => TyPointer,
 329             '?' => TyPoly,
 330             _ => err(fmt!("unknown type in conversion: %c", s.char_at(i)))
 331         };
 332 
 333         Parsed::new(t, i + 1)
 334     }

 505     pub fn conv_int(cv: Conv, i: int, buf: &mut ~str) {

  26 /// An atomically reference counted pointer.
  27 ///
  28 /// Enforces no shared-memory safety.
  29 pub struct UnsafeAtomicRcBox<T> {
  30     data: *mut libc::c_void,
  31 }
  32 
  33 struct AtomicRcBoxData<T> {
  34     count: AtomicUint,
  35     // An unwrapper uses this protocol to communicate with the "other" task that
  36     // drops the last refcount on an arc. Unfortunately this can't be a proper
  37     // pipe protocol because the unwrapper has to access both stages at once.
  38     // FIXME(#7544): Maybe use AtomicPtr instead (to avoid xchg in take() later)?
  39     unwrapper: AtomicOption<(comm::ChanOne<()>, comm::PortOne<bool>)>,
  40     // FIXME(#3224) should be able to make this non-option to save memory
  41     data: Option<T>,
  42 }

  44 unsafe fn new_inner<T: Send>(data: T, refcount: uint) -> *mut libc::c_void {
  45     let data = ~AtomicRcBoxData { count: AtomicUint::new(refcount),
  46                                   unwrapper: AtomicOption::empty(),
  47                                   data: Some(data) };
  48     cast::transmute(data)
  49 }

  52     pub fn new(data: T) -> UnsafeAtomicRcBox<T> {
  53         unsafe { UnsafeAtomicRcBox { data: new_inner(data, 1) } }
  54     }

  56     /// As new(), but returns an extra pre-cloned handle.
  57     pub fn new2(data: T) -> (UnsafeAtomicRcBox<T>, UnsafeAtomicRcBox<T>) {
  58         unsafe {
  59             let ptr = new_inner(data, 2);
  60             (UnsafeAtomicRcBox { data: ptr }, UnsafeAtomicRcBox { data: ptr })
  61         }
  62     }

  64     /// As new(), but returns a vector of as many pre-cloned handles as requested.
  65     pub fn newN(data: T, num_handles: uint) -> ~[UnsafeAtomicRcBox<T>] {
  66         unsafe {
  67             if num_handles == 0 {
  68                 ~[] // need to free data here
  69             } else {
  70                 let ptr = new_inner(data, num_handles);
  71                 vec::from_fn(num_handles, |_| UnsafeAtomicRcBox { data: ptr })
  72             }
  73         }
  74     }
  75 
  76     /// As newN(), but from an already-existing handle. Uses one xadd.
  77     pub fn cloneN(self, num_handles: uint) -> ~[UnsafeAtomicRcBox<T>] {
  78         if num_handles == 0 {
  79             ~[] // The "num_handles - 1" trick (below) fails in the 0 case.
  80         } else {
  81             unsafe {
  82                 let mut data: ~AtomicRcBoxData<T> = cast::transmute(self.data);
  83                 // Minus one because we are recycling the given handle's refcount.
  84                 let old_count = data.count.fetch_add(num_handles - 1, Acquire);
  85                 // let old_count = data.count.fetch_add(num_handles, Acquire);
  86                 assert!(old_count >= 1);
  87                 let ptr = cast::transmute(data);
  88                 cast::forget(self); // Don't run the destructor on this handle.
  89                 vec::from_fn(num_handles, |_| UnsafeAtomicRcBox { data: ptr })
  90             }
  91         }
  92     }

  94     #[inline]
  95     pub fn get(&self) -> *mut T {
  96         unsafe {
  97             let mut data: ~AtomicRcBoxData<T> = cast::transmute(self.data);
  98             assert!(data.count.load(Relaxed) > 0);
  99             let r: *mut T = data.data.get_mut_ref();
 100             cast::forget(data);
 101             return r;
 102         }
 103     }

 116     /// Wait until all other handles are dropped, then retrieve the enclosed
 117     /// data. See extra::arc::Arc for specific semantics documentation.
 118     /// If called when the task is already unkillable, unwrap will unkillably
 119     /// block; otherwise, an unwrapping task can be killed by linked failure.
 120     pub fn unwrap(self) -> T {
 121         let this = Cell::new(self); // argh
 122         do task::unkillable {
 123             unsafe {
 124                 let mut this = this.take();
 125                 let mut data: ~AtomicRcBoxData<T> = cast::transmute(this.data);
 126                 // Set up the unwrap protocol.
 127                 let (p1,c1) = comm::oneshot(); // ()
 128                 let (p2,c2) = comm::oneshot(); // bool
 129                 // Try to put our server end in the unwrapper slot.
 130                 // This needs no barrier -- it's protected by the release barrier on
 131                 // the xadd, and the acquire+release barrier in the destructor's xadd.
 132                 if data.unwrapper.fill(~(c1,p2), Relaxed).is_none() {
 133                     // Got in. Tell this handle's destructor not to run (we are now it).
 134                     this.data = ptr::mut_null();
 135                     // Drop our own reference.
 136                     let old_count = data.count.fetch_sub(1, Release);
 137                     assert!(old_count >= 1);
 138                     if old_count == 1 {
 139                         // We were the last owner. Can unwrap immediately.
 140                         // AtomicOption's destructor will free the server endpoint.
 141                         // FIXME(#3224): it should be like this
 142                         // let ~AtomicRcBoxData { data: user_data, _ } = data;
 143                         // user_data
 144                         data.data.take_unwrap()
 145                     } else {
 146                         // The *next* person who sees the refcount hit 0 will wake us.
 147                         let p1 = Cell::new(p1); // argh
 148                         // Unlike the above one, this cell is necessary. It will get
 149                         // taken either in the do block or in the finally block.
 150                         let c2_and_data = Cell::new((c2,data));
 151                         do (|| {
 152                             do task::rekillable { p1.take().recv(); }
 153                             // Got here. Back in the 'unkillable' without getting killed.
 154                             let (c2, data) = c2_and_data.take();
 155                             c2.send(true);
 156                             // FIXME(#3224): it should be like this
 157                             // let ~AtomicRcBoxData { data: user_data, _ } = data;
 158                             // user_data
 159                             let mut data = data;
 160                             data.data.take_unwrap()
 161                         }).finally {
 162                             if task::failing() {
 163                                 // Killed during wait. Because this might happen while
 164                                 // someone else still holds a reference, we can't free
 165                                 // the data now; the "other" last refcount will free it.
 166                                 let (c2, data) = c2_and_data.take();
 167                                 c2.send(false);
 168                                 cast::forget(data);
 169                             } else {
 170                                 assert!(c2_and_data.is_empty());
 171                             }
 172                         }
 173                     }
 174                 } else {
 175                     // If 'put' returns the server end back to us, we were rejected;
 176                     // someone else was trying to unwrap. Avoid guaranteed deadlock.
 177                     cast::forget(data);
 178                     fail!("Another task is already unwrapping this Arc!");
 179                 }
 180             }
 181         }
 182     }
 183 
 184     /// As unwrap above, but without blocking. Returns 'Left(self)' if this is
 185     /// not the last reference; 'Right(unwrapped_data)' if so.
 186     pub fn try_unwrap(self) -> Either<UnsafeAtomicRcBox<T>, T> {
 187         unsafe {
 188             let mut this = self; // FIXME(#4330) mutable self
 189             let mut data: ~AtomicRcBoxData<T> = cast::transmute(this.data);
 190             // This can of course race with anybody else who has a handle, but in
 191             // such a case, the returned count will always be at least 2. If we
 192             // see 1, no race was possible. All that matters is 1 or not-1.
 193             let count = data.count.load(Acquire);
 194             assert!(count >= 1);
 195             // The more interesting race is one with an unwrapper. They may have
 196             // already dropped their count -- but if so, the unwrapper pointer
 197             // will have been set first, which the barriers ensure we will see.
 198             // (Note: using is_empty(), not take(), to not free the unwrapper.)
 199             if count == 1 && data.unwrapper.is_empty(Acquire) {
 200                 // Tell this handle's destructor not to run (we are now it).
 201                 this.data = ptr::mut_null();
 202                 // FIXME(#3224) as above
 203                 Right(data.data.take_unwrap())
 204             } else {
 205                 cast::forget(data);
 206                 Left(this)
 207             }
 208         }
 209     }
 210 }

 212 impl<T: Send> Clone for UnsafeAtomicRcBox<T> {
 213     fn clone(&self) -> UnsafeAtomicRcBox<T> {
 214         unsafe {
 215             let mut data: ~AtomicRcBoxData<T> = cast::transmute(self.data);
 216             // This barrier might be unnecessary, but I'm not sure...
 217             let old_count = data.count.fetch_add(1, Acquire);
 218             assert!(old_count >= 1);
 219             cast::forget(data);
 220             return UnsafeAtomicRcBox { data: self.data };
 221         }
 222     }
 223 }

 225 #[unsafe_destructor]
 226 impl<T> Drop for UnsafeAtomicRcBox<T>{
 227     fn drop(&self) {
 228         unsafe {
 229             if self.data.is_null() {
 230                 return; // Happens when destructing an unwrapper's handle.
 231             }
 232             do task::unkillable {
 233                 let mut data: ~AtomicRcBoxData<T> = cast::transmute(self.data);
 234                 // Must be acquire+release, not just release, to make sure this
 235                 // doesn't get reordered to after the unwrapper pointer load.
 236                 let old_count = data.count.fetch_sub(1, SeqCst);
 237                 assert!(old_count >= 1);
 238                 if old_count == 1 {
 239                     // Were we really last, or should we hand off to an
 240                     // unwrapper? It's safe to not xchg because the unwrapper
 241                     // will set the unwrap lock *before* dropping his/her
 242                     // reference. In effect, being here means we're the only
 243                     // *awake* task with the data.
 244                     match data.unwrapper.take(Acquire) {
 245                         Some(~(message,response)) => {
 246                             // Send 'ready' and wait for a response.
 247                             message.send(());
 248                             // Unkillable wait. Message guaranteed to come.
 249                             if response.recv() {
 250                                 // Other task got the data.
 251                                 cast::forget(data);
 252                             } else {
 253                                 // Other task was killed. drop glue takes over.
 254                             }
 255                         }
 256                         None => {
 257                             // drop glue takes over.
 258                         }
 259                     }
 260                 } else {
 261                     cast::forget(data);
 262                 }
 263             }
 264         }
 265     }
 266 }

 271 /**
 272  * Enables a runtime assertion that no operation in the argument closure shall
 273  * use scheduler operations (yield, recv, spawn, etc). This is for use with
 274  * pthread mutexes, which may block the entire scheduler thread, rather than
 275  * just one task, and is hence prone to deadlocks if mixed with yielding.
 276  *
 277  * NOTE: THIS DOES NOT PROVIDE LOCKING, or any sort of critical-section
 278  * synchronization whatsoever. It only makes sense to use for CPU-local issues.
 279  */
 280 // FIXME(#8140) should not be pub
 281 pub unsafe fn atomically<U>(f: &fn() -> U) -> U {
 282     use rt::task::Task;
 283     use task::rt;
 284     use rt::local::Local;
 285     use rt::{context, OldTaskContext, TaskContext};
 286 
 287     match context() {
 288         OldTaskContext => {
 289             let t = rt::rust_get_task();
 290             do (|| {
 291                 rt::rust_task_inhibit_kill(t);
 292                 rt::rust_task_inhibit_yield(t);
 293                 f()
 294             }).finally {
 295                 rt::rust_task_allow_yield(t);
 296                 rt::rust_task_allow_kill(t);
 297             }
 298         }
 299         TaskContext => {
 300             let t = Local::unsafe_borrow::<Task>();
 301             do (|| {
 302                 (*t).death.inhibit_yield();
 303                 f()
 304             }).finally {
 305                 (*t).death.allow_yield();
 306             }
 307         }
 308         // FIXME(#3095): As in unkillable().
 309         _ => f()
 310     }
 311 }

 356 /**
 357  * An arc over mutable data that is protected by a lock. For library use only.
 358  *
 359  * # Safety note
 360  *
 361  * This uses a pthread mutex, not one that's aware of the userspace scheduler.
 362  * The user of an Exclusive must be careful not to invoke any functions that may
 363  * reschedule the task while holding the lock, or deadlock may result. If you
 364  * need to block or yield while accessing shared state, use extra::sync::RWArc.
 365  */
 366 pub struct Exclusive<T> {
 367     x: UnsafeAtomicRcBox<ExData<T>>
 368 }

 372     fn clone(&self) -> Exclusive<T> {
 373         Exclusive { x: self.x.clone() }
 374     }
 375 }

 377 impl<T:Send> Exclusive<T> {
 378     pub fn new(user_data: T) -> Exclusive<T> {
 379         let data = ExData {
 380             lock: LittleLock::new(),
 381             failed: false,
 382             data: user_data
 383         };
 384         Exclusive {
 385             x: UnsafeAtomicRcBox::new(data)
 386         }
 387     }

 389     // Exactly like std::arc::MutexArc,access(), but with the LittleLock
 390     // instead of a proper mutex. Same reason for being unsafe.
 391     //
 392     // Currently, scheduling operations (i.e., yielding, receiving on a pipe,
 393     // accessing the provided condition variable) are prohibited while inside
 394     // the Exclusive. Supporting that is a work in progress.
 395     #[inline]
 396     pub unsafe fn with<U>(&self, f: &fn(x: &mut T) -> U) -> U {
 397         let rec = self.x.get();
 398         do (*rec).lock.lock {
 399             if (*rec).failed {
 400                 fail!("Poisoned Exclusive::new - another task failed inside!");
 401             }
 402             (*rec).failed = true;
 403             let result = f(&mut (*rec).data);
 404             (*rec).failed = false;
 405             result
 406         }
 407     }

 411         do self.with |x| {
 412             f(cast::transmute_immut(x))
 413         }
 414     }

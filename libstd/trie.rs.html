  51 impl<T> Map<uint, T> for TrieMap<T> {
  52     /// Return a reference to the value corresponding to the key
  53     #[inline]
  54     fn find<'a>(&'a self, key: &uint) -> Option<&'a T> {
  55         let mut node: &'a TrieNode<T> = &self.root;
  56         let mut idx = 0;
  57         loop {
  58             match node.children[chunk(*key, idx)] {
  59               Internal(ref x) => node = &**x,
  60               External(stored, ref value) => {
  61                 if stored == *key {
  62                     return Some(value)
  63                 } else {
  64                     return None
  65                 }
  66               }
  67               Nothing => return None
  68             }
  69             idx += 1;
  70         }
  71     }
  72 }

  74 impl<T> MutableMap<uint, T> for TrieMap<T> {
  75     /// Return a mutable reference to the value corresponding to the key
  76     #[inline]
  77     fn find_mut<'a>(&'a mut self, key: &uint) -> Option<&'a mut T> {
  78         find_mut(&mut self.root.children[chunk(*key, 0)], *key, 1)
  79     }
  80 
  81     /// Insert a key-value pair from the map. If the key already had a value
  82     /// present in the map, that value is returned. Otherwise None is returned.
  83     fn swap(&mut self, key: uint, value: T) -> Option<T> {
  84         let ret = insert(&mut self.root.count,
  85                          &mut self.root.children[chunk(key, 0)],
  86                          key, value, 1);
  87         if ret.is_none() { self.length += 1 }
  88         ret
  89     }
  90 
  91     /// Removes a key from the map, returning the value at the key if the key
  92     /// was previously in the map.
  93     fn pop(&mut self, key: &uint) -> Option<T> {
  94         let ret = remove(&mut self.root.count,
  95                          &mut self.root.children[chunk(*key, 0)],
  96                          *key, 1);
  97         if ret.is_some() { self.length -= 1 }
  98         ret
  99     }
 100 }

 254 struct TrieNode<T> {

 265                  children: [Nothing, Nothing, Nothing, Nothing,
 266                             Nothing, Nothing, Nothing, Nothing,

 268                             Nothing, Nothing, Nothing, Nothing]}
 269     }

 272 impl<T> TrieNode<T> {
 273     fn each<'a>(&'a self, f: &fn(&uint, &'a T) -> bool) -> bool {
 274         for elt in self.children.iter() {
 275             match *elt {
 276                 Internal(ref x) => if !x.each(|i,t| f(i,t)) { return false },
 277                 External(k, ref v) => if !f(&k, v) { return false },
 278                 Nothing => ()
 279             }
 280         }
 281         true
 282     }
 283 
 284     fn each_reverse<'a>(&'a self, f: &fn(&uint, &'a T) -> bool) -> bool {
 285         for elt in self.children.rev_iter() {
 286             match *elt {
 287                 Internal(ref x) => if !x.each_reverse(|i,t| f(i,t)) { return false },
 288                 External(k, ref v) => if !f(&k, v) { return false },
 289                 Nothing => ()
 290             }
 291         }
 292         true
 293     }
 294 
 295     fn mutate_values<'a>(&'a mut self, f: &fn(&uint, &mut T) -> bool) -> bool {
 296         for child in self.children.mut_iter() {
 297             match *child {
 298                 Internal(ref mut x) => if !x.mutate_values(|i,t| f(i,t)) {
 299                     return false
 300                 },
 301                 External(k, ref mut v) => if !f(&k, v) { return false },
 302                 Nothing => ()
 303             }
 304         }
 305         true
 306     }
 307 }

 309 // if this was done via a trait, the key could be generic
 310 #[inline]
 311 fn chunk(n: uint, idx: uint) -> uint {
 312     let sh = uint::bits - (SHIFT * (idx + 1));
 313     (n >> sh) & MASK
 314 }
 315 
 316 fn find_mut<'r, T>(child: &'r mut Child<T>, key: uint, idx: uint) -> Option<&'r mut T> {
 317     match *child {
 318         External(_, ref mut value) => Some(value),
 319         Internal(ref mut x) => find_mut(&mut x.children[chunk(key, idx)], key, idx + 1),
 320         Nothing => None
 321     }
 322 }

 324 fn insert<T>(count: &mut uint, child: &mut Child<T>, key: uint, value: T,
 325              idx: uint) -> Option<T> {
 326     let mut tmp = Nothing;
 327     let ret;
 328     swap(&mut tmp, child);
 329 
 330     *child = match tmp {
 331       External(stored_key, stored_value) => {
 332           if stored_key == key {
 333               ret = Some(stored_value);
 334               External(stored_key, value)
 335           } else {
 336               // conflict - split the node
 337               let mut new = ~TrieNode::new();
 338               insert(&mut new.count,
 339                      &mut new.children[chunk(stored_key, idx)],
 340                      stored_key, stored_value, idx + 1);
 341               ret = insert(&mut new.count, &mut new.children[chunk(key, idx)],
 342                            key, value, idx + 1);
 343               Internal(new)
 344           }
 345       }
 346       Internal(x) => {
 347         let mut x = x;
 348         ret = insert(&mut x.count, &mut x.children[chunk(key, idx)], key,
 349                      value, idx + 1);
 350         Internal(x)
 351       }
 352       Nothing => {
 353         *count += 1;
 354         ret = None;
 355         External(key, value)
 356       }
 357     };
 358     return ret;
 359 }
 360 
 361 fn remove<T>(count: &mut uint, child: &mut Child<T>, key: uint,
 362              idx: uint) -> Option<T> {
 363     let (ret, this) = match *child {
 364       External(stored, _) if stored == key => {
 365         match replace(child, Nothing) {
 366             External(_, value) => (Some(value), true),
 367             _ => fail!()
 368         }
 369       }
 370       External(*) => (None, false),
 371       Internal(ref mut x) => {
 372           let ret = remove(&mut x.count, &mut x.children[chunk(key, idx)],
 373                            key, idx + 1);
 374           (ret, x.count == 0)
 375       }
 376       Nothing => (None, false)
 377     };
 378 
 379     if this {
 380         *child = Nothing;
 381         *count -= 1;
 382     }
 383     return ret;
 384 }

 386 /// Forward iterator over a map
 387 pub struct TrieMapIterator<'self, T> {
 388     priv stack: ~[vec::VecIterator<'self, Child<T>>],
 389     priv remaining_min: uint,
 390     priv remaining_max: uint
 391 }
 392 
 393 impl<'self, T> Iterator<(uint, &'self T)> for TrieMapIterator<'self, T> {
 394     fn next(&mut self) -> Option<(uint, &'self T)> {
 395         while !self.stack.is_empty() {
 396             match self.stack[self.stack.len() - 1].next() {
 397                 None => {
 398                     self.stack.pop();
 399                 }
 400                 Some(ref child) => {
 401                     match **child {
 402                         Internal(ref node) => {
 403                             self.stack.push(node.children.iter());
 404                         }
 405                         External(key, ref value) => {
 406                             self.remaining_max -= 1;
 407                             if self.remaining_min > 0 {
 408                                 self.remaining_min -= 1;
 409                             }
 410                             return Some((key, value));
 411                         }
 412                         Nothing => {}
 413                     }
 414                 }
 415             }
 416         }
 417         return None;
 418     }
 419 
 420     #[inline]
 421     fn size_hint(&self) -> (uint, Option<uint>) {
 422         (self.remaining_min, Some(self.remaining_max))
 423     }
 424 }

 441 #[cfg(test)]
 442 pub fn check_integrity<T>(trie: &TrieNode<T>) {
 443     assert!(trie.count != 0);
 444 
 445     let mut sum = 0;
 446 
 447     for x in trie.children.iter() {
 448         match *x {
 449           Nothing => (),
 450           Internal(ref y) => {
 451               check_integrity(&**y);
 452               sum += 1
 453           }
 454           External(_, _) => { sum += 1 }
 455         }
 456     }
 457 
 458     assert_eq!(sum, trie.count);
 459 }

  11  /*!
  12  
  13  C-string manipulation and management
  14  
  15  This modules provides the basic methods for creating and manipulating
  16  null-terminated strings for use with FFI calls (back to C). Most C APIs require
  17  that the string being passed to them is null-terminated, and by default rust's
  18  string types are *not* null terminated.
  19  
  20  The other problem with translating Rust strings to C strings is that Rust
  21  strings can validly contain a null-byte in the middle of the string (0 is a
  22  valid unicode codepoint). This means that not all Rust strings can actually be
  23  translated to C strings.
  24  
  25  # Creation of a C string
  26  
  27  A C string is managed through the `CString` type defined in this module. It
  28  "owns" the internal buffer of characters and will automatically deallocate the
  29  buffer when the string is dropped. The `ToCStr` trait is implemented for `&str`
  30  and `&[u8]`, but the conversions can fail due to some of the limitations
  31  explained above.
  32  
  33  This also means that currently whenever a C string is created, an allocation
  34  must be performed to place the data elsewhere (the lifetime of the C string is
  35  not tied to the lifetime of the original string/data buffer). If C strings are
  36  heavily used in applications, then caching may be advisable to prevent
  37  unnecessary amounts of allocations.
  38  
  39  An example of creating and using a C string would be:
  40  
  41  ```rust
  42  use std::libc;
  43  extern {
  44      fn puts(s: *libc::c_char);
  45  }
  46  
  47  let my_string = "Hello, world!";
  48  
  49  // Allocate the C string with an explicit local that owns the string. The
  50  // `c_buffer` pointer will be deallocated when `my_c_string` goes out of scope.
  51  let my_c_string = my_string.to_c_str();
  52  my_c_string.with_ref(|c_buffer| {
  53      unsafe { puts(c_buffer); }
  54  });
  55  
  56  // Don't save off the allocation of the C string, the `c_buffer` will be
  57  // deallocated when this block returns!
  58  my_string.with_c_str(|c_buffer| {
  59      unsafe { puts(c_buffer); }
  60  });
  61   ```
  62  
  63  */

  84  /// The representation of a C String.
  85  ///
  86  /// This structure wraps a `*libc::c_char`, and will automatically free the
  87  /// memory it is pointing to when it goes out of scope.
  88  pub struct CString {
  89      priv buf: *libc::c_char,
  90      priv owns_buffer_: bool,
  91  }
  92  
  93  impl Clone for CString {
  94      /// Clone this CString into a new, uniquely owned CString. For safety
  95      /// reasons, this is always a deep clone, rather than the usual shallow
  96      /// clone.
  97      fn clone(&self) -> CString {
  98          if self.buf.is_null() {
  99              CString { buf: self.buf, owns_buffer_: self.owns_buffer_ }
 100          } else {
 101              let len = self.len() + 1;
 102              let buf = unsafe { malloc_raw(len) } as *mut libc::c_char;
 103              unsafe { ptr::copy_nonoverlapping_memory(buf, self.buf, len); }
 104              CString { buf: buf as *libc::c_char, owns_buffer_: true }
 105          }
 106      }
 107  }

 109  impl Eq for CString {
 110      fn eq(&self, other: &CString) -> bool {
 111          if self.buf as uint == other.buf as uint {
 112              true
 113          } else if self.buf.is_null() || other.buf.is_null() {
 114              false
 115          } else {
 116              unsafe {
 117                  libc::strcmp(self.buf, other.buf) == 0
 118              }
 119          }
 120      }
 121  }

 251  /// A generic trait for converting a value to a CString.
 252  pub trait ToCStr {
 253      /// Copy the receiver into a CString.
 254      ///
 255      /// # Failure
 256      ///
 257      /// Fails the task if the receiver has an interior null.
 258      fn to_c_str(&self) -> CString;
 259  
 260      /// Unsafe variant of `to_c_str()` that doesn't check for nulls.
 261      unsafe fn to_c_str_unchecked(&self) -> CString;
 262  
 263      /// Work with a temporary CString constructed from the receiver.
 264      /// The provided `*libc::c_char` will be freed immediately upon return.
 265      ///
 266      /// # Example
 267      ///
 268      /// ```rust
 269      /// use std::libc;
 270      ///
 271      /// let s = "PATH".with_c_str(|path| unsafe {
 272      ///     libc::getenv(path)
 273      /// });
 274      /// ```
 275      ///
 276      /// # Failure
 277      ///
 278      /// Fails the task if the receiver has an interior null.
 279      #[inline]
 280      fn with_c_str<T>(&self, f: |*libc::c_char| -> T) -> T {
 281          self.to_c_str().with_ref(f)
 282      }
 283  
 284      /// Unsafe variant of `with_c_str()` that doesn't check for nulls.
 285      #[inline]
 286      unsafe fn with_c_str_unchecked<T>(&self, f: |*libc::c_char| -> T) -> T {
 287          self.to_c_str_unchecked().with_ref(f)
 288      }
 289  }

 313  // The length of the stack allocated buffer for `vec.with_c_str()`
 314  static BUF_LEN: uint = 128;
 315  
 316  impl<'a> ToCStr for &'a [u8] {
 317      fn to_c_str(&self) -> CString {
 318          let mut cs = unsafe { self.to_c_str_unchecked() };
 319          cs.with_mut_ref(|buf| check_for_null(*self, buf));
 320          cs
 321      }
 322  
 323      unsafe fn to_c_str_unchecked(&self) -> CString {
 324          let self_len = self.len();
 325          let buf = malloc_raw(self_len + 1);
 326  
 327          ptr::copy_memory(buf, self.as_ptr(), self_len);
 328          *buf.offset(self_len as int) = 0;
 329  
 330          CString::new(buf as *libc::c_char, true)
 331      }
 332  
 333      fn with_c_str<T>(&self, f: |*libc::c_char| -> T) -> T {
 334          unsafe { with_c_str(*self, true, f) }
 335      }
 336  
 337      unsafe fn with_c_str_unchecked<T>(&self, f: |*libc::c_char| -> T) -> T {
 338          with_c_str(*self, false, f)
 339      }
 340  }
 341  
 342  // Unsafe function that handles possibly copying the &[u8] into a stack array.
 343  unsafe fn with_c_str<T>(v: &[u8], checked: bool, f: |*libc::c_char| -> T) -> T {
 344      if v.len() < BUF_LEN {
 345          let mut buf: [u8, .. BUF_LEN] = mem::uninit();
 346          slice::bytes::copy_memory(buf, v);
 347          buf[v.len()] = 0;
 348  
 349          let buf = buf.as_mut_ptr();
 350          if checked {
 351              check_for_null(v, buf as *mut libc::c_char);
 352          }
 353  
 354          f(buf as *libc::c_char)
 355      } else if checked {
 356          v.to_c_str().with_ref(f)
 357      } else {
 358          v.to_c_str_unchecked().with_ref(f)
 359      }
 360  }
 361  
 362  #[inline]
 363  fn check_for_null(v: &[u8], buf: *mut libc::c_char) {
 364      for i in range(0, v.len()) {
 365          unsafe {
 366              let p = buf.offset(i as int);
 367              assert!(*p != 0);
 368          }
 369      }
 370  }

 372  /// External iterator for a CString's bytes.
 373  ///
 374  /// Use with the `std::iter` module.
 375  pub struct CChars<'a> {
 376      priv ptr: *libc::c_char,
 377      priv marker: marker::ContravariantLifetime<'a>,
 378  }
 379  
 380  impl<'a> Iterator<libc::c_char> for CChars<'a> {
 381      fn next(&mut self) -> Option<libc::c_char> {
 382          let ch = unsafe { *self.ptr };
 383          if ch == 0 {
 384              None
 385          } else {
 386              self.ptr = unsafe { self.ptr.offset(1) };
 387              Some(ch)
 388          }
 389      }
 390  }

 392  /// Parses a C "multistring", eg windows env values or
 393  /// the req->ptr result in a uv_fs_readdir() call.
 394  ///
 395  /// Optionally, a `count` can be passed in, limiting the
 396  /// parsing to only being done `count`-times.
 397  ///
 398  /// The specified closure is invoked with each string that
 399  /// is found, and the number of strings found is returned.
 400  pub unsafe fn from_c_multistring(buf: *libc::c_char,
 401                                   count: Option<uint>,
 402                                   f: |&CString|) -> uint {
 403  
 404      let mut curr_ptr: uint = buf as uint;
 405      let mut ctr = 0;
 406      let (limited_count, limit) = match count {
 407          Some(limit) => (true, limit),
 408          None => (false, 0)
 409      };
 410      while ((limited_count && ctr < limit) || !limited_count)
 411            && *(curr_ptr as *libc::c_char) != 0 as libc::c_char {
 412          let cstr = CString::new(curr_ptr as *libc::c_char, false);
 413          f(&cstr);
 414          curr_ptr += cstr.len() + 1;
 415          ctr += 1;
 416      }
 417      return ctr;
 418  }

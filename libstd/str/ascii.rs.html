 291  /// Convert the string to ASCII upper case:
 292  /// ASCII letters 'a' to 'z' are mapped to 'A' to 'Z',
 293  /// but non-ASCII letters are unchanged.
 294  #[inline]

 296      map_bytes(string, ASCII_UPPER_MAP)
 297  }
 298  
 299  /// Convert the string to ASCII lower case:
 300  /// ASCII letters 'A' to 'Z' are mapped to 'a' to 'z',
 301  /// but non-ASCII letters are unchanged.
 302  #[inline]
 303  pub fn to_ascii_lower(string: &str) -> ~str {
 304      map_bytes(string, ASCII_LOWER_MAP)
 305  }
 306  
 307  #[inline]
 308  fn map_bytes(string: &str, map: &'static [u8]) -> ~str {
 309      let len = string.len();
 310      let mut result = str::with_capacity(len);
 311      unsafe {
 312          do result.as_mut_buf |mut buf, _| {
 313              for c in string.as_bytes().iter() {
 314                  *buf = map[*c];
 315                  buf = ptr::mut_offset(buf, 1)
 316              }
 317          }
 318          str::raw::set_len(&mut result, len);
 319      }
 320      result
 321  }

 328      a.len() == b.len() && a.as_bytes().iter().zip(b.as_bytes().iter()).all(
 329          |(byte_a, byte_b)| ASCII_LOWER_MAP[*byte_a] == ASCII_LOWER_MAP[*byte_b])

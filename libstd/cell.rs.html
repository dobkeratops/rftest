  70  /// A mutable memory location with dynamically checked borrow rules
  71  pub struct RefCell<T> {

  73      priv borrow: BorrowFlag,
  74      priv nopod: marker::NoPod,
  75      priv noshare: marker::NoShare,

  77  
  78  // Values [1, MAX-1] represent the number of `Ref` active

 105      /// Attempts to immutably borrow the wrapped value.
 106      ///
 107      /// The borrow lasts until the returned `Ref` exits scope. Multiple
 108      /// immutable borrows can be taken out at the same time.
 109      ///
 110      /// Returns `None` if the value is currently mutably borrowed.
 111      pub fn try_borrow<'a>(&'a self) -> Option<Ref<'a, T>> {
 112          match self.borrow {
 113              WRITING => None,
 114              _ => {
 115                  unsafe { self.as_mut().borrow += 1; }
 116                  Some(Ref { parent: self })
 117              }
 118          }
 119      }
 120  
 121      /// Immutably borrows the wrapped value.
 122      ///
 123      /// The borrow lasts until the returned `Ref` exits scope. Multiple
 124      /// immutable borrows can be taken out at the same time.
 125      ///
 126      /// # Failure
 127      ///
 128      /// Fails if the value is currently mutably borrowed.
 129      pub fn borrow<'a>(&'a self) -> Ref<'a, T> {
 130          match self.try_borrow() {
 131              Some(ptr) => ptr,
 132              None => fail!("RefCell<T> already mutably borrowed")
 133          }
 134      }

 136      /// Mutably borrows the wrapped value.
 137      ///
 138      /// The borrow lasts until the returned `RefMut` exits scope. The value
 139      /// cannot be borrowed while this borrow is active.
 140      ///
 141      /// Returns `None` if the value is currently borrowed.
 142      pub fn try_borrow_mut<'a>(&'a self) -> Option<RefMut<'a, T>> {
 143          match self.borrow {
 144              UNUSED => unsafe {
 145                  let mut_self = self.as_mut();
 146                  mut_self.borrow = WRITING;
 147                  Some(RefMut { parent: mut_self })
 148              },
 149              _ => None
 150          }
 151      }
 152  
 153      /// Mutably borrows the wrapped value.
 154      ///
 155      /// The borrow lasts until the returned `RefMut` exits scope. The value
 156      /// cannot be borrowed while this borrow is active.
 157      ///
 158      /// # Failure

 161      pub fn borrow_mut<'a>(&'a self) -> RefMut<'a, T> {

 165          }
 166      }

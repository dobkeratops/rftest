   11 //! An unordered map and set type implemented as hash tables
   12 //!
   13 //! The tables use a keyed hash with new random keys generated for each container, so the ordering
   14 //! of a set of keys in a hash table is randomized.
   15

   17

   32 static INITIAL_CAPACITY: uint = 32u; // 2^5
   33 
   34 struct Bucket<K,V> {
   35     hash: uint,
   36     key: K,
   37     value: V,
   38 }
   39 
   40 /// A hash map implementation which uses linear probing along with the SipHash
   41 /// hash function for internal state. This means that the order of all hash maps
   42 /// is randomized by keying each hash map randomly on creation.
   43 ///
   44 /// It is required that the keys implement the `Eq` and `Hash` traits, although
   45 /// this can frequently be achieved by just implementing the `Eq` and
   46 /// `IterBytes` traits as `Hash` is automatically implemented for types that
   47 /// implement `IterBytes`.
   48 pub struct HashMap<K,V> {
   49     priv k0: u64,
   50     priv k1: u64,
   51     priv resize_at: uint,
   52     priv size: uint,
   53     priv buckets: ~[Option<Bucket<K, V>>],
   54 }
   55 
   56 // We could rewrite FoundEntry to have type Option<&Bucket<K, V>>
   57 // which would be nifty
   58 enum SearchResult {
   59     FoundEntry(uint), FoundHole(uint), TableFull
   60 }
   61 
   62 #[inline]
   63 fn resize_at(capacity: uint) -> uint {
   64     (capacity * 3) / 4
   65 }
   66 
   67 impl<K:Hash + Eq,V> HashMap<K, V> {
   68     #[inline]
   69     fn to_bucket(&self, h: uint) -> uint {
   70         // A good hash function with entropy spread over all of the
   71         // bits is assumed. SipHash is more than good enough.
   72         h % self.buckets.len()
   73     }
   74 
   75     #[inline]
   76     fn next_bucket(&self, idx: uint, len_buckets: uint) -> uint {
   77         (idx + 1) % len_buckets
   78     }
   79 
   80     #[inline]
   81     fn bucket_sequence(&self, hash: uint,
   82                        op: &fn(uint) -> bool) -> bool {
   83         let start_idx = self.to_bucket(hash);
   84         let len_buckets = self.buckets.len();
   85         let mut idx = start_idx;
   86         loop {
   87             if !op(idx) { return false; }
   88             idx = self.next_bucket(idx, len_buckets);
   89             if idx == start_idx {
   90                 return true;
   91             }
   92         }
   93     }
   94 
   95     #[inline]
   96     fn bucket_for_key(&self, k: &K) -> SearchResult {
   97         let hash = k.hash_keyed(self.k0, self.k1) as uint;
   98         self.bucket_for_key_with_hash(hash, k)
   99     }
  100 
  101     #[inline]
  102     fn bucket_for_key_equiv<Q:Hash + Equiv<K>>(&self, k: &Q)
  103                                                -> SearchResult {
  104         let hash = k.hash_keyed(self.k0, self.k1) as uint;
  105         self.bucket_for_key_with_hash_equiv(hash, k)
  106     }
  107 
  108     #[inline]
  109     fn bucket_for_key_with_hash(&self,
  110                                 hash: uint,
  111                                 k: &K)
  112                              -> SearchResult {
  113         let mut ret = TableFull;
  114         do self.bucket_sequence(hash) |i| {
  115             match self.buckets[i] {
  116                 Some(ref bkt) if bkt.hash == hash && *k == bkt.key => {
  117                     ret = FoundEntry(i); false
  118                 },
  119                 None => { ret = FoundHole(i); false }
  120                 _ => true,
  121             }
  122         };
  123         ret
  124     }
  125 
  126     #[inline]
  127     fn bucket_for_key_with_hash_equiv<Q:Equiv<K>>(&self,
  128                                                   hash: uint,
  129                                                   k: &Q)
  130                                                -> SearchResult {
  131         let mut ret = TableFull;
  132         do self.bucket_sequence(hash) |i| {
  133             match self.buckets[i] {
  134                 Some(ref bkt) if bkt.hash == hash && k.equiv(&bkt.key) => {
  135                     ret = FoundEntry(i); false
  136                 },
  137                 None => { ret = FoundHole(i); false }
  138                 _ => true,
  139             }
  140         };
  141         ret
  142     }
  143 
  144     /// Expand the capacity of the array to the next power of two
  145     /// and re-insert each of the existing buckets.
  146     #[inline]
  147     fn expand(&mut self) {
  148         let new_capacity = self.buckets.len() * 2;
  149         self.resize(new_capacity);
  150     }
  151 
  152     /// Expands the capacity of the array and re-insert each of the
  153     /// existing buckets.
  154     fn resize(&mut self, new_capacity: uint) {
  155         self.resize_at = resize_at(new_capacity);
  156 
  157         let old_buckets = replace(&mut self.buckets,
  158                                   vec::from_fn(new_capacity, |_| None));
  159 
  160         self.size = 0;
  161         // consume_rev_iter is more efficient
  162         for bucket in old_buckets.consume_rev_iter() {
  163             self.insert_opt_bucket(bucket);
  164         }
  165     }
  166 
  167     fn insert_opt_bucket(&mut self, bucket: Option<Bucket<K, V>>) {
  168         match bucket {
  169             Some(Bucket{hash: hash, key: key, value: value}) => {
  170                 self.insert_internal(hash, key, value);
  171             }
  172             None => {}
  173         }
  174     }
  175 
  176     #[inline]
  177     fn value_for_bucket<'a>(&'a self, idx: uint) -> &'a V {
  178         match self.buckets[idx] {
  179             Some(ref bkt) => &bkt.value,
  180             None => fail!("HashMap::find: internal logic error"),
  181         }
  182     }
  183 
  184     #[inline]
  185     fn mut_value_for_bucket<'a>(&'a mut self, idx: uint) -> &'a mut V {
  186         match self.buckets[idx] {
  187             Some(ref mut bkt) => &mut bkt.value,
  188             None => unreachable()
  189         }
  190     }
  191 
  192     /// Inserts the key value pair into the buckets.
  193     /// Assumes that there will be a bucket.
  194     /// True if there was no previous entry with that key
  195     fn insert_internal(&mut self, hash: uint, k: K, v: V) -> Option<V> {
  196         match self.bucket_for_key_with_hash(hash, &k) {
  197             TableFull => { fail!("Internal logic error"); }
  198             FoundHole(idx) => {
  199                 self.buckets[idx] = Some(Bucket{hash: hash, key: k,
  200                                                 value: v});
  201                 self.size += 1;
  202                 None
  203             }
  204             FoundEntry(idx) => {
  205                 match self.buckets[idx] {
  206                     None => { fail!("insert_internal: Internal logic error") }
  207                     Some(ref mut b) => {
  208                         b.hash = hash;
  209                         b.key = k;
  210                         Some(replace(&mut b.value, v))
  211                     }
  212                 }
  213             }
  214         }
  215     }
  216 
  217     fn pop_internal(&mut self, hash: uint, k: &K) -> Option<V> {
  218         // Removing from an open-addressed hashtable
  219         // is, well, painful.  The problem is that
  220         // the entry may lie on the probe path for other
  221         // entries, so removing it would make you think that
  222         // those probe paths are empty.
  223         //
  224         // To address this we basically have to keep walking,
  225         // re-inserting entries we find until we reach an empty
  226         // bucket.  We know we will eventually reach one because
  227         // we insert one ourselves at the beginning (the removed
  228         // entry).
  229         //
  230         // I found this explanation elucidating:
  231         // http://www.maths.lse.ac.uk/Courses/MA407/del-hash.pdf
  232         let mut idx = match self.bucket_for_key_with_hash(hash, k) {
  233             TableFull | FoundHole(_) => return None,
  234             FoundEntry(idx) => idx
  235         };
  236 
  237         let len_buckets = self.buckets.len();
  238         let bucket = self.buckets[idx].take();
  239 
  240         let value = do bucket.map_consume |bucket| {
  241             bucket.value
  242         };
  243 
  244         /* re-inserting buckets may cause changes in size, so remember
  245         what our new size is ahead of time before we start insertions */
  246         let size = self.size - 1;
  247         idx = self.next_bucket(idx, len_buckets);
  248         while self.buckets[idx].is_some() {
  249             let bucket = self.buckets[idx].take();
  250             self.insert_opt_bucket(bucket);
  251             idx = self.next_bucket(idx, len_buckets);
  252         }
  253         self.size = size;
  254 
  255         value
  256     }
  257 }

  262 }

  264 impl<K:Hash + Eq,V> Mutable for HashMap<K, V> {
  265     /// Clear the map, removing all key-value pairs.
  266     fn clear(&mut self) {
  267         for idx in range(0u, self.buckets.len()) {
  268             self.buckets[idx] = None;
  269         }
  270         self.size = 0;
  271     }
  272 }

  274 impl<K:Hash + Eq,V> Map<K, V> for HashMap<K, V> {
  275     /// Return a reference to the value corresponding to the key
  276     fn find<'a>(&'a self, k: &K) -> Option<&'a V> {
  277         match self.bucket_for_key(k) {
  278             FoundEntry(idx) => Some(self.value_for_bucket(idx)),
  279             TableFull | FoundHole(_) => None,
  280         }
  281     }
  282 }
  283 
  284 impl<K:Hash + Eq,V> MutableMap<K, V> for HashMap<K, V> {
  285     /// Return a mutable reference to the value corresponding to the key
  286     fn find_mut<'a>(&'a mut self, k: &K) -> Option<&'a mut V> {
  287         let idx = match self.bucket_for_key(k) {
  288             FoundEntry(idx) => idx,
  289             TableFull | FoundHole(_) => return None
  290         };
  291         Some(self.mut_value_for_bucket(idx))
  292     }
  293 
  294     /// Insert a key-value pair from the map. If the key already had a value
  295     /// present in the map, that value is returned. Otherwise None is returned.
  296     fn swap(&mut self, k: K, v: V) -> Option<V> {
  297         // this could be faster.
  298 
  299         if self.size >= self.resize_at {
  300             // n.b.: We could also do this after searching, so
  301             // that we do not resize if this call to insert is
  302             // simply going to update a key in place.  My sense
  303             // though is that it's worse to have to search through
  304             // buckets to find the right spot twice than to just
  305             // resize in this corner case.
  306             self.expand();
  307         }
  308 
  309         let hash = k.hash_keyed(self.k0, self.k1) as uint;
  310         self.insert_internal(hash, k, v)
  311     }
  312 
  313     /// Removes a key from the map, returning the value at the key if the key
  314     /// was previously in the map.
  315     fn pop(&mut self, k: &K) -> Option<V> {
  316         let hash = k.hash_keyed(self.k0, self.k1) as uint;
  317         self.pop_internal(hash, k)
  318     }
  319 }

  321 impl<K: Hash + Eq, V> HashMap<K, V> {
  322     /// Create an empty HashMap
  323     pub fn new() -> HashMap<K, V> {
  324         HashMap::with_capacity(INITIAL_CAPACITY)
  325     }
  326 
  327     /// Create an empty HashMap with space for at least `capacity`
  328     /// elements in the hash table.
  329     pub fn with_capacity(capacity: uint) -> HashMap<K, V> {
  330         let mut r = rand::task_rng();
  331         HashMap::with_capacity_and_keys(r.gen(), r.gen(), capacity)
  332     }
  333 
  334     /// Create an empty HashMap with space for at least `capacity`
  335     /// elements, using `k0` and `k1` as the keys.
  336     ///
  337     /// Warning: `k0` and `k1` are normally randomly generated, and
  338     /// are designed to allow HashMaps to be resistant to attacks that
  339     /// cause many collisions and very poor performance. Setting them
  340     /// manually using this function can expose a DoS attack vector.
  341     pub fn with_capacity_and_keys(k0: u64, k1: u64, capacity: uint) -> HashMap<K, V> {
  342         let cap = num::max(INITIAL_CAPACITY, capacity);
  343         HashMap {
  344             k0: k0, k1: k1,
  345             resize_at: resize_at(cap),
  346             size: 0,
  347             buckets: vec::from_fn(cap, |_| None)
  348         }
  349     }
  350 
  351     /// Reserve space for at least `n` elements in the hash table.
  352     pub fn reserve_at_least(&mut self, n: uint) {
  353         if n > self.buckets.len() {
  354             let buckets = n * 4 / 3 + 1;
  355             self.resize(uint::next_power_of_two(buckets));
  356         }
  357     }
  358 
  359     /// Modify and return the value corresponding to the key in the map, or
  360     /// insert and return a new value if it doesn't exist.
  361     pub fn mangle<'a,A>(&'a mut self, k: K, a: A, not_found: &fn(&K, A) -> V,
  362                         found: &fn(&K, &mut V, A)) -> &'a mut V {
  363         if self.size >= self.resize_at {
  364             // n.b.: We could also do this after searching, so
  365             // that we do not resize if this call to insert is
  366             // simply going to update a key in place.  My sense
  367             // though is that it's worse to have to search through
  368             // buckets to find the right spot twice than to just
  369             // resize in this corner case.
  370             self.expand();
  371         }
  372 
  373         let hash = k.hash_keyed(self.k0, self.k1) as uint;
  374         let idx = match self.bucket_for_key_with_hash(hash, &k) {
  375             TableFull => fail!("Internal logic error"),
  376             FoundEntry(idx) => { found(&k, self.mut_value_for_bucket(idx), a); idx }
  377             FoundHole(idx) => {
  378                 let v = not_found(&k, a);
  379                 self.buckets[idx] = Some(Bucket{hash: hash, key: k, value: v});
  380                 self.size += 1;
  381                 idx
  382             }
  383         };
  384 
  385         self.mut_value_for_bucket(idx)
  386     }
  387 
  388     /// Return the value corresponding to the key in the map, or insert
  389     /// and return the value if it doesn't exist.
  390     pub fn find_or_insert<'a>(&'a mut self, k: K, v: V) -> &'a mut V {
  391         self.mangle(k, v, |_k, a| a, |_k,_v,_a| ())
  392     }
  393 
  394     /// Return the value corresponding to the key in the map, or create,
  395     /// insert, and return a new value if it doesn't exist.
  396     pub fn find_or_insert_with<'a>(&'a mut self, k: K, f: &fn(&K) -> V)
  397                                -> &'a mut V {
  398         self.mangle(k, (), |k,_a| f(k), |_k,_v,_a| ())
  399     }
  400 
  401     /// Insert a key-value pair into the map if the key is not already present.
  402     /// Otherwise, modify the existing value for the key.
  403     /// Returns the new or modified value for the key.
  404     pub fn insert_or_update_with<'a>(&'a mut self, k: K, v: V,
  405                                      f: &fn(&K, &mut V)) -> &'a mut V {
  406         self.mangle(k, v, |_k,a| a, |k,v,_a| f(k,v))
  407     }
  408 
  409     /// Retrieves a value for the given key, failing if the key is not
  410     /// present.
  411     pub fn get<'a>(&'a self, k: &K) -> &'a V {
  412         match self.find(k) {
  413             Some(v) => v,
  414             None => fail!("No entry found for key: %?", k),
  415         }
  416     }
  417 
  418     /// Retrieves a (mutable) value for the given key, failing if the key
  419     /// is not present.

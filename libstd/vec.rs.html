   33  /// An owned, growable vector.
   34  ///
   35  /// # Examples
   36  ///
   37  /// ```rust
   38  /// # use std::vec::Vec;
   39  /// let mut vec = Vec::new();
   40  /// vec.push(1);
   41  /// vec.push(2);
   42  ///
   43  /// assert_eq!(vec.len(), 2);
   44  /// assert_eq!(vec.get(0), &1);
   45  ///
   46  /// assert_eq!(vec.pop(), Some(2));
   47  /// assert_eq!(vec.len(), 1);
   48  /// ```
   49  ///
   50  /// The `vec!` macro is provided to make initialization more convenient:
   51  ///
   52  /// ```rust
   53  /// let mut vec = vec!(1, 2, 3);
   54  /// vec.push(4);
   55  /// assert_eq!(vec, vec!(1, 2, 3, 4));
   56  /// ```
   57  #[unsafe_no_drop_flag]
   58  pub struct Vec<T> {
   59      len: uint,
   60      cap: uint,
   61      ptr: *mut T
   62  }
   63  
   64  impl<T> Vec<T> {
   65      /// Constructs a new, empty `Vec`.
   66      ///
   67      /// The vector will not allocate until elements are pushed onto it.
   68      ///
   69      /// # Example
   70      ///
   71      /// ```rust
   72      /// # use std::vec::Vec;
   73      /// let mut vec: Vec<int> = Vec::new();
   74      /// ```
   75      #[inline]
   76      pub fn new() -> Vec<T> {
   77          Vec { len: 0, cap: 0, ptr: 0 as *mut T }
   78      }
   79  
   80      /// Constructs a new, empty `Vec` with the specified capacity.
   81      ///
   82      /// The vector will be able to hold exactly `capacity` elements without
   83      /// reallocating. If `capacity` is 0, the vector will not allocate.
   84      ///
   85      /// # Example
   86      ///
   87      /// ```rust
   88      /// # use std::vec::Vec;
   89      /// let vec: Vec<int> = Vec::with_capacity(10);
   90      /// ```
   91      pub fn with_capacity(capacity: uint) -> Vec<T> {
   92          if capacity == 0 {
   93              Vec::new()
   94          } else {
   95              let size = capacity.checked_mul(&size_of::<T>()).expect("capacity overflow");
   96              let ptr = unsafe { malloc_raw(size) };
   97              Vec { len: 0, cap: capacity, ptr: ptr as *mut T }
   98          }
   99      }
  100  
  101      /// Creates and initializes a `Vec`.
  102      ///
  103      /// Creates a `Vec` of size `length` and initializes the elements to the
  104      /// value returned by the closure `op`.
  105      ///
  106      /// # Example
  107      ///
  108      /// ```rust
  109      /// # use std::vec::Vec;
  110      /// let vec = Vec::from_fn(3, |idx| idx * 2);
  111      /// assert_eq!(vec, vec!(0, 2, 4));
  112      /// ```
  113      pub fn from_fn(length: uint, op: |uint| -> T) -> Vec<T> {
  114          unsafe {
  115              let mut xs = Vec::with_capacity(length);
  116              while xs.len < length {
  117                  move_val_init(xs.as_mut_slice().unsafe_mut_ref(xs.len), op(xs.len));
  118                  xs.len += 1;
  119              }
  120              xs
  121          }
  122      }

  124      /// Create a `Vec<T>` directly from the raw constituents.
  125      ///
  126      /// This is highly unsafe:
  127      ///
  128      /// - if `ptr` is null, then `length` and `capacity` should be 0
  129      /// - `ptr` must point to an allocation of size `capacity`
  130      /// - there must be `length` valid instances of type `T` at the
  131      ///   beginning of that allocation
  132      /// - `ptr` must be allocated by the default `Vec` allocator
  133      pub unsafe fn from_raw_parts(length: uint, capacity: uint, ptr: *mut T) -> Vec<T> {
  134          Vec { len: length, cap: capacity, ptr: ptr }
  135      }
  136  
  137      /// Consumes the `Vec`, partitioning it based on a predcate.
  138      ///
  139      /// Partitions the `Vec` into two `Vec`s `(A,B)`, where all elements of `A`
  140      /// satisfy `f` and all elements of `B` do not. The order of elements is
  141      /// preserved.
  142      ///
  143      /// # Example
  144      ///
  145      /// ```rust
  146      /// let vec = vec!(1, 2, 3, 4);
  147      /// let (even, odd) = vec.partition(|&n| n % 2 == 0);
  148      /// assert_eq!(even, vec!(2, 4));
  149      /// assert_eq!(odd, vec!(1, 3));
  150      /// ```
  151      #[inline]
  152      pub fn partition(self, f: |&T| -> bool) -> (Vec<T>, Vec<T>) {
  153          let mut lefts  = Vec::new();
  154          let mut rights = Vec::new();
  155  
  156          for elt in self.move_iter() {
  157              if f(&elt) {
  158                  lefts.push(elt);
  159              } else {
  160                  rights.push(elt);
  161              }
  162          }
  163  
  164          (lefts, rights)
  165      }
  166  }

  168  impl<T: Clone> Vec<T> {
  169      /// Iterates over the `second` vector, copying each element and appending it to
  170      /// the `first`. Afterwards, the `first` is then returned for use again.
  171      ///
  172      /// # Example
  173      ///
  174      /// ```rust
  175      /// let vec = vec!(1, 2);
  176      /// let vec = vec.append([3, 4]);
  177      /// assert_eq!(vec, vec!(1, 2, 3, 4));
  178      /// ```
  179      #[inline]
  180      pub fn append(mut self, second: &[T]) -> Vec<T> {
  181          self.push_all(second);
  182          self
  183      }
  184  
  185      /// Constructs a `Vec` by cloning elements of a slice.
  186      ///
  187      /// # Example
  188      ///
  189      /// ```rust
  190      /// # use std::vec::Vec;
  191      /// let slice = [1, 2, 3];
  192      /// let vec = Vec::from_slice(slice);
  193      /// ```
  194      pub fn from_slice(values: &[T]) -> Vec<T> {
  195          values.iter().map(|x| x.clone()).collect()
  196      }
  197  
  198      /// Constructs a `Vec` with copies of a value.
  199      ///
  200      /// Creates a `Vec` with `length` copies of `value`.
  201      ///
  202      /// # Example
  203      /// ```rust
  204      /// # use std::vec::Vec;
  205      /// let vec = Vec::from_elem(3, "hi");
  206      /// println!("{}", vec); // prints [hi, hi, hi]
  207      /// ```
  208      pub fn from_elem(length: uint, value: T) -> Vec<T> {
  209          unsafe {
  210              let mut xs = Vec::with_capacity(length);
  211              while xs.len < length {
  212                  move_val_init(xs.as_mut_slice().unsafe_mut_ref(xs.len), value.clone());
  213                  xs.len += 1;
  214              }
  215              xs
  216          }
  217      }
  218  
  219      /// Appends all elements in a slice to the `Vec`.
  220      ///
  221      /// Iterates over the slice `other`, clones each element, and then appends
  222      /// it to this `Vec`. The `other` vector is traversed in-order.
  223      ///
  224      /// # Example
  225      ///
  226      /// ```rust
  227      /// let mut vec = vec!(1);
  228      /// vec.push_all([2, 3, 4]);
  229      /// assert_eq!(vec, vec!(1, 2, 3, 4));
  230      /// ```
  231      #[inline]
  232      pub fn push_all(&mut self, other: &[T]) {
  233          self.extend(other.iter().map(|e| e.clone()));
  234      }
  235  
  236      /// Grows the `Vec` in-place.
  237      ///
  238      /// Adds `n` copies of `value` to the `Vec`.
  239      ///
  240      /// # Example
  241      ///
  242      /// ```rust
  243      /// let mut vec = vec!("hello");
  244      /// vec.grow(2, & &"world");
  245      /// assert_eq!(vec, vec!("hello", "world", "world"));
  246      /// ```
  247      pub fn grow(&mut self, n: uint, value: &T) {
  248          let new_len = self.len() + n;
  249          self.reserve(new_len);
  250          let mut i: uint = 0u;
  251  
  252          while i < n {
  253              self.push((*value).clone());
  254              i += 1u;
  255          }
  256      }
  257  
  258      /// Sets the value of a vector element at a given index, growing the vector
  259      /// as needed.
  260      ///
  261      /// Sets the element at position `index` to `value`. If `index` is past the
  262      /// end of the vector, expands the vector by replicating `initval` to fill
  263      /// the intervening space.
  264      ///
  265      /// # Example
  266      ///
  267      /// ```rust
  268      /// let mut vec = vec!("a", "b", "c");
  269      /// vec.grow_set(1, & &"fill", "d");
  270      /// vec.grow_set(4, & &"fill", "e");
  271      /// assert_eq!(vec, vec!("a", "d", "c", "fill", "e"));
  272      /// ```
  273      pub fn grow_set(&mut self, index: uint, initval: &T, value: T) {
  274          let l = self.len();
  275          if index >= l {
  276              self.grow(index - l + 1u, initval);
  277          }
  278          *self.get_mut(index) = value;
  279      }

  281      /// Partitions a vector based on a predcate.
  282      ///
  283      /// Clones the elements of the vector, partitioning them into two `Vec`s
  284      /// `(A,B)`, where all elements of `A` satisfy `f` and all elements of `B`
  285      /// do not. The order of elements is preserved.
  286      ///
  287      /// # Example
  288      ///
  289      /// ```rust
  290      /// let vec = vec!(1, 2, 3, 4);
  291      /// let (even, odd) = vec.partitioned(|&n| n % 2 == 0);
  292      /// assert_eq!(even, vec!(2, 4));
  293      /// assert_eq!(odd, vec!(1, 3));
  294      /// ```
  295      pub fn partitioned(&self, f: |&T| -> bool) -> (Vec<T>, Vec<T>) {
  296          let mut lefts = Vec::new();
  297          let mut rights = Vec::new();
  298  
  299          for elt in self.iter() {
  300              if f(elt) {
  301                  lefts.push(elt.clone());
  302              } else {
  303                  rights.push(elt.clone());
  304              }
  305          }
  306  
  307          (lefts, rights)
  308      }
  309  }

  316      fn clone_from(&mut self, other: &Vec<T>) {
  317          // drop anything in self that will not be overwritten
  318          if self.len() > other.len() {
  319              self.truncate(other.len())
  320          }
  321  
  322          // reuse the contained values' allocations/resources.
  323          for (place, thing) in self.mut_iter().zip(other.iter()) {
  324              place.clone_from(thing)
  325          }
  326  
  327          // self.len <= other.len due to the truncate above, so the
  328          // slice here is always in-bounds.
  329          let len = self.len();
  330          self.extend(other.slice_from(len).iter().map(|x| x.clone()));
  331      }
  332  }

  385  impl<T> Vec<T> {
  386      /// Returns the number of elements the vector can hold without
  387      /// reallocating.
  388      ///
  389      /// # Example
  390      ///
  391      /// ```rust
  392      /// # use std::vec::Vec;
  393      /// let vec: Vec<int> = Vec::with_capacity(10);
  394      /// assert_eq!(vec.capacity(), 10);
  395      /// ```
  396      #[inline]
  397      pub fn capacity(&self) -> uint {
  398          self.cap
  399      }
  400  
  401       /// Reserves capacity for at least `n` additional elements in the given
  402       /// vector.
  403       ///
  404       /// # Failure
  405       ///
  406       /// Fails if the new capacity overflows `uint`.
  407       ///
  408       /// # Example
  409       ///
  410       /// ```rust
  411       /// # use std::vec::Vec;
  412       /// let mut vec: Vec<int> = vec!(1);
  413       /// vec.reserve_additional(10);
  414       /// assert!(vec.capacity() >= 11);
  415       /// ```
  416      pub fn reserve_additional(&mut self, extra: uint) {
  417          if self.cap - self.len < extra {
  418              match self.len.checked_add(&extra) {
  419                  None => fail!("Vec::reserve_additional: `uint` overflow"),
  420                  Some(new_cap) => self.reserve(new_cap)
  421              }
  422          }
  423      }
  424  
  425      /// Reserves capacity for at least `n` elements in the given vector.
  426      ///
  427      /// This function will over-allocate in order to amortize the allocation
  428      /// costs in scenarios where the caller may need to repeatedly reserve
  429      /// additional space.
  430      ///
  431      /// If the capacity for `self` is already equal to or greater than the
  432      /// requested capacity, then no action is taken.
  433      ///
  434      /// # Example
  435      ///
  436      /// ```rust
  437      /// let mut vec = vec!(1, 2, 3);
  438      /// vec.reserve(10);
  439      /// assert!(vec.capacity() >= 10);
  440      /// ```
  441      pub fn reserve(&mut self, capacity: uint) {
  442          if capacity >= self.len {
  443              self.reserve_exact(num::next_power_of_two(capacity))
  444          }
  445      }
  446  
  447      /// Reserves capacity for exactly `capacity` elements in the given vector.
  448      ///
  449      /// If the capacity for `self` is already equal to or greater than the
  450      /// requested capacity, then no action is taken.
  451      ///
  452      /// # Example
  453      ///
  454      /// ```rust
  455      /// # use std::vec::Vec;

  457      /// vec.reserve_exact(11);

  460      pub fn reserve_exact(&mut self, capacity: uint) {
  461          if capacity > self.cap {
  462              let size = capacity.checked_mul(&size_of::<T>()).expect("capacity overflow");
  463              self.cap = capacity;
  464              unsafe {
  465                  self.ptr = realloc_raw(self.ptr as *mut u8, size) as *mut T;
  466              }
  467          }
  468      }

  518      ///

  520      ///
  521      /// # Example

  876      /// elements after position i one position to the right.

  893          self.reserve(len + 1);

  896              // The spot to put the new value

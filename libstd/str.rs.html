  152      fn concat(&self) -> ~str {
  153          if self.is_empty() { return ~""; }
  154  
  155          let len = self.iter().map(|s| s.as_slice().len()).sum();
  156  
  157          let mut s = with_capacity(len);
  158  
  159          unsafe {
  160              do s.as_mut_buf |buf, _| {
  161                  let mut buf = buf;
  162                  for ss in self.iter() {
  163                      do ss.as_slice().as_imm_buf |ssbuf, sslen| {
  164                          ptr::copy_memory(buf, ssbuf, sslen);
  165                          buf = buf.offset(sslen as int);
  166                      }
  167                  }
  168              }
  169              raw::set_len(&mut s, len);
  170          }
  171          s
  172      }

  174      /// Concatenate a vector of strings, placing a given separator between each.
  175      fn connect(&self, sep: &str) -> ~str {
  176          if self.is_empty() { return ~""; }
  177  
  178          // concat is faster
  179          if sep.is_empty() { return self.concat(); }
  180  
  181          // this is wrong without the guarantee that `self` is non-empty
  182          let len = sep.len() * (self.len() - 1)
  183              + self.iter().map(|s| s.as_slice().len()).sum();
  184          let mut s = ~"";
  185          let mut first = true;
  186  
  187          s.reserve(len);
  188  
  189          unsafe {
  190              do s.as_mut_buf |buf, _| {
  191                  do sep.as_imm_buf |sepbuf, seplen| {
  192                      let mut buf = buf;
  193                      for ss in self.iter() {
  194                          do ss.as_slice().as_imm_buf |ssbuf, sslen| {
  195                              if first {
  196                                  first = false;
  197                              } else {
  198                                  ptr::copy_memory(buf, sepbuf, seplen);
  199                                  buf = buf.offset(seplen as int);
  200                              }
  201                              ptr::copy_memory(buf, ssbuf, sslen);
  202                              buf = buf.offset(sslen as int);
  203                          }
  204                      }
  205                  }
  206              }
  207              raw::set_len(&mut s, len);
  208          }
  209          s
  210      }
  211  }

  421              } else {

  426                  }

  546  /// Determines if a vector of bytes contains valid UTF-8
  547  pub fn is_utf8(v: &[u8]) -> bool {
  548      let mut i = 0u;
  549      let total = v.len();
  550      fn unsafe_get(xs: &[u8], i: uint) -> u8 {
  551          unsafe { *xs.unsafe_ref(i) }
  552      }
  553      while i < total {
  554          let v_i = unsafe_get(v, i);
  555          if v_i < 128u8 {
  556              i += 1u;
  557          } else {
  558              let w = utf8_char_width(v_i);
  559              if w == 0u { return false; }
  560  
  561              let nexti = i + w;
  562              if nexti > total { return false; }
  563  
  564              // 2-byte encoding is for codepoints  \u0080 to  \u07ff
  565              //        first  C2 80        last DF BF
  566              // 3-byte encoding is for codepoints  \u0800 to  \uffff
  567              //        first  E0 A0 80     last EF BF BF
  568              // 4-byte encoding is for codepoints \u10000 to \u10ffff
  569              //        first  F0 90 80 80  last F4 8F BF BF
  570              //
  571              // Use the UTF-8 syntax from the RFC
  572              //
  573              // https://tools.ietf.org/html/rfc3629
  574              // UTF8-1      = %x00-7F
  575              // UTF8-2      = %xC2-DF UTF8-tail
  576              // UTF8-3      = %xE0 %xA0-BF UTF8-tail / %xE1-EC 2( UTF8-tail ) /
  577              //               %xED %x80-9F UTF8-tail / %xEE-EF 2( UTF8-tail )
  578              // UTF8-4      = %xF0 %x90-BF 2( UTF8-tail ) / %xF1-F3 3( UTF8-tail ) /
  579              //               %xF4 %x80-8F 2( UTF8-tail )
  580              // UTF8-tail   = %x80-BF
  581              // --
  582              // This code allows surrogate pairs: \uD800 to \uDFFF -> ED A0 80 to ED BF BF
  583              match w {
  584                  2 => if unsafe_get(v, i + 1) & 192u8 != TAG_CONT_U8 {
  585                      return false
  586                  },
  587                  3 => match (v_i,
  588                              unsafe_get(v, i + 1),
  589                              unsafe_get(v, i + 2) & 192u8) {
  590                      (0xE0        , 0xA0 .. 0xBF, TAG_CONT_U8) => (),
  591                      (0xE1 .. 0xEF, 0x80 .. 0xBF, TAG_CONT_U8) => (),
  592                      _ => return false,
  593                  },
  594                  _ => match (v_i,
  595                              unsafe_get(v, i + 1),
  596                              unsafe_get(v, i + 2) & 192u8,
  597                              unsafe_get(v, i + 3) & 192u8) {
  598                      (0xF0        , 0x90 .. 0xBF, TAG_CONT_U8, TAG_CONT_U8) => (),
  599                      (0xF1 .. 0xF3, 0x80 .. 0xBF, TAG_CONT_U8, TAG_CONT_U8) => (),
  600                      (0xF4        , 0x80 .. 0x8F, TAG_CONT_U8, TAG_CONT_U8) => (),
  601                      _ => return false,
  602                  },
  603              }
  604  
  605              i = nexti;
  606          }
  607      }
  608      true
  609  }

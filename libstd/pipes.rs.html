  11 /*! Runtime support for message passing with protocol enforcement.
  12 
  13 
  14 Pipes consist of two endpoints. One endpoint can send messages and
  15 the other can receive messages. The set of legal messages and which
  16 directions they can flow at any given point are determined by a
  17 protocol. Below is an example protocol.
  18 
  19 ~~~ {.rust}
  20 proto! pingpong (
  21     ping: send {
  22         ping -> pong
  23     }
  24     pong: recv {
  25         pong -> ping
  26     }
  27 )
  28 ~~~
  29 
  30 The `proto!` syntax extension will convert this into a module called
  31 `pingpong`, which includes a set of types and functions that can be
  32 used to write programs that follow the pingpong protocol.
  33 
  34 */
  35 
  36 /* IMPLEMENTATION NOTES
  37 
  38 The initial design for this feature is available at:
  39 
  40 https://github.com/eholk/rust/wiki/Proposal-for-channel-contracts
  41 
  42 Much of the design in that document is still accurate. There are
  43 several components for the pipe implementation. First of all is the
  44 syntax extension. To see how that works, it is best see comments in
  45 libsyntax/ext/pipes.rs.
  46 
  47 This module includes two related pieces of the runtime
  48 implementation: support for unbounded and bounded
  49 protocols. The main difference between the two is the type of the
  50 buffer that is carried along in the endpoint data structures.
  51 
  52 
  53 The heart of the implementation is the packet type. It contains a
  54 header and a payload field. Much of the code in this module deals with
  55 the header field. This is where the synchronization information is
  56 stored. In the case of a bounded protocol, the header also includes a
  57 pointer to the buffer the packet is contained in.
  58 
  59 Packets represent a single message in a protocol. The payload field
  60 gets instatiated at the type of the message, which is usually an enum
  61 generated by the pipe compiler. Packets are conceptually single use,
  62 although in bounded protocols they are reused each time around the
  63 loop.
  64 
  65 
  66 Packets are usually handled through a send_packet_buffered or
  67 recv_packet_buffered object. Each packet is referenced by one
  68 send_packet and one recv_packet, and these wrappers enforce that only
  69 one end can send and only one end can receive. The structs also
  70 include a destructor that marks packets are terminated if the sender
  71 or receiver destroys the object before sending or receiving a value.
  72 
  73 The *_packet_buffered structs take two type parameters. The first is
  74 the message type for the current packet (or state). The second
  75 represents the type of the whole buffer. For bounded protocols, the
  76 protocol compiler generates a struct with a field for each protocol
  77 state. This generated struct is used as the buffer type parameter. For
  78 unbounded protocols, the buffer is simply one packet, so there is a
  79 shorthand struct called send_packet and recv_packet, where the buffer
  80 type is just `packet<T>`. Using the same underlying structure for both
  81 bounded and unbounded protocols allows for less code duplication.
  82 
  83 */

 152     // Returns the old state.
 153     pub unsafe fn mark_blocked(&mut self, this: *rust_task) -> State {
 154         rustrt::rust_task_ref(this);
 155         let old_task = swap_task(&mut self.blocked_task, this);

 160     pub unsafe fn unblock(&mut self) {
 161         let old_task = swap_task(&mut self.blocked_task, ptr::null());
 162         if !old_task.is_null() {
 163             rustrt::rust_task_deref(old_task)
 164         }
 165         match swap_state_acq(&mut self.state, Empty) {
 166           Empty | Blocked => (),
 167           Terminated => self.state = Terminated,
 168           Full => self.state = Full
 169         }
 170     }

 350 pub fn send<T,Tbuffer>(mut p: SendPacketBuffered<T,Tbuffer>,
 351                        payload: T)
 352                        -> bool {
 353     let header = p.header();
 354     let p_ = p.unwrap();
 355     let p = unsafe { &mut *p_ };
 356     assert_eq!(ptr::to_unsafe_ptr(&(p.header)), header);
 357     assert!(p.payload.is_none());
 358     p.payload = Some(payload);
 359     let old_state = swap_state_rel(&mut p.header.state, Full);
 360     match old_state {
 361         Empty => {
 362             // Yay, fastpath.
 363 
 364             // The receiver will eventually clean this up.
 365             //unsafe { forget(p); }
 366             return true;
 367         }
 368         Full => fail!("duplicate send"),
 369         Blocked => {
 370             debug!("waking up task for %?", p_);
 371             let old_task = swap_task(&mut p.header.blocked_task, ptr::null());
 372             if !old_task.is_null() {
 373                 unsafe {
 374                     rustrt::task_signal_event(
 375                         old_task,
 376                         ptr::to_unsafe_ptr(&(p.header)) as *libc::c_void);
 377                     rustrt::rust_task_deref(old_task);
 378                 }
 379             }
 380 
 381             // The receiver will eventually clean this up.
 382             //unsafe { forget(p); }
 383             return true;
 384         }
 385         Terminated => {
 386             // The receiver will never receive this. Rely on drop_glue
 387             // to clean everything up.
 388             return false;
 389         }
 390     }
 391 }

 393 /** Receives a message from a pipe.
 394 
 395 Fails if the sender closes the connection.
 396 
 397 */
 398 pub fn recv<T:Send,Tbuffer:Send>(
 399     p: RecvPacketBuffered<T, Tbuffer>) -> T {
 400     try_recv(p).expect("connection closed")
 401 }
 402 
 403 /** Attempts to receive a message from a pipe.
 404 
 405 Returns `None` if the sender has closed the connection without sending
 406 a message, or `Some(T)` if a message was received.
 407 
 408 */
 409 pub fn try_recv<T:Send,Tbuffer:Send>(mut p: RecvPacketBuffered<T, Tbuffer>)
 410                                        -> Option<T> {
 411     let p_ = p.unwrap();
 412     let p = unsafe { &mut *p_ };
 413 
 414     do (|| {
 415         try_recv_(p)
 416     }).finally {
 417         unsafe {
 418             if task::failing() {
 419                 p.header.state = Terminated;
 420                 let old_task = swap_task(&mut p.header.blocked_task, ptr::null());
 421                 if !old_task.is_null() {
 422                     rustrt::rust_task_deref(old_task);
 423                 }
 424             }
 425         }
 426     }
 427 }

 429 fn try_recv_<T:Send>(p: &mut Packet<T>) -> Option<T> {
 430     // optimistic path
 431     match p.header.state {
 432       Full => {
 433         let payload = p.payload.take();
 434         p.header.state = Empty;
 435         return Some(payload.unwrap())
 436       },
 437       Terminated => return None,
 438       _ => {}
 439     }
 440 
 441     // regular path
 442     let this = unsafe { rustrt::rust_get_task() };
 443     unsafe {
 444         rustrt::task_clear_event_reject(this);
 445         rustrt::rust_task_ref(this);
 446     };
 447     debug!("blocked = %x this = %x", p.header.blocked_task as uint,
 448            this as uint);
 449     let old_task = swap_task(&mut p.header.blocked_task, this);
 450     debug!("blocked = %x this = %x old_task = %x",
 451            p.header.blocked_task as uint,
 452            this as uint, old_task as uint);
 453     assert!(old_task.is_null());
 454     let mut first = true;
 455     let mut count = SPIN_COUNT;
 456     loop {
 457         unsafe {
 458             rustrt::task_clear_event_reject(this);
 459         }
 460 
 461         let old_state = swap_state_acq(&mut p.header.state,
 462                                        Blocked);
 463         match old_state {
 464           Empty => {
 465             debug!("no data available on %?, going to sleep.", p);
 466             if count == 0 {
 467                 wait_event(this);
 468             }
 469             else {
 470                 count -= 1;
 471                 // FIXME (#524): Putting the yield here destroys a lot
 472                 // of the benefit of spinning, since we still go into
 473                 // the scheduler at every iteration. However, without
 474                 // this everything spins too much because we end up
 475                 // sometimes blocking the thing we are waiting on.
 476                 task::yield();
 477             }
 478             debug!("woke up, p.state = %?", p.header.state);
 479           }
 480           Blocked => if first {
 481             fail!("blocking on already blocked packet")
 482           },
 483           Full => {
 484             let payload = p.payload.take();
 485             let old_task = swap_task(&mut p.header.blocked_task, ptr::null());
 486             if !old_task.is_null() {
 487                 unsafe {
 488                     rustrt::rust_task_deref(old_task);
 489                 }
 490             }
 491             p.header.state = Empty;
 492             return Some(payload.unwrap())
 493           }
 494           Terminated => {
 495             // This assert detects when we've accidentally unsafely
 496             // casted too big of a number to a state.
 497             assert_eq!(old_state, Terminated);
 498 
 499             let old_task = swap_task(&mut p.header.blocked_task, ptr::null());
 500             if !old_task.is_null() {
 501                 unsafe {
 502                     rustrt::rust_task_deref(old_task);
 503                 }
 504             }
 505             return None;
 506           }
 507         }
 508         first = false;
 509     }
 510 }

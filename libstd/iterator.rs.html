  391      /// Performs a fold operation over the entire iterator, returning the
  392      /// eventual state at the end of the iteration.
  393      ///
  394      /// # Example
  395      ///
  396      /// ~~~ {.rust}
  397      /// let a = [1, 2, 3, 4, 5];

  401      fn fold<B>(&mut self, init: B, f: &fn(B, A) -> B) -> B {
  402          let mut accum = init;
  403          loop {
  404              match self.next() {
  405                  Some(x) => { accum = f(accum, x); }
  406                  None    => { break; }
  407              }
  408          }
  409          accum

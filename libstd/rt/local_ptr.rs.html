 100      // Rationale for all of these functions being inline(never)
 101      //
 102      // The #[thread_local] annotation gets propagated all the way through to
 103      // LLVM, meaning the global is specially treated by LLVM to lower it to an
 104      // efficient sequence of instructions. This also involves dealing with fun
 105      // stuff in object files and whatnot. Regardless, it turns out this causes
 106      // trouble with green threads and lots of optimizations turned on. The
 107      // following case study was done on linux x86_64, but I would imagine that
 108      // other platforms are similar.
 109      //
 110      // On linux, the instruction sequence for loading the tls pointer global
 111      // looks like:
 112      //
 113      //      mov %fs:0x0, %rax
 114      //      mov -0x8(%rax), %rbx
 115      //
 116      // This code leads me to believe that (%fs:0x0) is a table, and then the
 117      // table contains the TLS values for the process. Hence, the slot at offset
 118      // -0x8 is the task TLS pointer. This leads us to the conclusion that this
 119      // table is the actual thread local part of each thread. The kernel sets up
 120      // the fs segment selector to point at the right region of memory for each
 121      // thread.
 122      //
 123      // Optimizations lead me to believe that this code is lowered to these
 124      // instructions in the LLVM codegen passes, because you'll see code like
 125      // this when everything is optimized:
 126      //
 127      //      mov %fs:0x0, %r14
 128      //      mov -0x8(%r14), %rbx
 129      //      // do something with %rbx, the rust Task pointer
 130      //
 131      //      ... // <- do more things
 132      //
 133      //      mov -0x8(%r14), %rbx
 134      //      // do something else with %rbx
 135      //
 136      // Note that the optimization done here is that the first load is not
 137      // duplicated during the lower instructions. This means that the %fs:0x0
 138      // memory location is only dereferenced once.
 139      //
 140      // Normally, this is actually a good thing! With green threads, however,
 141      // it's very possible for the code labeled "do more things" to context
 142      // switch to another thread. If this happens, then we *must* re-load %fs:0x0
 143      // because it's changed (we're on a different thread). If we don't re-load
 144      // the table location, then we'll be reading the original thread's TLS
 145      // values, not our thread's TLS values.
 146      //
 147      // Hence, we never inline these functions. By never inlining, we're
 148      // guaranteed that loading the table is a local decision which is forced to
 149      // *always* happen.

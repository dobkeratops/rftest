  96  /// Borrow the thread-local value from thread-local storage.
  97  /// While the value is borrowed it is not available in TLS.
  98  ///
  99  /// # Safety note
 100  ///
 101  /// Does not validate the pointer type.
 102  pub unsafe fn borrow<T>(f: &fn(&mut T)) {
 103      let mut value = take();
 104  
 105      // XXX: Need a different abstraction from 'finally' here to avoid unsafety
 106      let unsafe_ptr = cast::transmute_mut_region(&mut *value);
 107      let value_cell = Cell::new(value);
 108  
 109      do (|| {
 110          f(unsafe_ptr);
 111      }).finally {
 112          put(value_cell.take());
 113      }
 114  }

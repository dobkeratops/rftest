  96      // Rationale for all of these functions being inline(never)
  97      //
  98      // The #[thread_local] annotation gets propagated all the way through to
  99      // LLVM, meaning the global is specially treated by LLVM to lower it to an
 100      // efficient sequence of instructions. This also involves dealing with fun
 101      // stuff in object files and whatnot. Regardless, it turns out this causes
 102      // trouble with green threads and lots of optimizations turned on. The
 103      // following case study was done on linux x86_64, but I would imagine that
 104      // other platforms are similar.
 105      //
 106      // On linux, the instruction sequence for loading the tls pointer global
 107      // looks like:
 108      //
 109      //      mov %fs:0x0, %rax
 110      //      mov -0x8(%rax), %rbx
 111      //
 112      // This code leads me to believe that (%fs:0x0) is a table, and then the
 113      // table contains the TLS values for the process. Hence, the slot at offset
 114      // -0x8 is the task TLS pointer. This leads us to the conclusion that this
 115      // table is the actual thread local part of each thread. The kernel sets up
 116      // the fs segment selector to point at the right region of memory for each
 117      // thread.
 118      //
 119      // Optimizations lead me to believe that this code is lowered to these
 120      // instructions in the LLVM codegen passes, because you'll see code like
 121      // this when everything is optimized:
 122      //
 123      //      mov %fs:0x0, %r14
 124      //      mov -0x8(%r14), %rbx
 125      //      // do something with %rbx, the rust Task pointer
 126      //
 127      //      ... // <- do more things
 128      //
 129      //      mov -0x8(%r14), %rbx
 130      //      // do something else with %rbx
 131      //
 132      // Note that the optimization done here is that the first load is not
 133      // duplicated during the lower instructions. This means that the %fs:0x0
 134      // memory location is only dereferenced once.
 135      //
 136      // Normally, this is actually a good thing! With green threads, however,
 137      // it's very possible for the code labeled "do more things" to context
 138      // switch to another thread. If this happens, then we *must* re-load %fs:0x0
 139      // because it's changed (we're on a different thread). If we don't re-load
 140      // the table location, then we'll be reading the original thread's TLS
 141      // values, not our thread's TLS values.
 142      //
 143      // Hence, we never inline these functions. By never inlining, we're
 144      // guaranteed that loading the table is a local decision which is forced to
 145      // *always* happen.

  21  pub enum IpAddr {
  22      Ipv4Addr(u8, u8, u8, u8),
  23      Ipv6Addr(u16, u16, u16, u16, u16, u16, u16, u16)

  26  impl ToStr for IpAddr {
  27      fn to_str(&self) -> ~str {
  28          match *self {
  29              Ipv4Addr(a, b, c, d) =>
  30                  fmt!("%u.%u.%u.%u",
  31                      a as uint, b as uint, c as uint, d as uint),
  32  
  33              // Ipv4 Compatible address
  34              Ipv6Addr(0, 0, 0, 0, 0, 0, g, h) => {
  35                  let a = fmt!("%04x", g as uint);
  36                  let b = FromStrRadix::from_str_radix(a.slice(2, 4), 16).unwrap();
  37                  let a = FromStrRadix::from_str_radix(a.slice(0, 2), 16).unwrap();
  38                  let c = fmt!("%04x", h as uint);
  39                  let d = FromStrRadix::from_str_radix(c.slice(2, 4), 16).unwrap();
  40                  let c = FromStrRadix::from_str_radix(c.slice(0, 2), 16).unwrap();
  41  
  42                  fmt!("::%u.%u.%u.%u", a, b, c, d)
  43              }
  44  
  45              // Ipv4-Mapped address
  46              Ipv6Addr(0, 0, 0, 0, 0, 0xFFFF, g, h) => {
  47                  let a = fmt!("%04x", g as uint);
  48                  let b = FromStrRadix::from_str_radix(a.slice(2, 4), 16).unwrap();
  49                  let a = FromStrRadix::from_str_radix(a.slice(0, 2), 16).unwrap();
  50                  let c = fmt!("%04x", h as uint);
  51                  let d = FromStrRadix::from_str_radix(c.slice(2, 4), 16).unwrap();
  52                  let c = FromStrRadix::from_str_radix(c.slice(0, 2), 16).unwrap();
  53  
  54                  fmt!("::FFFF:%u.%u.%u.%u", a, b, c, d)
  55              }
  56  
  57              Ipv6Addr(a, b, c, d, e, f, g, h) =>
  58                  fmt!("%x:%x:%x:%x:%x:%x:%x:%x",
  59                      a as uint, b as uint, c as uint, d as uint,
  60                      e as uint, f as uint, g as uint, h as uint)
  61          }
  62      }
  63  }

  65  #[deriving(Eq, TotalEq, Clone)]
  66  pub struct SocketAddr {
  67      ip: IpAddr,
  68      port: Port,
  69  }

  72  impl ToStr for SocketAddr {
  73      fn to_str(&self) -> ~str {
  74          match self.ip {
  75              Ipv4Addr(*) => fmt!("%s:%u", self.ip.to_str(), self.port as uint),
  76              Ipv6Addr(*) => fmt!("[%s]:%u", self.ip.to_str(), self.port as uint),
  77          }
  78      }
  79  }
  80  
  81  struct Parser<'self> {
  82      // parsing as ASCII, so can use byte array
  83      s: &'self [u8],
  84      pos: uint,
  85  }
  86  
  87  impl<'self> Parser<'self> {
  88      fn new(s: &'self str) -> Parser<'self> {
  89          Parser {
  90              s: s.as_bytes(),
  91              pos: 0,
  92          }
  93      }
  94  
  95      fn is_eof(&self) -> bool {
  96          self.pos == self.s.len()
  97      }
  98  
  99      // Commit only if parser returns Some
 100      fn read_atomically<T>(&mut self, cb: &fn(&mut Parser) -> Option<T>) -> Option<T> {
 101          let pos = self.pos;
 102          let r = cb(self);
 103          if r.is_none() {
 104              self.pos = pos;
 105          }
 106          r
 107      }
 108  
 109      // Commit only if parser read till EOF
 110      fn read_till_eof<T>(&mut self, cb: &fn(&mut Parser) -> Option<T>) -> Option<T> {
 111          do self.read_atomically |p| {
 112              cb(p).filtered(|_| p.is_eof())
 113          }
 114      }
 115  
 116      // Return result of first successful parser
 117      fn read_or<T>(&mut self, parsers: &[&fn(&mut Parser) -> Option<T>]) -> Option<T> {
 118          for pf in parsers.iter() {
 119              match self.read_atomically(|p: &mut Parser| (*pf)(p)) {
 120                  Some(r) => return Some(r),
 121                  None => {}
 122              }
 123          }
 124          None
 125      }
 126  
 127      // Apply 3 parsers sequentially
 128      fn read_seq_3<A, B, C>(&mut self,
 129              pa: &fn(&mut Parser) -> Option<A>,
 130              pb: &fn(&mut Parser) -> Option<B>,
 131              pc: &fn(&mut Parser) -> Option<C>
 132          ) -> Option<(A, B, C)>
 133      {
 134          do self.read_atomically |p| {
 135              let a = pa(p);
 136              let b = if a.is_some() { pb(p) } else { None };
 137              let c = if b.is_some() { pc(p) } else { None };
 138              match (a, b, c) {
 139                  (Some(a), Some(b), Some(c)) => Some((a, b, c)),
 140                  _ => None
 141              }
 142          }
 143      }
 144  
 145      // Read next char
 146      fn read_char(&mut self) -> Option<char> {
 147          if self.is_eof() {
 148              None
 149          } else {
 150              let r = self.s[self.pos] as char;
 151              self.pos += 1;
 152              Some(r)
 153          }
 154      }
 155  
 156      // Return char and advance iff next char is equal to requested
 157      fn read_given_char(&mut self, c: char) -> Option<char> {
 158          do self.read_atomically |p| {
 159              p.read_char().filtered(|&next| next == c)
 160          }
 161      }
 162  
 163      // Read digit
 164      fn read_digit(&mut self, radix: u8) -> Option<u8> {
 165          fn parse_digit(c: char, radix: u8) -> Option<u8> {
 166              let c = c as u8;
 167              // assuming radix is either 10 or 16
 168              if c >= '0' as u8 && c <= '9' as u8 {
 169                  Some((c - '0' as u8) as u8)
 170              } else if radix > 10 && c >= 'a' as u8 && c < 'a' as u8 + (radix - 10) {
 171                  Some((c - 'a' as u8 + 10) as u8)
 172              } else if radix > 10 && c >= 'A' as u8 && c < 'A' as u8 + (radix - 10) {
 173                  Some((c - 'A' as u8 + 10) as u8)
 174              } else {
 175                  None
 176              }
 177          }
 178  
 179          do self.read_atomically |p| {
 180              p.read_char().and_then(|c| parse_digit(c, radix))
 181          }
 182      }
 183  
 184      fn read_number_impl(&mut self, radix: u8, max_digits: u32, upto: u32) -> Option<u32> {
 185          let mut r = 0u32;
 186          let mut digit_count = 0;
 187          loop {
 188              match self.read_digit(radix) {
 189                  Some(d) => {
 190                      r = r * (radix as u32) + (d as u32);
 191                      digit_count += 1;
 192                      if digit_count > max_digits || r >= upto {
 193                          return None
 194                      }
 195                  }
 196                  None => {
 197                      if digit_count == 0 {
 198                          return None
 199                      } else {
 200                          return Some(r)
 201                      }
 202                  }
 203              };
 204          }
 205      }
 206  
 207      // Read number, failing if max_digits of number value exceeded
 208      fn read_number(&mut self, radix: u8, max_digits: u32, upto: u32) -> Option<u32> {
 209          do self.read_atomically |p| {
 210              p.read_number_impl(radix, max_digits, upto)
 211          }
 212      }

 214      fn read_ipv4_addr_impl(&mut self) -> Option<IpAddr> {
 215          let mut bs = [0u8, ..4];
 216          let mut i = 0;
 217          while i < 4 {
 218              if i != 0 && self.read_given_char('.').is_none() {
 219                  return None;
 220              }
 221  
 222              let octet = self.read_number(10, 3, 0x100).map(|&n| n as u8);
 223              match octet {
 224                  Some(d) => bs[i] = d,
 225                  None => return None,
 226              };
 227              i += 1;
 228          }
 229          Some(Ipv4Addr(bs[0], bs[1], bs[2], bs[3]))
 230      }
 231  
 232      // Read IPv4 address
 233      fn read_ipv4_addr(&mut self) -> Option<IpAddr> {
 234          do self.read_atomically |p| {
 235              p.read_ipv4_addr_impl()
 236          }
 237      }

 239      fn read_ipv6_addr_impl(&mut self) -> Option<IpAddr> {
 240          fn ipv6_addr_from_head_tail(head: &[u16], tail: &[u16]) -> IpAddr {
 241              assert!(head.len() + tail.len() <= 8);
 242              let mut gs = [0u16, ..8];
 243              gs.copy_from(head);
 244              gs.mut_slice(8 - tail.len(), 8).copy_from(tail);
 245              Ipv6Addr(gs[0], gs[1], gs[2], gs[3], gs[4], gs[5], gs[6], gs[7])
 246          }
 247  
 248          fn read_groups(p: &mut Parser, groups: &mut [u16, ..8], limit: uint) -> (uint, bool) {
 249              let mut i = 0;
 250              while i < limit {
 251                  if i < limit - 1 {
 252                      let ipv4 = do p.read_atomically |p| {
 253                          if i == 0 || p.read_given_char(':').is_some() {
 254                              p.read_ipv4_addr()
 255                          } else {
 256                              None
 257                          }
 258                      };
 259                      match ipv4 {
 260                          Some(Ipv4Addr(a, b, c, d)) => {
 261                              groups[i + 0] = (a as u16 << 8) | (b as u16);
 262                              groups[i + 1] = (c as u16 << 8) | (d as u16);
 263                              return (i + 2, true);
 264                          }
 265                          _ => {}
 266                      }
 267                  }
 268  
 269                  let group = do p.read_atomically |p| {
 270                      if i == 0 || p.read_given_char(':').is_some() {
 271                          p.read_number(16, 4, 0x10000).map(|&n| n as u16)
 272                      } else {
 273                          None
 274                      }
 275                  };
 276                  match group {
 277                      Some(g) => groups[i] = g,
 278                      None => return (i, false)
 279                  }
 280                  i += 1;
 281              }
 282              (i, false)
 283          }
 284  
 285          let mut head = [0u16, ..8];
 286          let (head_size, head_ipv4) = read_groups(self, &mut head, 8);
 287  
 288          if head_size == 8 {
 289              return Some(Ipv6Addr(
 290                  head[0], head[1], head[2], head[3],
 291                  head[4], head[5], head[6], head[7]))
 292          }
 293  
 294          // IPv4 part is not allowed before `::`
 295          if head_ipv4 {
 296              return None
 297          }
 298  
 299          // read `::` if previous code parsed less than 8 groups
 300          if !self.read_given_char(':').is_some() || !self.read_given_char(':').is_some() {
 301              return None;
 302          }
 303  
 304          let mut tail = [0u16, ..8];
 305          let (tail_size, _) = read_groups(self, &mut tail, 8 - head_size);
 306          Some(ipv6_addr_from_head_tail(head.slice(0, head_size), tail.slice(0, tail_size)))
 307      }

 309      fn read_ipv6_addr(&mut self) -> Option<IpAddr> {
 310          do self.read_atomically |p| {
 311              p.read_ipv6_addr_impl()
 312          }
 313      }
 314  
 315      fn read_ip_addr(&mut self) -> Option<IpAddr> {
 316          let ipv4_addr = |p: &mut Parser| p.read_ipv4_addr();
 317          let ipv6_addr = |p: &mut Parser| p.read_ipv6_addr();
 318          self.read_or([ipv4_addr, ipv6_addr])
 319      }
 320  
 321      fn read_socket_addr(&mut self) -> Option<SocketAddr> {
 322          let ip_addr = |p: &mut Parser| {
 323              let ipv4_p = |p: &mut Parser| p.read_ip_addr();
 324              let ipv6_p = |p: &mut Parser| {
 325                  let open_br = |p: &mut Parser| p.read_given_char('[');
 326                  let ip_addr = |p: &mut Parser| p.read_ipv6_addr();
 327                  let clos_br = |p: &mut Parser| p.read_given_char(']');
 328                  p.read_seq_3::<char, IpAddr, char>(open_br, ip_addr, clos_br)
 329                          .map(|&t| match t { (_, ip, _) => ip })
 330              };
 331              p.read_or([ipv4_p, ipv6_p])
 332          };
 333          let colon = |p: &mut Parser| p.read_given_char(':');
 334          let port  = |p: &mut Parser| p.read_number(10, 5, 0x10000).map(|&n| n as u16);
 335  
 336          // host, colon, port
 337          self.read_seq_3::<IpAddr, char, u16>(ip_addr, colon, port)
 338                  .map(|&t| match t { (ip, _, port) => SocketAddr { ip: ip, port: port } })
 339      }
 340  }

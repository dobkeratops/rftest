 498                      let stream = Cell::new(stream);
 499                      // Start another task to handle the connection

 510              port.recv();
 511              connect(0, addr);
 512  
 513              fn connect(i: int, addr: SocketAddr) {
 514                  if i == MAX { return }
 515  
 516                  do spawntask {
 517                      rtdebug!("connecting");
 518                      let mut stream = TcpStream::connect(addr);
 519                      // Connect again before writing

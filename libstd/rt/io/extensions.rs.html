  27  pub trait ReaderUtil {
  28  
  29      /// Reads a single byte. Returns `None` on EOF.
  30      ///
  31      /// # Failure
  32      ///
  33      /// Raises the same conditions as the `read` method. Returns
  34      /// `None` if the condition is handled.
  35      fn read_byte(&mut self) -> Option<u8>;
  36  
  37      /// Reads `len` bytes and appends them to a vector.
  38      ///
  39      /// May push fewer than the requested number of bytes on error
  40      /// or EOF. Returns true on success, false on EOF or error.
  41      ///
  42      /// # Failure
  43      ///
  44      /// Raises the same conditions as `read`. Additionally raises `read_error`
  45      /// on EOF. If `read_error` is handled then `push_bytes` may push less
  46      /// than the requested number of bytes.
  47      fn push_bytes(&mut self, buf: &mut ~[u8], len: uint);
  48  
  49      /// Reads `len` bytes and gives you back a new vector of length `len`
  50      ///
  51      /// # Failure
  52      ///
  53      /// Raises the same conditions as `read`. Additionally raises `read_error`
  54      /// on EOF. If `read_error` is handled then the returned vector may
  55      /// contain less than the requested number of bytes.

  57

  60      /// # Failure

  62      /// Raises the same conditions as the `read` method.

 276  impl<T: Reader> ReaderUtil for T {
 277      fn read_byte(&mut self) -> Option<u8> {
 278          let mut buf = [0];
 279          match self.read(buf) {
 280              Some(0) => {
 281                  debug!("read 0 bytes. trying again");
 282                  self.read_byte()
 283              }
 284              Some(1) => Some(buf[0]),
 285              Some(_) => util::unreachable(),
 286              None => None
 287          }
 288      }
 289  
 290      fn push_bytes(&mut self, buf: &mut ~[u8], len: uint) {
 291          unsafe {
 292              let start_len = buf.len();
 293              let mut total_read = 0;
 294  
 295              buf.reserve_at_least(start_len + len);
 296              vec::raw::set_len(buf, start_len + len);
 297  
 298              do (|| {
 299                  while total_read < len {
 300                      let len = buf.len();
 301                      let slice = buf.mut_slice(start_len + total_read, len);
 302                      match self.read(slice) {
 303                          Some(nread) => {
 304                              total_read += nread;
 305                          }
 306                          None => {
 307                              read_error::cond.raise(standard_error(EndOfFile));
 308                              break;
 309                          }
 310                      }
 311                  }
 312              }).finally {
 313                  vec::raw::set_len(buf, start_len + total_read);
 314              }
 315          }
 316      }
 317  
 318      fn read_bytes(&mut self, len: uint) -> ~[u8] {
 319          let mut buf = vec::with_capacity(len);
 320          self.push_bytes(&mut buf, len);
 321          return buf;
 322      }
 323  
 324      fn read_to_end(&mut self) -> ~[u8] {
 325          let mut buf = vec::with_capacity(DEFAULT_BUF_SIZE);
 326          let mut keep_reading = true;
 327          do read_error::cond.trap(|e| {
 328              if e.kind == EndOfFile {
 329                  keep_reading = false;
 330              } else {
 331                  read_error::cond.raise(e)
 332              }
 333          }).inside {
 334              while keep_reading {
 335                  self.push_bytes(&mut buf, DEFAULT_BUF_SIZE)
 336              }
 337          }
 338          return buf;
 339      }
 340  }

 343      fn read_le_uint_n(&mut self, nbytes: uint) -> u64 {
 344          assert!(nbytes > 0 && nbytes <= 8);
 345  
 346          let mut val = 0u64;
 347          let mut pos = 0;
 348          let mut i = nbytes;
 349          while i > 0 {
 350              val += (self.read_u8() as u64) << pos;
 351              pos += 8;
 352              i -= 1;
 353          }
 354          val
 355      }

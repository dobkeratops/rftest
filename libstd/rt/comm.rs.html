   31  /// A combined refcount / BlockedTask-as-uint pointer.
   32  ///
   33  /// Can be equal to the following values:
   34  ///
   35  /// * 2 - both endpoints are alive
   36  /// * 1 - either the sender or the receiver is dead, determined by context
   37  /// * <ptr> - A pointer to a blocked Task (see BlockedTask::cast_{to,from}_uint)
   38  type State = uint;
   39  
   40  static STATE_BOTH: State = 2;
   41  static STATE_ONE: State = 1;
   42  
   43  /// The heap-allocated structure shared between two endpoints.
   44  struct Packet<T> {
   45      state: AtomicUint,
   46      payload: Option<T>,
   47  }

   61  pub fn oneshot<T: Send>() -> (PortOne<T>, ChanOne<T>) {
   62      let packet: ~Packet<T> = ~Packet {
   63          state: AtomicUint::new(STATE_BOTH),
   64          payload: None
   65      };
   66  
   67      unsafe {
   68          let packet: *mut Void = cast::transmute(packet);
   69          let port = PortOne {
   70              void_packet: packet,
   71              suppress_finalize: false
   72          };
   73          let chan = ChanOne {
   74              void_packet: packet,
   75              suppress_finalize: false
   76          };
   77          return (port, chan);
   78      }
   79  }

   91      /// Send a message on the one-shot channel. If a receiver task is blocked
   92      /// waiting for the message, will wake it up and reschedule to it.
   93      pub fn send(self, val: T) {
   94          self.try_send(val);
   95      }
   96  
   97      /// As `send`, but also returns whether or not the receiver endpoint is still open.
   98      pub fn try_send(self, val: T) -> bool {
   99          self.try_send_inner(val, true)
  100      }
  101  
  102      /// Send a message without immediately rescheduling to a blocked receiver.
  103      /// This can be useful in contexts where rescheduling is forbidden, or to
  104      /// optimize for when the sender expects to still have useful work to do.
  105      pub fn send_deferred(self, val: T) {
  106          self.try_send_deferred(val);
  107      }
  108  
  109      /// As `send_deferred` and `try_send` together.
  110      pub fn try_send_deferred(self, val: T) -> bool {
  111          self.try_send_inner(val, false)
  112      }
  113  
  114      // 'do_resched' configures whether the scheduler immediately switches to
  115      // the receiving task, or leaves the sending task still running.
  116      fn try_send_inner(self, val: T, do_resched: bool) -> bool {
  117          if do_resched {
  118              rtassert!(!rt::in_sched_context());
  119          }
  120  
  121          let mut this = self;
  122          let mut recvr_active = true;
  123          let packet = this.packet();
  124  
  125          unsafe {
  126  
  127              // Install the payload
  128              assert!((*packet).payload.is_none());
  129              (*packet).payload = Some(val);
  130  
  131              // Atomically swap out the old state to figure out what
  132              // the port's up to, issuing a release barrier to prevent
  133              // reordering of the payload write. This also issues an
  134              // acquire barrier that keeps the subsequent access of the
  135              // ~Task pointer from being reordered.
  136              let oldstate = (*packet).state.swap(STATE_ONE, SeqCst);
  137  
  138              // Suppress the synchronizing actions in the finalizer. We're
  139              // done with the packet. NB: In case of do_resched, this *must*
  140              // happen before waking up a blocked task (or be unkillable),
  141              // because we might get a kill signal during the reschedule.
  142              this.suppress_finalize = true;
  143  
  144              match oldstate {
  145                  STATE_BOTH => {
  146                      // Port is not waiting yet. Nothing to do
  147                      do Local::borrow::<Scheduler, ()> |sched| {
  148                          rtdebug!("non-rendezvous send");
  149                          sched.metrics.non_rendezvous_sends += 1;
  150                      }
  151                  }
  152                  STATE_ONE => {
  153                      do Local::borrow::<Scheduler, ()> |sched| {
  154                          rtdebug!("rendezvous send");
  155                          sched.metrics.rendezvous_sends += 1;
  156                      }
  157                      // Port has closed. Need to clean up.
  158                      let _packet: ~Packet<T> = cast::transmute(this.void_packet);
  159                      recvr_active = false;
  160                  }
  161                  task_as_state => {
  162                      // Port is blocked. Wake it up.
  163                      let recvr = BlockedTask::cast_from_uint(task_as_state);
  164                      if do_resched {
  165                          do recvr.wake().map_move |woken_task| {
  166                              Scheduler::run_task(woken_task);
  167                          };
  168                      } else {
  169                          let recvr = Cell::new(recvr);
  170                          do Local::borrow::<Scheduler, ()> |sched| {
  171                              sched.enqueue_blocked_task(recvr.take());
  172                          }
  173                      }
  174                  }
  175              }
  176          }
  177  
  178          return recvr_active;
  179      }
  180  }

  183      fn packet(&self) -> *mut Packet<T> {

  186              let p: *mut Packet<T> = &mut **p;
  187              return p;
  188          }
  189      }

  191      /// Wait for a message on the one-shot port. Fails if the send end is closed.
  192      pub fn recv(self) -> T {
  193          match self.try_recv() {
  194              Some(val) => val,
  195              None => {
  196                  fail!("receiving on closed channel");
  197              }
  198          }
  199      }
  200  
  201      /// As `recv`, but returns `None` if the send end is closed rather than failing.
  202      pub fn try_recv(self) -> Option<T> {
  203          let mut this = self;
  204  
  205          // Optimistic check. If data was sent already, we don't even need to block.
  206          // No release barrier needed here; we're not handing off our task pointer yet.
  207          if !this.optimistic_check() {
  208              // No data available yet.
  209              // Switch to the scheduler to put the ~Task into the Packet state.
  210              let sched = Local::take::<Scheduler>();
  211              do sched.deschedule_running_task_and_then |sched, task| {
  212                  this.block_on(sched, task);
  213              }
  214          }
  215  
  216          // Task resumes.
  217          this.recv_ready()
  218      }
  219  }

  223      fn optimistic_check(&mut self) -> bool {
  224          unsafe { (*self.packet()).state.load(Acquire) == STATE_ONE }
  225      }
  226  
  227      #[inline] #[cfg(test)]
  228      fn optimistic_check(&mut self) -> bool {
  229          // The optimistic check is never necessary for correctness. For testing
  230          // purposes, making it randomly return false simulates a racing sender.
  231          use rand::{Rand};
  232          let actually_check = do Local::borrow::<Scheduler, bool> |sched| {
  233              Rand::rand(&mut sched.rng)
  234          };
  235          if actually_check {
  236              unsafe { (*self.packet()).state.load(Acquire) == STATE_ONE }
  237          } else {
  238              false
  239          }
  240      }
  241  
  242      fn block_on(&mut self, sched: &mut Scheduler, task: BlockedTask) -> bool {
  243          unsafe {
  244              // Atomically swap the task pointer into the Packet state, issuing
  245              // an acquire barrier to prevent reordering of the subsequent read
  246              // of the payload. Also issues a release barrier to prevent
  247              // reordering of any previous writes to the task structure.
  248              let task_as_state = task.cast_to_uint();
  249              let oldstate = (*self.packet()).state.swap(task_as_state, SeqCst);
  250              match oldstate {
  251                  STATE_BOTH => {
  252                      // Data has not been sent. Now we're blocked.
  253                      rtdebug!("non-rendezvous recv");
  254                      sched.metrics.non_rendezvous_recvs += 1;
  255                      false
  256                  }
  257                  STATE_ONE => {
  258                      // Re-record that we are the only owner of the packet.
  259                      // No barrier needed, even if the task gets reawoken
  260                      // on a different core -- this is analogous to writing a
  261                      // payload; a barrier in enqueueing the task protects it.
  262                      // NB(#8132). This *must* occur before the enqueue below.
  263                      // FIXME(#6842, #8130) This is usually only needed for the
  264                      // assertion in recv_ready, except in the case of select().
  265                      // This won't actually ever have cacheline contention, but
  266                      // maybe should be optimized out with a cfg(test) anyway?
  267                      (*self.packet()).state.store(STATE_ONE, Relaxed);
  268  
  269                      rtdebug!("rendezvous recv");
  270                      sched.metrics.rendezvous_recvs += 1;
  271  
  272                      // Channel is closed. Switch back and check the data.
  273                      // NB: We have to drop back into the scheduler event loop here
  274                      // instead of switching immediately back or we could end up
  275                      // triggering infinite recursion on the scheduler's stack.
  276                      let recvr = BlockedTask::cast_from_uint(task_as_state);
  277                      sched.enqueue_blocked_task(recvr);
  278                      true
  279                  }
  280                  _ => rtabort!("can't block_on; a task is already blocked")
  281              }
  282          }
  283      }

  285      // This is the only select trait function that's not also used in recv.
  286      fn unblock_from(&mut self) -> bool {
  287          let packet = self.packet();
  288          unsafe {
  289              // In case the data is available, the acquire barrier here matches
  290              // the release barrier the sender used to release the payload.
  291              match (*packet).state.load(Acquire) {
  292                  // Impossible. We removed STATE_BOTH when blocking on it, and
  293                  // no self-respecting sender would put it back.
  294                  STATE_BOTH    => rtabort!("refcount already 2 in unblock_from"),
  295                  // Here, a sender already tried to wake us up. Perhaps they
  296                  // even succeeded! Data is available.
  297                  STATE_ONE     => true,
  298                  // Still registered as blocked. Need to "unblock" the pointer.
  299                  task_as_state => {
  300                      // In the window between the load and the CAS, a sender
  301                      // might take the pointer and set the refcount to ONE. If
  302                      // that happens, we shouldn't clobber that with BOTH!
  303                      // Acquire barrier again for the same reason as above.
  304                      match (*packet).state.compare_and_swap(task_as_state, STATE_BOTH,
  305                                                             Acquire) {
  306                          STATE_BOTH => rtabort!("refcount became 2 in unblock_from"),
  307                          STATE_ONE  => true, // Lost the race. Data available.
  308                          same_ptr   => {
  309                              // We successfully unblocked our task pointer.
  310                              assert!(task_as_state == same_ptr);
  311                              let handle = BlockedTask::cast_from_uint(task_as_state);
  312                              // Because we are already awake, the handle we
  313                              // gave to this port shall already be empty.
  314                              handle.assert_already_awake();
  315                              false
  316                          }
  317                      }
  318                  }
  319              }
  320          }
  321      }
  322  }

  326  impl<T> SelectPortInner<T> for PortOne<T> {
  327      fn recv_ready(self) -> Option<T> {
  328          let mut this = self;
  329          let packet = this.packet();
  330  
  331          // No further memory barrier is needed here to access the
  332          // payload. Some scenarios:
  333          //
  334          // 1) We encountered STATE_ONE above - the atomic_xchg was the acq barrier. We're fine.
  335          // 2) We encountered STATE_BOTH above and blocked. The sending task then ran us
  336          //    and ran on its thread. The sending task issued a read barrier when taking the
  337          //    pointer to the receiving task.
  338          // 3) We encountered STATE_BOTH above and blocked, but the receiving task (this task)
  339          //    is pinned to some other scheduler, so the sending task had to give us to
  340          //    a different scheduler for resuming. That send synchronized memory.
  341          unsafe {
  342              // See corresponding store() above in block_on for rationale.
  343              // FIXME(#8130) This can happen only in test builds.
  344              assert!((*packet).state.load(Relaxed) == STATE_ONE);
  345  
  346              let payload = (*packet).payload.take();
  347  
  348              // The sender has closed up shop. Drop the packet.
  349              let _packet: ~Packet<T> = cast::transmute(this.void_packet);
  350              // Suppress the synchronizing actions in the finalizer. We're done with the packet.
  351              this.suppress_finalize = true;
  352              return payload;
  353          }
  354      }
  355  }

  373  #[unsafe_destructor]
  374  impl<T> Drop for ChanOne<T> {
  375      fn drop(&self) {
  376          if self.suppress_finalize { return }
  377  
  378          unsafe {
  379              let this = cast::transmute_mut(self);
  380              let oldstate = (*this.packet()).state.swap(STATE_ONE, SeqCst);
  381              match oldstate {
  382                  STATE_BOTH => {
  383                      // Port still active. It will destroy the Packet.
  384                  },
  385                  STATE_ONE => {
  386                      let _packet: ~Packet<T> = cast::transmute(this.void_packet);
  387                  },
  388                  task_as_state => {
  389                      // The port is blocked waiting for a message we will never send. Wake it.
  390                      assert!((*this.packet()).payload.is_none());
  391                      let recvr = BlockedTask::cast_from_uint(task_as_state);
  392                      do recvr.wake().map_move |woken_task| {
  393                          Scheduler::run_task(woken_task);
  394                      };
  395                  }
  396              }
  397          }
  398      }
  399  }

  401  #[unsafe_destructor]
  402  impl<T> Drop for PortOne<T> {
  403      fn drop(&self) {
  404          if self.suppress_finalize { return }
  405  
  406          unsafe {
  407              let this = cast::transmute_mut(self);
  408              let oldstate = (*this.packet()).state.swap(STATE_ONE, SeqCst);
  409              match oldstate {
  410                  STATE_BOTH => {
  411                      // Chan still active. It will destroy the packet.
  412                  },
  413                  STATE_ONE => {
  414                      let _packet: ~Packet<T> = cast::transmute(this.void_packet);
  415                  }
  416                  task_as_state => {
  417                      // This case occurs during unwinding, when the blocked
  418                      // receiver was killed awake. The task can't still be
  419                      // blocked (we are it), but we need to free the handle.
  420                      let recvr = BlockedTask::cast_from_uint(task_as_state);
  421                      recvr.assert_already_awake();
  422                  }
  423              }
  424          }
  425      }
  426  }

  638          let (next_link_port, next_link_chan) = oneshot();
  639          next_link_chan.send(next_data_port);

  645  impl<T: Send> GenericPort<T> for SharedPort<T> {
  646      fn recv(&self) -> T {
  647          match self.try_recv() {
  648              Some(val) => val,
  649              None => {
  650                  fail!("receiving on a closed channel");
  651              }
  652          }
  653      }
  654  
  655      fn try_recv(&self) -> Option<T> {
  656          unsafe {
  657              let (next_link_port, next_link_chan) = oneshot();
  658              let link_port = (*self.next_link.get()).swap(~next_link_port, SeqCst);
  659              let link_port = link_port.unwrap();
  660              let data_port = link_port.recv();
  661              let (next_data_port, res) = match data_port.try_recv() {
  662                  Some(StreamPayload { val, next }) => {
  663                      (next, Some(val))
  664                  }
  665                  None => {
  666                      let (next_data_port, _) = oneshot();
  667                      (next_data_port, None)
  668                  }
  669              };
  670              next_link_chan.send(next_data_port);
  671              return res;
  672          }
  673      }
  674  }

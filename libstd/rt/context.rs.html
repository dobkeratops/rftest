  38     /// Create a new context that will resume execution by running ~fn()

  40         // XXX: Putting main into a ~ so it's a thin pointer and can
  41         // be passed to the spawn function.  Another unfortunate
  42         // allocation

  44 
  45         // The C-ABI function that is the task entry point
  46         extern fn task_start_wrapper(f: &~fn()) { (*f)() }
  47 
  48         let fp: *c_void = task_start_wrapper as *c_void;
  49         let argp: *c_void = unsafe { transmute::<&~fn(), *c_void>(&*start) };
  50         let sp: *uint = stack.end();
  51         let sp: *mut uint = unsafe { transmute_mut_unsafe(sp) };
  52         // Save and then immediately load the current context,
  53         // which we will then modify to call the given function when restored
  54         let mut regs = new_regs();
  55         unsafe {
  56             swap_registers(transmute_mut_region(&mut *regs), transmute_region(&*regs));
  57         };
  58 
  59         initialize_call_frame(&mut *regs, fp, argp, sp);
  60 
  61         return Context {
  62             start: Some(start),
  63             regs: regs
  64         }
  65     }

  86     // A helper to build a new task using the dynamically found
  87     // scheduler and task. Only works in GreenTask context.
  88     pub fn build_homed_child(f: ~fn(), home: SchedHome) -> ~Task {
  89         let f = Cell::new(f);
  90         let home = Cell::new(home);
  91         do Local::borrow::<Task, ~Task> |running_task| {
  92             let mut sched = running_task.sched.take_unwrap();
  93             let new_task = ~running_task.new_child_homed(&mut sched.stack_pool,
  94                                                          home.take(),
  95                                                          f.take());
  96             running_task.sched = Some(sched);
  97             new_task
  98         }
  99     }

 213     pub fn run(&mut self, f: &fn()) {
 214         rtdebug!("run called on task: %u", borrow::to_uint(self));
 215 
 216         // The only try/catch block in the world. Attempt to run the task's
 217         // client-specified code and catch any failures.
 218         do self.unwinder.try {
 219 
 220             // Run the task main function, then do some cleanup.
 221             do f.finally {
 222 
 223                 // Destroy task-local storage. This may run user dtors.
 224                 match self.storage {
 225                     LocalStorage(ptr, Some(ref dtor)) => {
 226                         (*dtor)(ptr)
 227                     }
 228                     _ => ()
 229                 }
 230 
 231                 // Destroy remaining boxes. Also may run user dtors.
 232                 unsafe { cleanup::annihilate(); }
 233             }
 234         }
 235 
 236         // FIXME(#7544): We pass the taskgroup into death so that it can be
 237         // dropped while the unkillable counter is set. This should not be
 238         // necessary except for an extraneous clone() in task/spawn.rs that
 239         // causes a killhandle to get dropped, which mustn't receive a kill
 240         // signal since we're outside of the unwinder's try() scope.
 241         // { let _ = self.taskgroup.take(); }
 242         self.death.collect_failure(!self.unwinder.unwinding, self.taskgroup.take());
 243         self.destroyed = true;
 244     }

 277     // Grab both the scheduler and the task from TLS and check if the
 278     // task is executing on an appropriate scheduler.
 279     pub fn on_appropriate_sched() -> bool {
 280         do Local::borrow::<Task,bool> |task| {
 281             let sched_id = task.sched.get_ref().sched_id();
 282             let sched_run_anything = task.sched.get_ref().run_anything;
 283             match task.task_type {
 284                 GreenTask(Some(~AnySched)) => {
 285                     rtdebug!("anysched task in sched check ****");
 286                     sched_run_anything
 287                 }
 288                 GreenTask(Some(~Sched(SchedHandle { sched_id: ref id, _ }))) => {
 289                     rtdebug!("homed task in sched check ****");
 290                     *id == sched_id
 291                 }
 292                 GreenTask(None) => {
 293                     rtabort!("task without home");
 294                 }
 295                 SchedTask => {
 296                     rtabort!("type error: expected: GreenTask, found: SchedTask");
 297                 }
 298             }
 299         }

 315     pub fn new(stack_pool: &mut StackPool, start: ~fn()) -> Coroutine {
 316         static MIN_STACK_SIZE: uint = 3000000; // XXX: Too much stack
 317 
 318         let start = Coroutine::build_start_wrapper(start);
 319         let mut stack = stack_pool.take_segment(MIN_STACK_SIZE);
 320         let initial_context = Context::new(start, &mut stack);
 321         Coroutine {
 322             current_stack_segment: stack,
 323             saved_context: initial_context
 324         }
 325     }
 326 
 327     pub fn empty() -> Coroutine {
 328         Coroutine {
 329             current_stack_segment: StackSegment::new(0),
 330             saved_context: Context::empty()
 331         }
 332     }
 333 
 334     fn build_start_wrapper(start: ~fn()) -> ~fn() {
 335         let start_cell = Cell::new(start);
 336         let wrapper: ~fn() = || {
 337             // First code after swap to this new context. Run our
 338             // cleanup job.
 339             unsafe {
 340 
 341                 // Again - might work while safe, or it might not.
 342                 do Local::borrow::<Scheduler,()> |sched| {
 343                     (sched).run_cleanup_job();
 344                 }
 345 
 346                 // To call the run method on a task we need a direct
 347                 // reference to it. The task is in TLS, so we can
 348                 // simply unsafe_borrow it to get this reference. We
 349                 // need to still have the task in TLS though, so we
 350                 // need to unsafe_borrow.
 351                 let task = Local::unsafe_borrow::<Task>();
 352 
 353                 do (*task).run {
 354                     // N.B. Removing `start` from the start wrapper
 355                     // closure by emptying a cell is critical for
 356                     // correctness. The ~Task pointer, and in turn the
 357                     // closure used to initialize the first call
 358                     // frame, is destroyed in the scheduler context,
 359                     // not task context. So any captured closures must
 360                     // not contain user-definable dtors that expect to
 361                     // be in task context. By moving `start` out of
 362                     // the closure, all the user code goes our of
 363                     // scope while the task is still running.
 364                     let start = start_cell.take();
 365                     start();
 366                 };
 367             }
 368 
 369             // We remove the sched from the Task in TLS right now.
 370             let sched = Local::take::<Scheduler>();
 371             // ... allowing us to give it away when performing a
 372             // scheduling operation.
 373             sched.terminate_current_task()
 374         };
 375         return wrapper;
 376     }

 390 // Just a sanity check to make sure we are catching a Rust-thrown exception
 391 static UNWIND_TOKEN: uintptr_t = 839147;
 392 
 393 impl Unwinder {
 394     pub fn try(&mut self, f: &fn()) {
 395         use unstable::raw::Closure;
 396 
 397         unsafe {
 398             let closure: Closure = transmute(f);
 399             let code = transmute(closure.code);
 400             let env = transmute(closure.env);
 401 
 402             let token = rust_try(try_fn, code, env);
 403             assert!(token == 0 || token == UNWIND_TOKEN);
 404         }
 405 
 406         extern fn try_fn(code: *c_void, env: *c_void) {
 407             unsafe {
 408                 let closure: Closure = Closure {
 409                     code: transmute(code),
 410                     env: transmute(env),
 411                 };
 412                 let closure: &fn() = transmute(closure);
 413                 closure();
 414             }
 415         }
 416 
 417         extern {
 418             #[rust_stack]
 419             fn rust_try(f: *u8, code: *c_void, data: *c_void) -> uintptr_t;
 420         }
 421     }
 422 
 423     pub fn begin_unwind(&mut self) -> ! {
 424         self.unwinding = true;
 425         unsafe {
 426             rust_begin_unwind(UNWIND_TOKEN);
 427             return transmute(());
 428         }
 429         extern {
 430             fn rust_begin_unwind(token: uintptr_t);
 431         }
 432     }
 433 }

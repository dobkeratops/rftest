  11  /*! The Rust Runtime, including the task scheduler and I/O
  12  
  13  The `rt` module provides the private runtime infrastructure necessary
  14  to support core language features like the exchange and local heap,
  15  the garbage collector, logging, local data and unwinding. It also
  16  implements the default task scheduler and task model. Initialization
  17  routines are provided for setting up runtime resources in common
  18  configurations, including that used by `rustc` when generating
  19  executables.
  20  
  21  It is intended that the features provided by `rt` can be factored in a
  22  way such that the core library can be built with different 'profiles'
  23  for different use cases, e.g. excluding the task scheduler. A number
  24  of runtime features though are critical to the functioning of the
  25  language and an implementation must be provided regardless of the
  26  execution environment.
  27  
  28  Of foremost importance is the global exchange heap, in the module
  29  `global_heap`. Very little practical Rust code can be written without
  30  access to the global heap. Unlike most of `rt` the global heap is
  31  truly a global resource and generally operates independently of the
  32  rest of the runtime.
  33  
  34  All other runtime features are task-local, including the local heap,
  35  the garbage collector, local storage, logging and the stack unwinder.
  36  
  37  The relationship between `rt` and the rest of the core library is
  38  not entirely clear yet and some modules will be moving into or
  39  out of `rt` as development proceeds.
  40  
  41  Several modules in `core` are clients of `rt`:
  42  
  43  * `std::task` - The user-facing interface to the Rust task model.
  44  * `std::task::local_data` - The interface to local data.
  45  * `std::gc` - The garbage collector.
  46  * `std::unstable::lang` - Miscellaneous lang items, some of which rely on `std::rt`.
  47  * `std::condition` - Uses local data.
  48  * `std::cleanup` - Local heap destruction.
  49  * `std::io` - In the future `std::io` will use an `rt` implementation.
  50  * `std::logging`
  51  * `std::pipes`
  52  * `std::comm`
  53  * `std::stackwalk`
  54  
  55  */

 234  fn run_(main: ~fn(), use_main_sched: bool) -> int {
 235      static DEFAULT_ERROR_CODE: int = 101;
 236  
 237      let nscheds = util::default_sched_threads();
 238  
 239      let main = Cell::new(main);
 240  
 241      // The shared list of sleeping schedulers.
 242      let sleepers = SleeperList::new();
 243  
 244      // Create a work queue for each scheduler, ntimes. Create an extra
 245      // for the main thread if that flag is set. We won't steal from it.
 246      let work_queues: ~[WorkQueue<~Task>] = vec::from_fn(nscheds, |_| WorkQueue::new());
 247  
 248      // The schedulers.
 249      let mut scheds = ~[];
 250      // Handles to the schedulers. When the main task ends these will be
 251      // sent the Shutdown message to terminate the schedulers.
 252      let mut handles = ~[];
 253  
 254      for work_queue in work_queues.iter() {
 255          rtdebug!("inserting a regular scheduler");
 256  
 257          // Every scheduler is driven by an I/O event loop.
 258          let loop_ = ~UvEventLoop::new();
 259          let mut sched = ~Scheduler::new(loop_,
 260                                          work_queue.clone(),
 261                                          work_queues.clone(),
 262                                          sleepers.clone());
 263          let handle = sched.make_handle();
 264  
 265          scheds.push(sched);
 266          handles.push(handle);
 267      }
 268  
 269      // If we need a main-thread task then create a main thread scheduler
 270      // that will reject any task that isn't pinned to it
 271      let main_sched = if use_main_sched {
 272  
 273          // Create a friend handle.
 274          let mut friend_sched = scheds.pop();
 275          let friend_handle = friend_sched.make_handle();
 276          scheds.push(friend_sched);
 277  
 278          // This scheduler needs a queue that isn't part of the stealee
 279          // set.
 280          let work_queue = WorkQueue::new();
 281  
 282          let main_loop = ~UvEventLoop::new();
 283          let mut main_sched = ~Scheduler::new_special(main_loop,
 284                                                       work_queue,
 285                                                       work_queues.clone(),
 286                                                       sleepers.clone(),
 287                                                       false,
 288                                                       Some(friend_handle));
 289          let main_handle = main_sched.make_handle();
 290          handles.push(main_handle);
 291          Some(main_sched)
 292      } else {
 293          None
 294      };
 295  
 296      // Create a shared cell for transmitting the process exit
 297      // code from the main task to this function.
 298      let exit_code = UnsafeArc::new(AtomicInt::new(0));
 299      let exit_code_clone = exit_code.clone();
 300  
 301      // When the main task exits, after all the tasks in the main
 302      // task tree, shut down the schedulers and set the exit code.
 303      let handles = Cell::new(handles);
 304      let on_exit: ~fn(bool) = |exit_success| {
 305          assert_once_ever!("last task exiting");
 306  
 307          let mut handles = handles.take();
 308          for handle in handles.mut_iter() {
 309              handle.send(Shutdown);
 310          }
 311  
 312          unsafe {
 313              let exit_code = if exit_success {
 314                  use rt::util;
 315  
 316                  // If we're exiting successfully, then return the global
 317                  // exit status, which can be set programmatically.
 318                  util::get_exit_status()
 319              } else {
 320                  DEFAULT_ERROR_CODE
 321              };
 322              (*exit_code_clone.get()).store(exit_code, SeqCst);
 323          }
 324      };
 325  
 326      let mut threads = ~[];
 327  
 328      let on_exit = Cell::new(on_exit);
 329  
 330      if !use_main_sched {
 331  
 332          // In the case where we do not use a main_thread scheduler we
 333          // run the main task in one of our threads.
 334  
 335          let mut main_task = ~Task::new_root(&mut scheds[0].stack_pool, None, main.take());
 336          main_task.death.on_exit = Some(on_exit.take());
 337          let main_task_cell = Cell::new(main_task);
 338  
 339          let sched = scheds.pop();
 340          let sched_cell = Cell::new(sched);
 341          let thread = do Thread::start {
 342              let sched = sched_cell.take();
 343              sched.bootstrap(main_task_cell.take());
 344          };
 345          threads.push(thread);
 346      }
 347  
 348      // Run each remaining scheduler in a thread.
 349      for sched in scheds.move_rev_iter() {
 350          rtdebug!("creating regular schedulers");
 351          let sched_cell = Cell::new(sched);
 352          let thread = do Thread::start {
 353              let mut sched = sched_cell.take();
 354              let bootstrap_task = ~do Task::new_root(&mut sched.stack_pool, None) || {
 355                  rtdebug!("boostraping a non-primary scheduler");
 356              };
 357              sched.bootstrap(bootstrap_task);
 358          };
 359          threads.push(thread);
 360      }
 361  
 362      // If we do have a main thread scheduler, run it now.
 363  
 364      if use_main_sched {
 365  
 366          rtdebug!("about to create the main scheduler task");
 367  
 368          let mut main_sched = main_sched.unwrap();
 369  
 370          let home = Sched(main_sched.make_handle());
 371          let mut main_task = ~Task::new_root_homed(&mut main_sched.stack_pool, None,
 372                                                    home, main.take());
 373          main_task.death.on_exit = Some(on_exit.take());
 374          rtdebug!("bootstrapping main_task");
 375  
 376          main_sched.bootstrap(main_task);
 377      }
 378  
 379      rtdebug!("waiting for threads");
 380  
 381      // Wait for schedulers
 382      for thread in threads.move_iter() {
 383          thread.join();
 384      }
 385  
 386      // Return the exit code
 387      unsafe {
 388          (*exit_code.get()).load(SeqCst)
 389      }
 390  }

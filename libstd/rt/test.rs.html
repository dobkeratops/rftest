  87      #[nolink]

  95      }
  96      static CTL_KERN: libc::c_int = 1;
  97      static KERN_MAXFILESPERPROC: libc::c_int = 29;
  98      static RLIMIT_NOFILE: libc::c_int = 8;

 100      pub unsafe fn raise_fd_limit() {
 101          // The strategy here is to fetch the current resource limits, read the kern.maxfilesperproc
 102          // sysctl value, and bump the soft resource limit for maxfiles up to the sysctl value.
 103          use ptr::{to_unsafe_ptr, to_mut_unsafe_ptr, mut_null};
 104          use sys::size_of_val;
 105          use os::last_os_error;
 106  
 107          // Fetch the kern.maxfilesperproc value
 108          let mut mib: [libc::c_int, ..2] = [CTL_KERN, KERN_MAXFILESPERPROC];
 109          let mut maxfiles: libc::c_int = 0;
 110          let mut size: libc::size_t = size_of_val(&maxfiles) as libc::size_t;
 111          if sysctl(to_mut_unsafe_ptr(&mut mib[0]), 2,
 112                    to_mut_unsafe_ptr(&mut maxfiles) as *mut libc::c_void,
 113                    to_mut_unsafe_ptr(&mut size),
 114                    mut_null(), 0) != 0 {
 115              let err = last_os_error();
 116              error!("raise_fd_limit: error calling sysctl: %s", err);
 117              return;
 118          }
 119  
 120          // Fetch the current resource limits
 121          let mut rlim = rlimit{rlim_cur: 0, rlim_max: 0};
 122          if getrlimit(RLIMIT_NOFILE, to_mut_unsafe_ptr(&mut rlim)) != 0 {
 123              let err = last_os_error();
 124              error!("raise_fd_limit: error calling getrlimit: %s", err);
 125              return;
 126          }
 127  
 128          // Bump the soft limit to the smaller of kern.maxfilesperproc and the hard limit
 129          rlim.rlim_cur = ::cmp::min(maxfiles as rlim_t, rlim.rlim_max);
 130  
 131          // Set our newly-increased resource limit
 132          if setrlimit(RLIMIT_NOFILE, to_unsafe_ptr(&rlim)) != 0 {
 133              let err = last_os_error();
 134              error!("raise_fd_limit: error calling setrlimit: %s", err);
 135              return;
 136          }
 137      }
 138  }

  72      /*!
  73       * darwin_fd_limit exists to work around an issue where launchctl on Mac OS X defaults the
  74       * rlimit maxfiles to 256/unlimited. The default soft limit of 256 ends up being far too low
  75       * for our multithreaded scheduler testing, depending on the number of cores available.
  76       *
  77       * This fixes issue #7772.
  78       */

  82      struct rlimit {
  83          rlim_cur: rlim_t,
  84          rlim_max: rlim_t

  93          fn setrlimit(resource: libc::c_int, rlp: *rlimit) -> libc::c_int;

  95      static CTL_KERN: libc::c_int = 1;
  96      static KERN_MAXFILESPERPROC: libc::c_int = 29;
  97      static RLIMIT_NOFILE: libc::c_int = 8;

  99      pub unsafe fn raise_fd_limit() {
 100          #[fixed_stack_segment]; #[inline(never)];
 101  
 102          // The strategy here is to fetch the current resource limits, read the kern.maxfilesperproc
 103          // sysctl value, and bump the soft resource limit for maxfiles up to the sysctl value.
 104          use ptr::{to_unsafe_ptr, to_mut_unsafe_ptr, mut_null};
 105          use sys::size_of_val;
 106          use os::last_os_error;
 107  
 108          // Fetch the kern.maxfilesperproc value
 109          let mut mib: [libc::c_int, ..2] = [CTL_KERN, KERN_MAXFILESPERPROC];
 110          let mut maxfiles: libc::c_int = 0;
 111          let mut size: libc::size_t = size_of_val(&maxfiles) as libc::size_t;
 112          if sysctl(to_mut_unsafe_ptr(&mut mib[0]), 2,
 113                    to_mut_unsafe_ptr(&mut maxfiles) as *mut libc::c_void,
 114                    to_mut_unsafe_ptr(&mut size),
 115                    mut_null(), 0) != 0 {
 116              let err = last_os_error();
 117              error!("raise_fd_limit: error calling sysctl: %s", err);
 118              return;
 119          }
 120  
 121          // Fetch the current resource limits
 122          let mut rlim = rlimit{rlim_cur: 0, rlim_max: 0};
 123          if getrlimit(RLIMIT_NOFILE, to_mut_unsafe_ptr(&mut rlim)) != 0 {
 124              let err = last_os_error();
 125              error!("raise_fd_limit: error calling getrlimit: %s", err);
 126              return;
 127          }
 128  
 129          // Bump the soft limit to the smaller of kern.maxfilesperproc and the hard limit
 130          rlim.rlim_cur = ::cmp::min(maxfiles as rlim_t, rlim.rlim_max);
 131  
 132          // Set our newly-increased resource limit
 133          if setrlimit(RLIMIT_NOFILE, to_unsafe_ptr(&rlim)) != 0 {
 134              let err = last_os_error();
 135              error!("raise_fd_limit: error calling setrlimit: %s", err);
 136              return;
 137          }
 138      }
 139  }

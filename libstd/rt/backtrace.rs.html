  49  // All rust symbols are in theory lists of "::"-separated identifiers. Some
  50  // assemblers, however, can't handle these characters in symbol names. To get
  51  // around this, we use C++-style mangling. The mangling method is:
  52  //
  53  // 1. Prefix the symbol with "_ZN"
  54  // 2. For each element of the path, emit the length plus the element
  55  // 3. End the path with "E"
  56  //
  57  // For example, "_ZN4testE" => "test" and "_ZN3foo3bar" => "foo::bar".
  58  //
  59  // We're the ones printing our backtraces, so we can't rely on anything else to
  60  // demangle our symbols. It's *much* nicer to look at demangled symbols, so
  61  // this function is implemented to give us nice pretty output.
  62  //
  63  // Note that this demangler isn't quite as fancy as it could be. We have lots
  64  // of other information in our symbols like hashes, version, type information,
  65  // etc. Additionally, this doesn't handle glue symbols at all.
  66  fn demangle(writer: &mut Writer, s: &str) -> IoResult<()> {
  67      // First validate the symbol. If it doesn't look like anything we're
  68      // expecting, we just print it literally. Note that we must handle non-rust
  69      // symbols because we could have any function in the backtrace.
  70      let mut valid = true;
  71      if s.len() > 4 && s.starts_with("_ZN") && s.ends_with("E") {
  72          let mut chars = s.slice(3, s.len() - 1).chars();
  73          while valid {
  74              let mut i = 0;
  75              for c in chars {
  76                  if c.is_digit() {
  77                      i = i * 10 + c as uint - '0' as uint;
  78                  } else {
  79                      break
  80                  }
  81              }
  82              if i == 0 {
  83                  valid = chars.next().is_none();
  84                  break
  85              } else if chars.by_ref().take(i - 1).len() != i - 1 {
  86                  valid = false;
  87              }
  88          }
  89      } else {
  90          valid = false;
  91      }
  92  
  93      // Alright, let's do this.
  94      if !valid {
  95          try!(writer.write_str(s));
  96      } else {
  97          let mut s = s.slice_from(3);
  98          let mut first = true;
  99          while s.len() > 1 {
 100              if !first {
 101                  try!(writer.write_str("::"));
 102              } else {
 103                  first = false;
 104              }
 105              let mut rest = s;
 106              while rest.char_at(0).is_digit() {
 107                  rest = rest.slice_from(1);
 108              }
 109              let i: uint = from_str(s.slice_to(s.len() - rest.len())).unwrap();
 110              s = rest.slice_from(i);
 111              rest = rest.slice_to(i);
 112              loop {
 113                  if rest.starts_with("$") {
 114                      macro_rules! demangle(
 115                          ($($pat:expr => $demangled:expr),*) => ({
 116                              $(if rest.starts_with($pat) {
 117                                  try!(writer.write_str($demangled));
 118                                  rest = rest.slice_from($pat.len());
 119                                } else)*
 120                              {
 121                                  try!(writer.write_str(rest));
 122                                  break;
 123                              }
 124  
 125                          })
 126                      )
 127                      // see src/librustc/back/link.rs for these mappings
 128                      demangle! (
 129                          "$SP$" => "@",
 130                          "$UP$" => "~",
 131                          "$RP$" => "*",
 132                          "$BP$" => "&",
 133                          "$LT$" => "<",
 134                          "$GT$" => ">",
 135                          "$LP$" => "(",
 136                          "$RP$" => ")",
 137                          "$C$"  => ",",
 138  
 139                          // in theory we can demangle any unicode code point, but
 140                          // for simplicity we just catch the common ones.
 141                          "$x20" => " ",
 142                          "$x27" => "'",
 143                          "$x5b" => "[",
 144                          "$x5d" => "]"
 145                      )
 146                  } else {
 147                      try!(writer.write_str(rest));
 148                      break;
 149                  }
 150              }
 151          }
 152      }
 153  
 154      Ok(())
 155  }

 157  /// Backtrace support built on libgcc with some extra OS-specific support
 158  ///
 159  /// Some methods of getting a backtrace:
 160  ///
 161  /// * The backtrace() functions on unix. It turns out this doesn't work very
 162  ///   well for green threads on OSX, and the address to symbol portion of it
 163  ///   suffers problems that are described below.
 164  ///
 165  /// * Using libunwind. This is more difficult than it sounds because libunwind
 166  ///   isn't installed everywhere by default. It's also a bit of a hefty library,
 167  ///   so possibly not the best option. When testing, libunwind was excellent at
 168  ///   getting both accurate backtraces and accurate symbols across platforms.
 169  ///   This route was not chosen in favor of the next option, however.
 170  ///
 171  /// * We're already using libgcc_s for exceptions in rust (triggering task
 172  ///   unwinding and running destructors on the stack), and it turns out that it
 173  ///   conveniently comes with a function that also gives us a backtrace. All of
 174  ///   these functions look like _Unwind_*, but it's not quite the full
 175  ///   repertoire of the libunwind API. Due to it already being in use, this was
 176  ///   the chosen route of getting a backtrace.
 177  ///
 178  /// After choosing libgcc_s for backtraces, the sad part is that it will only
 179  /// give us a stack trace of instruction pointers. Thankfully these instruction
 180  /// pointers are accurate (they work for green and native threads), but it's
 181  /// then up to us again to figure out how to translate these addresses to
 182  /// symbols. As with before, we have a few options. Before, that, a little bit
 183  /// of an interlude about symbols. This is my very limited knowledge about
 184  /// symbol tables, and this information is likely slightly wrong, but the
 185  /// general idea should be correct.
 186  ///
 187  /// When talking about symbols, it's helpful to know a few things about where
 188  /// symbols are located. Some symbols are located in the dynamic symbol table
 189  /// of the executable which in theory means that they're available for dynamic
 190  /// linking and lookup. Other symbols end up only in the local symbol table of
 191  /// the file. This loosely corresponds to pub and priv functions in Rust.
 192  ///
 193  /// Armed with this knowledge, we know that our solution for address to symbol
 194  /// translation will need to consult both the local and dynamic symbol tables.
 195  /// With that in mind, here's our options of translating an address to
 196  /// a symbol.
 197  ///
 198  /// * Use dladdr(). The original backtrace()-based idea actually uses dladdr()
 199  ///   behind the scenes to translate, and this is why backtrace() was not used.
 200  ///   Conveniently, this method works fantastically on OSX. It appears dladdr()
 201  ///   uses magic to consult the local symbol table, or we're putting everything
 202  ///   in the dynamic symbol table anyway. Regardless, for OSX, this is the
 203  ///   method used for translation. It's provided by the system and easy to do.o
 204  ///
 205  ///   Sadly, all other systems have a dladdr() implementation that does not
 206  ///   consult the local symbol table. This means that most functions are blank
 207  ///   because they don't have symbols. This means that we need another solution.
 208  ///
 209  /// * Use unw_get_proc_name(). This is part of the libunwind api (not the
 210  ///   libgcc_s version of the libunwind api), but involves taking a dependency
 211  ///   to libunwind. We may pursue this route in the future if we bundle
 212  ///   libunwind, but libunwind was unwieldy enough that it was not chosen at
 213  ///   this time to provide this functionality.
 214  ///
 215  /// * Shell out to a utility like `readelf`. Crazy though it may sound, it's a
 216  ///   semi-reasonable solution. The stdlib already knows how to spawn processes,
 217  ///   so in theory it could invoke readelf, parse the output, and consult the
 218  ///   local/dynamic symbol tables from there. This ended up not getting chosen
 219  ///   due to the craziness of the idea plus the advent of the next option.
 220  ///
 221  /// * Use `libbacktrace`. It turns out that this is a small library bundled in
 222  ///   the gcc repository which provides backtrace and symbol translation
 223  ///   functionality. All we really need from it is the backtrace functionality,
 224  ///   and we only really need this on everything that's not OSX, so this is the
 225  ///   chosen route for now.
 226  ///
 227  /// In summary, the current situation uses libgcc_s to get a trace of stack
 228  /// pointers, and we use dladdr() or libbacktrace to translate these addresses
 229  /// to symbols. This is a bit of a hokey implementation as-is, but it works for
 230  /// all unix platforms we support right now, so it at least gets the job done.

 248      #[inline(never)] // if we know this is a function call, we can skip it when
 249                       // tracing
 250      pub fn write(w: &mut Writer) -> IoResult<()> {
 251          // When using libbacktrace, we use some necessary global state, so we
 252          // need to prevent more than one thread from entering this block. This
 253          // is semi-reasonable in terms of printing anyway, and we know that all
 254          // I/O done here is blocking I/O, not green I/O, so we don't have to
 255          // worry about this being a native vs green mutex.
 256          static mut LOCK: StaticNativeMutex = NATIVE_MUTEX_INIT;
 257          let _g = unsafe { LOCK.lock() };
 258  
 259          try!(writeln!(w, "stack backtrace:"));
 260  
 261          let mut cx = Context { writer: w, last_error: None, idx: 0 };
 262          return match unsafe {
 263              uw::_Unwind_Backtrace(trace_fn,
 264                                    &mut cx as *mut Context as *libc::c_void)
 265          } {
 266              uw::_URC_NO_REASON => {
 267                  match cx.last_error {
 268                      Some(err) => Err(err),
 269                      None => Ok(())
 270                  }
 271              }
 272              _ => Ok(()),
 273          };

 275          extern fn trace_fn(ctx: *uw::_Unwind_Context,
 276                             arg: *libc::c_void) -> uw::_Unwind_Reason_Code {
 277              let cx: &mut Context = unsafe { cast::transmute(arg) };
 278              let ip = unsafe { uw::_Unwind_GetIP(ctx) as *libc::c_void };
 279              // dladdr() on osx gets whiny when we use FindEnclosingFunction, and
 280              // it appears to work fine without it, so we only use
 281              // FindEnclosingFunction on non-osx platforms. In doing so, we get a
 282              // slightly more accurate stack trace in the process.
 283              //
 284              // This is often because failure involves the last instruction of a
 285              // function being "call std::rt::begin_unwind", with no ret
 286              // instructions after it. This means that the return instruction
 287              // pointer points *outside* of the calling function, and by
 288              // unwinding it we go back to the original function.
 289              let ip = if cfg!(target_os = "macos") {
 290                  ip
 291              } else {
 292                  unsafe { uw::_Unwind_FindEnclosingFunction(ip) }
 293              };
 294  
 295              // Don't print out the first few frames (they're not user frames)
 296              cx.idx += 1;
 297              if cx.idx <= 0 { return uw::_URC_NO_REASON }
 298              // Don't print ginormous backtraces
 299              if cx.idx > 100 {
 300                  match write!(cx.writer, " ... <frames omitted>\n") {
 301                      Ok(()) => {}
 302                      Err(e) => { cx.last_error = Some(e); }
 303                  }
 304                  return uw::_URC_FAILURE
 305              }
 306  
 307              // Once we hit an error, stop trying to print more frames
 308              if cx.last_error.is_some() { return uw::_URC_FAILURE }
 309  
 310              match print(cx.writer, cx.idx, ip) {
 311                  Ok(()) => {}
 312                  Err(e) => { cx.last_error = Some(e); }
 313              }
 314  
 315              // keep going
 316              return uw::_URC_NO_REASON
 317          }
 318      }

 399          // The libbacktrace API supports creating a state, but it does not
 400          // support destroying a state. I personally take this to mean that a
 401          // state is meant to be created and then live forever.
 402          //
 403          // I would love to register an at_exit() handler which cleans up this
 404          // state, but libbacktrace provides no way to do so.
 405          //
 406          // With these constraints, this function has a statically cached state
 407          // that is calculated the first time this is requested. Remember that
 408          // backtracing all happens serially (one global lock).
 409          //
 410          // An additionally oddity in this function is that we initialize the
 411          // filename via self_exe_name() to pass to libbacktrace. It turns out
 412          // that on linux libbacktrace seamlessly gets the filename of the
 413          // current executable, but this fails on freebsd. by always providing
 414          // it, we make sure that libbacktrace never has a reason to not look up
 415          // the symbols. The libbacktrace API also states that the filename must
 416          // be in "permanent memory", so we copy it to a static and then use the
 417          // static as the pointer.
 418          unsafe fn init_state() -> *mut backtrace_state {
 419              static mut STATE: *mut backtrace_state = 0 as *mut backtrace_state;
 420              static mut LAST_FILENAME: [libc::c_char, ..256] = [0, ..256];
 421              if !STATE.is_null() { return STATE }
 422              let selfname = if cfg!(target_os = "freebsd") {
 423                  os::self_exe_name()
 424              } else {
 425                  None
 426              };

 428                  Some(path) => {
 429                      let bytes = path.as_vec();
 430                      if bytes.len() < LAST_FILENAME.len() {
 431                          let i = bytes.iter();
 432                          for (slot, val) in LAST_FILENAME.mut_iter().zip(i) {
 433                              *slot = *val as libc::c_char;
 434                          }
 435                          LAST_FILENAME.as_ptr()
 436                      } else {
 437                          ptr::null()
 438                      }
 439                  }
 440                  None => ptr::null(),
 441              };
 442              STATE = backtrace_create_state(filename, 0, error_cb,
 443                                             ptr::mut_null());

 445          }

 449          ////////////////////////////////////////////////////////////////////////

 451          // backtrace errors are currently swept under the rug, only I/O

 453          let state = unsafe { init_state() };

 455              return output(w, idx, addr, None)
 456          }

 463          };

 465              output(w, idx, addr, None)
 466          } else {
 467              output(w, idx, addr, Some(unsafe { CString::new(data, false) }))

 469      }

 473                s: Option<CString>) -> IoResult<()> {

  30 fn sockaddr_to_UvSocketAddr(addr: *uvll::sockaddr) -> UvSocketAddr {
  31     unsafe {
  32         assert!((is_ip4_addr(addr) || is_ip6_addr(addr)));
  33         assert!(!(is_ip4_addr(addr) && is_ip6_addr(addr)));
  34         match addr {
  35             _ if is_ip4_addr(addr) => UvIpv4SocketAddr(addr as *uvll::sockaddr_in),
  36             _ if is_ip6_addr(addr) => UvIpv6SocketAddr(addr as *uvll::sockaddr_in6),
  37             _ => fail!(),
  38         }
  39     }
  40 }

  42 fn socket_addr_as_uv_socket_addr<T>(addr: SocketAddr, f: &fn(UvSocketAddr) -> T) -> T {
  43     let malloc = match addr.ip {
  44         Ipv4Addr(*) => malloc_ip4_addr,
  45         Ipv6Addr(*) => malloc_ip6_addr,
  46     };
  47     let wrap = match addr.ip {
  48         Ipv4Addr(*) => UvIpv4SocketAddr,
  49         Ipv6Addr(*) => UvIpv6SocketAddr,
  50     };
  51     let free = match addr.ip {
  52         Ipv4Addr(*) => free_ip4_addr,
  53         Ipv6Addr(*) => free_ip6_addr,
  54     };
  55 
  56     let addr = unsafe { malloc(addr.ip.to_str(), addr.port as int) };
  57     do (|| {
  58         f(wrap(addr))
  59     }).finally {
  60         unsafe { free(addr) };
  61     }
  62 }

  64 fn uv_socket_addr_as_socket_addr<T>(addr: UvSocketAddr, f: &fn(SocketAddr) -> T) -> T {
  65     let ip_size = match addr {
  66         UvIpv4SocketAddr(*) => 4/*groups of*/ * 3/*digits separated by*/ + 3/*periods*/,
  67         UvIpv6SocketAddr(*) => 8/*groups of*/ * 4/*hex digits separated by*/ + 7 /*colons*/,
  68     };
  69     let ip_name = {
  70         let buf = vec::from_elem(ip_size + 1 /*null terminated*/, 0u8);
  71         unsafe {
  72             let buf_ptr = vec::raw::to_ptr(buf);
  73             match addr {
  74                 UvIpv4SocketAddr(addr) => uvll::ip4_name(addr, buf_ptr, ip_size as size_t),
  75                 UvIpv6SocketAddr(addr) => uvll::ip6_name(addr, buf_ptr, ip_size as size_t),
  76             }
  77         };
  78         buf
  79     };
  80     let ip_port = unsafe {
  81         let port = match addr {
  82             UvIpv4SocketAddr(addr) => uvll::ip4_port(addr),
  83             UvIpv6SocketAddr(addr) => uvll::ip6_port(addr),
  84         };
  85         port as u16
  86     };
  87     let ip_str = str::from_bytes_slice(ip_name).trim_right_chars(&'\x00');
  88     let ip = match addr {
  89         UvIpv4SocketAddr(*) => {
  90             let ip: ~[u8] =
  91                 ip_str.split_iter('.')
  92                       .transform(|s: &str| -> u8 { FromStr::from_str(s).unwrap() })
  93                       .collect();
  94             assert_eq!(ip.len(), 4);
  95             SocketAddr {
  96                 ip: Ipv4Addr(ip[0], ip[1], ip[2], ip[3]),
  97                 port: ip_port
  98             }
  99         },
 100         UvIpv6SocketAddr(*) => {
 101             let ip: ~[u16] = {
 102                 let expand_shorthand_and_convert = |s: &str| -> ~[~[u16]] {
 103                     let convert_each_segment = |s: &str| -> ~[u16] {
 104                         let read_hex_segment = |s: &str| -> u16 {
 105                             num::FromStrRadix::from_str_radix(s, 16u).unwrap()
 106                         };
 107                         match s {
 108                             "" => ~[],
 109                             // IPv4-Mapped/Compatible IPv6 Address?
 110                             s if s.find('.').is_some() => {
 111                                 let i = s.rfind(':').get_or_default(-1);
 112 
 113                                 let b = s.slice(i + 1, s.len()); // the ipv4 part
 114 
 115                                 let h = b.split_iter('.')
 116                                    .transform(|s: &str| -> u8 { FromStr::from_str(s).unwrap() })
 117                                    .transform(|s: u8| -> ~str { fmt!("%02x", s as uint) })
 118                                    .collect::<~[~str]>();
 119 
 120                                 if i == -1 {
 121                                     // Ipv4 Compatible Address (::x.x.x.x)
 122                                     // first 96 bits are zero leaving 32 bits
 123                                     // for the ipv4 part
 124                                     // (i.e ::127.0.0.1 == ::7F00:1)
 125                                     ~[num::FromStrRadix::from_str_radix(h[0] + h[1], 16).unwrap(),
 126                                       num::FromStrRadix::from_str_radix(h[2] + h[3], 16).unwrap()]
 127                                 } else {
 128                                     // Ipv4-Mapped Address (::FFFF:x.x.x.x)
 129                                     // first 80 bits are zero, followed by all ones
 130                                     // for the next 16 bits, leaving 32 bits for
 131                                     // the ipv4 part
 132                                     // (i.e ::FFFF:127.0.0.1 == ::FFFF:7F00:1)
 133                                     ~[1,
 134                                       num::FromStrRadix::from_str_radix(h[0] + h[1], 16).unwrap(),
 135                                       num::FromStrRadix::from_str_radix(h[2] + h[3], 16).unwrap()]
 136                                 }
 137                             },
 138                             s => s.split_iter(':').transform(read_hex_segment).collect()
 139                         }
 140                     };
 141                     s.split_str_iter("::").transform(convert_each_segment).collect()
 142                 };
 143                 match expand_shorthand_and_convert(ip_str) {
 144                     [x] => x, // no shorthand found
 145                     [l, r] => l + vec::from_elem(8 - l.len() - r.len(), 0u16) + r, // fill the gap
 146                     _ => fail!(), // impossible. only one shorthand allowed.
 147                 }
 148             };
 149             assert_eq!(ip.len(), 8);
 150             SocketAddr {
 151                 ip: Ipv6Addr(ip[0], ip[1], ip[2], ip[3], ip[4], ip[5], ip[6], ip[7]),
 152                 port: ip_port
 153             }
 154         },
 155     };
 156 
 157     // finally run the closure
 158     f(ip)
 159 }

 182 // uv_stream_t is the parent class of uv_tcp_t, uv_pipe_t, uv_tty_t

 184 pub struct StreamWatcher(*uvll::uv_stream_t);
 185 impl Watcher for StreamWatcher { }
 186 
 187 impl StreamWatcher {
 188     pub fn read_start(&mut self, alloc: AllocCallback, cb: ReadCallback) {
 189         {
 190             let data = self.get_watcher_data();
 191             data.alloc_cb = Some(alloc);
 192             data.read_cb = Some(cb);
 193         }
 194 
 195         unsafe { uvll::read_start(self.native_handle(), alloc_cb, read_cb); }
 196 
 197         extern fn alloc_cb(stream: *uvll::uv_stream_t, suggested_size: size_t) -> Buf {
 198             let mut stream_watcher: StreamWatcher = NativeHandle::from_native_handle(stream);
 199             let alloc_cb = stream_watcher.get_watcher_data().alloc_cb.get_ref();
 200             return (*alloc_cb)(suggested_size as uint);
 201         }
 202 
 203         extern fn read_cb(stream: *uvll::uv_stream_t, nread: ssize_t, buf: Buf) {
 204             rtdebug!("buf addr: %x", buf.base as uint);
 205             rtdebug!("buf len: %d", buf.len as int);
 206             let mut stream_watcher: StreamWatcher = NativeHandle::from_native_handle(stream);
 207             let cb = stream_watcher.get_watcher_data().read_cb.get_ref();
 208             let status = status_to_maybe_uv_error(stream_watcher, nread as c_int);
 209             (*cb)(stream_watcher, nread as int, buf, status);
 210         }
 211     }
 212 
 213     pub fn read_stop(&mut self) {
 214         // It would be nice to drop the alloc and read callbacks here,
 215         // but read_stop may be called from inside one of them and we
 216         // would end up freeing the in-use environment
 217         let handle = self.native_handle();
 218         unsafe { uvll::read_stop(handle); }
 219     }
 220 
 221     pub fn write(&mut self, buf: Buf, cb: ConnectionCallback) {
 222         {
 223             let data = self.get_watcher_data();
 224             assert!(data.write_cb.is_none());
 225             data.write_cb = Some(cb);
 226         }
 227 
 228         let req = WriteRequest::new();
 229         unsafe {
 230         assert_eq!(0, uvll::write(req.native_handle(), self.native_handle(), [buf], write_cb));
 231         }
 232 
 233         extern fn write_cb(req: *uvll::uv_write_t, status: c_int) {
 234             let write_request: WriteRequest = NativeHandle::from_native_handle(req);
 235             let mut stream_watcher = write_request.stream();
 236             write_request.delete();
 237             let cb = stream_watcher.get_watcher_data().write_cb.take_unwrap();
 238             let status = status_to_maybe_uv_error(stream_watcher, status);
 239             cb(stream_watcher, status);
 240         }
 241     }
 242 
 243     pub fn accept(&mut self, stream: StreamWatcher) {
 244         let self_handle = self.native_handle() as *c_void;
 245         let stream_handle = stream.native_handle() as *c_void;
 246         assert_eq!(0, unsafe { uvll::accept(self_handle, stream_handle) } );
 247     }

 249     pub fn close(self, cb: NullCallback) {
 250         {
 251             let mut this = self;
 252             let data = this.get_watcher_data();
 253             assert!(data.close_cb.is_none());
 254             data.close_cb = Some(cb);
 255         }
 256 
 257         unsafe { uvll::close(self.native_handle(), close_cb); }
 258 
 259         extern fn close_cb(handle: *uvll::uv_stream_t) {
 260             let mut stream_watcher: StreamWatcher = NativeHandle::from_native_handle(handle);
 261             stream_watcher.get_watcher_data().close_cb.take_unwrap()();
 262             stream_watcher.drop_watcher_data();
 263             unsafe { free_handle(handle as *c_void) }
 264         }
 265     }
 266 }

 292     pub fn bind(&mut self, address: SocketAddr) -> Result<(), UvError> {
 293         do socket_addr_as_uv_socket_addr(address) |addr| {
 294             let result = unsafe {

 307     pub fn connect(&mut self, address: SocketAddr, cb: ConnectionCallback) {
 308         unsafe {
 309             assert!(self.get_watcher_data().connect_cb.is_none());
 310             self.get_watcher_data().connect_cb = Some(cb);
 311 
 312             let connect_handle = ConnectRequest::new().native_handle();
 313             rtdebug!("connect_t: %x", connect_handle as uint);
 314             do socket_addr_as_uv_socket_addr(address) |addr| {
 315                 let result = match addr {
 316                     UvIpv4SocketAddr(addr) => uvll::tcp_connect(connect_handle,
 317                                                       self.native_handle(), addr, connect_cb),
 318                     UvIpv6SocketAddr(addr) => uvll::tcp_connect6(connect_handle,
 319                                                        self.native_handle(), addr, connect_cb),
 320                 };
 321                 assert_eq!(0, result);
 322             }

 417         extern fn recv_cb(handle: *uvll::uv_udp_t, nread: ssize_t, buf: Buf,
 418                           addr: *uvll::sockaddr, flags: c_uint) {
 419             // When there's no data to read the recv callback can be a no-op.
 420             // This can happen if read returns EAGAIN/EWOULDBLOCK. By ignoring
 421             // this we just drop back to kqueue and wait for the next callback.
 422             if nread == 0 {
 423                 return;
 424             }

 441         {

 445         }

 447         let req = UdpSendRequest::new();

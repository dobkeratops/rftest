 160      pub unsafe fn create(stack: uint, p: ~proc():Send) -> rust_thread {
 161          let arg: *mut libc::c_void = cast::transmute(p);
 162          // FIXME On UNIX, we guard against stack sizes that are too small but
 163          // that's because pthreads enforces that stacks are at least
 164          // PTHREAD_STACK_MIN bytes big.  Windows has no such lower limit, it's
 165          // just that below a certain threshold you can't do anything useful.
 166          // That threshold is application and architecture-specific, however.
 167          // For now, the only requirement is that it's big enough to hold the
 168          // red zone.  Round up to the next 64 kB because that's what the NT
 169          // kernel does, might as well make it explicit.  With the current
 170          // 20 kB red zone, that makes for a 64 kB minimum stack.
 171          let stack_size = (cmp::max(stack, RED_ZONE) + 0xfffe) & (-0xfffe - 1);
 172          let ret = CreateThread(ptr::mut_null(), stack_size as libc::size_t,
 173                                 super::thread_start, arg, 0, ptr::mut_null());
 174  
 175          if ret as uint == 0 {
 176              // be sure to not leak the closure
 177              let _p: ~proc():Send = cast::transmute(arg);
 178              fail!("failed to spawn native thread: {}", os::last_os_error());
 179          }
 180          return ret;
 181      }

 226      pub unsafe fn create(stack: uint, p: ~proc():Send) -> rust_thread {
 227          let mut native: libc::pthread_t = mem::uninit();
 228          let mut attr: libc::pthread_attr_t = mem::uninit();
 229          assert_eq!(pthread_attr_init(&mut attr), 0);
 230          assert_eq!(pthread_attr_setdetachstate(&mut attr,
 231                                                 PTHREAD_CREATE_JOINABLE), 0);
 232  
 233          // Reserve room for the red zone, the runtime's stack of last resort.
 234          let stack_size = cmp::max(stack, RED_ZONE + min_stack_size(&attr) as uint);
 235          match pthread_attr_setstacksize(&mut attr, stack_size as libc::size_t) {
 236              0 => {
 237              },
 238              libc::EINVAL => {
 239                  // EINVAL means |stack_size| is either too small or not a
 240                  // multiple of the system page size.  Because it's definitely
 241                  // >= PTHREAD_STACK_MIN, it must be an alignment issue.
 242                  // Round up to the neareast page and try again.
 243                  let page_size = os::page_size();
 244                  let stack_size = (stack_size + page_size - 1) & (-(page_size - 1) - 1);
 245                  assert_eq!(pthread_attr_setstacksize(&mut attr, stack_size as libc::size_t), 0);
 246              },
 247              errno => {
 248                  // This cannot really happen.
 249                  fail!("pthread_attr_setstacksize() error: {} ({})", os::last_os_error(), errno);
 250              },
 251          };
 252  
 253          let arg: *libc::c_void = cast::transmute(p);
 254          let ret = pthread_create(&mut native, &attr, super::thread_start, arg);
 255          assert_eq!(pthread_attr_destroy(&mut attr), 0);
 256  
 257          if ret != 0 {
 258              // be sure to not leak the closure
 259              let _p: ~proc():Send = cast::transmute(arg);
 260              fail!("failed to spawn native thread: {}", os::last_os_error());
 261          }
 262          native
 263      }

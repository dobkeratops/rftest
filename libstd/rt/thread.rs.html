  38  // This is the starting point of rust os threads. The first thing we do
  39  // is make sure that we don't trigger __morestack (also why this has a
  40  // no_split_stack annotation), and then we extract the main function
  41  // and invoke it.
  42  #[no_split_stack]
  43  extern fn thread_start(main: *libc::c_void) -> imp::rust_thread_return {
  44      use rt::stack;
  45      unsafe {
  46          stack::record_stack_bounds(0, uint::MAX);
  47          let f: ~proc() = cast::transmute(main);
  48          (*f)();
  49          cast::transmute(0 as imp::rust_thread_return)
  50      }
  51  }

  58      /// Starts execution of a new OS thread.
  59      ///
  60      /// This function will not wait for the thread to join, but a handle to the
  61      /// thread will be returned.
  62      ///
  63      /// Note that the handle returned is used to acquire the return value of the
  64      /// procedure `main`. The `join` function will wait for the thread to finish
  65      /// and return the value that `main` generated.
  66      ///
  67      /// Also note that the `Thread` returned will *always* wait for the thread
  68      /// to finish executing. This means that even if `join` is not explicitly
  69      /// called, when the `Thread` falls out of scope its destructor will block
  70      /// waiting for the OS thread.
  71      pub fn start<T: Send>(main: proc():Send -> T) -> Thread<T> {
  72          Thread::start_stack(DEFAULT_STACK_SIZE, main)
  73      }

  76      /// stack size for the new thread.
  77      pub fn start_stack<T: Send>(stack: uint, main: proc():Send -> T) -> Thread<T> {
  78  
  79          // We need the address of the packet to fill in to be stable so when
  80          // `main` fills it in it's still valid, so allocate an extra ~ box to do
  81          // so.
  82          let packet = ~None;

  90              native: native,

 159      pub unsafe fn create(stack: uint, p: ~proc():Send) -> rust_thread {
 160          let arg: *mut libc::c_void = cast::transmute(p);

 162          // that's because pthreads enforces that stacks are at least
 163          // PTHREAD_STACK_MIN bytes big.  Windows has no such lower limit, it's

 172                       super::thread_start, arg, 0, ptr::mut_null())

 218      pub unsafe fn create(stack: uint, p: ~proc():Send) -> rust_thread {
 219          let mut native: libc::pthread_t = mem::uninit();
 220          let mut attr: libc::pthread_attr_t = mem::uninit();
 221          assert_eq!(pthread_attr_init(&mut attr), 0);
 222          assert_eq!(pthread_attr_setdetachstate(&mut attr,
 223                                                 PTHREAD_CREATE_JOINABLE), 0);
 224  
 225          // Reserve room for the red zone, the runtime's stack of last resort.
 226          let stack_size = cmp::max(stack, RED_ZONE + min_stack_size(&attr) as uint);
 227          match pthread_attr_setstacksize(&mut attr, stack_size as libc::size_t) {
 228              0 => {
 229              },
 230              libc::EINVAL => {
 231                  // EINVAL means |stack_size| is either too small or not a
 232                  // multiple of the system page size.  Because it's definitely
 233                  // >= PTHREAD_STACK_MIN, it must be an alignment issue.
 234                  // Round up to the neareast page and try again.
 235                  let page_size = os::page_size();
 236                  let stack_size = (stack_size + page_size - 1) & (-(page_size - 1) - 1);
 237                  assert_eq!(pthread_attr_setstacksize(&mut attr, stack_size as libc::size_t), 0);
 238              },
 239              errno => {
 240                  // This cannot really happen.
 241                  fail!("pthread_attr_setstacksize() error: {} ({})", os::last_os_error(), errno);
 242              },
 243          };

 246          assert_eq!(pthread_create(&mut native, &attr,
 247                                    super::thread_start, arg), 0);
 248          native
 249      }

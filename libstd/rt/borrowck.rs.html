  73 unsafe fn fail_borrowed(box: *mut raw::Box<()>, file: *c_char, line: size_t) {

  81             }

  84             let mut msg = ~"borrowed";

  87                 if entry.box == box {

  90                     msg.push_str(filename);
  91                     msg.push_str(fmt!(":%u", entry.line as uint));
  92                     sep = " and at ";
  93                 }
  94             }
  95             do msg.as_c_str |msg_p| {
  96                 sys::begin_unwind_(msg_p as *c_char, file, line)
  97             }
  98         }
  99     }

 146 impl DebugPrints for io::fd_t {
 147     fn write_hex(&self, mut i: uint) {
 148         let letters = ['0', '1', '2', '3', '4', '5', '6', '7', '8',
 149                        '9', 'a', 'b', 'c', 'd', 'e', 'f'];
 150         static UINT_NIBBLES: uint = ::uint::bytes << 1;
 151         let mut buffer = [0_u8, ..UINT_NIBBLES+1];
 152         let mut c = UINT_NIBBLES;
 153         while c > 0 {
 154             c -= 1;
 155             buffer[c] = letters[i & 0xF] as u8;
 156             i >>= 4;
 157         }
 158         self.write(buffer.slice(0, UINT_NIBBLES));
 159     }

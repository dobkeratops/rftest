 137      fn write_hex(&self, mut i: uint) {
 138          let letters = ['0', '1', '2', '3', '4', '5', '6', '7', '8',
 139                         '9', 'a', 'b', 'c', 'd', 'e', 'f'];
 140          static UINT_NIBBLES: uint = ::uint::bytes << 1;
 141          let mut buffer = [0_u8, ..UINT_NIBBLES+1];
 142          let mut c = UINT_NIBBLES;
 143          while c > 0 {
 144              c -= 1;
 145              buffer[c] = letters[i & 0xF] as u8;
 146              i >>= 4;
 147          }
 148          self.write(buffer.slice(0, UINT_NIBBLES));
 149      }

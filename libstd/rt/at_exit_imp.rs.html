 25  // You'll note that these variables are *not* atomic, and this is done on
 26  // purpose. This module is designed to have init() called *once* in a
 27  // single-task context, and then run() is called only once in another
 28  // single-task context. As a result of this, only the `push` function is
 29  // thread-safe, and it assumes that the `init` function has run previously.
 30  static mut QUEUE: *mut Queue = 0 as *mut Queue;

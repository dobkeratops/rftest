 24  type Queue = Exclusive<~[proc():Send]>;

 26  // You'll note that these variables are *not* atomic, and this is done on
 27  // purpose. This module is designed to have init() called *once* in a
 28  // single-task context, and then run() is called only once in another
 29  // single-task context. As a result of this, only the `push` function is
 30  // thread-safe, and it assumes that the `init` function has run previously.
 31  static mut QUEUE: *mut Queue = 0 as *mut Queue;
 32  static mut RUNNING: bool = false;

 40      }
 41  }

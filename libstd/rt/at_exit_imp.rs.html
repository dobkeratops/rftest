 12  //!

 25  type Queue = Exclusive<Vec<proc():Send>>;
 26  
 27  // You'll note that these variables are *not* atomic, and this is done on
 28  // purpose. This module is designed to have init() called *once* in a
 29  // single-task context, and then run() is called only once in another
 30  // single-task context. As a result of this, only the `push` function is
 31  // thread-safe, and it assumes that the `init` function has run previously.
 32  static mut QUEUE: *mut Queue = 0 as *mut Queue;
 33  static mut RUNNING: bool = false;
 34  
 35  pub fn init() {
 36      unsafe {
 37          rtassert!(!RUNNING);
 38          rtassert!(QUEUE.is_null());
 39          let state: ~Queue = ~Exclusive::new(vec!());
 40          QUEUE = cast::transmute(state);
 41      }
 42  }
 43  
 44  pub fn push(f: proc():Send) {
 45      unsafe {
 46          rtassert!(!RUNNING);
 47          rtassert!(!QUEUE.is_null());
 48          let state: &mut Queue = cast::transmute(QUEUE);
 49          let mut f = Some(f);
 50          state.with(|arr|  {
 51              arr.push(f.take_unwrap());
 52          });
 53      }
 54  }
 55  
 56  pub fn run() {
 57      let vec = unsafe {
 58          rtassert!(!RUNNING);
 59          rtassert!(!QUEUE.is_null());
 60          RUNNING = true;
 61          let state: ~Queue = cast::transmute(QUEUE);
 62          QUEUE = 0 as *mut Queue;
 63          let mut vec = None;
 64          state.with(|arr| {
 65              vec = Some(mem::replace(arr, vec!()));
 66          });
 67          vec.take_unwrap()
 68      };
 69  
 70  
 71      for f in vec.move_iter() {
 72          f();
 73      }
 74  }

   34  /// The Scheduler is responsible for coordinating execution of Coroutines
   35  /// on a single thread. When the scheduler is running it is owned by
   36  /// thread local storage and the running task is owned by the
   37  /// scheduler.
   38  ///
   39  /// XXX: This creates too many callbacks to run_sched_once, resulting
   40  /// in too much allocation and too many events.
   41  pub struct Scheduler {
   42      /// There are N work queues, one per scheduler.
   43      priv work_queue: WorkQueue<~Task>,
   44      /// Work queues for the other schedulers. These are created by
   45      /// cloning the core work queues.
   46      work_queues: ~[WorkQueue<~Task>],
   47      /// The queue of incoming messages from other schedulers.
   48      /// These are enqueued by SchedHandles after which a remote callback
   49      /// is triggered to handle the message.
   50      priv message_queue: MessageQueue<SchedMessage>,
   51      /// A shared list of sleeping schedulers. We'll use this to wake
   52      /// up schedulers when pushing work onto the work queue.
   53      sleeper_list: SleeperList,
   54      /// Indicates that we have previously pushed a handle onto the
   55      /// SleeperList but have not yet received the Wake message.
   56      /// Being `true` does not necessarily mean that the scheduler is
   57      /// not active since there are multiple event sources that may
   58      /// wake the scheduler. It just prevents the scheduler from pushing
   59      /// multiple handles onto the sleeper list.
   60      priv sleepy: bool,
   61      /// A flag to indicate we've received the shutdown message and should
   62      /// no longer try to go to sleep, but exit instead.
   63      no_sleep: bool,
   64      stack_pool: StackPool,
   65      /// The event loop used to drive the scheduler and perform I/O
   66      event_loop: ~EventLoopObject,
   67      /// The scheduler runs on a special task.
   68      sched_task: Option<~Task>,
   69      /// An action performed after a context switch on behalf of the
   70      /// code running before the context switch
   71      priv cleanup_job: Option<CleanupJob>,
   72      metrics: SchedMetrics,
   73      /// Should this scheduler run any task, or only pinned tasks?
   74      run_anything: bool,
   75      /// If the scheduler shouldn't run some tasks, a friend to send
   76      /// them to.
   77      friend_handle: Option<SchedHandle>,
   78      /// A fast XorShift rng for scheduler use
   79      rng: XorShiftRng
   80  
   81  }

  145      // XXX: This may eventually need to be refactored so that
  146      // the scheduler itself doesn't have to call event_loop.run.

  150      // Take a main task to run, and a scheduler to run it in. Create a
  151      // scheduler task and bootstrap into it.
  152      pub fn bootstrap(~self, task: ~Task) {
  153  
  154          // Initialize the TLS key.
  155          local_ptr::init_tls_key();
  156  
  157          // Create a task for the scheduler with an empty context.
  158          let sched_task = ~Task::new_sched_task();
  159  
  160          // Now that we have an empty task struct for the scheduler
  161          // task, put it in TLS.
  162          Local::put::(sched_task);
  163  
  164          // Now, as far as all the scheduler state is concerned, we are
  165          // inside the "scheduler" context. So we can act like the
  166          // scheduler and resume the provided task.
  167          self.resume_task_immediately(task);
  168  
  169          // Now we are back in the scheduler context, having
  170          // successfully run the input task. Start by running the
  171          // scheduler. Grab it out of TLS - performing the scheduler
  172          // action will have given it away.
  173          let sched = Local::take::<Scheduler>();
  174  
  175          rtdebug!("starting scheduler %u", sched.sched_id());
  176  
  177          sched.run();
  178  
  179          // Now that we are done with the scheduler, clean up the
  180          // scheduler task. Do so by removing it from TLS and manually
  181          // cleaning up the memory it uses. As we didn't actually call
  182          // task.run() on the scheduler task we never get through all
  183          // the cleanup code it runs.
  184          let mut stask = Local::take::<Task>();
  185  
  186          rtdebug!("stopping scheduler %u", stask.sched.get_ref().sched_id());
  187  
  188          // Should not have any messages
  189          let message = stask.sched.get_mut_ref().message_queue.pop();
  190          assert!(message.is_none());
  191  
  192          stask.destroyed = true;
  193      }

  195      // This does not return a scheduler, as the scheduler is placed
  196      // inside the task.
  197      pub fn run(~self) {
  198  
  199          let mut self_sched = self;
  200  
  201          // Always run through the scheduler loop at least once so that
  202          // we enter the sleep state and can then be woken up by other
  203          // schedulers.
  204          self_sched.event_loop.callback(Scheduler::run_sched_once);
  205  
  206          // This is unsafe because we need to place the scheduler, with
  207          // the event_loop inside, inside our task. But we still need a
  208          // mutable reference to the event_loop to give it the "run"
  209          // command.
  210          unsafe {
  211              let event_loop: *mut ~EventLoopObject = &mut self_sched.event_loop;
  212  
  213              // Our scheduler must be in the task before the event loop
  214              // is started.
  215              let self_sched = Cell::new(self_sched);
  216              do Local::borrow::<Task,()> |stask| {
  217                  stask.sched = Some(self_sched.take());
  218              };
  219  
  220              (*event_loop).run();
  221          }
  222      }
  223  
  224      // One iteration of the scheduler loop, always run at least once.
  225  
  226      // The model for this function is that you continue through it
  227      // until you either use the scheduler while performing a schedule
  228      // action, in which case you give it away and do not return, or
  229      // you reach the end and sleep. In the case that a scheduler
  230      // action is performed the loop is evented such that this function
  231      // is called again.
  232      fn run_sched_once() {
  233  
  234          // When we reach the scheduler context via the event loop we
  235          // already have a scheduler stored in our local task, so we
  236          // start off by taking it. This is the only path through the
  237          // scheduler where we get the scheduler this way.
  238          let sched = Local::take::<Scheduler>();
  239  
  240          // Our first task is to read mail to see if we have important
  241          // messages.
  242  
  243          // 1) A wake message is easy, mutate sched struct and return
  244          //    it.
  245          // 2) A shutdown is also easy, shutdown.
  246          // 3) A pinned task - we resume immediately and do not return
  247          //    here.
  248          // 4) A message from another scheduler with a non-homed task
  249          //    to run here.
  250  
  251          let result = sched.interpret_message_queue();
  252          let sched = match result {
  253              Some(sched) => {
  254                  // We did not resume a task, so we returned.
  255                  sched
  256              }
  257              None => {
  258                  return;
  259              }
  260          };
  261  
  262          // Second activity is to try resuming a task from the queue.
  263  
  264          let result = sched.do_work();
  265          let mut sched = match result {
  266              Some(sched) => {
  267                  // Failed to dequeue a task, so we return.
  268                  sched
  269              }
  270              None => {
  271                  return;
  272              }
  273          };
  274  
  275          // If we got here then there was no work to do.
  276          // Generate a SchedHandle and push it to the sleeper list so
  277          // somebody can wake us up later.
  278          sched.metrics.wasted_turns += 1;
  279          if !sched.sleepy && !sched.no_sleep {
  280              rtdebug!("scheduler has no work to do, going to sleep");
  281              sched.metrics.sleepy_times += 1;
  282              sched.sleepy = true;
  283              let handle = sched.make_handle();
  284              sched.sleeper_list.push(handle);
  285          } else {
  286              rtdebug!("not sleeping, already doing so or no_sleep set");
  287          }
  288  
  289          // Finished a cycle without using the Scheduler. Place it back
  290          // in TLS.
  291          Local::put(sched);
  292      }

  294      pub fn make_handle(&mut self) -> SchedHandle {
  295          let remote = self.event_loop.remote_callback(Scheduler::run_sched_once);
  296  
  297          return SchedHandle {
  298              remote: remote,
  299              queue: self.message_queue.clone(),
  300              sched_id: self.sched_id()
  301          };
  302      }

  304      /// Schedule a task to be executed later.
  305      ///
  306      /// Pushes the task onto the work stealing queue and tells the
  307      /// event loop to run it later. Always use this instead of pushing
  308      /// to the work queue directly.
  309      pub fn enqueue_task(&mut self, task: ~Task) {
  310  
  311          let this = self;
  312  
  313          // We push the task onto our local queue clone.
  314          this.work_queue.push(task);
  315          this.event_loop.callback(Scheduler::run_sched_once);
  316  
  317          // We've made work available. Notify a
  318          // sleeping scheduler.
  319  
  320          // XXX: perf. Check for a sleeper without
  321          // synchronizing memory.  It's not critical
  322          // that we always find it.
  323  
  324          // XXX: perf. If there's a sleeper then we
  325          // might as well just send it the task
  326          // directly instead of pushing it to the
  327          // queue. That is essentially the intent here
  328          // and it is less work.
  329          match this.sleeper_list.pop() {
  330              Some(handle) => {
  331                  let mut handle = handle;
  332                  handle.send(Wake)
  333              }
  334              None => { (/* pass */) }
  335          };
  336      }
  337  
  338      /// As enqueue_task, but with the possibility for the blocked task to
  339      /// already have been killed.
  340      pub fn enqueue_blocked_task(&mut self, blocked_task: BlockedTask) {
  341          do blocked_task.wake().map_move |task| {
  342              self.enqueue_task(task);
  343          };
  344      }

  347  
  348      // This function returns None if the scheduler is "used", or it
  349      // returns the still-available scheduler.
  350      fn interpret_message_queue(~self) -> Option<~Scheduler> {
  351  
  352          let mut this = self;
  353          match this.message_queue.pop() {
  354              Some(PinnedTask(task)) => {
  355                  this.event_loop.callback(Scheduler::run_sched_once);
  356                  let mut task = task;
  357                  task.give_home(Sched(this.make_handle()));
  358                  this.resume_task_immediately(task);
  359                  return None;
  360              }
  361              Some(TaskFromFriend(task)) => {
  362                  this.event_loop.callback(Scheduler::run_sched_once);
  363                  rtdebug!("got a task from a friend. lovely!");
  364                  return this.sched_schedule_task(task);
  365              }
  366              Some(Wake) => {
  367                  this.event_loop.callback(Scheduler::run_sched_once);
  368                  this.sleepy = false;
  369                  return Some(this);
  370              }
  371              Some(Shutdown) => {
  372                  this.event_loop.callback(Scheduler::run_sched_once);
  373                  if this.sleepy {
  374                      // There may be an outstanding handle on the
  375                      // sleeper list.  Pop them all to make sure that's
  376                      // not the case.
  377                      loop {
  378                          match this.sleeper_list.pop() {
  379                              Some(handle) => {
  380                                  let mut handle = handle;
  381                                  handle.send(Wake);
  382                              }
  383                              None => break
  384                          }
  385                      }
  386                  }
  387                  // No more sleeping. After there are no outstanding
  388                  // event loop references we will shut down.
  389                  this.no_sleep = true;
  390                  this.sleepy = false;
  391                  // YYY: Does a shutdown count as a "use" of the
  392                  // scheduler? This seems to work - so I'm leaving it
  393                  // this way despite not having a solid rational for
  394                  // why I should return the scheduler here.
  395                  return Some(this);
  396              }
  397              None => {
  398                  return Some(this);
  399              }
  400          }
  401      }

  403      /// Given an input Coroutine sends it back to its home scheduler.
  404      fn send_task_home(task: ~Task) {
  405          let mut task = task;
  406          let mut home = task.take_unwrap_home();
  407          match home {
  408              Sched(ref mut home_handle) => {
  409                  home_handle.send(PinnedTask(task));
  410              }
  411              AnySched => {
  412                  rtabort!("error: cannot send anysched task home");
  413              }
  414          }
  415      }

  417      /// Take a non-homed task we aren't allowed to run here and send
  418      /// it to the designated friend scheduler to execute.
  419      fn send_to_friend(&mut self, task: ~Task) {
  420          rtdebug!("sending a task to friend");
  421          match self.friend_handle {
  422              Some(ref mut handle) => {
  423                  handle.send(TaskFromFriend(task));
  424              }
  425              None => {
  426                  rtabort!("tried to send task to a friend but scheduler has no friends");
  427              }
  428          }
  429      }
  430  
  431      // Workstealing: In this iteration of the runtime each scheduler
  432      // thread has a distinct work queue. When no work is available
  433      // locally, make a few attempts to steal work from the queues of
  434      // other scheduler threads. If a few steals fail we end up in the
  435      // old "no work" path which is fine.
  436  
  437      // First step in the process is to find a task. This function does
  438      // that by first checking the local queue, and if there is no work
  439      // there, trying to steal from the remote work queues.
  440      fn find_work(&mut self) -> Option<~Task> {
  441          rtdebug!("scheduler looking for work");
  442          match self.work_queue.pop() {
  443              Some(task) => {
  444                  rtdebug!("found a task locally");
  445                  return Some(task)
  446              }
  447              None => {
  448                  // Our naive stealing, try kinda hard.
  449                  rtdebug!("scheduler trying to steal");
  450                  let _len = self.work_queues.len();
  451                  return self.try_steals(2);
  452              }
  453          }
  454      }
  455  
  456      // With no backoff try stealing n times from the queues the
  457      // scheduler knows about. This naive implementation can steal from
  458      // our own queue or from other special schedulers.
  459      fn try_steals(&mut self, n: uint) -> Option<~Task> {
  460          for _ in range(0, n) {
  461              let index = self.rng.gen_uint_range(0, self.work_queues.len());
  462              let work_queues = &mut self.work_queues;
  463              match work_queues[index].steal() {
  464                  Some(task) => {
  465                      rtdebug!("found task by stealing"); return Some(task)
  466                  }
  467                  None => ()
  468              }
  469          };
  470          rtdebug!("giving up on stealing");
  471          return None;
  472      }
  473  
  474      // Given a task, execute it correctly.
  475      fn process_task(~self, task: ~Task) -> Option<~Scheduler> {
  476          let mut this = self;
  477          let mut task = task;
  478  
  479          rtdebug!("processing a task");
  480  
  481          let home = task.take_unwrap_home();
  482          match home {
  483              Sched(home_handle) => {
  484                  if home_handle.sched_id != this.sched_id() {
  485                      rtdebug!("sending task home");
  486                      task.give_home(Sched(home_handle));
  487                      Scheduler::send_task_home(task);
  488                      return Some(this);
  489                  } else {
  490                      rtdebug!("running task here");
  491                      task.give_home(Sched(home_handle));
  492                      this.resume_task_immediately(task);
  493                      return None;
  494                  }
  495              }
  496              AnySched if this.run_anything => {
  497                  rtdebug!("running anysched task here");
  498                  task.give_home(AnySched);
  499                  this.resume_task_immediately(task);
  500                  return None;
  501              }
  502              AnySched => {
  503                  rtdebug!("sending task to friend");
  504                  task.give_home(AnySched);
  505                  this.send_to_friend(task);
  506                  return Some(this);
  507              }
  508          }
  509      }
  510  
  511      // Bundle the helpers together.
  512      fn do_work(~self) -> Option<~Scheduler> {
  513          let mut this = self;
  514  
  515          rtdebug!("scheduler calling do work");
  516          match this.find_work() {
  517              Some(task) => {
  518                  rtdebug!("found some work! processing the task");
  519                  return this.process_task(task);
  520              }
  521              None => {
  522                  rtdebug!("no work was found, returning the scheduler struct");
  523                  return Some(this);
  524              }
  525          }
  526      }
  527  
  528      /// Called by a running task to end execution, after which it will
  529      /// be recycled by the scheduler for reuse in a new task.
  530      pub fn terminate_current_task(~self) {
  531          // Similar to deschedule running task and then, but cannot go through
  532          // the task-blocking path. The task is already dying.
  533          let mut this = self;
  534          let stask = this.sched_task.take_unwrap();
  535          do this.change_task_context(stask) |sched, mut dead_task| {
  536              let coroutine = dead_task.coroutine.take_unwrap();
  537              coroutine.recycle(&mut sched.stack_pool);
  538          }
  539      }
  540  
  541      // Scheduling a task requires a few checks to make sure the task
  542      // ends up in the appropriate location. The run_anything flag on
  543      // the scheduler and the home on the task need to be checked. This
  544      // helper performs that check. It takes a function that specifies
  545      // how to queue the the provided task if that is the correct
  546      // action. This is a "core" function that requires handling the
  547      // returned Option correctly.
  548  
  549      pub fn schedule_task(~self, task: ~Task,
  550                           schedule_fn: ~fn(sched: ~Scheduler, task: ~Task))
  551          -> Option<~Scheduler> {
  552  
  553          // is the task home?
  554          let is_home = task.is_home_no_tls(&self);
  555  
  556          // does the task have a home?
  557          let homed = task.homed();
  558  
  559          let mut this = self;
  560  
  561          if is_home || (!homed && this.run_anything) {
  562              // here we know we are home, execute now OR we know we
  563              // aren't homed, and that this sched doesn't care
  564              rtdebug!("task: %u is on ok sched, executing", to_uint(task));
  565              schedule_fn(this, task);
  566              return None;
  567          } else if !homed && !this.run_anything {
  568              // the task isn't homed, but it can't be run here
  569              this.send_to_friend(task);
  570              return Some(this);
  571          } else {
  572              // task isn't home, so don't run it here, send it home
  573              Scheduler::send_task_home(task);
  574              return Some(this);
  575          }
  576      }

  578      // There are two contexts in which schedule_task can be called:
  579      // inside the scheduler, and inside a task. These contexts handle
  580      // executing the task slightly differently. In the scheduler
  581      // context case we want to receive the scheduler as an input, and
  582      // manually deal with the option. In the task context case we want
  583      // to use TLS to find the scheduler, and deal with the option
  584      // inside the helper.
  585  
  586      pub fn sched_schedule_task(~self, task: ~Task) -> Option<~Scheduler> {
  587          do self.schedule_task(task) |sched, next_task| {
  588              sched.resume_task_immediately(next_task);
  589          }
  590      }

  592      // Task context case - use TLS.
  593      pub fn run_task(task: ~Task) {
  594          let sched = Local::take::<Scheduler>();
  595          let opt = do sched.schedule_task(task) |sched, next_task| {
  596              do sched.switch_running_tasks_and_then(next_task) |sched, last_task| {
  597                  sched.enqueue_blocked_task(last_task);
  598              }
  599          };
  600          opt.map_move(Local::put);
  601      }

  603      // The primary function for changing contexts. In the current
  604      // design the scheduler is just a slightly modified GreenTask, so
  605      // all context swaps are from Task to Task. The only difference
  606      // between the various cases is where the inputs come from, and
  607      // what is done with the resulting task. That is specified by the
  608      // cleanup function f, which takes the scheduler and the
  609      // old task as inputs.
  610  
  611      pub fn change_task_context(~self,
  612                                 next_task: ~Task,
  613                                 f: &fn(&mut Scheduler, ~Task)) {
  614          let mut this = self;
  615  
  616          // The current task is grabbed from TLS, not taken as an input.
  617          let current_task: ~Task = Local::take::<Task>();
  618  
  619          // Check that the task is not in an atomically() section (e.g.,
  620          // holding a pthread mutex, which could deadlock the scheduler).
  621          current_task.death.assert_may_sleep();
  622  
  623          // These transmutes do something fishy with a closure.
  624          let f_fake_region = unsafe {
  625              transmute::<&fn(&mut Scheduler, ~Task),
  626                          &fn(&mut Scheduler, ~Task)>(f)
  627          };
  628          let f_opaque = ClosureConverter::from_fn(f_fake_region);
  629  
  630          // The current task is placed inside an enum with the cleanup
  631          // function. This enum is then placed inside the scheduler.
  632          this.enqueue_cleanup_job(GiveTask(current_task, f_opaque));
  633  
  634          // The scheduler is then placed inside the next task.
  635          let mut next_task = next_task;
  636          next_task.sched = Some(this);
  637  
  638          // However we still need an internal mutable pointer to the
  639          // original task. The strategy here was "arrange memory, then
  640          // get pointers", so we crawl back up the chain using
  641          // transmute to eliminate borrowck errors.
  642          unsafe {
  643  
  644              let sched: &mut Scheduler =
  645                  transmute_mut_region(*next_task.sched.get_mut_ref());
  646  
  647              let current_task: &mut Task = match sched.cleanup_job {
  648                  Some(GiveTask(ref task, _)) => {
  649                      transmute_mut_region(*transmute_mut_unsafe(task))
  650                  }
  651                  Some(DoNothing) => {
  652                      rtabort!("no next task");
  653                  }
  654                  None => {
  655                      rtabort!("no cleanup job");
  656                  }
  657              };
  658  
  659              let (current_task_context, next_task_context) =
  660                  Scheduler::get_contexts(current_task, next_task);
  661  
  662              // Done with everything - put the next task in TLS. This
  663              // works because due to transmute the borrow checker
  664              // believes that we have no internal pointers to
  665              // next_task.
  666              Local::put(next_task);
  667  
  668              // The raw context swap operation. The next action taken
  669              // will be running the cleanup job from the context of the
  670              // next task.
  671              Context::swap(current_task_context, next_task_context);
  672          }
  673  
  674          // When the context swaps back to this task we immediately
  675          // run the cleanup job, as expected by the previously called
  676          // swap_contexts function.
  677          unsafe {
  678              let sched = Local::unsafe_borrow::<Scheduler>();
  679              (*sched).run_cleanup_job();
  680  
  681              // Must happen after running the cleanup job (of course).
  682              let task = Local::unsafe_borrow::<Task>();
  683              (*task).death.check_killed((*task).unwinder.unwinding);
  684          }
  685      }

  687      // Old API for task manipulation implemented over the new core
  688      // function.
  689  
  690      pub fn resume_task_immediately(~self, task: ~Task) {
  691          do self.change_task_context(task) |sched, stask| {
  692              sched.sched_task = Some(stask);
  693          }
  694      }

  696      pub fn resume_blocked_task_immediately(~self, blocked_task: BlockedTask) {
  697          match blocked_task.wake() {
  698              Some(task) => self.resume_task_immediately(task),
  699              None => Local::put(self),
  700          };
  701      }

  703      /// Block a running task, context switch to the scheduler, then pass the
  704      /// blocked task to a closure.
  705      ///
  706      /// # Safety note
  707      ///
  708      /// The closure here is a *stack* closure that lives in the
  709      /// running task.  It gets transmuted to the scheduler's lifetime
  710      /// and called while the task is blocked.
  711      ///
  712      /// This passes a Scheduler pointer to the fn after the context switch
  713      /// in order to prevent that fn from performing further scheduling operations.
  714      /// Doing further scheduling could easily result in infinite recursion.
  715      pub fn deschedule_running_task_and_then(~self, f: &fn(&mut Scheduler, BlockedTask)) {
  716          // Trickier - we need to get the scheduler task out of self
  717          // and use it as the destination.
  718          let mut this = self;
  719          let stask = this.sched_task.take_unwrap();
  720          // Otherwise this is the same as below.
  721          this.switch_running_tasks_and_then(stask, f);
  722      }

  724      pub fn switch_running_tasks_and_then(~self, next_task: ~Task,
  725                                           f: &fn(&mut Scheduler, BlockedTask)) {
  726          // This is where we convert the BlockedTask-taking closure into one
  727          // that takes just a Task, and is aware of the block-or-killed protocol.
  728          do self.change_task_context(next_task) |sched, task| {
  729              // Task might need to receive a kill signal instead of blocking.
  730              // We can call the "and_then" only if it blocks successfully.
  731              match BlockedTask::try_block(task) {
  732                  Left(killed_task) => sched.enqueue_task(killed_task),
  733                  Right(blocked_task) => f(sched, blocked_task),
  734              }
  735          }
  736      }

  738      // A helper that looks up the scheduler and runs a task later by
  739      // enqueuing it.
  740      pub fn run_task_later(next_task: ~Task) {
  741          // We aren't performing a scheduler operation, so we want to
  742          // put the Scheduler back when we finish.
  743          let next_task = Cell::new(next_task);
  744          do Local::borrow::<Scheduler,()> |sched| {
  745              sched.enqueue_task(next_task.take());
  746          };
  747      }

  750      // swap. This is unsafe - we are getting mutable internal
  751      // references to keep even when we don't own the tasks. It looks
  752      // kinda safe because we are doing transmutes before passing in
  753      // the arguments.
  754      pub fn get_contexts<'a>(current_task: &mut Task, next_task: &mut Task) ->

  758          let next_task_context =
  759              &mut next_task.coroutine.get_mut_ref().saved_context;
  760          unsafe {
  761              (transmute_mut_region(current_task_context),
  762               transmute_mut_region(next_task_context))
  763          }
  764      }

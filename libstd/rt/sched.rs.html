   32 /// The Scheduler is responsible for coordinating execution of Coroutines
   33 /// on a single thread. When the scheduler is running it is owned by
   34 /// thread local storage and the running task is owned by the
   35 /// scheduler.
   36 ///
   37 /// XXX: This creates too many callbacks to run_sched_once, resulting
   38 /// in too much allocation and too many events.
   39 pub struct Scheduler {
   40     /// A queue of available work. Under a work-stealing policy there
   41     /// is one per Scheduler.
   42     work_queue: WorkQueue<~Task>,
   43     /// The queue of incoming messages from other schedulers.
   44     /// These are enqueued by SchedHandles after which a remote callback
   45     /// is triggered to handle the message.
   46     priv message_queue: MessageQueue<SchedMessage>,
   47     /// A shared list of sleeping schedulers. We'll use this to wake
   48     /// up schedulers when pushing work onto the work queue.
   49     sleeper_list: SleeperList,
   50     /// Indicates that we have previously pushed a handle onto the
   51     /// SleeperList but have not yet received the Wake message.
   52     /// Being `true` does not necessarily mean that the scheduler is
   53     /// not active since there are multiple event sources that may
   54     /// wake the scheduler. It just prevents the scheduler from pushing
   55     /// multiple handles onto the sleeper list.
   56     priv sleepy: bool,
   57     /// A flag to indicate we've received the shutdown message and should
   58     /// no longer try to go to sleep, but exit instead.
   59     no_sleep: bool,
   60     stack_pool: StackPool,
   61     /// The event loop used to drive the scheduler and perform I/O
   62     event_loop: ~EventLoopObject,
   63     /// The scheduler runs on a special task.
   64     sched_task: Option<~Task>,
   65     /// An action performed after a context switch on behalf of the
   66     /// code running before the context switch
   67     priv cleanup_job: Option<CleanupJob>,
   68     metrics: SchedMetrics,
   69     /// Should this scheduler run any task, or only pinned tasks?
   70     run_anything: bool,
   71     /// If the scheduler shouldn't run some tasks, a friend to send
   72     /// them to.
   73     friend_handle: Option<SchedHandle>
   74 }

  137     // Take a main task to run, and a scheduler to run it in. Create a
  138     // scheduler task and bootstrap into it.
  139     pub fn bootstrap(~self, task: ~Task) {
  140 
  141         // Initialize the TLS key.
  142         local_ptr::init_tls_key();
  143 
  144         // Create a task for the scheduler with an empty context.
  145         let sched_task = ~Task::new_sched_task();
  146 
  147         // Now that we have an empty task struct for the scheduler
  148         // task, put it in TLS.
  149         Local::put::(sched_task);
  150 
  151         // Now, as far as all the scheduler state is concerned, we are
  152         // inside the "scheduler" context. So we can act like the
  153         // scheduler and resume the provided task.
  154         self.resume_task_immediately(task);
  155 
  156         // Now we are back in the scheduler context, having
  157         // successfully run the input task. Start by running the
  158         // scheduler. Grab it out of TLS - performing the scheduler
  159         // action will have given it away.
  160         let sched = Local::take::<Scheduler>();
  161 
  162         rtdebug!("starting scheduler %u", sched.sched_id());
  163 
  164         sched.run();
  165 
  166         // Now that we are done with the scheduler, clean up the
  167         // scheduler task. Do so by removing it from TLS and manually
  168         // cleaning up the memory it uses. As we didn't actually call
  169         // task.run() on the scheduler task we never get through all
  170         // the cleanup code it runs.
  171         let mut stask = Local::take::<Task>();
  172 
  173         rtdebug!("stopping scheduler %u", stask.sched.get_ref().sched_id());
  174 
  175         // Should not have any messages
  176         let message = stask.sched.get_mut_ref().message_queue.pop();
  177         assert!(message.is_none());
  178 
  179         stask.destroyed = true;
  180     }

  182     // This does not return a scheduler, as the scheduler is placed
  183     // inside the task.
  184     pub fn run(~self) {
  185 
  186         let mut self_sched = self;
  187 
  188         // Always run through the scheduler loop at least once so that
  189         // we enter the sleep state and can then be woken up by other
  190         // schedulers.
  191         self_sched.event_loop.callback(Scheduler::run_sched_once);
  192 
  193         // This is unsafe because we need to place the scheduler, with
  194         // the event_loop inside, inside our task. But we still need a
  195         // mutable reference to the event_loop to give it the "run"
  196         // command.
  197         unsafe {
  198             let event_loop: *mut ~EventLoopObject = &mut self_sched.event_loop;
  199 
  200             // Our scheduler must be in the task before the event loop
  201             // is started.
  202             let self_sched = Cell::new(self_sched);
  203             do Local::borrow::<Task,()> |stask| {
  204                 stask.sched = Some(self_sched.take());
  205             };
  206 
  207             (*event_loop).run();
  208         }
  209     }
  210 
  211     // One iteration of the scheduler loop, always run at least once.
  212 
  213     // The model for this function is that you continue through it
  214     // until you either use the scheduler while performing a schedule
  215     // action, in which case you give it away and do not return, or
  216     // you reach the end and sleep. In the case that a scheduler
  217     // action is performed the loop is evented such that this function
  218     // is called again.
  219     fn run_sched_once() {
  220 
  221         // When we reach the scheduler context via the event loop we
  222         // already have a scheduler stored in our local task, so we
  223         // start off by taking it. This is the only path through the
  224         // scheduler where we get the scheduler this way.
  225         let sched = Local::take::<Scheduler>();
  226 
  227         // Our first task is to read mail to see if we have important
  228         // messages.
  229 
  230         // 1) A wake message is easy, mutate sched struct and return
  231         //    it.
  232         // 2) A shutdown is also easy, shutdown.
  233         // 3) A pinned task - we resume immediately and do not return
  234         //    here.
  235         // 4) A message from another scheduler with a non-homed task
  236         //    to run here.
  237 
  238         let result = sched.interpret_message_queue();
  239         let sched = match result {
  240             Some(sched) => {
  241                 // We did not resume a task, so we returned.
  242                 sched
  243             }
  244             None => {
  245                 return;
  246             }
  247         };
  248 
  249         // Second activity is to try resuming a task from the queue.
  250 
  251         let result = sched.resume_task_from_queue();
  252         let mut sched = match result {
  253             Some(sched) => {
  254                 // Failed to dequeue a task, so we return.
  255                 sched
  256             }
  257             None => {
  258                 return;
  259             }
  260         };
  261 
  262         // If we got here then there was no work to do.
  263         // Generate a SchedHandle and push it to the sleeper list so
  264         // somebody can wake us up later.
  265         sched.metrics.wasted_turns += 1;
  266         if !sched.sleepy && !sched.no_sleep {
  267             rtdebug!("scheduler has no work to do, going to sleep");
  268             sched.metrics.sleepy_times += 1;
  269             sched.sleepy = true;
  270             let handle = sched.make_handle();
  271             sched.sleeper_list.push(handle);
  272         } else {
  273             rtdebug!("not sleeping, already doing so or no_sleep set");
  274         }
  275 
  276         // Finished a cycle without using the Scheduler. Place it back
  277         // in TLS.
  278         Local::put(sched);
  279     }

  285             remote: remote,
  286             queue: self.message_queue.clone(),
  287             sched_id: self.sched_id()
  288         };
  289     }

  291     /// Schedule a task to be executed later.
  292     ///
  293     /// Pushes the task onto the work stealing queue and tells the
  294     /// event loop to run it later. Always use this instead of pushing
  295     /// to the work queue directly.
  296     pub fn enqueue_task(&mut self, task: ~Task) {
  297 
  298         let this = self;
  299 
  300         // We push the task onto our local queue clone.
  301         this.work_queue.push(task);
  302         this.event_loop.callback(Scheduler::run_sched_once);
  303 
  304         // We've made work available. Notify a
  305         // sleeping scheduler.
  306 
  307         // XXX: perf. Check for a sleeper without
  308         // synchronizing memory.  It's not critical
  309         // that we always find it.
  310 
  311         // XXX: perf. If there's a sleeper then we
  312         // might as well just send it the task
  313         // directly instead of pushing it to the
  314         // queue. That is essentially the intent here
  315         // and it is less work.
  316         match this.sleeper_list.pop() {
  317             Some(handle) => {
  318                 let mut handle = handle;
  319                 handle.send(Wake)
  320             }
  321             None => { (/* pass */) }
  322         };
  323     }
  324 
  325     /// As enqueue_task, but with the possibility for the blocked task to
  326     /// already have been killed.
  327     pub fn enqueue_blocked_task(&mut self, blocked_task: BlockedTask) {
  328         do blocked_task.wake().map_consume |task| {
  329             self.enqueue_task(task);
  330         };
  331     }

  335     // This function returns None if the scheduler is "used", or it
  336     // returns the still-available scheduler.
  337     fn interpret_message_queue(~self) -> Option<~Scheduler> {
  338 
  339         let mut this = self;
  340         match this.message_queue.pop() {
  341             Some(PinnedTask(task)) => {
  342                 let mut task = task;
  343                 this.event_loop.callback(Scheduler::run_sched_once);
  344                 task.give_home(Sched(this.make_handle()));
  345                 this.resume_task_immediately(task);
  346                 return None;
  347             }
  348             Some(TaskFromFriend(task)) => {
  349                 this.event_loop.callback(Scheduler::run_sched_once);
  350                 rtdebug!("got a task from a friend. lovely!");
  351                 return this.sched_schedule_task(task);
  352             }
  353             Some(Wake) => {
  354                 this.sleepy = false;
  355                 return Some(this);
  356             }
  357             Some(Shutdown) => {
  358                 if this.sleepy {
  359                     // There may be an outstanding handle on the
  360                     // sleeper list.  Pop them all to make sure that's
  361                     // not the case.
  362                     loop {
  363                         match this.sleeper_list.pop() {
  364                             Some(handle) => {
  365                                 let mut handle = handle;
  366                                 handle.send(Wake);
  367                             }
  368                             None => break
  369                         }
  370                     }
  371                 }
  372                 // No more sleeping. After there are no outstanding
  373                 // event loop references we will shut down.
  374                 this.no_sleep = true;
  375                 this.sleepy = false;
  376                 // YYY: Does a shutdown count as a "use" of the
  377                 // scheduler? This seems to work - so I'm leaving it
  378                 // this way despite not having a solid rational for
  379                 // why I should return the scheduler here.
  380                 return Some(this);
  381             }
  382             None => {
  383                 return Some(this);
  384             }
  385         }
  386     }

  388     /// Given an input Coroutine sends it back to its home scheduler.
  389     fn send_task_home(task: ~Task) {
  390         let mut task = task;
  391         let mut home = task.take_unwrap_home();
  392         match home {
  393             Sched(ref mut home_handle) => {
  394                 home_handle.send(PinnedTask(task));
  395             }
  396             AnySched => {
  397                 rtabort!("error: cannot send anysched task home");
  398             }
  399         }
  400     }
  401 
  402     /// Take a non-homed task we aren't allowed to run here and send
  403     /// it to the designated friend scheduler to execute.
  404     fn send_to_friend(&mut self, task: ~Task) {
  405         rtdebug!("sending a task to friend");
  406         match self.friend_handle {
  407             Some(ref mut handle) => {
  408                 handle.send(TaskFromFriend(task));
  409             }
  410             None => {
  411                 rtabort!("tried to send task to a friend but scheduler has no friends");
  412             }
  413         }
  414     }
  415 
  416     // Resume a task from the queue - but also take into account that
  417     // it might not belong here.
  418 
  419     // If we perform a scheduler action we give away the scheduler ~
  420     // pointer, if it is still available we return it.
  421 
  422     fn resume_task_from_queue(~self) -> Option<~Scheduler> {
  423 
  424         let mut this = self;
  425 
  426         match this.work_queue.pop() {
  427             Some(task) => {
  428                 let mut task = task;
  429                 let home = task.take_unwrap_home();
  430                 match home {
  431                     Sched(home_handle) => {
  432                         if home_handle.sched_id != this.sched_id() {
  433                             task.give_home(Sched(home_handle));
  434                             Scheduler::send_task_home(task);
  435                             return Some(this);
  436                         } else {
  437                             this.event_loop.callback(Scheduler::run_sched_once);
  438                             task.give_home(Sched(home_handle));
  439                             this.resume_task_immediately(task);
  440                             return None;
  441                         }
  442                     }
  443                     AnySched if this.run_anything => {
  444                         this.event_loop.callback(Scheduler::run_sched_once);
  445                         task.give_home(AnySched);
  446                         this.resume_task_immediately(task);
  447                         return None;
  448                     }
  449                     AnySched => {
  450                         task.give_home(AnySched);

  454                 }

  462     /// Called by a running task to end execution, after which it will
  463     /// be recycled by the scheduler for reuse in a new task.
  464     pub fn terminate_current_task(~self) {
  465         // Similar to deschedule running task and then, but cannot go through
  466         // the task-blocking path. The task is already dying.
  467         let mut this = self;
  468         let stask = this.sched_task.take_unwrap();
  469         do this.change_task_context(stask) |sched, mut dead_task| {
  470             let coroutine = dead_task.coroutine.take_unwrap();
  471             coroutine.recycle(&mut sched.stack_pool);
  472         }
  473     }

  475     // Scheduling a task requires a few checks to make sure the task
  476     // ends up in the appropriate location. The run_anything flag on
  477     // the scheduler and the home on the task need to be checked. This
  478     // helper performs that check. It takes a function that specifies
  479     // how to queue the the provided task if that is the correct
  480     // action. This is a "core" function that requires handling the
  481     // returned Option correctly.
  482 
  483     pub fn schedule_task(~self, task: ~Task,
  484                          schedule_fn: ~fn(sched: ~Scheduler, task: ~Task))
  485         -> Option<~Scheduler> {
  486 
  487         // is the task home?
  488         let is_home = task.is_home_no_tls(&self);
  489 
  490         // does the task have a home?
  491         let homed = task.homed();
  492 
  493         let mut this = self;
  494 
  495         if is_home || (!homed && this.run_anything) {
  496             // here we know we are home, execute now OR we know we
  497             // aren't homed, and that this sched doesn't care
  498             rtdebug!("task: %u is on ok sched, executing", to_uint(task));
  499             schedule_fn(this, task);
  500             return None;
  501         } else if !homed && !this.run_anything {
  502             // the task isn't homed, but it can't be run here
  503             this.send_to_friend(task);
  504             return Some(this);
  505         } else {
  506             // task isn't home, so don't run it here, send it home
  507             Scheduler::send_task_home(task);
  508             return Some(this);
  509         }
  510     }

  512     // There are two contexts in which schedule_task can be called:
  513     // inside the scheduler, and inside a task. These contexts handle
  514     // executing the task slightly differently. In the scheduler
  515     // context case we want to receive the scheduler as an input, and
  516     // manually deal with the option. In the task context case we want
  517     // to use TLS to find the scheduler, and deal with the option
  518     // inside the helper.
  519 
  520     pub fn sched_schedule_task(~self, task: ~Task) -> Option<~Scheduler> {
  521         do self.schedule_task(task) |sched, next_task| {
  522             sched.resume_task_immediately(next_task);
  523         }
  524     }

  526     // Task context case - use TLS.
  527     pub fn run_task(task: ~Task) {
  528         let sched = Local::take::<Scheduler>();
  529         let opt = do sched.schedule_task(task) |sched, next_task| {
  530             do sched.switch_running_tasks_and_then(next_task) |sched, last_task| {
  531                 sched.enqueue_blocked_task(last_task);
  532             }
  533         };
  534         opt.map_consume(Local::put);
  535     }

  537     // The primary function for changing contexts. In the current
  538     // design the scheduler is just a slightly modified GreenTask, so
  539     // all context swaps are from Task to Task. The only difference
  540     // between the various cases is where the inputs come from, and
  541     // what is done with the resulting task. That is specified by the
  542     // cleanup function f, which takes the scheduler and the
  543     // old task as inputs.
  544 
  545     pub fn change_task_context(~self,
  546                                next_task: ~Task,
  547                                f: &fn(&mut Scheduler, ~Task)) {
  548         let mut this = self;
  549 
  550         // The current task is grabbed from TLS, not taken as an input.
  551         let current_task: ~Task = Local::take::<Task>();
  552 
  553         // Check that the task is not in an atomically() section (e.g.,
  554         // holding a pthread mutex, which could deadlock the scheduler).
  555         current_task.death.assert_may_sleep();
  556 
  557         // These transmutes do something fishy with a closure.
  558         let f_fake_region = unsafe {
  559             transmute::<&fn(&mut Scheduler, ~Task),
  560                         &fn(&mut Scheduler, ~Task)>(f)
  561         };
  562         let f_opaque = ClosureConverter::from_fn(f_fake_region);
  563 
  564         // The current task is placed inside an enum with the cleanup
  565         // function. This enum is then placed inside the scheduler.
  566         this.enqueue_cleanup_job(GiveTask(current_task, f_opaque));
  567 
  568         // The scheduler is then placed inside the next task.
  569         let mut next_task = next_task;
  570         next_task.sched = Some(this);
  571 
  572         // However we still need an internal mutable pointer to the
  573         // original task. The strategy here was "arrange memory, then
  574         // get pointers", so we crawl back up the chain using
  575         // transmute to eliminate borrowck errors.
  576         unsafe {
  577 
  578             let sched: &mut Scheduler =
  579                 transmute_mut_region(*next_task.sched.get_mut_ref());
  580 
  581             let current_task: &mut Task = match sched.cleanup_job {
  582                 Some(GiveTask(ref task, _)) => {
  583                     transmute_mut_region(*transmute_mut_unsafe(task))
  584                 }
  585                 Some(DoNothing) => {
  586                     rtabort!("no next task");
  587                 }
  588                 None => {
  589                     rtabort!("no cleanup job");
  590                 }
  591             };
  592 
  593             let (current_task_context, next_task_context) =
  594                 Scheduler::get_contexts(current_task, next_task);
  595 
  596             // Done with everything - put the next task in TLS. This
  597             // works because due to transmute the borrow checker
  598             // believes that we have no internal pointers to
  599             // next_task.
  600             Local::put(next_task);
  601 
  602             // The raw context swap operation. The next action taken
  603             // will be running the cleanup job from the context of the
  604             // next task.
  605             Context::swap(current_task_context, next_task_context);
  606         }
  607 
  608         // When the context swaps back to this task we immediately
  609         // run the cleanup job, as expected by the previously called
  610         // swap_contexts function.
  611         unsafe {
  612             let sched = Local::unsafe_borrow::<Scheduler>();
  613             (*sched).run_cleanup_job();
  614 
  615             // Must happen after running the cleanup job (of course).
  616             let task = Local::unsafe_borrow::<Task>();
  617             (*task).death.check_killed((*task).unwinder.unwinding);
  618         }
  619     }

  637     /// Block a running task, context switch to the scheduler, then pass the
  638     /// blocked task to a closure.
  639     ///
  640     /// # Safety note
  641     ///
  642     /// The closure here is a *stack* closure that lives in the
  643     /// running task.  It gets transmuted to the scheduler's lifetime
  644     /// and called while the task is blocked.
  645     ///
  646     /// This passes a Scheduler pointer to the fn after the context switch
  647     /// in order to prevent that fn from performing further scheduling operations.
  648     /// Doing further scheduling could easily result in infinite recursion.
  649     pub fn deschedule_running_task_and_then(~self, f: &fn(&mut Scheduler, BlockedTask)) {
  650         // Trickier - we need to get the scheduler task out of self
  651         // and use it as the destination.
  652         let mut this = self;
  653         let stask = this.sched_task.take_unwrap();
  654         // Otherwise this is the same as below.
  655         this.switch_running_tasks_and_then(stask, f);
  656     }

  658     pub fn switch_running_tasks_and_then(~self, next_task: ~Task,
  659                                          f: &fn(&mut Scheduler, BlockedTask)) {
  660         // This is where we convert the BlockedTask-taking closure into one
  661         // that takes just a Task, and is aware of the block-or-killed protocol.
  662         do self.change_task_context(next_task) |sched, task| {
  663             // Task might need to receive a kill signal instead of blocking.
  664             // We can call the "and_then" only if it blocks successfully.
  665             match BlockedTask::try_block(task) {
  666                 Left(killed_task) => sched.enqueue_task(killed_task),
  667                 Right(blocked_task) => f(sched, blocked_task),
  668             }
  669         }
  670     }

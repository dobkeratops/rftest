  32  /// Parse an individual log level that is either a number or a symbolic log level
  33  fn parse_log_level(level: &str) -> Option<u32> {
  34      let num = from_str::<u32>(level);
  35      let mut log_level;
  36      match num {
  37          Some(num) => {
  38              if num < MAX_LOG_LEVEL {
  39                  log_level = Some(num);
  40              } else {
  41                  log_level = Some(MAX_LOG_LEVEL);
  42              }
  43          }
  44          _ => {
  45              let position = log_level_names.iter().position(|&name| name == level);
  46              match position {
  47                  Some(position) => {
  48                      log_level = Some(u32::min(MAX_LOG_LEVEL, (position + 1) as u32))
  49                  },
  50                  _ => {
  51                      log_level = None;
  52                  }
  53              }
  54          }
  55      }
  56      log_level
  57  }

  59  /// Parse a logging specification string (e.g: "crate1,crate2::mod3,crate3::x=1")
  60  /// and return a vector with log directives.
  61  /// Valid log levels are 0-255, with the most likely ones being 1-4 (defined in std::).
  62  /// Also supports string log levels of error, warn, info, and debug
  63  fn parse_logging_spec(spec: ~str) -> ~[LogDirective]{
  64      let mut dirs = ~[];
  65      for s in spec.split_iter(',') {
  66          let parts: ~[&str] = s.split_iter('=').collect();
  67          let mut log_level;
  68          let mut name = Some(parts[0].to_owned());
  69          match parts.len() {
  70              1 => {
  71                  //if the single argument is a log-level string or number,
  72                  //treat that as a global fallback
  73                  let possible_log_level = parse_log_level(parts[0]);
  74                  match possible_log_level {
  75                      Some(num) => {
  76                          name = None;
  77                          log_level = num;
  78                      },
  79                      _ => {
  80                          log_level = MAX_LOG_LEVEL
  81                      }
  82                  }
  83              }
  84              2 => {
  85                  let possible_log_level = parse_log_level(parts[1]);
  86                  match possible_log_level {
  87                      Some(num) => {
  88                          log_level = num;
  89                      },
  90                      _ => {
  91                          dumb_println(fmt!("warning: invalid logging spec \
  92                                             '%s', ignoring it", parts[1]));
  93                          loop;
  94                      }
  95                  }
  96              },
  97              _ => {
  98                  dumb_println(fmt!("warning: invalid logging spec '%s',\
  99                                    ignoring it", s));
 100                  loop;
 101              }
 102          }
 103          let dir = LogDirective {name: name, level: log_level};
 104          dirs.push(dir);
 105      }
 106      return dirs;
 107  }

 111  fn update_entry(dirs: &[LogDirective], entry: *mut ModEntry) -> u32 {
 112      let mut new_lvl: u32 = DEFAULT_LOG_LEVEL;
 113      let mut longest_match = -1i;
 114      unsafe {
 115          for dir in dirs.iter() {
 116              match dir.name {
 117                  None => {
 118                      if longest_match == -1 {
 119                          longest_match = 0;
 120                          new_lvl = dir.level;
 121                      }
 122                  }
 123                  Some(ref dir_name) => {
 124                      let name = from_c_str((*entry).name);
 125                      let len = dir_name.len() as int;
 126                      if name.starts_with(*dir_name) &&
 127                          len >= longest_match {
 128                          longest_match = len;
 129                          new_lvl = dir.level;
 130                      }
 131                  }
 132              };
 133          }
 134          *(*entry).log_level = new_lvl;
 135      }
 136      if longest_match >= 0 { return 1; } else { return 0; }
 137  }

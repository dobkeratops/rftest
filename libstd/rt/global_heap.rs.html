  37  pub unsafe fn malloc_raw(size: uint) -> *mut u8 {
  38      // `malloc(0)` may allocate, but it may also return a null pointer
  39      // http://pubs.opengroup.org/onlinepubs/9699919799/functions/malloc.html
  40      if size == 0 {

  49      }
  50  }

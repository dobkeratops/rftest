 213  /// Takes each element in the `Iterator`: if it is an `Err`, no further
 214  /// elements are taken, and the `Err` is returned. Should no `Err` occur, a
 215  /// vector containing the values of each `Result` is returned.
 216  ///
 217  /// Here is an example which increments every integer in a vector,
 218  /// checking for overflow:
 219  ///
 220  ///     fn inc_conditionally(x: uint) -> Result<uint, &'static str> {
 221  ///         if x == uint::MAX { return Err("overflow"); }
 222  ///         else { return Ok(x+1u); }
 223  ///     }
 224  ///     let v = [1u, 2, 3];
 225  ///     let res = collect(v.iter().map(|&x| inc_conditionally(x)));
 226  ///     assert!(res == Ok(~[2u, 3, 4]));
 227  #[inline]
 228  pub fn collect<T, E, Iter: Iterator<Result<T, E>>, V: FromIterator<T>>(iter: Iter) -> Result<V, E> {
 229      // FIXME(#11084): This should be twice as fast once this bug is closed.
 230      let mut iter = iter.scan(None, |state, x| {
 231          match x {
 232              Ok(x) => Some(x),
 233              Err(err) => {
 234                  *state = Some(err);
 235                  None
 236              }
 237          }
 238      });
 239  
 240      let v: V = FromIterator::from_iterator(iter.by_ref());
 241  
 242      match iter.state {
 243          Some(err) => Err(err),
 244          None => Ok(v),
 245      }
 246  }

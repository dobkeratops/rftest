  354      fn classify(&self) -> FPCategory {
  355          static EXP_MASK: u64 = 0x7ff0000000000000;
  356          static MAN_MASK: u64 = 0x000fffffffffffff;
  357

  360              (0, 0)        => FPZero,
  361              (_, 0)        => FPSubnormal,
  362              (0, EXP_MASK) => FPInfinite,

  365          }

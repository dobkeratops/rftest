 21  impl BitCount for i8 {
 22      /// Counts the number of bits set. Wraps LLVM's `ctpop` intrinsic.
 23      #[inline]
 24      fn population_count(&self) -> i8 { unsafe { intrinsics::ctpop8(*self) } }
 25  
 26      /// Counts the number of leading zeros. Wraps LLVM's `ctlz` intrinsic.
 27      #[inline]
 28      fn leading_zeros(&self) -> i8 { unsafe { intrinsics::ctlz8(*self) } }
 29  
 30      /// Counts the number of trailing zeros. Wraps LLVM's `cttz` intrinsic.
 31      #[inline]
 32      fn trailing_zeros(&self) -> i8 { unsafe { intrinsics::cttz8(*self) } }
 33  }

  27  /// A flag that specifies whether to use exponential (scientific) notation.
  28  pub enum ExponentFormat {
  29      /// Do not use exponential notation.
  30      ExpNone,
  31      /// Use exponential notation with the exponent having a base of 10 and the
  32      /// exponent sign being `e` or `E`. For example, 1000 would be printed
  33      /// 1e3.
  34      ExpDec,
  35      /// Use exponential notation with the exponent having a base of 2 and the
  36      /// exponent sign being `p` or `P`. For example, 8 would be printed 1p3.
  37      ExpBin,
  38  }

 146  /**
 147   * Converts an integral number to its string representation as a byte vector.
 148   * This is meant to be a common base implementation for all integral string
 149   * conversion functions like `to_str()` or `to_str_radix()`.
 150   *
 151   * # Arguments
 152   * - `num`           - The number to convert. Accepts any number that
 153   *                     implements the numeric traits.
 154   * - `radix`         - Base to use. Accepts only the values 2-36.
 155   * - `sign`          - How to emit the sign. Options are:
 156   *     - `SignNone`: No sign at all. Basically emits `abs(num)`.
 157   *     - `SignNeg`:  Only `-` on negative values.
 158   *     - `SignAll`:  Both `+` on positive, and `-` on negative numbers.
 159   * - `f`             - a callback which will be invoked for each ascii character
 160   *                     which composes the string representation of this integer
 161   *
 162   * # Return value
 163   * A tuple containing the byte vector, and a boolean flag indicating
 164   * whether it represents a special value like `inf`, `-inf`, `NaN` or not.
 165   * It returns a tuple because there can be ambiguity between a special value
 166   * and a number representation at higher bases.
 167   *
 168   * # Failure
 169   * - Fails if `radix` < 2 or `radix` > 36.
 170   */
 171  pub fn int_to_str_bytes_common<T: Int>(num: T, radix: uint, sign: SignFormat, f: |u8|) {
 172      assert!(2 <= radix && radix <= 36);
 173  
 174      let _0: T = Zero::zero();
 175  
 176      let neg = num < _0;
 177      let radix_gen: T = cast(radix).unwrap();
 178  
 179      let mut deccum = num;
 180      // This is just for integral types, the largest of which is a u64. The
 181      // smallest base that we can have is 2, so the most number of digits we're
 182      // ever going to have is 64
 183      let mut buf = [0u8, ..64];
 184      let mut cur = 0;
 185  
 186      // Loop at least once to make sure at least a `0` gets emitted.
 187      loop {
 188          // Calculate the absolute value of each digit instead of only
 189          // doing it once for the whole number because a
 190          // representable negative number doesn't necessary have an
 191          // representable additive inverse of the same type
 192          // (See twos complement). But we assume that for the
 193          // numbers [-35 .. 0] we always have [0 .. 35].
 194          let current_digit_signed = deccum % radix_gen;
 195          let current_digit = if current_digit_signed < _0 {
 196              -current_digit_signed
 197          } else {
 198              current_digit_signed
 199          };
 200          buf[cur] = match current_digit.to_u8().unwrap() {
 201              i @ 0..9 => '0' as u8 + i,
 202              i        => 'a' as u8 + (i - 10),
 203          };
 204          cur += 1;
 205  
 206          deccum = deccum / radix_gen;
 207          // No more digits to calculate for the non-fractional part -> break
 208          if deccum == _0 { break; }
 209      }
 210  
 211      // Decide what sign to put in front
 212      match sign {
 213          SignNeg | SignAll if neg => { f('-' as u8); }
 214          SignAll => { f('+' as u8); }
 215          _ => ()
 216      }
 217  
 218      // We built the number in reverse order, so un-reverse it here
 219      while cur > 0 {
 220          cur -= 1;
 221          f(buf[cur]);
 222      }
 223  }

 225  /**
 226   * Converts a number to its string representation as a byte vector.
 227   * This is meant to be a common base implementation for all numeric string
 228   * conversion functions like `to_str()` or `to_str_radix()`.
 229   *
 230   * # Arguments
 231   * - `num`           - The number to convert. Accepts any number that
 232   *                     implements the numeric traits.
 233   * - `radix`         - Base to use. Accepts only the values 2-36. If the exponential notation
 234   *                     is used, then this base is only used for the significand. The exponent
 235   *                     itself always printed using a base of 10.
 236   * - `negative_zero` - Whether to treat the special value `-0` as
 237   *                     `-0` or as `+0`.
 238   * - `sign`          - How to emit the sign. See `SignFormat`.
 239   * - `digits`        - The amount of digits to use for emitting the fractional
 240   *                     part, if any. See `SignificantDigits`.
 241   * - `exp_format`   - Whether or not to use the exponential (scientific) notation.
 242   *                    See `ExponentFormat`.
 243   * - `exp_capital`   - Whether or not to use a capital letter for the exponent sign, if
 244   *                     exponential notation is desired.
 245   *
 246   * # Return value
 247   * A tuple containing the byte vector, and a boolean flag indicating
 248   * whether it represents a special value like `inf`, `-inf`, `NaN` or not.
 249   * It returns a tuple because there can be ambiguity between a special value
 250   * and a number representation at higher bases.
 251   *
 252   * # Failure
 253   * - Fails if `radix` < 2 or `radix` > 36.
 254   * - Fails if `radix` > 14 and `exp_format` is `ExpDec` due to conflict
 255   *   between digit and exponent sign `'e'`.
 256   * - Fails if `radix` > 25 and `exp_format` is `ExpBin` due to conflict
 257   *   between digit and exponent sign `'p'`.
 258   */
 259  pub fn float_to_str_bytes_common<T:NumCast+Zero+One+Eq+Ord+Float+Round+
 260                                    Div<T,T>+Neg<T>+Rem<T,T>+Mul<T,T>>(
 261          num: T, radix: uint, negative_zero: bool,
 262          sign: SignFormat, digits: SignificantDigits, exp_format: ExponentFormat, exp_upper: bool
 263          ) -> (~[u8], bool) {
 264      assert!(2 <= radix && radix <= 36);
 265      match exp_format {
 266          ExpDec if radix >= DIGIT_E_RADIX       // decimal exponent 'e'
 267            => fail!("float_to_str_bytes_common: radix {} incompatible with \
 268                      use of 'e' as decimal exponent", radix),
 269          ExpBin if radix >= DIGIT_P_RADIX       // binary exponent 'p'
 270            => fail!("float_to_str_bytes_common: radix {} incompatible with \
 271                      use of 'p' as binary exponent", radix),
 272          _ => ()
 273      }
 274  
 275      let _0: T = Zero::zero();
 276      let _1: T = One::one();
 277  
 278      match num.classify() {
 279          FPNaN => { return ("NaN".as_bytes().to_owned(), true); }
 280          FPInfinite if num > _0 => {
 281              return match sign {
 282                  SignAll => ("+inf".as_bytes().to_owned(), true),
 283                  _       => ("inf".as_bytes().to_owned(), true)
 284              };
 285          }
 286          FPInfinite if num < _0 => {
 287              return match sign {
 288                  SignNone => ("inf".as_bytes().to_owned(), true),
 289                  _        => ("-inf".as_bytes().to_owned(), true),
 290              };
 291          }
 292          _ => {}
 293      }
 294  
 295      let neg = num < _0 || (negative_zero && _1 / num == Float::neg_infinity());
 296      let mut buf: ~[u8] = ~[];
 297      let radix_gen: T   = cast(radix as int).unwrap();
 298  
 299      let (num, exp) = match exp_format {
 300          ExpNone => (num, 0i32),
 301          ExpDec | ExpBin => {
 302              if num == _0 {
 303                  (num, 0i32)
 304              } else {
 305                  let (exp, exp_base) = match exp_format {
 306                      ExpDec => (num.abs().log10().floor(), cast::<f64, T>(10.0f64).unwrap()),
 307                      ExpBin => (num.abs().log2().floor(), cast::<f64, T>(2.0f64).unwrap()),
 308                      ExpNone => unreachable!()
 309                  };
 310  
 311                  (num / exp_base.powf(&exp), cast::<T, i32>(exp).unwrap())
 312              }
 313          }
 314      };
 315  
 316      // First emit the non-fractional part, looping at least once to make
 317      // sure at least a `0` gets emitted.
 318      let mut deccum = num.trunc();
 319      loop {
 320          // Calculate the absolute value of each digit instead of only
 321          // doing it once for the whole number because a
 322          // representable negative number doesn't necessary have an
 323          // representable additive inverse of the same type
 324          // (See twos complement). But we assume that for the
 325          // numbers [-35 .. 0] we always have [0 .. 35].
 326          let current_digit = (deccum % radix_gen).abs();
 327  
 328          // Decrease the deccumulator one digit at a time
 329          deccum = deccum / radix_gen;
 330          deccum = deccum.trunc();
 331  
 332          buf.push(char::from_digit(current_digit.to_int().unwrap() as uint, radix)
 333               .unwrap() as u8);
 334  
 335          // No more digits to calculate for the non-fractional part -> break
 336          if deccum == _0 { break; }
 337      }
 338  
 339      // If limited digits, calculate one digit more for rounding.
 340      let (limit_digits, digit_count, exact) = match digits {
 341          DigAll          => (false, 0u,      false),
 342          DigMax(count)   => (true,  count+1, false),
 343          DigExact(count) => (true,  count+1, true)
 344      };
 345  
 346      // Decide what sign to put in front
 347      match sign {
 348          SignNeg | SignAll if neg => {
 349              buf.push('-' as u8);
 350          }
 351          SignAll => {
 352              buf.push('+' as u8);
 353          }
 354          _ => ()
 355      }
 356  
 357      buf.reverse();
 358  
 359      // Remember start of the fractional digits.
 360      // Points one beyond end of buf if none get generated,
 361      // or at the '.' otherwise.
 362      let start_fractional_digits = buf.len();
 363  
 364      // Now emit the fractional part, if any
 365      deccum = num.fract();
 366      if deccum != _0 || (limit_digits && exact && digit_count > 0) {
 367          buf.push('.' as u8);
 368          let mut dig = 0u;
 369  
 370          // calculate new digits while
 371          // - there is no limit and there are digits left
 372          // - or there is a limit, it's not reached yet and
 373          //   - it's exact
 374          //   - or it's a maximum, and there are still digits left
 375          while (!limit_digits && deccum != _0)
 376             || (limit_digits && dig < digit_count && (
 377                     exact
 378                  || (!exact && deccum != _0)
 379                )
 380          ) {
 381              // Shift first fractional digit into the integer part
 382              deccum = deccum * radix_gen;
 383  
 384              // Calculate the absolute value of each digit.
 385              // See note in first loop.
 386              let current_digit = deccum.trunc().abs();
 387  
 388              buf.push(char::from_digit(
 389                  current_digit.to_int().unwrap() as uint, radix).unwrap() as u8);
 390  
 391              // Decrease the deccumulator one fractional digit at a time
 392              deccum = deccum.fract();
 393              dig += 1u;
 394          }
 395  
 396          // If digits are limited, and that limit has been reached,
 397          // cut off the one extra digit, and depending on its value
 398          // round the remaining ones.
 399          if limit_digits && dig == digit_count {
 400              let ascii2value = |chr: u8| {
 401                  char::to_digit(chr as char, radix).unwrap()
 402              };
 403              let value2ascii = |val: uint| {
 404                  char::from_digit(val, radix).unwrap() as u8
 405              };
 406  
 407              let extra_digit = ascii2value(buf.pop().unwrap());
 408              if extra_digit >= radix / 2 { // -> need to round
 409                  let mut i: int = buf.len() as int - 1;
 410                  loop {
 411                      // If reached left end of number, have to
 412                      // insert additional digit:
 413                      if i < 0
 414                      || buf[i as uint] == '-' as u8
 415                      || buf[i as uint] == '+' as u8 {
 416                          buf.insert((i + 1) as uint, value2ascii(1));
 417                          break;
 418                      }
 419  
 420                      // Skip the '.'
 421                      if buf[i as uint] == '.' as u8 { i -= 1; continue; }
 422  
 423                      // Either increment the digit,
 424                      // or set to 0 if max and carry the 1.
 425                      let current_digit = ascii2value(buf[i as uint]);
 426                      if current_digit < (radix - 1) {
 427                          buf[i as uint] = value2ascii(current_digit+1);
 428                          break;
 429                      } else {
 430                          buf[i as uint] = value2ascii(0);
 431                          i -= 1;
 432                      }
 433                  }
 434              }
 435          }
 436      }
 437  
 438      // if number of digits is not exact, remove all trailing '0's up to
 439      // and including the '.'
 440      if !exact {
 441          let buf_max_i = buf.len() - 1;
 442  
 443          // index to truncate from
 444          let mut i = buf_max_i;
 445  
 446          // discover trailing zeros of fractional part
 447          while i > start_fractional_digits && buf[i] == '0' as u8 {
 448              i -= 1;
 449          }
 450  
 451          // Only attempt to truncate digits if buf has fractional digits
 452          if i >= start_fractional_digits {
 453              // If buf ends with '.', cut that too.
 454              if buf[i] == '.' as u8 { i -= 1 }
 455  
 456              // only resize buf if we actually remove digits
 457              if i < buf_max_i {
 458                  buf = buf.slice(0, i + 1).to_owned();
 459              }
 460          }
 461      } // If exact and trailing '.', just cut that
 462      else {
 463          let max_i = buf.len() - 1;
 464          if buf[max_i] == '.' as u8 {
 465              buf = buf.slice(0, max_i).to_owned();
 466          }
 467      }
 468  
 469      match exp_format {
 470          ExpNone => (),
 471          _ => {
 472              buf.push(match exp_format {
 473                  ExpDec if exp_upper => 'E',
 474                  ExpDec if !exp_upper => 'e',
 475                  ExpBin if exp_upper => 'P',
 476                  ExpBin if !exp_upper => 'p',
 477                  _ => unreachable!()
 478              } as u8);
 479  
 480              int_to_str_bytes_common(exp, 10, sign, |c| buf.push(c));
 481          }
 482      }
 483  
 484      (buf, false)
 485  }
 486  
 487  /**
 488   * Converts a number to its string representation. This is a wrapper for
 489   * `to_str_bytes_common()`, for details see there.
 490   */
 491  #[inline]
 492  pub fn float_to_str_common<T:NumCast+Zero+One+Eq+Ord+NumStrConv+Float+Round+
 493                               Div<T,T>+Neg<T>+Rem<T,T>+Mul<T,T>>(
 494          num: T, radix: uint, negative_zero: bool,
 495          sign: SignFormat, digits: SignificantDigits, exp_format: ExponentFormat, exp_capital: bool
 496          ) -> (~str, bool) {
 497      let (bytes, special) = float_to_str_bytes_common(num, radix,
 498                                 negative_zero, sign, digits, exp_format, exp_capital);
 499      (str::from_utf8_owned(bytes).unwrap(), special)
 500  }
 501  
 502  // Some constants for from_str_bytes_common's input validation,
 503  // they define minimum radix values for which the character is a valid digit.
 504  static DIGIT_P_RADIX: uint = ('p' as uint) - ('a' as uint) + 11u;
 505  static DIGIT_I_RADIX: uint = ('i' as uint) - ('a' as uint) + 11u;
 506  static DIGIT_E_RADIX: uint = ('e' as uint) - ('a' as uint) + 11u;
 507  
 508  /**
 509   * Parses a byte slice as a number. This is meant to
 510   * be a common base implementation for all numeric string conversion
 511   * functions like `from_str()` or `from_str_radix()`.
 512   *
 513   * # Arguments
 514   * - `buf`        - The byte slice to parse.
 515   * - `radix`      - Which base to parse the number as. Accepts 2-36.
 516   * - `negative`   - Whether to accept negative numbers.
 517   * - `fractional` - Whether to accept numbers with fractional parts.
 518   * - `special`    - Whether to accept special values like `inf`
 519   *                  and `NaN`. Can conflict with `radix`, see Failure.
 520   * - `exponent`   - Which exponent format to accept. Options are:
 521   *     - `ExpNone`: No Exponent, accepts just plain numbers like `42` or
 522   *                  `-8.2`.
 523   *     - `ExpDec`:  Accepts numbers with a decimal exponent like `42e5` or
 524   *                  `8.2E-2`. The exponent string itself is always base 10.
 525   *                  Can conflict with `radix`, see Failure.
 526   *     - `ExpBin`:  Accepts numbers with a binary exponent like `42P-8` or
 527   *                  `FFp128`. The exponent string itself is always base 10.
 528   *                  Can conflict with `radix`, see Failure.
 529   * - `empty_zero` - Whether to accept an empty `buf` as a 0 or not.
 530   * - `ignore_underscores` - Whether all underscores within the string should
 531   *                          be ignored.
 532   *
 533   * # Return value
 534   * Returns `Some(n)` if `buf` parses to a number n without overflowing, and
 535   * `None` otherwise, depending on the constraints set by the remaining
 536   * arguments.
 537   *
 538   * # Failure
 539   * - Fails if `radix` < 2 or `radix` > 36.
 540   * - Fails if `radix` > 14 and `exponent` is `ExpDec` due to conflict
 541   *   between digit and exponent sign `'e'`.
 542   * - Fails if `radix` > 25 and `exponent` is `ExpBin` due to conflict
 543   *   between digit and exponent sign `'p'`.
 544   * - Fails if `radix` > 18 and `special == true` due to conflict
 545   *   between digit and lowest first character in `inf` and `NaN`, the `'i'`.
 546   */
 547  pub fn from_str_bytes_common<T:NumCast+Zero+One+Eq+Ord+Div<T,T>+
 548                                      Mul<T,T>+Sub<T,T>+Neg<T>+Add<T,T>+
 549                                      NumStrConv+Clone>(
 550          buf: &[u8], radix: uint, negative: bool, fractional: bool,
 551          special: bool, exponent: ExponentFormat, empty_zero: bool,
 552          ignore_underscores: bool
 553          ) -> Option<T> {
 554      match exponent {
 555          ExpDec if radix >= DIGIT_E_RADIX       // decimal exponent 'e'
 556            => fail!("from_str_bytes_common: radix {} incompatible with \
 557                      use of 'e' as decimal exponent", radix),
 558          ExpBin if radix >= DIGIT_P_RADIX       // binary exponent 'p'
 559            => fail!("from_str_bytes_common: radix {} incompatible with \
 560                      use of 'p' as binary exponent", radix),
 561          _ if special && radix >= DIGIT_I_RADIX // first digit of 'inf'
 562            => fail!("from_str_bytes_common: radix {} incompatible with \
 563                      special values 'inf' and 'NaN'", radix),
 564          _ if (radix as int) < 2
 565            => fail!("from_str_bytes_common: radix {} to low, \
 566                      must lie in the range [2, 36]", radix),
 567          _ if (radix as int) > 36
 568            => fail!("from_str_bytes_common: radix {} to high, \
 569                      must lie in the range [2, 36]", radix),
 570          _ => ()
 571      }
 572  
 573      let _0: T = Zero::zero();
 574      let _1: T = One::one();
 575      let radix_gen: T = cast(radix as int).unwrap();
 576  
 577      let len = buf.len();
 578  
 579      if len == 0 {
 580          if empty_zero {
 581              return Some(_0);
 582          } else {
 583              return None;
 584          }
 585      }
 586  
 587      if special {
 588          if buf == INF_BUF || buf == POS_INF_BUF {
 589              return NumStrConv::inf();
 590          } else if buf == NEG_INF_BUF {
 591              if negative {
 592                  return NumStrConv::neg_inf();
 593              } else {
 594                  return None;
 595              }
 596          } else if buf == NAN_BUF {
 597              return NumStrConv::nan();
 598          }
 599      }
 600  
 601      let (start, accum_positive) = match buf[0] as char {
 602        '-' if !negative => return None,
 603        '-' => (1u, false),
 604        '+' => (1u, true),
 605         _  => (0u, true)
 606      };
 607  
 608      // Initialize accumulator with signed zero for floating point parsing to
 609      // work
 610      let mut accum      = if accum_positive { _0.clone() } else { -_1 * _0};
 611      let mut last_accum = accum.clone(); // Necessary to detect overflow
 612      let mut i          = start;
 613      let mut exp_found  = false;
 614  
 615      // Parse integer part of number
 616      while i < len {
 617          let c = buf[i] as char;
 618  
 619          match char::to_digit(c, radix) {
 620              Some(digit) => {
 621                  // shift accum one digit left
 622                  accum = accum * radix_gen.clone();
 623  
 624                  // add/subtract current digit depending on sign
 625                  if accum_positive {
 626                      accum = accum + cast(digit as int).unwrap();
 627                  } else {
 628                      accum = accum - cast(digit as int).unwrap();
 629                  }
 630  
 631                  // Detect overflow by comparing to last value, except
 632                  // if we've not seen any non-zero digits.
 633                  if last_accum != _0 {
 634                      if accum_positive && accum <= last_accum { return NumStrConv::inf(); }
 635                      if !accum_positive && accum >= last_accum { return NumStrConv::neg_inf(); }
 636  
 637                      // Detect overflow by reversing the shift-and-add proccess
 638                      if accum_positive &&
 639                          (last_accum != ((accum - cast(digit as int).unwrap())/radix_gen.clone())) {
 640                          return NumStrConv::inf();
 641                      }
 642                      if !accum_positive &&
 643                          (last_accum != ((accum + cast(digit as int).unwrap())/radix_gen.clone())) {
 644                          return NumStrConv::neg_inf();
 645                      }
 646                  }
 647                  last_accum = accum.clone();
 648              }
 649              None => match c {
 650                  '_' if ignore_underscores => {}
 651                  'e' | 'E' | 'p' | 'P' => {
 652                      exp_found = true;
 653                      break;                       // start of exponent
 654                  }
 655                  '.' if fractional => {
 656                      i += 1u;                     // skip the '.'
 657                      break;                       // start of fractional part
 658                  }
 659                  _ => return None                 // invalid number
 660              }
 661          }
 662  
 663          i += 1u;
 664      }
 665  
 666      // Parse fractional part of number
 667      // Skip if already reached start of exponent
 668      if !exp_found {
 669          let mut power = _1.clone();
 670  
 671          while i < len {
 672              let c = buf[i] as char;
 673  
 674              match char::to_digit(c, radix) {
 675                  Some(digit) => {
 676                      // Decrease power one order of magnitude
 677                      power = power / radix_gen;
 678  
 679                      let digit_t: T = cast(digit).unwrap();
 680  
 681                      // add/subtract current digit depending on sign
 682                      if accum_positive {
 683                          accum = accum + digit_t * power;
 684                      } else {
 685                          accum = accum - digit_t * power;
 686                      }
 687  
 688                      // Detect overflow by comparing to last value
 689                      if accum_positive && accum < last_accum { return NumStrConv::inf(); }
 690                      if !accum_positive && accum > last_accum { return NumStrConv::neg_inf(); }
 691                      last_accum = accum.clone();
 692                  }
 693                  None => match c {
 694                      '_' if ignore_underscores => {}
 695                      'e' | 'E' | 'p' | 'P' => {
 696                          exp_found = true;
 697                          break;                   // start of exponent
 698                      }
 699                      _ => return None             // invalid number
 700                  }
 701              }
 702  
 703              i += 1u;
 704          }
 705      }
 706  
 707      // Special case: buf not empty, but does not contain any digit in front
 708      // of the exponent sign -> number is empty string
 709      if i == start {
 710          if empty_zero {
 711              return Some(_0);
 712          } else {
 713              return None;
 714          }
 715      }
 716  
 717      let mut multiplier = _1.clone();
 718  
 719      if exp_found {
 720          let c = buf[i] as char;
 721          let base: T = match (c, exponent) {
 722              // c is never _ so don't need to handle specially
 723              ('e', ExpDec) | ('E', ExpDec) => cast(10u).unwrap(),
 724              ('p', ExpBin) | ('P', ExpBin) => cast(2u).unwrap(),
 725              _ => return None // char doesn't fit given exponent format
 726          };
 727  
 728          // parse remaining bytes as decimal integer,
 729          // skipping the exponent char
 730          let exp: Option<int> = from_str_bytes_common(
 731              buf.slice(i+1, len), 10, true, false, false, ExpNone, false,
 732              ignore_underscores);
 733  
 734          match exp {
 735              Some(exp_pow) => {
 736                  multiplier = if exp_pow < 0 {
 737                      _1 / num::pow(base, (-exp_pow.to_int().unwrap()) as uint)
 738                  } else {
 739                      num::pow(base, exp_pow.to_int().unwrap() as uint)
 740                  }
 741              }
 742              None => return None // invalid exponent -> invalid number
 743          }
 744      }
 745  
 746      Some(accum * multiplier)
 747  }
 748  
 749  /**
 750   * Parses a string as a number. This is a wrapper for
 751   * `from_str_bytes_common()`, for details see there.
 752   */
 753  #[inline]
 754  pub fn from_str_common<T:NumCast+Zero+One+Eq+Ord+Div<T,T>+Mul<T,T>+
 755                                Sub<T,T>+Neg<T>+Add<T,T>+NumStrConv+Clone>(
 756          buf: &str, radix: uint, negative: bool, fractional: bool,
 757          special: bool, exponent: ExponentFormat, empty_zero: bool,
 758          ignore_underscores: bool
 759          ) -> Option<T> {
 760      from_str_bytes_common(buf.as_bytes(), radix, negative,
 761                            fractional, special, exponent, empty_zero,
 762                            ignore_underscores)
 763  }

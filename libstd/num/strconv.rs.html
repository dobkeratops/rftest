 111  /**
 112   * Converts an integral number to its string representation as a byte vector.
 113   * This is meant to be a common base implementation for all integral string
 114   * conversion functions like `to_str()` or `to_str_radix()`.
 115   *
 116   * # Arguments
 117   * - `num`           - The number to convert. Accepts any number that
 118   *                     implements the numeric traits.
 119   * - `radix`         - Base to use. Accepts only the values 2-36.
 120   * - `sign`          - How to emit the sign. Options are:
 121   *     - `SignNone`: No sign at all. Basically emits `abs(num)`.
 122   *     - `SignNeg`:  Only `-` on negative values.
 123   *     - `SignAll`:  Both `+` on positive, and `-` on negative numbers.
 124   * - `f`             - a callback which will be invoked for each ascii character
 125   *                     which composes the string representation of this integer
 126   *
 127   * # Return value
 128   * A tuple containing the byte vector, and a boolean flag indicating
 129   * whether it represents a special value like `inf`, `-inf`, `NaN` or not.
 130   * It returns a tuple because there can be ambiguity between a special value
 131   * and a number representation at higher bases.
 132   *
 133   * # Failure
 134   * - Fails if `radix` < 2 or `radix` > 36.
 135   */
 136  pub fn int_to_str_bytes_common<T: Int>(num: T, radix: uint, sign: SignFormat, f: |u8|) {
 137      assert!(2 <= radix && radix <= 36);
 138  
 139      let _0: T = Zero::zero();
 140  
 141      let neg = num < _0;
 142      let radix_gen: T = cast(radix).unwrap();
 143  
 144      let mut deccum = num;
 145      // This is just for integral types, the largest of which is a u64. The
 146      // smallest base that we can have is 2, so the most number of digits we're
 147      // ever going to have is 64
 148      let mut buf = [0u8, ..64];
 149      let mut cur = 0;
 150  
 151      // Loop at least once to make sure at least a `0` gets emitted.
 152      loop {
 153          // Calculate the absolute value of each digit instead of only
 154          // doing it once for the whole number because a
 155          // representable negative number doesn't necessary have an
 156          // representable additive inverse of the same type
 157          // (See twos complement). But we assume that for the
 158          // numbers [-35 .. 0] we always have [0 .. 35].
 159          let current_digit_signed = deccum % radix_gen;
 160          let current_digit = if current_digit_signed < _0 {
 161              -current_digit_signed
 162          } else {
 163              current_digit_signed
 164          };
 165          buf[cur] = match current_digit.to_u8().unwrap() {
 166              i @ 0..9 => '0' as u8 + i,
 167              i        => 'a' as u8 + (i - 10),
 168          };
 169          cur += 1;
 170  
 171          deccum = deccum / radix_gen;
 172          // No more digits to calculate for the non-fractional part -> break
 173          if deccum == _0 { break; }
 174      }
 175  
 176      // Decide what sign to put in front
 177      match sign {
 178          SignNeg | SignAll if neg => { f('-' as u8); }
 179          SignAll => { f('+' as u8); }
 180          _ => ()
 181      }
 182  
 183      // We built the number in reverse order, so un-reverse it here
 184      while cur > 0 {
 185          cur -= 1;
 186          f(buf[cur]);
 187      }
 188  }

 190  /**
 191   * Converts a number to its string representation as a byte vector.
 192   * This is meant to be a common base implementation for all numeric string
 193   * conversion functions like `to_str()` or `to_str_radix()`.
 194   *
 195   * # Arguments
 196   * - `num`           - The number to convert. Accepts any number that
 197   *                     implements the numeric traits.
 198   * - `radix`         - Base to use. Accepts only the values 2-36. If the exponential notation
 199   *                     is used, then this base is only used for the significand. The exponent
 200   *                     itself always printed using a base of 10.
 201   * - `negative_zero` - Whether to treat the special value `-0` as
 202   *                     `-0` or as `+0`.
 203   * - `sign`          - How to emit the sign. Options are:
 204   *     - `SignNone`: No sign at all. The exponent sign is also omitted.
 205   *     - `SignNeg`:  Only `-` on negative values.
 206   *     - `SignAll`:  Both `+` on positive, and `-` on negative numbers.
 207   * - `digits`        - The amount of digits to use for emitting the
 208   *                     fractional part, if any. Options are:
 209   *     - `DigAll`:         All calculatable digits. Beware of bignums or
 210   *                         fractions!
 211   *     - `DigMax(uint)`:   Maximum N digits, truncating any trailing zeros.
 212   *     - `DigExact(uint)`: Exactly N digits.
 213   * - `exp_format`   - Whether or not to use the exponential (scientific) notation.
 214   *                    Options are:
 215   *     - `ExpNone`: Do not use the exponential notation.
 216   *     - `ExpDec`:  Use the exponential notation with the exponent having a base of 10,
 217   *                  and exponent sign being `'e'` or `'E'` depending on the value of
 218   *                  the `exp_upper` argument. E.g. the number 1000 would be printed as 1e3.
 219   *     - `ExpBin`:  Use the exponential notation with the exponent having a base of 2,
 220   *                  and exponent sign being `'p'` or `'P'` depending on the value of
 221   *                  the `exp_upper` argument. E.g. the number 8 would be printed as 1p3.
 222   * - `exp_capital`   - Whether or not to use a capital letter for the exponent sign, if
 223   *                     exponential notation is desired.
 224   *
 225   * # Return value
 226   * A tuple containing the byte vector, and a boolean flag indicating
 227   * whether it represents a special value like `inf`, `-inf`, `NaN` or not.
 228   * It returns a tuple because there can be ambiguity between a special value
 229   * and a number representation at higher bases.
 230   *
 231   * # Failure
 232   * - Fails if `radix` < 2 or `radix` > 36.
 233   * - Fails if `radix` > 14 and `exp_format` is `ExpDec` due to conflict
 234   *   between digit and exponent sign `'e'`.
 235   * - Fails if `radix` > 25 and `exp_format` is `ExpBin` due to conflict
 236   *   between digit and exponent sign `'p'`.
 237   */
 238  pub fn float_to_str_bytes_common<T:NumCast+Zero+One+Eq+Ord+Float+Round+
 239                                    Div<T,T>+Neg<T>+Rem<T,T>+Mul<T,T>>(
 240          num: T, radix: uint, negative_zero: bool,
 241          sign: SignFormat, digits: SignificantDigits, exp_format: ExponentFormat, exp_upper: bool
 242          ) -> (~[u8], bool) {
 243      assert!(2 <= radix && radix <= 36);
 244      match exp_format {
 245          ExpDec if radix >= DIGIT_E_RADIX       // decimal exponent 'e'
 246            => fail!("float_to_str_bytes_common: radix {} incompatible with \
 247                      use of 'e' as decimal exponent", radix),
 248          ExpBin if radix >= DIGIT_P_RADIX       // binary exponent 'p'
 249            => fail!("float_to_str_bytes_common: radix {} incompatible with \
 250                      use of 'p' as binary exponent", radix),
 251          _ => ()
 252      }
 253  
 254      let _0: T = Zero::zero();
 255      let _1: T = One::one();
 256  
 257      match num.classify() {
 258          FPNaN => { return ("NaN".as_bytes().to_owned(), true); }
 259          FPInfinite if num > _0 => {
 260              return match sign {
 261                  SignAll => ("+inf".as_bytes().to_owned(), true),
 262                  _       => ("inf".as_bytes().to_owned(), true)
 263              };
 264          }
 265          FPInfinite if num < _0 => {
 266              return match sign {
 267                  SignNone => ("inf".as_bytes().to_owned(), true),
 268                  _        => ("-inf".as_bytes().to_owned(), true),
 269              };
 270          }
 271          _ => {}
 272      }
 273  
 274      let neg = num < _0 || (negative_zero && _1 / num == Float::neg_infinity());
 275      let mut buf: ~[u8] = ~[];
 276      let radix_gen: T   = cast(radix as int).unwrap();
 277  
 278      let (num, exp) = match exp_format {
 279          ExpNone => (num, 0i32),
 280          ExpDec | ExpBin => {
 281              if num == _0 {
 282                  (num, 0i32)
 283              } else {
 284                  let (exp, exp_base) = match exp_format {
 285                      ExpDec => (num.abs().log10().floor(), cast::<f64, T>(10.0f64).unwrap()),
 286                      ExpBin => (num.abs().log2().floor(), cast::<f64, T>(2.0f64).unwrap()),
 287                      ExpNone => unreachable!()
 288                  };
 289  
 290                  (num / exp_base.powf(&exp), cast::<T, i32>(exp).unwrap())
 291              }
 292          }
 293      };
 294  
 295      // First emit the non-fractional part, looping at least once to make
 296      // sure at least a `0` gets emitted.
 297      let mut deccum = num.trunc();
 298      loop {
 299          // Calculate the absolute value of each digit instead of only
 300          // doing it once for the whole number because a
 301          // representable negative number doesn't necessary have an
 302          // representable additive inverse of the same type
 303          // (See twos complement). But we assume that for the
 304          // numbers [-35 .. 0] we always have [0 .. 35].
 305          let current_digit = (deccum % radix_gen).abs();
 306  
 307          // Decrease the deccumulator one digit at a time
 308          deccum = deccum / radix_gen;
 309          deccum = deccum.trunc();
 310  
 311          buf.push(char::from_digit(current_digit.to_int().unwrap() as uint, radix)
 312               .unwrap() as u8);
 313  
 314          // No more digits to calculate for the non-fractional part -> break
 315          if deccum == _0 { break; }
 316      }
 317  
 318      // If limited digits, calculate one digit more for rounding.
 319      let (limit_digits, digit_count, exact) = match digits {
 320          DigAll          => (false, 0u,      false),
 321          DigMax(count)   => (true,  count+1, false),
 322          DigExact(count) => (true,  count+1, true)
 323      };
 324  
 325      // Decide what sign to put in front
 326      match sign {
 327          SignNeg | SignAll if neg => {
 328              buf.push('-' as u8);
 329          }
 330          SignAll => {
 331              buf.push('+' as u8);
 332          }
 333          _ => ()
 334      }
 335  
 336      buf.reverse();
 337  
 338      // Remember start of the fractional digits.
 339      // Points one beyond end of buf if none get generated,
 340      // or at the '.' otherwise.
 341      let start_fractional_digits = buf.len();
 342  
 343      // Now emit the fractional part, if any
 344      deccum = num.fract();
 345      if deccum != _0 || (limit_digits && exact && digit_count > 0) {
 346          buf.push('.' as u8);
 347          let mut dig = 0u;
 348  
 349          // calculate new digits while
 350          // - there is no limit and there are digits left
 351          // - or there is a limit, it's not reached yet and
 352          //   - it's exact
 353          //   - or it's a maximum, and there are still digits left
 354          while (!limit_digits && deccum != _0)
 355             || (limit_digits && dig < digit_count && (
 356                     exact
 357                  || (!exact && deccum != _0)
 358                )
 359          ) {
 360              // Shift first fractional digit into the integer part
 361              deccum = deccum * radix_gen;
 362  
 363              // Calculate the absolute value of each digit.
 364              // See note in first loop.
 365              let current_digit = deccum.trunc().abs();
 366  
 367              buf.push(char::from_digit(
 368                  current_digit.to_int().unwrap() as uint, radix).unwrap() as u8);
 369  
 370              // Decrease the deccumulator one fractional digit at a time
 371              deccum = deccum.fract();
 372              dig += 1u;
 373          }
 374  
 375          // If digits are limited, and that limit has been reached,
 376          // cut off the one extra digit, and depending on its value
 377          // round the remaining ones.
 378          if limit_digits && dig == digit_count {
 379              let ascii2value = |chr: u8| {
 380                  char::to_digit(chr as char, radix).unwrap()
 381              };
 382              let value2ascii = |val: uint| {
 383                  char::from_digit(val, radix).unwrap() as u8
 384              };
 385  
 386              let extra_digit = ascii2value(buf.pop().unwrap());
 387              if extra_digit >= radix / 2 { // -> need to round
 388                  let mut i: int = buf.len() as int - 1;
 389                  loop {
 390                      // If reached left end of number, have to
 391                      // insert additional digit:
 392                      if i < 0
 393                      || buf[i] == '-' as u8
 394                      || buf[i] == '+' as u8 {
 395                          buf.insert((i + 1) as uint, value2ascii(1));
 396                          break;
 397                      }
 398  
 399                      // Skip the '.'
 400                      if buf[i] == '.' as u8 { i -= 1; continue; }
 401  
 402                      // Either increment the digit,
 403                      // or set to 0 if max and carry the 1.
 404                      let current_digit = ascii2value(buf[i]);
 405                      if current_digit < (radix - 1) {
 406                          buf[i] = value2ascii(current_digit+1);
 407                          break;
 408                      } else {
 409                          buf[i] = value2ascii(0);
 410                          i -= 1;
 411                      }
 412                  }
 413              }
 414          }
 415      }
 416  
 417      // if number of digits is not exact, remove all trailing '0's up to
 418      // and including the '.'
 419      if !exact {
 420          let buf_max_i = buf.len() - 1;
 421  
 422          // index to truncate from
 423          let mut i = buf_max_i;
 424  
 425          // discover trailing zeros of fractional part
 426          while i > start_fractional_digits && buf[i] == '0' as u8 {
 427              i -= 1;
 428          }
 429  
 430          // Only attempt to truncate digits if buf has fractional digits
 431          if i >= start_fractional_digits {
 432              // If buf ends with '.', cut that too.
 433              if buf[i] == '.' as u8 { i -= 1 }
 434  
 435              // only resize buf if we actually remove digits
 436              if i < buf_max_i {
 437                  buf = buf.slice(0, i + 1).to_owned();
 438              }
 439          }
 440      } // If exact and trailing '.', just cut that
 441      else {
 442          let max_i = buf.len() - 1;
 443          if buf[max_i] == '.' as u8 {
 444              buf = buf.slice(0, max_i).to_owned();
 445          }
 446      }
 447  
 448      match exp_format {
 449          ExpNone => (),
 450          _ => {
 451              buf.push(match exp_format {
 452                  ExpDec if exp_upper => 'E',
 453                  ExpDec if !exp_upper => 'e',
 454                  ExpBin if exp_upper => 'P',
 455                  ExpBin if !exp_upper => 'p',
 456                  _ => unreachable!()
 457              } as u8);
 458  
 459              int_to_str_bytes_common(exp, 10, sign, |c| buf.push(c));
 460          }
 461      }
 462  
 463      (buf, false)
 464  }
 465  
 466  /**
 467   * Converts a number to its string representation. This is a wrapper for
 468   * `to_str_bytes_common()`, for details see there.
 469   */
 470  #[inline]
 471  pub fn float_to_str_common<T:NumCast+Zero+One+Eq+Ord+NumStrConv+Float+Round+
 472                               Div<T,T>+Neg<T>+Rem<T,T>+Mul<T,T>>(
 473          num: T, radix: uint, negative_zero: bool,
 474          sign: SignFormat, digits: SignificantDigits, exp_format: ExponentFormat, exp_capital: bool
 475          ) -> (~str, bool) {
 476      let (bytes, special) = float_to_str_bytes_common(num, radix,
 477                                 negative_zero, sign, digits, exp_format, exp_capital);
 478      (str::from_utf8_owned(bytes).unwrap(), special)
 479  }
 480  
 481  // Some constants for from_str_bytes_common's input validation,
 482  // they define minimum radix values for which the character is a valid digit.
 483  static DIGIT_P_RADIX: uint = ('p' as uint) - ('a' as uint) + 11u;
 484  static DIGIT_I_RADIX: uint = ('i' as uint) - ('a' as uint) + 11u;
 485  static DIGIT_E_RADIX: uint = ('e' as uint) - ('a' as uint) + 11u;
 486  
 487  /**
 488   * Parses a byte slice as a number. This is meant to
 489   * be a common base implementation for all numeric string conversion
 490   * functions like `from_str()` or `from_str_radix()`.
 491   *
 492   * # Arguments
 493   * - `buf`        - The byte slice to parse.
 494   * - `radix`      - Which base to parse the number as. Accepts 2-36.
 495   * - `negative`   - Whether to accept negative numbers.
 496   * - `fractional` - Whether to accept numbers with fractional parts.
 497   * - `special`    - Whether to accept special values like `inf`
 498   *                  and `NaN`. Can conflict with `radix`, see Failure.
 499   * - `exponent`   - Which exponent format to accept. Options are:
 500   *     - `ExpNone`: No Exponent, accepts just plain numbers like `42` or
 501   *                  `-8.2`.
 502   *     - `ExpDec`:  Accepts numbers with a decimal exponent like `42e5` or
 503   *                  `8.2E-2`. The exponent string itself is always base 10.
 504   *                  Can conflict with `radix`, see Failure.
 505   *     - `ExpBin`:  Accepts numbers with a binary exponent like `42P-8` or
 506   *                  `FFp128`. The exponent string itself is always base 10.
 507   *                  Can conflict with `radix`, see Failure.
 508   * - `empty_zero` - Whether to accept an empty `buf` as a 0 or not.
 509   * - `ignore_underscores` - Whether all underscores within the string should
 510   *                          be ignored.
 511   *
 512   * # Return value
 513   * Returns `Some(n)` if `buf` parses to a number n without overflowing, and
 514   * `None` otherwise, depending on the constraints set by the remaining
 515   * arguments.
 516   *
 517   * # Failure
 518   * - Fails if `radix` < 2 or `radix` > 36.
 519   * - Fails if `radix` > 14 and `exponent` is `ExpDec` due to conflict
 520   *   between digit and exponent sign `'e'`.
 521   * - Fails if `radix` > 25 and `exponent` is `ExpBin` due to conflict
 522   *   between digit and exponent sign `'p'`.
 523   * - Fails if `radix` > 18 and `special == true` due to conflict
 524   *   between digit and lowest first character in `inf` and `NaN`, the `'i'`.
 525   */
 526  pub fn from_str_bytes_common<T:NumCast+Zero+One+Eq+Ord+Div<T,T>+
 527                                      Mul<T,T>+Sub<T,T>+Neg<T>+Add<T,T>+
 528                                      NumStrConv+Clone>(
 529          buf: &[u8], radix: uint, negative: bool, fractional: bool,
 530          special: bool, exponent: ExponentFormat, empty_zero: bool,
 531          ignore_underscores: bool
 532          ) -> Option<T> {
 533      match exponent {
 534          ExpDec if radix >= DIGIT_E_RADIX       // decimal exponent 'e'
 535            => fail!("from_str_bytes_common: radix {} incompatible with \
 536                      use of 'e' as decimal exponent", radix),
 537          ExpBin if radix >= DIGIT_P_RADIX       // binary exponent 'p'
 538            => fail!("from_str_bytes_common: radix {} incompatible with \
 539                      use of 'p' as binary exponent", radix),
 540          _ if special && radix >= DIGIT_I_RADIX // first digit of 'inf'
 541            => fail!("from_str_bytes_common: radix {} incompatible with \
 542                      special values 'inf' and 'NaN'", radix),
 543          _ if (radix as int) < 2
 544            => fail!("from_str_bytes_common: radix {} to low, \
 545                      must lie in the range [2, 36]", radix),
 546          _ if (radix as int) > 36
 547            => fail!("from_str_bytes_common: radix {} to high, \
 548                      must lie in the range [2, 36]", radix),
 549          _ => ()
 550      }
 551  
 552      let _0: T = Zero::zero();
 553      let _1: T = One::one();
 554      let radix_gen: T = cast(radix as int).unwrap();
 555  
 556      let len = buf.len();
 557  
 558      if len == 0 {
 559          if empty_zero {
 560              return Some(_0);
 561          } else {
 562              return None;
 563          }
 564      }
 565  
 566      if special {
 567          if buf == INF_BUF || buf == POS_INF_BUF {
 568              return NumStrConv::inf();
 569          } else if buf == NEG_INF_BUF {
 570              if negative {
 571                  return NumStrConv::neg_inf();
 572              } else {
 573                  return None;
 574              }
 575          } else if buf == NAN_BUF {
 576              return NumStrConv::nan();
 577          }
 578      }
 579  
 580      let (start, accum_positive) = match buf[0] as char {
 581        '-' if !negative => return None,
 582        '-' => (1u, false),
 583        '+' => (1u, true),
 584         _  => (0u, true)
 585      };
 586  
 587      // Initialize accumulator with signed zero for floating point parsing to
 588      // work
 589      let mut accum      = if accum_positive { _0.clone() } else { -_1 * _0};
 590      let mut last_accum = accum.clone(); // Necessary to detect overflow
 591      let mut i          = start;
 592      let mut exp_found  = false;
 593  
 594      // Parse integer part of number
 595      while i < len {
 596          let c = buf[i] as char;
 597  
 598          match char::to_digit(c, radix) {
 599              Some(digit) => {
 600                  // shift accum one digit left
 601                  accum = accum * radix_gen.clone();
 602  
 603                  // add/subtract current digit depending on sign
 604                  if accum_positive {
 605                      accum = accum + cast(digit as int).unwrap();
 606                  } else {
 607                      accum = accum - cast(digit as int).unwrap();
 608                  }
 609  
 610                  // Detect overflow by comparing to last value, except
 611                  // if we've not seen any non-zero digits.
 612                  if last_accum != _0 {
 613                      if accum_positive && accum <= last_accum { return NumStrConv::inf(); }
 614                      if !accum_positive && accum >= last_accum { return NumStrConv::neg_inf(); }
 615  
 616                      // Detect overflow by reversing the shift-and-add proccess
 617                      if accum_positive &&
 618                          (last_accum != ((accum - cast(digit as int).unwrap())/radix_gen.clone())) {
 619                          return NumStrConv::inf();
 620                      }
 621                      if !accum_positive &&
 622                          (last_accum != ((accum + cast(digit as int).unwrap())/radix_gen.clone())) {
 623                          return NumStrConv::neg_inf();
 624                      }
 625                  }
 626                  last_accum = accum.clone();
 627              }
 628              None => match c {
 629                  '_' if ignore_underscores => {}
 630                  'e' | 'E' | 'p' | 'P' => {
 631                      exp_found = true;
 632                      break;                       // start of exponent
 633                  }
 634                  '.' if fractional => {
 635                      i += 1u;                     // skip the '.'
 636                      break;                       // start of fractional part
 637                  }
 638                  _ => return None                 // invalid number
 639              }
 640          }
 641  
 642          i += 1u;
 643      }
 644  
 645      // Parse fractional part of number
 646      // Skip if already reached start of exponent
 647      if !exp_found {
 648          let mut power = _1.clone();
 649  
 650          while i < len {
 651              let c = buf[i] as char;
 652  
 653              match char::to_digit(c, radix) {
 654                  Some(digit) => {
 655                      // Decrease power one order of magnitude
 656                      power = power / radix_gen;
 657  
 658                      let digit_t: T = cast(digit).unwrap();
 659  
 660                      // add/subtract current digit depending on sign
 661                      if accum_positive {
 662                          accum = accum + digit_t * power;
 663                      } else {
 664                          accum = accum - digit_t * power;
 665                      }
 666  
 667                      // Detect overflow by comparing to last value
 668                      if accum_positive && accum < last_accum { return NumStrConv::inf(); }
 669                      if !accum_positive && accum > last_accum { return NumStrConv::neg_inf(); }
 670                      last_accum = accum.clone();
 671                  }
 672                  None => match c {
 673                      '_' if ignore_underscores => {}
 674                      'e' | 'E' | 'p' | 'P' => {
 675                          exp_found = true;
 676                          break;                   // start of exponent
 677                      }
 678                      _ => return None             // invalid number
 679                  }
 680              }
 681  
 682              i += 1u;
 683          }
 684      }
 685  
 686      // Special case: buf not empty, but does not contain any digit in front
 687      // of the exponent sign -> number is empty string
 688      if i == start {
 689          if empty_zero {
 690              return Some(_0);
 691          } else {
 692              return None;
 693          }
 694      }
 695  
 696      let mut multiplier = _1.clone();
 697  
 698      if exp_found {
 699          let c = buf[i] as char;
 700          let base: T = match (c, exponent) {
 701              // c is never _ so don't need to handle specially
 702              ('e', ExpDec) | ('E', ExpDec) => cast(10u).unwrap(),
 703              ('p', ExpBin) | ('P', ExpBin) => cast(2u).unwrap(),
 704              _ => return None // char doesn't fit given exponent format
 705          };
 706  
 707          // parse remaining bytes as decimal integer,
 708          // skipping the exponent char
 709          let exp: Option<int> = from_str_bytes_common(
 710              buf.slice(i+1, len), 10, true, false, false, ExpNone, false,
 711              ignore_underscores);
 712  
 713          match exp {
 714              Some(exp_pow) => {
 715                  multiplier = if exp_pow < 0 {
 716                      _1 / num::pow(base, (-exp_pow.to_int().unwrap()) as uint)
 717                  } else {
 718                      num::pow(base, exp_pow.to_int().unwrap() as uint)
 719                  }
 720              }
 721              None => return None // invalid exponent -> invalid number
 722          }
 723      }
 724  
 725      Some(accum * multiplier)
 726  }
 727  
 728  /**
 729   * Parses a string as a number. This is a wrapper for
 730   * `from_str_bytes_common()`, for details see there.
 731   */
 732  #[inline]
 733  pub fn from_str_common<T:NumCast+Zero+One+Eq+Ord+Div<T,T>+Mul<T,T>+
 734                                Sub<T,T>+Neg<T>+Add<T,T>+NumStrConv+Clone>(
 735          buf: &str, radix: uint, negative: bool, fractional: bool,
 736          special: bool, exponent: ExponentFormat, empty_zero: bool,
 737          ignore_underscores: bool
 738          ) -> Option<T> {
 739      from_str_bytes_common(buf.as_bytes(), radix, negative,
 740                            fractional, special, exponent, empty_zero,
 741                            ignore_underscores)
 742  }

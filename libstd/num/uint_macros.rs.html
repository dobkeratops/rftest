 159      /// Calculates the Greatest Common Divisor (GCD) of the number and `other`
 160      #[inline]
 161      fn gcd(&self, other: &$T) -> $T {
 162          // Use Euclid's algorithm
 163          let mut m = *self;
 164          let mut n = *other;
 165          while m != 0 {
 166              let temp = m;
 167              m = n % temp;
 168              n = temp;
 169          }
 170          n
 171      }

 174      #[inline]
 175      fn lcm(&self, other: &$T) -> $T {
 176          (*self * *other) / self.gcd(other)
 177      }

 267  /// Convert to a string as a byte slice in a given base.
 268  #[inline]
 269  pub fn to_str_bytes<U>(n: $T, radix: uint, f: &fn(v: &[u8]) -> U) -> U {
 270      // The radix can be as low as 2, so we need at least 64 characters for a
 271      // base 2 number.
 272      let mut buf = [0u8, ..64];
 273      let mut cur = 0;
 274      do strconv::int_to_str_bytes_common(n, radix, strconv::SignNone) |i| {
 275          buf[cur] = i;
 276          cur += 1;
 277      }
 278      f(buf.slice(0, cur))
 279  }

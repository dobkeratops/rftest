 211  impl Integer for $T {
 212      ///
 213      /// Floored integer division
 214      ///
 215      /// # Examples
 216      ///
 217      /// ~~~
 218      /// assert!(( 8).div_floor( 3) ==  2);
 219      /// assert!(( 8).div_floor(-3) == -3);
 220      /// assert!((-8).div_floor( 3) == -3);
 221      /// assert!((-8).div_floor(-3) ==  2);
 222      ///
 223      /// assert!(( 1).div_floor( 2) ==  0);
 224      /// assert!(( 1).div_floor(-2) == -1);
 225      /// assert!((-1).div_floor( 2) == -1);
 226      /// assert!((-1).div_floor(-2) ==  0);
 227      /// ~~~
 228      ///
 229      #[inline]
 230      fn div_floor(&self, other: &$T) -> $T {
 231          // Algorithm from [Daan Leijen. _Division and Modulus for Computer Scientists_,
 232          // December 2001](http://research.microsoft.com/pubs/151917/divmodnote-letter.pdf)
 233          match self.div_rem(other) {
 234              (d, r) if (r > 0 && *other < 0)
 235                     || (r < 0 && *other > 0) => d - 1,
 236              (d, _)                          => d,
 237          }
 238      }
 239  
 240      ///
 241      /// Integer modulo, satisfying:
 242      ///
 243      /// ~~~
 244      /// assert!(n.div_floor(d) * d + n.mod_floor(d) == n)
 245      /// ~~~
 246      ///
 247      /// # Examples
 248      ///
 249      /// ~~~
 250      /// assert!(( 8).mod_floor( 3) ==  2);
 251      /// assert!(( 8).mod_floor(-3) == -1);
 252      /// assert!((-8).mod_floor( 3) ==  1);
 253      /// assert!((-8).mod_floor(-3) == -2);
 254      ///
 255      /// assert!(( 1).mod_floor( 2) ==  1);
 256      /// assert!(( 1).mod_floor(-2) == -1);
 257      /// assert!((-1).mod_floor( 2) ==  1);
 258      /// assert!((-1).mod_floor(-2) == -1);
 259      /// ~~~
 260      ///
 261      #[inline]
 262      fn mod_floor(&self, other: &$T) -> $T {
 263          // Algorithm from [Daan Leijen. _Division and Modulus for Computer Scientists_,
 264          // December 2001](http://research.microsoft.com/pubs/151917/divmodnote-letter.pdf)
 265          match *self % *other {
 266              r if (r > 0 && *other < 0)
 267                || (r < 0 && *other > 0) => r + *other,
 268              r                          => r,
 269          }
 270      }
 271  
 272      /// Calculates `div_floor` and `mod_floor` simultaneously
 273      #[inline]
 274      fn div_mod_floor(&self, other: &$T) -> ($T,$T) {
 275          // Algorithm from [Daan Leijen. _Division and Modulus for Computer Scientists_,
 276          // December 2001](http://research.microsoft.com/pubs/151917/divmodnote-letter.pdf)
 277          match self.div_rem(other) {
 278              (d, r) if (r > 0 && *other < 0)
 279                     || (r < 0 && *other > 0) => (d - 1, r + *other),
 280              (d, r)                          => (d, r),
 281          }
 282      }
 283  
 284      /// Calculates `div` (`\`) and `rem` (`%`) simultaneously
 285      #[inline]
 286      fn div_rem(&self, other: &$T) -> ($T,$T) {
 287          (*self / *other, *self % *other)
 288      }

 290      ///
 291      /// Calculates the Greatest Common Divisor (GCD) of the number and `other`
 292      ///
 293      /// The result is always positive
 294      ///
 295      #[inline]
 296      fn gcd(&self, other: &$T) -> $T {
 297          // Use Euclid's algorithm
 298          let mut m = *self;
 299          let mut n = *other;
 300          while m != 0 {
 301              let temp = m;
 302              m = n % temp;
 303              n = temp;
 304          }
 305          n.abs()
 306      }
 307  
 308      ///
 309      /// Calculates the Lowest Common Multiple (LCM) of the number and `other`
 310      ///
 311      #[inline]
 312      fn lcm(&self, other: &$T) -> $T {
 313          ((*self * *other) / self.gcd(other)).abs() // should not have to recaluculate abs
 314      }
 315

 318      fn is_multiple_of(&self, other: &$T) -> bool { *self % *other == 0 }
 319

 412  /// Convert to a string as a byte slice in a given base.
 413  #[inline]
 414  pub fn to_str_bytes<U>(n: $T, radix: uint, f: &fn(v: &[u8]) -> U) -> U {
 415      // The radix can be as low as 2, so we need at least 64 characters for a
 416      // base 2 number, and then we need another for a possible '-' character.
 417      let mut buf = [0u8, ..65];
 418      let mut cur = 0;
 419      do strconv::int_to_str_bytes_common(n, radix, strconv::SignNeg) |i| {
 420          buf[cur] = i;
 421          cur += 1;
 422      }
 423      f(buf.slice(0, cur))
 424  }

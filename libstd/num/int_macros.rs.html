  35  ///
  36  /// Iterate through a range with a given step value.
  37  ///
  38  /// Let `term` denote the closed interval `[stop-step,stop]` if `r` is Closed;
  39  /// otherwise `term` denotes the half-open interval `[stop-step,stop)`.
  40  /// Iterates through the range `[x_0, x_1, ..., x_n]` where
  41  /// `x_j == start + step*j`, and `x_n` lies in the interval `term`.
  42  ///
  43  /// If no such nonnegative integer `n` exists, then the iteration range
  44  /// is empty.
  45  ///
  46  fn range_step_core(start: $T, stop: $T, step: $T, r: Range, it: &fn($T) -> bool) -> bool {
  47      let mut i = start;
  48      if step == 0 {
  49          fail!(~"range_step called with step == 0");
  50      } else if step == (1 as $T) { // elide bounds check to tighten loop
  51          while i < stop {
  52              if !it(i) { return false; }
  53              // no need for overflow check;
  54              // cannot have i + 1 > max_value because i < stop <= max_value
  55              i += (1 as $T);
  56          }
  57      } else if step == (-1 as $T) { // elide bounds check to tighten loop
  58          while i > stop {
  59              if !it(i) { return false; }
  60              // no need for underflow check;
  61              // cannot have i - 1 < min_value because i > stop >= min_value
  62              i -= (1 as $T);
  63          }
  64      } else if step > 0 { // ascending
  65          while i < stop {
  66              if !it(i) { return false; }
  67              // avoiding overflow. break if i + step > max_value
  68              if i > max_value - step { return true; }
  69              i += step;
  70          }
  71      } else { // descending
  72          while i > stop {
  73              if !it(i) { return false; }
  74              // avoiding underflow. break if i + step < min_value
  75              if i < min_value - step { return true; }
  76              i += step;
  77          }
  78      }
  79      match r {
  80          HalfOpen => return true,
  81          Closed => return (i != stop || it(i))
  82      }
  83  }

  85  #[inline]
  86  ///
  87  /// Iterate through the range [`start`..`stop`) with a given step value.
  88  ///
  89  /// Iterates through the range `[x_0, x_1, ..., x_n]` where
  90  /// * `x_i == start + step*i`, and
  91  /// * `n` is the greatest nonnegative integer such that `x_n < stop`
  92  ///
  93  /// (If no such `n` exists, then the iteration range is empty.)
  94  ///
  95  /// # Arguments
  96  ///
  97  /// * `start` - lower bound, inclusive
  98  /// * `stop` - higher bound, exclusive
  99  ///
 100  /// # Examples
 101  /// ~~~
 102  /// let mut sum = 0;
 103  /// for int::range(1, 5) |i| {
 104  ///     sum += i;
 105  /// }
 106  /// assert!(sum == 10);
 107  /// ~~~
 108  ///
 109  pub fn range_step(start: $T, stop: $T, step: $T, it: &fn($T) -> bool) -> bool {
 110      range_step_core(start, stop, step, HalfOpen, it)
 111  }

 290  impl Integer for $T {
 291      ///
 292      /// Floored integer division
 293      ///
 294      /// # Examples
 295      ///
 296      /// ~~~
 297      /// assert!(( 8).div_floor( 3) ==  2);
 298      /// assert!(( 8).div_floor(-3) == -3);
 299      /// assert!((-8).div_floor( 3) == -3);
 300      /// assert!((-8).div_floor(-3) ==  2);
 301      ///
 302      /// assert!(( 1).div_floor( 2) ==  0);
 303      /// assert!(( 1).div_floor(-2) == -1);
 304      /// assert!((-1).div_floor( 2) == -1);
 305      /// assert!((-1).div_floor(-2) ==  0);
 306      /// ~~~
 307      ///
 308      #[inline]
 309      fn div_floor(&self, other: &$T) -> $T {
 310          // Algorithm from [Daan Leijen. _Division and Modulus for Computer Scientists_,
 311          // December 2001](http://research.microsoft.com/pubs/151917/divmodnote-letter.pdf)
 312          match self.div_rem(other) {
 313              (d, r) if (r > 0 && *other < 0)
 314                     || (r < 0 && *other > 0) => d - 1,
 315              (d, _)                          => d,
 316          }
 317      }
 318  
 319      ///
 320      /// Integer modulo, satisfying:
 321      ///
 322      /// ~~~
 323      /// assert!(n.div_floor(d) * d + n.mod_floor(d) == n)
 324      /// ~~~
 325      ///
 326      /// # Examples
 327      ///
 328      /// ~~~
 329      /// assert!(( 8).mod_floor( 3) ==  2);
 330      /// assert!(( 8).mod_floor(-3) == -1);
 331      /// assert!((-8).mod_floor( 3) ==  1);
 332      /// assert!((-8).mod_floor(-3) == -2);
 333      ///
 334      /// assert!(( 1).mod_floor( 2) ==  1);
 335      /// assert!(( 1).mod_floor(-2) == -1);
 336      /// assert!((-1).mod_floor( 2) ==  1);
 337      /// assert!((-1).mod_floor(-2) == -1);
 338      /// ~~~
 339      ///
 340      #[inline]
 341      fn mod_floor(&self, other: &$T) -> $T {
 342          // Algorithm from [Daan Leijen. _Division and Modulus for Computer Scientists_,
 343          // December 2001](http://research.microsoft.com/pubs/151917/divmodnote-letter.pdf)
 344          match *self % *other {
 345              r if (r > 0 && *other < 0)
 346                || (r < 0 && *other > 0) => r + *other,
 347              r                          => r,
 348          }
 349      }
 350  
 351      /// Calculates `div_floor` and `mod_floor` simultaneously
 352      #[inline]
 353      fn div_mod_floor(&self, other: &$T) -> ($T,$T) {
 354          // Algorithm from [Daan Leijen. _Division and Modulus for Computer Scientists_,
 355          // December 2001](http://research.microsoft.com/pubs/151917/divmodnote-letter.pdf)
 356          match self.div_rem(other) {
 357              (d, r) if (r > 0 && *other < 0)
 358                     || (r < 0 && *other > 0) => (d - 1, r + *other),
 359              (d, r)                          => (d, r),
 360          }
 361      }

 363      /// Calculates `div` (`\`) and `rem` (`%`) simultaneously
 364      #[inline]
 365      fn div_rem(&self, other: &$T) -> ($T,$T) {
 366          (*self / *other, *self % *other)
 367      }

 369      ///
 370      /// Calculates the Greatest Common Divisor (GCD) of the number and `other`
 371      ///
 372      /// The result is always positive
 373      ///
 374      #[inline]
 375      fn gcd(&self, other: &$T) -> $T {
 376          // Use Euclid's algorithm
 377          let mut m = *self;
 378          let mut n = *other;
 379          while m != 0 {
 380              let temp = m;
 381              m = n % temp;
 382              n = temp;
 383          }
 384          n.abs()
 385      }
 386  
 387      ///
 388      /// Calculates the Lowest Common Multiple (LCM) of the number and `other`
 389      ///
 390      #[inline]
 391      fn lcm(&self, other: &$T) -> $T {
 392          ((*self * *other) / self.gcd(other)).abs() // should not have to recaluculate abs
 393      }

 503  /// Convert to a string as a byte slice in a given base.
 504  #[inline]
 505  pub fn to_str_bytes<U>(n: $T, radix: uint, f: &fn(v: &[u8]) -> U) -> U {
 506      // The radix can be as low as 2, so we need at least 64 characters for a
 507      // base 2 number, and then we need another for a possible '-' character.
 508      let mut buf = [0u8, ..65];
 509      let mut cur = 0;
 510      do strconv::int_to_str_bytes_common(n, radix, strconv::SignNeg) |i| {
 511          buf[cur] = i;
 512          cur += 1;
 513      }
 514      f(buf.slice(0, cur))
 515  }

 521  }

 523  /// Convert to a string in a given base.
 524  #[inline]
 525  pub fn to_str_radix(num: $T, radix: uint) -> ~str {
 526      let mut buf: ~[u8] = ~[];
 527      do strconv::int_to_str_bytes_common(num, radix, strconv::SignNeg) |i| {
 528          buf.push(i);
 529      }
 530      // We know we generated valid utf-8, so we don't need to go through that
 531      // check.
 532      unsafe { str::raw::from_bytes_owned(buf) }
 533  }

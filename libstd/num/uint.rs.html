  79      /// A convenience form for basic repetition. Given a uint `x`,
  80      /// `do x.times { ... }` executes the given block x times.
  81      ///
  82      /// Equivalent to `for uint::range(0, x) |_| { ... }`.
  83      ///
  84      /// Not defined on all integer types to permit unambiguous
  85      /// use with integer literals of inferred integer-type as
  86      /// the self-value (eg. `do 100.times { ... }`).
  87      ///
  88      fn times(&self, it: &fn()) {

  90          while i > 0 {
  91              it();

  93          }
  94      }

  97  /// Returns the smallest power of 2 greater than or equal to `n`
  98  #[inline]
  99  pub fn next_power_of_two(n: uint) -> uint {
 100      let halfbits: uint = sys::size_of::<uint>() * 4u;
 101      let mut tmp: uint = n - 1u;
 102      let mut shift: uint = 1u;
 103      while shift <= halfbits { tmp |= tmp >> shift; shift <<= 1u; }
 104      tmp + 1u
 105  }

  94  /// Returns the smallest power of 2 greater than or equal to `n`
  95  #[inline]
  96  pub fn next_power_of_two(n: uint) -> uint {
  97      let halfbits: uint = sys::size_of::<uint>() * 4u;
  98      let mut tmp: uint = n - 1u;
  99      let mut shift: uint = 1u;
 100      while shift <= halfbits { tmp |= tmp >> shift; shift <<= 1u; }
 101      return tmp + 1u;
 102  }

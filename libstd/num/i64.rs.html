 30  impl Bitwise for i64 {
 31      /// Returns the number of ones in the binary representation of the number.
 32      #[inline]
 33      fn count_ones(&self) -> i64 { unsafe { intrinsics::ctpop64(*self) } }
 34  
 35      /// Returns the number of leading zeros in the in the binary representation
 36      /// of the number.
 37      #[inline]
 38      fn leading_zeros(&self) -> i64 { unsafe { intrinsics::ctlz64(*self) } }
 39  
 40      /// Counts the number of trailing zeros.
 41      #[inline]
 42      fn trailing_zeros(&self) -> i64 { unsafe { intrinsics::cttz64(*self) } }
 43  }

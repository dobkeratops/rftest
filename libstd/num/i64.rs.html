 23  impl BitCount for i64 {
 24      /// Counts the number of bits set. Wraps LLVM's `ctpop` intrinsic.
 25      #[inline]
 26      fn population_count(&self) -> i64 { unsafe { intrinsics::ctpop64(*self) } }
 27  
 28      /// Counts the number of leading zeros. Wraps LLVM's `ctlz` intrinsic.
 29      #[inline]
 30      fn leading_zeros(&self) -> i64 { unsafe { intrinsics::ctlz64(*self) } }
 31  
 32      /// Counts the number of trailing zeros. Wraps LLVM's `cttz` intrinsic.
 33      #[inline]
 34      fn trailing_zeros(&self) -> i64 { unsafe { intrinsics::cttz64(*self) } }
 35  }

 37  impl CheckedAdd for i64 {
 38      #[inline]
 39      fn checked_add(&self, v: &i64) -> Option<i64> {
 40          unsafe {
 41              let (x, y) = intrinsics::i64_add_with_overflow(*self, *v);
 42              if y { None } else { Some(x) }
 43          }
 44      }
 45  }

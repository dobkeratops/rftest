  184  /// Raises a value to the power of exp, using exponentiation by squaring.
  185  ///
  186  /// # Example
  187  ///
  188  /// ```rust
  189  /// use std::num;

  193  #[inline]
  194  pub fn pow<T: One + Mul<T, T>>(mut base: T, mut exp: uint) -> T {
  195      if exp == 1 { base }
  196      else {
  197          let mut acc = one::<T>();
  198          while exp > 0 {
  199              if (exp & 1) == 1 {
  200                  acc = acc * base;
  201              }
  202              base = base * base;
  203              exp = exp >> 1;
  204          }
  205          acc
  206      }
  207  }

  301  /// Returns the smallest power of 2 greater than or equal to `n`.
  302  #[inline]
  303  pub fn next_power_of_two<T: Unsigned + Int>(n: T) -> T {
  304      let halfbits: T = cast(size_of::<T>() * 4).unwrap();
  305      let mut tmp: T = n - one();
  306      let mut shift: T = one();

  308          tmp = tmp | (tmp >> shift);
  309          shift = shift << one();
  310      }
  311      tmp + one()
  312  }

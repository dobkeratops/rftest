  152  /// Raises a value to the power of exp, using exponentiation by squaring.
  153  ///
  154  /// # Example
  155  ///
  156  /// ```rust
  157  /// use std::num;
  158  ///
  159  /// assert_eq!(num::pow(2, 4), 16);
  160  /// ```
  161  #[inline]
  162  pub fn pow<T: One + Mul<T, T>>(mut base: T, mut exp: uint) -> T {
  163      if exp == 1 { base }
  164      else {
  165          let mut acc = one::<T>();
  166          while exp > 0 {
  167              if (exp & 1) == 1 {
  168                  acc = acc * base;
  169              }
  170              base = base * base;
  171              exp = exp >> 1;
  172          }
  173          acc
  174      }
  175  }

  268  /// Returns the smallest power of 2 greater than or equal to `n`.
  269  #[inline]
  270  pub fn next_power_of_two<T: Unsigned + Int>(n: T) -> T {
  271      let halfbits: T = cast(size_of::<T>() * 4).unwrap();
  272      let mut tmp: T = n - one();
  273      let mut shift: T = one();
  274      while shift <= halfbits {
  275          tmp = tmp | (tmp >> shift);
  276          shift = shift << one();
  277      }
  278      tmp + one()
  279  }

  303 ///
  304 /// Compute the exponentiation of an integer by another integer as a float
  305 ///
  306 /// # Arguments
  307 ///
  308 /// * x - The base
  309 /// * pow - The exponent
  310 ///
  311 /// # Return value
  312 ///
  313 /// `NaN` if both `x` and `pow` are `0u`, otherwise `x^pow`
  314 ///
  315 pub fn pow_with_uint(base: uint, pow: uint) -> float {
  316     if base == 0u {
  317         if pow == 0u {
  318             return NaN as float;
  319         }
  320         return 0.;
  321     }
  322     let mut my_pow     = pow;
  323     let mut total      = 1f;
  324     let mut multiplier = base as float;
  325     while (my_pow > 0u) {
  326         if my_pow % 2u == 1u {
  327             total = total * multiplier;
  328         }
  329         my_pow     /= 2u;
  330         multiplier *= multiplier;
  331     }
  332     return total;
  333 }

 54  /// Returns `base` raised to the power of `exponent`
 55  pub fn pow(base: int, exponent: uint) -> int {
 56      if exponent == 0u {
 57          //Not mathemtically true if ~[base == 0]
 58          return 1;
 59      }
 60      if base == 0 { return 0; }
 61      let mut my_pow  = exponent;
 62      let mut acc     = 1;
 63      let mut multiplier = base;
 64      while(my_pow > 0u) {
 65          if my_pow % 2u == 1u {
 66              acc *= multiplier;
 67          }
 68          my_pow     /= 2u;
 69          multiplier *= multiplier;
 70      }
 71      return acc;
 72  }

 79      assert!((pow(-1, 0u) == 1));

 83      assert!((pow(4, 9u) == 262144));
 84  }

 122  /// Returns `base` raised to the power of `exponent`
 123  pub fn pow(base: int, exponent: uint) -> int {
 124      if exponent == 0u {
 125          //Not mathemtically true if ~[base == 0]
 126          return 1;
 127      }
 128      if base == 0 { return 0; }
 129      let mut my_pow  = exponent;
 130      let mut acc     = 1;
 131      let mut multiplier = base;
 132      while(my_pow > 0u) {
 133          if my_pow % 2u == 1u {
 134              acc *= multiplier;
 135          }
 136          my_pow     /= 2u;
 137          multiplier *= multiplier;
 138      }
 139      return acc;
 140  }

 54 /// Returns `base` raised to the power of `exponent`
 55 pub fn pow(base: int, exponent: uint) -> int {
 56     if exponent == 0u {
 57         //Not mathemtically true if ~[base == 0]
 58         return 1;
 59     }
 60     if base == 0 { return 0; }
 61     let mut my_pow  = exponent;
 62     let mut acc     = 1;
 63     let mut multiplier = base;
 64     while(my_pow > 0u) {
 65         if my_pow % 2u == 1u {
 66             acc *= multiplier;
 67         }
 68         my_pow     /= 2u;
 69         multiplier *= multiplier;
 70     }
 71     return acc;
 72 }

 74 #[test]
 75 fn test_pow() {
 76     assert!((pow(0, 0u) == 1));
 77     assert!((pow(0, 1u) == 0));
 78     assert!((pow(0, 2u) == 0));
 79     assert!((pow(-1, 0u) == 1));
 80     assert!((pow(1, 0u) == 1));
 81     assert!((pow(-3, 2u) == 9));
 82     assert!((pow(-3, 3u) == -27));
 83     assert!((pow(4, 9u) == 262144));
 84 }

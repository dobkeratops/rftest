  355      fn classify(&self) -> FPCategory {
  356          static EXP_MASK: u32 = 0x7f800000;
  357          static MAN_MASK: u32 = 0x007fffff;
  358  
  359          let bits: u32 = unsafe {::cast::transmute(*self)};

  361              (0, 0)        => FPZero,
  362              (_, 0)        => FPSubnormal,
  363              (0, EXP_MASK) => FPInfinite,

  365              _             => FPNormal,
  366          }

  205  impl Orderable for f32 {
  206      /// Returns `NaN` if either of the numbers are `NaN`.
  207      #[inline]
  208      fn min(&self, other: &f32) -> f32 {
  209          match () {
  210              _ if self.is_nan()  => *self,
  211              _ if other.is_nan() => *other,
  212              _ if *self < *other => *self,
  213              _                   => *other,
  214          }
  215      }
  216  
  217      /// Returns `NaN` if either of the numbers are `NaN`.
  218      #[inline]
  219      fn max(&self, other: &f32) -> f32 {
  220          match () {
  221              _ if self.is_nan()  => *self,
  222              _ if other.is_nan() => *other,
  223              _ if *self > *other => *self,
  224              _                   => *other,
  225          }
  226      }
  227  
  228      /// Returns the number constrained within the range `mn <= self <= mx`.
  229      /// If any of the numbers are `NaN` then `NaN` is returned.
  230      #[inline]

  235              _ if !(*self >= *mn) => *mn,
  236              _                    => *self,
  237          }
  238      }

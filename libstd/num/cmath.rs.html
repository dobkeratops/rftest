  66          pub fn ldexp_radix(n: c_double, i: c_int) -> c_double;

  68          pub fn tan(n: c_double) -> c_double;
  69          pub fn tanh(n: c_double) -> c_double;
  70          pub fn tgamma(n: c_double) -> c_double;

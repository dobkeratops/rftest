 11 //! Operations and constants for `i16`
 12 
 13 use num::BitCount;
 14 use unstable::intrinsics;
 15 
 16 pub use self::generated::*;
 17 
 18 int_module!(i16, 16)
 19 
 20 impl BitCount for i16 {
 21     /// Counts the number of bits set. Wraps LLVM's `ctpop` intrinsic.
 22     #[inline]
 23     fn population_count(&self) -> i16 { unsafe { intrinsics::ctpop16(*self) } }
 24 
 25     /// Counts the number of leading zeros. Wraps LLVM's `ctlz` intrinsic.
 26     #[inline]
 27     fn leading_zeros(&self) -> i16 { unsafe { intrinsics::ctlz16(*self) } }
 28 
 29     /// Counts the number of trailing zeros. Wraps LLVM's `cttz` intrinsic.
 30     #[inline]
 31     fn trailing_zeros(&self) -> i16 { unsafe { intrinsics::cttz16(*self) } }
 32 }

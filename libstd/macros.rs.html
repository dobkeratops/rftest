 11  #[macro_escape];
 12  
 13  macro_rules! rterrln (
 14      ($( $arg:expr),+) => ( {
 15          ::rt::util::dumb_println(fmt!( $($arg),+ ));
 16      } )
 17  )
 18  
 19  // Some basic logging
 20  macro_rules! rtdebug_ (
 21      ($( $arg:expr),+) => ( {
 22          rterrln!( $($arg),+ )
 23      } )
 24  )
 25  
 26  // An alternate version with no output, for turning off logging. An
 27  // earlier attempt that did not call the fmt! macro was insufficient,
 28  // as a case of the "let bind each variable" approach eventually
 29  // failed without an error message describing the invocation site.
 30  macro_rules! rtdebug (
 31      ($( $arg:expr),+) => ( {
 32          let _x = fmt!( $($arg),+ );
 33      })
 34  )
 35  
 36  macro_rules! rtassert (
 37      ( $arg:expr ) => ( {
 38          if !$arg {
 39              rtabort!("assertion failed: %s", stringify!($arg));
 40          }
 41      } )
 42  )
 43  
 44  
 45  macro_rules! rtabort(
 46      ($( $msg:expr),+) => ( {
 47          ::rt::util::abort(fmt!($($msg),+));
 48      } )
 49  )

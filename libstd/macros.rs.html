 11  #[macro_escape];
 12  
 13  macro_rules! rterrln (
 14      ($( $arg:expr),+) => ( {
 15          ::rt::util::dumb_println(fmt!( $($arg),+ ));
 16      } )
 17  )
 18  
 19  // Some basic logging. Enabled by passing `--cfg rtdebug` to the libstd build.
 20  macro_rules! rtdebug (
 21      ($( $arg:expr),+) => ( {
 22          if cfg!(rtdebug) {
 23              rterrln!( $($arg),+ )
 24          }
 25      })
 26  )
 27  
 28  macro_rules! rtassert (
 29      ( $arg:expr ) => ( {
 30          if !$arg {
 31              rtabort!("assertion failed: %s", stringify!($arg));
 32          }
 33      } )
 34  )
 35  
 36  
 37  macro_rules! rtabort(
 38      ($( $msg:expr),+) => ( {
 39          ::rt::util::abort(fmt!($($msg),+));
 40      } )
 41  )

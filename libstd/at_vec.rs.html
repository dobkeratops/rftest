  24  /// Returns the number of elements the vector can hold without reallocating
  25  #[inline]
  26  pub fn capacity<T>(v: @[T]) -> uint {
  27      unsafe {
  28          let box = v.repr();
  29          (*box).data.alloc / sys::size_of::<T>()
  30      }
  31  }

  33  /**
  34   * Builds a vector by calling a provided function with an argument
  35   * function that pushes an element to the back of a vector.
  36   * The initial size for the vector may optionally be specified
  37   *
  38   * # Arguments
  39   *
  40   * * size - An option, maybe containing initial size of the vector to reserve
  41   * * builder - A function that will construct the vector. It receives
  42   *             as an argument a function that will push an element
  43   *             onto the vector being constructed.
  44   */
  45  #[inline]
  46  pub fn build<A>(size: Option<uint>, builder: &fn(push: &fn(v: A))) -> @[A] {
  47      let mut vec = @[];
  48      unsafe { raw::reserve(&mut vec, size.unwrap_or(4)); }
  49      builder(|x| unsafe { raw::push(&mut vec, x) });
  50      vec
  51  }

  53  // Appending
  54  
  55  /// Iterates over the `rhs` vector, copying each element and appending it to the
  56  /// `lhs`. Afterwards, the `lhs` is then returned for use again.
  57  #[inline]
  58  pub fn append<T:Clone>(lhs: @[T], rhs: &[T]) -> @[T] {
  59      do build(Some(lhs.len() + rhs.len())) |push| {
  60          for x in lhs.iter() {
  61              push((*x).clone());
  62          }
  63          for elt in rhs.iter() {
  64              push(elt.clone());
  65          }
  66      }
  67  }

  70  /// Apply a function to each element of a vector and return the results
  71  pub fn map<T, U>(v: &[T], f: &fn(x: &T) -> U) -> @[U] {
  72      do build(Some(v.len())) |push| {
  73          for elem in v.iter() {
  74              push(f(elem));
  75          }
  76      }
  77  }

  79  /**
  80   * Creates and initializes an immutable vector.
  81   *
  82   * Creates an immutable vector of size `n_elts` and initializes the elements
  83   * to the value returned by the function `op`.
  84   */
  85  pub fn from_fn<T>(n_elts: uint, op: &fn(uint) -> T) -> @[T] {
  86      do build(Some(n_elts)) |push| {
  87          let mut i: uint = 0u;
  88          while i < n_elts { push(op(i)); i += 1u; }
  89      }
  90  }

  92  /**
  93   * Creates and initializes an immutable vector.
  94   *
  95   * Creates an immutable vector of size `n_elts` and initializes the elements
  96   * to the value `t`.
  97   */
  98  pub fn from_elem<T:Clone>(n_elts: uint, t: T) -> @[T] {
  99      do build(Some(n_elts)) |push| {
 100          let mut i: uint = 0u;
 101          while i < n_elts {
 102              push(t.clone());
 103              i += 1u;
 104          }
 105      }
 106  }

 168       * Sets the length of a vector
 169       *
 170       * This will explicitly set the size of the vector, without actually
 171       * modifying its buffers, so it is up to the caller to ensure that
 172       * the vector is actually the specified size.

 174      #[inline]
 175      pub unsafe fn set_len<T>(v: &mut @[T], new_len: uint) {
 176          let repr: *mut Box<Vec<T>> = cast::transmute_copy(v);
 177          (*repr).data.fill = new_len * sys::size_of::<T>();
 178      }

 180      /**
 181       * Pushes a new value onto this vector.
 182       */
 183      #[inline]
 184      pub unsafe fn push<T>(v: &mut @[T], initval: T) {
 185          let full = {
 186              let repr: *Box<Vec<T>> = cast::transmute_copy(v);
 187              (*repr).data.alloc > (*repr).data.fill
 188          };
 189          if full {
 190              push_fast(v, initval);
 191          } else {
 192              push_slow(v, initval);
 193          }
 194      }

 196      #[inline] // really pretty please
 197      unsafe fn push_fast<T>(v: &mut @[T], initval: T) {
 198          let repr: *mut Box<Vec<T>> = cast::transmute_copy(v);
 199          let amt = v.len();
 200          (*repr).data.fill += sys::size_of::<T>();
 201          let p = ptr::offset(&(*repr).data.data as *T, amt as int) as *mut T;
 202          move_val_init(&mut(*p), initval);
 203      }

 205      unsafe fn push_slow<T>(v: &mut @[T], initval: T) {
 206          reserve_at_least(v, v.len() + 1u);
 207          push_fast(v, initval);
 208      }

 210      /**
 211       * Reserves capacity for exactly `n` elements in the given vector.
 212       *
 213       * If the capacity for `v` is already equal to or greater than the
 214       * requested capacity, then no action is taken.
 215       *
 216       * # Arguments
 217       *
 218       * * v - A vector
 219       * * n - The number of elements to reserve space for
 220       */
 221      pub unsafe fn reserve<T>(v: &mut @[T], n: uint) {
 222          // Only make the (slow) call into the runtime if we have to
 223          if capacity(*v) < n {
 224              let ptr: *mut *mut Box<Vec<()>> = transmute(v);
 225              let ty = intrinsics::get_tydesc::<T>();
 226              return reserve_raw(ty, ptr, n);
 227          }
 228      }

 230      // Implementation detail. Shouldn't be public
 231      #[allow(missing_doc)]
 232      pub fn reserve_raw(ty: *TyDesc, ptr: *mut *mut Box<Vec<()>>, n: uint) {
 233          // check for `uint` overflow
 234          unsafe {
 235              if n > (**ptr).data.alloc / (*ty).size {
 236                  let alloc = n * (*ty).size;
 237                  let total_size = alloc + sys::size_of::<Vec<()>>();
 238                  if alloc / (*ty).size != n || total_size < alloc {
 239                      fail!("vector size is too large: %u", n);
 240                  }
 241                  (*ptr) = local_realloc(*ptr as *(), total_size) as *mut Box<Vec<()>>;
 242                  (**ptr).data.alloc = alloc;
 243              }
 244          }
 245  
 246          fn local_realloc(ptr: *(), size: uint) -> *() {
 247              use rt::local::Local;
 248              use rt::task::Task;
 249  
 250              do Local::borrow |task: &mut Task| {
 251                  task.heap.realloc(ptr as *libc::c_void, size) as *()
 252              }
 253          }
 254      }

 256      /**
 257       * Reserves capacity for at least `n` elements in the given vector.
 258       *
 259       * This function will over-allocate in order to amortize the
 260       * allocation costs in scenarios where the caller may need to
 261       * repeatedly reserve additional space.
 262       *
 263       * If the capacity for `v` is already equal to or greater than the
 264       * requested capacity, then no action is taken.
 265       *
 266       * # Arguments
 267       *
 268       * * v - A vector
 269       * * n - The number of elements to reserve space for
 270       */

 272          reserve(v, uint::next_power_of_two(n));
 273      }

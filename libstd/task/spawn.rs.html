  11  /*!**************************************************************************
  12   * Spawning & linked failure
  13   *
  14   * Several data structures are involved in task management to allow properly
  15   * propagating failure across linked/supervised tasks.
  16   *
  17   * (1) The "taskgroup_arc" is an unsafe::exclusive which contains a hashset of
  18   *     all tasks that are part of the group. Some tasks are 'members', which
  19   *     means if they fail, they will kill everybody else in the taskgroup.
  20   *     Other tasks are 'descendants', which means they will not kill tasks
  21   *     from this group, but can be killed by failing members.
  22   *
  23   *     A new one of these is created each spawn_linked or spawn_supervised.
  24   *
  25   * (2) The "tcb" is a per-task control structure that tracks a task's spawn
  26   *     configuration. It contains a reference to its taskgroup_arc, a
  27   *     reference to its node in the ancestor list (below), a flag for
  28   *     whether it's part of the 'main'/'root' taskgroup, and an optionally
  29   *     configured notification port. These are stored in TLS.
  30   *
  31   * (3) The "ancestor_list" is a cons-style list of unsafe::exclusives which
  32   *     tracks 'generations' of taskgroups -- a group's ancestors are groups
  33   *     which (directly or transitively) spawn_supervised-ed them. Each task
  34   *     is recorded in the 'descendants' of each of its ancestor groups.
  35   *
  36   *     Spawning a supervised task is O(n) in the number of generations still
  37   *     alive, and exiting (by success or failure) that task is also O(n).
  38   *
  39   * This diagram depicts the references between these data structures:
  40   *
  41   *          linked_________________________________
  42   *        ___/                   _________         \___
  43   *       /   \                  | group X |        /   \
  44   *      (  A  ) - - - - - - - > | {A,B} {}|< - - -(  B  )
  45   *       \___/                  |_________|        \___/
  46   *      unlinked
  47   *         |      __ (nil)
  48   *         |      //|                         The following code causes this:
  49   *         |__   //   /\         _________
  50   *        /   \ //    ||        | group Y |     fn taskA() {
  51   *       (  C  )- - - ||- - - > |{C} {D,E}|         spawn(taskB);
  52   *        \___/      /  \=====> |_________|         spawn_unlinked(taskC);
  53   *      supervise   /gen \                          ...
  54   *         |    __  \ 00 /                      }
  55   *         |    //|  \__/                       fn taskB() { ... }
  56   *         |__ //     /\         _________      fn taskC() {
  57   *        /   \/      ||        | group Z |         spawn_supervised(taskD);
  58   *       (  D  )- - - ||- - - > | {D} {E} |         ...
  59   *        \___/      /  \=====> |_________|     }
  60   *      supervise   /gen \                      fn taskD() {
  61   *         |    __  \ 01 /                          spawn_supervised(taskE);
  62   *         |    //|  \__/                           ...
  63   *         |__ //                _________      }
  64   *        /   \/                | group W |     fn taskE() { ... }
  65   *       (  E  )- - - - - - - > | {E}  {} |
  66   *        \___/                 |_________|
  67   *
  68   *        "tcb"               "taskgroup_arc"
  69   *             "ancestor_list"
  70   *

 149  // One of these per group of linked-failure tasks.
 150  struct TaskGroupData {
 151      // All tasks which might kill this group. When this is empty, the group
 152      // can be "GC"ed (i.e., its link in the ancestor list can be removed).
 153      members:     TaskSet,
 154      // All tasks unidirectionally supervised by (directly or transitively)
 155      // tasks in this group.
 156      descendants: TaskSet,
 157  }
 158  type TaskGroupArc = Exclusive<Option<TaskGroupData>>;
 159  
 160  type TaskGroupInner<'self> = &'self mut Option<TaskGroupData>;
 161  
 162  // A taskgroup is 'dead' when nothing can cause it to fail; only members can.
 163  fn taskgroup_is_dead(tg: &TaskGroupData) -> bool {
 164      tg.members.is_empty()
 165  }

 167  // A list-like structure by which taskgroups keep track of all ancestor groups
 168  // which may kill them. Needed for tasks to be able to remove themselves from
 169  // ancestor groups upon exit. The list has a node for each "generation", and
 170  // ends either at the root taskgroup (which has no ancestors) or at a
 171  // taskgroup which was spawned-unlinked. Tasks from intermediate generations
 172  // have references to the middle of the list; when intermediate generations
 173  // die, their node in the list will be collected at a descendant's spawn-time.
 174  struct AncestorNode {
 175      // Since the ancestor list is recursive, we end up with references to
 176      // exclusives within other exclusives. This is dangerous business (if
 177      // circular references arise, deadlock and memory leaks are imminent).
 178      // Hence we assert that this counter monotonically decreases as we
 179      // approach the tail of the list.
 180      generation:     uint,
 181      // Handle to the tasks in the group of the current generation.
 182      parent_group:   TaskGroupArc,
 183      // Recursive rest of the list.
 184      ancestors:      AncestorList,
 185  }
 186  
 187  struct AncestorList(Option<Exclusive<AncestorNode>>);
 188  
 189  // Accessors for taskgroup arcs and ancestor arcs that wrap the unsafety.
 190  #[inline]
 191  fn access_group<U>(x: &TaskGroupArc, blk: &fn(TaskGroupInner) -> U) -> U {
 192      unsafe {
 193          x.with(blk)
 194      }
 195  }
 196  
 197  #[inline]
 198  fn access_ancestors<U>(x: &Exclusive<AncestorNode>,
 199                         blk: &fn(x: &mut AncestorNode) -> U) -> U {
 200      unsafe {
 201          x.with(blk)
 202      }
 203  }

 206  fn check_generation(younger: uint, older: uint) { assert!(younger > older); }
 207  #[inline] #[cfg(not(test))]
 208  fn check_generation(_younger: uint, _older: uint) { }

 210  #[inline] #[cfg(test)]
 211  fn incr_generation(ancestors: &AncestorList) -> uint {
 212      ancestors.map_default(0, |arc| access_ancestors(arc, |a| a.generation+1))
 213  }
 214  #[inline] #[cfg(not(test))]
 215  fn incr_generation(_ancestors: &AncestorList) -> uint { 0 }

 217  // Iterates over an ancestor list.
 218  // (1) Runs forward_blk on each ancestral taskgroup in the list
 219  // (2) If forward_blk "break"s, runs optional bail_blk on all ancestral
 220  //     taskgroups that forward_blk already ran on successfully (Note: bail_blk
 221  //     is NOT called on the block that forward_blk broke on!).
 222  // (3) As a bonus, coalesces away all 'dead' taskgroup nodes in the list.
 223  fn each_ancestor(list:        &mut AncestorList,
 224                   bail_blk:    &fn(TaskGroupInner),
 225                   forward_blk: &fn(TaskGroupInner) -> bool)
 226                -> bool {
 227      // "Kickoff" call - there was no last generation.
 228      return !coalesce(list, bail_blk, forward_blk, uint::max_value);
 229  
 230      // Recursively iterates, and coalesces afterwards if needed. Returns
 231      // whether or not unwinding is needed (i.e., !successful iteration).
 232      fn coalesce(list:            &mut AncestorList,
 233                  bail_blk:        &fn(TaskGroupInner),
 234                  forward_blk:     &fn(TaskGroupInner) -> bool,
 235                  last_generation: uint) -> bool {
 236          let (coalesce_this, early_break) =
 237              iterate(list, bail_blk, forward_blk, last_generation);
 238          // What should our next ancestor end up being?
 239          if coalesce_this.is_some() {
 240              // Needed coalesce. Our next ancestor becomes our old
 241              // ancestor's next ancestor. ("next = old_next->next;")
 242              *list = coalesce_this.unwrap();
 243          }
 244          return early_break;
 245      }
 246  
 247      // Returns an optional list-to-coalesce and whether unwinding is needed.
 248      // Option<ancestor_list>:
 249      //     Whether or not the ancestor taskgroup being iterated over is
 250      //     dead or not; i.e., it has no more tasks left in it, whether or not
 251      //     it has descendants. If dead, the caller shall coalesce it away.
 252      // bool:
 253      //     True if the supplied block did 'break', here or in any recursive
 254      //     calls. If so, must call the unwinder on all previous nodes.
 255      fn iterate(ancestors:       &mut AncestorList,
 256                 bail_blk:        &fn(TaskGroupInner),
 257                 forward_blk:     &fn(TaskGroupInner) -> bool,
 258                 last_generation: uint)
 259              -> (Option<AncestorList>, bool) {
 260          // At each step of iteration, three booleans are at play which govern
 261          // how the iteration should behave.
 262          // 'nobe_is_dead' - Should the list should be coalesced at this point?
 263          //                  Largely unrelated to the other two.
 264          // 'need_unwind'  - Should we run the bail_blk at this point? (i.e.,
 265          //                  do_continue was false not here, but down the line)
 266          // 'do_continue'  - Did the forward_blk succeed at this point? (i.e.,
 267          //                  should we recurse? or should our callers unwind?)
 268  
 269          let forward_blk = Cell::new(forward_blk);
 270  
 271          // The map defaults to None, because if ancestors is None, we're at
 272          // the end of the list, which doesn't make sense to coalesce.
 273          do ancestors.map_default((None,false)) |ancestor_arc| {
 274              // NB: Takes a lock! (this ancestor node)
 275              do access_ancestors(ancestor_arc) |nobe| {
 276                  // Argh, but we couldn't give it to coalesce() otherwise.
 277                  let forward_blk = forward_blk.take();
 278                  // Check monotonicity
 279                  check_generation(last_generation, nobe.generation);
 280                  /*##########################################################*
 281                   * Step 1: Look at this ancestor group (call iterator block).
 282                   *##########################################################*/
 283                  let mut nobe_is_dead = false;
 284                  let do_continue =
 285                      // NB: Takes a lock! (this ancestor node's parent group)
 286                      do access_group(&nobe.parent_group) |tg_opt| {
 287                          // Decide whether this group is dead. Note that the
 288                          // group being *dead* is disjoint from it *failing*.
 289                          nobe_is_dead = match *tg_opt {
 290                              Some(ref tg) => taskgroup_is_dead(tg),
 291                              None => nobe_is_dead
 292                          };
 293                          // Call iterator block. (If the group is dead, it's
 294                          // safe to skip it. This will leave our TaskHandle
 295                          // hanging around in the group even after it's freed,
 296                          // but that's ok because, by virtue of the group being
 297                          // dead, nobody will ever kill-all (for) over it.)
 298                          if nobe_is_dead { true } else { forward_blk(tg_opt) }
 299                      };
 300                  /*##########################################################*
 301                   * Step 2: Recurse on the rest of the list; maybe coalescing.
 302                   *##########################################################*/
 303                  // 'need_unwind' is only set if blk returned true above, *and*
 304                  // the recursive call early-broke.
 305                  let mut need_unwind = false;
 306                  if do_continue {
 307                      // NB: Takes many locks! (ancestor nodes & parent groups)
 308                      need_unwind = coalesce(&mut nobe.ancestors, |tg| bail_blk(tg),
 309                                             forward_blk, nobe.generation);
 310                  }
 311                  /*##########################################################*
 312                   * Step 3: Maybe unwind; compute return info for our caller.
 313                   *##########################################################*/
 314                  if need_unwind && !nobe_is_dead {
 315                      do access_group(&nobe.parent_group) |tg_opt| {
 316                          bail_blk(tg_opt)
 317                      }
 318                  }
 319                  // Decide whether our caller should unwind.
 320                  need_unwind = need_unwind || !do_continue;
 321                  // Tell caller whether or not to coalesce and/or unwind
 322                  if nobe_is_dead {
 323                      // Swap the list out here; the caller replaces us with it.
 324                      let rest = util::replace(&mut nobe.ancestors,
 325                                               AncestorList(None));
 326                      (Some(rest), need_unwind)
 327                  } else {
 328                      (None, need_unwind)
 329                  }
 330              }
 331          }
 332      }
 333  }

 335  // One of these per task.
 336  pub struct Taskgroup {
 337      // List of tasks with whose fates this one's is intertwined.
 338      tasks:      TaskGroupArc, // 'none' means the group has failed.
 339      // Lists of tasks who will kill us if they fail, but whom we won't kill.
 340      ancestors:  AncestorList,
 341      is_main:    bool,
 342      notifier:   Option<AutoNotify>,
 343  }
 344  
 345  impl Drop for Taskgroup {
 346      // Runs on task exit.
 347      fn drop(&self) {
 348          unsafe {
 349              // FIXME(#4330) Need self by value to get mutability.
 350              let this: &mut Taskgroup = transmute(self);
 351  
 352              // If we are failing, the whole taskgroup needs to die.
 353              do RuntimeGlue::with_task_handle_and_failing |me, failing| {
 354                  if failing {
 355                      for x in this.notifier.mut_iter() {
 356                          x.failed = true;
 357                      }
 358                      // Take everybody down with us.
 359                      do access_group(&self.tasks) |tg| {
 360                          kill_taskgroup(tg, &me, self.is_main);
 361                      }
 362                  } else {
 363                      // Remove ourselves from the group(s).
 364                      do access_group(&self.tasks) |tg| {
 365                          leave_taskgroup(tg, &me, true);
 366                      }
 367                  }
 368                  // It doesn't matter whether this happens before or after dealing
 369                  // with our own taskgroup, so long as both happen before we die.
 370                  // We remove ourself from every ancestor we can, so no cleanup; no
 371                  // break.
 372                  do each_ancestor(&mut this.ancestors, |_| {}) |ancestor_group| {
 373                      leave_taskgroup(ancestor_group, &me, false);
 374                      true
 375                  };
 376              }
 377          }

 417                             is_member: bool) -> bool {

 431  fn leave_taskgroup(state: TaskGroupInner, me: &TaskHandle,

 444  // NB: Runs in destructor/post-exit context. Can't 'fail'.
 445  fn kill_taskgroup(state: TaskGroupInner, me: &TaskHandle, is_main: bool) {
 446      unsafe {
 447          // NB: We could do the killing iteration outside of the group arc, by
 448          // having "let mut newstate" here, swapping inside, and iterating
 449          // after. But that would let other exiting tasks fall-through and exit
 450          // while we were trying to kill them, causing potential
 451          // use-after-free. A task's presence in the arc guarantees it's alive
 452          // only while we hold the lock, so if we're failing, all concurrently
 453          // exiting tasks must wait for us. To do it differently, we'd have to
 454          // use the runtime's task refcounting, but that could leave task
 455          // structs around long after their task exited.
 456          let newstate = util::replace(state, None);
 457          // Might already be None, if Somebody is failing simultaneously.
 458          // That's ok; only one task needs to do the dirty work. (Might also
 459          // see 'None' if Somebody already failed and we got a kill signal.)
 460          if newstate.is_some() {
 461              let TaskGroupData { members: members, descendants: descendants } =
 462                  newstate.unwrap();
 463              for sibling in members.move_iter() {
 464                  // Skip self - killing ourself won't do much good.
 465                  if &sibling != me {
 466                      RuntimeGlue::kill_task(sibling);
 467                  }
 468              }
 469              for child in descendants.move_iter() {
 470                  assert!(&child != me);
 471                  RuntimeGlue::kill_task(child);
 472              }
 473              // Only one task should ever do this.
 474              if is_main {
 475                  RuntimeGlue::kill_all_tasks(me);
 476              }
 477              // Do NOT restore state to Some(..)! It stays None to indicate
 478              // that the whole taskgroup is failing, to forbid new spawns.
 479          }
 480          // (note: multiple tasks may reach this point)
 481      }
 482  }

 484  // FIXME (#2912): Work around core-vs-coretest function duplication. Can't use
 485  // a proper closure because the #[test]s won't understand. Have to fake it.
 486  fn taskgroup_key() -> local_data::Key<@@mut Taskgroup> {
 487      unsafe { cast::transmute(-2) }
 488  }

 531      fn with_my_taskgroup<U>(blk: &fn(&Taskgroup) -> U) -> U {
 532          if in_green_task_context() {
 533              unsafe {
 534                  // Can't use safe borrow, because creating new hashmaps for the
 535                  // tasksets requires an rng, which needs to borrow the sched.
 536                  let me = Local::unsafe_borrow::<Task>();
 537                  blk(match (*me).taskgroup {
 538                      None => {
 539                          // First task in its (unlinked/unsupervised) taskgroup.
 540                          // Lazily initialize.
 541                          let mut members = TaskSet::new();

 543                          members.insert(NewTask(my_handle));
 544                          let tasks = Exclusive::new(Some(TaskGroupData {
 545                              members: members,
 546                              descendants: TaskSet::new(),
 547                          }));

 550                          // behaviour.

 554                      }
 555                      Some(ref group) => group,
 556                  })
 557              }
 558          } else {
 559              rtabort!("spawning in bad context")
 560          }
 561      }
 562  }

 564  // Returns 'None' in the case where the child's TG should be lazily initialized.
 565  fn gen_child_taskgroup(linked: bool, supervised: bool)
 566      -> Option<(TaskGroupArc, AncestorList, bool)> {
 567      // FIXME(#7544): Not safe to lazily initialize in the old runtime. Remove
 568      // this context check once 'spawn_raw_oldsched' is gone.
 569      if linked || supervised {
 570          // with_my_taskgroup will lazily initialize the parent's taskgroup if
 571          // it doesn't yet exist. We don't want to call it in the unlinked case.
 572          do RuntimeGlue::with_my_taskgroup |spawner_group| {
 573              let ancestors = AncestorList(spawner_group.ancestors.map(|x| x.clone()));
 574              if linked {
 575                  // Child is in the same group as spawner.
 576                  // Child's ancestors are spawner's ancestors.
 577                  // Propagate main-ness.
 578                  Some((spawner_group.tasks.clone(), ancestors, spawner_group.is_main))
 579              } else {
 580                  // Child is in a separate group from spawner.
 581                  let g = Exclusive::new(Some(TaskGroupData {
 582                      members:     TaskSet::new(),
 583                      descendants: TaskSet::new(),
 584                  }));
 585                  let a = if supervised {
 586                      let new_generation = incr_generation(&ancestors);
 587                      assert!(new_generation < uint::max_value);
 588                      // Child's ancestors start with the spawner.
 589                      // Build a new node in the ancestor list.
 590                      AncestorList(Some(Exclusive::new(AncestorNode {
 591                          generation: new_generation,
 592                          parent_group: spawner_group.tasks.clone(),
 593                          ancestors: ancestors,
 594                      })))
 595                  } else {
 596                      // Child has no ancestors.
 597                      AncestorList(None)
 598                  };
 599                  Some((g, a, false))
 600              }
 601          }
 602      } else {
 603          None
 604      }
 605  }
 606  
 607  // Set up membership in taskgroup and descendantship in all ancestor
 608  // groups. If any enlistment fails, Some task was already failing, so
 609  // don't let the child task run, and undo every successful enlistment.
 610  fn enlist_many(child: TaskHandle, child_arc: &TaskGroupArc,
 611                 ancestors: &mut AncestorList) -> bool {
 612      // Join this taskgroup.
 613      let mut result = do access_group(child_arc) |child_tg| {
 614          enlist_in_taskgroup(child_tg, child.clone(), true) // member
 615      };
 616      if result {
 617          // Unwinding function in case any ancestral enlisting fails
 618          let bail: &fn(TaskGroupInner) = |tg| { leave_taskgroup(tg, &child, false) };
 619          // Attempt to join every ancestor group.
 620          result = do each_ancestor(ancestors, bail) |ancestor_tg| {
 621              // Enlist as a descendant, not as an actual member.
 622              // Descendants don't kill ancestor groups on failure.
 623              enlist_in_taskgroup(ancestor_tg, child.clone(), false)
 624          };
 625          // If any ancestor group fails, need to exit this group too.
 626          if !result {
 627              do access_group(child_arc) |child_tg| {
 628                  leave_taskgroup(child_tg, &child, true); // member
 629              }
 630          }
 631      }
 632      result
 633  }

 636      if in_green_task_context() {
 637          spawn_raw_newsched(opts, f)
 638      } else {
 639          fail!("can't spawn from this context")
 640      }
 641  }

 643  fn spawn_raw_newsched(mut opts: TaskOpts, f: ~fn()) {
 644      use rt::sched::*;
 645  
 646      let child_data = Cell::new(gen_child_taskgroup(opts.linked, opts.supervised));
 647      let indestructible = opts.indestructible;
 648  
 649      let child_wrapper: ~fn() = || {
 650          // Child task runs this code.
 651  
 652          // If child data is 'None', the enlist is vacuously successful.
 653          let enlist_success = do child_data.take().map_move_default(true) |child_data| {
 654              let child_data = Cell::new(child_data); // :(
 655              do Local::borrow::<Task, bool> |me| {
 656                  let (child_tg, ancestors, is_main) = child_data.take();
 657                  let mut ancestors = ancestors;
 658                  // FIXME(#7544): Optimize out the xadd in this clone, somehow.
 659                  let handle = me.death.kill_handle.get_ref().clone();
 660                  // Atomically try to get into all of our taskgroups.
 661                  if enlist_many(NewTask(handle), &child_tg, &mut ancestors) {
 662                      // Got in. We can run the provided child body, and can also run
 663                      // the taskgroup's exit-time-destructor afterward.
 664                      me.taskgroup = Some(Taskgroup(child_tg, ancestors, is_main, None));
 665                      true
 666                  } else {
 667                      false
 668                  }
 669              }
 670          };
 671          // Should be run after the local-borrowed task is returned.
 672          if enlist_success {
 673              if indestructible {
 674                  do unkillable { f() }
 675              } else {
 676                  f()
 677              }
 678          }
 679      };
 680  
 681      let mut task = unsafe {
 682          if opts.sched.mode != SingleThreaded {
 683              if opts.watched {
 684                  Task::build_child(opts.stack_size, child_wrapper)
 685              } else {
 686                  Task::build_root(opts.stack_size, child_wrapper)
 687              }
 688          } else {
 689              // Creating a 1:1 task:thread ...
 690              let sched = Local::unsafe_borrow::<Scheduler>();
 691              let sched_handle = (*sched).make_handle();
 692  
 693              // Since this is a 1:1 scheduler we create a queue not in
 694              // the stealee set. The run_anything flag is set false
 695              // which will disable stealing.
 696              let work_queue = WorkQueue::new();
 697  
 698              // Create a new scheduler to hold the new task
 699              let new_loop = ~UvEventLoop::new();
 700              let mut new_sched = ~Scheduler::new_special(new_loop,
 701                                                          work_queue,
 702                                                          (*sched).work_queues.clone(),
 703                                                          (*sched).sleeper_list.clone(),
 704                                                          false,
 705                                                          Some(sched_handle));
 706              let mut new_sched_handle = new_sched.make_handle();
 707  
 708              // Allow the scheduler to exit when the pinned task exits
 709              new_sched_handle.send(Shutdown);
 710  
 711              // Pin the new task to the new scheduler
 712              let new_task = if opts.watched {
 713                  Task::build_homed_child(opts.stack_size, child_wrapper, Sched(new_sched_handle))
 714              } else {
 715                  Task::build_homed_root(opts.stack_size, child_wrapper, Sched(new_sched_handle))
 716              };
 717  
 718              // Create a task that will later be used to join with the new scheduler
 719              // thread when it is ready to terminate
 720              let (thread_port, thread_chan) = oneshot();
 721              let thread_port_cell = Cell::new(thread_port);
 722              let join_task = do Task::build_child(None) {
 723                  rtdebug!("running join task");
 724                  let thread_port = thread_port_cell.take();
 725                  let thread: Thread = thread_port.recv();
 726                  thread.join();
 727              };
 728  
 729              // Put the scheduler into another thread
 730              let new_sched_cell = Cell::new(new_sched);
 731              let orig_sched_handle_cell = Cell::new((*sched).make_handle());
 732              let join_task_cell = Cell::new(join_task);
 733  
 734              let thread = do Thread::start {
 735                  let mut new_sched = new_sched_cell.take();
 736                  let mut orig_sched_handle = orig_sched_handle_cell.take();
 737                  let join_task = join_task_cell.take();
 738  
 739                  let bootstrap_task = ~do Task::new_root(&mut new_sched.stack_pool, None) || {
 740                      rtdebug!("boostrapping a 1:1 scheduler");
 741                  };
 742                  new_sched.bootstrap(bootstrap_task);
 743  
 744                  rtdebug!("enqueing join_task");
 745                  // Now tell the original scheduler to join with this thread
 746                  // by scheduling a thread-joining task on the original scheduler
 747                  orig_sched_handle.send(TaskFromFriend(join_task));
 748  
 749                  // NB: We can't simply send a message from here to another task
 750                  // because this code isn't running in a task and message passing doesn't
 751                  // work outside of tasks. Hence we're sending a scheduler message
 752                  // to execute a new task directly to a scheduler.
 753              };
 754  
 755              // Give the thread handle to the join task
 756              thread_chan.send(thread);
 757  
 758              // When this task is enqueued on the current scheduler it will then get
 759              // forwarded to the scheduler to which it is pinned
 760              new_task
 761          }
 762      };
 763  
 764      if opts.notify_chan.is_some() {
 765          let notify_chan = opts.notify_chan.take_unwrap();
 766          let notify_chan = Cell::new(notify_chan);
 767          let on_exit: ~fn(bool) = |success| {
 768              notify_chan.take().send(
 769                  if success { Success } else { Failure }
 770              )
 771          };
 772          task.death.on_exit = Some(on_exit);
 773      }
 774  
 775      task.name = opts.name.take();
 776      rtdebug!("spawn calling run_task");
 777      Scheduler::run_task(task);
 778  
 779  }

   11 /*!
   12  * Task management.
   13  *
   14  * An executing Rust program consists of a tree of tasks, each with their own
   15  * stack, and sole ownership of their allocated heap data. Tasks communicate
   16  * with each other using ports and channels.
   17  *
   18  * When a task fails, that failure will propagate to its parent (the task
   19  * that spawned it) and the parent will fail as well. The reverse is not
   20  * true: when a parent task fails its children will continue executing. When
   21  * the root (main) task fails, all tasks fail, and then so does the entire
   22  * process.
   23  *
   24  * Tasks may execute in parallel and are scheduled automatically by the
   25  * runtime.
   26  *
   27  * # Example
   28  *
   29  * ~~~
   30  * do spawn {
   31  *     log(error, "Hello, World!");
   32  * }
   33  * ~~~
   34  */

   59 /**
   60  * Indicates the manner in which a task exited.
   61  *
   62  * A task that completes without failing is considered to exit successfully.
   63  * Supervised ancestors and linked siblings may yet fail after this task
   64  * succeeds. Also note that in such a case, it may be nondeterministic whether
   65  * linked failure or successful exit happen first.
   66  *
   67  * If you wish for this result's delivery to block until all linked and/or
   68  * children tasks complete, recommend using a result future.
   69  */

   97 /**
   98  * Task configuration options
   99  *
  100  * # Fields
  101  *
  102  * * linked - Propagate failure bidirectionally between child and parent.
  103  *            True by default. If both this and 'supervised' are false, then
  104  *            either task's failure will not affect the other ("unlinked").
  105  *
  106  * * supervised - Propagate failure unidirectionally from parent to child,
  107  *                but not from child to parent. False by default.
  108  *
  109  * * watched - Make parent task collect exit status notifications from child
  110  *             before reporting its own exit status. (This delays the parent
  111  *             task's death and cleanup until after all transitively watched
  112  *             children also exit.) True by default.
  113  *
  114  * * indestructible - Configures the task to ignore kill signals received from
  115  *                    linked failure. This may cause process hangs during
  116  *                    failure if not used carefully, but causes task blocking
  117  *                    code paths (e.g. port recv() calls) to be faster by 2
  118  *                    atomic operations. False by default.
  119  *
  120  * * notify_chan - Enable lifecycle notifications on the given channel
  121  *
  122  * * name - A name for the task-to-be, for identification in failure messages.
  123  *
  124  * * sched - Specify the configuration of a new scheduler to create the task
  125  *           in
  126  *
  127  *     By default, every task is created in the same scheduler as its
  128  *     parent, where it is scheduled cooperatively with all other tasks
  129  *     in that scheduler. Some specialized applications may want more
  130  *     control over their scheduling, in which case they can be spawned
  131  *     into a new scheduler with the specific properties required.
  132  *
  133  *     This is of particular importance for libraries which want to call
  134  *     into foreign code that blocks. Without doing so in a different
  135  *     scheduler other tasks will be impeded or even blocked indefinitely.
  136  */
  137 pub struct TaskOpts {
  138     linked: bool,
  139     supervised: bool,
  140     watched: bool,
  141     indestructible: bool,
  142     notify_chan: Option<Chan<TaskResult>>,
  143     name: Option<~str>,
  144     sched: SchedOpts,
  145     stack_size: Option<uint>
  146 }

  252     /**
  253      * Get a future representing the exit status of the task.
  254      *
  255      * Taking the value of the future will block until the child task
  256      * terminates. The future-receiving callback specified will be called
  257      * *before* the task is spawned; as such, do not invoke .get() within the
  258      * closure; rather, store it in an outer variable/list for later use.
  259      *
  260      * Note that the future returning by this function is only useful for
  261      * obtaining the value of the next task to be spawning with the
  262      * builder. If additional tasks are spawned with the same builder
  263      * then a new result future must be obtained prior to spawning each
  264      * task.
  265      *
  266      * # Failure
  267      * Fails if a future_result was already set for this task.
  268      */
  269     pub fn future_result(&mut self, blk: &fn(v: Port<TaskResult>)) {
  270         // FIXME (#3725): Once linked failure and notification are
  271         // handled in the library, I can imagine implementing this by just
  272         // registering an arbitrary number of task::on_exit handlers and
  273         // sending out messages.
  274 
  275         if self.opts.notify_chan.is_some() {
  276             fail!("Can't set multiple future_results for one task!");
  277         }
  278 
  279         // Construct the future and give it to the caller.
  280         let (notify_pipe_po, notify_pipe_ch) = stream::<TaskResult>();
  281 
  282         blk(notify_pipe_po);
  283 
  284         // Reconfigure self to use a notify channel.
  285         self.opts.notify_chan = Some(notify_pipe_ch);
  286     }

  299     /**
  300      * Add a wrapper to the body of the spawned task.
  301      *
  302      * Before the task is spawned it is passed through a 'body generator'
  303      * function that may perform local setup operations as well as wrap
  304      * the task body in remote setup operations. With this the behavior
  305      * of tasks can be extended in simple ways.
  306      *
  307      * This function augments the current body generator with a new body
  308      * generator by applying the task body which results from the
  309      * existing body generator to the new body generator.
  310      */
  311     pub fn add_wrapper(&mut self, wrapper: ~fn(v: ~fn()) -> ~fn()) {
  312         let prev_gen_body = self.gen_body.take();
  313         let prev_gen_body = match prev_gen_body {
  314             Some(gen) => gen,
  315             None => {
  316                 let f: ~fn(~fn()) -> ~fn() = |body| body;
  317                 f
  318             }
  319         };
  320         let prev_gen_body = Cell::new(prev_gen_body);
  321         let next_gen_body = {
  322             let f: ~fn(~fn()) -> ~fn() = |body| {
  323                 let prev_gen_body = prev_gen_body.take();
  324                 wrapper(prev_gen_body(body))
  325             };
  326             f
  327         };
  328         self.gen_body = Some(next_gen_body);
  329     }

  331     /**
  332      * Creates and executes a new child task
  333      *
  334      * Sets up a new task with its own call stack and schedules it to run
  335      * the provided unique closure. The task has the properties and behavior
  336      * specified by the task_builder.
  337      *
  338      * # Failure
  339      *
  340      * When spawning into a new scheduler, the number of threads requested
  341      * must be greater than zero.
  342      */
  343     pub fn spawn(&mut self, f: ~fn()) {
  344         let gen_body = self.gen_body.take();
  345         let notify_chan = self.opts.notify_chan.take();
  346         let name = self.opts.name.take();
  347         let x = self.consume();
  348         let opts = TaskOpts {
  349             linked: x.opts.linked,
  350             supervised: x.opts.supervised,
  351             watched: x.opts.watched,
  352             indestructible: x.opts.indestructible,
  353             notify_chan: notify_chan,
  354             name: name,
  355             sched: x.opts.sched,
  356             stack_size: x.opts.stack_size
  357         };
  358         let f = match gen_body {
  359             Some(gen) => {
  360                 gen(f)
  361             }
  362             None => {
  363                 f
  364             }
  365         };
  366         spawn::spawn_raw(opts, f);
  367     }

  377     /**
  378      * Execute a function in another task and return either the return value
  379      * of the function or result::err.
  380      *
  381      * # Return value
  382      *
  383      * If the function executed successfully then try returns result::ok
  384      * containing the value returned by the function. If the function fails
  385      * then try returns result::err containing nil.
  386      *
  387      * # Failure
  388      * Fails if a future_result was already set for this task.
  389      */
  390     pub fn try<T:Send>(&mut self, f: ~fn() -> T) -> Result<T,()> {
  391         let (po, ch) = stream::<T>();
  392         let mut result = None;
  393 
  394         self.future_result(|r| { result = Some(r); });
  395 
  396         do self.spawn {
  397             ch.send(f());
  398         }
  399 
  400         match result.unwrap().recv() {
  401             Success => result::Ok(po.recv()),
  402             Failure => result::Err(())
  403         }
  404     }
  405 }

  555 pub fn failing() -> bool {
  556     //! True if the running task has failed
  557 
  558     use rt::task::Task;
  559

  561         local.unwinder.unwinding
  562     }
  563 }

  565 /**
  566  * Temporarily make the task unkillable
  567  *
  568  * # Example
  569  *
  570  * ~~~
  571  * do task::unkillable {
  572  *     // detach / yield / destroy must all be called together
  573  *     rustrt::rust_port_detach(po);
  574  *     // This must not result in the current task being killed
  575  *     task::yield();
  576  *     rustrt::rust_port_destroy(po);
  577  * }
  578  * ~~~
  579  */
  580 pub fn unkillable<U>(f: &fn() -> U) -> U {
  581     use rt::task::Task;
  582 
  583     unsafe {
  584         if in_green_task_context() {
  585             // The inhibits/allows might fail and need to borrow the task.
  586             let t = Local::unsafe_borrow::<Task>();
  587             do (|| {
  588                 (*t).death.inhibit_kill((*t).unwinder.unwinding);
  589                 f()
  590             }).finally {
  591                 (*t).death.allow_kill((*t).unwinder.unwinding);
  592             }
  593         } else {
  594             // FIXME(#3095): This should be an rtabort as soon as the scheduler
  595             // no longer uses a workqueue implemented with an Exclusive.
  596             f()
  597         }
  598     }
  599 }

  601 /// The inverse of unkillable. Only ever to be used nested in unkillable().
  602 pub unsafe fn rekillable<U>(f: &fn() -> U) -> U {
  603     use rt::task::Task;
  604 
  605     if in_green_task_context() {
  606         let t = Local::unsafe_borrow::<Task>();
  607         do (|| {
  608             (*t).death.allow_kill((*t).unwinder.unwinding);
  609             f()
  610         }).finally {
  611             (*t).death.inhibit_kill((*t).unwinder.unwinding);
  612         }
  613     } else {
  614         // FIXME(#3095): As in unkillable().
  615         f()
  616     }
  617 }

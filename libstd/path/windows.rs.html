   48  /// Represents a Windows path
   49  // Notes for Windows path impl:
   50  // The MAX_PATH is 260, but 253 is the practical limit due to some API bugs
   51  // See http://msdn.microsoft.com/en-us/library/windows/desktop/aa365247.aspx for good information
   52  // about windows paths.
   53  // That same page puts a bunch of restrictions on allowed characters in a path.
   54  // `\foo.txt` means "relative to current drive", but will not be considered to be absolute here
   55  // as `âP | P.join("\foo.txt") != "\foo.txt"`.
   56  // `C:` is interesting, that means "the current directory on drive C".
   57  // Long absolute paths need to have \\?\ prefix (or, for UNC, \\?\UNC\). I think that can be
   58  // ignored for now, though, and only added in a hypothetical .to_pwstr() function.
   59  // However, if a path is parsed that has \\?\, this needs to be preserved as it disables the
   60  // processing of "." and ".." components and / as a separator.
   61  // Experimentally, \\?\foo is not the same thing as \foo.
   62  // Also, \\foo is not valid either (certainly not equivalent to \foo).
   63  // Similarly, C:\\Users is not equivalent to C:\Users, although C:\Users\\foo is equivalent
   64  // to C:\Users\foo. In fact the command prompt treats C:\\foo\bar as UNC path. But it might be
   65  // best to just ignore that and normalize it to C:\foo\bar.
   66  //
   67  // Based on all this, I think the right approach is to do the following:
   68  // * Require valid utf-8 paths. Windows API may use WCHARs, but we don't, and utf-8 is convertible
   69  // to UTF-16 anyway (though does Windows use UTF-16 or UCS-2? Not sure).
   70  // * Parse the prefixes \\?\UNC\, \\?\, and \\.\ explicitly.
   71  // * If \\?\UNC\, treat following two path components as server\share. Don't error for missing
   72  //   server\share.
   73  // * If \\?\, parse disk from following component, if present. Don't error for missing disk.
   74  // * If \\.\, treat rest of path as just regular components. I don't know how . and .. are handled
   75  //   here, they probably aren't, but I'm not going to worry about that.
   76  // * Else if starts with \\, treat following two components as server\share. Don't error for missing
   77  //   server\share.
   78  // * Otherwise, attempt to parse drive from start of path.
   79  //
   80  // The only error condition imposed here is valid utf-8. All other invalid paths are simply
   81  // preserved by the data structure; let the Windows API error out on them.
   82  #[deriving(Clone)]
   83  pub struct Path {
   84      priv repr: ~str, // assumed to never be empty
   85      priv prefix: Option<PathPrefix>,
   86      priv sepidx: Option<uint> // index of the final separator in the non-prefix portion of repr
   87  }

  169      /// See `GenericPathUnsafe::set_filename_unchecekd`.
  170      ///
  171      /// # Failure
  172      ///
  173      /// Fails if not valid UTF-8.
  174      unsafe fn set_filename_unchecked<T: BytesContainer>(&mut self, filename: T) {
  175          let filename = filename.container_as_str().unwrap();
  176          match self.sepidx_or_prefix_len() {
  177              None if ".." == self.repr => {
  178                  let mut s = str::with_capacity(3 + filename.len());
  179                  s.push_str("..");
  180                  s.push_char(SEP);
  181                  s.push_str(filename);
  182                  self.update_normalized(s);
  183              }
  184              None => {
  185                  self.update_normalized(filename);
  186              }
  187              Some((_,idxa,end)) if self.repr.slice(idxa,end) == ".." => {
  188                  let mut s = str::with_capacity(end + 1 + filename.len());
  189                  s.push_str(self.repr.slice_to(end));
  190                  s.push_char(SEP);
  191                  s.push_str(filename);
  192                  self.update_normalized(s);
  193              }
  194              Some((idxb,idxa,_)) if self.prefix == Some(DiskPrefix) && idxa == self.prefix_len() => {
  195                  let mut s = str::with_capacity(idxb + filename.len());
  196                  s.push_str(self.repr.slice_to(idxb));
  197                  s.push_str(filename);
  198                  self.update_normalized(s);
  199              }
  200              Some((idxb,_,_)) => {
  201                  let mut s = str::with_capacity(idxb + 1 + filename.len());
  202                  s.push_str(self.repr.slice_to(idxb));
  203                  s.push_char(SEP);
  204                  s.push_str(filename);
  205                  self.update_normalized(s);
  206              }
  207          }
  208      }

  210      /// See `GenericPathUnsafe::push_unchecked`.
  211      ///
  212      /// Concatenating two Windows Paths is rather complicated.
  213      /// For the most part, it will behave as expected, except in the case of
  214      /// pushing a volume-relative path, e.g. `C:foo.txt`. Because we have no
  215      /// concept of per-volume cwds like Windows does, we can't behave exactly
  216      /// like Windows will. Instead, if the receiver is an absolute path on
  217      /// the same volume as the new path, it will be treated as the cwd that
  218      /// the new path is relative to. Otherwise, the new path will be treated
  219      /// as if it were absolute and will replace the receiver outright.
  220      unsafe fn push_unchecked<T: BytesContainer>(&mut self, path: T) {
  221          let path = path.container_as_str().unwrap();
  222          fn is_vol_abs(path: &str, prefix: Option<PathPrefix>) -> bool {
  223              // assume prefix is Some(DiskPrefix)
  224              let rest = path.slice_from(prefix_len(prefix));
  225              !rest.is_empty() && rest[0].is_ascii() && is_sep(rest[0] as char)
  226          }
  227          fn shares_volume(me: &Path, path: &str) -> bool {
  228              // path is assumed to have a prefix of Some(DiskPrefix)
  229              match me.prefix {
  230                  Some(DiskPrefix) => me.repr[0] == path[0].to_ascii().to_upper().to_byte(),
  231                  Some(VerbatimDiskPrefix) => me.repr[4] == path[0].to_ascii().to_upper().to_byte(),
  232                  _ => false
  233              }
  234          }
  235          fn is_sep_(prefix: Option<PathPrefix>, u: u8) -> bool {
  236              if prefix_is_verbatim(prefix) { is_sep_verbatim(u as char) }
  237              else { is_sep(u as char) }
  238          }
  239  
  240          fn replace_path(me: &mut Path, path: &str, prefix: Option<PathPrefix>) {
  241              let newpath = Path::normalize__(path, prefix);
  242              me.repr = match newpath {
  243                  Some(p) => p,
  244                  None => path.to_owned()
  245              };
  246              me.prefix = prefix;
  247              me.update_sepidx();
  248          }
  249          fn append_path(me: &mut Path, path: &str) {
  250              // appends a path that has no prefix
  251              // if me is verbatim, we need to pre-normalize the new path
  252              let path_ = if is_verbatim(me) { Path::normalize__(path, None) }
  253                          else { None };
  254              let pathlen = path_.as_ref().map_or(path.len(), |p| p.len());
  255              let mut s = str::with_capacity(me.repr.len() + 1 + pathlen);
  256              s.push_str(me.repr);
  257              let plen = me.prefix_len();
  258              // if me is "C:" we don't want to add a path separator
  259              match me.prefix {
  260                  Some(DiskPrefix) if me.repr.len() == plen => (),
  261                  _ if !(me.repr.len() > plen && me.repr[me.repr.len()-1] == SEP_BYTE) => {
  262                      s.push_char(SEP);
  263                  }
  264                  _ => ()
  265              }
  266              match path_ {
  267                  None => s.push_str(path),
  268                  Some(p) => s.push_str(p)
  269              };
  270              me.update_normalized(s)
  271          }
  272  
  273          if !path.is_empty() {
  274              let prefix = parse_prefix(path);
  275              match prefix {
  276                  Some(DiskPrefix) if !is_vol_abs(path, prefix) && shares_volume(self, path) => {
  277                      // cwd-relative path, self is on the same volume
  278                      append_path(self, path.slice_from(prefix_len(prefix)));
  279                  }
  280                  Some(_) => {
  281                      // absolute path, or cwd-relative and self is not same volume
  282                      replace_path(self, path, prefix);
  283                  }
  284                  None if !path.is_empty() && is_sep_(self.prefix, path[0]) => {
  285                      // volume-relative path
  286                      if self.prefix.is_some() {
  287                          // truncate self down to the prefix, then append
  288                          let n = self.prefix_len();
  289                          self.repr.truncate(n);
  290                          append_path(self, path);
  291                      } else {
  292                          // we have no prefix, so nothing to be relative to
  293                          replace_path(self, path, prefix);
  294                      }
  295                  }
  296                  None => {
  297                      // relative path
  298                      append_path(self, path);
  299                  }
  300              }
  301          }
  302      }

  963  fn parse_prefix<'a>(mut path: &'a str) -> Option<PathPrefix> {
  964      if path.starts_with("\\\\") {
  965          // \\
  966          path = path.slice_from(2);
  967          if path.starts_with("?\\") {
  968              // \\?\
  969              path = path.slice_from(2);
  970              if path.starts_with("UNC\\") {
  971                  // \\?\UNC\server\share
  972                  path = path.slice_from(4);
  973                  let (idx_a, idx_b) = match parse_two_comps(path, is_sep_verbatim) {
  974                      Some(x) => x,
  975                      None => (path.len(), 0)
  976                  };
  977                  return Some(VerbatimUNCPrefix(idx_a, idx_b));
  978              } else {
  979                  // \\?\path
  980                  let idx = path.find('\\');
  981                  if idx == Some(2) && path[1] == ':' as u8 {
  982                      let c = path[0];
  983                      if c.is_ascii() && ::char::is_alphabetic(c as char) {
  984                          // \\?\C:\ path
  985                          return Some(VerbatimDiskPrefix);
  986                      }
  987                  }
  988                  let idx = idx.unwrap_or(path.len());
  989                  return Some(VerbatimPrefix(idx));
  990              }
  991          } else if path.starts_with(".\\") {
  992              // \\.\path
  993              path = path.slice_from(2);
  994              let idx = path.find('\\').unwrap_or(path.len());
  995              return Some(DeviceNSPrefix(idx));
  996          }
  997          match parse_two_comps(path, is_sep) {
  998              Some((idx_a, idx_b)) if idx_a > 0 && idx_b > 0 => {
  999                  // \\server\share
 1000                  return Some(UNCPrefix(idx_a, idx_b));
 1001              }
 1002              _ => ()
 1003          }
 1004      } else if path.len() > 1 && path[1] == ':' as u8 {
 1005          // C:
 1006          let c = path[0];
 1007          if c.is_ascii() && ::char::is_alphabetic(c as char) {
 1008              return Some(DiskPrefix);
 1009          }
 1010      }
 1011      return None;

 1026  fn normalize_helper<'a>(s: &'a str, prefix: Option<PathPrefix>) -> (bool,Option<~[&'a str]>) {
 1027      let f = if !prefix_is_verbatim(prefix) { is_sep } else { is_sep_verbatim };
 1028      let is_abs = s.len() > prefix_len(prefix) && f(s.char_at(prefix_len(prefix)));
 1029      let s_ = s.slice_from(prefix_len(prefix));
 1030      let s_ = if is_abs { s_.slice_from(1) } else { s_ };
 1031  
 1032      if is_abs && s_.is_empty() {
 1033          return (is_abs, match prefix {
 1034              Some(DiskPrefix) | None => (if is_sep_verbatim(s.char_at(prefix_len(prefix))) { None }
 1035                                          else { Some(~[]) }),
 1036              Some(_) => Some(~[]), // need to trim the trailing separator
 1037          });
 1038      }
 1039      let mut comps: ~[&'a str] = ~[];
 1040      let mut n_up = 0u;
 1041      let mut changed = false;
 1042      for comp in s_.split(f) {
 1043          if comp.is_empty() { changed = true }
 1044          else if comp == "." { changed = true }
 1045          else if comp == ".." {
 1046              let has_abs_prefix = match prefix {
 1047                  Some(DiskPrefix) => false,
 1048                  Some(_) => true,
 1049                  None => false
 1050              };
 1051              if (is_abs || has_abs_prefix) && comps.is_empty() { changed = true }
 1052              else if comps.len() == n_up { comps.push(".."); n_up += 1 }
 1053              else { comps.pop().unwrap(); changed = true }
 1054          } else { comps.push(comp) }
 1055      }
 1056      if !changed && !prefix_is_verbatim(prefix) {
 1057          changed = s.find(is_sep).is_some();
 1058      }
 1059      if changed {
 1060          if comps.is_empty() && !is_abs && prefix.is_none() {
 1061              if s == "." {
 1062                  return (is_abs, None);
 1063              }
 1064              comps.push(".");
 1065          }
 1066          (is_abs, Some(comps))
 1067      } else {
 1068          (is_abs, None)
 1069      }
 1070  }

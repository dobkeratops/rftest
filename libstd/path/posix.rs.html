  126      unsafe fn set_filename_unchecked<T: BytesContainer>(&mut self, filename: T) {
  127          let filename = filename.container_as_bytes();
  128          match self.sepidx {
  129              None if bytes!("..") == self.repr => {
  130                  let mut v = slice::with_capacity(3 + filename.len());
  131                  v.push_all(dot_dot_static);
  132                  v.push(SEP_BYTE);
  133                  v.push_all(filename);
  134                  self.repr = Path::normalize(v);
  135              }
  136              None => {
  137                  self.repr = Path::normalize(filename);
  138              }
  139              Some(idx) if self.repr.slice_from(idx+1) == bytes!("..") => {
  140                  let mut v = slice::with_capacity(self.repr.len() + 1 + filename.len());
  141                  v.push_all(self.repr);
  142                  v.push(SEP_BYTE);
  143                  v.push_all(filename);
  144                  self.repr = Path::normalize(v);
  145              }
  146              Some(idx) => {
  147                  let mut v = slice::with_capacity(idx + 1 + filename.len());
  148                  v.push_all(self.repr.slice_to(idx+1));
  149                  v.push_all(filename);
  150                  self.repr = Path::normalize(v);
  151              }
  152          }

  237      fn is_ancestor_of(&self, other: &Path) -> bool {
  238          if self.is_absolute() != other.is_absolute() {
  239              false
  240          } else {
  241              let mut ita = self.components();
  242              let mut itb = other.components();
  243              if bytes!(".") == self.repr {
  244                  return match itb.next() {
  245                      None => true,
  246                      Some(b) => b != bytes!("..")
  247                  };
  248              }
  249              loop {
  250                  match (ita.next(), itb.next()) {
  251                      (None, _) => break,
  252                      (Some(a), Some(b)) if a == b => { continue },
  253                      (Some(a), _) if a == bytes!("..") => {
  254                          // if ita contains only .. components, it's an ancestor
  255                          return ita.all(|x| x == bytes!(".."));
  256                      }
  257                      _ => return false
  258                  }
  259              }
  260              true
  261          }
  262      }

  264      fn path_relative_from(&self, base: &Path) -> Option<Path> {
  265          if self.is_absolute() != base.is_absolute() {
  266              if self.is_absolute() {
  267                  Some(self.clone())
  268              } else {
  269                  None
  270              }
  271          } else {
  272              let mut ita = self.components();
  273              let mut itb = base.components();
  274              let mut comps = ~[];
  275              loop {
  276                  match (ita.next(), itb.next()) {
  277                      (None, None) => break,
  278                      (Some(a), None) => {
  279                          comps.push(a);
  280                          comps.extend(ita.by_ref());
  281                          break;
  282                      }
  283                      (None, _) => comps.push(dot_dot_static),
  284                      (Some(a), Some(b)) if comps.is_empty() && a == b => (),
  285                      (Some(a), Some(b)) if b == bytes!(".") => comps.push(a),
  286                      (Some(_), Some(b)) if b == bytes!("..") => return None,
  287                      (Some(a), Some(_)) => {
  288                          comps.push(dot_dot_static);
  289                          for _ in itb {
  290                              comps.push(dot_dot_static);
  291                          }
  292                          comps.push(a);
  293                          comps.extend(ita.by_ref());
  294                          break;
  295                      }
  296                  }
  297              }
  298              Some(Path::new(comps.connect_vec(&SEP_BYTE)))
  299          }
  300      }

  335      /// Returns a normalized byte vector representation of a path, by removing all empty
  336      /// components, and unnecessary . and .. components.
  337      fn normalize<V: Vector<u8>+CloneableVector<u8>>(v: V) -> ~[u8] {
  338          // borrowck is being very picky
  339          let val = {
  340              let is_abs = !v.as_slice().is_empty() && v.as_slice()[0] == SEP_BYTE;
  341              let v_ = if is_abs { v.as_slice().slice_from(1) } else { v.as_slice() };
  342              let comps = normalize_helper(v_, is_abs);
  343              match comps {
  344                  None => None,
  345                  Some(comps) => {
  346                      if is_abs && comps.is_empty() {
  347                          Some(~[SEP_BYTE])
  348                      } else {
  349                          let n = if is_abs { comps.len() } else { comps.len() - 1} +
  350                                  comps.iter().map(|v| v.len()).sum();
  351                          let mut v = slice::with_capacity(n);
  352                          let mut it = comps.move_iter();
  353                          if !is_abs {
  354                              match it.next() {
  355                                  None => (),
  356                                  Some(comp) => v.push_all(comp)
  357                              }
  358                          }
  359                          for comp in it {
  360                              v.push(SEP_BYTE);
  361                              v.push_all(comp);
  362                          }
  363                          Some(v)
  364                      }
  365                  }
  366              }
  367          };
  368          match val {
  369              None => v.into_owned(),
  370              Some(val) => val
  371          }
  372      }

  417  // None result means the byte vector didn't need normalizing
  418  fn normalize_helper<'a>(v: &'a [u8], is_abs: bool) -> Option<~[&'a [u8]]> {
  419      if is_abs && v.as_slice().is_empty() {
  420          return None;
  421      }
  422      let mut comps: ~[&'a [u8]] = ~[];
  423      let mut n_up = 0u;
  424      let mut changed = false;
  425      for comp in v.split(is_sep_byte) {
  426          if comp.is_empty() { changed = true }
  427          else if comp == bytes!(".") { changed = true }

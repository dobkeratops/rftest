  11 /*!
  12 
  13 Operations on the ubiquitous `Option` type.
  14 
  15 Type `Option` represents an optional value.
  16 
  17 Every `Option<T>` value can either be `Some(T)` or `None`. Where in other
  18 languages you might use a nullable type, in Rust you would use an option
  19 type.
  20 
  21 Options are most commonly used with pattern matching to query the presence
  22 of a value and take action, always accounting for the `None` case.
  23 
  24 # Example
  25 
  26 ~~~
  27 let msg = Some(~"howdy");
  28 
  29 // Take a reference to the contained string
  30 match msg {
  31     Some(ref m) => io::println(m),
  32     None => ()
  33 }
  34 
  35 // Remove the contained string, destroying the Option
  36 let unwrapped_msg = match msg {
  37     Some(m) => m,
  38     None => ~"default message"
  39 };
  40 ~~~
  41 
  42 */

  53 /// The option type
  54 #[deriving(Clone, DeepClone, Eq)]
  55 pub enum Option<T> {
  56     None,
  57     Some(T),
  58 }

  60 impl<T:Ord> Ord for Option<T> {
  61     fn lt(&self, other: &Option<T>) -> bool {
  62         match (self, other) {
  63             (&None, &None) => false,
  64             (&None, &Some(_)) => true,
  65             (&Some(_), &None) => false,
  66             (&Some(ref a), &Some(ref b)) => *a < *b
  67         }
  68     }
  69 
  70     fn le(&self, other: &Option<T>) -> bool {
  71         match (self, other) {
  72             (&None, &None) => true,
  73             (&None, &Some(_)) => true,
  74             (&Some(_), &None) => false,
  75             (&Some(ref a), &Some(ref b)) => *a <= *b
  76         }
  77     }
  78 
  79     fn ge(&self, other: &Option<T>) -> bool {
  80         !(self < other)
  81     }
  82 
  83     fn gt(&self, other: &Option<T>) -> bool {
  84         !(self <= other)
  85     }
  86 }
  87 
  88 impl<T:Clone+Add<T,T>> Add<Option<T>, Option<T>> for Option<T> {
  89     #[inline]
  90     fn add(&self, other: &Option<T>) -> Option<T> {
  91         match (&*self, &*other) {
  92             (&None, &None) => None,
  93             (_, &None) => (*self).clone(),
  94             (&None, _) => (*other).clone(),
  95             (&Some(ref lhs), &Some(ref rhs)) => Some(*lhs + *rhs)
  96         }
  97     }
  98 }

 119     /// Returns true if the option equals `None`
 120     #[inline]
 121     pub fn is_none(&self) -> bool {
 122         match *self { None => true, Some(_) => false }
 123     }

 125     /// Returns true if the option contains a `Some` value
 126     #[inline]
 127     pub fn is_some(&self) -> bool { !self.is_none() }

 129     /// Update an optional value by optionally running its content through a
 130     /// function that returns an option.
 131     #[inline]
 132     pub fn chain<U>(self, f: &fn(t: T) -> Option<U>) -> Option<U> {
 133         match self {
 134             Some(t) => f(t),
 135             None => None
 136         }
 137     }

 139     /// Returns the leftmost Some() value, or None if both are None.

 142         match self {
 143             Some(opta) => Some(opta),
 144             _ => optb
 145         }
 146     }

 154             Some(ref x) => f(x),

 170     /// Filters an optional value using given function.
 171     #[inline(always)]
 172     pub fn filtered(self, f: &fn(t: &T) -> bool) -> Option<T> {
 173         match self {
 174             Some(x) => if(f(&x)) {Some(x)} else {None},
 175             None => None
 176         }
 177     }

 179     /// Maps a `Some` value from one type to another by reference
 180     #[inline]
 181     pub fn map<'a, U>(&'a self, f: &fn(&'a T) -> U) -> Option<U> {
 182         match *self { Some(ref x) => Some(f(x)), None => None }
 183     }

 185     /// Maps a `Some` value from one type to another by a mutable reference
 186     #[inline]
 187     pub fn map_mut<'a, U>(&'a mut self, f: &fn(&'a mut T) -> U) -> Option<U> {
 188         match *self { Some(ref mut x) => Some(f(x)), None => None }
 189     }

 191     /// Maps a `Some` value from one type to another by a mutable reference,
 192     /// or returns a default value.
 193     #[inline]
 194     pub fn map_mut_default<'a, U>(&'a mut self, def: U, f: &fn(&'a mut T) -> U) -> U {
 195         match *self { Some(ref mut x) => f(x), None => def }
 196     }
 197 
 198     /// As `map`, but consumes the option and gives `f` ownership to avoid
 199     /// copying.
 200     #[inline]
 201     pub fn map_consume<U>(self, f: &fn(v: T) -> U) -> Option<U> {
 202         match self { None => None, Some(v) => Some(f(v)) }
 203     }

 205     /// Applies a function to the contained value or returns a default
 206     #[inline]
 207     pub fn map_default<'a, U>(&'a self, def: U, f: &fn(&'a T) -> U) -> U {
 208         match *self { None => def, Some(ref t) => f(t) }
 209     }
 210 
 211     /// As `map_default`, but consumes the option and gives `f`
 212     /// ownership to avoid copying.
 213     #[inline]
 214     pub fn map_consume_default<U>(self, def: U, f: &fn(v: T) -> U) -> U {
 215         match self { None => def, Some(v) => f(v) }
 216     }

 218     /// Take the value out of the option, leaving a `None` in its place.
 219     #[inline]
 220     pub fn take(&mut self) -> Option<T> {
 221         util::replace(self, None)
 222     }
 223 
 224     /// As `map_consume`, but swaps a None into the original option rather
 225     /// than consuming it by-value.
 226     #[inline]
 227     pub fn take_map<U>(&mut self, blk: &fn(T) -> U) -> Option<U> {
 228         self.take().map_consume(blk)
 229     }
 230 
 231     /// As `map_consume_default`, but swaps a None into the original option

 233     #[inline]
 234     pub fn take_map_default<U> (&mut self, def: U, blk: &fn(T) -> U) -> U {
 235         self.take().map_consume_default(def, blk)
 236     }

 238     /// Apply a function to the contained value or do nothing.
 239     /// Returns true if the contained value was mutated.
 240     pub fn mutate(&mut self, f: &fn(T) -> T) -> bool {
 241         if self.is_some() {
 242             *self = Some(f(self.take_unwrap()));
 243             true
 244         } else { false }
 245     }
 246 
 247     /// Apply a function to the contained value or set it to a default.
 248     /// Returns true if the contained value was mutated, or false if set to the default.
 249     pub fn mutate_default(&mut self, def: T, f: &fn(T) -> T) -> bool {
 250         if self.is_some() {
 251             *self = Some(f(self.take_unwrap()));
 252             true
 253         } else {
 254             *self = Some(def);
 255             false
 256         }
 257     }

 260     Gets an immutable reference to the value inside an option.
 261 
 262     # Failure
 263 
 264     Fails if the value equals `None`
 265 
 266     # Safety note

 279     }

 303     #[inline]
 304     pub fn unwrap(self) -> T {
 305         /*!
 306         Moves a value out of an option type and returns it.
 307 
 308         Useful primarily for getting strings, vectors and unique pointers out
 309         of option types without copying them.
 310 
 311         # Failure
 312 
 313         Fails if the value equals `None`.
 314 
 315         # Safety note
 316 
 317         In general, because this function may fail, its use is discouraged.
 318         Instead, prefer to use pattern matching and handle the `None`
 319         case explicitly.
 320          */
 321         match self {
 322           Some(x) => x,
 323           None => fail!("option::unwrap `None`"),
 324         }
 325     }
 326 
 327     /**
 328      * The option dance. Moves a value out of an option type and returns it,
 329      * replacing the original with `None`.
 330      *
 331      * # Failure
 332      *
 333      * Fails if the value equals `None`.
 334      */
 335     #[inline]
 336     pub fn take_unwrap(&mut self) -> T {
 337         if self.is_none() { fail!("option::take_unwrap `None`") }
 338         self.take().unwrap()
 339     }
 340 
 341     /**
 342      * Gets the value out of an option, printing a specified message on
 343      * failure
 344      *
 345      * # Failure
 346      *
 347      * Fails if the value equals `None`
 348      */
 349     #[inline]
 350     pub fn expect(self, reason: &str) -> T {
 351         match self {
 352           Some(val) => val,
 353           None => fail!(reason.to_owned()),
 354         }
 355     }
 356 
 357     /**
 358     Gets the value out of an option
 359 
 360     # Failure
 361 
 362     Fails if the value equals `None`
 363 
 364     # Safety note
 365 
 366     In general, because this function may fail, its use is discouraged
 367     (calling `get` on `None` is akin to dereferencing a null pointer).
 368     Instead, prefer to use pattern matching and handle the `None`
 369     case explicitly.
 370     */
 371     #[inline]
 372     pub fn get(self) -> T {
 373         match self {
 374           Some(x) => return x,
 375           None => fail!("option::get `None`")
 376         }
 377     }

 381     pub fn get_or_default(self, def: T) -> T {

 385     /// Applies a function zero or more times until the result is `None`.
 386     #[inline]
 387     pub fn while_some(self, blk: &fn(v: T) -> Option<T>) {
 388         let mut opt = self;
 389         while opt.is_some() {
 390             opt = blk(opt.unwrap());
 391         }
 392     }

   57 pub trait GenericPath {
   58     /// Converts a string to a Path
   59     fn from_str(&str) -> Self;
   60 
   61     /// Returns the directory component of `self`, as a string
   62     fn dirname(&self) -> ~str;
   63     /// Returns the file component of `self`, as a string option.
   64     /// Returns None if `self` names a directory.
   65     fn filename(&self) -> Option<~str>;
   66     /// Returns the stem of the file component of `self`, as a string option.
   67     /// The stem is the slice of a filename starting at 0 and ending just before
   68     /// the last '.' in the name.
   69     /// Returns None if `self` names a directory.
   70     fn filestem(&self) -> Option<~str>;
   71     /// Returns the type of the file component of `self`, as a string option.
   72     /// The file type is the slice of a filename starting just after the last
   73     /// '.' in the name and ending at the last index in the filename.
   74     /// Returns None if `self` names a directory.
   75     fn filetype(&self) -> Option<~str>;
   76 
   77     /// Returns a new path consisting of `self` with the parent directory component replaced
   78     /// with the given string.
   79     fn with_dirname(&self, (&str)) -> Self;
   80     /// Returns a new path consisting of `self` with the file component replaced
   81     /// with the given string.
   82     fn with_filename(&self, (&str)) -> Self;
   83     /// Returns a new path consisting of `self` with the file stem replaced
   84     /// with the given string.
   85     fn with_filestem(&self, (&str)) -> Self;
   86     /// Returns a new path consisting of `self` with the file type replaced
   87     /// with the given string.
   88     fn with_filetype(&self, (&str)) -> Self;
   89 
   90     /// Returns the directory component of `self`, as a new path.
   91     /// If `self` has no parent, returns `self`.
   92     fn dir_path(&self) -> Self;
   93     /// Returns the file component of `self`, as a new path.
   94     /// If `self` names a directory, returns the empty path.
   95     fn file_path(&self) -> Self;
   96 
   97     /// Returns a new Path whose parent directory is `self` and whose
   98     /// file component is the given string.
   99     fn push(&self, (&str)) -> Self;
  100     /// Returns a new Path consisting of the given path, made relative to `self`.
  101     fn push_rel(&self, (&Self)) -> Self;
  102     /// Returns a new Path consisting of the path given by the given vector
  103     /// of strings, relative to `self`.
  104     fn push_many<S: Str>(&self, (&[S])) -> Self;
  105     /// Identical to `dir_path` except in the case where `self` has only one
  106     /// component. In this case, `pop` returns the empty path.
  107     fn pop(&self) -> Self;
  108 
  109     /// The same as `push_rel`, except that the directory argument must not
  110     /// contain directory separators in any of its components.
  111     fn unsafe_join(&self, (&Self)) -> Self;
  112     /// On Unix, always returns false. On Windows, returns true iff `self`'s
  113     /// file stem is one of: `con` `aux` `com1` `com2` `com3` `com4`
  114     /// `lpt1` `lpt2` `lpt3` `prn` `nul`
  115     fn is_restricted(&self) -> bool;
  116 
  117     /// Returns a new path that names the same file as `self`, without containing
  118     /// any '.', '..', or empty components. On Windows, uppercases the drive letter
  119     /// as well.
  120     fn normalize(&self) -> Self;
  121 
  122     /// Returns `true` if `self` is an absolute path.
  123     fn is_absolute(&self) -> bool;
  124 
  125     /// True if `self` is an ancestor of `other`. See `test_is_ancestor_of` for examples
  126     fn is_ancestor_of(&self, (&Self)) -> bool;
  127 }

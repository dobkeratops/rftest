  22  fn combine(seek: SeekStyle, cur: uint, end: uint, offset: i64) -> IoResult<u64> {
  23      // compute offset as signed and clamp to prevent overflow
  24      let pos = match seek {
  25          SeekSet => 0,
  26          SeekEnd => end,
  27          SeekCur => cur,
  28      } as i64;
  29  
  30      if offset + pos < 0 {
  31          Err(IoError {
  32              kind: io::InvalidInput,
  33              desc: "invalid seek to a negative offset",
  34              detail: None
  35          })
  36      } else {
  37          Ok((offset + pos) as u64)
  38      }
  39  }

  81  impl Writer for MemWriter {
  82      fn write(&mut self, buf: &[u8]) -> IoResult<()> {
  83          // Make sure the internal buffer is as least as big as where we
  84          // currently are
  85          let difference = self.pos as i64 - self.buf.len() as i64;
  86          if difference > 0 {
  87              self.buf.grow(difference as uint, &0);
  88          }
  89  
  90          // Figure out what bytes will be used to overwrite what's currently
  91          // there (left), and what will be appended on the end (right)
  92          let cap = self.buf.len() - self.pos;
  93          let (left, right) = if cap <= buf.len() {
  94              (buf.slice_to(cap), buf.slice_from(cap))
  95          } else {
  96              (buf, &[])
  97          };
  98  
  99          // Do the necessary writes
 100          if left.len() > 0 {
 101              slice::bytes::copy_memory(self.buf.mut_slice_from(self.pos), left);
 102          }
 103          if right.len() > 0 {
 104              self.buf.push_all(right);
 105          }
 106  
 107          // Bump us forward
 108          self.pos += buf.len();
 109          Ok(())
 110      }
 111  }

 167          if self.eof() { return Err(io::standard_error(io::EndOfFile)) }

 176          self.pos += write_len;
 177          assert!(self.pos <= self.buf.len());

 179          return Ok(write_len);

 238      fn write(&mut self, buf: &[u8]) -> IoResult<()> {
 239          // raises a condition if the entire write does not fit in the buffer

 242              return Err(IoError {
 243                  kind: io::OtherIoError,

 245                  detail: None
 246              })
 247          }
 248  
 249          slice::bytes::copy_memory(self.buf.mut_slice_from(self.pos), buf);

 252      }

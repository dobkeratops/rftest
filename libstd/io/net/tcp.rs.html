  11  //! TCP network connections
  12  //!
  13  //! This module contains the ability to open a TCP stream to a socket address,
  14  //! as well as creating a socket server to accept incoming connections. The
  15  //! destination and binding addresses can either be an IPv4 or IPv6 address.
  16  //!
  17  //! A TCP connection implements the `Reader` and `Writer` traits, while the TCP
  18  //! listener (socket server) implements the `Listener` and `Acceptor` traits.
  19  
  20  #[deny(missing_doc)];
  21

  29  /// A structure which represents a TCP stream between a local socket and a
  30  /// remote socket.
  31  ///
  32  /// # Example
  33  ///
  34  /// ```rust
  35  /// # #[allow(unused_must_use)];
  36  /// use std::io::net::tcp::TcpStream;
  37  /// use std::io::net::ip::{Ipv4Addr, SocketAddr};
  38  ///
  39  /// let addr = SocketAddr { ip: Ipv4Addr(127, 0, 0, 1), port: 34254 };
  40  /// let mut stream = TcpStream::connect(addr);
  41  ///
  42  /// stream.write([1]);
  43  /// let mut buf = [0];
  44  /// stream.read(buf);
  45  /// drop(stream); // close the connection
  46  /// ```
  47  pub struct TcpStream {
  48      priv obj: ~RtioTcpStream
  49  }
  50  
  51  impl TcpStream {
  52      fn new(s: ~RtioTcpStream) -> TcpStream {
  53          TcpStream { obj: s }
  54      }
  55  
  56      /// Creates a TCP connection to a remote socket address.
  57      ///
  58      /// If no error is encountered, then `Ok(stream)` is returned.
  59      pub fn connect(addr: SocketAddr) -> IoResult<TcpStream> {
  60          LocalIo::maybe_raise(|io| {
  61              io.tcp_connect(addr).map(TcpStream::new)
  62          })
  63      }
  64  
  65      /// Returns the socket address of the remote peer of this TCP connection.
  66      pub fn peer_name(&mut self) -> IoResult<SocketAddr> {
  67          self.obj.peer_name()
  68      }
  69  
  70      /// Returns the socket address of the local half of this TCP connection.
  71      pub fn socket_name(&mut self) -> IoResult<SocketAddr> {
  72          self.obj.socket_name()
  73      }
  74  }
  75  
  76  impl Clone for TcpStream {
  77      /// Creates a new handle to this TCP stream, allowing for simultaneous reads
  78      /// and writes of this connection.
  79      ///
  80      /// The underlying TCP stream will not be closed until all handles to the
  81      /// stream have been deallocated. All handles will also follow the same
  82      /// stream, but two concurrent reads will not receive the same data.
  83      /// Instead, the first read will receive the first packet received, and the
  84      /// second read will receive the second packet.
  85      fn clone(&self) -> TcpStream {
  86          TcpStream { obj: self.obj.clone() }
  87      }
  88  }

  90  impl Reader for TcpStream {
  91      fn read(&mut self, buf: &mut [u8]) -> IoResult<uint> { self.obj.read(buf) }
  92  }

  94  impl Writer for TcpStream {
  95      fn write(&mut self, buf: &[u8]) -> IoResult<()> { self.obj.write(buf) }
  96  }

  98  /// A structure representing a socket server. This listener is used to create a
  99  /// `TcpAcceptor` which can be used to accept sockets on a local port.
 100  ///
 101  /// # Example
 102  ///
 103  /// ```rust
 104  /// # fn main() {}
 105  /// # fn foo() {
 106  /// # #[allow(unused_must_use, dead_code)];
 107  /// use std::io::net::tcp::TcpListener;
 108  /// use std::io::net::ip::{Ipv4Addr, SocketAddr};
 109  /// use std::io::{Acceptor, Listener};
 110  ///
 111  /// let addr = SocketAddr { ip: Ipv4Addr(127, 0, 0, 1), port: 80 };
 112  /// let listener = TcpListener::bind(addr);
 113  ///
 114  /// // bind the listener to the specified address
 115  /// let mut acceptor = listener.listen();
 116  ///
 117  /// // accept connections and process them
 118  /// # fn handle_client<T>(_: T) {}
 119  /// for stream in acceptor.incoming() {
 120  ///     spawn(proc() {
 121  ///         handle_client(stream);
 122  ///     });
 123  /// }
 124  ///
 125  /// // close the socket server
 126  /// drop(acceptor);
 127  /// # }
 128  /// ```
 129  pub struct TcpListener {
 130      priv obj: ~RtioTcpListener
 131  }
 132  
 133  impl TcpListener {
 134      /// Creates a new `TcpListener` which will be bound to the specified local
 135      /// socket address. This listener is not ready for accepting connections,
 136      /// `listen` must be called on it before that's possible.
 137      ///
 138      /// Binding with a port number of 0 will request that the OS assigns a port
 139      /// to this listener. The port allocated can be queried via the
 140      /// `socket_name` function.
 141      pub fn bind(addr: SocketAddr) -> IoResult<TcpListener> {
 142          LocalIo::maybe_raise(|io| {
 143              io.tcp_bind(addr).map(|l| TcpListener { obj: l })
 144          })
 145      }
 146  
 147      /// Returns the local socket address of this listener.
 148      pub fn socket_name(&mut self) -> IoResult<SocketAddr> {
 149          self.obj.socket_name()
 150      }
 151  }

 153  impl Listener<TcpStream, TcpAcceptor> for TcpListener {
 154      fn listen(self) -> IoResult<TcpAcceptor> {
 155          self.obj.listen().map(|acceptor| TcpAcceptor { obj: acceptor })
 156      }
 157  }

 159  /// The accepting half of a TCP socket server. This structure is created through
 160  /// a `TcpListener`'s `listen` method, and this object can be used to accept new
 161  /// `TcpStream` instances.
 162  pub struct TcpAcceptor {
 163      priv obj: ~RtioTcpAcceptor
 164  }
 165  
 166  impl Acceptor<TcpStream> for TcpAcceptor {
 167      fn accept(&mut self) -> IoResult<TcpStream> {
 168          self.obj.accept().map(TcpStream::new)
 169      }
 170  }

  69  struct Parser<'a> {
  70      // parsing as ASCII, so can use byte array
  71      s: &'a [u8],
  72      pos: uint,
  73  }
  74  
  75  impl<'a> Parser<'a> {
  76      fn new(s: &'a str) -> Parser<'a> {
  77          Parser {
  78              s: s.as_bytes(),
  79              pos: 0,
  80          }
  81      }
  82  
  83      fn is_eof(&self) -> bool {
  84          self.pos == self.s.len()
  85      }
  86  
  87      // Commit only if parser returns Some
  88      fn read_atomically<T>(&mut self, cb: |&mut Parser| -> Option<T>)
  89                         -> Option<T> {
  90          let pos = self.pos;
  91          let r = cb(self);
  92          if r.is_none() {
  93              self.pos = pos;
  94          }
  95          r
  96      }
  97  
  98      // Commit only if parser read till EOF
  99      fn read_till_eof<T>(&mut self, cb: |&mut Parser| -> Option<T>)
 100                       -> Option<T> {
 101          self.read_atomically(|p| cb(p).filtered(|_| p.is_eof()))
 102      }
 103  
 104      // Return result of first successful parser
 105      fn read_or<T>(&mut self, parsers: &[|&mut Parser| -> Option<T>])
 106                 -> Option<T> {
 107          for pf in parsers.iter() {
 108              match self.read_atomically(|p: &mut Parser| (*pf)(p)) {
 109                  Some(r) => return Some(r),
 110                  None => {}
 111              }
 112          }
 113          None
 114      }
 115  
 116      // Apply 3 parsers sequentially
 117      fn read_seq_3<A,
 118                    B,
 119                    C>(
 120                    &mut self,
 121                    pa: |&mut Parser| -> Option<A>,
 122                    pb: |&mut Parser| -> Option<B>,
 123                    pc: |&mut Parser| -> Option<C>)
 124                    -> Option<(A, B, C)> {
 125          self.read_atomically(|p| {
 126              let a = pa(p);
 127              let b = if a.is_some() { pb(p) } else { None };
 128              let c = if b.is_some() { pc(p) } else { None };
 129              match (a, b, c) {
 130                  (Some(a), Some(b), Some(c)) => Some((a, b, c)),
 131                  _ => None
 132              }
 133          })
 134      }
 135  
 136      // Read next char
 137      fn read_char(&mut self) -> Option<char> {
 138          if self.is_eof() {
 139              None
 140          } else {
 141              let r = self.s[self.pos] as char;
 142              self.pos += 1;
 143              Some(r)
 144          }
 145      }
 146  
 147      // Return char and advance iff next char is equal to requested
 148      fn read_given_char(&mut self, c: char) -> Option<char> {
 149          self.read_atomically(|p| {
 150              p.read_char().filtered(|&next| next == c)
 151          })
 152      }
 153  
 154      // Read digit
 155      fn read_digit(&mut self, radix: u8) -> Option<u8> {
 156          fn parse_digit(c: char, radix: u8) -> Option<u8> {
 157              let c = c as u8;
 158              // assuming radix is either 10 or 16
 159              if c >= '0' as u8 && c <= '9' as u8 {
 160                  Some(c - '0' as u8)
 161              } else if radix > 10 && c >= 'a' as u8 && c < 'a' as u8 + (radix - 10) {
 162                  Some(c - 'a' as u8 + 10)
 163              } else if radix > 10 && c >= 'A' as u8 && c < 'A' as u8 + (radix - 10) {
 164                  Some(c - 'A' as u8 + 10)
 165              } else {
 166                  None
 167              }
 168          }
 169  
 170          self.read_atomically(|p| {
 171              p.read_char().and_then(|c| parse_digit(c, radix))
 172          })
 173      }
 174  
 175      fn read_number_impl(&mut self, radix: u8, max_digits: u32, upto: u32) -> Option<u32> {
 176          let mut r = 0u32;
 177          let mut digit_count = 0;
 178          loop {
 179              match self.read_digit(radix) {
 180                  Some(d) => {
 181                      r = r * (radix as u32) + (d as u32);
 182                      digit_count += 1;
 183                      if digit_count > max_digits || r >= upto {
 184                          return None
 185                      }
 186                  }
 187                  None => {
 188                      if digit_count == 0 {
 189                          return None
 190                      } else {
 191                          return Some(r)
 192                      }
 193                  }
 194              };
 195          }
 196      }
 197  
 198      // Read number, failing if max_digits of number value exceeded
 199      fn read_number(&mut self, radix: u8, max_digits: u32, upto: u32) -> Option<u32> {
 200          self.read_atomically(|p| p.read_number_impl(radix, max_digits, upto))
 201      }

 203      fn read_ipv4_addr_impl(&mut self) -> Option<IpAddr> {
 204          let mut bs = [0u8, ..4];
 205          let mut i = 0;
 206          while i < 4 {
 207              if i != 0 && self.read_given_char('.').is_none() {
 208                  return None;
 209              }
 210  
 211              let octet = self.read_number(10, 3, 0x100).map(|n| n as u8);
 212              match octet {
 213                  Some(d) => bs[i] = d,
 214                  None => return None,
 215              };
 216              i += 1;
 217          }
 218          Some(Ipv4Addr(bs[0], bs[1], bs[2], bs[3]))
 219      }

 221      // Read IPv4 address
 222      fn read_ipv4_addr(&mut self) -> Option<IpAddr> {
 223          self.read_atomically(|p| p.read_ipv4_addr_impl())
 224      }

 226      fn read_ipv6_addr_impl(&mut self) -> Option<IpAddr> {
 227          fn ipv6_addr_from_head_tail(head: &[u16], tail: &[u16]) -> IpAddr {
 228              assert!(head.len() + tail.len() <= 8);
 229              let mut gs = [0u16, ..8];
 230              gs.copy_from(head);
 231              gs.mut_slice(8 - tail.len(), 8).copy_from(tail);
 232              Ipv6Addr(gs[0], gs[1], gs[2], gs[3], gs[4], gs[5], gs[6], gs[7])
 233          }
 234  
 235          fn read_groups(p: &mut Parser, groups: &mut [u16, ..8], limit: uint) -> (uint, bool) {
 236              let mut i = 0;
 237              while i < limit {
 238                  if i < limit - 1 {
 239                      let ipv4 = p.read_atomically(|p| {
 240                          if i == 0 || p.read_given_char(':').is_some() {
 241                              p.read_ipv4_addr()
 242                          } else {
 243                              None
 244                          }
 245                      });
 246                      match ipv4 {
 247                          Some(Ipv4Addr(a, b, c, d)) => {
 248                              groups[i + 0] = (a as u16 << 8) | (b as u16);
 249                              groups[i + 1] = (c as u16 << 8) | (d as u16);
 250                              return (i + 2, true);
 251                          }
 252                          _ => {}
 253                      }
 254                  }
 255  
 256                  let group = p.read_atomically(|p| {
 257                      if i == 0 || p.read_given_char(':').is_some() {
 258                          p.read_number(16, 4, 0x10000).map(|n| n as u16)
 259                      } else {
 260                          None
 261                      }
 262                  });
 263                  match group {
 264                      Some(g) => groups[i] = g,
 265                      None => return (i, false)
 266                  }
 267                  i += 1;
 268              }
 269              (i, false)
 270          }
 271  
 272          let mut head = [0u16, ..8];
 273          let (head_size, head_ipv4) = read_groups(self, &mut head, 8);
 274  
 275          if head_size == 8 {
 276              return Some(Ipv6Addr(
 277                  head[0], head[1], head[2], head[3],
 278                  head[4], head[5], head[6], head[7]))
 279          }
 280  
 281          // IPv4 part is not allowed before `::`
 282          if head_ipv4 {
 283              return None
 284          }
 285  
 286          // read `::` if previous code parsed less than 8 groups
 287          if !self.read_given_char(':').is_some() || !self.read_given_char(':').is_some() {
 288              return None;
 289          }
 290  
 291          let mut tail = [0u16, ..8];
 292          let (tail_size, _) = read_groups(self, &mut tail, 8 - head_size);
 293          Some(ipv6_addr_from_head_tail(head.slice(0, head_size), tail.slice(0, tail_size)))
 294      }

 296      fn read_ipv6_addr(&mut self) -> Option<IpAddr> {
 297          self.read_atomically(|p| p.read_ipv6_addr_impl())
 298      }

 300      fn read_ip_addr(&mut self) -> Option<IpAddr> {
 301          let ipv4_addr = |p: &mut Parser| p.read_ipv4_addr();
 302          let ipv6_addr = |p: &mut Parser| p.read_ipv6_addr();
 303          self.read_or([ipv4_addr, ipv6_addr])
 304      }
 305  
 306      fn read_socket_addr(&mut self) -> Option<SocketAddr> {
 307          let ip_addr = |p: &mut Parser| {
 308              let ipv4_p = |p: &mut Parser| p.read_ip_addr();
 309              let ipv6_p = |p: &mut Parser| {
 310                  let open_br = |p: &mut Parser| p.read_given_char('[');
 311                  let ip_addr = |p: &mut Parser| p.read_ipv6_addr();
 312                  let clos_br = |p: &mut Parser| p.read_given_char(']');
 313                  p.read_seq_3::<char, IpAddr, char>(open_br, ip_addr, clos_br)
 314                          .map(|t| match t { (_, ip, _) => ip })
 315              };
 316              p.read_or([ipv4_p, ipv6_p])
 317          };
 318          let colon = |p: &mut Parser| p.read_given_char(':');
 319          let port  = |p: &mut Parser| p.read_number(10, 5, 0x10000).map(|n| n as u16);
 320  
 321          // host, colon, port
 322          self.read_seq_3::<IpAddr, char, u16>(ip_addr, colon, port)
 323                  .map(|t| match t { (ip, _, port) => SocketAddr { ip: ip, port: port } })
 324      }
 325  }

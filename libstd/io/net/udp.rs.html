  47  impl Clone for UdpSocket {
  48      /// Creates a new handle to this UDP socket, allowing for simultaneous reads
  49      /// and writes of the socket.
  50      ///
  51      /// The underlying UDP socket will not be closed until all handles to the
  52      /// socket have been deallocated. Two concurrent reads will not receive the
  53      /// same data.  Instead, the first read will receive the first packet
  54      /// received, and the second read will receive the second packet.
  55      fn clone(&self) -> UdpSocket {
  56          UdpSocket { obj: self.obj.clone() }
  57      }
  58  }

  73  impl Reader for UdpStream {
  74      fn read(&mut self, buf: &mut [u8]) -> IoResult<uint> {
  75          let peer = self.connected_to;
  76          self.as_socket(|sock| {
  77              match sock.recvfrom(buf) {
  78                  Ok((_nread, src)) if src != peer => Ok(0),
  79                  Ok((nread, _src)) => Ok(nread),
  80                  Err(e) => Err(e),
  81              }
  82          })
  83      }
  84  }

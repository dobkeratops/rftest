  27  /// An iterator that reads a single byte on each iteration,
  28  /// until `.read_byte()` returns `EndOfFile`.
  29  ///
  30  /// # Notes about the Iteration Protocol
  31  ///
  32  /// The `Bytes` may yield `None` and thus terminate
  33  /// an iteration, but continue to yield elements if iteration
  34  /// is attempted again.
  35  ///
  36  /// # Error
  37  ///
  38  /// Any error other than `EndOfFile` that is produced by the underlying Reader
  39  /// is returned by the iterator and should be handled by the caller.
  40  pub struct Bytes<'r, T> {
  41      priv reader: &'r mut T,
  42  }
  43  
  44  impl<'r, R: Reader> Bytes<'r, R> {
  45      pub fn new(r: &'r mut R) -> Bytes<'r, R> {
  46          Bytes { reader: r }
  47      }
  48  }

  50  impl<'r, R: Reader> Iterator<IoResult<u8>> for Bytes<'r, R> {
  51      #[inline]
  52      fn next(&mut self) -> Option<IoResult<u8>> {
  53          match self.reader.read_byte() {
  54              Ok(x) => Some(Ok(x)),
  55              Err(IoError { kind: io::EndOfFile, .. }) => None,
  56              Err(e) => Some(Err(e))
  57          }
  58      }
  59  }

  61  pub fn u64_to_le_bytes<T>(n: u64, size: uint, f: |v: &[u8]| -> T) -> T {
  62      use mem::{to_le16, to_le32, to_le64};
  63      use cast::transmute;
  64  
  65      // LLVM fails to properly optimize this when using shifts instead of the to_le* intrinsics
  66      assert!(size <= 8u);
  67      match size {
  68        1u => f(&[n as u8]),
  69        2u => f(unsafe { transmute::<i16, [u8, ..2]>(to_le16(n as i16)) }),
  70        4u => f(unsafe { transmute::<i32, [u8, ..4]>(to_le32(n as i32)) }),
  71        8u => f(unsafe { transmute::<i64, [u8, ..8]>(to_le64(n as i64)) }),
  72        _ => {
  73  
  74          let mut bytes: ~[u8] = ~[];
  75          let mut i = size;
  76          let mut n = n;
  77          while i > 0u {
  78              bytes.push((n & 255_u64) as u8);
  79              n >>= 8_u64;
  80              i -= 1u;
  81          }
  82          f(bytes)
  83        }
  84      }
  85  }

  87  pub fn u64_to_be_bytes<T>(n: u64, size: uint, f: |v: &[u8]| -> T) -> T {
  88      use mem::{to_be16, to_be32, to_be64};
  89      use cast::transmute;
  90  
  91      // LLVM fails to properly optimize this when using shifts instead of the to_be* intrinsics
  92      assert!(size <= 8u);
  93      match size {
  94        1u => f(&[n as u8]),
  95        2u => f(unsafe { transmute::<i16, [u8, ..2]>(to_be16(n as i16)) }),
  96        4u => f(unsafe { transmute::<i32, [u8, ..4]>(to_be32(n as i32)) }),
  97        8u => f(unsafe { transmute::<i64, [u8, ..8]>(to_be64(n as i64)) }),
  98        _ => {
  99          let mut bytes: ~[u8] = ~[];
 100          let mut i = size;
 101          while i > 0u {
 102              let shift = ((i - 1u) * 8u) as u64;
 103              bytes.push((n >> shift) as u8);
 104              i -= 1u;
 105          }
 106          f(bytes)
 107        }
 108      }
 109  }

 111  pub fn u64_from_be_bytes(data: &[u8],
 112                           start: uint,
 113                           size: uint)
 114                        -> u64 {
 115      use ptr::{copy_nonoverlapping_memory};
 116      use mem::from_be64;
 117      use slice::MutableVector;
 118  
 119      assert!(size <= 8u);
 120  
 121      if data.len() - start < size {
 122          fail!("index out of bounds");
 123      }
 124  
 125      let mut buf = [0u8, ..8];
 126      unsafe {
 127          let ptr = data.as_ptr().offset(start as int);
 128          let out = buf.as_mut_ptr();
 129          copy_nonoverlapping_memory(out.offset((8 - size) as int), ptr, size);
 130          from_be64(*(out as *i64)) as u64
 131      }
 132  }

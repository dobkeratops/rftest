  57  impl Reader for ChanReader {
  58      fn read(&mut self, buf: &mut [u8]) -> IoResult<uint> {
  59          let mut num_read = 0;
  60          loop {
  61              match self.buf {
  62                  Some(ref prev) => {
  63                      let dst = buf.mut_slice_from(num_read);
  64                      let src = prev.slice_from(self.pos);
  65                      let count = cmp::min(dst.len(), src.len());
  66                      bytes::copy_memory(dst, src.slice_to(count));
  67                      num_read += count;
  68                      self.pos += count;
  69                  },
  70                  None => (),
  71              };
  72              if num_read == buf.len() || self.closed {
  73                  break;
  74              }
  75              self.pos = 0;
  76              self.buf = self.rx.recv_opt();
  77              self.closed = self.buf.is_none();
  78          }
  79          if self.closed && num_read == 0 {
  80              Err(io::standard_error(io::EndOfFile))
  81          } else {
  82              Ok(num_read)
  83          }
  84      }
  85  }

  44      fn read(&mut self, buf: &mut [u8]) -> io::IoResult<uint> {

  46              return Err(io::standard_error(io::EndOfFile));
  47          }

 134      fn read(&mut self, buf: &mut [u8]) -> io::IoResult<uint> {
 135          loop {
 136              let err = match self.cur_reader {
 137                  Some(ref mut r) => {

 139                          Ok(len) => return Ok(len),
 140                          Err(ref e) if e.kind == io::EndOfFile => None,
 141                          Err(e) => Some(e),
 142                      }
 143                  }
 144                  None => break
 145              };
 146              self.cur_reader = self.readers.next();
 147              match err {
 148                  Some(e) => return Err(e),
 149                  None => {}
 150              }
 151          }
 152          Err(io::standard_error(io::EndOfFile))
 153      }

 185  /// Copies all data from a `Reader` to a `Writer`.
 186  pub fn copy<R: Reader, W: Writer>(r: &mut R, w: &mut W) -> io::IoResult<()> {
 187      let mut buf = [0, ..super::DEFAULT_BUF_SIZE];
 188      loop {
 189          let len = match r.read(buf) {
 190              Ok(len) => len,
 191              Err(ref e) if e.kind == io::EndOfFile => return Ok(()),
 192              Err(e) => return Err(e),
 193          };
 194          try!(w.write(buf.slice_to(len)));
 195      }
 196  }

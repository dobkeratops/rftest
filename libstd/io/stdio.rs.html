  11  /*! Non-blocking access to stdin, stdout, and stderr.
  12  
  13  This module provides bindings to the local event loop's TTY interface, using it
  14  to offer synchronous but non-blocking versions of stdio. These handles can be
  15  inspected for information about terminal dimensions or for related information
  16  about the stream or terminal to which it is attached.
  17  
  18  # Example
  19  
  20  ```rust
  21  # #[allow(unused_must_use)];
  22  use std::io;
  23  
  24  let mut out = io::stdout();
  25  out.write(bytes!("Hello, world!"));
  26  ```
  27  
  28  */

  78  fn src<T>(fd: libc::c_int, readable: bool, f: |StdSource| -> T) -> T {
  79      LocalIo::maybe_raise(|io| {
  80          Ok(match io.tty_open(fd, readable) {
  81              Ok(tty) => f(TTY(tty)),
  82              Err(_) => f(File(io.fs_from_raw_fd(fd, DontClose))),
  83          })
  84      }).unwrap()
  85  }

  87  /// Creates a new non-blocking handle to the stdin of the current process.
  88  ///
  89  /// The returned handled is buffered by default with a `BufferedReader`. If
  90  /// buffered access is not desired, the `stdin_raw` function is provided to
  91  /// provided unbuffered access to stdin.
  92  ///
  93  /// Care should be taken when creating multiple handles to the stdin of a
  94  /// process. Beause this is a buffered reader by default, it's possible for
  95  /// pending input to be unconsumed in one reader and unavailable to other
  96  /// readers. It is recommended that only one handle at a time is created for the
  97  /// stdin of a process.
  98  ///
  99  /// See `stdout()` for more notes about this function.
 100  pub fn stdin() -> BufferedReader<StdReader> {
 101      BufferedReader::new(stdin_raw())
 102  }

 104  /// Creates a new non-blocking handle to the stdin of the current process.
 105  ///
 106  /// Unlike `stdin()`, the returned reader is *not* a buffered reader.
 107  ///
 108  /// See `stdout()` for more notes about this function.
 109  pub fn stdin_raw() -> StdReader {
 110      src(libc::STDIN_FILENO, true, |src| StdReader { inner: src })
 111  }

 113  /// Creates a line-buffered handle to the stdout of the current process.
 114  ///
 115  /// Note that this is a fairly expensive operation in that at least one memory
 116  /// allocation is performed. Additionally, this must be called from a runtime
 117  /// task context because the stream returned will be a non-blocking object using
 118  /// the local scheduler to perform the I/O.

 122  /// no synchronization is performed to ensure a sane output.
 123  pub fn stdout() -> LineBufferedWriter<StdWriter> {
 124      LineBufferedWriter::new(stdout_raw())
 125  }

 148  fn reset_helper(w: ~Writer,
 149                  f: |&mut Task, ~Writer| -> Option<~Writer>) -> Option<~Writer> {
 150      let mut t = Local::borrow(None::<Task>);
 151      // Be sure to flush any pending output from the writer
 152      match f(t.get(), w) {
 153          Some(mut w) => {
 154              drop(t);
 155              // FIXME: is failing right here?
 156              w.flush().unwrap();
 157              Some(w)
 158          }
 159          None => None
 160      }
 161  }

 163  /// Resets the task-local stdout handle to the specified writer
 164  ///
 165  /// This will replace the current task's stdout handle, returning the old
 166  /// handle. All future calls to `print` and friends will emit their output to
 167  /// this specified handle.
 168  ///
 169  /// Note that this does not need to be called for all new tasks; the default
 170  /// output handle is to the process's stdout stream.
 171  pub fn set_stdout(stdout: ~Writer) -> Option<~Writer> {
 172      reset_helper(stdout, |t, w| replace(&mut t.stdout, Some(w)))
 173  }

 175  /// Resets the task-local stderr handle to the specified writer
 176  ///
 177  /// This will replace the current task's stderr handle, returning the old
 178  /// handle. Currently, the stderr handle is used for printing failure messages
 179  /// during task failure.
 180  ///
 181  /// Note that this does not need to be called for all new tasks; the default
 182  /// output handle is to the process's stderr stream.
 183  pub fn set_stderr(stderr: ~Writer) -> Option<~Writer> {
 184      reset_helper(stderr, |t, w| replace(&mut t.stderr, Some(w)))
 185  }

 187  // Helper to access the local task's stdout handle
 188  //
 189  // Note that this is not a safe function to expose because you can create an
 190  // aliased pointer very easily:
 191  //
 192  //  with_task_stdout(|io1| {
 193  //      with_task_stdout(|io2| {
 194  //          // io1 aliases io2
 195  //      })
 196  //  })
 197  fn with_task_stdout(f: |&mut Writer| -> IoResult<()> ) {
 198      let task: Option<~Task> = Local::try_take();
 199      let result = match task {
 200          Some(mut task) => {
 201              // Printing may run arbitrary code, so ensure that the task is in
 202              // TLS to allow all std services. Note that this means a print while
 203              // printing won't use the task's normal stdout handle, but this is
 204              // necessary to ensure safety (no aliasing).
 205              let mut my_stdout = task.stdout.take();
 206              Local::put(task);
 207  
 208              if my_stdout.is_none() {
 209                  my_stdout = Some(~stdout() as ~Writer);
 210              }
 211              let ret = f(*my_stdout.get_mut_ref());
 212  
 213              // Note that we need to be careful when putting the stdout handle
 214              // back into the task. If the handle was set to `Some` while
 215              // printing, then we can run aribitrary code when destroying the
 216              // previous handle. This means that the local task needs to be in
 217              // TLS while we do this.
 218              //
 219              // To protect against this, we do a little dance in which we
 220              // temporarily take the task, swap the handles, put the task in TLS,
 221              // and only then drop the previous handle.
 222              let mut t = Local::borrow(None::<Task>);
 223              let prev = replace(&mut t.get().stdout, my_stdout);
 224              drop(t);
 225              drop(prev);
 226              ret
 227          }

 229          None => {
 230              struct Stdout;
 231              impl Writer for Stdout {
 232                  fn write(&mut self, data: &[u8]) -> IoResult<()> {
 233                      unsafe {
 234                          libc::write(libc::STDOUT_FILENO,
 235                                      data.as_ptr() as *libc::c_void,
 236                                      data.len() as libc::size_t);
 237                      }
 238                      Ok(()) // just ignore the results
 239                  }
 240              }
 241              let mut io = Stdout;
 242              f(&mut io as &mut Writer)
 243          }
 244      };
 245  
 246      match result {
 247          Ok(()) => {}
 248          Err(e) => fail!("failed printing to stdout: {}", e),
 249      }
 250  }

 252  /// Flushes the local task's stdout handle.
 253  ///
 254  /// By default, this stream is a line-buffering stream, so flushing may be
 255  /// necessary to ensure that all output is printed to the screen (if there are
 256  /// no newlines printed).
 257  ///
 258  /// Note that logging macros do not use this stream. Using the logging macros
 259  /// will emit output to stderr, and while they are line buffered the log
 260  /// messages are always terminated in a newline (no need to flush).
 261  pub fn flush() {
 262      with_task_stdout(|io| io.flush())
 263  }

 265  /// Prints a string to the stdout of the current process. No newline is emitted

 268      with_task_stdout(|io| io.write(s.as_bytes()))

 296  impl Reader for StdReader {
 297      fn read(&mut self, buf: &mut [u8]) -> IoResult<uint> {
 298          let ret = match self.inner {
 299              TTY(ref mut tty) => {
 300                  // Flush the task-local stdout so that weird issues like a
 301                  // print!'d prompt not being shown until after the user hits
 302                  // enter.
 303                  flush();
 304                  tty.read(buf)
 305              },
 306              File(ref mut file) => file.read(buf).map(|i| i as uint),
 307          };
 308          match ret {
 309              // When reading a piped stdin, libuv will return 0-length reads when
 310              // stdin reaches EOF. For pretty much all other streams it will
 311              // return an actual EOF error, but apparently for stdin it's a
 312              // little different. Hence, here we convert a 0 length read to an
 313              // end-of-file indicator so the caller knows to stop reading.
 314              Ok(0) => { Err(standard_error(EndOfFile)) }
 315              ret @ Ok(..) | ret @ Err(..) => ret,
 316          }
 317      }
 318  }

 325  impl StdWriter {
 326      /// Gets the size of this output window, if possible. This is typically used
 327      /// when the writer is attached to something like a terminal, this is used
 328      /// to fetch the dimensions of the terminal.
 329      ///
 330      /// If successful, returns `Ok((width, height))`.
 331      ///
 332      /// # Error
 333      ///
 334      /// This function will return an error if the output stream is not actually
 335      /// connected to a TTY instance, or if querying the TTY instance fails.
 336      pub fn winsize(&mut self) -> IoResult<(int, int)> {
 337          match self.inner {
 338              TTY(ref mut tty) => tty.get_winsize(),
 339              File(..) => {
 340                  Err(IoError {
 341                      kind: OtherIoError,
 342                      desc: "stream is not a tty",
 343                      detail: None,
 344                  })
 345              }
 346          }
 347      }

 349      /// Controls whether this output stream is a "raw stream" or simply a normal
 350      /// stream.

 353      ///
 354      /// This function will return an error if the output stream is not actually
 355      /// connected to a TTY instance, or if querying the TTY instance fails.
 356      pub fn set_raw(&mut self, raw: bool) -> IoResult<()> {
 357          match self.inner {

  60  /// Listener provides a receiver to listen for registered signals.
  61  ///
  62  /// Listener automatically unregisters its handles once it is out of scope.
  63  /// However, clients can still unregister signums manually.
  64  ///
  65  /// # Example
  66  ///
  67  /// ```rust,no_run
  68  /// # #![allow(unused_must_use)]
  69  /// use std::io::signal::{Listener, Interrupt};
  70  ///
  71  /// let mut listener = Listener::new();
  72  /// listener.register(Interrupt);
  73  ///
  74  /// loop {

  76  ///         Interrupt => println!("Got Interrupt'ed"),

  79  /// }

  84      /// This is where all the handles send signums, which are received by

 106      /// Listen for a signal, returning true when successfully registered for
 107      /// signum. Signals can be received using `recv()`.
 108      ///
 109      /// Once a signal is registered, this listener will continue to receive
 110      /// notifications of signals until it is unregistered. This occurs
 111      /// regardless of the number of other listeners registered in other tasks
 112      /// (or on this task).
 113      ///
 114      /// Signals are still received if there is no task actively waiting for
 115      /// a signal, and a later call to `recv` will return the signal that was
 116      /// received while no task was waiting on it.
 117      ///
 118      /// # Error
 119      ///
 120      /// If this function fails to register a signal handler, then an error will
 121      /// be returned.
 122      pub fn register(&mut self, signum: Signum) -> io::IoResult<()> {
 123          if self.handles.iter().any(|&(sig, _)| sig == signum) {
 124              return Ok(()); // self is already listening to signum, so succeed
 125          }
 126          match LocalIo::maybe_raise(|io| {
 127              io.signal(signum, self.tx.clone())
 128          }) {
 129              Ok(handle) => {
 130                  self.handles.push((signum, handle));
 131                  Ok(())
 132              }
 133              Err(e) => Err(e)
 134          }
 135      }

 137      /// Unregisters a signal. If this listener currently had a handler
 138      /// registered for the signal, then it will stop receiving any more
 139      /// notification about the signal. If the signal has already been received,
 140      /// it may still be returned by `recv`.
 141      pub fn unregister(&mut self, signum: Signum) {

 143              Some(i) => drop(self.handles.remove(i)),
 144              None => {}
 145          }
 146      }

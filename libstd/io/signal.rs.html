  61  /// Listener provides a receiver to listen for registered signals.
  62  ///
  63  /// Listener automatically unregisters its handles once it is out of scope.
  64  /// However, clients can still unregister signums manually.
  65  ///
  66  /// # Example
  67  ///
  68  /// ```rust,no_run
  69  /// # #![allow(unused_must_use)]
  70  /// use std::io::signal::{Listener, Interrupt};
  71  ///
  72  /// let mut listener = Listener::new();
  73  /// listener.register(Interrupt);
  74  ///
  75  /// loop {
  76  ///     match listener.rx.recv() {
  77  ///         Interrupt => println!("Got Interrupt'ed"),
  78  ///         _ => (),
  79  ///     }
  80  /// }
  81  /// ```
  82  pub struct Listener {
  83      /// A map from signums to handles to keep the handles in memory
  84      handles: Vec<(Signum, ~RtioSignal:Send)>,
  85      /// This is where all the handles send signums, which are received by
  86      /// the clients from the receiver.
  87      tx: Sender<Signum>,
  88  
  89      /// Clients of Listener can `recv()` on this receiver. This is exposed to
  90      /// allow selection over it as well as manipulation of the receiver
  91      /// directly.
  92      pub rx: Receiver<Signum>,
  93  }

 107      /// Listen for a signal, returning true when successfully registered for
 108      /// signum. Signals can be received using `recv()`.
 109      ///
 110      /// Once a signal is registered, this listener will continue to receive
 111      /// notifications of signals until it is unregistered. This occurs
 112      /// regardless of the number of other listeners registered in other tasks
 113      /// (or on this task).
 114      ///
 115      /// Signals are still received if there is no task actively waiting for
 116      /// a signal, and a later call to `recv` will return the signal that was
 117      /// received while no task was waiting on it.
 118      ///
 119      /// # Error
 120      ///
 121      /// If this function fails to register a signal handler, then an error will
 122      /// be returned.
 123      pub fn register(&mut self, signum: Signum) -> io::IoResult<()> {
 124          if self.handles.iter().any(|&(sig, _)| sig == signum) {
 125              return Ok(()); // self is already listening to signum, so succeed
 126          }
 127          match LocalIo::maybe_raise(|io| {
 128              io.signal(signum, self.tx.clone())
 129          }) {
 130              Ok(handle) => {
 131                  self.handles.push((signum, handle));
 132                  Ok(())
 133              }
 134              Err(e) => Err(e)
 135          }
 136      }
 137  
 138      /// Unregisters a signal. If this listener currently had a handler
 139      /// registered for the signal, then it will stop receiving any more
 140      /// notification about the signal. If the signal has already been received,
 141      /// it may still be returned by `recv`.
 142      pub fn unregister(&mut self, signum: Signum) {
 143          match self.handles.iter().position(|&(i, _)| i == signum) {
 144              Some(i) => drop(self.handles.remove(i)),
 145              None => {}
 146          }
 147      }
 148  }

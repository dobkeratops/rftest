 136 pub fn begin_unwind_(msg: *c_char, file: *c_char, line: size_t) -> ! {
 137     use either::Left;
 138     use option::{Some, None};
 139     use rt::{context, OldTaskContext, TaskContext};
 140     use rt::task::Task;
 141     use rt::local::Local;
 142     use rt::logging::Logger;
 143     use str::Str;
 144 
 145     let context = context();
 146     match context {
 147         OldTaskContext => {
 148             unsafe {
 149                 rustrt::rust_upcall_fail(msg, file, line);
 150                 cast::transmute(())
 151             }
 152         }
 153         _ => {
 154             unsafe {
 155                 // XXX: Bad re-allocations. fail! needs some refactoring
 156                 let msg = str::raw::from_c_str(msg);
 157                 let file = str::raw::from_c_str(file);
 158 
 159                 // XXX: Logging doesn't work correctly in non-task context because it
 160                 // invokes the local heap
 161                 if context == TaskContext {
 162                     // XXX: Logging doesn't work here - the check to call the log
 163                     // function never passes - so calling the log function directly.
 164                     do Local::borrow::<Task, ()> |task| {
 165                         let msg = match task.name {
 166                             Some(ref name) =>
 167                                 fmt!("task '%s' failed at '%s', %s:%i",
 168                                      name.as_slice(), msg, file, line as int),
 169                             None =>
 170                                 fmt!("task <unnamed> failed at '%s', %s:%i",
 171                                      msg, file, line as int)
 172                         };
 173 
 174                         task.logger.log(Left(msg));
 175                     }
 176                 } else {
 177                     rterrln!("failed in non-task context at '%s', %s:%i",
 178                              msg, file, line as int);
 179                 }
 180 
 181                 let task = Local::unsafe_borrow::<Task>();
 182                 if (*task).unwinder.unwinding {
 183                     rtabort!("unwinding again");
 184                 }
 185                 (*task).unwinder.begin_unwind();
 186             }
 187         }
 188     }

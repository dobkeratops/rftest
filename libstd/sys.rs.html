 137          // XXX: Bad re-allocations. fail! needs some refactoring
 138          let msg = str::raw::from_c_str(msg);

 144              // XXX: Logging doesn't work here - the check to call the log

 146              do Local::borrow::<Task, ()> |task| {
 147                  let msg = match task.name {
 148                      Some(ref name) =>
 149                      fmt!("task '%s' failed at '%s', %s:%i",
 150                           name.as_slice(), msg, file, line as int),
 151                      None =>
 152                      fmt!("task <unnamed> failed at '%s', %s:%i",
 153                           msg, file, line as int)
 154                  };
 155  
 156                  task.logger.log(Left(msg));
 157              }
 158          } else {
 159              rterrln!("failed in non-task context at '%s', %s:%i",
 160                       msg, file, line as int);
 161          }
 162  
 163          let task = Local::unsafe_borrow::<Task>();
 164          if (*task).unwinder.unwinding {
 165              rtabort!("unwinding again");
 166          }

 168      }

 203      pub fn write_vec_range(&mut self, ptr: *(), len: uint, inner: *TyDesc) -> bool {
 204          let mut p = ptr as *u8;
 205          let (sz, al) = unsafe { ((*inner).size, (*inner).align) };
 206          try!(self, self.writer.write(['[' as u8]));
 207          let mut first = true;
 208          let mut left = len;
 209          // unit structs have 0 size, and don't loop forever.
 210          let dec = if sz == 0 {1} else {sz};
 211          while left > 0 {
 212              if first {
 213                  first = false;
 214              } else {
 215                  try!(self, self.writer.write(", ".as_bytes()));
 216              }
 217              self.visit_ptr_inner(p as *u8, inner);
 218              p = align(unsafe { p.offset(sz as int) as uint }, al) as *u8;
 219              left -= dec;
 220          }
 221          try!(self, self.writer.write([']' as u8]));
 222          true
 223      }

 226          self.write_vec_range(&v.data, v.fill, inner)
 227      }

 229      fn write_escaped_char(&mut self, ch: char, is_str: bool) -> bool {
 230          try!(self, match ch {
 231              '\t' => self.writer.write("\\t".as_bytes()),
 232              '\r' => self.writer.write("\\r".as_bytes()),
 233              '\n' => self.writer.write("\\n".as_bytes()),
 234              '\\' => self.writer.write("\\\\".as_bytes()),
 235              '\'' => {
 236                  if is_str {
 237                      self.writer.write("'".as_bytes())
 238                  } else {
 239                      self.writer.write("\\'".as_bytes())
 240                  }
 241              }
 242              '"' => {
 243                  if is_str {
 244                      self.writer.write("\\\"".as_bytes())
 245                  } else {
 246                      self.writer.write("\"".as_bytes())
 247                  }
 248              }
 249              '\x20'..'\x7e' => self.writer.write([ch as u8]),
 250              _ => {
 251                  char::escape_unicode(ch, |c| {
 252                      let _ = self.writer.write([c as u8]);
 253                  });
 254                  Ok(())
 255              }
 256          });
 257          return true;
 258      }
 259  }

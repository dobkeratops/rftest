 127          self.ptr = adjustment(self.ptr);

 134      }

 178      pub fn write_escaped_slice(&mut self, slice: &str) {
 179          self.writer.write(['"' as u8]);
 180          for ch in slice.iter() {
 181              self.write_escaped_char(ch, true);
 182          }
 183          self.writer.write(['"' as u8]);
 184      }

 186      pub fn write_mut_qualifier(&mut self, mtbl: uint) {
 187          if mtbl == 0 {
 188              self.writer.write("mut ".as_bytes());
 189          } else if mtbl == 1 {
 190              // skip, this is ast::m_imm
 191          } else {
 192              fail!("invalid mutability value");
 193          }
 194      }

 196      pub fn write_vec_range(&mut self,
 197                             _mtbl: uint,
 198                             ptr: *(),
 199                             len: uint,
 200                             inner: *TyDesc)
 201                             -> bool {
 202          let mut p = ptr as *u8;
 203          let (sz, al) = unsafe { ((*inner).size, (*inner).align) };
 204          self.writer.write(['[' as u8]);
 205          let mut first = true;
 206          let mut left = len;
 207          // unit structs have 0 size, and don't loop forever.
 208          let dec = if sz == 0 {1} else {sz};
 209          while left > 0 {
 210              if first {
 211                  first = false;
 212              } else {
 213                  self.writer.write(", ".as_bytes());
 214              }
 215              self.visit_ptr_inner(p as *c_void, inner);
 216              p = align(unsafe { ptr::offset(p, sz as int) as uint }, al) as *u8;
 217              left -= dec;
 218          }
 219          self.writer.write([']' as u8]);
 220          true
 221      }

 223      pub fn write_unboxed_vec_repr(&mut self,
 224                                    mtbl: uint,
 225                                    v: &raw::Vec<()>,
 226                                    inner: *TyDesc)
 227                                    -> bool {
 228          self.write_vec_range(mtbl, ptr::to_unsafe_ptr(&v.data),
 229                               v.fill, inner)
 230      }

 232      fn write_escaped_char(&mut self, ch: char, is_str: bool) {
 233          match ch {
 234              '\t' => self.writer.write("\\t".as_bytes()),
 235              '\r' => self.writer.write("\\r".as_bytes()),
 236              '\n' => self.writer.write("\\n".as_bytes()),
 237              '\\' => self.writer.write("\\\\".as_bytes()),
 238              '\'' => {
 239                  if is_str {
 240                      self.writer.write("'".as_bytes())
 241                  } else {
 242                      self.writer.write("\\'".as_bytes())
 243                  }
 244              }
 245              '"' => {
 246                  if is_str {
 247                      self.writer.write("\\\"".as_bytes())
 248                  } else {
 249                      self.writer.write("\"".as_bytes())
 250                  }
 251              }
 252              '\x20'..'\x7e' => self.writer.write([ch as u8]),
 253              _ => {
 254                  do char::escape_unicode(ch) |c| {
 255                      self.writer.write([c as u8]);
 256                  }
 257              }
 258          }
 259      }

  43  impl EscapedCharWriter for @Writer {
  44      fn write_escaped_char(&self, ch: char) {
  45          match ch {
  46              '\t' => self.write_str("\\t"),
  47              '\r' => self.write_str("\\r"),
  48              '\n' => self.write_str("\\n"),
  49              '\\' => self.write_str("\\\\"),
  50              '\'' => self.write_str("\\'"),
  51              '"' => self.write_str("\\\""),
  52              '\x20'..'\x7e' => self.write_char(ch),
  53              _ => {
  54                  do char::escape_unicode(ch) |c| {
  55                      self.write_char(c);
  56                  }
  57              }
  58          }
  59      }
  60  }

 177      pub fn write_escaped_slice(&self, slice: &str) {
 178          self.writer.write_char('"');
 179          for ch in slice.iter() {

 181          }

 185      pub fn write_mut_qualifier(&self, mtbl: uint) {
 186          if mtbl == 0 {
 187              self.writer.write_str("mut ");
 188          } else if mtbl == 1 {
 189              // skip, this is ast::m_imm
 190          } else {
 191              assert_eq!(mtbl, 2);
 192              self.writer.write_str("const ");
 193          }
 194      }

 196      pub fn write_vec_range(&self,
 197                             _mtbl: uint,
 198                             ptr: *(),
 199                             len: uint,
 200                             inner: *TyDesc)
 201                             -> bool {
 202          let mut p = ptr as *u8;
 203          let (sz, al) = unsafe { ((*inner).size, (*inner).align) };
 204          self.writer.write_char('[');
 205          let mut first = true;
 206          let mut left = len;
 207          // unit structs have 0 size, and don't loop forever.
 208          let dec = if sz == 0 {1} else {sz};
 209          while left > 0 {
 210              if first {
 211                  first = false;
 212              } else {
 213                  self.writer.write_str(", ");
 214              }
 215              self.visit_ptr_inner(p as *c_void, inner);
 216              p = align(ptr::offset(p, sz as int) as uint, al) as *u8;
 217              left -= dec;
 218          }
 219          self.writer.write_char(']');
 220          true
 221      }

 227                                    -> bool {

 438      fn visit_enter_enum(&self,
 439                          _n_variants: uint,
 440                          get_disr: extern unsafe fn(ptr: *Opaque) -> int,
 441                          _sz: uint,
 442                          _align: uint) -> bool {
 443          let var_stk: &mut ~[VariantState] = self.var_stk;
 444          let disr = unsafe {
 445              get_disr(transmute(*self.ptr))
 446          };
 447          var_stk.push(SearchingFor(disr));
 448          true
 449      }

 451      fn visit_enter_enum_variant(&self, _variant: uint,
 452                                  disr_val: int,
 453                                  n_fields: uint,
 454                                  name: &str) -> bool {
 455          let mut write = false;
 456          match self.var_stk.pop() {
 457              SearchingFor(sought) => {
 458                  if disr_val == sought {
 459                      self.var_stk.push(Matched);
 460                      write = true;
 461                  } else {
 462                      self.var_stk.push(SearchingFor(sought));
 463                  }
 464              }
 465              Matched | AlreadyFound => {
 466                  self.var_stk.push(AlreadyFound);
 467              }
 468          }
 469  
 470          if write {
 471              self.writer.write_str(name);
 472              if n_fields > 0 {
 473                  self.writer.write_char('(');
 474              }
 475          }
 476          true
 477      }

 479      fn visit_enum_variant_field(&self,
 480                                  i: uint,
 481                                  _offset: uint,
 482                                  inner: *TyDesc)
 483                                  -> bool {
 484          match self.var_stk[self.var_stk.len() - 1] {

  28  //! A mostly lock-free multi-producer, single consumer queue.
  29  //!
  30  //! This module contains an implementation of a concurrent MPSC queue. This
  31  //! queue can be used to share data between tasks, and is also used as the
  32  //! building block of channels in rust.
  33  //!
  34  //! Note that the current implementation of this queue has a caveat of the `pop`
  35  //! method, and see the method for more information about it. Due to this
  36  //! caveat, this queue may not be appropriate for all use-cases.
  37  
  38  // http://www.1024cores.net/home/lock-free-algorithms
  39  //                         /queues/non-intrusive-mpsc-node-based-queue

  48  /// A result of the `pop` function.
  49  pub enum PopResult<T> {
  50      /// Some data has been popped
  51      Data(T),
  52      /// The queue is empty
  53      Empty,
  54      /// The queue is in an inconsistent state. Popping data should succeed, but
  55      /// some pushers have yet to make enough progress in order allow a pop to
  56      /// succeed. It is recommended that a pop() occur "in the near future" in
  57      /// order to see if the sender has made progress or not
  58      Inconsistent,
  59  }

  62      next: AtomicPtr<Node<T>>,

  66  /// The multi-producer single-consumer structure. This is not cloneable, but it
  67  /// may be safely shared so long as it is guaranteed that there is only one
  68  /// popper at a time (many pushers are allowed).
  69  pub struct Queue<T> {

  72  }

  94      /// Pushes a new value onto this queue.
  95      pub fn push(&mut self, t: T) {
  96          unsafe {
  97              let n = Node::new(Some(t));
  98              let prev = self.head.swap(n, AcqRel);
  99              (*prev).next.store(n, Release);
 100          }
 101      }

 103      /// Pops some data from this queue.
 104      ///
 105      /// Note that the current implementation means that this function cannot
 106      /// return `Option<T>`. It is possible for this queue to be in an
 107      /// inconsistent state where many pushes have succeeded and completely
 108      /// finished, but pops cannot return `Some(t)`. This inconsistent state
 109      /// happens when a pusher is pre-empted at an inopportune moment.
 110      ///
 111      /// This inconsistent state means that this queue does indeed have data, but
 112      /// it does not currently have access to it at this time.
 113      pub fn pop(&mut self) -> PopResult<T> {
 114          unsafe {
 115              let tail = self.tail;
 116              let next = (*tail).next.load(Acquire);
 117  
 118              if !next.is_null() {
 119                  self.tail = next;
 120                  assert!((*tail).value.is_none());
 121                  assert!((*next).value.is_some());
 122                  let ret = (*next).value.take_unwrap();
 123                  let _: ~Node<T> = cast::transmute(tail);
 124                  return Data(ret);
 125              }
 126  
 127              if self.head.load(Acquire) == tail {Empty} else {Inconsistent}
 128          }
 129      }
 130  
 131      /// Attempts to pop data from this queue, but doesn't attempt too hard. This
 132      /// will canonicalize inconsistent states to a `None` value.
 133      pub fn casual_pop(&mut self) -> Option<T> {
 134          match self.pop() {
 135              Data(t) => Some(t),
 136              Empty | Inconsistent => None,
 137          }
 138      }
 139  }

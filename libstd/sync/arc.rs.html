  11  //! Atomically reference counted data
  12  //!
  13  //! This modules contains the implementation of an atomically reference counted
  14  //! pointer for the purpose of sharing data between tasks. This is obviously a
  15  //! very unsafe primitive to use, but it has its use cases when implementing
  16  //! concurrent data structures and similar tasks.
  17  //!
  18  //! Great care must be taken to ensure that data races do not arise through the
  19  //! usage of `UnsafeArc`, and this often requires some form of external
  20  //! synchronization. The only guarantee provided to you by this class is that
  21  //! the underlying data will remain valid (not free'd) so long as the reference
  22  //! count is greater than one.
  23

  33  /// An atomically reference counted pointer.
  34  ///
  35  /// Enforces no shared-memory safety.
  36  #[unsafe_no_drop_flag]
  37  pub struct UnsafeArc<T> {
  38      data: *mut ArcData<T>,
  39  }

  44  }

  46  unsafe fn new_inner<T: Send>(data: T, refcount: uint) -> *mut ArcData<T> {
  47      let data = ~ArcData {
  48                      count: AtomicUint::new(refcount),
  49                      data: Unsafe::new(data)
  50                   };

  52  }

  54  impl<T: Send> UnsafeArc<T> {
  55      /// Creates a new `UnsafeArc` which wraps the given data.
  56      pub fn new(data: T) -> UnsafeArc<T> {
  57          unsafe { UnsafeArc { data: new_inner(data, 1) } }
  58      }

  60      /// As new(), but returns an extra pre-cloned handle.
  61      pub fn new2(data: T) -> (UnsafeArc<T>, UnsafeArc<T>) {
  62          unsafe {
  63              let ptr = new_inner(data, 2);
  64              (UnsafeArc { data: ptr }, UnsafeArc { data: ptr })
  65          }
  66      }
  67  
  68      /// As new(), but returns a vector of as many pre-cloned handles as
  69      /// requested.
  70      pub fn newN(data: T, num_handles: uint) -> ~[UnsafeArc<T>] {
  71          unsafe {
  72              if num_handles == 0 {
  73                  ~[] // need to free data here
  74              } else {
  75                  let ptr = new_inner(data, num_handles);
  76                  slice::from_fn(num_handles, |_| UnsafeArc { data: ptr })
  77              }
  78          }
  79      }

 113  impl<T: Send> Clone for UnsafeArc<T> {
 114      fn clone(&self) -> UnsafeArc<T> {
 115          unsafe {
 116              // Using a relaxed ordering is alright here, as knowledge of the original reference
 117              // prevents other threads from erroneously deleting the object.
 118              //
 119              // As explained in the [Boost documentation][1],
 120              //  Increasing the reference counter can always be done with memory_order_relaxed: New
 121              //  references to an object can only be formed from an existing reference, and passing
 122              //  an existing reference from one thread to another must already provide any required
 123              //  synchronization.
 124              // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
 125              let old_count = (*self.data).count.fetch_add(1, Relaxed);
 126              // FIXME(#12049): this needs some sort of debug assertion
 127              if cfg!(test) { assert!(old_count >= 1); }
 128              return UnsafeArc { data: self.data };
 129          }
 130      }
 131  }

 133  #[unsafe_destructor]
 134  impl<T> Drop for UnsafeArc<T>{
 135      fn drop(&mut self) {
 136          unsafe {
 137              // Happens when destructing an unwrapper's handle and from
 138              // `#[unsafe_no_drop_flag]`
 139              if self.data.is_null() {
 140                  return
 141              }
 142              // Because `fetch_sub` is already atomic, we do not need to synchronize with other
 143              // threads unless we are going to delete the object.
 144              let old_count = (*self.data).count.fetch_sub(1, Release);
 145              // FIXME(#12049): this needs some sort of debug assertion
 146              if cfg!(test) { assert!(old_count >= 1); }
 147              if old_count == 1 {
 148                  // This fence is needed to prevent reordering of use of the data and deletion of
 149                  // the data. Because it is marked `Release`, the decreasing of the reference count
 150                  // sychronizes with this `Acquire` fence. This means that use of the data happens
 151                  // before decreasing the refernce count, which happens before this fence, which
 152                  // happens before the deletion of the data.
 153                  //
 154                  // As explained in the [Boost documentation][1],
 155                  //  It is important to enforce any possible access to the object in one thread
 156                  //  (through an existing reference) to *happen before* deleting the object in a
 157                  //  different thread. This is achieved by a "release" operation after dropping a
 158                  //  reference (any access to the object through this reference must obviously
 159                  //  happened before), and an "acquire" operation before deleting the object.
 160                  // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
 161                  fence(Acquire);
 162                  let _: ~ArcData<T> = cast::transmute(self.data);
 163              }
 164          }
 165      }
 166  }

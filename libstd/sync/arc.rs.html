  11  //! Atomically reference counted data
  12  //!
  13  //! This modules contains the implementation of an atomically reference counted
  14  //! pointer for the purpose of sharing data between tasks. This is obviously a
  15  //! very unsafe primitive to use, but it has its use cases when implementing
  16  //! concurrent data structures and similar tasks.
  17  //!
  18  //! Great care must be taken to ensure that data races do not arise through the
  19  //! usage of `UnsafeArc`, and this often requires some form of external
  20  //! synchronization. The only guarantee provided to you by this class is that
  21  //! the underlying data will remain valid (not free'd) so long as the reference
  22  //! count is greater than one.
  23

  47  unsafe fn new_inner<T: Send>(data: T, refcount: uint) -> *mut ArcData<T> {
  48      let data = ~ArcData {
  49                      count: AtomicUint::new(refcount),
  50                      data: Unsafe::new(data)
  51                   };

  53  }

  69      /// As new(), but returns a vector of as many pre-cloned handles as
  70      /// requested.

  73              if num_handles == 0 {
  74                  ~[] // need to free data here

  77                  let v = Vec::from_fn(num_handles, |_| UnsafeArc { data: ptr });

  80          }
  81      }

 115  impl<T: Send> Clone for UnsafeArc<T> {
 116      fn clone(&self) -> UnsafeArc<T> {
 117          unsafe {
 118              // Using a relaxed ordering is alright here, as knowledge of the original reference
 119              // prevents other threads from erroneously deleting the object.
 120              //
 121              // As explained in the [Boost documentation][1],
 122              //  Increasing the reference counter can always be done with memory_order_relaxed: New
 123              //  references to an object can only be formed from an existing reference, and passing
 124              //  an existing reference from one thread to another must already provide any required
 125              //  synchronization.
 126              // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
 127              let old_count = (*self.data).count.fetch_add(1, Relaxed);
 128              // FIXME(#12049): this needs some sort of debug assertion
 129              if cfg!(test) { assert!(old_count >= 1); }
 130              return UnsafeArc { data: self.data };
 131          }
 132      }
 133  }

 135  #[unsafe_destructor]
 136  impl<T> Drop for UnsafeArc<T>{
 137      fn drop(&mut self) {
 138          unsafe {
 139              // Happens when destructing an unwrapper's handle and from
 140              // `#[unsafe_no_drop_flag]`
 141              if self.data.is_null() {
 142                  return
 143              }
 144              // Because `fetch_sub` is already atomic, we do not need to synchronize with other
 145              // threads unless we are going to delete the object.
 146              let old_count = (*self.data).count.fetch_sub(1, Release);
 147              // FIXME(#12049): this needs some sort of debug assertion
 148              if cfg!(test) { assert!(old_count >= 1); }
 149              if old_count == 1 {
 150                  // This fence is needed to prevent reordering of use of the data and deletion of
 151                  // the data. Because it is marked `Release`, the decreasing of the reference count
 152                  // sychronizes with this `Acquire` fence. This means that use of the data happens
 153                  // before decreasing the refernce count, which happens before this fence, which
 154                  // happens before the deletion of the data.
 155                  //
 156                  // As explained in the [Boost documentation][1],
 157                  //  It is important to enforce any possible access to the object in one thread
 158                  //  (through an existing reference) to *happen before* deleting the object in a
 159                  //  different thread. This is achieved by a "release" operation after dropping a
 160                  //  reference (any access to the object through this reference must obviously
 161                  //  happened before), and an "acquire" operation before deleting the object.
 162                  // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
 163                  fence(Acquire);
 164                  let _: ~ArcData<T> = cast::transmute(self.data);
 165              }
 166          }
 167      }
 168  }

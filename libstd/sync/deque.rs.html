  11  //! A (mostly) lock-free concurrent work-stealing deque
  12  //!
  13  //! This module contains an implementation of the Chase-Lev work stealing deque
  14  //! described in "Dynamic Circular Work-Stealing Deque". The implementation is
  15  //! heavily based on the pseudocode found in the paper.
  16  //!
  17  //! This implementation does not want to have the restriction of a garbage
  18  //! collector for reclamation of buffers, and instead it uses a shared pool of
  19  //! buffers. This shared pool is required for correctness in this
  20  //! implementation.
  21  //!
  22  //! The only lock-synchronized portions of this deque are the buffer allocation
  23  //! and deallocation portions. Otherwise all operations are lock-free.
  24  //!
  25  //! # Example
  26  //!
  27  //!     use std::rt::deque::BufferPool;
  28  //!
  29  //!     let mut pool = BufferPool::new();
  30  //!     let (mut worker, mut stealer) = pool.deque();
  31  //!
  32  //!     // Only the worker may push/pop
  33  //!     worker.push(1);
  34  //!     worker.pop();
  35  //!
  36  //!     // Stealers take data from the other end of the deque
  37  //!     worker.push(1);
  38  //!     stealer.steal();
  39  //!
  40  //!     // Stealers can be cloned to have many stealers stealing in parallel
  41  //!     worker.push(1);
  42  //!     let mut stealer2 = stealer.clone();
  43  //!     stealer2.steal();
  44  
  45  // NB: the "buffer pool" strategy is not done for speed, but rather for
  46  //     correctness. For more info, see the comment on `swap_buffer`
  47  
  48  // FIXME: all atomic operations in this module use a SeqCst ordering. That is
  49  //      probably overkill

  66  // Once the queue is less than 1/K full, then it will be downsized. Note that
  67  // the deque requires that this number be less than 2.
  68  static K: int = 4;
  69  
  70  // Minimum number of bits that a buffer size should be. No buffer will resize to
  71  // under this value, and all deques will initially contain a buffer of this
  72  // size.
  73  //
  74  // The size in question is 1 << MIN_BITS
  75  static MIN_BITS: int = 7;
  76  
  77  struct Deque<T> {
  78      bottom: AtomicInt,
  79      top: AtomicInt,
  80      array: AtomicPtr<Buffer<T>>,
  81      pool: BufferPool<T>,
  82  }
  83  
  84  /// Worker half of the work-stealing deque. This worker has exclusive access to
  85  /// one side of the deque, and uses `push` and `pop` method to manipulate it.
  86  ///
  87  /// There may only be one worker per deque.
  88  pub struct Worker<T> {
  89      deque: UnsafeArc<Deque<T>>,
  90  }
  91  
  92  /// The stealing half of the work-stealing deque. Stealers have access to the
  93  /// opposite end of the deque from the worker, and they only have access to the
  94  /// `steal` method.
  95  pub struct Stealer<T> {
  96      deque: UnsafeArc<Deque<T>>,
  97  }
  98  
  99  /// When stealing some data, this is an enumeration of the possible outcomes.
 100  #[deriving(Eq, Show)]
 101  pub enum Stolen<T> {
 102      /// The deque was empty at the time of stealing
 103      Empty,
 104      /// The stealer lost the race for stealing data, and a retry may return more
 105      /// data.
 106      Abort,
 107      /// The stealer has successfully stolen some data.
 108      Data(T),
 109  }
 110  
 111  /// The allocation pool for buffers used by work-stealing deques. Right now this
 112  /// structure is used for reclamation of memory after it is no longer in use by
 113  /// deques.
 114  ///
 115  /// This data structure is protected by a mutex, but it is rarely used. Deques
 116  /// will only use this structure when allocating a new buffer or deallocating a
 117  /// previous one.
 118  pub struct BufferPool<T> {
 119      pool: Exclusive<~[~Buffer<T>]>,
 120  }
 121  
 122  /// An internal buffer used by the chase-lev deque. This structure is actually
 123  /// implemented as a circular buffer, and is used as the intermediate storage of
 124  /// the data in the deque.
 125  ///
 126  /// This type is implemented with *T instead of ~[T] for two reasons:
 127  ///
 128  ///   1. There is nothing safe about using this buffer. This easily allows the
 129  ///      same value to be read twice in to rust, and there is nothing to
 130  ///      prevent this. The usage by the deque must ensure that one of the
 131  ///      values is forgotten. Furthermore, we only ever want to manually run
 132  ///      destructors for values in this buffer (on drop) because the bounds
 133  ///      are defined by the deque it's owned by.
 134  ///
 135  ///   2. We can certainly avoid bounds checks using *T instead of ~[T], although
 136  ///      LLVM is probably pretty good at doing this already.
 137  struct Buffer<T> {
 138      storage: *T,
 139      log_size: int,
 140  }

 142  impl<T: Send> BufferPool<T> {
 143      /// Allocates a new buffer pool which in turn can be used to allocate new
 144      /// deques.
 145      pub fn new() -> BufferPool<T> {
 146          BufferPool { pool: Exclusive::new(~[]) }
 147      }
 148  
 149      /// Allocates a new work-stealing deque which will send/receiving memory to
 150      /// and from this buffer pool.
 151      pub fn deque(&mut self) -> (Worker<T>, Stealer<T>) {
 152          let (a, b) = UnsafeArc::new2(Deque::new(self.clone()));
 153          (Worker { deque: a }, Stealer { deque: b })
 154      }
 155  
 156      fn alloc(&mut self, bits: int) -> ~Buffer<T> {
 157          unsafe {
 158              self.pool.with(|pool| {
 159                  match pool.iter().position(|x| x.size() >= (1 << bits)) {
 160                      Some(i) => pool.remove(i).unwrap(),
 161                      None => ~Buffer::new(bits)
 162                  }
 163              })
 164          }
 165      }
 166  
 167      fn free(&mut self, buf: ~Buffer<T>) {
 168          unsafe {
 169              let mut buf = Some(buf);
 170              self.pool.with(|pool| {
 171                  let buf = buf.take_unwrap();
 172                  match pool.iter().position(|v| v.size() > buf.size()) {
 173                      Some(i) => pool.insert(i, buf),
 174                      None => pool.push(buf),
 175                  }
 176              })
 177          }
 178      }
 179  }

 183  }

 222  // Almost all of this code can be found directly in the paper so I'm not
 223  // personally going to heavily comment what's going on here.
 224  
 225  impl<T: Send> Deque<T> {
 226      fn new(mut pool: BufferPool<T>) -> Deque<T> {
 227          let buf = pool.alloc(MIN_BITS);
 228          Deque {
 229              bottom: AtomicInt::new(0),
 230              top: AtomicInt::new(0),
 231              array: AtomicPtr::new(unsafe { cast::transmute(buf) }),
 232              pool: pool,
 233          }
 234      }
 235  
 236      unsafe fn push(&mut self, data: T) {
 237          let mut b = self.bottom.load(SeqCst);
 238          let t = self.top.load(SeqCst);
 239          let mut a = self.array.load(SeqCst);
 240          let size = b - t;
 241          if size >= (*a).size() - 1 {
 242              // You won't find this code in the chase-lev deque paper. This is
 243              // alluded to in a small footnote, however. We always free a buffer
 244              // when growing in order to prevent leaks.
 245              a = self.swap_buffer(b, a, (*a).resize(b, t, 1));
 246              b = self.bottom.load(SeqCst);
 247          }
 248          (*a).put(b, data);
 249          self.bottom.store(b + 1, SeqCst);
 250      }
 251  
 252      unsafe fn pop(&mut self) -> Option<T> {
 253          let b = self.bottom.load(SeqCst);
 254          let a = self.array.load(SeqCst);
 255          let b = b - 1;
 256          self.bottom.store(b, SeqCst);
 257          let t = self.top.load(SeqCst);
 258          let size = b - t;
 259          if size < 0 {
 260              self.bottom.store(t, SeqCst);
 261              return None;
 262          }
 263          let data = (*a).get(b);
 264          if size > 0 {
 265              self.maybe_shrink(b, t);
 266              return Some(data);
 267          }
 268          if self.top.compare_and_swap(t, t + 1, SeqCst) == t {
 269              self.bottom.store(t + 1, SeqCst);
 270              return Some(data);
 271          } else {
 272              self.bottom.store(t + 1, SeqCst);
 273              cast::forget(data); // someone else stole this value
 274              return None;
 275          }
 276      }
 277  
 278      unsafe fn steal(&mut self) -> Stolen<T> {
 279          let t = self.top.load(SeqCst);
 280          let old = self.array.load(SeqCst);
 281          let b = self.bottom.load(SeqCst);
 282          let a = self.array.load(SeqCst);
 283          let size = b - t;
 284          if size <= 0 { return Empty }
 285          if size % (*a).size() == 0 {
 286              if a == old && t == self.top.load(SeqCst) {
 287                  return Empty
 288              }
 289              return Abort
 290          }
 291          let data = (*a).get(t);
 292          if self.top.compare_and_swap(t, t + 1, SeqCst) == t {
 293              Data(data)
 294          } else {
 295              cast::forget(data); // someone else stole this value
 296              Abort
 297          }
 298      }
 299  
 300      unsafe fn maybe_shrink(&mut self, b: int, t: int) {
 301          let a = self.array.load(SeqCst);
 302          if b - t < (*a).size() / K && b - t > (1 << MIN_BITS) {
 303              self.swap_buffer(b, a, (*a).resize(b, t, -1));
 304          }
 305      }
 306  
 307      // Helper routine not mentioned in the paper which is used in growing and
 308      // shrinking buffers to swap in a new buffer into place. As a bit of a
 309      // recap, the whole point that we need a buffer pool rather than just
 310      // calling malloc/free directly is that stealers can continue using buffers
 311      // after this method has called 'free' on it. The continued usage is simply
 312      // a read followed by a forget, but we must make sure that the memory can
 313      // continue to be read after we flag this buffer for reclamation.
 314      unsafe fn swap_buffer(&mut self, b: int, old: *mut Buffer<T>,
 315                            buf: Buffer<T>) -> *mut Buffer<T> {
 316          let newbuf: *mut Buffer<T> = cast::transmute(~buf);
 317          self.array.store(newbuf, SeqCst);
 318          let ss = (*newbuf).size();
 319          self.bottom.store(b + ss, SeqCst);
 320          let t = self.top.load(SeqCst);
 321          if self.top.compare_and_swap(t, t + ss, SeqCst) != t {
 322              self.bottom.store(b, SeqCst);
 323          }
 324          self.pool.free(cast::transmute(old));
 325          return newbuf;
 326      }
 327  }

 330  #[unsafe_destructor]
 331  impl<T: Send> Drop for Deque<T> {
 332      fn drop(&mut self) {
 333          let t = self.top.load(SeqCst);
 334          let b = self.bottom.load(SeqCst);
 335          let a = self.array.load(SeqCst);
 336          // Free whatever is leftover in the dequeue, and then move the buffer
 337          // back into the pool.
 338          for i in range(t, b) {
 339              let _: T = unsafe { (*a).get(i) };
 340          }
 341          self.pool.free(unsafe { cast::transmute(a) });
 342      }
 343  }

 345  impl<T: Send> Buffer<T> {
 346      unsafe fn new(log_size: int) -> Buffer<T> {
 347          let size = (1 << log_size) * mem::size_of::<T>();
 348          let buffer = libc::malloc(size as libc::size_t);
 349          assert!(!buffer.is_null());
 350          Buffer {
 351              storage: buffer as *T,
 352              log_size: log_size,
 353          }
 354      }
 355  
 356      fn size(&self) -> int { 1 << self.log_size }
 357  
 358      // Apparently LLVM cannot optimize (foo % (1 << bar)) into this implicitly
 359      fn mask(&self) -> int { (1 << self.log_size) - 1 }
 360  
 361      // This does not protect against loading duplicate values of the same cell,
 362      // nor does this clear out the contents contained within. Hence, this is a
 363      // very unsafe method which the caller needs to treat specially in case a
 364      // race is lost.
 365      unsafe fn get(&self, i: int) -> T {
 366          ptr::read(self.storage.offset(i & self.mask()))
 367      }
 368  
 369      // Unsafe because this unsafely overwrites possibly uninitialized or
 370      // initialized data.
 371      unsafe fn put(&mut self, i: int, t: T) {
 372          let ptr = self.storage.offset(i & self.mask());
 373          ptr::copy_nonoverlapping_memory(ptr as *mut T, &t as *T, 1);
 374          cast::forget(t);
 375      }
 376  
 377      // Again, unsafe because this has incredibly dubious ownership violations.
 378      // It is assumed that this buffer is immediately dropped.
 379      unsafe fn resize(&self, b: int, t: int, delta: int) -> Buffer<T> {
 380          let mut buf = Buffer::new(self.log_size + delta);
 381          for i in range(t, b) {
 382              buf.put(i, self.get(i));
 383          }
 384          return buf;
 385      }
 386  }

 388  #[unsafe_destructor]
 389  impl<T: Send> Drop for Buffer<T> {
 390      fn drop(&mut self) {
 391          // It is assumed that all buffers are empty on drop.
 392          unsafe { libc::free(self.storage as *mut libc::c_void) }
 393      }
 394  }

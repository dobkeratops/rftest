  30  //! A single-producer single-consumer concurrent queue
  31  //!
  32  //! This module contains the implementation of an SPSC queue which can be used
  33  //! concurrently between two tasks. This data structure is safe to use and
  34  //! enforces the semantics that there is one pusher and one popper.

  53  /// but it can be safely shared in an UnsafeArc if it is guaranteed that there
  54  /// is only one popper and one pusher touching the queue at any one point in
  55  /// time.

  84  impl<T: Send> Queue<T> {
  85      /// Creates a new queue. The producer returned is connected to the consumer
  86      /// to push all data to the consumer.
  87      ///
  88      /// # Arguments
  89      ///
  90      ///   * `bound` - This queue implementation is implemented with a linked
  91      ///               list, and this means that a push is always a malloc. In
  92      ///               order to amortize this cost, an internal cache of nodes is
  93      ///               maintained to prevent a malloc from always being
  94      ///               necessary. This bound is the limit on the size of the
  95      ///               cache (if desired). If the value is 0, then the cache has
  96      ///               no bound. Otherwise, the cache will never grow larger than
  97      ///               `bound` (although the queue itself could be much larger.
  98      pub fn new(bound: uint) -> Queue<T> {
  99          let n1 = Node::new();
 100          let n2 = Node::new();
 101          unsafe { (*n1).next.store(n2, Relaxed) }
 102          Queue {
 103              tail: n2,
 104              tail_prev: AtomicPtr::new(n1),
 105              head: n2,
 106              first: n1,
 107              tail_copy: n1,
 108              cache_bound: bound,
 109              cache_additions: AtomicUint::new(0),
 110              cache_subtractions: AtomicUint::new(0),
 111          }
 112      }
 113  
 114      /// Pushes a new value onto this queue. Note that to use this function
 115      /// safely, it must be externally guaranteed that there is only one pusher.
 116      pub fn push(&mut self, t: T) {
 117          unsafe {
 118              // Acquire a node (which either uses a cached one or allocates a new
 119              // one), and then append this to the 'head' node.
 120              let n = self.alloc();
 121              assert!((*n).value.is_none());
 122              (*n).value = Some(t);
 123              (*n).next.store(0 as *mut Node<T>, Relaxed);
 124              (*self.head).next.store(n, Release);
 125              self.head = n;
 126          }
 127      }
 128  
 129      unsafe fn alloc(&mut self) -> *mut Node<T> {
 130          // First try to see if we can consume the 'first' node for our uses.
 131          // We try to avoid as many atomic instructions as possible here, so
 132          // the addition to cache_subtractions is not atomic (plus we're the
 133          // only one subtracting from the cache).
 134          if self.first != self.tail_copy {
 135              if self.cache_bound > 0 {
 136                  let b = self.cache_subtractions.load(Relaxed);
 137                  self.cache_subtractions.store(b + 1, Relaxed);
 138              }
 139              let ret = self.first;
 140              self.first = (*ret).next.load(Relaxed);
 141              return ret;
 142          }
 143          // If the above fails, then update our copy of the tail and try
 144          // again.
 145          self.tail_copy = self.tail_prev.load(Acquire);
 146          if self.first != self.tail_copy {
 147              if self.cache_bound > 0 {
 148                  let b = self.cache_subtractions.load(Relaxed);
 149                  self.cache_subtractions.store(b + 1, Relaxed);
 150              }
 151              let ret = self.first;
 152              self.first = (*ret).next.load(Relaxed);
 153              return ret;
 154          }
 155          // If all of that fails, then we have to allocate a new node
 156          // (there's nothing in the node cache).
 157          Node::new()
 158      }

 160      /// Attempts to pop a value from this queue. Remember that to use this type
 161      /// safely you must ensure that there is only one popper at a time.
 162      pub fn pop(&mut self) -> Option<T> {
 163          unsafe {
 164              // The `tail` node is not actually a used node, but rather a
 165              // sentinel from where we should start popping from. Hence, look at
 166              // tail's next field and see if we can use it. If we do a pop, then
 167              // the current tail node is a candidate for going into the cache.
 168              let tail = self.tail;
 169              let next = (*tail).next.load(Acquire);
 170              if next.is_null() { return None }
 171              assert!((*next).value.is_some());
 172              let ret = (*next).value.take();
 173  
 174              self.tail = next;
 175              if self.cache_bound == 0 {
 176                  self.tail_prev.store(tail, Release);
 177              } else {
 178                  // FIXME: this is dubious with overflow.
 179                  let additions = self.cache_additions.load(Relaxed);
 180                  let subtractions = self.cache_subtractions.load(Relaxed);
 181                  let size = additions - subtractions;
 182  
 183                  if size < self.cache_bound {
 184                      self.tail_prev.store(tail, Release);
 185                      self.cache_additions.store(additions + 1, Relaxed);
 186                  } else {
 187                      (*self.tail_prev.load(Relaxed)).next.store(next, Relaxed);
 188                      // We have successfully erased all references to 'tail', so
 189                      // now we can safely drop it.
 190                      let _: ~Node<T> = cast::transmute(tail);
 191                  }
 192              }
 193              return ret;
 194          }
 195      }

 197      /// Attempts to peek at the head of the queue, returning `None` if the queue
 198      /// has no data currently
 199      pub fn peek<'a>(&'a mut self) -> Option<&'a mut T> {
 200          // This is essentially the same as above with all the popping bits
 201          // stripped out.
 202          unsafe {
 203              let tail = self.tail;
 204              let next = (*tail).next.load(Acquire);
 205              if next.is_null() { return None }
 206              return (*next).value.as_mut();
 207          }
 208      }

  87  ///
  88  /// Indicates whether a character is whitespace. Whitespace is defined in
  89  /// terms of the Unicode General Categories 'Zs', 'Zl', 'Zp'
  90  /// additional 'Cc'-category control codes in the range [0x09, 0x0d]
  91  ///
  92  #[inline]
  93  pub fn is_whitespace(c: char) -> bool {
  94      c == ' '
  95          || ('\x09' <= c && c <= '\x0d')
  96          || general_category::Zs(c)
  97          || general_category::Zl(c)
  98          || general_category::Zp(c)
  99  }

 126  ///
 127  /// # Return value
 128  ///
 129  /// Returns `true` if `c` is a valid digit under `radix`, and `false`
 130  /// otherwise.
 131  ///
 132  /// # Failure
 133  ///
 134  /// Fails if given a `radix` > 36.
 135  ///
 136  /// # Note
 137  ///
 138  /// This just wraps `to_digit()`.
 139  ///
 140  #[inline]
 141  pub fn is_digit_radix(c: char, radix: uint) -> bool {

 143          Some(_) => true,
 144          None    => false,
 145      }
 146  }

 148  ///
 149  /// Convert a char to the corresponding digit.
 150  ///
 151  /// # Return value
 152  ///
 153  /// If `c` is between '0' and '9', the corresponding value
 154  /// between 0 and 9. If `c` is 'a' or 'A', 10. If `c` is
 155  /// 'b' or 'B', 11, etc. Returns none if the char does not
 156  /// refer to a digit in the given radix.
 157  ///
 158  /// # Failure
 159  ///
 160  /// Fails if given a `radix` outside the range `[0..36]`.
 161  ///
 162  #[inline]
 163  pub fn to_digit(c: char, radix: uint) -> Option<uint> {
 164      if radix > 36 {
 165          fail!("to_digit: radix %? is to high (maximum 36)", radix);
 166      }
 167      let val = match c {
 168        '0' .. '9' => c as uint - ('0' as uint),
 169        'a' .. 'z' => c as uint + 10u - ('a' as uint),
 170        'A' .. 'Z' => c as uint + 10u - ('A' as uint),
 171        _ => return None,
 172      };
 173      if val < radix { Some(val) }
 174      else { None }
 175  }

 177  ///
 178  /// Converts a number to the character representing it.
 179  ///
 180  /// # Return value
 181  ///
 182  /// Returns `Some(char)` if `num` represents one digit under `radix`,
 183  /// using one character of `0-9` or `a-z`, or `None` if it doesn't.
 184  ///
 185  /// # Failure
 186  ///
 187  /// Fails if given an `radix` > 36.
 188  ///
 189  #[inline]
 190  pub fn from_digit(num: uint, radix: uint) -> Option<char> {
 191      if radix > 36 {
 192          fail!("from_digit: radix %? is to high (maximum 36)", num);
 193      }
 194      if num < radix {
 195          if num < 10 {
 196              Some(('0' as uint + num) as char)
 197          } else {
 198              Some(('a' as uint + num - 10u) as char)
 199          }
 200      } else {
 201          None
 202      }
 203  }

 205  ///
 206  /// Return the hexadecimal unicode escape of a char.
 207  ///
 208  /// The rules are as follows:
 209  ///
 210  /// - chars in [0,0xff] get 2-digit escapes: `\\xNN`
 211  /// - chars in [0x100,0xffff] get 4-digit escapes: `\\uNNNN`
 212  /// - chars above 0x10000 get 8-digit escapes: `\\UNNNNNNNN`
 213  ///
 214  pub fn escape_unicode(c: char, f: &fn(char)) {
 215      // avoid calling str::to_str_radix because we don't really need to allocate
 216      // here.
 217      f('\\');
 218      let pad = cond!(
 219          (c <= '\xff')   { f('x'); 2 }
 220          (c <= '\uffff') { f('u'); 4 }
 221          _               { f('U'); 8 }
 222      );
 223      do int::range_step(4 * (pad - 1), -1, -4) |offset| {
 224          match ((c as u32) >> offset) & 0xf {
 225              i @ 0 .. 9 => { f('0' + i as char); }
 226              i => { f('a' + (i - 10) as char); }
 227          }
 228          true
 229      };
 230  }

 232  ///
 233  /// Return a 'default' ASCII and C++11-like char-literal escape of a char.
 234  ///
 235  /// The default is chosen with a bias toward producing literals that are
 236  /// legal in a variety of languages, including C++11 and similar C-family
 237  /// languages. The exact rules are:
 238  ///
 239  /// - Tab, CR and LF are escaped as '\t', '\r' and '\n' respectively.
 240  /// - Single-quote, double-quote and backslash chars are backslash-escaped.
 241  /// - Any other chars in the range [0x20,0x7e] are not escaped.
 242  /// - Any other chars are given hex unicode escapes; see `escape_unicode`.
 243  ///
 244  pub fn escape_default(c: char, f: &fn(char)) {
 245      match c {
 246          '\t' => { f('\\'); f('t'); }
 247          '\r' => { f('\\'); f('r'); }
 248          '\n' => { f('\\'); f('n'); }
 249          '\\' => { f('\\'); f('\\'); }
 250          '\'' => { f('\\'); f('\''); }
 251          '"'  => { f('\\'); f('"'); }
 252          '\x20' .. '\x7e' => { f(c); }
 253          _ => c.escape_unicode(f),
 254      }
 255  }

 257  /// Returns the amount of bytes this character would need if encoded in utf8
 258  pub fn len_utf8_bytes(c: char) -> uint {
 259      static MAX_ONE_B:   uint = 128u;
 260      static MAX_TWO_B:   uint = 2048u;
 261      static MAX_THREE_B: uint = 65536u;
 262      static MAX_FOUR_B:  uint = 2097152u;
 263  
 264      let code = c as uint;
 265      cond!(
 266          (code < MAX_ONE_B)   { 1u }
 267          (code < MAX_TWO_B)   { 2u }
 268          (code < MAX_THREE_B) { 3u }
 269          (code < MAX_FOUR_B)  { 4u }
 270          _ { fail!("invalid character!") }
 271      )
 272  }

 327      fn encode_utf8<'a>(&self, dst: &'a mut [u8]) -> uint {
 328          let code = *self as uint;
 329          if code < MAX_ONE_B {
 330              dst[0] = code as u8;
 331              return 1;
 332          } else if code < MAX_TWO_B {
 333              dst[0] = (code >> 6u & 31u | TAG_TWO_B) as u8;
 334              dst[1] = (code & 63u | TAG_CONT) as u8;
 335              return 2;
 336          } else if code < MAX_THREE_B {
 337              dst[0] = (code >> 12u & 15u | TAG_THREE_B) as u8;
 338              dst[1] = (code >> 6u & 63u | TAG_CONT) as u8;
 339              dst[2] = (code & 63u | TAG_CONT) as u8;
 340              return 3;
 341          } else {
 342              dst[0] = (code >> 18u & 7u | TAG_FOUR_B) as u8;
 343              dst[1] = (code >> 12u & 63u | TAG_CONT) as u8;
 344              dst[2] = (code >> 6u & 63u | TAG_CONT) as u8;
 345              dst[3] = (code & 63u | TAG_CONT) as u8;
 346              return 4;
 347          }
 348      }
 349  }

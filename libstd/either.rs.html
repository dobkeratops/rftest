  26 /// The either type
  27 #[deriving(Clone, Eq)]
  28 pub enum Either<T, U> {
  29     Left(T),
  30     Right(U)

  33 /// Applies a function based on the given either value
  34 ///
  35 /// If `value` is left(T) then `f_left` is applied to its contents, if
  36 /// `value` is right(U) then `f_right` is applied to its contents, and the
  37 /// result is returned.
  38 #[inline]
  39 pub fn either<T, U, V>(f_left: &fn(&T) -> V,
  40                        f_right: &fn(&U) -> V, value: &Either<T, U>) -> V {
  41     match *value {
  42         Left(ref l) => f_left(l),
  43         Right(ref r) => f_right(r)
  44     }
  45 }

  47 /// Extracts from a vector of either all the left values
  48 pub fn lefts<T:Clone,U>(eithers: &[Either<T, U>]) -> ~[T] {
  49     do vec::build_sized(eithers.len()) |push| {
  50         for elt in eithers.iter() {
  51             match *elt {
  52                 Left(ref l) => { push((*l).clone()); }
  53                 _ => { /* fallthrough */ }
  54             }
  55         }
  56     }
  57 }
  58 
  59 /// Extracts from a vector of either all the right values
  60 pub fn rights<T, U: Clone>(eithers: &[Either<T, U>]) -> ~[U] {
  61     do vec::build_sized(eithers.len()) |push| {
  62         for elt in eithers.iter() {
  63             match *elt {
  64                 Right(ref r) => { push((*r).clone()); }
  65                 _ => { /* fallthrough */ }
  66             }
  67         }
  68     }
  69 }

  71 /// Extracts from a vector of either all the left values and right values
  72 ///
  73 /// Returns a structure containing a vector of left values and a vector of
  74 /// right values.
  75 pub fn partition<T, U>(eithers: ~[Either<T, U>]) -> (~[T], ~[U]) {
  76     let mut lefts: ~[T] = ~[];
  77     let mut rights: ~[U] = ~[];
  78     for elt in eithers.consume_iter() {
  79         match elt {
  80             Left(l) => lefts.push(l),
  81             Right(r) => rights.push(r)
  82         }
  83     }
  84     return (lefts, rights);
  85 }

  87 /// Flips between left and right of a given either
  88 #[inline]
  89 pub fn flip<T, U>(eith: Either<T, U>) -> Either<U, T> {
  90     match eith {
  91         Right(r) => Left(r),
  92         Left(l) => Right(l)
  93     }
  94 }

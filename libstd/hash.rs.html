  12   * Implementation of SipHash 2-4
  13   *
  14   * See: http://131002.net/siphash/
  15   *
  16   * Consider this as a main "general-purpose" hash for all hashtables: it

  36  /**
  37   * Types that can meaningfully be hashed should implement this.
  38   *
  39   * Note that this trait is likely to change somewhat as it is
  40   * closely related to `to_bytes::IterBytes` and in almost all
  41   * cases presently the two are (and must be) used together.
  42   *
  43   * In general, most types only need to implement `IterBytes`,
  44   * and the implementation of `Hash` below will take care of
  45   * the rest. This is the recommended approach, since constructing
  46   * good keyed hash functions is quite difficult.
  47   */
  48  pub trait Hash {
  49      /**
  50       * Compute a "keyed" hash of the value implementing the trait,
  51       * taking `k0` and `k1` as "keying" parameters that randomize or
  52       * otherwise perturb the hash function in such a way that a
  53       * hash table built using such "keyed hash functions" cannot
  54       * be made to perform linearly by an attacker controlling the
  55       * hashtable's contents.
  56       *
  57       * In practical terms, we implement this using the SipHash 2-4
  58       * function and require most types to only implement the
  59       * IterBytes trait, that feeds SipHash.
  60       */
  61      fn hash_keyed(&self, k0: u64, k1: u64) -> u64;
  62  }

 202  struct SipState {
 203      k0: u64,
 204      k1: u64,
 205      length: uint, // how many bytes we've processed
 206      v0: u64,      // hash state
 207      v1: u64,
 208      v2: u64,
 209      v3: u64,
 210      tail: [u8, ..8], // unprocessed bytes
 211      ntail: uint,  // how many bytes in tail are valid
 212  }

 223              v2: 0,

 233  // sadly, these macro definitions can't appear later,
 234  // because they're needed in the following defs;
 235  // this design could be improved.
 236  
 237  macro_rules! u8to64_le (
 238      ($buf:expr, $i:expr) =>
 239      ($buf[0+$i] as u64 |
 240       $buf[1+$i] as u64 << 8 |
 241       $buf[2+$i] as u64 << 16 |
 242       $buf[3+$i] as u64 << 24 |
 243       $buf[4+$i] as u64 << 32 |
 244       $buf[5+$i] as u64 << 40 |
 245       $buf[6+$i] as u64 << 48 |
 246       $buf[7+$i] as u64 << 56)
 247  )
 248  
 249  macro_rules! rotl (
 250      ($x:expr, $b:expr) =>
 251      (($x << $b) | ($x >> (64 - $b)))
 252  )
 253  
 254  macro_rules! compress (
 255      ($v0:expr, $v1:expr, $v2:expr, $v3:expr) =>
 256      ({
 257          $v0 += $v1; $v1 = rotl!($v1, 13); $v1 ^= $v0;
 258          $v0 = rotl!($v0, 32);
 259          $v2 += $v3; $v3 = rotl!($v3, 16); $v3 ^= $v2;
 260          $v0 += $v3; $v3 = rotl!($v3, 21); $v3 ^= $v0;
 261          $v2 += $v1; $v1 = rotl!($v1, 17); $v1 ^= $v2;
 262          $v2 = rotl!($v2, 32);
 263      })
 264  )

 267  impl Writer for SipState {
 268      // Methods for io::writer
 269      #[inline]
 270      fn write(&mut self, msg: &[u8]) {
 271          let length = msg.len();
 272          self.length += length;
 273  
 274          let mut needed = 0u;
 275  
 276          if self.ntail != 0 {
 277              needed = 8 - self.ntail;
 278  
 279              if length < needed {
 280                  let mut t = 0;
 281                  while t < length {
 282                      self.tail[self.ntail+t] = msg[t];
 283                      t += 1;
 284                  }
 285                  self.ntail += length;
 286                  return;
 287              }
 288  
 289              let mut t = 0;
 290              while t < needed {
 291                  self.tail[self.ntail+t] = msg[t];
 292                  t += 1;
 293              }
 294  
 295              let m = u8to64_le!(self.tail, 0);
 296  
 297              self.v3 ^= m;
 298              compress!(self.v0, self.v1, self.v2, self.v3);
 299              compress!(self.v0, self.v1, self.v2, self.v3);
 300              self.v0 ^= m;
 301  
 302              self.ntail = 0;
 303          }
 304  
 305          // Buffered tail is now flushed, process new input.
 306          let len = length - needed;
 307          let end = len & (!0x7);
 308          let left = len & 0x7;
 309  
 310          let mut i = needed;
 311          while i < end {
 312              let mi = u8to64_le!(msg, i);
 313  
 314              self.v3 ^= mi;
 315              compress!(self.v0, self.v1, self.v2, self.v3);
 316              compress!(self.v0, self.v1, self.v2, self.v3);
 317              self.v0 ^= mi;
 318  
 319              i += 8;
 320          }
 321  
 322          let mut t = 0u;
 323          while t < left {
 324              self.tail[t] = msg[i+t];
 325              t += 1
 326          }
 327          self.ntail = left;
 328      }
 329

 331          // No-op
 332      }
 333  }

 341      #[inline]
 342      fn result_u64(&mut self) -> u64 {
 343          let mut v0 = self.v0;
 344          let mut v1 = self.v1;
 345          let mut v2 = self.v2;
 346          let mut v3 = self.v3;
 347  
 348          let mut b : u64 = (self.length as u64 & 0xff) << 56;
 349  
 350          if self.ntail > 0 { b |= self.tail[0] as u64 <<  0; }
 351          if self.ntail > 1 { b |= self.tail[1] as u64 <<  8; }
 352          if self.ntail > 2 { b |= self.tail[2] as u64 << 16; }
 353          if self.ntail > 3 { b |= self.tail[3] as u64 << 24; }
 354          if self.ntail > 4 { b |= self.tail[4] as u64 << 32; }
 355          if self.ntail > 5 { b |= self.tail[5] as u64 << 40; }
 356          if self.ntail > 6 { b |= self.tail[6] as u64 << 48; }
 357  
 358          v3 ^= b;
 359          compress!(v0, v1, v2, v3);
 360          compress!(v0, v1, v2, v3);
 361          v0 ^= b;
 362  
 363          v2 ^= 0xff;
 364          compress!(v0, v1, v2, v3);
 365          compress!(v0, v1, v2, v3);
 366          compress!(v0, v1, v2, v3);
 367          compress!(v0, v1, v2, v3);
 368  
 369          return (v0 ^ v1 ^ v2 ^ v3);
 370      }

 372      fn result_bytes(&mut self) -> ~[u8] {
 373          let h = self.result_u64();
 374          ~[(h >> 0) as u8,
 375            (h >> 8) as u8,
 376            (h >> 16) as u8,
 377            (h >> 24) as u8,
 378            (h >> 32) as u8,
 379            (h >> 40) as u8,
 380            (h >> 48) as u8,
 381            (h >> 56) as u8,
 382          ]
 383      }

 385      fn result_str(&mut self) -> ~str {
 386          let r = self.result_bytes();
 387          let mut s = ~"";
 388          for b in r.iter() {
 389              s.push_str(uint::to_str_radix(*b as uint, 16u));
 390          }
 391          s
 392      }

  20  pub unsafe fn transmute_copy<T, U>(src: &T) -> U {
  21      let mut dest: U = intrinsics::uninit();
  22      let dest_ptr: *mut u8 = transmute(&mut dest);

  24      intrinsics::memcpy32(dest_ptr, src_ptr, sys::size_of::<U>() as u32);
  25      dest
  26  }

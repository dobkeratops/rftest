  16  /// Casts the value at `src` to U. The two types must have the same length.
  17  #[cfg(target_word_size = "32")]
  18  #[inline]
  19  pub unsafe fn transmute_copy<T, U>(src: &T) -> U {
  20      let mut dest: U = intrinsics::uninit();
  21      let dest_ptr: *mut u8 = transmute(&mut dest);
  22      let src_ptr: *u8 = transmute(src);
  23      intrinsics::memcpy32(dest_ptr, src_ptr, sys::size_of::<U>() as u32);
  24      dest
  25  }

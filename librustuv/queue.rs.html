  11  //! A concurrent queue used to signal remote event loops
  12  //!
  13  //! This queue implementation is used to send tasks among event loops. This is
  14  //! backed by a multi-producer/single-consumer queue from libstd and uv_async_t
  15  //! handles (to wake up a remote event loop).
  16  //!
  17  //! The uv_async_t is stored next to the event loop, so in order to not keep the
  18  //! event loop alive we use uv_ref and uv_unref in order to control when the
  19  //! async handle is active or not.

  58  extern fn async_cb(handle: *uvll::uv_async_t, status: c_int) {
  59      assert_eq!(status, 0);
  60      let pool: &mut QueuePool = unsafe {
  61          cast::transmute(uvll::get_data_for_uv_handle(handle))
  62      };
  63      let state: &mut State = unsafe { cast::transmute(pool.queue.get()) };
  64  
  65      // Remember that there is no guarantee about how many times an async
  66      // callback is called with relation to the number of sends, so process the
  67      // entire queue in a loop.
  68      loop {
  69          match state.queue.pop() {
  70              mpsc::Data(Task(task)) => {
  71                  let _ = task.wake().map(|t| t.reawaken());
  72              }
  73              mpsc::Data(Increment) => unsafe {
  74                  if pool.refcnt == 0 {
  75                      uvll::uv_ref(state.handle);
  76                  }
  77                  pool.refcnt += 1;
  78              },
  79              mpsc::Data(Decrement) => unsafe {
  80                  pool.refcnt -= 1;
  81                  if pool.refcnt == 0 {
  82                      uvll::uv_unref(state.handle);
  83                  }
  84              },
  85              mpsc::Empty | mpsc::Inconsistent => break
  86          };
  87      }
  88  
  89      // If the refcount is now zero after processing the queue, then there is no
  90      // longer a reference on the async handle and it is possible that this event
  91      // loop can exit. What we're not guaranteed, however, is that a producer in
  92      // the middle of dropping itself is yet done with the handle. It could be
  93      // possible that we saw their Decrement message but they have yet to signal
  94      // on the async handle. If we were to return immediately, the entire uv loop
  95      // could be destroyed meaning the call to uv_async_send would abort()
  96      //
  97      // In order to fix this, an OS mutex is used to wait for the other end to
  98      // finish before we continue. The drop block on a handle will acquire a
  99      // mutex and then drop it after both the push and send have been completed.
 100      // If we acquire the mutex here, then we are guaranteed that there are no
 101      // longer any senders which are holding on to their handles, so we can
 102      // safely allow the event loop to exit.
 103      if pool.refcnt == 0 {
 104          unsafe {
 105              let _l = state.lock.lock();
 106          }
 107      }
 108  }

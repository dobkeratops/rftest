  23  // uv_stream_t instance, and all I/O operations assume that it's already located

  46  impl StreamWatcher {
  47      // Creates a new helper structure which should be then embedded into another
  48      // watcher. This provides the generic read/write methods on streams.
  49      //
  50      // This structure will *not* close the stream when it is dropped. It is up
  51      // to the enclosure structure to be sure to call the close method (which
  52      // will block the task). Note that this is also required to prevent memory
  53      // leaks.
  54      //
  55      // It should also be noted that the `data` field of the underlying uv handle
  56      // will be manipulated on each of the methods called on this watcher.
  57      // Wrappers should ensure to always reset the field to an appropriate value
  58      // if they rely on the field to perform an action.
  59      pub fn new(stream: *uvll::uv_stream_t) -> StreamWatcher {
  60          StreamWatcher {
  61              handle: stream,
  62              last_write_req: None,
  63          }
  64      }

  66      pub fn read(&mut self, buf: &mut [u8]) -> Result<uint, UvError> {
  67          // This read operation needs to get canceled on an unwind via libuv's
  68          // uv_read_stop function
  69          let _f = ForbidUnwind::new("stream read");
  70  
  71          let mut rcx = ReadContext {
  72              buf: Some(slice_to_uv_buf(buf)),
  73              result: 0,
  74              task: None,
  75          };
  76          // When reading a TTY stream on windows, libuv will invoke alloc_cb
  77          // immediately as part of the call to alloc_cb. What this means is that
  78          // we must be ready for this to happen (by setting the data in the uv
  79          // handle). In theory this otherwise doesn't need to happen until after
  80          // the read is succesfully started.
  81          unsafe {
  82              uvll::set_data_for_uv_handle(self.handle, &rcx)
  83          }
  84  
  85          // Send off the read request, but don't block until we're sure that the
  86          // read request is queued.
  87          match unsafe {
  88              uvll::uv_read_start(self.handle, alloc_cb, read_cb)
  89          } {
  90              0 => {
  91                  let loop_ = unsafe { uvll::get_loop_for_uv_handle(self.handle) };
  92                  wait_until_woken_after(&mut rcx.task, &Loop::wrap(loop_), || {});
  93                  match rcx.result {
  94                      n if n < 0 => Err(UvError(n as c_int)),
  95                      n => Ok(n as uint),
  96                  }
  97              }
  98              n => Err(UvError(n))
  99          }
 100      }

 102      pub fn write(&mut self, buf: &[u8]) -> Result<(), UvError> {
 103          // The ownership of the write request is dubious if this function
 104          // unwinds. I believe that if the write_cb fails to re-schedule the task
 105          // then the write request will be leaked.
 106          let _f = ForbidUnwind::new("stream write");
 107  
 108          // Prepare the write request, either using a cached one or allocating a
 109          // new one
 110          let mut req = match self.last_write_req.take() {
 111              Some(req) => req, None => Request::new(uvll::UV_WRITE),
 112          };
 113          req.set_data(ptr::null::<()>());
 114  
 115          // Send off the request, but be careful to not block until we're sure
 116          // that the write reqeust is queued. If the reqeust couldn't be queued,
 117          // then we should return immediately with an error.
 118          match unsafe {
 119              uvll::uv_write(req.handle, self.handle, [slice_to_uv_buf(buf)],
 120                             write_cb)
 121          } {
 122              0 => {
 123                  let mut wcx = WriteContext { result: 0, task: None, };
 124                  req.defuse(); // uv callback now owns this request
 125  
 126                  let loop_ = unsafe { uvll::get_loop_for_uv_handle(self.handle) };
 127                  wait_until_woken_after(&mut wcx.task, &Loop::wrap(loop_), || {
 128                      req.set_data(&wcx);
 129                  });
 130                  self.last_write_req = Some(Request::wrap(req.handle));
 131                  match wcx.result {
 132                      0 => Ok(()),
 133                      n => Err(UvError(n)),
 134                  }
 135              }
 136              n => Err(UvError(n)),
 137          }
 138      }
 139  }

 141  // This allocation callback expects to be invoked once and only once. It will
 142  // unwrap the buffer in the ReadContext stored in the stream and return it. This
 143  // will fail if it is called more than once.
 144  extern fn alloc_cb(stream: *uvll::uv_stream_t, _hint: size_t, buf: *mut Buf) {
 145      uvdebug!("alloc_cb");

 147          let rcx: &mut ReadContext =
 148              cast::transmute(uvll::get_data_for_uv_handle(stream));
 149          *buf = rcx.buf.take().expect("stream alloc_cb called more than once");
 150      }
 151  }

 153  // When a stream has read some data, we will always forcibly stop reading and
 154  // return all the data read (even if it didn't fill the whole buffer).
 155  extern fn read_cb(handle: *uvll::uv_stream_t, nread: ssize_t, _buf: *Buf) {
 156      uvdebug!("read_cb {}", nread);
 157      assert!(nread != uvll::ECANCELED as ssize_t);
 158      let rcx: &mut ReadContext = unsafe {
 159          cast::transmute(uvll::get_data_for_uv_handle(handle))
 160      };
 161      // Stop reading so that no read callbacks are
 162      // triggered before the user calls `read` again.
 163      // FIXME: Is there a performance impact to calling
 164      // stop here?
 165      unsafe { assert_eq!(uvll::uv_read_stop(handle), 0); }
 166      rcx.result = nread;
 167  
 168      wakeup(&mut rcx.task);
 169  }

 171  // Unlike reading, the WriteContext is stored in the uv_write_t request. Like
 172  // reading, however, all this does is wake up the blocked task after squirreling
 173  // away the error code as a result.
 174  extern fn write_cb(req: *uvll::uv_write_t, status: c_int) {
 175      let mut req = Request::wrap(req);
 176      assert!(status != uvll::ECANCELED);
 177      // Remember to not free the request because it is re-used between writes on
 178      // the same stream.
 179      let wcx: &mut WriteContext = unsafe { req.get_data() };
 180      wcx.result = status;
 181      req.defuse();
 182  
 183      wakeup(&mut wcx.task);
 184  }

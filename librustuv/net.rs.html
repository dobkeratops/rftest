   39  pub fn sockaddr_to_addr(storage: &libc::sockaddr_storage,
   40                          len: uint) -> ip::SocketAddr {
   41      match storage.ss_family as c_int {
   42          libc::AF_INET => {
   43              assert!(len as uint >= mem::size_of::<libc::sockaddr_in>());
   44              let storage: &libc::sockaddr_in = unsafe {
   45                  cast::transmute(storage)
   46              };
   47              let addr = storage.sin_addr.s_addr as u32;
   48              let a = (addr >>  0) as u8;
   49              let b = (addr >>  8) as u8;
   50              let c = (addr >> 16) as u8;
   51              let d = (addr >> 24) as u8;
   52              ip::SocketAddr {
   53                  ip: ip::Ipv4Addr(a, b, c, d),
   54                  port: ntohs(storage.sin_port),
   55              }
   56          }
   57          libc::AF_INET6 => {
   58              assert!(len as uint >= mem::size_of::<libc::sockaddr_in6>());
   59              let storage: &libc::sockaddr_in6 = unsafe {
   60                  cast::transmute(storage)
   61              };
   62              let a = ntohs(storage.sin6_addr.s6_addr[0]);
   63              let b = ntohs(storage.sin6_addr.s6_addr[1]);
   64              let c = ntohs(storage.sin6_addr.s6_addr[2]);
   65              let d = ntohs(storage.sin6_addr.s6_addr[3]);
   66              let e = ntohs(storage.sin6_addr.s6_addr[4]);
   67              let f = ntohs(storage.sin6_addr.s6_addr[5]);
   68              let g = ntohs(storage.sin6_addr.s6_addr[6]);
   69              let h = ntohs(storage.sin6_addr.s6_addr[7]);
   70              ip::SocketAddr {
   71                  ip: ip::Ipv6Addr(a, b, c, d, e, f, g, h),
   72                  port: ntohs(storage.sin6_port),
   73              }
   74          }
   75          n => {
   76              fail!("unknown family {}", n);
   77          }
   78      }
   79  }

   81  fn addr_to_sockaddr(addr: ip::SocketAddr) -> (libc::sockaddr_storage, uint) {
   82      unsafe {
   83          let mut storage: libc::sockaddr_storage = mem::init();
   84          let len = match addr.ip {
   85              ip::Ipv4Addr(a, b, c, d) => {
   86                  let storage: &mut libc::sockaddr_in =
   87                      cast::transmute(&mut storage);
   88                  (*storage).sin_family = libc::AF_INET as libc::sa_family_t;
   89                  (*storage).sin_port = htons(addr.port);
   90                  (*storage).sin_addr = libc::in_addr {
   91                      s_addr: (d as u32 << 24) |
   92                              (c as u32 << 16) |
   93                              (b as u32 <<  8) |
   94                              (a as u32 <<  0)
   95                  };
   96                  mem::size_of::<libc::sockaddr_in>()
   97              }
   98              ip::Ipv6Addr(a, b, c, d, e, f, g, h) => {
   99                  let storage: &mut libc::sockaddr_in6 =
  100                      cast::transmute(&mut storage);
  101                  storage.sin6_family = libc::AF_INET6 as libc::sa_family_t;
  102                  storage.sin6_port = htons(addr.port);
  103                  storage.sin6_addr = libc::in6_addr {
  104                      s6_addr: [
  105                          htons(a),
  106                          htons(b),
  107                          htons(c),
  108                          htons(d),
  109                          htons(e),
  110                          htons(f),
  111                          htons(g),
  112                          htons(h),
  113                      ]
  114                  };
  115                  mem::size_of::<libc::sockaddr_in6>()
  116              }
  117          };
  118          return (storage, len);
  119      }
  120  }

  202      pub fn connect(io: &mut UvIoFactory,
  203                     address: ip::SocketAddr,
  204                     timeout: Option<u64>) -> Result<TcpWatcher, UvError> {
  205          struct Ctx {
  206              status: c_int,
  207              task: Option<BlockedTask>,
  208              timer: Option<~TimerWatcher>,
  209          }
  210  
  211          let tcp = TcpWatcher::new(io);
  212          let (addr, _len) = addr_to_sockaddr(address);
  213          let mut req = Request::new(uvll::UV_CONNECT);
  214          let result = unsafe {
  215              let addr_p = &addr as *libc::sockaddr_storage;
  216              uvll::uv_tcp_connect(req.handle, tcp.handle,
  217                                   addr_p as *libc::sockaddr,
  218                                   connect_cb)
  219          };
  220          return match result {
  221              0 => {
  222                  req.defuse(); // uv callback now owns this request
  223                  let mut cx = Ctx { status: -1, task: None, timer: None };
  224                  match timeout {
  225                      Some(t) => {
  226                          let mut timer = TimerWatcher::new(io);
  227                          timer.start(timer_cb, t, 0);
  228                          cx.timer = Some(timer);
  229                      }
  230                      None => {}
  231                  }
  232                  wait_until_woken_after(&mut cx.task, &io.loop_, || {
  233                      let data = &cx as *_;
  234                      match cx.timer {
  235                          Some(ref mut timer) => unsafe { timer.set_data(data) },
  236                          None => {}
  237                      }
  238                      req.set_data(data);
  239                  });
  240                  // Make sure an erroneously fired callback doesn't have access
  241                  // to the context any more.
  242                  req.set_data(0 as *int);
  243  
  244                  // If we failed because of a timeout, drop the TcpWatcher as
  245                  // soon as possible because it's data is now set to null and we
  246                  // want to cancel the callback ASAP.
  247                  match cx.status {
  248                      0 => Ok(tcp),
  249                      n => { drop(tcp); Err(UvError(n)) }
  250                  }
  251              }
  252              n => Err(UvError(n))
  253          };
  254  
  255          extern fn timer_cb(handle: *uvll::uv_timer_t, status: c_int) {
  256              // Don't close the corresponding tcp request, just wake up the task
  257              // and let RAII take care of the pending watcher.
  258              assert_eq!(status, 0);
  259              let cx: &mut Ctx = unsafe {
  260                  &mut *(uvll::get_data_for_uv_handle(handle) as *mut Ctx)
  261              };
  262              cx.status = uvll::ECANCELED;
  263              wakeup(&mut cx.task);
  264          }
  265  
  266          extern fn connect_cb(req: *uvll::uv_connect_t, status: c_int) {
  267              // This callback can be invoked with ECANCELED if the watcher is
  268              // closed by the timeout callback. In that case we just want to free
  269              // the request and be along our merry way.
  270              let req = Request::wrap(req);
  271              if status == uvll::ECANCELED { return }
  272  
  273              let cx: &mut Ctx = unsafe { req.get_data() };
  274              cx.status = status;
  275              match cx.timer {
  276                  Some(ref mut t) => t.stop(),
  277                  None => {}
  278              }
  279              // Note that the timer callback doesn't cancel the connect request
  280              // (that's the job of uv_close()), so it's possible for this
  281              // callback to get triggered after the timeout callback fires, but
  282              // before the task wakes up. In that case, we did indeed
  283              // successfully connect, but we don't need to wake someone up. We
  284              // updated the status above (correctly so), and the task will pick
  285              // up on this when it wakes up.
  286              if cx.task.is_some() {
  287                  wakeup(&mut cx.task);
  288              }
  289          }
  290      }
  291  }

  577  impl rtio::RtioUdpSocket for UdpWatcher {
  578      fn recvfrom(&mut self, buf: &mut [u8])
  579          -> Result<(uint, ip::SocketAddr), IoError>
  580      {
  581          struct Ctx {
  582              task: Option<BlockedTask>,
  583              buf: Option<Buf>,
  584              result: Option<(ssize_t, Option<ip::SocketAddr>)>,
  585          }
  586          let loop_ = self.uv_loop();
  587          let m = self.fire_homing_missile();
  588          let _g = self.read_access.grant(m);
  589  
  590          let a = match unsafe {
  591              uvll::uv_udp_recv_start(self.handle, alloc_cb, recv_cb)
  592          } {
  593              0 => {
  594                  let mut cx = Ctx {
  595                      task: None,
  596                      buf: Some(slice_to_uv_buf(buf)),
  597                      result: None,
  598                  };
  599                  let handle = self.handle;
  600                  wait_until_woken_after(&mut cx.task, &loop_, || {
  601                      unsafe { uvll::set_data_for_uv_handle(handle, &cx) }
  602                  });
  603                  match cx.result.take_unwrap() {
  604                      (n, _) if n < 0 =>
  605                          Err(uv_error_to_io_error(UvError(n as c_int))),
  606                      (n, addr) => Ok((n as uint, addr.unwrap()))
  607                  }
  608              }
  609              n => Err(uv_error_to_io_error(UvError(n)))
  610          };
  611          return a;
  612  
  613          extern fn alloc_cb(handle: *uvll::uv_udp_t,
  614                             _suggested_size: size_t,
  615                             buf: *mut Buf) {
  616              unsafe {
  617                  let cx: &mut Ctx =
  618                      cast::transmute(uvll::get_data_for_uv_handle(handle));
  619                  *buf = cx.buf.take().expect("recv alloc_cb called more than once")
  620              }
  621          }
  622  
  623          extern fn recv_cb(handle: *uvll::uv_udp_t, nread: ssize_t, buf: *Buf,
  624                            addr: *libc::sockaddr, _flags: c_uint) {
  625              assert!(nread != uvll::ECANCELED as ssize_t);
  626              let cx: &mut Ctx = unsafe {
  627                  cast::transmute(uvll::get_data_for_uv_handle(handle))
  628              };
  629  
  630              // When there's no data to read the recv callback can be a no-op.
  631              // This can happen if read returns EAGAIN/EWOULDBLOCK. By ignoring
  632              // this we just drop back to kqueue and wait for the next callback.
  633              if nread == 0 {
  634                  cx.buf = Some(unsafe { *buf });
  635                  return
  636              }
  637  
  638              unsafe {
  639                  assert_eq!(uvll::uv_udp_recv_stop(handle), 0)
  640              }
  641  
  642              let cx: &mut Ctx = unsafe {
  643                  cast::transmute(uvll::get_data_for_uv_handle(handle))
  644              };
  645              let addr = if addr == ptr::null() {
  646                  None
  647              } else {
  648                  let len = mem::size_of::<libc::sockaddr_storage>();
  649                  Some(sockaddr_to_addr(unsafe { cast::transmute(addr) }, len))
  650              };
  651              cx.result = Some((nread, addr));
  652              wakeup(&mut cx.task);
  653          }
  654      }
  655  
  656      fn sendto(&mut self, buf: &[u8], dst: ip::SocketAddr) -> Result<(), IoError> {
  657          struct Ctx { task: Option<BlockedTask>, result: c_int }
  658  
  659          let m = self.fire_homing_missile();
  660          let loop_ = self.uv_loop();
  661          let _g = self.write_access.grant(m);
  662  
  663          let mut req = Request::new(uvll::UV_UDP_SEND);
  664          let buf = slice_to_uv_buf(buf);
  665          let (addr, _len) = addr_to_sockaddr(dst);
  666          let result = unsafe {
  667              let addr_p = &addr as *libc::sockaddr_storage;
  668              uvll::uv_udp_send(req.handle, self.handle, [buf],
  669                                addr_p as *libc::sockaddr, send_cb)
  670          };
  671  
  672          return match result {
  673              0 => {
  674                  req.defuse(); // uv callback now owns this request
  675                  let mut cx = Ctx { task: None, result: 0 };
  676                  wait_until_woken_after(&mut cx.task, &loop_, || {
  677                      req.set_data(&cx);
  678                  });
  679                  match cx.result {
  680                      0 => Ok(()),
  681                      n => Err(uv_error_to_io_error(UvError(n)))
  682                  }
  683              }
  684              n => Err(uv_error_to_io_error(UvError(n)))
  685          };
  686  
  687          extern fn send_cb(req: *uvll::uv_udp_send_t, status: c_int) {
  688              let req = Request::wrap(req);
  689              assert!(status != uvll::ECANCELED);
  690              let cx: &mut Ctx = unsafe { req.get_data() };
  691              cx.result = status;
  692              wakeup(&mut cx.task);
  693          }
  694      }

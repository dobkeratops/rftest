  87      pub fn connect(io: &mut UvIoFactory, name: &CString)
  88          -> Result<PipeWatcher, UvError>
  89      {
  90          struct Ctx { task: Option<BlockedTask>, result: libc::c_int, }
  91          let mut cx = Ctx { task: None, result: 0 };
  92          let mut req = Request::new(uvll::UV_CONNECT);
  93          let pipe = PipeWatcher::new(io, false);
  94  
  95          wait_until_woken_after(&mut cx.task, &io.loop_, || {
  96              unsafe {
  97                  uvll::uv_pipe_connect(req.handle,
  98                                        pipe.handle(),
  99                                        name.with_ref(|p| p),
 100                                        connect_cb)
 101              }
 102              req.set_data(&cx);
 103              req.defuse(); // uv callback now owns this request
 104          });
 105          return match cx.result {
 106              0 => Ok(pipe),
 107              n => Err(UvError(n))
 108          };

 111              let req = Request::wrap(req);

 114              cx.result = status;
 115              wakeup(&mut cx.task);

 175          -> Result<~PipeListener, UvError>
 176      {

 178          match unsafe {
 179              uvll::uv_pipe_bind(pipe.handle(), name.with_ref(|p| p))

 181              0 => {
 182                  // If successful, unwrap the PipeWatcher because we control how
 183                  // we close the pipe differently. We can't rely on

 187                      home: io.make_handle(),
 188                      pipe: pipe.unwrap(),
 189                      incoming: rx,
 190                      outgoing: tx,
 191                  };

 194              n => Err(UvError(n))

  65  impl Drop for UvEventLoop {
  66      fn drop(&mut self) {
  67          // Must first destroy the pool of handles before we destroy the loop
  68          // because otherwise the contained async handle will be destroyed after
  69          // the loop is free'd (use-after-free). We also must free the uv handle
  70          // after the loop has been closed because during the closing of the loop
  71          // the handle is required to be used apparently.
  72          let handle = self.uvio.handle_pool.get_ref().handle();
  73          drop(self.uvio.handle_pool.take());
  74          self.uvio.loop_.close();
  75          unsafe { uvll::free_handle(handle) }
  76      }
  77  }

  11  /*!
  12  
  13  Bindings to libuv, along with the default implementation of `std::rt::rtio`.
  14  
  15  UV types consist of the event loop (Loop), Watchers, Requests and
  16  Callbacks.
  17  
  18  Watchers and Requests encapsulate pointers to uv *handles*, which have
  19  subtyping relationships with each other.  This subtyping is reflected
  20  in the bindings with explicit or implicit coercions. For example, an
  21  upcast from TcpWatcher to StreamWatcher is done with
  22  `tcp_watcher.as_stream()`. In other cases a callback on a specific
  23  type of watcher will be passed a watcher of a supertype.
  24  
  25  Currently all use of Request types (connect/write requests) are
  26  encapsulated in the bindings and don't need to be dealt with by the
  27  caller.
  28  
  29  # Safety note
  30  
  31  Due to the complex lifecycle of uv handles, as well as compiler bugs,
  32  this module is not memory safe and requires explicit memory management,
  33  via `close` and `delete` methods.
  34  
  35  */

 134  /// A type that wraps a uv handle
 135  pub trait UvHandle<T> {
 136      fn uv_handle(&self) -> *T;
 137  
 138      fn uv_loop(&self) -> Loop {
 139          Loop::wrap(unsafe { uvll::get_loop_for_uv_handle(self.uv_handle()) })
 140      }
 141  
 142      // FIXME(#8888) dummy self
 143      fn alloc(_: Option<Self>, ty: uvll::uv_handle_type) -> *T {
 144          unsafe {
 145              let handle = uvll::malloc_handle(ty);
 146              assert!(!handle.is_null());
 147              handle as *T
 148          }
 149      }
 150  
 151      unsafe fn from_uv_handle<'a>(h: &'a *T) -> &'a mut Self {
 152          cast::transmute(uvll::get_data_for_uv_handle(*h))
 153      }
 154  
 155      fn install(~self) -> ~Self {
 156          unsafe {
 157              let myptr = cast::transmute::<&~Self, &*u8>(&self);
 158              uvll::set_data_for_uv_handle(self.uv_handle(), *myptr);
 159          }
 160          self
 161      }
 162  
 163      fn close_async_(&mut self) {
 164          // we used malloc to allocate all handles, so we must always have at
 165          // least a callback to free all the handles we allocated.
 166          extern fn close_cb(handle: *uvll::uv_handle_t) {
 167              unsafe { uvll::free_handle(handle) }
 168          }
 169  
 170          unsafe {
 171              uvll::set_data_for_uv_handle(self.uv_handle(), null::<()>());
 172              uvll::uv_close(self.uv_handle() as *uvll::uv_handle_t, close_cb)
 173          }
 174      }
 175  
 176      fn close(&mut self) {
 177          let mut slot = None;
 178  
 179          unsafe {
 180              uvll::uv_close(self.uv_handle() as *uvll::uv_handle_t, close_cb);
 181              uvll::set_data_for_uv_handle(self.uv_handle(), ptr::null::<()>());
 182  
 183              wait_until_woken_after(&mut slot, &self.uv_loop(), || {
 184                  uvll::set_data_for_uv_handle(self.uv_handle(), &slot);
 185              })
 186          }
 187  
 188          extern fn close_cb(handle: *uvll::uv_handle_t) {
 189              unsafe {
 190                  let data = uvll::get_data_for_uv_handle(handle);
 191                  uvll::free_handle(handle);
 192                  if data == ptr::null() { return }
 193                  let slot: &mut Option<BlockedTask> = cast::transmute(data);
 194                  wakeup(slot);
 195              }
 196          }
 197      }
 198  }

 242  fn wait_until_woken_after(slot: *mut Option<BlockedTask>,
 243                            loop_: &Loop,
 244                            f: ||) {
 245      let _f = ForbidUnwind::new("wait_until_woken_after");
 246      unsafe {
 247          assert!((*slot).is_none());
 248          let task: ~Task = Local::take();
 249          loop_.modify_blockers(1);
 250          task.deschedule(1, |task| {
 251              *slot = Some(task);
 252              f();
 253              Ok(())
 254          });
 255          loop_.modify_blockers(-1);
 256      }
 257  }

 259  fn wakeup(slot: &mut Option<BlockedTask>) {
 260      assert!(slot.is_some());
 261      let _ = slot.take_unwrap().wake().map(|t| t.reawaken());
 262  }

 394  pub fn uv_error_to_io_error(uverr: UvError) -> IoError {
 395      unsafe {
 396          // Importing error constants
 397  
 398          // uv error descriptions are static
 399          let UvError(errcode) = uverr;
 400          let c_desc = uvll::uv_strerror(errcode);
 401          let desc = str::raw::c_str_to_static_slice(c_desc);
 402  
 403          let kind = match errcode {
 404              uvll::UNKNOWN => io::OtherIoError,
 405              uvll::OK => io::OtherIoError,
 406              uvll::EOF => io::EndOfFile,
 407              uvll::EACCES => io::PermissionDenied,
 408              uvll::ECONNREFUSED => io::ConnectionRefused,
 409              uvll::ECONNRESET => io::ConnectionReset,
 410              uvll::ENOTCONN => io::NotConnected,
 411              uvll::ENOENT => io::FileNotFound,
 412              uvll::EPIPE => io::BrokenPipe,
 413              uvll::ECONNABORTED => io::ConnectionAborted,
 414              uvll::EADDRNOTAVAIL => io::ConnectionRefused,
 415              err => {
 416                  uvdebug!("uverr.code {}", err as int);
 417                  // FIXME: Need to map remaining uv error types
 418                  io::OtherIoError
 419              }
 420          };
 421  
 422          IoError {
 423              kind: kind,
 424              desc: desc,
 425              detail: None
 426          }
 427      }
 428  }

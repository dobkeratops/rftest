  71      fn sleep(&mut self, msecs: u64) {
  72          // As with all of the below functions, we must be extra careful when
  73          // destroying the previous action. If the previous action was a channel,
  74          // destroying it could invoke a context switch. For these situtations,
  75          // we must temporarily un-home ourselves, then destroy the action, and
  76          // then re-home again.
  77          let missile = self.fire_homing_missile();
  78          self.id += 1;
  79          self.stop();
  80          let _missile = match mem::replace(&mut self.action, None) {
  81              None => missile, // no need to do a homing dance
  82              Some(action) => {
  83                  drop(missile);      // un-home ourself
  84                  drop(action);       // destroy the previous action
  85                  self.fire_homing_missile()  // re-home ourself
  86              }
  87          };
  88  
  89          // If the descheduling operation unwinds after the timer has been
  90          // started, then we need to call stop on the timer.
  91          let _f = ForbidUnwind::new("timer");
  92  
  93          self.action = Some(WakeTask);
  94          wait_until_woken_after(&mut self.blocker, &self.uv_loop(), || {
  95              self.start(msecs, 0);
  96          });
  97          self.stop();
  98      }

 100      fn oneshot(&mut self, msecs: u64) -> Receiver<()> {
 101          let (tx, rx) = channel();
 102  
 103          // similarly to the destructor, we must drop the previous action outside
 104          // of the homing missile
 105          let _prev_action = {
 106              let _m = self.fire_homing_missile();
 107              self.id += 1;
 108              self.stop();
 109              self.start(msecs, 0);
 110              mem::replace(&mut self.action, Some(SendOnce(tx)))
 111          };
 112  
 113          return rx;
 114      }

 133  extern fn timer_cb(handle: *uvll::uv_timer_t, status: c_int) {
 134      let _f = ForbidSwitch::new("timer callback can't switch");
 135      assert_eq!(status, 0);
 136      let timer: &mut TimerWatcher = unsafe { UvHandle::from_uv_handle(&handle) };
 137  
 138      match timer.action.take_unwrap() {
 139          WakeTask => {
 140              let task = timer.blocker.take_unwrap();
 141              let _ = task.wake().map(|t| t.reawaken());
 142          }
 143          SendOnce(chan) => { let _ = chan.try_send(()); }
 144          SendMany(chan, id) => {
 145              let _ = chan.try_send(());
 146  
 147              // Note that the above operation could have performed some form of
 148              // scheduling. This means that the timer may have decided to insert
 149              // some other action to happen. This 'id' keeps track of the updates
 150              // to the timer, so we only reset the action back to sending on this
 151              // channel if the id has remained the same. This is essentially a
 152              // bug in that we have mutably aliasable memory, but that's libuv
 153              // for you. We're guaranteed to all be running on the same thread,
 154              // so there's no need for any synchronization here.
 155              if timer.id == id {
 156                  timer.action = Some(SendMany(chan, id));
 157              }
 158          }
 159      }
 160  }

 162  impl Drop for TimerWatcher {
 163      fn drop(&mut self) {
 164          // note that this drop is a little subtle. Dropping a channel which is
 165          // held internally may invoke some scheduling operations. We can't take
 166          // the channel unless we're on the home scheduler, but once we're on the
 167          // home scheduler we should never move. Hence, we take the timer's
 168          // action item and then move it outside of the homing block.
 169          let _action = {
 170              let _m = self.fire_homing_missile();
 171              self.stop();
 172              self.close();
 173              self.action.take()
 174          };
 175      }
 176  }

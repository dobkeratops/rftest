  11  /*!
  12   * Support for matching file paths against Unix shell style patterns.
  13   *
  14   * The `glob` and `glob_with` functions, in concert with the `Paths`
  15   * type, allow querying the filesystem for all files that match a particular
  16   * pattern - just like the libc `glob` function (for an example see the `glob`
  17   * documentation). The methods on the `Pattern` type provide functionality
  18   * for checking if individual paths match a particular pattern - in a similar
  19   * manner to the libc `fnmatch` function
  20   *
  21   * For consistency across platforms, and for Windows support, this module
  22   * is implemented entirely in Rust rather than deferring to the libc
  23   * `glob`/`fnmatch` functions.
  24   */
  25

  39  /**
  40   * An iterator that yields Paths from the filesystem that match a particular
  41   * pattern - see the `glob` function for more details.
  42   */
  43  pub struct Paths {
  44      root: Path,
  45      dir_patterns: Vec<Pattern>,
  46      options: MatchOptions,
  47      todo: Vec<(Path,uint)>,
  48  }

  50  ///
  51  /// Return an iterator that produces all the Paths that match the given pattern,
  52  /// which may be absolute or relative to the current working directory.
  53  ///
  54  /// is method uses the default match options and is equivalent to calling
  55  /// `glob_with(pattern, MatchOptions::new())`. Use `glob_with` directly if you
  56  /// want to use non-default match options.
  57  ///
  58  /// # Example
  59  ///
  60  /// Consider a directory `/media/pictures` containing only the files `kittens.jpg`,
  61  /// `puppies.jpg` and `hamsters.gif`:
  62  ///
  63  /// ```rust
  64  /// use glob::glob;
  65  ///
  66  /// for path in glob("/media/pictures/*.jpg") {
  67  ///     println!("{}", path.display());
  68  /// }
  69  /// ```
  70  ///
  71  /// The above code will print:
  72  ///
  73  /// ```ignore
  74  /// /media/pictures/kittens.jpg
  75  /// /media/pictures/puppies.jpg
  76  /// ```
  77  ///
  78  pub fn glob(pattern: &str) -> Paths {
  79      glob_with(pattern, MatchOptions::new())
  80  }

  82  /**
  83   * Return an iterator that produces all the Paths that match the given pattern,
  84   * which may be absolute or relative to the current working directory.
  85   *
  86   * This function accepts Unix shell style patterns as described by `Pattern::new(..)`.
  87   * The options given are passed through unchanged to `Pattern::matches_with(..)` with
  88   * the exception that `require_literal_separator` is always set to `true` regardless of the
  89   * value passed to this function.
  90   *
  91   * Paths are yielded in alphabetical order, as absolute paths.
  92   */
  93  pub fn glob_with(pattern: &str, options: MatchOptions) -> Paths {
  94      #[cfg(windows)]
  95      fn check_windows_verbatim(p: &Path) -> bool { path::windows::is_verbatim(p) }
  96      #[cfg(not(windows))]
  97      fn check_windows_verbatim(_: &Path) -> bool { false }
  98  
  99      // calculate root this way to handle volume-relative Windows paths correctly
 100      let mut root = os::getcwd();
 101      let pat_root = Path::new(pattern).root_path();
 102      if pat_root.is_some() {
 103          if check_windows_verbatim(pat_root.get_ref()) {
 104              // FIXME: How do we want to handle verbatim paths? I'm inclined to return nothing,
 105              // since we can't very well find all UNC shares with a 1-letter server name.
 106              return Paths {
 107                  root: root,
 108                  dir_patterns: Vec::new(),
 109                  options: options,
 110                  todo: Vec::new(),
 111              };
 112          }
 113          root.push(pat_root.get_ref());
 114      }
 115  
 116      let root_len = pat_root.map_or(0u, |p| p.as_vec().len());
 117      let dir_patterns = pattern.slice_from(cmp::min(root_len, pattern.len()))
 118                         .split_terminator(is_sep)
 119                         .map(|s| Pattern::new(s))
 120                         .collect();
 121  
 122      let todo = list_dir_sorted(&root).move_iter().map(|x|(x,0u)).collect();
 123  
 124      Paths {
 125          root: root,
 126          dir_patterns: dir_patterns,
 127          options: options,
 128          todo: todo,
 129      }
 130  }

 132  impl Iterator<Path> for Paths {
 133  
 134      fn next(&mut self) -> Option<Path> {
 135          loop {
 136              if self.dir_patterns.is_empty() || self.todo.is_empty() {
 137                  return None;
 138              }
 139  
 140              let (path,idx) = self.todo.pop().unwrap();
 141              let ref pattern = *self.dir_patterns.get(idx);
 142  
 143              if pattern.matches_with(match path.filename_str() {
 144                  // this ugly match needs to go here to avoid a borrowck error
 145                  None => {
 146                      // FIXME (#9639): How do we handle non-utf8 filenames? Ignore them for now
 147                      // Ideally we'd still match them against a *
 148                      continue;
 149                  }
 150                  Some(x) => x
 151              }, self.options) {
 152                  if idx == self.dir_patterns.len() - 1 {
 153                      // it is not possible for a pattern to match a directory *AND* its children
 154                      // so we don't need to check the children
 155                      return Some(path);
 156                  } else {
 157                      self.todo.extend(list_dir_sorted(&path).move_iter().map(|x|(x,idx+1)));
 158                  }
 159              }
 160          }
 161      }
 162  
 163  }

 165  fn list_dir_sorted(path: &Path) -> Vec<Path> {
 166      match fs::readdir(path) {
 167          Ok(mut children) => {
 168              children.sort_by(|p1, p2| p2.filename().cmp(&p1.filename()));
 169              children.move_iter().collect()
 170          }
 171          Err(..) => Vec::new()
 172      }
 173  }

 205  impl Pattern {
 206  
 207      /**
 208       * This function compiles Unix shell style patterns: `?` matches any single
 209       * character, `*` matches any (possibly empty) sequence of characters and
 210       * `[...]` matches any character inside the brackets, unless the first
 211       * character is `!` in which case it matches any character except those
 212       * between the `!` and the `]`. Character sequences can also specify ranges
 213       * of characters, as ordered by Unicode, so e.g. `[0-9]` specifies any
 214       * character between 0 and 9 inclusive.
 215       *
 216       * The metacharacters `?`, `*`, `[`, `]` can be matched by using brackets
 217       * (e.g. `[?]`).  When a `]` occurs immediately following `[` or `[!` then
 218       * it is interpreted as being part of, rather then ending, the character
 219       * set, so `]` and NOT `]` can be matched by `[]]` and `[!]]` respectively.
 220       * The `-` character can be specified inside a character sequence pattern by
 221       * placing it at the start or the end, e.g. `[abc-]`.
 222       *
 223       * When a `[` does not have a closing `]` before the end of the string then
 224       * the `[` will be treated literally.
 225       */
 226      pub fn new(pattern: &str) -> Pattern {
 227  
 228          let chars = pattern.chars().collect::<~[_]>();
 229          let mut tokens = Vec::new();
 230          let mut i = 0;
 231  
 232          while i < chars.len() {
 233              match chars[i] {
 234                  '?' => {
 235                      tokens.push(AnyChar);
 236                      i += 1;
 237                  }
 238                  '*' => {
 239                      // *, **, ***, ****, ... are all equivalent
 240                      while i < chars.len() && chars[i] == '*' {
 241                          i += 1;
 242                      }
 243                      tokens.push(AnySequence);
 244                  }
 245                  '[' => {
 246  
 247                      if i <= chars.len() - 4 && chars[i + 1] == '!' {
 248                          match chars.slice_from(i + 3).position_elem(&']') {
 249                              None => (),
 250                              Some(j) => {
 251                                  let chars = chars.slice(i + 2, i + 3 + j);
 252                                  let cs = parse_char_specifiers(chars);
 253                                  tokens.push(AnyExcept(cs));
 254                                  i += j + 4;
 255                                  continue;
 256                              }
 257                          }
 258                      }
 259                      else if i <= chars.len() - 3 && chars[i + 1] != '!' {
 260                          match chars.slice_from(i + 2).position_elem(&']') {
 261                              None => (),
 262                              Some(j) => {
 263                                  let cs = parse_char_specifiers(chars.slice(i + 1, i + 2 + j));
 264                                  tokens.push(AnyWithin(cs));
 265                                  i += j + 3;
 266                                  continue;
 267                              }
 268                          }
 269                      }
 270  
 271                      // if we get here then this is not a valid range pattern
 272                      tokens.push(Char('['));
 273                      i += 1;
 274                  }
 275                  c => {
 276                      tokens.push(Char(c));
 277                      i += 1;
 278                  }
 279              }
 280          }
 281  
 282          Pattern { tokens: tokens }
 283      }
 284  
 285      /**
 286       * Escape metacharacters within the given string by surrounding them in
 287       * brackets. The resulting string will, when compiled into a `Pattern`,
 288       * match the input string and nothing else.
 289       */
 290      pub fn escape(s: &str) -> ~str {
 291          let mut escaped = ~"";
 292          for c in s.chars() {
 293              match c {
 294                  // note that ! does not need escaping because it is only special inside brackets
 295                  '?' | '*' | '[' | ']' => {
 296                      escaped.push_char('[');
 297                      escaped.push_char(c);
 298                      escaped.push_char(']');
 299                  }
 300                  c => {
 301                      escaped.push_char(c);
 302                  }
 303              }
 304          }
 305          escaped
 306      }
 307  
 308      /**
 309       * Return if the given `str` matches this `Pattern` using the default
 310       * match options (i.e. `MatchOptions::new()`).
 311       *
 312       * # Example
 313       *
 314       * ```rust
 315       * use glob::Pattern;
 316       *
 317       * assert!(Pattern::new("c?t").matches("cat"));
 318       * assert!(Pattern::new("k[!e]tteh").matches("kitteh"));
 319       * assert!(Pattern::new("d*g").matches("doog"));
 320       * ```
 321       */

 323          self.matches_with(str, MatchOptions::new())
 324      }

 338       * Return if the given `str` matches this `Pattern` using the specified match options.

 343  
 344      /**
 345       * Return if the given `Path`, when converted to a `str`, matches this `Pattern`
 346       * using the specified match options.
 347       */
 348      pub fn matches_path_with(&self, path: &Path, options: MatchOptions) -> bool {
 349          // FIXME (#9639): This needs to handle non-utf8 paths
 350          path.as_str().map_or(false, |s| {
 351              self.matches_with(s, options)
 352          })
 353      }

 355      fn matches_from(&self,
 356                      prev_char: Option<char>,
 357                      mut file: &str,
 358                      i: uint,
 359                      options: MatchOptions) -> MatchResult {
 360  
 361          let prev_char = Cell::new(prev_char);
 362  
 363          let require_literal = |c| {
 364              (options.require_literal_separator && is_sep(c)) ||
 365              (options.require_literal_leading_dot && c == '.'
 366               && is_sep(prev_char.get().unwrap_or('/')))
 367          };
 368  
 369          for (ti, token) in self.tokens.slice_from(i).iter().enumerate() {
 370              match *token {
 371                  AnySequence => {
 372                      loop {
 373                          match self.matches_from(prev_char.get(), file, i + ti + 1, options) {
 374                              SubPatternDoesntMatch => (), // keep trying
 375                              m => return m,
 376                          }
 377  
 378                          if file.is_empty() {
 379                              return EntirePatternDoesntMatch;
 380                          }
 381  
 382                          let (some_c, next) = file.slice_shift_char();
 383                          if require_literal(some_c.unwrap()) {
 384                              return SubPatternDoesntMatch;
 385                          }
 386                          prev_char.set(some_c);
 387                          file = next;
 388                      }
 389                  }
 390                  _ => {
 391                      if file.is_empty() {
 392                          return EntirePatternDoesntMatch;
 393                      }
 394  
 395                      let (some_c, next) = file.slice_shift_char();
 396                      let c = some_c.unwrap();
 397                      let matches = match *token {
 398                          AnyChar => {
 399                              !require_literal(c)
 400                          }
 401                          AnyWithin(ref specifiers) => {
 402                              !require_literal(c) &&
 403                                  in_char_specifiers(specifiers.as_slice(),
 404                                                     c,
 405                                                     options)
 406                          }
 407                          AnyExcept(ref specifiers) => {
 408                              !require_literal(c) &&
 409                                  !in_char_specifiers(specifiers.as_slice(),
 410                                                      c,
 411                                                      options)
 412                          }
 413                          Char(c2) => {
 414                              chars_eq(c, c2, options.case_sensitive)
 415                          }
 416                          AnySequence => {
 417                              unreachable!()
 418                          }
 419                      };
 420                      if !matches {
 421                          return SubPatternDoesntMatch;
 422                      }
 423                      prev_char.set(some_c);
 424                      file = next;
 425                  }
 426              }
 427          }
 428  
 429          if file.is_empty() {
 430              Match
 431          } else {
 432              SubPatternDoesntMatch
 433          }
 434      }
 435  
 436  }

 438  fn parse_char_specifiers(s: &[char]) -> Vec<CharSpecifier> {
 439      let mut cs = Vec::new();
 440      let mut i = 0;
 441      while i < s.len() {
 442          if i + 3 <= s.len() && s[i + 1] == '-' {
 443              cs.push(CharRange(s[i], s[i + 2]));
 444              i += 3;
 445          } else {
 446              cs.push(SingleChar(s[i]));
 447              i += 1;
 448          }
 449      }
 450      cs
 451  }
 452  
 453  fn in_char_specifiers(specifiers: &[CharSpecifier], c: char, options: MatchOptions) -> bool {
 454  
 455      for &specifier in specifiers.iter() {
 456          match specifier {
 457              SingleChar(sc) => {
 458                  if chars_eq(c, sc, options.case_sensitive) {
 459                      return true;
 460                  }
 461              }
 462              CharRange(start, end) => {
 463  
 464                  // FIXME: work with non-ascii chars properly (issue #1347)

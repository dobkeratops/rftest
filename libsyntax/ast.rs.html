   23 // an identifier contains a Name (index into the interner
   24 // table) and a SyntaxContext to track renaming and

   33 /// A SyntaxContext represents a chain of macro-expandings
   34 /// and renamings. Each macro expansion corresponds to
   35 /// a fresh uint
   36 
   37 // I'm representing this syntax context as an index into
   38 // a table, in order to work around a compiler bug
   39 // that's causing unreleased memory to cause core dumps
   40 // and also perhaps to save some work in destructor checks.
   41 // the special uint '0' will be used to indicate an empty
   42 // syntax context.
   43 
   44 // this uint is a reference to a table stored in thread-local
   45 // storage.
   46 pub type SyntaxContext = uint;

  232                 }

  236                 MetaList(ref no, ref miso) => {

  506 // When the main rust parser encounters a syntax-extension invocation, it
  507 // parses the arguments to the invocation as a token-tree. This is a very
  508 // loose structure, such that all sorts of different AST-fragments can
  509 // be passed to syntax extensions using a uniform type.
  510 //
  511 // If the syntax extension is an MBE macro, it will attempt to match its
  512 // LHS "matchers" against the provided token tree, and if it finds a
  513 // match, will transcribe the RHS token tree, splicing in any captured
  514 // macro_parser::matched_nonterminals into the tt_nonterminals it finds.
  515 //
  516 // The RHS of an MBE macro is the only place a tt_nonterminal or tt_seq
  517 // makes any real sense. You could write them elsewhere but nothing
  518 // else knows what to do with them, so you'll probably get a syntax
  519 // error.
  520 //
  521 #[deriving(Clone, Eq, Encodable, Decodable, IterBytes)]
  522 #[doc="For macro invocations; parsing is delegated to the macro"]
  523 pub enum token_tree {
  524     // a single token
  525     tt_tok(span, ::parse::token::Token),
  526     // a delimited sequence (the delimiters appear as the first
  527     // and last elements of the vector)
  528     tt_delim(@mut ~[token_tree]),
  529     // These only make sense for right-hand-sides of MBE macros:
  530 
  531     // a kleene-style repetition sequence with a span, a tt_forest,
  532     // an optional separator (?), and a boolean where true indicates
  533     // zero or more (*), and false indicates one or more (+).
  534     tt_seq(span, @mut ~[token_tree], Option<::parse::token::Token>, bool),
  535 
  536     // a syntactic variable that will be filled in by macro expansion.
  537     tt_nonterminal(span, ident)
  538 }

  540 //
  541 // Matchers are nodes defined-by and recognized-by the main rust parser and
  542 // language, but they're only ever found inside syntax-extension invocations;
  543 // indeed, the only thing that ever _activates_ the rules in the rust parser
  544 // for parsing a matcher is a matcher looking for the 'matchers' nonterminal
  545 // itself. Matchers represent a small sub-language for pattern-matching
  546 // token-trees, and are thus primarily used by the macro-defining extension
  547 // itself.
  548 //
  549 // match_tok
  550 // ---------
  551 //
  552 //     A matcher that matches a single token, denoted by the token itself. So
  553 //     long as there's no $ involved.
  554 //
  555 //
  556 // match_seq
  557 // ---------
  558 //
  559 //     A matcher that matches a sequence of sub-matchers, denoted various
  560 //     possible ways:
  561 //
  562 //             $(M)*       zero or more Ms
  563 //             $(M)+       one or more Ms
  564 //             $(M),+      one or more comma-separated Ms
  565 //             $(A B C);*  zero or more semi-separated 'A B C' seqs
  566 //
  567 //
  568 // match_nonterminal
  569 // -----------------
  570 //
  571 //     A matcher that matches one of a few interesting named rust
  572 //     nonterminals, such as types, expressions, items, or raw token-trees. A
  573 //     black-box matcher on expr, for example, binds an expr to a given ident,
  574 //     and that ident can re-occur as an interpolation in the RHS of a
  575 //     macro-by-example rule. For example:
  576 //
  577 //        $foo:expr   =>     1 + $foo    // interpolate an expr
  578 //        $foo:tt     =>     $foo        // interpolate a token-tree
  579 //        $foo:tt     =>     bar! $foo   // only other valid interpolation
  580 //                                       // is in arg position for another
  581 //                                       // macro
  582 //
  583 // As a final, horrifying aside, note that macro-by-example's input is
  584 // also matched by one of these matchers. Holy self-referential! It is matched
  585 // by an match_seq, specifically this one:
  586 //
  587 //                   $( $lhs:matchers => $rhs:tt );+
  588 //
  589 // If you understand that, you have closed to loop and understand the whole
  590 // macro system. Congratulations.
  591 //
  592 pub type matcher = spanned<matcher_>;

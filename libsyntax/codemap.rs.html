  11  /*!
  12  
  13  The CodeMap tracks all the source code used within a single crate, mapping
  14  from integer byte positions to the original source code location. Each bit of
  15  source parsed during crate parsing (typically files, in-memory strings, or
  16  various bits of macro expansion) cover a continuous range of bytes in the
  17  CodeMap and are represented by FileMaps. Byte positions are stored in `spans`
  18  and used pervasively in the compiler. They are absolute positions within the
  19  CodeMap, which upon request can be converted to line and column information,
  20  source code snippets, etc.
  21  
  22  */
  23

  78  /**

 197  /// A single source in the CodeMap
 198  pub struct FileMap {
 199      /// The name of the file that the source came from, source that doesn't
 200      /// originate from files has names between angle brackets by convention,
 201      /// e.g. `<anon>`
 202      name: FileName,
 203      /// Extra information used by qquote
 204      substr: FileSubstr,
 205      /// The complete source code
 206      src: @str,
 207      /// The start position of this source in the CodeMap
 208      start_pos: BytePos,
 209      /// Locations of lines beginnings in the source code
 210      lines: @mut ~[BytePos],
 211      /// Locations of multi-byte characters in the source code
 212      multibyte_chars: @mut ~[MultiByteChar],
 213  }
 214  
 215  impl FileMap {
 216      // EFFECT: register a start-of-line offset in the
 217      // table of line-beginnings.
 218      // UNCHECKED INVARIANT: these offsets must be added in the right
 219      // order and must be in the right places; there is shared knowledge
 220      // about what ends a line between this file and parse.rs
 221      pub fn next_line(&self, pos: BytePos) {
 222          // the new charpos must be > the last one (or it's the first one).
 223          let lines = &mut *self.lines;
 224          assert!((lines.len() == 0) || (lines[lines.len() - 1] < pos))
 225          lines.push(pos);
 226      }
 227  
 228      // get a line from the list of pre-computed line-beginnings
 229      pub fn get_line(&self, line: int) -> ~str {
 230          let begin: BytePos = self.lines[line] - self.start_pos;
 231          let begin = begin.to_uint();
 232          let slice = self.src.slice_from(begin);
 233          match slice.find('\n') {
 234              Some(e) => slice.slice_to(e).to_owned(),
 235              None => slice.to_owned()
 236          }
 237      }

 265      pub fn new_filemap_w_substr(&self,
 266                                  filename: FileName,
 267                                  substr: FileSubstr,
 268                                  src: @str)
 269                                  -> @FileMap {
 270          let files = &mut *self.files;
 271          let start_pos = if files.len() == 0 {
 272              0
 273          } else {
 274              let last_start = files.last().start_pos.to_uint();
 275              let last_len = files.last().src.len();
 276              last_start + last_len
 277          };
 278  
 279          let filemap = @FileMap {
 280              name: filename, substr: substr, src: src,
 281              start_pos: BytePos(start_pos),
 282              lines: @mut ~[],
 283              multibyte_chars: @mut ~[],
 284          };
 285  
 286          files.push(filemap);
 287  
 288          return filemap;
 289      }

 359          let begin = self.lookup_byte_offset(sp.lo);

 363          // it's testing isn't true for all spans in the AST, so to allow the

 381  impl CodeMap {
 382      fn lookup_filemap_idx(&self, pos: BytePos) -> uint {
 383          let files = &*self.files;
 384          let len = files.len();
 385          let mut a = 0u;
 386          let mut b = len;
 387          while b - a > 1u {
 388              let m = (a + b) / 2u;
 389              if self.files[m].start_pos > pos {
 390                  b = m;
 391              } else {
 392                  a = m;
 393              }
 394          }
 395          if (a >= len) {
 396              fail!("position %u does not resolve to a source location", pos.to_uint())
 397          }
 398  
 399          return a;
 400      }
 401  
 402      fn lookup_line(&self, pos: BytePos) -> FileMapAndLine
 403      {
 404          let idx = self.lookup_filemap_idx(pos);
 405          let f = self.files[idx];
 406          let mut a = 0u;
 407          let lines = &*f.lines;
 408          let mut b = lines.len();
 409          while b - a > 1u {
 410              let m = (a + b) / 2u;
 411              if lines[m] > pos { b = m; } else { a = m; }
 412          }
 413          return FileMapAndLine {fm: f, line: a};
 414      }
 415  
 416      fn lookup_pos(&self, pos: BytePos) -> Loc {
 417          let FileMapAndLine {fm: f, line: a} = self.lookup_line(pos);
 418          let line = a + 1u; // Line numbers start at 1
 419          let chpos = self.bytepos_to_local_charpos(pos);
 420          let linebpos = f.lines[a];
 421          let linechpos = self.bytepos_to_local_charpos(linebpos);
 422          debug!("codemap: byte pos %? is on the line at byte pos %?",
 423                 pos, linebpos);
 424          debug!("codemap: char pos %? is on the line at char pos %?",
 425                 chpos, linechpos);
 426          debug!("codemap: byte is on line: %?", line);
 427          assert!(chpos >= linechpos);
 428          return Loc {
 429              file: f,
 430              line: line,
 431              col: chpos - linechpos
 432          };
 433      }

  11  /*!
  12  
  13  The CodeMap tracks all the source code used within a single crate, mapping
  14  from integer byte positions to the original source code location. Each bit of
  15  source parsed during crate parsing (typically files, in-memory strings, or
  16  various bits of macro expansion) cover a continuous range of bytes in the
  17  CodeMap and are represented by FileMaps. Byte positions are stored in `spans`
  18  and used pervasively in the compiler. They are absolute positions within the
  19  CodeMap, which upon request can be converted to line and column information,
  20  source code snippets, etc.
  21  
  22  */

 203  /// A single source in the CodeMap
 204  pub struct FileMap {
 205      /// The name of the file that the source came from, source that doesn't
 206      /// originate from files has names between angle brackets by convention,
 207      /// e.g. `<anon>`
 208      name: FileName,
 209      /// The complete source code
 210      src: ~str,
 211      /// The start position of this source in the CodeMap
 212      start_pos: BytePos,
 213      /// Locations of lines beginnings in the source code
 214      lines: RefCell<Vec<BytePos> >,
 215      /// Locations of multi-byte characters in the source code
 216      multibyte_chars: RefCell<Vec<MultiByteChar> >,
 217  }
 218  
 219  impl FileMap {
 220      // EFFECT: register a start-of-line offset in the
 221      // table of line-beginnings.
 222      // UNCHECKED INVARIANT: these offsets must be added in the right
 223      // order and must be in the right places; there is shared knowledge
 224      // about what ends a line between this file and parse.rs
 225      // WARNING: pos param here is the offset relative to start of CodeMap,
 226      // and CodeMap will append a newline when adding a filemap without a newline at the end,
 227      // so the safe way to call this is with value calculated as
 228      // filemap.start_pos + newline_offset_relative_to_the_start_of_filemap.
 229      pub fn next_line(&self, pos: BytePos) {
 230          // the new charpos must be > the last one (or it's the first one).
 231          let mut lines = self.lines.borrow_mut();;
 232          let line_len = lines.len();
 233          assert!(line_len == 0 || (*lines.get(line_len - 1) < pos))
 234          lines.push(pos);
 235      }
 236  
 237      // get a line from the list of pre-computed line-beginnings
 238      pub fn get_line(&self, line: int) -> ~str {
 239          let mut lines = self.lines.borrow_mut();
 240          let begin: BytePos = *lines.get(line as uint) - self.start_pos;
 241          let begin = begin.to_uint();
 242          let slice = self.src.slice_from(begin);
 243          match slice.find('\n') {
 244              Some(e) => slice.slice_to(e).to_owned(),
 245              None => slice.to_owned()
 246          }
 247      }

 274      pub fn new_filemap(&self, filename: FileName, src: ~str) -> Rc<FileMap> {
 275          let mut files = self.files.borrow_mut();
 276          let start_pos = match files.last() {
 277              None => 0,
 278              Some(last) => last.start_pos.to_uint() + last.src.len(),
 279          };
 280  
 281          // Remove utf-8 BOM if any.
 282          // FIXME #12884: no efficient/safe way to remove from the start of a string
 283          // and reuse the allocation.
 284          let mut src = if src.starts_with("\ufeff") {
 285              src.as_slice().slice_from(3).into_owned()
 286          } else {
 287              src
 288          };
 289  
 290          // Append '\n' in case it's not already there.
 291          // This is a workaround to prevent CodeMap.lookup_filemap_idx from accidentally
 292          // overflowing into the next filemap in case the last byte of span is also the last
 293          // byte of filemap, which leads to incorrect results from CodeMap.span_to_*.
 294          if src.len() > 0 && !src.ends_with("\n") {
 295              src.push_char('\n');
 296          }
 297  
 298          let filemap = Rc::new(FileMap {
 299              name: filename,
 300              src: src,
 301              start_pos: Pos::from_uint(start_pos),
 302              lines: RefCell::new(Vec::new()),
 303              multibyte_chars: RefCell::new(Vec::new()),
 304          });
 305  
 306          files.push(filemap.clone());
 307  
 308          filemap
 309      }

 380      fn lookup_filemap_idx(&self, pos: BytePos) -> uint {
 381          let files = self.files.borrow();
 382          let files = files;
 383          let len = files.len();
 384          let mut a = 0u;
 385          let mut b = len;
 386          while b - a > 1u {
 387              let m = (a + b) / 2u;
 388              if files.get(m).start_pos > pos {
 389                  b = m;
 390              } else {
 391                  a = m;
 392              }
 393          }
 394          // There can be filemaps with length 0. These have the same start_pos as the previous
 395          // filemap, but are not the filemaps we want (because they are length 0, they cannot
 396          // contain what we are looking for). So, rewind until we find a useful filemap.
 397          loop {
 398              let lines = files.get(a).lines.borrow();
 399              let lines = lines;
 400              if lines.len() > 0 {
 401                  break;
 402              }
 403              if a == 0 {
 404                  fail!("position {} does not resolve to a source location", pos.to_uint());
 405              }
 406              a -= 1;
 407          }
 408          if a >= len {
 409              fail!("position {} does not resolve to a source location", pos.to_uint())
 410          }
 411  
 412          return a;
 413      }

 415      fn lookup_line(&self, pos: BytePos) -> FileMapAndLine {
 416          let idx = self.lookup_filemap_idx(pos);
 417  
 418          let files = self.files.borrow();
 419          let f = files.get(idx).clone();
 420          let mut a = 0u;
 421          {
 422              let mut lines = f.lines.borrow_mut();
 423              let mut b = lines.len();
 424              while b - a > 1u {
 425                  let m = (a + b) / 2u;
 426                  if *lines.get(m) > pos { b = m; } else { a = m; }
 427              }
 428          }
 429          FileMapAndLine {fm: f, line: a}
 430      }

 432      fn lookup_pos(&self, pos: BytePos) -> Loc {
 433          let FileMapAndLine {fm: f, line: a} = self.lookup_line(pos);

  11  /*!
  12  
  13  The CodeMap tracks all the source code used within a single crate, mapping
  14  from integer byte positions to the original source code location. Each bit of
  15  source parsed during crate parsing (typically files, in-memory strings, or
  16  various bits of macro expansion) cover a continuous range of bytes in the
  17  CodeMap and are represented by FileMaps. Byte positions are stored in `spans`
  18  and used pervasively in the compiler. They are absolute positions within the
  19  CodeMap, which upon request can be converted to line and column information,
  20  source code snippets, etc.
  21  
  22  */
  23

  33  /// A byte offset. Keep this small (currently 32-bits), as AST contains
  34  /// a lot of them.
  35  #[deriving(Clone, Eq, TotalEq, Hash, Ord, Show)]
  36  pub struct BytePos(pub u32);
  37  
  38  /// A character offset. Because of multibyte utf8 characters, a byte offset
  39  /// is not equivalent to a character offset. The CodeMap will convert BytePos
  40  /// values to CharPos values as necessary.
  41  #[deriving(Eq, Hash, Ord, Show)]
  42  pub struct CharPos(pub uint);
  43  
  44  // FIXME: Lots of boilerplate in these impls, but so far my attempts to fix
  45  // have been unsuccessful
  46  
  47  impl Pos for BytePos {
  48      fn from_uint(n: uint) -> BytePos { BytePos(n as u32) }
  49      fn to_uint(&self) -> uint { let BytePos(n) = *self; n as uint }
  50  }

  52  impl Add<BytePos, BytePos> for BytePos {
  53      fn add(&self, rhs: &BytePos) -> BytePos {

  56  }

  58  impl Sub<BytePos, BytePos> for BytePos {

  69  impl Add<CharPos,CharPos> for CharPos {

  81  /**
  82  Spans represent a region of code, used for error reporting. Positions in spans
  83  are *absolute* positions from the beginning of the codemap, not positions
  84  relative to FileMaps. Methods on the CodeMap can be used to relate spans back
  85  to the original source.
  86  */
  87  #[deriving(Clone, Show, Hash)]
  88  pub struct Span {
  89      pub lo: BytePos,
  90      pub hi: BytePos,
  91      /// Information about where the macro came from, if this piece of
  92      /// code was created by a macro expansion.
  93      pub expn_info: Option<@ExpnInfo>
  94  }

 189  /// Extra information for tracking macro expansion of spans
 190  #[deriving(Hash, Show)]

 226  /// A single source in the CodeMap
 227  pub struct FileMap {
 228      /// The name of the file that the source came from, source that doesn't
 229      /// originate from files has names between angle brackets by convention,
 230      /// e.g. `<anon>`
 231      pub name: FileName,
 232      /// The complete source code
 233      pub src: ~str,
 234      /// The start position of this source in the CodeMap
 235      pub start_pos: BytePos,
 236      /// Locations of lines beginnings in the source code
 237      pub lines: RefCell<Vec<BytePos> >,
 238      /// Locations of multi-byte characters in the source code
 239      pub multibyte_chars: RefCell<Vec<MultiByteChar> >,
 240  }
 241  
 242  impl FileMap {
 243      // EFFECT: register a start-of-line offset in the
 244      // table of line-beginnings.
 245      // UNCHECKED INVARIANT: these offsets must be added in the right
 246      // order and must be in the right places; there is shared knowledge
 247      // about what ends a line between this file and parse.rs
 248      // WARNING: pos param here is the offset relative to start of CodeMap,
 249      // and CodeMap will append a newline when adding a filemap without a newline at the end,
 250      // so the safe way to call this is with value calculated as
 251      // filemap.start_pos + newline_offset_relative_to_the_start_of_filemap.
 252      pub fn next_line(&self, pos: BytePos) {
 253          // the new charpos must be > the last one (or it's the first one).
 254          let mut lines = self.lines.borrow_mut();;
 255          let line_len = lines.len();
 256          assert!(line_len == 0 || (*lines.get(line_len - 1) < pos))
 257          lines.push(pos);
 258      }
 259  
 260      // get a line from the list of pre-computed line-beginnings
 261      pub fn get_line(&self, line: int) -> ~str {
 262          let mut lines = self.lines.borrow_mut();
 263          let begin: BytePos = *lines.get(line as uint) - self.start_pos;
 264          let begin = begin.to_uint();
 265          let slice = self.src.slice_from(begin);
 266          match slice.find('\n') {
 267              Some(e) => slice.slice_to(e).to_owned(),
 268              None => slice.to_owned()
 269          }
 270      }
 271  
 272      pub fn record_multibyte_char(&self, pos: BytePos, bytes: uint) {
 273          assert!(bytes >=2 && bytes <= 4);

 275              pos: pos,

 278          self.multibyte_chars.borrow_mut().push(mbc);
 279      }

 297      pub fn new_filemap(&self, filename: FileName, src: ~str) -> Rc<FileMap> {
 298          let mut files = self.files.borrow_mut();
 299          let start_pos = match files.last() {
 300              None => 0,
 301              Some(last) => last.start_pos.to_uint() + last.src.len(),
 302          };
 303  
 304          // Remove utf-8 BOM if any.
 305          // FIXME #12884: no efficient/safe way to remove from the start of a string
 306          // and reuse the allocation.
 307          let mut src = if src.starts_with("\ufeff") {
 308              src.as_slice().slice_from(3).into_owned()
 309          } else {
 310              src
 311          };
 312  
 313          // Append '\n' in case it's not already there.
 314          // This is a workaround to prevent CodeMap.lookup_filemap_idx from accidentally
 315          // overflowing into the next filemap in case the last byte of span is also the last
 316          // byte of filemap, which leads to incorrect results from CodeMap.span_to_*.
 317          if src.len() > 0 && !src.ends_with("\n") {
 318              src.push_char('\n');
 319          }
 320  
 321          let filemap = Rc::new(FileMap {
 322              name: filename,
 323              src: src,
 324              start_pos: Pos::from_uint(start_pos),
 325              lines: RefCell::new(Vec::new()),
 326              multibyte_chars: RefCell::new(Vec::new()),
 327          });
 328  
 329          files.push(filemap.clone());
 330  
 331          filemap
 332      }

 379      pub fn span_to_snippet(&self, sp: Span) -> Option<~str> {
 380          let begin = self.lookup_byte_offset(sp.lo);
 381          let end = self.lookup_byte_offset(sp.hi);
 382  
 383          // FIXME #8256: this used to be an assert but whatever precondition
 384          // it's testing isn't true for all spans in the AST, so to allow the
 385          // caller to not have to fail (and it can't catch it since the CodeMap
 386          // isn't sendable), return None
 387          if begin.fm.start_pos != end.fm.start_pos {
 388              None
 389          } else {
 390              Some(begin.fm.src.slice( begin.pos.to_uint(), end.pos.to_uint()).to_owned())
 391          }
 392      }

 403      fn lookup_filemap_idx(&self, pos: BytePos) -> uint {
 404          let files = self.files.borrow();
 405          let files = files;
 406          let len = files.len();
 407          let mut a = 0u;
 408          let mut b = len;
 409          while b - a > 1u {
 410              let m = (a + b) / 2u;
 411              if files.get(m).start_pos > pos {
 412                  b = m;
 413              } else {
 414                  a = m;
 415              }
 416          }
 417          // There can be filemaps with length 0. These have the same start_pos as the previous
 418          // filemap, but are not the filemaps we want (because they are length 0, they cannot

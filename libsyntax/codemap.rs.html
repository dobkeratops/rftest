  11 /*!
  12 
  13 The CodeMap tracks all the source code used within a single crate, mapping
  14 from integer byte positions to the original source code location. Each bit of
  15 source parsed during crate parsing (typically files, in-memory strings, or
  16 various bits of macro expansion) cover a continuous range of bytes in the
  17 CodeMap and are represented by FileMaps. Byte positions are stored in `spans`
  18 and used pervasively in the compiler. They are absolute positions within the
  19 CodeMap, which upon request can be converted to line and column information,
  20 source code snippets, etc.
  21 
  22 */

 211 /// A single source in the CodeMap

 224     lines: @mut ~[BytePos],

 226     multibyte_chars: @mut ~[MultiByteChar],

 229 impl FileMap {
 230     // EFFECT: register a start-of-line offset in the
 231     // table of line-beginnings.
 232     // UNCHECKED INVARIANT: these offsets must be added in the right
 233     // order and must be in the right places; there is shared knowledge
 234     // about what ends a line between this file and parse.rs
 235     pub fn next_line(&self, pos: BytePos) {
 236         // the new charpos must be > the last one (or it's the first one).
 237         let lines = &mut *self.lines;
 238         assert!((lines.len() == 0) || (lines[lines.len() - 1] < pos))
 239         lines.push(pos);
 240     }
 241 
 242     // get a line from the list of pre-computed line-beginnings
 243     pub fn get_line(&self, line: int) -> ~str {
 244         let begin: BytePos = self.lines[line] - self.start_pos;
 245         let begin = begin.to_uint();
 246         let slice = self.src.slice_from(begin);
 247         match slice.find('\n') {
 248             Some(e) => slice.slice_to(e).to_owned(),
 249             None => slice.to_owned()
 250         }
 251     }

 395 impl CodeMap {
 396     fn lookup_filemap_idx(&self, pos: BytePos) -> uint {
 397         let files = &*self.files;
 398         let len = files.len();
 399         let mut a = 0u;
 400         let mut b = len;
 401         while b - a > 1u {
 402             let m = (a + b) / 2u;
 403             if self.files[m].start_pos > pos {
 404                 b = m;
 405             } else {
 406                 a = m;
 407             }
 408         }
 409         if (a >= len) {
 410             fail!("position %u does not resolve to a source location", pos.to_uint())
 411         }
 412 
 413         return a;
 414     }
 415 
 416     fn lookup_line(&self, pos: BytePos) -> FileMapAndLine
 417     {
 418         let idx = self.lookup_filemap_idx(pos);
 419         let f = self.files[idx];
 420         let mut a = 0u;
 421         let lines = &*f.lines;
 422         let mut b = lines.len();
 423         while b - a > 1u {
 424             let m = (a + b) / 2u;
 425             if lines[m] > pos { b = m; } else { a = m; }
 426         }
 427         return FileMapAndLine {fm: f, line: a};
 428     }
 429 
 430     fn lookup_pos(&self, pos: BytePos) -> Loc {
 431         let FileMapAndLine {fm: f, line: a} = self.lookup_line(pos);
 432         let line = a + 1u; // Line numbers start at 1
 433         let chpos = self.bytepos_to_local_charpos(pos);
 434         let linebpos = f.lines[a];
 435         let linechpos = self.bytepos_to_local_charpos(linebpos);
 436         debug!("codemap: byte pos %? is on the line at byte pos %?",
 437                pos, linebpos);
 438         debug!("codemap: char pos %? is on the line at char pos %?",
 439                chpos, linechpos);
 440         debug!("codemap: byte is on line: %?", line);
 441         assert!(chpos >= linechpos);
 442         return Loc {
 443             file: f,
 444             line: line,
 445             col: chpos - linechpos
 446         };
 447     }

 464     // Converts an absolute BytePos to a CharPos relative to the file it is
 465     // located in

  25  pub trait AttrMetaMethods {
  26      // This could be changed to `fn check_name(&self, name: InternedString) ->

  31      /// #[foo="bar"] and #[foo(bar)]

  40      fn meta_item_list<'a>(&'a self) -> Option<&'a [@MetaItem]>;
  41  
  42      /**
  43       * If the meta item is a name-value type with a string value then returns
  44       * a tuple containing the name and string value, otherwise `None`
  45       */

  47  }

  62  impl AttrMetaMethods for MetaItem {
  63      fn name(&self) -> InternedString {
  64          match self.node {
  65              MetaWord(ref n) => (*n).clone(),
  66              MetaNameValue(ref n, _) => (*n).clone(),
  67              MetaList(ref n, _) => (*n).clone(),
  68          }
  69      }
  70  
  71      fn value_str(&self) -> Option<InternedString> {
  72          match self.node {
  73              MetaNameValue(_, ref v) => {
  74                  match v.node {
  75                      ast::LitStr(ref s, _) => Some((*s).clone()),
  76                      _ => None,
  77                  }
  78              },
  79              _ => None
  80          }
  81      }
  82  
  83      fn meta_item_list<'a>(&'a self) -> Option<&'a [@MetaItem]> {
  84          match self.node {
  85              MetaList(_, ref l) => Some(l.as_slice()),
  86              _ => None
  87          }
  88      }
  89  
  90      fn name_str_pair(&self) -> Option<(InternedString,InternedString)> {
  91          self.value_str().map(|s| (self.name(), s))
  92      }
  93  }

 113  impl AttributeMethods for Attribute {
 114      /// Extract the MetaItem from inside this Attribute.
 115      fn meta(&self) -> @MetaItem {
 116          self.node.value
 117      }
 118  
 119      /// Convert self to a normal #[doc="foo"] comment, if it is a
 120      /// comment like `///` or `/** */`. (Returns self unchanged for
 121      /// non-sugared doc attributes.)
 122      fn desugar_doc(&self) -> Attribute {
 123          if self.node.is_sugared_doc {
 124              let comment = self.value_str().unwrap();
 125              let meta = mk_name_value_item_str(
 126                  InternedString::new("doc"),
 127                  token::intern_and_get_ident(strip_doc_comment_decoration(
 128                          comment.get())));
 129              mk_attr(meta)
 130          } else {
 131              *self
 132          }
 133      }
 134  }

 215  pub fn sort_meta_items(items: &[@MetaItem]) -> Vec<@MetaItem> {
 216      // This is sort of stupid here, but we need to sort by
 217      // human-readable strings.
 218      let mut v = items.iter()
 219          .map(|&mi| (mi.name(), mi))
 220          .collect::<Vec<(InternedString, @MetaItem)> >();
 221  
 222      v.sort_by(|&(ref a, _), &(ref b, _)| a.cmp(b));
 223  
 224      // There doesn't seem to be a more optimal way to do this
 225      v.move_iter().map(|(_, m)| {
 226          match m.node {
 227              MetaList(ref n, ref mis) => {
 228                  @Spanned {
 229                      node: MetaList((*n).clone(),
 230                                     sort_meta_items(mis.as_slice())),
 231                      .. /*bad*/ (*m).clone()
 232                  }
 233              }
 234              _ => m
 235          }
 236      }).collect()
 237  }

 269  /// True if something like #[inline] is found in the list of attrs.
 270  pub fn find_inline_attr(attrs: &[Attribute]) -> InlineAttr {
 271      // FIXME (#2809)---validate the usage of #[inline] and #[inline]
 272      attrs.iter().fold(InlineNone, |ia,attr| {
 273          match attr.node.value.node {
 274            MetaWord(ref n) if n.equiv(&("inline")) => InlineHint,
 275            MetaList(ref n, ref items) if n.equiv(&("inline")) => {
 276              if contains_name(items.as_slice(), "always") {
 277                  InlineAlways
 278              } else if contains_name(items.as_slice(), "never") {
 279                  InlineNever
 280              } else {
 281                  InlineHint
 282              }
 283            }
 284            _ => ia
 285          }
 286      })
 287  }

 289  /// Tests if any `cfg(...)` meta items in `metas` match `cfg`. e.g.
 290  ///
 291  /// test_cfg(`[foo="a", bar]`, `[cfg(foo), cfg(bar)]`) == true
 292  /// test_cfg(`[foo="a", bar]`, `[cfg(not(bar))]`) == false
 293  /// test_cfg(`[foo="a", bar]`, `[cfg(bar, foo="a")]`) == true
 294  /// test_cfg(`[foo="a", bar]`, `[cfg(bar, foo="b")]`) == false
 295  pub fn test_cfg<AM: AttrMetaMethods, It: Iterator<AM>>
 296      (cfg: &[@MetaItem], mut metas: It) -> bool {
 297      // having no #[cfg(...)] attributes counts as matching.
 298      let mut no_cfgs = true;
 299  
 300      // this would be much nicer as a chain of iterator adaptors, but
 301      // this doesn't work.
 302      let some_cfg_matches = metas.any(|mi| {
 303          debug!("testing name: {}", mi.name());
 304          if mi.name().equiv(&("cfg")) { // it is a #[cfg()] attribute
 305              debug!("is cfg");
 306              no_cfgs = false;
 307               // only #[cfg(...)] ones are understood.
 308              match mi.meta_item_list() {
 309                  Some(cfg_meta) => {
 310                      debug!("is cfg(...)");
 311                      cfg_meta.iter().all(|cfg_mi| {
 312                          debug!("cfg({}[...])", cfg_mi.name());
 313                          match cfg_mi.node {
 314                              ast::MetaList(ref s, ref not_cfgs)
 315                              if s.equiv(&("not")) => {
 316                                  debug!("not!");
 317                                  // inside #[cfg(not(...))], so these need to all
 318                                  // not match.
 319                                  !not_cfgs.iter().all(|mi| {
 320                                      debug!("cfg(not({}[...]))", mi.name());
 321                                      contains(cfg, *mi)
 322                                  })
 323                              }
 324                              _ => contains(cfg, *cfg_mi)
 325                          }
 326                      })
 327                  }
 328                  None => false
 329              }
 330          } else {
 331              false
 332          }
 333      });
 334      debug!("test_cfg (no_cfgs={}, some_cfg_matches={})", no_cfgs, some_cfg_matches);
 335      no_cfgs || some_cfg_matches
 336  }

 355  /// Find the first stability attribute. `None` if none exists.
 356  pub fn find_stability<AM: AttrMetaMethods, It: Iterator<AM>>(mut metas: It)
 357                        -> Option<Stability> {
 358      for m in metas {
 359          let level = match m.name().get() {
 360              "deprecated" => Deprecated,
 361              "experimental" => Experimental,
 362              "unstable" => Unstable,
 363              "stable" => Stable,
 364              "frozen" => Frozen,
 365              "locked" => Locked,
 366              _ => continue // not a stability level
 367          };
 368  
 369          return Some(Stability {
 370                  level: level,
 371                  text: m.value_str()
 372              });
 373      }
 374      None
 375  }

 390  /**
 391   * Fold this over attributes to parse #[repr(...)] forms.
 392   *
 393   * Valid repr contents: any of the primitive integral type names (see
 394   * `int_type_of_word`, below) to specify the discriminant type; and `C`, to use
 395   * the same discriminant size that the corresponding C enum would.  These are
 396   * not allowed on univariant or zero-variant enums, which have no discriminant.
 397   *
 398   * If a discriminant type is so specified, then the discriminant will be
 399   * present (before fields, if any) with that type; reprensentation
 400   * optimizations which would remove it will not be done.
 401   */
 402  pub fn find_repr_attr(diagnostic: &SpanHandler, attr: @ast::MetaItem, acc: ReprAttr)
 403      -> ReprAttr {
 404      let mut acc = acc;
 405      match attr.node {
 406          ast::MetaList(ref s, ref items) if s.equiv(&("repr")) => {
 407              for item in items.iter() {
 408                  match item.node {
 409                      ast::MetaWord(ref word) => {
 410                          let hint = match word.get() {
 411                              // Can't use "extern" because it's not a lexical identifier.
 412                              "C" => ReprExtern,
 413                              _ => match int_type_of_word(word.get()) {
 414                                  Some(ity) => ReprInt(item.span, ity),
 415                                  None => {
 416                                      // Not a word we recognize
 417                                      diagnostic.span_err(item.span,
 418                                                          "unrecognized representation hint");
 419                                      ReprAny
 420                                  }
 421                              }
 422                          };
 423                          if hint != ReprAny {
 424                              if acc == ReprAny {
 425                                  acc = hint;
 426                              } else if acc != hint {
 427                                  diagnostic.span_warn(item.span,
 428                                                       "conflicting representation hint ignored")
 429                              }
 430                          }
 431                      }
 432                      // Not a word:
 433                      _ => diagnostic.span_err(item.span, "unrecognized representation hint")
 434                  }
 435              }
 436          }
 437          // Not a "repr" hint: ignore.
 438          _ => { }
 439      }
 440      acc
 441  }

 443  fn int_type_of_word(s: &str) -> Option<IntType> {
 444      match s {
 445          "i8" => Some(SignedInt(ast::TyI8)),
 446          "u8" => Some(UnsignedInt(ast::TyU8)),
 447          "i16" => Some(SignedInt(ast::TyI16)),
 448          "u16" => Some(UnsignedInt(ast::TyU16)),
 449          "i32" => Some(SignedInt(ast::TyI32)),
 450          "u32" => Some(UnsignedInt(ast::TyU32)),
 451          "i64" => Some(SignedInt(ast::TyI64)),
 452          "u64" => Some(UnsignedInt(ast::TyU64)),
 453          "int" => Some(SignedInt(ast::TyI)),
 454          "uint" => Some(UnsignedInt(ast::TyU)),
 455          _ => None
 456      }

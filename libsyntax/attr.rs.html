 200 pub fn sort_meta_items(items: &[@MetaItem]) -> ~[@MetaItem] {
 201     // This is sort of stupid here, but we need to sort by
 202     // human-readable strings.
 203     let mut v = items.iter()
 204         .transform(|&mi| (mi.name(), mi))
 205         .collect::<~[(@str, @MetaItem)]>();
 206 
 207     do extra::sort::quick_sort(v) |&(a, _), &(b, _)| {
 208         a <= b
 209     }
 210 
 211     // There doesn't seem to be a more optimal way to do this
 212     do v.consume_iter().transform |(_, m)| {
 213         match m.node {
 214             MetaList(n, ref mis) => {
 215                 @spanned {
 216                     node: MetaList(n, sort_meta_items(*mis)),
 217                     .. /*bad*/ (*m).clone()

 219             }
 220             _ => m

 270 /// test_cfg(`[foo="a", bar]`, `[cfg(foo), cfg(bar)]`) == true

 308         } else {
 309             false
 310         }
 311     };
 312     debug!("test_cfg (no_cfgs=%?, some_cfg_matches=%?)", no_cfgs, some_cfg_matches);

 314 }

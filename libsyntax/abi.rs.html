 199     pub fn for_arch(&self, arch: Architecture) -> Option<Abi> {
 200         // NB---Single platform ABIs come first

 216     pub fn check_valid(&self) -> Option<(Abi, Abi)> {
 217         let mut abis = ~[];
 218         do self.each |abi| { abis.push(abi); true };
 219 
 220         for (i, abi) in abis.iter().enumerate() {
 221             let data = abi.data();
 222             for other_abi in abis.slice(0, i).iter() {
 223                 let other_data = other_abi.data();
 224                 debug!("abis=(%?,%?) datas=(%?,%?)",
 225                        abi, data.abi_arch,
 226                        other_abi, other_data.abi_arch);
 227                 match (&data.abi_arch, &other_data.abi_arch) {
 228                     (&AllArch, &AllArch) => {
 229                         // Two cross-architecture ABIs
 230                         return Some((*abi, *other_abi));
 231                     }
 232                     (_, &RustArch) |
 233                     (&RustArch, _) => {
 234                         // Cannot combine Rust or Rust-Intrinsic with
 235                         // anything else.
 236                         return Some((*abi, *other_abi));
 237                     }
 238                     (&Archs(is), &Archs(js)) if (is & js) != 0 => {
 239                         // Two ABIs for same architecture
 240                         return Some((*abi, *other_abi));
 241                     }
 242                     _ => {}
 243                 }
 244             }
 245         }
 246 
 247         return None;
 248     }
 249 }

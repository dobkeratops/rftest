 260  fn highlight_lines(cm: @codemap::CodeMap,
 261                     sp: Span,
 262                     lvl: level,
 263                     lines: @codemap::FileLines) {
 264      let fm = lines.file;
 265  
 266      // arbitrarily only print up to six lines of the error
 267      let max_lines = 6u;
 268      let mut elided = false;
 269      let mut display_lines = /* FIXME (#2543) */ lines.lines.clone();
 270      if display_lines.len() > max_lines {
 271          display_lines = display_lines.slice(0u, max_lines).to_owned();
 272          elided = true;
 273      }
 274      // Print the offending lines
 275      for line in display_lines.iter() {
 276          io::stderr().write_str(fmt!("%s:%u ", fm.name, *line + 1u));
 277          let s = fm.get_line(*line as int) + "\n";
 278          io::stderr().write_str(s);
 279      }
 280      if elided {
 281          let last_line = display_lines[display_lines.len() - 1u];
 282          let s = fmt!("%s:%u ", fm.name, last_line + 1u);
 283          let mut indent = s.len();
 284          let mut out = ~"";
 285          while indent > 0u {
 286              out.push_char(' ');
 287              indent -= 1u;
 288          }
 289          out.push_str("...\n");
 290          io::stderr().write_str(out);
 291      }
 292  
 293      // FIXME (#3260)
 294      // If there's one line at fault we can easily point to the problem
 295      if lines.lines.len() == 1u {
 296          let lo = cm.lookup_char_pos(sp.lo);
 297          let mut digits = 0u;
 298          let mut num = (lines.lines[0] + 1u) / 10u;
 299  
 300          // how many digits must be indent past?
 301          while num > 0u { num /= 10u; digits += 1u; }
 302  
 303          // indent past |name:## | and the 0-offset column location
 304          let left = fm.name.len() + digits + lo.col.to_uint() + 3u;
 305          let mut s = ~"";
 306          // Skip is the number of characters we need to skip because they are
 307          // part of the 'filename:line ' part of the previous line.
 308          let skip = fm.name.len() + digits + 3u;
 309          do skip.times() {
 310              s.push_char(' ');
 311          }
 312          let orig = fm.get_line(lines.lines[0] as int);
 313          for pos in range(0u, left-skip) {
 314              let curChar = (orig[pos] as char);
 315              // Whenever a tab occurs on the previous line, we insert one on
 316              // the error-point-squiggly-line as well (instead of a space).
 317              // That way the squiggly line will usually appear in the correct
 318              // position.
 319              match curChar {
 320                  '\t' => s.push_char('\t'),
 321                  _ => s.push_char(' '),
 322              };
 323          }
 324          io::stderr().write_str(s);
 325          let mut s = ~"^";
 326          let hi = cm.lookup_char_pos(sp.hi);
 327          if hi.col != lo.col {
 328              // the ^ already takes up one space
 329              let num_squigglies = hi.col.to_uint()-lo.col.to_uint()-1u;
 330              do num_squigglies.times() {
 331                  s.push_char('~')
 332              }
 333          }
 334          print_maybe_styled(s + "\n", term::attr::ForegroundColor(diagnosticcolor(lvl)));
 335      }
 336  }

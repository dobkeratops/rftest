  26  pub enum RenderSpan {
  27      /// A FullSpan renders with both with an initial line for the
  28      /// message, prefixed by file:linenum, followed by a summary of

  43      fn is_full_span(&self) -> bool {
  44          match self {

 108  // a handler deals with errors; certain errors
 109  // (fatal, bug, unimpl) may cause immediate exit,
 110  // others log errors for later reporting.
 111  pub struct Handler {
 112      err_count: Cell<uint>,
 113      emit: RefCell<~Emitter:Send>,
 114  }
 115  
 116  impl Handler {
 117      pub fn fatal(&self, msg: &str) -> ! {
 118          self.emit.borrow_mut().emit(None, msg, Fatal);
 119          fail!(FatalError);
 120      }
 121      pub fn err(&self, msg: &str) {
 122          self.emit.borrow_mut().emit(None, msg, Error);
 123          self.bump_err_count();
 124      }
 125      pub fn bump_err_count(&self) {
 126          self.err_count.set(self.err_count.get() + 1u);
 127      }
 128      pub fn err_count(&self) -> uint {
 129          self.err_count.get()
 130      }
 131      pub fn has_errors(&self) -> bool {
 132          self.err_count.get()> 0u
 133      }
 134      pub fn abort_if_errors(&self) {
 135          let s;
 136          match self.err_count.get() {
 137            0u => return,
 138            1u => s = ~"aborting due to previous error",
 139            _  => {
 140              s = format!("aborting due to {} previous errors",
 141                       self.err_count.get());
 142            }
 143          }
 144          self.fatal(s);
 145      }
 146      pub fn warn(&self, msg: &str) {
 147          self.emit.borrow_mut().emit(None, msg, Warning);
 148      }
 149      pub fn note(&self, msg: &str) {
 150          self.emit.borrow_mut().emit(None, msg, Note);
 151      }
 152      pub fn bug(&self, msg: &str) -> ! {
 153          self.emit.borrow_mut().emit(None, msg, Bug);
 154          fail!(ExplicitBug);
 155      }
 156      pub fn unimpl(&self, msg: &str) -> ! {
 157          self.bug(~"unimplemented " + msg);
 158      }
 159      pub fn emit(&self,
 160                  cmsp: Option<(&codemap::CodeMap, Span)>,
 161                  msg: &str,
 162                  lvl: Level) {
 163          self.emit.borrow_mut().emit(cmsp, msg, lvl);
 164      }
 165      pub fn custom_emit(&self, cm: &codemap::CodeMap,
 166                         sp: RenderSpan, msg: &str, lvl: Level) {
 167          self.emit.borrow_mut().custom_emit(cm, sp, msg, lvl);
 168      }
 169  }

 221  fn print_maybe_styled(w: &mut EmitterWriter,
 222                        msg: &str,
 223                        color: term::attr::Attr) -> io::IoResult<()> {
 224      match w.dst {
 225          Terminal(ref mut t) => {
 226              try!(t.attr(color));
 227              try!(t.write_str(msg));
 228              try!(t.reset());
 229              Ok(())
 230          }
 231          Raw(ref mut w) => {
 232              w.write_str(msg)
 233          }
 234      }
 235  }

 237  fn print_diagnostic(dst: &mut EmitterWriter,
 238                      topic: &str, lvl: Level, msg: &str) -> io::IoResult<()> {
 239      if !topic.is_empty() {
 240          try!(write!(&mut dst.dst, "{} ", topic));
 241      }
 242  
 243      try!(print_maybe_styled(dst, format!("{}: ", lvl.to_str()),
 244                              term::attr::ForegroundColor(lvl.color())));
 245      try!(print_maybe_styled(dst, format!("{}\n", msg), term::attr::Bold));
 246      Ok(())
 247  }

 258  impl EmitterWriter {
 259      pub fn stderr() -> EmitterWriter {
 260          let stderr = io::stderr();
 261          if stderr.get_ref().isatty() {
 262              let dst = match term::Terminal::new(stderr.unwrap()) {
 263                  Ok(t) => Terminal(t),
 264                  Err(..) => Raw(~io::stderr()),
 265              };
 266              EmitterWriter { dst: dst }
 267          } else {
 268              EmitterWriter { dst: Raw(~stderr) }
 269          }
 270      }

 286  impl Emitter for EmitterWriter {
 287      fn emit(&mut self,
 288              cmsp: Option<(&codemap::CodeMap, Span)>,
 289              msg: &str,
 290              lvl: Level) {
 291          let error = match cmsp {
 292              Some((cm, sp)) => emit(self, cm, FullSpan(sp), msg, lvl, false),
 293              None => print_diagnostic(self, "", lvl, msg),
 294          };
 295  
 296          match error {
 297              Ok(()) => {}
 298              Err(e) => fail!("failed to print diagnostics: {}", e),
 299          }
 300      }

 302      fn custom_emit(&mut self, cm: &codemap::CodeMap,
 303                     sp: RenderSpan, msg: &str, lvl: Level) {
 304          match emit(self, cm, sp, msg, lvl, true) {
 305              Ok(()) => {}
 306              Err(e) => fail!("failed to print diagnostics: {}", e),
 307          }
 308      }
 309  }

 311  fn emit(dst: &mut EmitterWriter, cm: &codemap::CodeMap, rsp: RenderSpan,
 312          msg: &str, lvl: Level, custom: bool) -> io::IoResult<()> {
 313      let sp = rsp.span();
 314      let ss = cm.span_to_str(sp);
 315      let lines = cm.span_to_lines(sp);
 316      if custom {
 317          // we want to tell compiletest/runtest to look at the last line of the
 318          // span (since `custom_highlight_lines` displays an arrow to the end of
 319          // the span)
 320          let span_end = Span { lo: sp.hi, hi: sp.hi, expn_info: sp.expn_info};
 321          let ses = cm.span_to_str(span_end);
 322          try!(print_diagnostic(dst, ses, lvl, msg));
 323          if rsp.is_full_span() {
 324              try!(custom_highlight_lines(dst, cm, sp, lvl, lines));
 325          }
 326      } else {
 327          try!(print_diagnostic(dst, ss, lvl, msg));
 328          if rsp.is_full_span() {
 329              try!(highlight_lines(dst, cm, sp, lvl, lines));
 330          }
 331      }
 332      print_macro_backtrace(dst, cm, sp)
 333  }

 335  fn highlight_lines(err: &mut EmitterWriter,
 336                     cm: &codemap::CodeMap,
 337                     sp: Span,
 338                     lvl: Level,
 339                     lines: codemap::FileLines) -> io::IoResult<()> {
 340      let fm = &*lines.file;
 341  
 342      let mut elided = false;
 343      let mut display_lines = lines.lines.as_slice();
 344      if display_lines.len() > MAX_LINES {
 345          display_lines = display_lines.slice(0u, MAX_LINES);
 346          elided = true;
 347      }
 348      // Print the offending lines
 349      for line in display_lines.iter() {
 350          try!(write!(&mut err.dst, "{}:{} {}\n", fm.name, *line + 1,
 351                      fm.get_line(*line as int)));
 352      }
 353      if elided {
 354          let last_line = display_lines[display_lines.len() - 1u];
 355          let s = format!("{}:{} ", fm.name, last_line + 1u);
 356          try!(write!(&mut err.dst, "{0:1$}...\n", "", s.len()));
 357      }
 358  
 359      // FIXME (#3260)
 360      // If there's one line at fault we can easily point to the problem
 361      if lines.lines.len() == 1u {
 362          let lo = cm.lookup_char_pos(sp.lo);
 363          let mut digits = 0u;
 364          let mut num = (*lines.lines.get(0) + 1u) / 10u;
 365  
 366          // how many digits must be indent past?
 367          while num > 0u { num /= 10u; digits += 1u; }
 368  
 369          // indent past |name:## | and the 0-offset column location
 370          let left = fm.name.len() + digits + lo.col.to_uint() + 3u;
 371          let mut s = ~"";
 372          // Skip is the number of characters we need to skip because they are
 373          // part of the 'filename:line ' part of the previous line.
 374          let skip = fm.name.len() + digits + 3u;
 375          for _ in range(0, skip) { s.push_char(' '); }
 376          let orig = fm.get_line(*lines.lines.get(0) as int);
 377          for pos in range(0u, left-skip) {
 378              let cur_char = orig[pos] as char;
 379              // Whenever a tab occurs on the previous line, we insert one on
 380              // the error-point-squiggly-line as well (instead of a space).
 381              // That way the squiggly line will usually appear in the correct
 382              // position.
 383              match cur_char {
 384                  '\t' => s.push_char('\t'),
 385                  _ => s.push_char(' '),
 386              };
 387          }
 388          try!(write!(&mut err.dst, "{}", s));
 389          let mut s = ~"^";
 390          let hi = cm.lookup_char_pos(sp.hi);
 391          if hi.col != lo.col {
 392              // the ^ already takes up one space
 393              let num_squigglies = hi.col.to_uint()-lo.col.to_uint()-1u;
 394              for _ in range(0, num_squigglies) { s.push_char('~'); }
 395          }
 396          try!(print_maybe_styled(err, s + "\n",
 397                                  term::attr::ForegroundColor(lvl.color())));
 398      }
 399      Ok(())
 400  }
 401  
 402  // Here are the differences between this and the normal `highlight_lines`:
 403  // `custom_highlight_lines` will always put arrow on the last byte of the
 404  // span (instead of the first byte). Also, when the span is too long (more
 405  // than 6 lines), `custom_highlight_lines` will print the first line, then
 406  // dot dot dot, then last line, whereas `highlight_lines` prints the first
 407  // six lines.
 408  fn custom_highlight_lines(w: &mut EmitterWriter,
 409                            cm: &codemap::CodeMap,
 410                            sp: Span,
 411                            lvl: Level,
 412                            lines: codemap::FileLines) -> io::IoResult<()> {
 413      let fm = &*lines.file;
 414  
 415      let lines = lines.lines.as_slice();
 416      if lines.len() > MAX_LINES {
 417          try!(write!(&mut w.dst, "{}:{} {}\n", fm.name,
 418                      lines[0] + 1, fm.get_line(lines[0] as int)));
 419          try!(write!(&mut w.dst, "...\n"));
 420          let last_line = lines[lines.len()-1];
 421          try!(write!(&mut w.dst, "{}:{} {}\n", fm.name,
 422                      last_line + 1, fm.get_line(last_line as int)));
 423      } else {
 424          for line in lines.iter() {
 425              try!(write!(&mut w.dst, "{}:{} {}\n", fm.name,
 426                          *line + 1, fm.get_line(*line as int)));
 427          }
 428      }
 429      let last_line_start = format!("{}:{} ", fm.name, lines[lines.len()-1]+1);
 430      let hi = cm.lookup_char_pos(sp.hi);
 431      // Span seems to use half-opened interval, so subtract 1
 432      let skip = last_line_start.len() + hi.col.to_uint() - 1;
 433      let mut s = ~"";
 434      for _ in range(0, skip) { s.push_char(' '); }
 435      s.push_char('^');
 436      print_maybe_styled(w, s + "\n", term::attr::ForegroundColor(lvl.color()))

 299  fn highlight_lines(err: &mut EmitterWriter,
 300                     cm: &codemap::CodeMap,
 301                     sp: Span,
 302                     lvl: Level,
 303                     lines: codemap::FileLines) -> io::IoResult<()> {
 304      let fm = &*lines.file;
 305  
 306      let mut elided = false;
 307      let mut display_lines = lines.lines.as_slice();
 308      if display_lines.len() > MAX_LINES {
 309          display_lines = display_lines.slice(0u, MAX_LINES);
 310          elided = true;
 311      }
 312      // Print the offending lines
 313      for line in display_lines.iter() {
 314          try!(write!(&mut err.dst, "{}:{} {}\n", fm.name, *line + 1,
 315                      fm.get_line(*line as int)));
 316      }
 317      if elided {
 318          let last_line = display_lines[display_lines.len() - 1u];
 319          let s = format!("{}:{} ", fm.name, last_line + 1u);
 320          try!(write!(&mut err.dst, "{0:1$}...\n", "", s.len()));
 321      }
 322  
 323      // FIXME (#3260)
 324      // If there's one line at fault we can easily point to the problem
 325      if lines.lines.len() == 1u {
 326          let lo = cm.lookup_char_pos(sp.lo);
 327          let mut digits = 0u;
 328          let mut num = (*lines.lines.get(0) + 1u) / 10u;
 329  
 330          // how many digits must be indent past?
 331          while num > 0u { num /= 10u; digits += 1u; }
 332  
 333          // indent past |name:## | and the 0-offset column location
 334          let left = fm.name.len() + digits + lo.col.to_uint() + 3u;
 335          let mut s = ~"";
 336          // Skip is the number of characters we need to skip because they are
 337          // part of the 'filename:line ' part of the previous line.
 338          let skip = fm.name.len() + digits + 3u;
 339          for _ in range(0, skip) { s.push_char(' '); }
 340          let orig = fm.get_line(*lines.lines.get(0) as int);
 341          for pos in range(0u, left-skip) {
 342              let cur_char = orig[pos] as char;
 343              // Whenever a tab occurs on the previous line, we insert one on
 344              // the error-point-squiggly-line as well (instead of a space).

 346              // position.
 347              match cur_char {
 348                  '\t' => s.push_char('\t'),
 349                  _ => s.push_char(' '),
 350              };
 351          }
 352          try!(write!(&mut err.dst, "{}", s));
 353          let mut s = ~"^";
 354          let hi = cm.lookup_char_pos(sp.hi);
 355          if hi.col != lo.col {
 356              // the ^ already takes up one space
 357              let num_squigglies = hi.col.to_uint()-lo.col.to_uint()-1u;
 358              for _ in range(0, num_squigglies) { s.push_char('~'); }
 359          }
 360          try!(print_maybe_styled(err, s + "\n",
 361                                  term::attr::ForegroundColor(lvl.color())));
 362      }

 366  // Here are the differences between this and the normal `highlight_lines`:
 367  // `custom_highlight_lines` will always put arrow on the last byte of the
 368  // span (instead of the first byte). Also, when the span is too long (more
 369  // than 6 lines), `custom_highlight_lines` will print the first line, then
 370  // dot dot dot, then last line, whereas `highlight_lines` prints the first
 371  // six lines.

 373                            cm: &codemap::CodeMap,
 374                            sp: Span,
 375                            lvl: Level,
 376                            lines: codemap::FileLines) -> io::IoResult<()> {

 381          try!(write!(&mut w.dst, "{}:{} {}\n", fm.name,
 382                      lines[0] + 1, fm.get_line(lines[0] as int)));
 383          try!(write!(&mut w.dst, "...\n"));
 384          let last_line = lines[lines.len()-1];
 385          try!(write!(&mut w.dst, "{}:{} {}\n", fm.name,
 386                      last_line + 1, fm.get_line(last_line as int)));
 387      } else {
 388          for line in lines.iter() {
 389              try!(write!(&mut w.dst, "{}:{} {}\n", fm.name,
 390                          *line + 1, fm.get_line(*line as int)));
 391          }
 392      }
 393      let last_line_start = format!("{}:{} ", fm.name, lines[lines.len()-1]+1);
 394      let hi = cm.lookup_char_pos(sp.hi);
 395      // Span seems to use half-opened interval, so subtract 1
 396      let skip = last_line_start.len() + hi.col.to_uint() - 1;

 398      for _ in range(0, skip) { s.push_char(' '); }

 256 fn highlight_lines(cm: @codemap::CodeMap,
 257                    sp: span, lvl: level,
 258                    lines: @codemap::FileLines) {
 259     let fm = lines.file;
 260 
 261     // arbitrarily only print up to six lines of the error
 262     let max_lines = 6u;
 263     let mut elided = false;
 264     let mut display_lines = /* FIXME (#2543) */ lines.lines.clone();
 265     if display_lines.len() > max_lines {
 266         display_lines = display_lines.slice(0u, max_lines).to_owned();
 267         elided = true;
 268     }
 269     // Print the offending lines
 270     for line in display_lines.iter() {
 271         io::stderr().write_str(fmt!("%s:%u ", fm.name, *line + 1u));
 272         let s = fm.get_line(*line as int) + "\n";
 273         io::stderr().write_str(s);
 274     }
 275     if elided {
 276         let last_line = display_lines[display_lines.len() - 1u];
 277         let s = fmt!("%s:%u ", fm.name, last_line + 1u);
 278         let mut indent = s.len();
 279         let mut out = ~"";
 280         while indent > 0u {
 281             out.push_char(' ');
 282             indent -= 1u;
 283         }
 284         out.push_str("...\n");
 285         io::stderr().write_str(out);
 286     }
 287 
 288     // FIXME (#3260)
 289     // If there's one line at fault we can easily point to the problem
 290     if lines.lines.len() == 1u {
 291         let lo = cm.lookup_char_pos(sp.lo);
 292         let mut digits = 0u;
 293         let mut num = (lines.lines[0] + 1u) / 10u;
 294 
 295         // how many digits must be indent past?
 296         while num > 0u { num /= 10u; digits += 1u; }
 297 
 298         // indent past |name:## | and the 0-offset column location
 299         let left = fm.name.len() + digits + lo.col.to_uint() + 3u;
 300         let mut s = ~"";
 301         // Skip is the number of characters we need to skip because they are
 302         // part of the 'filename:line ' part of the previous line.
 303         let skip = fm.name.len() + digits + 3u;
 304         do skip.times() {
 305             s.push_char(' ');
 306         }
 307         let orig = fm.get_line(lines.lines[0] as int);
 308         for pos in range(0u, left-skip) {
 309             let curChar = (orig[pos] as char);
 310             // Whenever a tab occurs on the previous line, we insert one on
 311             // the error-point-squiggly-line as well (instead of a space).
 312             // That way the squiggly line will usually appear in the correct
 313             // position.
 314             match curChar {
 315                 '\t' => s.push_char('\t'),
 316                 _ => s.push_char(' '),
 317             };
 318         }
 319         io::stderr().write_str(s);
 320         let mut s = ~"^";
 321         let hi = cm.lookup_char_pos(sp.hi);
 322         if hi.col != lo.col {
 323             // the ^ already takes up one space
 324             let num_squigglies = hi.col.to_uint()-lo.col.to_uint()-1u;
 325             do num_squigglies.times() {
 326                 s.push_char('~')
 327             }
 328         }
 329         print_maybe_styled(s + "\n", term::attr::ForegroundColor(diagnosticcolor(lvl)));
 330     }
 331 }

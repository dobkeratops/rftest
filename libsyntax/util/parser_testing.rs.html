  18 // map a string to tts, using a made-up filename: return both the token_trees
  19 // and the ParseSess
  20 pub fn string_to_tts_and_sess (source_str : @str) -> (~[ast::token_tree],@mut ParseSess) {
  21     let ps = new_parse_sess(None);
  22     (filemap_to_tts(ps,string_to_filemap(ps,source_str,@"bogofile")),ps)
  23 }

  89 // does the given string match the pattern? whitespace in the first string
  90 // may be deleted or replaced with other whitespace to match the pattern.
  91 // this function is unicode-ignorant; fortunately, the careful design of
  92 // UTF-8 mitigates this ignorance.  In particular, this function only collapses
  93 // sequences of \n, \r, ' ', and \t, but it should otherwise tolerate unicode
  94 // chars. Unsurprisingly, it doesn't do NKF-normalization(?).
  95 pub fn matches_codepattern(a : &str, b : &str) -> bool {
  96     let mut idx_a = 0;
  97     let mut idx_b = 0;
  98     loop {
  99         if (idx_a == a.len() && idx_b == b.len()) {
 100             return true;
 101         }
 102         else if (idx_a == a.len()) {return false;}
 103         else if (idx_b == b.len()) {
 104             // maybe the stuff left in a is all ws?
 105             if (is_whitespace(a.char_at(idx_a))) {
 106                 return (scan_for_non_ws_or_end(a,idx_a) == a.len());
 107             } else {
 108                 return false;
 109             }
 110         }
 111         // ws in both given and pattern:
 112         else if (is_whitespace(a.char_at(idx_a))
 113            && is_whitespace(b.char_at(idx_b))) {
 114             idx_a = scan_for_non_ws_or_end(a,idx_a);
 115             idx_b = scan_for_non_ws_or_end(b,idx_b);
 116         }
 117         // ws in given only:
 118         else if (is_whitespace(a.char_at(idx_a))) {
 119             idx_a = scan_for_non_ws_or_end(a,idx_a);
 120         }
 121         // *don't* silently eat ws in expected only.
 122         else if (a.char_at(idx_a) == b.char_at(idx_b)) {
 123             idx_a += 1;
 124             idx_b += 1;
 125         }
 126         else {
 127             return false;
 128         }
 129     }
 130 }

 132 // given a string and an index, return the first uint >= idx
 133 // that is a non-ws-char or is outside of the legal range of
 134 // the string.
 135 fn scan_for_non_ws_or_end(a : &str, idx: uint) -> uint {
 136     let mut i = idx;
 137     let len = a.len();
 138     while ((i < len) && (is_whitespace(a.char_at(i)))) {
 139         i += 1;
 140     }
 141     i
 142 }

 144 // copied from lexer.
 145 pub fn is_whitespace(c: char) -> bool {
 146     return c == ' ' || c == '\t' || c == '\r' || c == '\n';
 147 }

 150 mod test {
 151     use super::*;
 152 
 153     #[test] fn eqmodws() {
 154         assert_eq!(matches_codepattern("",""),true);
 155         assert_eq!(matches_codepattern("","a"),false);
 156         assert_eq!(matches_codepattern("a",""),false);
 157         assert_eq!(matches_codepattern("a","a"),true);
 158         assert_eq!(matches_codepattern("a b","a   \n\t\r  b"),true);
 159         assert_eq!(matches_codepattern("a b ","a   \n\t\r  b"),true);
 160         assert_eq!(matches_codepattern("a b","a   \n\t\r  b "),false);
 161         assert_eq!(matches_codepattern("a   b","a b"),true);
 162         assert_eq!(matches_codepattern("ab","a b"),false);
 163         assert_eq!(matches_codepattern("a   b","ab"),true);
 164     }

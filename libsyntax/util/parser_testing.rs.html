  76  // does the given string match the pattern? whitespace in the first string
  77  // may be deleted or replaced with other whitespace to match the pattern.
  78  // this function is unicode-ignorant; fortunately, the careful design of
  79  // UTF-8 mitigates this ignorance.  In particular, this function only collapses
  80  // sequences of \n, \r, ' ', and \t, but it should otherwise tolerate unicode
  81  // chars. Unsurprisingly, it doesn't do NKF-normalization(?).
  82  pub fn matches_codepattern(a : &str, b : &str) -> bool {
  83      let mut idx_a = 0;
  84      let mut idx_b = 0;
  85      loop {
  86          if idx_a == a.len() && idx_b == b.len() {
  87              return true;
  88          }
  89          else if idx_a == a.len() {return false;}
  90          else if idx_b == b.len() {
  91              // maybe the stuff left in a is all ws?
  92              if is_whitespace(a.char_at(idx_a)) {
  93                  return scan_for_non_ws_or_end(a,idx_a) == a.len();
  94              } else {
  95                  return false;
  96              }
  97          }
  98          // ws in both given and pattern:
  99          else if is_whitespace(a.char_at(idx_a))
 100             && is_whitespace(b.char_at(idx_b)) {
 101              idx_a = scan_for_non_ws_or_end(a,idx_a);
 102              idx_b = scan_for_non_ws_or_end(b,idx_b);
 103          }
 104          // ws in given only:
 105          else if is_whitespace(a.char_at(idx_a)) {
 106              idx_a = scan_for_non_ws_or_end(a,idx_a);
 107          }
 108          // *don't* silently eat ws in expected only.
 109          else if a.char_at(idx_a) == b.char_at(idx_b) {
 110              idx_a += 1;
 111              idx_b += 1;
 112          }
 113          else {
 114              return false;
 115          }
 116      }
 117  }
 118  
 119  // given a string and an index, return the first uint >= idx
 120  // that is a non-ws-char or is outside of the legal range of
 121  // the string.
 122  fn scan_for_non_ws_or_end(a : &str, idx: uint) -> uint {
 123      let mut i = idx;
 124      let len = a.len();
 125      while (i < len) && (is_whitespace(a.char_at(i))) {
 126          i += 1;
 127      }
 128      i
 129  }

 131  // copied from lexer.
 132  pub fn is_whitespace(c: char) -> bool {
 133      return c == ' ' || c == '\t' || c == '\r' || c == '\n';
 134  }

 137  mod test {
 138      use super::*;
 139  
 140      #[test] fn eqmodws() {
 141          assert_eq!(matches_codepattern("",""),true);
 142          assert_eq!(matches_codepattern("","a"),false);
 143          assert_eq!(matches_codepattern("a",""),false);
 144          assert_eq!(matches_codepattern("a","a"),true);
 145          assert_eq!(matches_codepattern("a b","a   \n\t\r  b"),true);
 146          assert_eq!(matches_codepattern("a b ","a   \n\t\r  b"),true);
 147          assert_eq!(matches_codepattern("a b","a   \n\t\r  b "),false);
 148          assert_eq!(matches_codepattern("a   b","a b"),true);
 149          assert_eq!(matches_codepattern("ab","a b"),false);
 150          assert_eq!(matches_codepattern("a   b","ab"),true);
 151      }

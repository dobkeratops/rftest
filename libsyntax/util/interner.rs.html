  47      pub fn intern(&self, val: T) -> Name {
  48          let mut map = self.map.borrow_mut();
  49          match (*map).find(&val) {
  50              Some(&idx) => return idx,
  51              None => (),
  52          }
  53

  55          let new_idx = (*vect).len() as Name;

  59      }

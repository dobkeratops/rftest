  19  // Context-passing AST walker. Each overridden visit method has full control
  20  // over what happens with its node, it can do its own traversal of the node's
  21  // children (potentially passing in different contexts to each), call
  22  // visit::visit_* to apply the default traversal algorithm (again, it can
  23  // override the context), or prevent deeper traversal by doing nothing.
  24  //
  25  // Note: it is an important invariant that the default visitor walks the body
  26  // of a function in "execution order" (more concretely, reverse post-order
  27  // with respect to the CFG implied by the AST), meaning that if AST node A may
  28  // execute before AST node B, then A is visited first.  The borrow checker in
  29  // particular relies on this property.
  30  
  31  pub enum fn_kind<'self> {
  32      // fn foo() or extern "Abi" fn foo()
  33      fk_item_fn(ident, &'self Generics, purity, AbiSet),
  34  
  35      // fn foo(&self)
  36      fk_method(ident, &'self Generics, &'self method),
  37  
  38      // @fn(x, y) { ... }
  39      fk_anon(ast::Sigil),
  40  
  41      // |x, y| ...
  42      fk_fn_block,
  43  }
  44  
  45  pub fn name_of_fn(fk: &fn_kind) -> ident {
  46      match *fk {
  47        fk_item_fn(name, _, _, _) | fk_method(name, _, _) => {
  48            name
  49        }
  50        fk_anon(*) | fk_fn_block(*) => parse::token::special_idents::anon,
  51      }
  52  }

  54  pub fn generics_of_fn(fk: &fn_kind) -> Generics {
  55      match *fk {
  56          fk_item_fn(_, generics, _, _) |
  57          fk_method(_, generics, _) => {
  58              (*generics).clone()
  59          }
  60          fk_anon(*) | fk_fn_block(*) => {
  61              Generics {
  62                  lifetimes: opt_vec::Empty,
  63                  ty_params: opt_vec::Empty,
  64              }
  65          }
  66      }
  67  }

  95  pub fn visit_mod<E:Clone>(visitor: @Visitor<E>, module: &_mod, env: E) {
  96      for view_item in module.view_items.iter() {
  97          visitor.visit_view_item(view_item, env.clone())
  98      }
  99      for item in module.items.iter() {
 100          visitor.visit_item(*item, env.clone())
 101      }
 102  }

 123  pub fn visit_item<E:Clone>(visitor: @Visitor<E>, item: &item, env: E) {
 124      match item.node {
 125          item_static(ref typ, _, expr) => {
 126              visitor.visit_ty(typ, env.clone());
 127              visitor.visit_expr(expr, env);
 128          }
 129          item_fn(ref declaration, purity, abi, ref generics, ref body) => {
 130              visitor.visit_fn(&fk_item_fn(item.ident, generics, purity, abi),
 131                               declaration,
 132                               body,
 133                               item.span,
 134                               item.id,
 135                               env)
 136          }
 137          item_mod(ref module) => {
 138              visitor.visit_mod(module, item.span, item.id, env)
 139          }
 140          item_foreign_mod(ref foreign_module) => {
 141              for view_item in foreign_module.view_items.iter() {
 142                  visitor.visit_view_item(view_item, env.clone())
 143              }
 144              for foreign_item in foreign_module.items.iter() {
 145                  visitor.visit_foreign_item(*foreign_item, env.clone())
 146              }
 147          }
 148          item_ty(ref typ, ref type_parameters) => {
 149              visitor.visit_ty(typ, env.clone());
 150              visitor.visit_generics(type_parameters, env)
 151          }
 152          item_enum(ref enum_definition, ref type_parameters) => {
 153              visitor.visit_generics(type_parameters, env.clone());
 154              visit_enum_def(visitor, enum_definition, type_parameters, env)
 155          }
 156          item_impl(ref type_parameters,
 157                    ref trait_references,
 158                    ref typ,
 159                    ref methods) => {
 160              visitor.visit_generics(type_parameters, env.clone());
 161              for trait_reference in trait_references.iter() {
 162                  visit_trait_ref(visitor, trait_reference, env.clone())
 163              }
 164              visitor.visit_ty(typ, env.clone());
 165              for method in methods.iter() {
 166                  visit_method_helper(visitor, *method, env.clone())
 167              }
 168          }
 169          item_struct(struct_definition, ref generics) => {
 170              visitor.visit_generics(generics, env.clone());
 171              visitor.visit_struct_def(struct_definition,
 172                                       item.ident,
 173                                       generics,
 174                                       item.id,
 175                                       env)
 176          }
 177          item_trait(ref generics, ref trait_paths, ref methods) => {
 178              visitor.visit_generics(generics, env.clone());
 179              for trait_path in trait_paths.iter() {
 180                  visit_path(visitor, &trait_path.path, env.clone())
 181              }
 182              for method in methods.iter() {
 183                  visitor.visit_trait_method(method, env.clone())
 184              }
 185          }
 186          item_mac(ref macro) => visit_mac(visitor, macro, env),
 187      }
 188  }

 190  pub fn visit_enum_def<E:Clone>(visitor: @Visitor<E>,
 191                                 enum_definition: &ast::enum_def,
 192                                 generics: &Generics,
 193                                 env: E) {
 194      for variant in enum_definition.variants.iter() {
 195          match variant.node.kind {
 196              tuple_variant_kind(ref variant_arguments) => {
 197                  for variant_argument in variant_arguments.iter() {
 198                      visitor.visit_ty(&variant_argument.ty, env.clone())
 199                  }
 200              }
 201              struct_variant_kind(struct_definition) => {
 202                  visitor.visit_struct_def(struct_definition,
 203                                           variant.node.name,
 204                                           generics,
 205                                           variant.node.id,
 206                                           env.clone())
 207              }
 208          }
 209      }
 210  }

 216  pub fn visit_ty<E:Clone>(visitor: @Visitor<E>, typ: &Ty, env: E) {
 217      match typ.node {
 218          ty_box(ref mutable_type) | ty_uniq(ref mutable_type) |
 219          ty_vec(ref mutable_type) | ty_ptr(ref mutable_type) |
 220          ty_rptr(_, ref mutable_type) => {
 221              visitor.visit_ty(mutable_type.ty, env)
 222          }
 223          ty_tup(ref tuple_element_types) => {
 224              for tuple_element_type in tuple_element_types.iter() {
 225                  visitor.visit_ty(tuple_element_type, env.clone())
 226              }
 227          }
 228          ty_closure(ref function_declaration) => {
 229               for argument in function_declaration.decl.inputs.iter() {
 230                  visitor.visit_ty(&argument.ty, env.clone())
 231               }
 232               visitor.visit_ty(&function_declaration.decl.output, env.clone());
 233               for bounds in function_declaration.bounds.iter() {
 234                  visit_ty_param_bounds(visitor, bounds, env.clone())
 235               }
 236          }
 237          ty_bare_fn(ref function_declaration) => {
 238              for argument in function_declaration.decl.inputs.iter() {
 239                  visitor.visit_ty(&argument.ty, env.clone())
 240              }
 241              visitor.visit_ty(&function_declaration.decl.output, env.clone())
 242          }
 243          ty_path(ref path, ref bounds, _) => {
 244              visit_path(visitor, path, env.clone());
 245              for bounds in bounds.iter() {
 246                  visit_ty_param_bounds(visitor, bounds, env.clone())
 247              }
 248          }
 249          ty_fixed_length_vec(ref mutable_type, expression) => {
 250              visitor.visit_ty(mutable_type.ty, env.clone());
 251              visitor.visit_expr(expression, env)
 252          }
 253          ty_nil | ty_bot | ty_mac(_) | ty_infer => ()
 254      }
 255  }

 263  pub fn visit_pat<E:Clone>(visitor: @Visitor<E>, pattern: &pat, env: E) {
 264      match pattern.node {
 265          pat_enum(ref path, ref children) => {
 266              visit_path(visitor, path, env.clone());
 267              for children in children.iter() {
 268                  for child in children.iter() {
 269                      visitor.visit_pat(*child, env.clone())
 270                  }
 271              }
 272          }
 273          pat_struct(ref path, ref fields, _) => {
 274              visit_path(visitor, path, env.clone());
 275              for field in fields.iter() {
 276                  visitor.visit_pat(field.pat, env.clone())
 277              }
 278          }
 279          pat_tup(ref tuple_elements) => {
 280              for tuple_element in tuple_elements.iter() {
 281                  visitor.visit_pat(*tuple_element, env.clone())
 282              }
 283          }
 284          pat_box(subpattern) |
 285          pat_uniq(subpattern) |
 286          pat_region(subpattern) => {
 287              visitor.visit_pat(subpattern, env)
 288          }
 289          pat_ident(_, ref path, ref optional_subpattern) => {
 290              visit_path(visitor, path, env.clone());
 291              match *optional_subpattern {
 292                  None => {}
 293                  Some(subpattern) => visitor.visit_pat(subpattern, env),
 294              }
 295          }
 296          pat_lit(expression) => visitor.visit_expr(expression, env),
 297          pat_range(lower_bound, upper_bound) => {
 298              visitor.visit_expr(lower_bound, env.clone());
 299              visitor.visit_expr(upper_bound, env)
 300          }
 301          pat_wild => (),
 302          pat_vec(ref prepattern, ref slice_pattern, ref postpatterns) => {
 303              for prepattern in prepattern.iter() {
 304                  visitor.visit_pat(*prepattern, env.clone())
 305              }
 306              for slice_pattern in slice_pattern.iter() {
 307                  visitor.visit_pat(*slice_pattern, env.clone())
 308              }
 309              for postpattern in postpatterns.iter() {
 310                  visitor.visit_pat(*postpattern, env.clone())
 311              }
 312          }
 313      }
 314  }

 349  pub fn visit_fn_decl<E:Clone>(visitor: @Visitor<E>,
 350                                function_declaration: &fn_decl,
 351                                env: E) {
 352      for argument in function_declaration.inputs.iter() {
 353          visitor.visit_pat(argument.pat, env.clone());
 354          visitor.visit_ty(&argument.ty, env.clone())
 355      }
 356      visitor.visit_ty(&function_declaration.output, env)
 357  }

 359  // Note: there is no visit_method() method in the visitor, instead override
 360  // visit_fn() and check for fk_method().  I named this visit_method_helper()
 361  // because it is not a default impl of any method, though I doubt that really
 362  // clarifies anything. - Niko
 363  pub fn visit_method_helper<E:Clone>(visitor: @Visitor<E>,
 364                                      method: &method,
 365                                      env: E) {
 366      visitor.visit_fn(&fk_method(method.ident, &method.generics, method),
 367                       &method.decl,
 368                       &method.body,
 369                       method.span,
 370                       method.id,
 371                       env)
 372  }

 374  pub fn visit_fn<E:Clone>(visitor: @Visitor<E>,
 375                           function_kind: &fn_kind,
 376                           function_declaration: &fn_decl,
 377                           function_body: &Block,
 378                           _: span,
 379                           _: NodeId,
 380                           env: E) {
 381      visit_fn_decl(visitor, function_declaration, env.clone());

 383      visitor.visit_generics(&generics, env.clone());

 385  }

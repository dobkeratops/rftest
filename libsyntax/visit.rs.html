  18  // Context-passing AST walker. Each overridden visit method has full control
  19  // over what happens with its node, it can do its own traversal of the node's
  20  // children (potentially passing in different contexts to each), call
  21  // visit::visit_* to apply the default traversal algorithm (again, it can
  22  // override the context), or prevent deeper traversal by doing nothing.
  23  //
  24  // Note: it is an important invariant that the default visitor walks the body
  25  // of a function in "execution order" (more concretely, reverse post-order
  26  // with respect to the CFG implied by the AST), meaning that if AST node A may
  27  // execute before AST node B, then A is visited first.  The borrow checker in
  28  // particular relies on this property.
  29  
  30  pub enum FnKind<'a> {
  31      // fn foo() or extern "Abi" fn foo()
  32      FkItemFn(Ident, &'a Generics, Purity, Abi),
  33  
  34      // fn foo(&self)
  35      FkMethod(Ident, &'a Generics, &'a Method),
  36  
  37      // |x, y| ...
  38      // proc(x, y) ...
  39      FkFnBlock,
  40  }
  41  
  42  pub fn name_of_fn(fk: &FnKind) -> Ident {
  43      match *fk {
  44          FkItemFn(name, _, _, _) | FkMethod(name, _, _) => name,
  45          FkFnBlock(..) => parse::token::special_idents::invalid
  46      }
  47  }

  49  pub fn generics_of_fn(fk: &FnKind) -> Generics {
  50      match *fk {
  51          FkItemFn(_, generics, _, _) |
  52          FkMethod(_, generics, _) => {
  53              (*generics).clone()
  54          }
  55          FkFnBlock(..) => {
  56              Generics {
  57                  lifetimes: Vec::new(),
  58                  ty_params: OwnedSlice::empty(),
  59              }
  60          }
  61      }
  62  }

  82      fn visit_fn(&mut self, fk: &FnKind, fd: &FnDecl, b: &Block, s: Span, n: NodeId, e: E) {

  84      }
  85      fn visit_ty_method(&mut self, t: &TypeMethod, e: E) { walk_ty_method(self, t, e) }
  86      fn visit_trait_method(&mut self, t: &TraitMethod, e: E) { walk_trait_method(self, t, e) }
  87      fn visit_struct_def(&mut self, s: &StructDef, i: Ident, g: &Generics, n: NodeId, e: E) {
  88          walk_struct_def(self, s, i, g, n, e)
  89      }
  90      fn visit_struct_field(&mut self, s: &StructField, e: E) { walk_struct_field(self, s, e) }
  91      fn visit_variant(&mut self, v: &Variant, g: &Generics, e: E) { walk_variant(self, v, g, e) }
  92      fn visit_opt_lifetime_ref(&mut self,

 110          /*! Visits a declaration of a lifetime */
 111      }
 112      fn visit_explicit_self(&mut self, es: &ExplicitSelf, e: E) {
 113          walk_explicit_self(self, es, e)
 114      }
 115      fn visit_mac(&mut self, macro: &Mac, e: E) {

 138  pub fn walk_mod<E: Clone, V: Visitor<E>>(visitor: &mut V, module: &Mod, env: E) {
 139      for view_item in module.view_items.iter() {
 140          visitor.visit_view_item(view_item, env.clone())
 141      }
 142  
 143      for item in module.items.iter() {
 144          visitor.visit_item(*item, env.clone())
 145      }
 146  }

 148  pub fn walk_view_item<E: Clone, V: Visitor<E>>(visitor: &mut V, vi: &ViewItem, env: E) {
 149      match vi.node {
 150          ViewItemExternCrate(name, _, _) => {
 151              visitor.visit_ident(vi.span, name, env)
 152          }
 153          ViewItemUse(ref paths) => {
 154              for vp in paths.iter() {
 155                  match vp.node {
 156                      ViewPathSimple(ident, ref path, id) => {
 157                          visitor.visit_ident(vp.span, ident, env.clone());
 158                          visitor.visit_path(path, id, env.clone());
 159                      }
 160                      ViewPathGlob(ref path, id) => {
 161                          visitor.visit_path(path, id, env.clone());
 162                      }
 163                      ViewPathList(ref path, ref list, _) => {
 164                          for id in list.iter() {
 165                              visitor.visit_ident(id.span, id.node.name, env.clone())
 166                          }
 167                          walk_path(visitor, path, env.clone());
 168                      }
 169                  }
 170              }
 171          }
 172      }
 173  }

 178      match local.init {
 179          None => {}
 180          Some(initializer) => visitor.visit_expr(initializer, env),
 181      }
 182  }

 203  pub fn walk_item<E: Clone, V: Visitor<E>>(visitor: &mut V, item: &Item, env: E) {
 204      visitor.visit_ident(item.span, item.ident, env.clone());
 205      match item.node {
 206          ItemStatic(typ, _, expr) => {
 207              visitor.visit_ty(typ, env.clone());
 208              visitor.visit_expr(expr, env);
 209          }
 210          ItemFn(declaration, purity, abi, ref generics, body) => {
 211              visitor.visit_fn(&FkItemFn(item.ident, generics, purity, abi),
 212                               declaration,
 213                               body,
 214                               item.span,
 215                               item.id,
 216                               env)
 217          }
 218          ItemMod(ref module) => {
 219              visitor.visit_mod(module, item.span, item.id, env)
 220          }
 221          ItemForeignMod(ref foreign_module) => {
 222              for view_item in foreign_module.view_items.iter() {
 223                  visitor.visit_view_item(view_item, env.clone())
 224              }
 225              for foreign_item in foreign_module.items.iter() {
 226                  visitor.visit_foreign_item(*foreign_item, env.clone())
 227              }
 228          }
 229          ItemTy(typ, ref type_parameters) => {
 230              visitor.visit_ty(typ, env.clone());
 231              visitor.visit_generics(type_parameters, env)
 232          }
 233          ItemEnum(ref enum_definition, ref type_parameters) => {
 234              visitor.visit_generics(type_parameters, env.clone());
 235              walk_enum_def(visitor, enum_definition, type_parameters, env)
 236          }
 237          ItemImpl(ref type_parameters,
 238                   ref trait_reference,
 239                   typ,
 240                   ref methods) => {
 241              visitor.visit_generics(type_parameters, env.clone());
 242              match *trait_reference {
 243                  Some(ref trait_reference) => walk_trait_ref_helper(visitor,
 244                                                                     trait_reference, env.clone()),
 245                  None => ()
 246              }
 247              visitor.visit_ty(typ, env.clone());
 248              for method in methods.iter() {
 249                  walk_method_helper(visitor, *method, env.clone())
 250              }
 251          }
 252          ItemStruct(struct_definition, ref generics) => {
 253              visitor.visit_generics(generics, env.clone());
 254              visitor.visit_struct_def(struct_definition,
 255                                       item.ident,
 256                                       generics,
 257                                       item.id,
 258                                       env)
 259          }
 260          ItemTrait(ref generics, ref trait_paths, ref methods) => {
 261              visitor.visit_generics(generics, env.clone());
 262              for trait_path in trait_paths.iter() {
 263                  visitor.visit_path(&trait_path.path,
 264                                     trait_path.ref_id,
 265                                     env.clone())
 266              }
 267              for method in methods.iter() {
 268                  visitor.visit_trait_method(method, env.clone())
 269              }
 270          }
 271          ItemMac(ref macro) => visitor.visit_mac(macro, env),
 272      }
 273  }

 284  pub fn walk_variant<E: Clone, V: Visitor<E>>(visitor: &mut V,
 285                                               variant: &Variant,
 286                                               generics: &Generics,
 287                                               env: E) {
 288      visitor.visit_ident(variant.span, variant.node.name, env.clone());
 289  
 290      match variant.node.kind {
 291          TupleVariantKind(ref variant_arguments) => {
 292              for variant_argument in variant_arguments.iter() {
 293                  visitor.visit_ty(variant_argument.ty, env.clone())
 294              }
 295          }
 296          StructVariantKind(struct_definition) => {
 297              visitor.visit_struct_def(struct_definition,
 298                                       variant.node.name,
 299                                       generics,
 300                                       variant.node.id,
 301                                       env.clone())
 302          }
 303      }
 304      match variant.node.disr_expr {
 305          Some(expr) => visitor.visit_expr(expr, env),
 306          None => ()
 307      }
 308  }

 314  pub fn walk_ty<E: Clone, V: Visitor<E>>(visitor: &mut V, typ: &Ty, env: E) {
 315      match typ.node {
 316          TyUniq(ty) | TyVec(ty) | TyBox(ty) => {
 317              visitor.visit_ty(ty, env)
 318          }
 319          TyPtr(ref mutable_type) => {
 320              visitor.visit_ty(mutable_type.ty, env)
 321          }
 322          TyRptr(ref lifetime, ref mutable_type) => {
 323              visitor.visit_opt_lifetime_ref(typ.span, lifetime, env.clone());
 324              visitor.visit_ty(mutable_type.ty, env)
 325          }
 326          TyTup(ref tuple_element_types) => {
 327              for &tuple_element_type in tuple_element_types.iter() {
 328                  visitor.visit_ty(tuple_element_type, env.clone())
 329              }
 330          }
 331          TyClosure(ref function_declaration) => {
 332              for argument in function_declaration.decl.inputs.iter() {
 333                  visitor.visit_ty(argument.ty, env.clone())
 334              }
 335              visitor.visit_ty(function_declaration.decl.output, env.clone());
 336              for bounds in function_declaration.bounds.iter() {
 337                  walk_ty_param_bounds(visitor, bounds, env.clone())
 338              }
 339              visitor.visit_opt_lifetime_ref(
 340                  typ.span,
 341                  &function_declaration.region,
 342                  env.clone());
 343              walk_lifetime_decls(visitor, &function_declaration.lifetimes,
 344                                  env.clone());
 345          }
 346          TyBareFn(ref function_declaration) => {
 347              for argument in function_declaration.decl.inputs.iter() {
 348                  visitor.visit_ty(argument.ty, env.clone())
 349              }
 350              visitor.visit_ty(function_declaration.decl.output, env.clone());
 351              walk_lifetime_decls(visitor, &function_declaration.lifetimes,
 352                                  env.clone());
 353          }
 354          TyPath(ref path, ref bounds, id) => {
 355              visitor.visit_path(path, id, env.clone());
 356              for bounds in bounds.iter() {
 357                  walk_ty_param_bounds(visitor, bounds, env.clone())
 358              }
 359          }
 360          TyFixedLengthVec(ty, expression) => {
 361              visitor.visit_ty(ty, env.clone());
 362              visitor.visit_expr(expression, env)
 363          }
 364          TyTypeof(expression) => {
 365              visitor.visit_expr(expression, env)
 366          }
 367          TyNil | TyBot | TyInfer => {}
 368      }
 369  }

 379  pub fn walk_path<E: Clone, V: Visitor<E>>(visitor: &mut V, path: &Path, env: E) {
 380      for segment in path.segments.iter() {
 381          visitor.visit_ident(path.span, segment.identifier, env.clone());
 382  
 383          for &typ in segment.types.iter() {
 384              visitor.visit_ty(typ, env.clone());
 385          }
 386          for lifetime in segment.lifetimes.iter() {
 387              visitor.visit_lifetime_ref(lifetime, env.clone());
 388          }
 389      }
 390  }

 392  pub fn walk_pat<E: Clone, V: Visitor<E>>(visitor: &mut V, pattern: &Pat, env: E) {
 393      match pattern.node {
 394          PatEnum(ref path, ref children) => {
 395              visitor.visit_path(path, pattern.id, env.clone());
 396              for children in children.iter() {
 397                  for child in children.iter() {
 398                      visitor.visit_pat(*child, env.clone())

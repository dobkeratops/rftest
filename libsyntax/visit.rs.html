  19  // Context-passing AST walker. Each overridden visit method has full control
  20  // over what happens with its node, it can do its own traversal of the node's
  21  // children (potentially passing in different contexts to each), call
  22  // visit::visit_* to apply the default traversal algorithm (again, it can
  23  // override the context), or prevent deeper traversal by doing nothing.
  24  //
  25  // Note: it is an important invariant that the default visitor walks the body
  26  // of a function in "execution order" (more concretely, reverse post-order
  27  // with respect to the CFG implied by the AST), meaning that if AST node A may
  28  // execute before AST node B, then A is visited first.  The borrow checker in
  29  // particular relies on this property.
  30  
  31  pub enum fn_kind<'self> {
  32      // fn foo() or extern "Abi" fn foo()
  33      fk_item_fn(Ident, &'self Generics, purity, AbiSet),
  34  
  35      // fn foo(&self)
  36      fk_method(Ident, &'self Generics, &'self method),
  37  
  38      // @fn(x, y) { ... }
  39      fk_anon(ast::Sigil),
  40  
  41      // |x, y| ...
  42      fk_fn_block,
  43  }
  44  
  45  pub fn name_of_fn(fk: &fn_kind) -> Ident {
  46      match *fk {
  47        fk_item_fn(name, _, _, _) | fk_method(name, _, _) => {
  48            name
  49        }
  50        fk_anon(*) | fk_fn_block(*) => parse::token::special_idents::anon,
  51      }
  52  }

  54  pub fn generics_of_fn(fk: &fn_kind) -> Generics {
  55      match *fk {
  56          fk_item_fn(_, generics, _, _) |
  57          fk_method(_, generics, _) => {
  58              (*generics).clone()
  59          }
  60          fk_anon(*) | fk_fn_block(*) => {
  61              Generics {
  62                  lifetimes: opt_vec::Empty,
  63                  ty_params: opt_vec::Empty,
  64              }
  65          }
  66      }

  69  pub trait Visitor<E:Clone> {
  70      fn visit_mod(&mut self, m:&_mod, _s:Span, _n:NodeId, e:E) { walk_mod(self, m, e) }
  71      fn visit_view_item(&mut self, i:&view_item, e:E) { walk_view_item(self, i, e) }
  72      fn visit_foreign_item(&mut self, i:@foreign_item, e:E) { walk_foreign_item(self, i, e) }
  73      fn visit_item(&mut self, i:@item, e:E) { walk_item(self, i, e) }
  74      fn visit_local(&mut self, l:@Local, e:E) { walk_local(self, l, e) }
  75      fn visit_block(&mut self, b:&Block, e:E) { walk_block(self, b, e) }
  76      fn visit_stmt(&mut self, s:@Stmt, e:E) { walk_stmt(self, s, e) }
  77      fn visit_arm(&mut self, a:&Arm, e:E) { walk_arm(self, a, e) }
  78      fn visit_pat(&mut self, p:@Pat, e:E) { walk_pat(self, p, e) }
  79      fn visit_decl(&mut self, d:@Decl, e:E) { walk_decl(self, d, e) }
  80      fn visit_expr(&mut self, ex:@Expr, e:E) { walk_expr(self, ex, e) }
  81      fn visit_expr_post(&mut self, _ex:@Expr, _e:E) { }
  82      fn visit_ty(&mut self, _t:&Ty, _e:E) { }
  83      fn visit_generics(&mut self, g:&Generics, e:E) { walk_generics(self, g, e) }
  84      fn visit_fn(&mut self, fk:&fn_kind, fd:&fn_decl, b:&Block, s:Span, n:NodeId, e:E) {
  85          walk_fn(self, fk, fd, b, s, n , e)
  86      }
  87      fn visit_ty_method(&mut self, t:&TypeMethod, e:E) { walk_ty_method(self, t, e) }
  88      fn visit_trait_method(&mut self, t:&trait_method, e:E) { walk_trait_method(self, t, e) }
  89      fn visit_struct_def(&mut self, s:@struct_def, i:Ident, g:&Generics, n:NodeId, e:E) {
  90          walk_struct_def(self, s, i, g, n, e)
  91      }
  92      fn visit_struct_field(&mut self, s:@struct_field, e:E) { walk_struct_field(self, s, e) }
  93  }

 159  pub fn walk_mod<E:Clone, V:Visitor<E>>(visitor: &mut V, module: &_mod, env: E) {
 160      for view_item in module.view_items.iter() {
 161          visitor.visit_view_item(view_item, env.clone())
 162      }
 163      for item in module.items.iter() {
 164          visitor.visit_item(*item, env.clone())
 165      }
 166  }

 187  pub fn walk_item<E:Clone, V:Visitor<E>>(visitor: &mut V, item: &item, env: E) {
 188      match item.node {
 189          item_static(ref typ, _, expr) => {
 190              visitor.visit_ty(typ, env.clone());
 191              visitor.visit_expr(expr, env);
 192          }
 193          item_fn(ref declaration, purity, abi, ref generics, ref body) => {
 194              visitor.visit_fn(&fk_item_fn(item.ident, generics, purity, abi),
 195                               declaration,
 196                               body,
 197                               item.span,
 198                               item.id,
 199                               env)
 200          }
 201          item_mod(ref module) => {
 202              visitor.visit_mod(module, item.span, item.id, env)
 203          }
 204          item_foreign_mod(ref foreign_module) => {
 205              for view_item in foreign_module.view_items.iter() {
 206                  visitor.visit_view_item(view_item, env.clone())
 207              }
 208              for foreign_item in foreign_module.items.iter() {
 209                  visitor.visit_foreign_item(*foreign_item, env.clone())
 210              }
 211          }
 212          item_ty(ref typ, ref type_parameters) => {
 213              visitor.visit_ty(typ, env.clone());
 214              visitor.visit_generics(type_parameters, env)
 215          }
 216          item_enum(ref enum_definition, ref type_parameters) => {
 217              visitor.visit_generics(type_parameters, env.clone());
 218              walk_enum_def(visitor, enum_definition, type_parameters, env)
 219          }
 220          item_impl(ref type_parameters,
 221                    ref trait_references,
 222                    ref typ,
 223                    ref methods) => {
 224              visitor.visit_generics(type_parameters, env.clone());
 225              for trait_reference in trait_references.iter() {
 226                  walk_trait_ref(visitor, trait_reference, env.clone())
 227              }
 228              visitor.visit_ty(typ, env.clone());
 229              for method in methods.iter() {
 230                  walk_method_helper(visitor, *method, env.clone())
 231              }
 232          }
 233          item_struct(struct_definition, ref generics) => {
 234              visitor.visit_generics(generics, env.clone());
 235              visitor.visit_struct_def(struct_definition,
 236                                       item.ident,
 237                                       generics,
 238                                       item.id,
 239                                       env)
 240          }
 241          item_trait(ref generics, ref trait_paths, ref methods) => {
 242              visitor.visit_generics(generics, env.clone());
 243              for trait_path in trait_paths.iter() {
 244                  walk_path(visitor, &trait_path.path, env.clone())
 245              }
 246              for method in methods.iter() {
 247                  visitor.visit_trait_method(method, env.clone())
 248              }
 249          }
 250          item_mac(ref macro) => walk_mac(visitor, macro, env),
 251      }
 252  }

 254  pub fn walk_enum_def<E:Clone, V:Visitor<E>>(visitor: &mut V,
 255                                 enum_definition: &ast::enum_def,
 256                                 generics: &Generics,
 257                                 env: E) {
 258      for variant in enum_definition.variants.iter() {
 259          match variant.node.kind {
 260              tuple_variant_kind(ref variant_arguments) => {
 261                  for variant_argument in variant_arguments.iter() {
 262                      visitor.visit_ty(&variant_argument.ty, env.clone())
 263                  }
 264              }
 265              struct_variant_kind(struct_definition) => {
 266                  visitor.visit_struct_def(struct_definition,
 267                                           variant.node.name,
 268                                           generics,
 269                                           variant.node.id,
 270                                           env.clone())
 271              }
 272          }
 273      }
 274  }

 280  pub fn walk_ty<E:Clone, V:Visitor<E>>(visitor: &mut V, typ: &Ty, env: E) {
 281      match typ.node {
 282          ty_box(ref mutable_type) | ty_uniq(ref mutable_type) |
 283          ty_vec(ref mutable_type) | ty_ptr(ref mutable_type) |
 284          ty_rptr(_, ref mutable_type) => {
 285              visitor.visit_ty(mutable_type.ty, env)
 286          }
 287          ty_tup(ref tuple_element_types) => {
 288              for tuple_element_type in tuple_element_types.iter() {
 289                  visitor.visit_ty(tuple_element_type, env.clone())
 290              }
 291          }
 292          ty_closure(ref function_declaration) => {
 293               for argument in function_declaration.decl.inputs.iter() {
 294                  visitor.visit_ty(&argument.ty, env.clone())
 295               }
 296               visitor.visit_ty(&function_declaration.decl.output, env.clone());
 297               for bounds in function_declaration.bounds.iter() {
 298                  walk_ty_param_bounds(visitor, bounds, env.clone())
 299               }
 300          }
 301          ty_bare_fn(ref function_declaration) => {
 302              for argument in function_declaration.decl.inputs.iter() {
 303                  visitor.visit_ty(&argument.ty, env.clone())
 304              }
 305              visitor.visit_ty(&function_declaration.decl.output, env.clone())
 306          }
 307          ty_path(ref path, ref bounds, _) => {
 308              walk_path(visitor, path, env.clone());
 309              for bounds in bounds.iter() {
 310                  walk_ty_param_bounds(visitor, bounds, env.clone())
 311              }
 312          }
 313          ty_fixed_length_vec(ref mutable_type, expression) => {
 314              visitor.visit_ty(mutable_type.ty, env.clone());
 315              visitor.visit_expr(expression, env)
 316          }
 317          ty_typeof(expression) => {
 318              visitor.visit_expr(expression, env)
 319          }
 320          ty_nil | ty_bot | ty_mac(_) | ty_infer => ()
 321      }
 322  }

 332  pub fn walk_pat<E:Clone, V:Visitor<E>>(visitor: &mut V, pattern: &Pat, env: E) {
 333      match pattern.node {
 334          PatEnum(ref path, ref children) => {
 335              walk_path(visitor, path, env.clone());
 336              for children in children.iter() {
 337                  for child in children.iter() {
 338                      visitor.visit_pat(*child, env.clone())
 339                  }
 340              }
 341          }
 342          PatStruct(ref path, ref fields, _) => {
 343              walk_path(visitor, path, env.clone());
 344              for field in fields.iter() {
 345                  visitor.visit_pat(field.pat, env.clone())
 346              }
 347          }
 348          PatTup(ref tuple_elements) => {
 349              for tuple_element in tuple_elements.iter() {
 350                  visitor.visit_pat(*tuple_element, env.clone())
 351              }
 352          }
 353          PatBox(subpattern) |
 354          PatUniq(subpattern) |
 355          PatRegion(subpattern) => {
 356              visitor.visit_pat(subpattern, env)
 357          }
 358          PatIdent(_, ref path, ref optional_subpattern) => {
 359              walk_path(visitor, path, env.clone());
 360              match *optional_subpattern {
 361                  None => {}
 362                  Some(subpattern) => visitor.visit_pat(subpattern, env),
 363              }
 364          }
 365          PatLit(expression) => visitor.visit_expr(expression, env),
 366          PatRange(lower_bound, upper_bound) => {
 367              visitor.visit_expr(lower_bound, env.clone());
 368              visitor.visit_expr(upper_bound, env)
 369          }
 370          PatWild => (),
 371          PatVec(ref prepattern, ref slice_pattern, ref postpatterns) => {
 372              for prepattern in prepattern.iter() {
 373                  visitor.visit_pat(*prepattern, env.clone())
 374              }
 375              for slice_pattern in slice_pattern.iter() {
 376                  visitor.visit_pat(*slice_pattern, env.clone())
 377              }
 378              for postpattern in postpatterns.iter() {
 379                  visitor.visit_pat(*postpattern, env.clone())
 380              }
 381          }
 382      }
 383  }

  420     fn visit_view_item(@mut self, view_item: &view_item, env: ()) {
  421         match view_item.node {
  422             view_item_extern_mod(_, _, node_id) => {
  423                 (self.visit_callback)(node_id)
  424             }
  425             view_item_use(ref view_paths) => {
  426                 for view_path in view_paths.iter() {
  427                     match view_path.node {
  428                         view_path_simple(_, _, node_id) |
  429                         view_path_glob(_, node_id) => {
  430                             (self.visit_callback)(node_id)
  431                         }
  432                         view_path_list(_, ref paths, node_id) => {
  433                             (self.visit_callback)(node_id);
  434                             for path in paths.iter() {
  435                                 (self.visit_callback)(path.node.id)
  436                             }
  437                         }
  438                     }
  439                 }
  440             }
  441         }
  442         visit::visit_view_item(self as @Visitor<()>, view_item, env)
  443     }

  456             }
  457         }

  461             item_enum(ref enum_definition, _) => {
  462                 for variant in enum_definition.variants.iter() {
  463                     (self.visit_callback)(variant.node.id)
  464                 }
  465             }
  466             _ => {}
  467         }
  468 
  469         visit::visit_item(self as @Visitor<()>, item, env);
  470 
  471         self.visited_outermost = false

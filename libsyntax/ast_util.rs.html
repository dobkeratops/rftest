 232  /// Generate a "pretty" name for an `impl` from its type and trait.
 233  /// This is designed so that symbols of `impl`'d methods give some
 234  /// hint of where they came from, (previously they would all just be
 235  /// listed as `__extensions__::method_name::hash`, with no indication
 236  /// of the type).
 237  pub fn impl_pretty_name(trait_ref: &Option<TraitRef>, ty: &Ty) -> Ident {
 238      let mut pretty = pprust::ty_to_str(ty);
 239      match *trait_ref {
 240          Some(ref trait_ref) => {
 241              pretty.push_char('.');
 242              pretty.push_str(pprust::path_to_str(&trait_ref.path));
 243          }
 244          None => {}
 245      }
 246      token::gensym_ident(pretty)
 247  }

 270                  explicit_self: m.explicit_self,

 278  pub fn split_trait_methods(trait_methods: &[TraitMethod])
 279      -> (Vec<TypeMethod> , Vec<@Method> ) {
 280      let mut reqd = Vec::new();
 281      let mut provd = Vec::new();
 282      for trt_method in trait_methods.iter() {
 283          match *trt_method {
 284              Required(ref tm) => reqd.push((*tm).clone()),
 285              Provided(m) => provd.push(m)
 286          }
 287      };
 288      (reqd, provd)
 289  }

 354  pub struct IdVisitor<'a, O> {

 359  
 360  impl<'a, O: IdVisitingOperation> IdVisitor<'a, O> {
 361      fn visit_generics_helper(&self, generics: &Generics) {
 362          for type_parameter in generics.ty_params.iter() {
 363              self.operation.visit_id(type_parameter.id)
 364          }
 365          for lifetime in generics.lifetimes.iter() {
 366              self.operation.visit_id(lifetime.id)
 367          }
 368      }
 369  }
 370  
 371  impl<'a, O: IdVisitingOperation> Visitor<()> for IdVisitor<'a, O> {
 372      fn visit_mod(&mut self,
 373                   module: &Mod,
 374                   _: Span,
 375                   node_id: NodeId,
 376                   env: ()) {
 377          self.operation.visit_id(node_id);
 378          visit::walk_mod(self, module, env)
 379      }
 380  
 381      fn visit_view_item(&mut self, view_item: &ViewItem, env: ()) {
 382          match view_item.node {
 383              ViewItemExternCrate(_, _, node_id) => {
 384                  self.operation.visit_id(node_id)
 385              }
 386              ViewItemUse(ref view_paths) => {
 387                  for view_path in view_paths.iter() {
 388                      match view_path.node {
 389                          ViewPathSimple(_, _, node_id) |
 390                          ViewPathGlob(_, node_id) => {
 391                              self.operation.visit_id(node_id)
 392                          }
 393                          ViewPathList(_, ref paths, node_id) => {
 394                              self.operation.visit_id(node_id);
 395                              for path in paths.iter() {
 396                                  self.operation.visit_id(path.node.id)
 397                              }
 398                          }
 399                      }
 400                  }
 401              }
 402          }
 403          visit::walk_view_item(self, view_item, env)
 404      }
 405  
 406      fn visit_foreign_item(&mut self, foreign_item: &ForeignItem, env: ()) {

 408          visit::walk_foreign_item(self, foreign_item, env)
 409      }

 411      fn visit_item(&mut self, item: &Item, env: ()) {
 412          if !self.pass_through_items {
 413              if self.visited_outermost {
 414                  return
 415              } else {
 416                  self.visited_outermost = true
 417              }
 418          }
 419  
 420          self.operation.visit_id(item.id);
 421          match item.node {
 422              ItemEnum(ref enum_definition, _) => {
 423                  for variant in enum_definition.variants.iter() {
 424                      self.operation.visit_id(variant.node.id)
 425                  }
 426              }
 427              _ => {}
 428          }
 429  
 430          visit::walk_item(self, item, env);
 431  
 432          self.visited_outermost = false
 433      }

 474  
 475      fn visit_fn(&mut self,
 476                  function_kind: &visit::FnKind,
 477                  function_declaration: &FnDecl,
 478                  block: &Block,
 479                  span: Span,
 480                  node_id: NodeId,
 481                  env: ()) {
 482          if !self.pass_through_items {
 483              match *function_kind {
 484                  visit::FkMethod(..) if self.visited_outermost => return,
 485                  visit::FkMethod(..) => self.visited_outermost = true,
 486                  _ => {}
 487              }
 488          }
 489  
 490          self.operation.visit_id(node_id);
 491  
 492          match *function_kind {
 493              visit::FkItemFn(_, generics, _, _) |
 494              visit::FkMethod(_, generics, _) => {
 495                  self.visit_generics_helper(generics)
 496              }
 497              visit::FkFnBlock => {}

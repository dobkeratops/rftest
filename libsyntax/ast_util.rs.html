  393  struct IdVisitor {
  394      visit_callback: @fn(NodeId),
  395      pass_through_items: bool,
  396      visited_outermost: bool,
  397  }
  398  
  399  impl IdVisitor {
  400      fn visit_generics_helper(@mut self, generics: &Generics) {
  401          for type_parameter in generics.ty_params.iter() {
  402              (self.visit_callback)(type_parameter.id)
  403          }
  404          for lifetime in generics.lifetimes.iter() {
  405              (self.visit_callback)(lifetime.id)
  406          }
  407      }
  408  }
  409  
  410  impl Visitor<()> for IdVisitor {
  411      fn visit_mod(@mut self,
  412                   module: &_mod,
  413                   _span: span,
  414                   node_id: NodeId,
  415                   env: ()) {
  416          (self.visit_callback)(node_id);
  417          visit::visit_mod(self as @Visitor<()>, module, env)
  418      }
  419  
  420      fn visit_view_item(@mut self, view_item: &view_item, env: ()) {
  421          match view_item.node {
  422              view_item_extern_mod(_, _, _, node_id) => {
  423                  (self.visit_callback)(node_id)
  424              }
  425              view_item_use(ref view_paths) => {
  426                  for view_path in view_paths.iter() {
  427                      match view_path.node {
  428                          view_path_simple(_, _, node_id) |
  429                          view_path_glob(_, node_id) => {
  430                              (self.visit_callback)(node_id)
  431                          }
  432                          view_path_list(_, ref paths, node_id) => {
  433                              (self.visit_callback)(node_id);
  434                              for path in paths.iter() {
  435                                  (self.visit_callback)(path.node.id)
  436                              }
  437                          }
  438                      }
  439                  }
  440              }
  441          }
  442          visit::visit_view_item(self as @Visitor<()>, view_item, env)
  443      }
  444  
  445      fn visit_foreign_item(@mut self, foreign_item: @foreign_item, env: ()) {
  446          (self.visit_callback)(foreign_item.id);
  447          visit::visit_foreign_item(self as @Visitor<()>, foreign_item, env)
  448      }
  449  
  450      fn visit_item(@mut self, item: @item, env: ()) {
  451          if !self.pass_through_items {
  452              if self.visited_outermost {
  453                  return
  454              } else {
  455                  self.visited_outermost = true
  456              }
  457          }
  458  
  459          (self.visit_callback)(item.id);
  460          match item.node {
  461              item_enum(ref enum_definition, _) => {
  462                  for variant in enum_definition.variants.iter() {
  463                      (self.visit_callback)(variant.node.id)
  464                  }
  465              }
  466              _ => {}

  646  pub fn walk_pat(pat: @pat, it: &fn(@pat) -> bool) -> bool {
  647      if !it(pat) {
  648          return false;
  649      }

  652          pat_ident(_, _, Some(p)) => walk_pat(p, it),
  653          pat_struct(_, ref fields, _) => {
  654              fields.iter().advance(|f| walk_pat(f.pat, |p| it(p)))
  655          }

  658          }

  660              walk_pat(s, it)
  661          }
  662          pat_vec(ref before, ref slice, ref after) => {
  663              before.iter().advance(|&p| walk_pat(p, |p| it(p))) &&
  664                  slice.iter().advance(|&p| walk_pat(p, |p| it(p))) &&
  665                  after.iter().advance(|&p| walk_pat(p, |p| it(p)))
  666          }
  667          pat_wild | pat_lit(_) | pat_range(_, _) | pat_ident(_, _, _) |
  668          pat_enum(_, _) => {
  669              true
  670          }
  671      }

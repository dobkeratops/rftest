   76  //used in noop_fold_item and noop_fold_crate and noop_fold_crate_directive
   77  fn fold_meta_item_(mi: @MetaItem, fld: @ast_fold) -> @MetaItem {
   78      @spanned {
   79          node:
   80              match mi.node {
   81                  MetaWord(id) => MetaWord(id),
   82                  MetaList(id, ref mis) => {
   83                      let fold_meta_item = |x| fold_meta_item_(x, fld);
   84                      MetaList(
   85                          id,
   86                          mis.map(|e| fold_meta_item(*e))
   87                      )
   88                  }
   89                  MetaNameValue(id, s) => MetaNameValue(id, s)
   90              },
   91          span: fld.new_span(mi.span) }
   92  }

  128  fn fold_tts(tts : &[token_tree], fld: @ast_fold) -> ~[token_tree] {
  129      do tts.map |tt| {
  130          match *tt {
  131              tt_tok(span, ref tok) =>
  132              tt_tok(span,maybe_fold_ident(tok,fld)),
  133              tt_delim(ref tts) =>
  134              tt_delim(@mut fold_tts(**tts, fld)),
  135              tt_seq(span, ref pattern, ref sep, is_optional) =>
  136              tt_seq(span,
  137                     @mut fold_tts(**pattern, fld),
  138                     sep.map(|tok|maybe_fold_ident(tok,fld)),
  139                     is_optional),
  140              tt_nonterminal(sp,ref ident) =>
  141              tt_nonterminal(sp,fld.fold_ident(*ident))
  142          }
  143      }
  144  }

  274  pub fn noop_fold_item_underscore(i: &item_, fld: @ast_fold) -> item_ {
  275      match *i {
  276          item_static(ref t, m, e) => item_static(fld.fold_ty(t), m, fld.fold_expr(e)),
  277          item_fn(ref decl, purity, abi, ref generics, ref body) => {
  278              item_fn(
  279                  fold_fn_decl(decl, fld),
  280                  purity,
  281                  abi,
  282                  fold_generics(generics, fld),
  283                  fld.fold_block(body)
  284              )
  285          }
  286          item_mod(ref m) => item_mod(fld.fold_mod(m)),
  287          item_foreign_mod(ref nm) => {
  288              item_foreign_mod(fld.fold_foreign_mod(nm))
  289          }
  290          item_ty(ref t, ref generics) => {
  291              item_ty(fld.fold_ty(t), fold_generics(generics, fld))
  292          }
  293          item_enum(ref enum_definition, ref generics) => {
  294              item_enum(
  295                  ast::enum_def {
  296                      variants: do enum_definition.variants.map |x| {
  297                          fld.fold_variant(x)
  298                      },
  299                  },
  300                  fold_generics(generics, fld))
  301          }
  302          item_struct(ref struct_def, ref generics) => {
  303              let struct_def = fold_struct_def(*struct_def, fld);
  304              item_struct(struct_def, /* FIXME (#2543) */ (*generics).clone())
  305          }
  306          item_impl(ref generics, ref ifce, ref ty, ref methods) => {
  307              item_impl(
  308                  fold_generics(generics, fld),
  309                  ifce.map(|p| fold_trait_ref(p, fld)),
  310                  fld.fold_ty(ty),
  311                  methods.map(|x| fld.fold_method(*x))
  312              )
  313          }
  314          item_trait(ref generics, ref traits, ref methods) => {
  315              let methods = do methods.map |method| {
  316                  match *method {
  317                      required(*) => (*method).clone(),
  318                      provided(method) => provided(fld.fold_method(method))
  319                  }
  320              };
  321              item_trait(
  322                  fold_generics(generics, fld),
  323                  traits.map(|p| fold_trait_ref(p, fld)),
  324                  methods
  325              )
  326          }
  327          item_mac(ref m) => {
  328              // FIXME #2888: we might actually want to do something here.
  329              // ... okay, we're doing something. It would probably be nicer
  330              // to add something to the ast_fold trait, but I'll defer
  331              // that work.
  332              item_mac(fold_mac_(m,fld))
  333          }
  334      }
  335  }

  382  pub fn noop_fold_block(b: &Block, fld: @ast_fold) -> Block {
  383      let view_items = b.view_items.map(|x| fld.fold_view_item(x));
  384      let mut stmts = ~[];
  385      for stmt in b.stmts.iter() {
  386          match fld.fold_stmt(*stmt) {
  387              None => {}
  388              Some(stmt) => stmts.push(stmt)
  389          }
  390      }
  391      ast::Block {
  392          view_items: view_items,
  393          stmts: stmts,

  30          view_paths.iter().map(|view_path| {
  31              let inner_view_path = match view_path.node {
  32                  ViewPathSimple(ref ident, ref path, node_id) => {

  37                  ViewPathGlob(ref path, node_id) => {

  41                      ViewPathList(self.fold_path(path),
  42                                   path_list_idents.map(|path_list_ident| {
  43                                      let id = self.new_id(path_list_ident.node
  44                                                                          .id);

  47                                              name: path_list_ident.node

  63          }).collect()

 146      fn fold_ty(&mut self, t: P<Ty>) -> P<Ty> {
 147          let node = match t.node {
 148              TyNil | TyBot | TyInfer => t.node.clone(),
 149              TyBox(ty) => TyBox(self.fold_ty(ty)),
 150              TyUniq(ty) => TyUniq(self.fold_ty(ty)),
 151              TyVec(ty) => TyVec(self.fold_ty(ty)),
 152              TyPtr(ref mt) => TyPtr(fold_mt(mt, self)),
 153              TyRptr(ref region, ref mt) => {
 154                  TyRptr(fold_opt_lifetime(region, self), fold_mt(mt, self))
 155              }
 156              TyClosure(ref f) => {
 157                  TyClosure(@ClosureTy {
 158                      sigil: f.sigil,
 159                      purity: f.purity,
 160                      region: fold_opt_lifetime(&f.region, self),
 161                      onceness: f.onceness,
 162                      bounds: fold_opt_bounds(&f.bounds, self),
 163                      decl: self.fold_fn_decl(f.decl),
 164                      lifetimes: f.lifetimes.map(|l| fold_lifetime(l, self)),
 165                  })
 166              }
 167              TyBareFn(ref f) => {
 168                  TyBareFn(@BareFnTy {
 169                      lifetimes: f.lifetimes.map(|l| fold_lifetime(l, self)),
 170                      purity: f.purity,
 171                      abis: f.abis,
 172                      decl: self.fold_fn_decl(f.decl)
 173                  })
 174              }
 175              TyTup(ref tys) => TyTup(tys.map(|&ty| self.fold_ty(ty))),
 176              TyPath(ref path, ref bounds, id) => {
 177                  TyPath(self.fold_path(path),
 178                         fold_opt_bounds(bounds, self),
 179                         self.new_id(id))
 180              }
 181              TyFixedLengthVec(ty, e) => {
 182                  TyFixedLengthVec(self.fold_ty(ty), self.fold_expr(e))
 183              }
 184              TyTypeof(expr) => TyTypeof(self.fold_expr(expr)),
 185          };
 186          P(Ty {
 187              id: self.new_id(t.id),
 188              span: self.new_span(t.span),
 189              node: node,
 190          })
 191      }

 211      fn fold_variant(&mut self, v: &Variant) -> P<Variant> {
 212          let kind;
 213          match v.node.kind {
 214              TupleVariantKind(ref variant_args) => {
 215                  kind = TupleVariantKind(variant_args.map(|x|
 216                      fold_variant_arg_(x, self)))
 217              }
 218              StructVariantKind(ref struct_def) => {
 219                  kind = StructVariantKind(@ast::StructDef {
 220                      fields: struct_def.fields.iter()
 221                          .map(|f| self.fold_struct_field(f)).collect(),
 222                      ctor_id: struct_def.ctor_id.map(|c| self.new_id(c))
 223                  })
 224              }
 225          }
 226  
 227          let attrs = v.node.attrs.map(|x| fold_attribute_(*x, self));
 228  
 229          let de = match v.node.disr_expr {
 230            Some(e) => Some(self.fold_expr(e)),
 231            None => None
 232          };
 233          let node = ast::Variant_ {
 234              name: v.node.name,
 235              attrs: attrs,
 236              kind: kind,
 237              id: self.new_id(v.node.id),
 238              disr_expr: de,
 239              vis: v.node.vis,
 240          };
 241          P(Spanned {
 242              node: node,
 243              span: self.new_span(v.span),
 244          })
 245      }

 354  // build a new vector of tts by appling the Folder's fold_ident to
 355  // all of the identifiers in the token trees.
 356  //
 357  // This is part of hygiene magic. As far as hygiene is concerned, there
 358  // are three types of let pattern bindings or loop labels:
 359  //      - those defined and used in non-macro part of the program
 360  //      - those used as part of macro invocation arguments
 361  //      - those defined and used inside macro definitions
 362  // Lexically, type 1 and 2 are in one group and type 3 the other. If they
 363  // clash, in order for let and loop label to work hygienically, one group
 364  // or the other needs to be renamed. The problem is that type 2 and 3 are
 365  // parsed together (inside the macro expand function). After being parsed and
 366  // AST being constructed, they can no longer be distinguished from each other.
 367  //
 368  // For that reason, type 2 let bindings and loop labels are actually renamed
 369  // in the form of tokens instead of AST nodes, here. There are wasted effort
 370  // since many token::IDENT are not necessary part of let bindings and most
 371  // token::LIFETIME are certainly not loop labels. But we can't tell in their
 372  // token form. So this is less ideal and hacky but it works.
 373  pub fn fold_tts<T: Folder>(tts: &[TokenTree], fld: &mut T) -> Vec<TokenTree> {
 374      tts.iter().map(|tt| {
 375          match *tt {
 376              TTTok(span, ref tok) =>
 377              TTTok(span,maybe_fold_ident(tok,fld)),
 378              TTDelim(tts) => TTDelim(@fold_tts(tts.as_slice(), fld)),
 379              TTSeq(span, pattern, ref sep, is_optional) =>
 380              TTSeq(span,
 381                    @fold_tts(pattern.as_slice(), fld),
 382                    sep.as_ref().map(|tok|maybe_fold_ident(tok,fld)),
 383                    is_optional),
 384              TTNonterminal(sp,ref ident) =>
 385              TTNonterminal(sp,fld.fold_ident(*ident))
 386          }
 387      }).collect()
 388  }

 391  fn maybe_fold_ident<T: Folder>(t: &token::Token, fld: &mut T) -> token::Token {

 397          _ => (*t).clone()
 398      }
 399  }

 545  pub fn noop_fold_item_underscore<T: Folder>(i: &Item_, folder: &mut T) -> Item_ {
 546      match *i {
 547          ItemStatic(t, m, e) => {
 548              ItemStatic(folder.fold_ty(t), m, folder.fold_expr(e))
 549          }
 550          ItemFn(decl, purity, abi, ref generics, body) => {
 551              ItemFn(
 552                  folder.fold_fn_decl(decl),
 553                  purity,
 554                  abi,
 555                  fold_generics(generics, folder),
 556                  folder.fold_block(body)
 557              )
 558          }
 559          ItemMod(ref m) => ItemMod(folder.fold_mod(m)),
 560          ItemForeignMod(ref nm) => ItemForeignMod(folder.fold_foreign_mod(nm)),
 561          ItemTy(t, ref generics) => {
 562              ItemTy(folder.fold_ty(t), fold_generics(generics, folder))
 563          }
 564          ItemEnum(ref enum_definition, ref generics) => {
 565              ItemEnum(
 566                  ast::EnumDef {
 567                      variants: enum_definition.variants.map(|&x| {
 568                          folder.fold_variant(x)
 569                      }),
 570                  },
 571                  fold_generics(generics, folder))
 572          }
 573          ItemStruct(ref struct_def, ref generics) => {
 574              let struct_def = fold_struct_def(*struct_def, folder);
 575              ItemStruct(struct_def, fold_generics(generics, folder))
 576          }
 577          ItemImpl(ref generics, ref ifce, ty, ref methods) => {
 578              ItemImpl(fold_generics(generics, folder),
 579                       ifce.as_ref().map(|p| fold_trait_ref(p, folder)),
 580                       folder.fold_ty(ty),
 581                       methods.map(|x| folder.fold_method(*x))
 582              )
 583          }
 584          ItemTrait(ref generics, ref traits, ref methods) => {
 585              let methods = methods.map(|method| {
 586                  match *method {
 587                      Required(ref m) => Required(folder.fold_type_method(m)),
 588                      Provided(method) => Provided(folder.fold_method(method))
 589                  }
 590              });
 591              ItemTrait(fold_generics(generics, folder),
 592                        traits.map(|p| fold_trait_ref(p, folder)),
 593                        methods)
 594          }
 595          ItemMac(ref m) => ItemMac(folder.fold_mac(m)),
 596      }
 597  }

 689  pub fn noop_fold_pat<T: Folder>(p: @Pat, folder: &mut T) -> @Pat {
 690      let node = match p.node {
 691          PatWild => PatWild,
 692          PatWildMulti => PatWildMulti,
 693          PatIdent(binding_mode, ref pth, ref sub) => {
 694              PatIdent(binding_mode,
 695                       folder.fold_path(pth),
 696                       sub.map(|x| folder.fold_pat(x)))
 697          }
 698          PatLit(e) => PatLit(folder.fold_expr(e)),
 699          PatEnum(ref pth, ref pats) => {
 700              PatEnum(folder.fold_path(pth),
 701                      pats.as_ref().map(|pats| pats.map(|x| folder.fold_pat(*x))))
 702          }
 703          PatStruct(ref pth, ref fields, etc) => {
 704              let pth_ = folder.fold_path(pth);
 705              let fs = fields.map(|f| {
 706                  ast::FieldPat {
 707                      ident: f.ident,
 708                      pat: folder.fold_pat(f.pat)
 709                  }
 710              });
 711              PatStruct(pth_, fs, etc)
 712          }
 713          PatTup(ref elts) => PatTup(elts.map(|x| folder.fold_pat(*x))),
 714          PatUniq(inner) => PatUniq(folder.fold_pat(inner)),
 715          PatRegion(inner) => PatRegion(folder.fold_pat(inner)),
 716          PatRange(e1, e2) => {
 717              PatRange(folder.fold_expr(e1), folder.fold_expr(e2))
 718          },
 719          PatVec(ref before, ref slice, ref after) => {
 720              PatVec(before.map(|x| folder.fold_pat(*x)),
 721                      slice.map(|x| folder.fold_pat(x)),
 722                      after.map(|x| folder.fold_pat(*x)))
 723          }
 724      };
 725  
 726      @Pat {
 727          id: folder.new_id(p.id),
 728          span: folder.new_span(p.span),
 729          node: node,
 730      }
 731  }

 849  pub fn noop_fold_stmt<T: Folder>(s: &Stmt, folder: &mut T) -> SmallVector<@Stmt> {
 850      let nodes = match s.node {
 851          StmtDecl(d, nid) => {
 852              folder.fold_decl(d).move_iter()
 853                      .map(|d| StmtDecl(d, folder.new_id(nid)))
 854                      .collect()
 855          }
 856          StmtExpr(e, nid) => {
 857              SmallVector::one(StmtExpr(folder.fold_expr(e), folder.new_id(nid)))
 858          }
 859          StmtSemi(e, nid) => {
 860              SmallVector::one(StmtSemi(folder.fold_expr(e), folder.new_id(nid)))
 861          }
 862          StmtMac(ref mac, semi) => SmallVector::one(StmtMac(folder.fold_mac(mac), semi))
 863      };
 864  
 865      nodes.move_iter().map(|node| @Spanned {

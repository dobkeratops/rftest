 376  // build a new vector of tts by appling the Folder's fold_ident to
 377  // all of the identifiers in the token trees.
 378  //
 379  // This is part of hygiene magic. As far as hygiene is concerned, there
 380  // are three types of let pattern bindings or loop labels:
 381  //      - those defined and used in non-macro part of the program
 382  //      - those used as part of macro invocation arguments
 383  //      - those defined and used inside macro definitions
 384  // Lexically, type 1 and 2 are in one group and type 3 the other. If they
 385  // clash, in order for let and loop label to work hygienically, one group
 386  // or the other needs to be renamed. The problem is that type 2 and 3 are
 387  // parsed together (inside the macro expand function). After being parsed and
 388  // AST being constructed, they can no longer be distinguished from each other.
 389  //
 390  // For that reason, type 2 let bindings and loop labels are actually renamed
 391  // in the form of tokens instead of AST nodes, here. There are wasted effort
 392  // since many token::IDENT are not necessary part of let bindings and most
 393  // token::LIFETIME are certainly not loop labels. But we can't tell in their
 394  // token form. So this is less ideal and hacky but it works.
 395  pub fn fold_tts<T: Folder>(tts: &[TokenTree], fld: &mut T) -> Vec<TokenTree> {
 396      tts.iter().map(|tt| {
 397          match *tt {
 398              TTTok(span, ref tok) =>
 399              TTTok(span,maybe_fold_ident(tok,fld)),
 400              TTDelim(ref tts) => TTDelim(Rc::new(fold_tts(tts.as_slice(), fld))),
 401              TTSeq(span, ref pattern, ref sep, is_optional) =>
 402              TTSeq(span,
 403                    Rc::new(fold_tts(pattern.as_slice(), fld)),
 404                    sep.as_ref().map(|tok|maybe_fold_ident(tok,fld)),
 405                    is_optional),
 406              TTNonterminal(sp,ref ident) =>
 407              TTNonterminal(sp,fld.fold_ident(*ident))
 408          }
 409      }).collect()
 410  }

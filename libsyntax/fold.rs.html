   76  //used in noop_fold_item and noop_fold_crate and noop_fold_crate_directive
   77  fn fold_meta_item_(mi: @MetaItem, fld: @ast_fold) -> @MetaItem {
   78      @spanned {
   79          node:
   80              match mi.node {
   81                  MetaWord(id) => MetaWord(id),
   82                  MetaList(id, ref mis) => {
   83                      let fold_meta_item = |x| fold_meta_item_(x, fld);
   84                      MetaList(
   85                          id,
   86                          mis.map(|e| fold_meta_item(*e))
   87                      )
   88                  }
   89                  MetaNameValue(id, s) => MetaNameValue(id, s)
   90              },
   91          span: fld.new_span(mi.span) }
   92  }

  128  fn fold_tts(tts : &[token_tree], fld: @ast_fold) -> ~[token_tree] {
  129      do tts.map |tt| {
  130          match *tt {
  131              tt_tok(span, ref tok) =>
  132              tt_tok(span,maybe_fold_ident(tok,fld)),
  133              tt_delim(ref tts) =>
  134              tt_delim(@mut fold_tts(**tts, fld)),
  135              tt_seq(span, ref pattern, ref sep, is_optional) =>
  136              tt_seq(span,
  137                     @mut fold_tts(**pattern, fld),
  138                     sep.map(|tok|maybe_fold_ident(tok,fld)),
  139                     is_optional),
  140              tt_nonterminal(sp,ref ident) =>
  141              tt_nonterminal(sp,fld.fold_ident(*ident))
  142          }
  143      }
  144  }

  273  pub fn noop_fold_item_underscore(i: &item_, fld: @ast_fold) -> item_ {
  274      match *i {
  275          item_static(ref t, m, e) => item_static(fld.fold_ty(t), m, fld.fold_expr(e)),
  276          item_fn(ref decl, purity, abi, ref generics, ref body) => {
  277              item_fn(
  278                  fold_fn_decl(decl, fld),
  279                  purity,
  280                  abi,
  281                  fold_generics(generics, fld),
  282                  fld.fold_block(body)
  283              )
  284          }
  285          item_mod(ref m) => item_mod(fld.fold_mod(m)),
  286          item_foreign_mod(ref nm) => {
  287              item_foreign_mod(fld.fold_foreign_mod(nm))
  288          }
  289          item_ty(ref t, ref generics) => {
  290              item_ty(fld.fold_ty(t), fold_generics(generics, fld))
  291          }
  292          item_enum(ref enum_definition, ref generics) => {
  293              item_enum(
  294                  ast::enum_def {
  295                      variants: do enum_definition.variants.map |x| {
  296                          fld.fold_variant(x)
  297                      },
  298                  },
  299                  fold_generics(generics, fld))
  300          }
  301          item_struct(ref struct_def, ref generics) => {
  302              let struct_def = fold_struct_def(*struct_def, fld);
  303              item_struct(struct_def, /* FIXME (#2543) */ (*generics).clone())
  304          }
  305          item_impl(ref generics, ref ifce, ref ty, ref methods) => {
  306              item_impl(
  307                  fold_generics(generics, fld),
  308                  ifce.map(|p| fold_trait_ref(p, fld)),
  309                  fld.fold_ty(ty),
  310                  methods.map(|x| fld.fold_method(*x))
  311              )
  312          }
  313          item_trait(ref generics, ref traits, ref methods) => {
  314              let methods = do methods.map |method| {
  315                  match *method {
  316                      required(*) => (*method).clone(),
  317                      provided(method) => provided(fld.fold_method(method))
  318                  }
  319              };
  320              item_trait(
  321                  fold_generics(generics, fld),
  322                  traits.map(|p| fold_trait_ref(p, fld)),
  323                  methods
  324              )
  325          }
  326          item_mac(ref m) => {
  327              // FIXME #2888: we might actually want to do something here.
  328              // ... okay, we're doing something. It would probably be nicer
  329              // to add something to the ast_fold trait, but I'll defer
  330              // that work.
  331              item_mac(fold_mac_(m,fld))
  332          }
  333      }
  334  }

  380  pub fn noop_fold_block(b: &Block, fld: @ast_fold) -> Block {
  381      let view_items = b.view_items.map(|x| fld.fold_view_item(x));
  382      let mut stmts = ~[];
  383      for stmt in b.stmts.iter() {
  384          match fld.fold_stmt(*stmt) {
  385              None => {}
  386              Some(stmt) => stmts.push(stmt)
  387          }
  388      }
  389      ast::Block {
  390          view_items: view_items,
  391          stmts: stmts,
  392          expr: b.expr.map(|x| fld.fold_expr(*x)),
  393          id: fld.new_id(b.id),
  394          rules: b.rules,

  27      fn fold_view_paths(&self, view_paths: &[@view_path]) -> ~[@view_path] {
  28          view_paths.map(|view_path| {
  29              let inner_view_path = match view_path.node {
  30                  view_path_simple(ref ident, ref path, node_id) => {
  31                      view_path_simple(ident.clone(),
  32                                       self.fold_path(path),
  33                                       self.new_id(node_id))
  34                  }
  35                  view_path_glob(ref path, node_id) => {
  36                      view_path_glob(self.fold_path(path), self.new_id(node_id))
  37                  }
  38                  view_path_list(ref path, ref path_list_idents, node_id) => {
  39                      view_path_list(self.fold_path(path),
  40                                     path_list_idents.map(|path_list_ident| {
  41                                      let id = self.new_id(path_list_ident.node
  42                                                                          .id);
  43                                      Spanned {
  44                                          node: path_list_ident_ {
  45                                              name: path_list_ident.node
  46                                                                   .name
  47                                                                   .clone(),
  48                                              id: id,
  49                                          },
  50                                          span: self.new_span(
  51                                              path_list_ident.span)
  52                                     }
  53                                    }),
  54                                    self.new_id(node_id))
  55                  }
  56              };
  57              @Spanned {
  58                  node: inner_view_path,
  59                  span: self.new_span(view_path.span),
  60              }
  61          })
  62      }

  64      fn fold_view_item(&self, vi: &view_item) -> view_item {
  65          let inner_view_item = match vi.node {
  66              view_item_extern_mod(ref ident,
  67                                   string,
  68                                   ref meta_items,

  71                                       string,

  73                                       self.new_id(node_id))
  74              }

  77              }
  78          };

  81              attrs: vi.attrs.map(|a| fold_attribute_(*a, self)),

  84          }

 173      fn fold_pat(&self, p: @Pat) -> @Pat {
 174          let node = match p.node {
 175              PatWild => PatWild,
 176              PatIdent(binding_mode, ref pth, ref sub) => {
 177                  PatIdent(binding_mode,
 178                           self.fold_path(pth),
 179                           sub.map_move(|x| self.fold_pat(x)))
 180              }
 181              PatLit(e) => PatLit(self.fold_expr(e)),
 182              PatEnum(ref pth, ref pats) => {
 183                  PatEnum(self.fold_path(pth),
 184                          pats.map(|pats| pats.map(|x| self.fold_pat(*x))))
 185              }
 186              PatStruct(ref pth, ref fields, etc) => {
 187                  let pth_ = self.fold_path(pth);
 188                  let fs = do fields.map |f| {
 189                      ast::FieldPat {
 190                          ident: f.ident,
 191                          pat: self.fold_pat(f.pat)
 192                      }
 193                  };
 194                  PatStruct(pth_, fs, etc)
 195              }
 196              PatTup(ref elts) => PatTup(elts.map(|x| self.fold_pat(*x))),
 197              PatBox(inner) => PatBox(self.fold_pat(inner)),
 198              PatUniq(inner) => PatUniq(self.fold_pat(inner)),
 199              PatRegion(inner) => PatRegion(self.fold_pat(inner)),
 200              PatRange(e1, e2) => {
 201                  PatRange(self.fold_expr(e1), self.fold_expr(e2))
 202              },
 203              PatVec(ref before, ref slice, ref after) => {
 204                  PatVec(before.map(|x| self.fold_pat(*x)),
 205                         slice.map_move(|x| self.fold_pat(x)),
 206                         after.map(|x| self.fold_pat(*x)))
 207              }
 208          };
 209  
 210          @Pat {
 211              id: self.new_id(p.id),
 212              span: self.new_span(p.span),
 213              node: node,
 214          }
 215      }

 217      fn fold_decl(&self, d: @Decl) -> Option<@Decl> {
 218          let node = match d.node {
 219              DeclLocal(ref l) => Some(DeclLocal(self.fold_local(*l))),
 220              DeclItem(it) => {
 221                  match self.fold_item(it) {
 222                      Some(it_folded) => Some(DeclItem(it_folded)),
 223                      None => None,
 224                  }
 225              }
 226          };
 227  
 228          node.map_move(|node| {
 229              @Spanned {
 230                  node: node,
 231                  span: d.span,
 232              }
 233          })
 234      }

 240      fn fold_ty(&self, t: &Ty) -> Ty {
 241          let node = match t.node {
 242              ty_nil | ty_bot | ty_infer => t.node.clone(),
 243              ty_box(ref mt) => ty_box(fold_mt(mt, self)),
 244              ty_uniq(ref mt) => ty_uniq(fold_mt(mt, self)),
 245              ty_vec(ref mt) => ty_vec(fold_mt(mt, self)),
 246              ty_ptr(ref mt) => ty_ptr(fold_mt(mt, self)),
 247              ty_rptr(region, ref mt) => ty_rptr(region, fold_mt(mt, self)),
 248              ty_closure(ref f) => {
 249                  ty_closure(@TyClosure {
 250                      sigil: f.sigil,
 251                      purity: f.purity,
 252                      region: f.region,
 253                      onceness: f.onceness,
 254                      bounds: fold_opt_bounds(&f.bounds, self),
 255                      decl: fold_fn_decl(&f.decl, self),
 256                      lifetimes: f.lifetimes.map(|l| fold_lifetime(l, self)),
 257                  })
 258              }
 259              ty_bare_fn(ref f) => {
 260                  ty_bare_fn(@TyBareFn {
 261                      lifetimes: f.lifetimes.map(|l| fold_lifetime(l, self)),
 262                      purity: f.purity,
 263                      abis: f.abis,
 264                      decl: fold_fn_decl(&f.decl, self)
 265                  })
 266              }
 267              ty_tup(ref tys) => ty_tup(tys.map(|ty| self.fold_ty(ty))),
 268              ty_path(ref path, ref bounds, id) => {
 269                  ty_path(self.fold_path(path),
 270                          fold_opt_bounds(bounds, self),
 271                          self.new_id(id))
 272              }
 273              ty_fixed_length_vec(ref mt, e) => {
 274                  ty_fixed_length_vec(fold_mt(mt, self), self.fold_expr(e))
 275              }
 276              ty_mac(ref mac) => ty_mac(self.fold_mac(mac)),
 277              ty_typeof(expr) => ty_typeof(self.fold_expr(expr)),
 278          };
 279          Ty {
 280              id: self.new_id(t.id),
 281              span: self.new_span(t.span),
 282              node: node,
 283          }
 284      }

 305      fn fold_variant(&self, v: &variant) -> variant {
 306          let kind;
 307          match v.node.kind {
 308              tuple_variant_kind(ref variant_args) => {
 309                  kind = tuple_variant_kind(variant_args.map(|x|
 310                      fold_variant_arg_(x, self)))
 311              }
 312              struct_variant_kind(ref struct_def) => {
 313                  kind = struct_variant_kind(@ast::struct_def {
 314                      fields: struct_def.fields.iter()
 315                          .map(|f| self.fold_struct_field(*f)).collect(),
 316                      ctor_id: struct_def.ctor_id.map(|c| self.new_id(*c))
 317                  })
 318              }
 319          }
 320  
 321          let fold_attribute = |x| fold_attribute_(x, self);
 322          let attrs = v.node.attrs.map(|x| fold_attribute(*x));
 323  
 324          let de = match v.node.disr_expr {
 325            Some(e) => Some(self.fold_expr(e)),
 326            None => None
 327          };
 328          let node = ast::variant_ {
 329              name: v.node.name,
 330              attrs: attrs,
 331              kind: kind,
 332              id: self.new_id(v.node.id),
 333              disr_expr: de,
 334              vis: v.node.vis,
 335          };
 336          Spanned {
 337              node: node,
 338              span: self.new_span(v.span),
 339          }
 340      }

 437  // build a new vector of tts by appling the ast_fold's fold_ident to
 438  // all of the identifiers in the token trees.
 439  pub fn fold_tts<T:ast_fold>(tts: &[token_tree], fld: &T) -> ~[token_tree] {
 440      do tts.map |tt| {
 441          match *tt {
 442              tt_tok(span, ref tok) =>
 443              tt_tok(span,maybe_fold_ident(tok,fld)),
 444              tt_delim(ref tts) => tt_delim(@mut fold_tts(**tts, fld)),
 445              tt_seq(span, ref pattern, ref sep, is_optional) =>
 446              tt_seq(span,
 447                     @mut fold_tts(**pattern, fld),
 448                     sep.map(|tok|maybe_fold_ident(tok,fld)),
 449                     is_optional),
 450              tt_nonterminal(sp,ref ident) =>
 451              tt_nonterminal(sp,fld.fold_ident(*ident))
 452          }
 453      }
 454  }

 456  // apply ident folder if it's an ident, otherwise leave it alone

 614  pub fn noop_fold_item_underscore<T:ast_fold>(i: &item_, folder: &T) -> item_ {
 615      match *i {
 616          item_static(ref t, m, e) => {
 617              item_static(folder.fold_ty(t), m, folder.fold_expr(e))
 618          }
 619          item_fn(ref decl, purity, abi, ref generics, ref body) => {
 620              item_fn(
 621                  fold_fn_decl(decl, folder),
 622                  purity,
 623                  abi,
 624                  fold_generics(generics, folder),
 625                  folder.fold_block(body)
 626              )
 627          }
 628          item_mod(ref m) => item_mod(folder.fold_mod(m)),
 629          item_foreign_mod(ref nm) => {
 630              item_foreign_mod(folder.fold_foreign_mod(nm))
 631          }
 632          item_ty(ref t, ref generics) => {
 633              item_ty(folder.fold_ty(t),
 634                      fold_generics(generics, folder))
 635          }
 636          item_enum(ref enum_definition, ref generics) => {
 637              item_enum(
 638                  ast::enum_def {
 639                      variants: do enum_definition.variants.map |x| {
 640                          folder.fold_variant(x)
 641                      },
 642                  },
 643                  fold_generics(generics, folder))
 644          }
 645          item_struct(ref struct_def, ref generics) => {
 646              let struct_def = fold_struct_def(*struct_def, folder);
 647              item_struct(struct_def, fold_generics(generics, folder))
 648          }
 649          item_impl(ref generics, ref ifce, ref ty, ref methods) => {
 650              item_impl(fold_generics(generics, folder),
 651                        ifce.map(|p| fold_trait_ref(p, folder)),
 652                        folder.fold_ty(ty),
 653                        methods.map(|x| folder.fold_method(*x))
 654              )
 655          }
 656          item_trait(ref generics, ref traits, ref methods) => {
 657              let methods = do methods.map |method| {
 658                  match *method {
 659                      required(ref m) => required(folder.fold_type_method(m)),
 660                      provided(method) => provided(folder.fold_method(method))
 661                  }
 662              };
 663              item_trait(fold_generics(generics, folder),
 664                         traits.map(|p| fold_trait_ref(p, folder)),
 665                         methods)
 666          }
 667          item_mac(ref m) => item_mac(folder.fold_mac(m)),
 668      }
 669  }

 720  pub fn noop_fold_expr<T:ast_fold>(e: @ast::Expr, folder: &T) -> @ast::Expr {
 721      let fold_field = |x| fold_field_(x, folder);
 722  
 723      let node = match e.node {
 724          ExprVstore(e, v) => {
 725              ExprVstore(folder.fold_expr(e), v)
 726          }
 727          ExprVec(ref exprs, mutt) => {
 728              ExprVec(folder.map_exprs(|x| folder.fold_expr(x), *exprs), mutt)
 729          }
 730          ExprRepeat(expr, count, mutt) => {
 731              ExprRepeat(folder.fold_expr(expr), folder.fold_expr(count), mutt)
 732          }
 733          ExprTup(ref elts) => ExprTup(elts.map(|x| folder.fold_expr(*x))),
 734          ExprCall(f, ref args, blk) => {
 735              ExprCall(folder.fold_expr(f),
 736                       folder.map_exprs(|x| folder.fold_expr(x), *args),
 737                       blk)
 738          }
 739          ExprMethodCall(callee_id, f, i, ref tps, ref args, blk) => {
 740              ExprMethodCall(
 741                  folder.new_id(callee_id),
 742                  folder.fold_expr(f),
 743                  folder.fold_ident(i),
 744                  tps.map(|x| folder.fold_ty(x)),
 745                  folder.map_exprs(|x| folder.fold_expr(x), *args),
 746                  blk
 747              )
 748          }
 749          ExprBinary(callee_id, binop, lhs, rhs) => {
 750              ExprBinary(folder.new_id(callee_id),
 751                         binop,
 752                         folder.fold_expr(lhs),
 753                         folder.fold_expr(rhs))
 754          }
 755          ExprUnary(callee_id, binop, ohs) => {
 756              ExprUnary(folder.new_id(callee_id), binop, folder.fold_expr(ohs))
 757          }
 758          ExprDoBody(f) => ExprDoBody(folder.fold_expr(f)),
 759          ExprLit(_) => e.node.clone(),
 760          ExprCast(expr, ref ty) => {
 761              ExprCast(folder.fold_expr(expr), folder.fold_ty(ty))
 762          }
 763          ExprAddrOf(m, ohs) => ExprAddrOf(m, folder.fold_expr(ohs)),
 764          ExprIf(cond, ref tr, fl) => {
 765              ExprIf(folder.fold_expr(cond),
 766                     folder.fold_block(tr),
 767                     fl.map_move(|x| folder.fold_expr(x)))
 768          }
 769          ExprWhile(cond, ref body) => {
 770              ExprWhile(folder.fold_expr(cond), folder.fold_block(body))
 771          }
 772          ExprForLoop(pat, iter, ref body, ref maybe_ident) => {
 773              ExprForLoop(folder.fold_pat(pat),
 774                          folder.fold_expr(iter),
 775                          folder.fold_block(body),
 776                          maybe_ident.map_move(|i| folder.fold_ident(i)))
 777          }
 778          ExprLoop(ref body, opt_ident) => {
 779              ExprLoop(folder.fold_block(body),
 780                       opt_ident.map_move(|x| folder.fold_ident(x)))
 781          }
 782          ExprMatch(expr, ref arms) => {
 783              ExprMatch(folder.fold_expr(expr),
 784                        arms.map(|x| folder.fold_arm(x)))
 785          }
 786          ExprFnBlock(ref decl, ref body) => {
 787              ExprFnBlock(
 788                  fold_fn_decl(decl, folder),
 789                  folder.fold_block(body)
 790              )
 791          }
 792          ExprBlock(ref blk) => ExprBlock(folder.fold_block(blk)),
 793          ExprAssign(el, er) => {
 794              ExprAssign(folder.fold_expr(el), folder.fold_expr(er))
 795          }
 796          ExprAssignOp(callee_id, op, el, er) => {
 797              ExprAssignOp(folder.new_id(callee_id),
 798                           op,
 799                           folder.fold_expr(el),
 800                           folder.fold_expr(er))
 801          }
 802          ExprField(el, id, ref tys) => {
 803              ExprField(folder.fold_expr(el), folder.fold_ident(id),
 804                        tys.map(|x| folder.fold_ty(x)))
 805          }
 806          ExprIndex(callee_id, el, er) => {
 807              ExprIndex(folder.new_id(callee_id),
 808                        folder.fold_expr(el),
 809                        folder.fold_expr(er))
 810          }
 811          ExprPath(ref pth) => ExprPath(folder.fold_path(pth)),
 812          ExprSelf => ExprSelf,
 813          ExprLogLevel => ExprLogLevel,
 814          ExprBreak(opt_ident) => ExprBreak(opt_ident),
 815          ExprAgain(opt_ident) => ExprAgain(opt_ident),
 816          ExprRet(ref e) => {
 817              ExprRet(e.map_move(|x| folder.fold_expr(x)))
 818          }
 819          ExprInlineAsm(ref a) => {
 820              ExprInlineAsm(inline_asm {
 821                  inputs: a.inputs.map(|&(c, input)| (c, folder.fold_expr(input))),
 822                  outputs: a.outputs.map(|&(c, out)| (c, folder.fold_expr(out))),
 823                  .. (*a).clone()
 824              })
 825          }
 826          ExprMac(ref mac) => ExprMac(folder.fold_mac(mac)),
 827          ExprStruct(ref path, ref fields, maybe_expr) => {
 828              ExprStruct(folder.fold_path(path),
 829                         fields.map(|x| fold_field(*x)),
 830                         maybe_expr.map_move(|x| folder.fold_expr(x)))
 831          },
 832          ExprParen(ex) => ExprParen(folder.fold_expr(ex))
 833      };
 834  
 835      @Expr {
 836          id: folder.new_id(e.id),
 837          node: node,
 838          span: folder.new_span(e.span),
 839      }
 840  }

 842  pub fn noop_fold_stmt<T:ast_fold>(s: &Stmt, folder: &T) -> Option<@Stmt> {
 843      let node = match s.node {
 844          StmtDecl(d, nid) => {
 845              match folder.fold_decl(d) {
 846                  Some(d) => Some(StmtDecl(d, folder.new_id(nid))),
 847                  None => None,
 848              }
 849          }
 850          StmtExpr(e, nid) => {
 851              Some(StmtExpr(folder.fold_expr(e), folder.new_id(nid)))
 852          }
 853          StmtSemi(e, nid) => {
 854              Some(StmtSemi(folder.fold_expr(e), folder.new_id(nid)))
 855          }
 856          StmtMac(ref mac, semi) => Some(StmtMac(folder.fold_mac(mac), semi))
 857      };
 858  
 859      node.map_move(|node| @Spanned {
 860          node: node,
 861          span: folder.new_span(s.span),
 862      })
 863  }

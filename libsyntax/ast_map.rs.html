  36  pub fn path_to_str_with_sep(p: &[path_elt], sep: &str, itr: @ident_interner)
  37                           -> ~str {
  38      let strs = do p.map |e| {
  39          match *e {
  40            path_mod(s) => itr.get(s.name),
  41            path_name(s) => itr.get(s.name)
  42          }
  43      };
  44      strs.connect(sep)
  45  }

  47  pub fn path_ident_to_str(p: &path, i: ident, itr: @ident_interner) -> ~str {
  48      if p.is_empty() {
  49          itr.get(i.name).to_owned()
  50      } else {
  51          fmt!("%s::%s", path_to_str(*p, itr), itr.get(i.name))
  52      }
  53  }

  55  pub fn path_to_str(p: &[path_elt], itr: @ident_interner) -> ~str {
  56      path_to_str_with_sep(p, "::", itr)
  57  }

  91  impl Ctx {
  92      fn extend(@mut self, elt: ident) -> @path {
  93          @vec::append(self.path.clone(), [path_name(elt)])
  94      }

  96      fn map_method(@mut self,
  97                    impl_did: def_id,
  98                    impl_path: @path,
  99                    m: @method,
 100                    is_provided: bool) {
 101          let entry = if is_provided {
 102              node_trait_method(@provided(m), impl_did, impl_path)
 103          } else {
 104              node_method(m, impl_did, impl_path)
 105          };
 106          self.map.insert(m.id, entry);
 107          self.map.insert(m.self_id, node_local(special_idents::self_));
 108      }

 110      fn map_struct_def(@mut self,
 111                        struct_def: @ast::struct_def,
 112                        parent_node: ast_node,
 113                        ident: ast::ident) {
 114          let p = self.extend(ident);
 115  
 116          // If this is a tuple-like struct, register the constructor.
 117          match struct_def.ctor_id {
 118              None => {}
 119              Some(ctor_id) => {
 120                  match parent_node {
 121                      node_item(item, _) => {
 122                          self.map.insert(ctor_id,
 123                                          node_struct_ctor(struct_def,
 124                                                           item,
 125                                                           p));
 126                      }
 127                      _ => fail!("struct def parent wasn't an item")
 128                  }
 129              }
 130          }
 131      }

 133      fn map_expr(@mut self, ex: @expr) {
 134          self.map.insert(ex.id, node_expr(ex));
 135  
 136          // Expressions which are or might be calls:
 137          {
 138              let r = ex.get_callee_id();
 139              for callee_id in r.iter() {
 140                  self.map.insert(*callee_id, node_callee_scope(ex));
 141              }
 142          }
 143  
 144          visit::visit_expr(self as @mut Visitor<()>, ex, ());
 145      }

 170      fn map_pat(@mut self, pat: @pat) {
 171          match pat.node {
 172              pat_ident(_, ref path, _) => {
 173                  // Note: this is at least *potentially* a pattern...
 174                  self.map.insert(pat.id,
 175                                  node_local(ast_util::path_to_ident(path)));
 176              }
 177              _ => ()

 180          visit::visit_pat(self as @mut Visitor<()>, pat, ());
 181      }

 184  impl Visitor<()> for Ctx {
 185      fn visit_item(@mut self, i: @item, _: ()) {
 186          // clone is FIXME #2543
 187          let item_path = @self.path.clone();
 188          self.map.insert(i.id, node_item(i, item_path));
 189          match i.node {
 190              item_impl(_, _, _, ref ms) => {
 191                  let impl_did = ast_util::local_def(i.id);
 192                  for m in ms.iter() {
 193                      self.map_method(impl_did, self.extend(i.ident), *m, false)
 194                  }
 195              }
 196              item_enum(ref enum_definition, _) => {
 197                  for v in (*enum_definition).variants.iter() {
 198                      // FIXME #2543: bad clone
 199                      self.map.insert(v.node.id,
 200                                      node_variant((*v).clone(),
 201                                                   i,
 202                                                   self.extend(i.ident)));
 203                  }
 204              }
 205              item_foreign_mod(ref nm) => {
 206                  for nitem in nm.items.iter() {
 207                      // Compute the visibility for this native item.
 208                      let visibility = match nitem.vis {
 209                          public => public,
 210                          private => private,
 211                          inherited => i.vis
 212                      };
 213  
 214                      self.map.insert(nitem.id,
 215                                      node_foreign_item(*nitem,
 216                                                        nm.abis,
 217                                                        visibility,
 218                                                        // FIXME (#2543)
 219                                                        if nm.sort ==
 220                                                              ast::named {
 221                                                          self.extend(i.ident)
 222                                                        } else {
 223                                                          // Anonymous extern
 224                                                          // mods go in the
 225                                                          // parent scope.
 226                                                          @self.path.clone()
 227                                                        }));
 228                  }
 229              }
 230              item_struct(struct_def, _) => {
 231                  self.map_struct_def(struct_def,
 232                                      node_item(i, item_path),
 233                                      i.ident)
 234              }
 235              item_trait(_, ref traits, ref methods) => {
 236                  for p in traits.iter() {
 237                      self.map.insert(p.ref_id, node_item(i, item_path));
 238                  }
 239                  for tm in methods.iter() {
 240                      let id = ast_util::trait_method_to_ty_method(tm).id;
 241                      let d_id = ast_util::local_def(i.id);
 242                      self.map.insert(id,
 243                                      node_trait_method(@(*tm).clone(),
 244                                                        d_id,
 245                                                        item_path));
 246                  }
 247              }
 248              _ => {}
 249          }
 250  
 251          match i.node {
 252              item_mod(_) | item_foreign_mod(_) => {
 253                  self.path.push(path_mod(i.ident));
 254              }
 255              _ => self.path.push(path_name(i.ident))
 256          }
 257          visit::visit_item(self as @mut Visitor<()>, i, ());
 258          self.path.pop();
 259      }

 373          diag: diag,

 375      visit::visit_crate(cx as @mut Visitor<()>, c, ());
 376      cx.map
 377  }

 379  // Used for items loaded from external crate that are being inlined into this
 380  // crate.  The `path` should be the path to the item but should not include
 381  // the item itself.
 382  pub fn map_decoded_item(diag: @mut span_handler,
 383                          map: map,
 384                          path: path,
 385                          ii: &inlined_item) {
 386      // I believe it is ok for the local IDs of inlined items from other crates
 387      // to overlap with the local ids from this crate, so just generate the ids
 388      // starting from 0.
 389      let cx = @mut Ctx {
 390          map: map,
 391          path: path.clone(),
 392          diag: diag,
 393      };
 394  
 395      // methods get added to the AST map when their impl is visited.  Since we
 396      // don't decode and instantiate the impl, but just the method, we have to
 397      // add it to the table now:
 398      match *ii {
 399          ii_item(*) => {} // fallthrough
 400          ii_foreign(i) => {
 401              cx.map.insert(i.id, node_foreign_item(i,
 402                                                    AbiSet::Intrinsic(),
 403                                                    i.vis,    // Wrong but OK
 404                                                    @path));
 405          }
 406          ii_method(impl_did, is_provided, m) => {
 407              cx.map_method(impl_did, @path, m, is_provided);
 408          }
 409      }
 410  
 411      // visit the item / method contents and add those to the map:
 412      ii.accept((), cx as @mut Visitor<()>);
 413  }

 415  pub fn node_id_to_str(map: map, id: NodeId, itr: @ident_interner) -> ~str {
 416      match map.find(&id) {
 417        None => {
 418          fmt!("unknown node (id=%d)", id)
 419        }
 420        Some(&node_item(item, path)) => {
 421          let path_str = path_ident_to_str(path, item.ident, itr);
 422          let item_str = match item.node {
 423            item_static(*) => ~"static",
 424            item_fn(*) => ~"fn",
 425            item_mod(*) => ~"mod",
 426            item_foreign_mod(*) => ~"foreign mod",
 427            item_ty(*) => ~"ty",
 428            item_enum(*) => ~"enum",
 429            item_struct(*) => ~"struct",
 430            item_trait(*) => ~"trait",
 431            item_impl(*) => ~"impl",
 432            item_mac(*) => ~"macro"
 433          };
 434          fmt!("%s %s (id=%?)", item_str, path_str, id)
 435        }
 436        Some(&node_foreign_item(item, abi, _, path)) => {
 437          fmt!("foreign item %s with abi %? (id=%?)",
 438               path_ident_to_str(path, item.ident, itr), abi, id)
 439        }
 440        Some(&node_method(m, _, path)) => {
 441          fmt!("method %s in %s (id=%?)",
 442               itr.get(m.ident.name), path_to_str(*path, itr), id)
 443        }
 444        Some(&node_trait_method(ref tm, _, path)) => {
 445          let m = ast_util::trait_method_to_ty_method(&**tm);
 446          fmt!("method %s in %s (id=%?)",
 447               itr.get(m.ident.name), path_to_str(*path, itr), id)
 448        }
 449        Some(&node_variant(ref variant, _, path)) => {
 450          fmt!("variant %s in %s (id=%?)",
 451               itr.get(variant.node.name.name), path_to_str(*path, itr), id)
 452        }
 453        Some(&node_expr(expr)) => {
 454          fmt!("expr %s (id=%?)", pprust::expr_to_str(expr, itr), id)
 455        }
 456        Some(&node_callee_scope(expr)) => {
 457          fmt!("callee_scope %s (id=%?)", pprust::expr_to_str(expr, itr), id)
 458        }
 459        Some(&node_stmt(stmt)) => {
 460          fmt!("stmt %s (id=%?)",
 461               pprust::stmt_to_str(stmt, itr), id)
 462        }
 463        Some(&node_arg) => {

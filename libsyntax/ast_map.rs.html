 183  pub struct Map {
 184      /// NodeIds are sequential integers from 0, so we can be
 185      /// super-compact by storing them in a vector. Not everything with
 186      /// a NodeId is in the map, but empirically the occupancy is about
 187      /// 75-80%, so there's not too much overhead (certainly less than
 188      /// a hashmap, since they (at the time of writing) have a maximum
 189      /// of 75% occupancy).
 190      ///
 191      /// Also, indexing is pretty quick when you've got a vector and
 192      /// plain old integers.
 193      map: RefCell<Vec<MapEntry> >
 194  }

 196  impl Map {
 197      fn find_entry(&self, id: NodeId) -> Option<MapEntry> {
 198          let map = self.map.borrow();
 199          if map.len() > id as uint {
 200              Some(*map.get(id as uint))
 201          } else {
 202              None
 203          }
 204      }

 206      /// Retrieve the Node corresponding to `id`, failing if it cannot
 207      /// be found.
 208      pub fn get(&self, id: NodeId) -> Node {
 209          match self.find(id) {
 210              Some(node) => node,
 211              None => fail!("couldn't find node id {} in the AST map", id)
 212          }
 213      }

 215      /// Retrieve the Node corresponding to `id`, returning None if
 216      /// cannot be found.
 217      pub fn find(&self, id: NodeId) -> Option<Node> {
 218          self.find_entry(id).and_then(|x| x.to_node())

 223      pub fn get_parent(&self, id: NodeId) -> NodeId {
 224          self.find_entry(id).and_then(|x| x.parent()).unwrap_or(id)

 275      pub fn get_path_elem(&self, id: NodeId) -> PathElem {
 276          match self.get(id) {
 277              NodeItem(item) => {
 278                  match item.node {
 279                      ItemMod(_) | ItemForeignMod(_) => {
 280                          PathMod(item.ident.name)
 281                      }
 282                      _ => PathName(item.ident.name)
 283                  }
 284              }
 285              NodeForeignItem(i) => PathName(i.ident.name),
 286              NodeMethod(m) => PathName(m.ident.name),
 287              NodeTraitMethod(tm) => match *tm {
 288                  Required(ref m) => PathName(m.ident.name),
 289                  Provided(ref m) => PathName(m.ident.name)
 290              },
 291              NodeVariant(v) => PathName(v.node.name.name),

 310      fn with_path_next<T>(&self, id: NodeId, next: LinkedPath, f: |PathElems| -> T) -> T {
 311          let parent = self.get_parent(id);
 312          let parent = match self.find_entry(id) {
 313              Some(EntryForeignItem(..)) | Some(EntryVariant(..)) => {
 314                  // Anonymous extern items, enum variants and struct ctors
 315                  // go in the parent scope.
 316                  self.get_parent(parent)
 317              }
 318              // But tuple struct ctors don't have names, so use the path of its
 319              // parent, the struct item. Similarly with closure expressions.
 320              Some(EntryStructCtor(..)) | Some(EntryExpr(..)) => {
 321                  return self.with_path_next(parent, next, f);
 322              }
 323              _ => parent
 324          };
 325          if parent == id {
 326              match self.find_entry(id) {
 327                  Some(RootInlinedParent(data)) => {
 328                      f(Values(data.path.iter()).chain(next))
 329                  }
 330                  _ => f(Values([].iter()).chain(next))
 331              }
 332          } else {
 333              self.with_path_next(parent, Some(&LinkedPathNode {
 334                  node: self.get_path_elem(id),
 335                  next: next
 336              }), f)
 337          }
 338      }

 340      pub fn with_attrs<T>(&self, id: NodeId, f: |Option<&[Attribute]>| -> T) -> T {
 341          let attrs = match self.get(id) {
 342              NodeItem(i) => Some(i.attrs.as_slice()),
 343              NodeForeignItem(fi) => Some(fi.attrs.as_slice()),
 344              NodeTraitMethod(tm) => match *tm {
 345                  Required(ref type_m) => Some(type_m.attrs.as_slice()),
 346                  Provided(m) => Some(m.attrs.as_slice())
 347              },
 348              NodeMethod(m) => Some(m.attrs.as_slice()),
 349              NodeVariant(ref v) => Some(v.node.attrs.as_slice()),
 350              // unit/tuple structs take the attributes straight from
 351              // the struct definition.
 352              // FIXME(eddyb) make this work again (requires access to the map).
 353              NodeStructCtor(_) => {
 354                  return self.with_attrs(self.get_parent(id), f);
 355              }
 356              _ => None
 357          };
 358          f(attrs)
 359      }

 396  pub struct Ctx<'a, F> {
 397      map: &'a Map,
 398      // The node in which we are currently mapping (an item or a method).
 399      // When equal to DUMMY_NODE_ID, the next mapped node becomes the parent.
 400      parent: NodeId,
 401      fold_ops: F
 402  }
 403  
 404  impl<'a, F> Ctx<'a, F> {
 405      fn insert(&self, id: NodeId, entry: MapEntry) {
 406          (*self.map.map.borrow_mut()).grow_set(id as uint, &NotPresent, entry);
 407      }
 408  }
 409  
 410  impl<'a, F: FoldOps> Folder for Ctx<'a, F> {
 411      fn new_id(&mut self, id: NodeId) -> NodeId {
 412          let id = self.fold_ops.new_id(id);
 413          if self.parent == DUMMY_NODE_ID {
 414              self.parent = id;
 415          }
 416          id
 417      }
 418  
 419      fn new_span(&mut self, span: Span) -> Span {
 420          self.fold_ops.new_span(span)
 421      }
 422  
 423      fn fold_item(&mut self, i: @Item) -> SmallVector<@Item> {
 424          let parent = self.parent;
 425          self.parent = DUMMY_NODE_ID;
 426  
 427          let i = fold::noop_fold_item(i, self).expect_one("expected one item");
 428          assert_eq!(self.parent, i.id);
 429  
 430          match i.node {
 431              ItemImpl(_, _, _, ref ms) => {
 432                  for &m in ms.iter() {
 433                      self.insert(m.id, EntryMethod(self.parent, m));
 434                  }
 435              }
 436              ItemEnum(ref enum_definition, _) => {
 437                  for &v in enum_definition.variants.iter() {
 438                      self.insert(v.node.id, EntryVariant(self.parent, v));
 439                  }
 440              }
 441              ItemForeignMod(ref nm) => {
 442                  for &nitem in nm.items.iter() {
 443                      self.insert(nitem.id, EntryForeignItem(self.parent, nitem));
 444                  }
 445              }
 446              ItemStruct(struct_def, _) => {
 447                  // If this is a tuple-like struct, register the constructor.
 448                  match struct_def.ctor_id {
 449                      Some(ctor_id) => {
 450                          self.insert(ctor_id, EntryStructCtor(self.parent,
 451                                                               struct_def));
 452                      }
 453                      None => {}

  30  pub enum path_elt {
  31      path_mod(Ident),
  32      path_name(Ident),
  33  
  34      // A pretty name can come from an `impl` block. We attempt to select a
  35      // reasonable name for debuggers to see, but to guarantee uniqueness with
  36      // other paths the hash should also be taken into account during symbol
  37      // generation.
  38      path_pretty_name(Ident, u64),
  39  }

  41  impl path_elt {
  42      pub fn ident(&self) -> Ident {
  43          match *self {
  44              path_mod(ident)            |
  45              path_name(ident)           |
  46              path_pretty_name(ident, _) => ident
  47          }
  48      }
  49  }

  51  pub type path = ~[path_elt];
  52  
  53  pub fn path_to_str_with_sep(p: &[path_elt], sep: &str, itr: @ident_interner)
  54                           -> ~str {
  55      let strs = do p.map |e| {
  56          match *e {
  57              path_mod(s) | path_name(s) | path_pretty_name(s, _) => {
  58                  itr.get(s.name)
  59              }
  60          }
  61      };
  62      strs.connect(sep)
  63  }

  65  pub fn path_ident_to_str(p: &path, i: Ident, itr: @ident_interner) -> ~str {
  66      if p.is_empty() {
  67          itr.get(i.name).to_owned()
  68      } else {
  69          fmt!("%s::%s", path_to_str(*p, itr), itr.get(i.name))
  70      }
  71  }
  72  
  73  pub fn path_to_str(p: &[path_elt], itr: @ident_interner) -> ~str {
  74      path_to_str_with_sep(p, "::", itr)
  75  }
  76  
  77  pub fn path_elt_to_str(pe: path_elt, itr: @ident_interner) -> ~str {
  78      match pe {
  79          path_mod(s) | path_name(s) | path_pretty_name(s, _) => {
  80              itr.get(s.name).to_owned()
  81          }
  82      }
  83  }

  85  pub fn impl_pretty_name(trait_ref: &Option<trait_ref>,
  86                          ty: &Ty, default: Ident) -> path_elt {
  87      let itr = get_ident_interner();
  88      let ty_ident = match ty.node {
  89          ty_path(ref path, _, _) => path.segments.last().identifier,
  90          _ => default
  91      };
  92      let hash = (trait_ref, ty).hash();
  93      match *trait_ref {
  94          None => path_pretty_name(ty_ident, hash),
  95          Some(ref trait_ref) => {
  96              // XXX: this dollar sign is actually a relic of being one of the
  97              //      very few valid symbol names on unix. These kinds of
  98              //      details shouldn't be exposed way up here in the ast.
  99              let s = fmt!("%s$%s",
 100                           itr.get(trait_ref.path.segments.last().identifier.name),
 101                           itr.get(ty_ident.name));
 102              path_pretty_name(Ident::new(itr.gensym(s)), hash)
 103          }
 104      }
 105  }

 107  #[deriving(Clone)]
 108  pub enum ast_node {
 109      node_item(@item, @path),
 110      node_foreign_item(@foreign_item, AbiSet, visibility, @path),
 111      node_trait_method(@trait_method, DefId /* trait did */,
 112                        @path /* path to the trait */),
 113      node_method(@method, DefId /* impl did */, @path /* path to the impl */),
 114      node_variant(variant, @item, @path),
 115      node_expr(@Expr),
 116      node_stmt(@Stmt),
 117      node_arg(@Pat),
 118      node_local(Ident),
 119      node_block(Block),
 120      node_struct_ctor(@struct_def, @item, @path),
 121      node_callee_scope(@Expr)
 122  }

 124  impl ast_node {
 125      pub fn with_attrs<T>(&self, f: &fn(Option<&[Attribute]>) -> T) -> T {
 126          let attrs = match *self {
 127              node_item(i, _) => Some(i.attrs.as_slice()),
 128              node_foreign_item(fi, _, _, _) => Some(fi.attrs.as_slice()),
 129              node_trait_method(tm, _, _) => match *tm {
 130                  required(ref type_m) => Some(type_m.attrs.as_slice()),
 131                  provided(m) => Some(m.attrs.as_slice())
 132              },
 133              node_method(m, _, _) => Some(m.attrs.as_slice()),
 134              node_variant(ref v, _, _) => Some(v.node.attrs.as_slice()),
 135              // unit/tuple structs take the attributes straight from
 136              // the struct definition.
 137              node_struct_ctor(_, strct, _) => Some(strct.attrs.as_slice()),
 138              _ => None
 139          };
 140          f(attrs)
 141      }
 142  }

 152  impl Ctx {
 153      fn extend(&self, elt: path_elt) -> @path {
 154          @vec::append(self.path.clone(), [elt])
 155      }
 156  
 157      fn map_method(&mut self,
 158                    impl_did: DefId,
 159                    impl_path: @path,
 160                    m: @method,
 161                    is_provided: bool) {
 162          let entry = if is_provided {
 163              node_trait_method(@provided(m), impl_did, impl_path)
 164          } else {
 165              node_method(m, impl_did, impl_path)
 166          };
 167          self.map.insert(m.id, entry);
 168          self.map.insert(m.self_id, node_local(special_idents::self_));
 169      }
 170  
 171      fn map_struct_def(&mut self,
 172                        struct_def: @ast::struct_def,
 173                        parent_node: ast_node,
 174                        ident: ast::Ident) {
 175          let p = self.extend(path_name(ident));
 176  
 177          // If this is a tuple-like struct, register the constructor.
 178          match struct_def.ctor_id {
 179              None => {}
 180              Some(ctor_id) => {
 181                  match parent_node {
 182                      node_item(item, _) => {
 183                          self.map.insert(ctor_id,
 184                                          node_struct_ctor(struct_def,
 185                                                           item,
 186                                                           p));
 187                      }
 188                      _ => fail!("struct def parent wasn't an item")
 189                  }
 190              }
 191          }
 192      }

 194      fn map_expr(&mut self, ex: @Expr) {
 195          self.map.insert(ex.id, node_expr(ex));
 196  
 197          // Expressions which are or might be calls:
 198          {
 199              let r = ex.get_callee_id();
 200              for callee_id in r.iter() {
 201                  self.map.insert(*callee_id, node_callee_scope(ex));
 202              }
 203          }
 204  
 205          visit::walk_expr(self, ex, ());
 206      }
 207  
 208      fn map_fn(&mut self,
 209                fk: &visit::fn_kind,
 210                decl: &fn_decl,
 211                body: &Block,
 212                sp: codemap::Span,
 213                id: NodeId) {
 214          for a in decl.inputs.iter() {
 215              self.map.insert(a.id, node_arg(a.pat));
 216          }
 217          match *fk {
 218              visit::fk_method(name, _, _) => { self.path.push(path_name(name)) }
 219              _ => {}
 220          }
 221          visit::walk_fn(self, fk, decl, body, sp, id, ());
 222          match *fk {
 223              visit::fk_method(*) => { self.path.pop(); }
 224              _ => {}
 225          }
 226      }

 229          self.map.insert(stmt_id(stmt), node_stmt(stmt));
 230          visit::walk_stmt(self, stmt, ());
 231      }

 239      fn map_pat(&mut self, pat: @Pat) {
 240          match pat.node {
 241              PatIdent(_, ref path, _) => {
 242                  // Note: this is at least *potentially* a pattern...
 243                  self.map.insert(pat.id,
 244                                  node_local(ast_util::path_to_ident(path)));
 245              }
 246              _ => ()
 247          }
 248  
 249          visit::walk_pat(self, pat, ());
 250      }
 251  }

 253  impl Visitor<()> for Ctx {
 254      fn visit_item(&mut self, i: @item, _: ()) {
 255          // clone is FIXME #2543
 256          let item_path = @self.path.clone();
 257          self.map.insert(i.id, node_item(i, item_path));
 258          match i.node {
 259              item_impl(_, ref maybe_trait, ref ty, ref ms) => {
 260                  // Right now the ident on impls is __extensions__ which isn't
 261                  // very pretty when debugging, so attempt to select a better
 262                  // name to use.
 263                  let elt = impl_pretty_name(maybe_trait, ty, i.ident);
 264  
 265                  let impl_did = ast_util::local_def(i.id);
 266                  for m in ms.iter() {
 267                      let extended = { self.extend(elt) };
 268                      self.map_method(impl_did, extended, *m, false)
 269                  }
 270  
 271                  self.path.push(elt);
 272              }
 273              item_enum(ref enum_definition, _) => {
 274                  for v in (*enum_definition).variants.iter() {
 275                      let elt = path_name(i.ident);
 276                      // FIXME #2543: bad clone
 277                      self.map.insert(v.node.id,
 278                                      node_variant((*v).clone(),
 279                                                   i,
 280                                                   self.extend(elt)));
 281                  }
 282              }
 283              item_foreign_mod(ref nm) => {
 284                  for nitem in nm.items.iter() {
 285                      // Compute the visibility for this native item.
 286                      let visibility = match nitem.vis {
 287                          public => public,
 288                          private => private,
 289                          inherited => i.vis
 290                      };
 291  
 292                      self.map.insert(nitem.id,
 293                                      node_foreign_item(*nitem,
 294                                                        nm.abis,
 295                                                        visibility,
 296                                                        // FIXME (#2543)
 297                                                        if nm.sort ==
 298                                                              ast::named {
 299                                                            let e = path_name(
 300                                                                i.ident);
 301                                                            self.extend(e)
 302                                                        } else {
 303                                                          // Anonymous extern
 304                                                          // mods go in the
 305                                                          // parent scope.
 306                                                          @self.path.clone()
 307                                                        }));
 308                  }
 309              }
 310              item_struct(struct_def, _) => {
 311                  self.map_struct_def(struct_def,
 312                                      node_item(i, item_path),
 313                                      i.ident)
 314              }
 315              item_trait(_, ref traits, ref methods) => {
 316                  for p in traits.iter() {
 317                      self.map.insert(p.ref_id, node_item(i, item_path));
 318                  }
 319                  for tm in methods.iter() {
 320                      let ext = { self.extend(path_name(i.ident)) };
 321                      let d_id = ast_util::local_def(i.id);
 322                      match *tm {
 323                          required(ref m) => {
 324                              let entry =
 325                                  node_trait_method(@(*tm).clone(), d_id, ext);
 326                              self.map.insert(m.id, entry);
 327                          }
 328                          provided(m) => {
 329                              self.map_method(d_id, ext, m, true);
 330                          }
 331                      }
 332                  }
 333              }
 334              _ => {}
 335          }
 336  
 337          match i.node {
 338              item_mod(_) | item_foreign_mod(_) => {
 339                  self.path.push(path_mod(i.ident));
 340              }
 341              item_impl(*) => {} // this was guessed above.
 342              _ => self.path.push(path_name(i.ident))
 343          }
 344          visit::walk_item(self, i, ());
 345          self.path.pop();
 346      }

 390  // Used for items loaded from external crate that are being inlined into this
 391  // crate.  The `path` should be the path to the item but should not include
 392  // the item itself.
 393  pub fn map_decoded_item(diag: @mut span_handler,
 394                          map: map,
 395                          path: path,
 396                          ii: &inlined_item) {
 397      // I believe it is ok for the local IDs of inlined items from other crates
 398      // to overlap with the local ids from this crate, so just generate the ids
 399      // starting from 0.
 400      let cx = @mut Ctx {
 401          map: map,
 402          path: path.clone(),
 403          diag: diag,
 404      };
 405  
 406      // methods get added to the AST map when their impl is visited.  Since we
 407      // don't decode and instantiate the impl, but just the method, we have to
 408      // add it to the table now:
 409      match *ii {
 410          ii_item(*) => {} // fallthrough
 411          ii_foreign(i) => {
 412              cx.map.insert(i.id, node_foreign_item(i,
 413                                                    AbiSet::Intrinsic(),
 414                                                    i.vis,    // Wrong but OK
 415                                                    @path));
 416          }
 417          ii_method(impl_did, is_provided, m) => {
 418              cx.map_method(impl_did, @path, m, is_provided);
 419          }
 420      }
 421  
 422      // visit the item / method contents and add those to the map:
 423      ii.accept((), cx);
 424  }

 426  pub fn node_id_to_str(map: map, id: NodeId, itr: @ident_interner) -> ~str {
 427      match map.find(&id) {
 428        None => {
 429          fmt!("unknown node (id=%d)", id)
 430        }
 431        Some(&node_item(item, path)) => {
 432          let path_str = path_ident_to_str(path, item.ident, itr);
 433          let item_str = match item.node {
 434            item_static(*) => ~"static",
 435            item_fn(*) => ~"fn",
 436            item_mod(*) => ~"mod",
 437            item_foreign_mod(*) => ~"foreign mod",
 438            item_ty(*) => ~"ty",
 439            item_enum(*) => ~"enum",
 440            item_struct(*) => ~"struct",
 441            item_trait(*) => ~"trait",
 442            item_impl(*) => ~"impl",
 443            item_mac(*) => ~"macro"
 444          };
 445          fmt!("%s %s (id=%?)", item_str, path_str, id)
 446        }
 447        Some(&node_foreign_item(item, abi, _, path)) => {
 448          fmt!("foreign item %s with abi %? (id=%?)",
 449               path_ident_to_str(path, item.ident, itr), abi, id)
 450        }
 451        Some(&node_method(m, _, path)) => {
 452          fmt!("method %s in %s (id=%?)",
 453               itr.get(m.ident.name), path_to_str(*path, itr), id)
 454        }
 455        Some(&node_trait_method(ref tm, _, path)) => {
 456          let m = ast_util::trait_method_to_ty_method(&**tm);
 457          fmt!("method %s in %s (id=%?)",

 364  pub fn get_exprs_from_tts(cx: @ExtCtxt,
 365                            sp: span,
 366                            tts: &[ast::token_tree]) -> ~[@ast::expr] {
 367      let p = parse::new_parser_from_tts(cx.parse_sess(),
 368                                         cx.cfg(),
 369                                         tts.to_owned());
 370      let mut es = ~[];
 371      while *p.token != token::EOF {
 372          if es.len() != 0 && !p.eat(&token::COMMA) {
 373              cx.span_fatal(sp, "expected token: `,`");
 374          }
 375          es.push(p.parse_expr());
 376      }
 377      es
 378  }

 380  // in order to have some notion of scoping for macros,
 381  // we want to implement the notion of a transformation
 382  // environment.

 384  // This environment maps Names to Transformers.
 385  // Initially, this includes macro definitions and
 386  // block directives.
 387  
 388  
 389  
 390  // Actually, the following implementation is parameterized
 391  // by both key and value types.
 392  
 393  //impl question: how to implement it? Initially, the
 394  // env will contain only macros, so it might be painful
 395  // to add an empty frame for every context. Let's just
 396  // get it working, first....
 397  
 398  // NB! the mutability of the underlying maps means that
 399  // if expansion is out-of-order, a deeper scope may be
 400  // able to refer to a macro that was added to an enclosing
 401  // scope lexically later than the deeper scope.
 402  
 403  // Note on choice of representation: I've been pushed to
 404  // use a top-level managed pointer by some difficulties
 405  // with pushing and popping functionally, and the ownership
 406  // issues.  As a result, the values returned by the table
 407  // also need to be managed; the &'self ... type that Maps
 408  // return won't work for things that need to get outside
 409  // of that managed pointer.  The easiest way to do this
 410  // is just to insist that the values in the tables are
 411  // managed to begin with.

 413  // a transformer env is either a base map or a map on top
 414  // of another chain.
 415  pub enum MapChain<K,V> {
 416      BaseMapChain(~HashMap<K,@V>),
 417      ConsMapChain(~HashMap<K,@V>,@mut MapChain<K,V>)
 418  }
 419  
 420  
 421  // get the map from an env frame
 422  impl <K: Eq + Hash + IterBytes + 'static, V: 'static> MapChain<K,V>{
 423      // Constructor. I don't think we need a zero-arg one.
 424      fn new(init: ~HashMap<K,@V>) -> @mut MapChain<K,V> {
 425          @mut BaseMapChain(init)
 426      }
 427  
 428      // add a new frame to the environment (functionally)
 429      fn push_frame (@mut self) -> @mut MapChain<K,V> {
 430          @mut ConsMapChain(~HashMap::new() ,self)
 431      }

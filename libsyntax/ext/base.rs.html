 438  /// Extract a string literal from the macro expanded version of `expr`,
 439  /// emitting `err_msg` if `expr` is not a string literal. This does not stop
 440  /// compilation on error, merely emits a non-fatal error and returns None.
 441  pub fn expr_to_str(cx: &mut ExtCtxt, expr: @ast::Expr, err_msg: &str)
 442                     -> Option<(InternedString, ast::StrStyle)> {
 443      // we want to be able to handle e.g. concat("foo", "bar")
 444      let expr = cx.expand_expr(expr);
 445      match expr.node {

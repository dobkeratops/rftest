 163  // The SyntaxEnv is the environment that's threaded through the expansion
 164  // of macros. It contains bindings for macros, and also a special binding

 430  pub fn get_single_str_from_tts(cx: @ExtCtxt,
 431                                 sp: Span,
 432                                 tts: &[ast::token_tree],
 433                                 name: &str)
 434                                 -> @str {
 435      if tts.len() != 1 {
 436          cx.span_fatal(sp, fmt!("%s takes 1 argument.", name));
 437      }
 438  
 439      match tts[0] {
 440          ast::tt_tok(_, token::LIT_STR(ident)) => cx.str_of(ident),

 442      }

 445  pub fn get_exprs_from_tts(cx: @ExtCtxt,
 446                            sp: Span,
 447                            tts: &[ast::token_tree]) -> ~[@ast::Expr] {
 448      let p = parse::new_parser_from_tts(cx.parse_sess(),
 449                                         cx.cfg(),
 450                                         tts.to_owned());
 451      let mut es = ~[];

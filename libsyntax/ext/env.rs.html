 45      let msg = match exprs.len() {

 47          2 => expr_to_str(ext_cx, exprs[1], "expected string literal"),
 48          _ => ext_cx.span_fatal(sp, "env! takes 1 or 2 arguments")
 49      };
 50  
 51      let e = match os::getenv(var) {
 52          None => ext_cx.span_fatal(sp, msg),

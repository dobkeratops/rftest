  53  impl Context {
  54      /// Parses the arguments from the given list of tokens, returning None if
  55      /// there's a parse error so we can continue parsing other fmt! expressions.
  56      fn parse_args(&mut self, sp: span,
  57                    tts: &[ast::token_tree]) -> Option<@ast::expr> {
  58          let p = rsparse::new_parser_from_tts(self.ecx.parse_sess(),
  59                                               self.ecx.cfg(),
  60                                               tts.to_owned());
  61          if *p.token == token::EOF {
  62              self.ecx.span_err(sp, "ifmt! expects at least one argument");
  63              return None;
  64          }
  65          let fmtstr = p.parse_expr();
  66          let mut named = false;
  67          while *p.token != token::EOF {
  68              if !p.eat(&token::COMMA) {
  69                  self.ecx.span_err(sp, "expected token: `,`");
  70                  return None;
  71              }
  72              if named || (token::is_ident(p.token) &&
  73                           p.look_ahead(1, |t| *t == token::EQ)) {
  74                  named = true;
  75                  let ident = match *p.token {
  76                      token::IDENT(i, _) => {
  77                          p.bump();
  78                          i
  79                      }
  80                      _ if named => {
  81                          self.ecx.span_err(*p.span,
  82                                            "expected ident, positional arguments \
  83                                             cannot follow named arguments");
  84                          return None;
  85                      }
  86                      _ => {
  87                          self.ecx.span_err(*p.span,
  88                                            fmt!("expected ident for named \
  89                                                  argument, but found `%s`",
  90                                                 p.this_token_to_str()));
  91                          return None;
  92                      }
  93                  };
  94                  let name = self.ecx.str_of(ident);
  95                  p.expect(&token::EQ);
  96                  let e = p.parse_expr();
  97                  match self.names.find(&name) {
  98                      None => {}
  99                      Some(prev) => {
 100                          self.ecx.span_err(e.span, fmt!("duplicate argument \
 101                                                          named `%s`", name));
 102                          self.ecx.parse_sess.span_diagnostic.span_note(
 103                              prev.span, "previously here");
 104                          loop
 105                      }
 106                  }
 107                  self.names.insert(name, e);
 108              } else {
 109                  self.args.push(p.parse_expr());
 110                  self.arg_types.push(None);
 111              }
 112          }
 113          return Some(fmtstr);
 114      }

 116      /// Verifies one piece of a parse string. All errors are not emitted as
 117      /// fatal so we can continue giving errors about this and possibly other
 118      /// format strings.
 119      fn verify_piece(&mut self, p: &parse::Piece) {
 120          match *p {
 121              parse::String(*) => {}
 122              parse::CurrentArgument => {
 123                  if self.nest_level == 0 {
 124                      self.ecx.span_err(self.fmtsp,
 125                                        "`#` reference used with nothing to \
 126                                         reference back to");
 127                  }
 128              }
 129              parse::Argument(ref arg) => {
 130                  // width/precision first, if they have implicit positional
 131                  // parameters it makes more sense to consume them first.
 132                  self.verify_count(arg.format.width);
 133                  self.verify_count(arg.format.precision);
 134  
 135                  // argument second, if it's an implicit positional parameter
 136                  // it's written second, so it should come after width/precision.
 137                  let pos = match arg.position {
 138                      parse::ArgumentNext => {
 139                          let i = self.next_arg;
 140                          if self.check_positional_ok() {
 141                              self.next_arg += 1;
 142                          }
 143                          Left(i)
 144                      }
 145                      parse::ArgumentIs(i) => Left(i),
 146                      parse::ArgumentNamed(s) => Right(s.to_managed()),
 147                  };
 148                  let ty = if arg.format.ty == "" {
 149                      Unknown
 150                  } else { Known(arg.format.ty.to_managed()) };
 151                  self.verify_arg_type(pos, ty);
 152  
 153                  // and finally the method being applied
 154                  match arg.method {
 155                      None => {}
 156                      Some(ref method) => { self.verify_method(pos, *method); }
 157                  }
 158              }
 159          }
 160      }

 162      fn verify_pieces(&mut self, pieces: &[parse::Piece]) {
 163          for piece in pieces.iter() {
 164              self.verify_piece(piece);
 165          }
 166      }

 168      fn verify_count(&mut self, c: parse::Count) {
 169          match c {
 170              parse::CountImplied | parse::CountIs(*) => {}
 171              parse::CountIsParam(i) => {
 172                  self.verify_arg_type(Left(i), Unsigned);
 173              }
 174              parse::CountIsNextParam => {
 175                  if self.check_positional_ok() {
 176                      self.verify_arg_type(Left(self.next_arg), Unsigned);
 177                      self.next_arg += 1;
 178                  }
 179              }
 180          }
 181      }

 183      fn check_positional_ok(&mut self) -> bool {
 184          if self.nest_level != 0 {
 185              self.ecx.span_err(self.fmtsp, "cannot use implicit positional \
 186                                             arguments nested inside methods");
 187              false
 188          } else {
 189              true
 190          }
 191      }

 193      fn verify_method(&mut self, pos: Either<uint, @str>, m: &parse::Method) {
 194          self.nest_level += 1;
 195          match *m {
 196              parse::Plural(_, ref arms, ref default) => {
 197                  let mut seen_cases = HashSet::new();
 198                  self.verify_arg_type(pos, Unsigned);
 199                  for arm in arms.iter() {
 200                      if !seen_cases.insert(arm.selector) {
 201                          match arm.selector {
 202                              Left(name) => {
 203                                  self.ecx.span_err(self.fmtsp,
 204                                                    fmt!("duplicate selector \
 205                                                         `%?`", name));
 206                              }
 207                              Right(idx) => {
 208                                  self.ecx.span_err(self.fmtsp,
 209                                                    fmt!("duplicate selector \
 210                                                         `=%u`", idx));
 211                              }
 212                          }
 213                      }
 214                      self.verify_pieces(arm.result);
 215                  }
 216                  self.verify_pieces(*default);
 217              }
 218              parse::Select(ref arms, ref default) => {
 219                  self.verify_arg_type(pos, String);
 220                  let mut seen_cases = HashSet::new();
 221                  for arm in arms.iter() {
 222                      if !seen_cases.insert(arm.selector) {
 223                          self.ecx.span_err(self.fmtsp,
 224                                            fmt!("duplicate selector `%s`",
 225                                                 arm.selector));
 226                      } else if arm.selector == "" {
 227                          self.ecx.span_err(self.fmtsp,
 228                                            "empty selector in `select`");
 229                      }
 230                      self.verify_pieces(arm.result);
 231                  }
 232                  self.verify_pieces(*default);
 233              }
 234          }
 235          self.nest_level -= 1;
 236      }

 238      fn verify_arg_type(&mut self, arg: Either<uint, @str>, ty: ArgumentType) {
 239          match arg {
 240              Left(arg) => {
 241                  if arg < 0 || self.args.len() <= arg {
 242                      let msg = fmt!("invalid reference to argument `%u` (there \
 243                                      are %u arguments)", arg, self.args.len());
 244                      self.ecx.span_err(self.fmtsp, msg);
 245                      return;
 246                  }
 247                  self.verify_same(self.args[arg].span, ty, self.arg_types[arg]);
 248                  if ty != Unknown || self.arg_types[arg].is_none() {
 249                      self.arg_types[arg] = Some(ty);
 250                  }
 251              }
 252  
 253              Right(name) => {
 254                  let span = match self.names.find(&name) {
 255                      Some(e) => e.span,
 256                      None => {
 257                          let msg = fmt!("There is no argument named `%s`", name);
 258                          self.ecx.span_err(self.fmtsp, msg);
 259                          return;
 260                      }
 261                  };
 262                  self.verify_same(span, ty,
 263                                   self.name_types.find(&name).map(|&x| *x));
 264                  if ty != Unknown || !self.name_types.contains_key(&name) {
 265                      self.name_types.insert(name, ty);
 266                  }
 267                  // Assign this named argument a slot in the arguments array if
 268                  // it hasn't already been assigned a slot.
 269                  if !self.name_positions.contains_key(&name) {
 270                      let slot = self.name_positions.len();
 271                      self.name_positions.insert(name, slot);
 272                  }
 273              }
 274          }
 275      }

 277      /// When we're keeping track of the types that are declared for certain
 278      /// arguments, we assume that `None` means we haven't seen this argument
 279      /// yet, `Some(None)` means that we've seen the argument, but no format was
 280      /// specified, and `Some(Some(x))` means that the argument was declared to
 281      /// have type `x`.
 282      ///
 283      /// Obviously `Some(Some(x)) != Some(Some(y))`, but we consider it true
 284      /// that: `Some(None) == Some(Some(x))`
 285      fn verify_same(&self, sp: span, ty: ArgumentType,
 286                     before: Option<ArgumentType>) {
 287          if ty == Unknown { return }
 288          let cur = match before {
 289              Some(Unknown) | None => return,
 290              Some(t) => t,
 291          };
 292          if ty == cur { return }
 293          match (cur, ty) {
 294              (Known(cur), Known(ty)) => {
 295                  self.ecx.span_err(sp,
 296                                    fmt!("argument redeclared with type `%s` when \
 297                                          it was previously `%s`", ty, cur));
 298              }
 299              (Known(cur), _) => {
 300                  self.ecx.span_err(sp,
 301                                    fmt!("argument used to format with `%s` was \
 302                                          attempted to not be used for formatting",
 303                                          cur));
 304              }
 305              (_, Known(ty)) => {
 306                  self.ecx.span_err(sp,
 307                                    fmt!("argument previously used as a format \
 308                                          argument attempted to be used as `%s`",
 309                                          ty));
 310              }
 311              (_, _) => {
 312                  self.ecx.span_err(sp, "argument declared with multiple formats");
 313              }
 314          }
 315      }

 317      /// Translate a `parse::Piece` to a static `rt::Piece`
 318      fn trans_piece(&mut self, piece: &parse::Piece) -> @ast::expr {
 319          let sp = self.fmtsp;
 320          let parsepath = |s: &str| {
 321              ~[self.ecx.ident_of("std"), self.ecx.ident_of("fmt"),
 322                self.ecx.ident_of("parse"), self.ecx.ident_of(s)]
 323          };
 324          let rtpath = |s: &str| {
 325              ~[self.ecx.ident_of("std"), self.ecx.ident_of("fmt"),
 326                self.ecx.ident_of("rt"), self.ecx.ident_of(s)]
 327          };
 328          let ctpath = |s: &str| {
 329              ~[self.ecx.ident_of("std"), self.ecx.ident_of("fmt"),
 330                self.ecx.ident_of("parse"), self.ecx.ident_of(s)]
 331          };
 332          let none = || {
 333              let p = self.ecx.path(sp, ~[self.ecx.ident_of("None")]);
 334              self.ecx.expr_path(p)
 335          };
 336          let some = |e: @ast::expr| {
 337              self.ecx.expr_call_ident(sp, self.ecx.ident_of("Some"), ~[e])
 338          };
 339          let trans_count = |c: parse::Count| {
 340              match c {
 341                  parse::CountIs(i) => {
 342                      self.ecx.expr_call_global(sp, ctpath("CountIs"),
 343                                                ~[self.ecx.expr_uint(sp, i)])
 344                  }
 345                  parse::CountIsParam(i) => {
 346                      self.ecx.expr_call_global(sp, ctpath("CountIsParam"),
 347                                                ~[self.ecx.expr_uint(sp, i)])
 348                  }
 349                  parse::CountImplied => {
 350                      let path = self.ecx.path_global(sp, ctpath("CountImplied"));
 351                      self.ecx.expr_path(path)
 352                  }
 353                  parse::CountIsNextParam => {
 354                      let path = self.ecx.path_global(sp, ctpath("CountIsNextParam"));
 355                      self.ecx.expr_path(path)
 356                  }
 357              }
 358          };
 359          let trans_method = |method: &parse::Method| {
 360              let method = match *method {
 361                  parse::Select(ref arms, ref default) => {
 362                      let arms = arms.iter().map(|arm| {
 363                          let p = self.ecx.path_global(sp, rtpath("SelectArm"));
 364                          let result = arm.result.iter().map(|p| {
 365                              self.trans_piece(p)
 366                          }).collect();
 367                          let s = arm.selector.to_managed();
 368                          let selector = self.ecx.expr_str(sp, s);
 369                          self.ecx.expr_struct(sp, p, ~[
 370                              self.ecx.field_imm(sp,
 371                                                 self.ecx.ident_of("selector"),
 372                                                 selector),
 373                              self.ecx.field_imm(sp, self.ecx.ident_of("result"),
 374                                                 self.ecx.expr_vec_slice(sp, result)),
 375                          ])
 376                      }).collect();
 377                      let default = default.iter().map(|p| {
 378                          self.trans_piece(p)
 379                      }).collect();
 380                      self.ecx.expr_call_global(sp, rtpath("Select"), ~[
 381                          self.ecx.expr_vec_slice(sp, arms),
 382                          self.ecx.expr_vec_slice(sp, default),
 383                      ])
 384                  }
 385                  parse::Plural(offset, ref arms, ref default) => {
 386                      let offset = match offset {
 387                          Some(i) => { some(self.ecx.expr_uint(sp, i)) }
 388                          None => { none() }
 389                      };
 390                      let arms = arms.iter().map(|arm| {
 391                          let p = self.ecx.path_global(sp, rtpath("PluralArm"));
 392                          let result = arm.result.iter().map(|p| {
 393                              self.trans_piece(p)
 394                          }).collect();
 395                          let (lr, selarg) = match arm.selector {
 396                              Left(t) => {
 397                                  let p = ctpath(fmt!("%?", t));
 398                                  let p = self.ecx.path_global(sp, p);
 399                                  (self.ecx.ident_of("Left"),
 400                                   self.ecx.expr_path(p))
 401                              }
 402                              Right(i) => {
 403                                  (self.ecx.ident_of("Right"),
 404                                   self.ecx.expr_uint(sp, i))
 405                              }
 406                          };
 407                          let selector = self.ecx.expr_call_ident(sp,
 408                                  lr, ~[selarg]);
 409                          self.ecx.expr_struct(sp, p, ~[
 410                              self.ecx.field_imm(sp,
 411                                                 self.ecx.ident_of("selector"),
 412                                                 selector),
 413                              self.ecx.field_imm(sp, self.ecx.ident_of("result"),
 414                                                 self.ecx.expr_vec_slice(sp, result)),
 415                          ])
 416                      }).collect();
 417                      let default = default.iter().map(|p| {
 418                          self.trans_piece(p)
 419                      }).collect();
 420                      self.ecx.expr_call_global(sp, rtpath("Plural"), ~[
 421                          offset,
 422                          self.ecx.expr_vec_slice(sp, arms),
 423                          self.ecx.expr_vec_slice(sp, default),
 424                      ])
 425                  }
 426              };
 427              let life = self.ecx.lifetime(sp, self.ecx.ident_of("static"));
 428              let ty = self.ecx.ty_path(self.ecx.path_all(
 429                  sp,
 430                  true,
 431                  rtpath("Method"),
 432                  Some(life),
 433                  ~[]
 434              ), None);
 435              let st = ast::item_static(ty, ast::m_imm, method);
 436              let static_name = self.ecx.ident_of(fmt!("__static_method_%u",
 437                                                       self.method_statics.len()));
 438              // Flag these statics as `address_insignificant` so LLVM can
 439              // merge duplicate globals as much as possible (which we're

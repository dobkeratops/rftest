  36  impl<'a> ParserAnyMacro<'a> {
  37      /// Make sure we don't have any tokens left to parse, so we don't
  38      /// silently drop anything. `allow_semi` is so that "optional"
  39      /// semilons at the end of normal expressions aren't complained
  40      /// about e.g. the semicolon in `macro_rules! kapow( () => {
  41      /// fail!(); } )` doesn't get picked up by .parse_expr(), but it's
  42      /// allowed to be there.
  43      fn ensure_complete_parse(&self, allow_semi: bool) {
  44          let mut parser = self.parser.borrow_mut();
  45          if allow_semi && parser.token == SEMI {
  46              parser.bump()
  47          }
  48          if parser.token != EOF {
  49              let token_str = parser.this_token_to_str();
  50              let msg = format!("macro expansion ignores token `{}` and any \
  51                                 following",
  52                                token_str);
  53              let span = parser.span;
  54              parser.span_err(span, msg);
  55          }
  56      }
  57  }

  60      fn make_expr(&self) -> @ast::Expr {

  65      fn make_items(&self) -> SmallVector<@ast::Item> {
  66          let mut ret = SmallVector::zero();
  67          loop {
  68              let mut parser = self.parser.borrow_mut();
  69              let attrs = parser.parse_outer_attributes();
  70              match parser.parse_item(attrs) {
  71                  Some(item) => ret.push(item),
  72                  None => break
  73              }
  74          }
  75          self.ensure_complete_parse(false);
  76          ret
  77      }

  79          let attrs = self.parser.borrow_mut().parse_outer_attributes();
  80          let ret = self.parser.borrow_mut().parse_stmt(attrs);
  81          self.ensure_complete_parse(true);

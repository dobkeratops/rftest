  37  impl<'a> ParserAnyMacro<'a> {
  38      /// Make sure we don't have any tokens left to parse, so we don't
  39      /// silently drop anything. `allow_semi` is so that "optional"
  40      /// semilons at the end of normal expressions aren't complained
  41      /// about e.g. the semicolon in `macro_rules! kapow( () => {
  42      /// fail!(); } )` doesn't get picked up by .parse_expr(), but it's
  43      /// allowed to be there.
  44      fn ensure_complete_parse(&self, allow_semi: bool) {
  45          let mut parser = self.parser.borrow_mut();
  46          if allow_semi && parser.token == SEMI {
  47              parser.bump()
  48          }
  49          if parser.token != EOF {
  50              let token_str = parser.this_token_to_str();
  51              let msg = format!("macro expansion ignores token `{}` and any \
  52                                 following",
  53                                token_str);
  54              let span = parser.span;
  55              parser.span_err(span, msg);
  56          }
  57      }
  58  }

  61      fn make_expr(&self) -> @ast::Expr {
  62          let ret = self.parser.borrow_mut().parse_expr();
  63          self.ensure_complete_parse(true);
  64          ret
  65      }
  66      fn make_items(&self) -> SmallVector<@ast::Item> {
  67          let mut ret = SmallVector::zero();
  68          loop {
  69              let mut parser = self.parser.borrow_mut();
  70              let attrs = parser.parse_outer_attributes();
  71              match parser.parse_item(attrs) {
  72                  Some(item) => ret.push(item),
  73                  None => break
  74              }
  75          }
  76          self.ensure_complete_parse(false);
  77          ret
  78      }
  79      fn make_stmt(&self) -> @ast::Stmt {
  80          let attrs = self.parser.borrow_mut().parse_outer_attributes();
  81          let ret = self.parser.borrow_mut().parse_stmt(attrs);
  82          self.ensure_complete_parse(true);
  83          ret
  84      }
  85  }

 109  fn generic_extension(cx: &ExtCtxt,
 110                       sp: Span,
 111                       name: Ident,
 112                       arg: &[ast::TokenTree],
 113                       lhses: &[Rc<NamedMatch>],
 114                       rhses: &[Rc<NamedMatch>])
 115                       -> MacResult {

 117          println!("{}! \\{ {} \\}",
 118                   token::get_ident(name),
 119                   print::pprust::tt_to_str(&TTDelim(Rc::new(arg.iter()
 120                                                                .map(|x| (*x).clone())
 121                                                                .collect()))));

 123  
 124      // Which arm's failure should we report? (the one furthest along)
 125      let mut best_fail_spot = DUMMY_SP;
 126      let mut best_fail_msg = ~"internal error: ran no matchers";
 127  
 128      for (i, lhs) in lhses.iter().enumerate() { // try each arm's matchers
 129          match **lhs {

 131              // `None` is because we're not interpolating

 133                                          None,
 134                                          arg.iter()
 135                                             .map(|x| (*x).clone())
 136                                             .collect());

 138                Success(named_matches) => {
 139                  let rhs = match *rhses[i] {
 140                      // okay, what's your transcriber?

 142                          match *tt {

 144                              TTDelim(ref tts) => {

 146                                        .iter()
 147                                        .map(|x| (*x).clone())
 148                                        .collect()
 149                              }
 150                              _ => cx.span_fatal(
 151                                  sp, "macro rhs must be delimited")
 152                          }
 153                      },
 154                      _ => cx.span_bug(sp, "bad thing in rhs")
 155                  };

 157                  let trncbr = new_tt_reader(&cx.parse_sess().span_diagnostic,
 158                                             Some(named_matches),
 159                                             rhs);
 160                  let p = Parser(cx.parse_sess(), cx.cfg(), ~trncbr);

 162                  // Weird, but useful for X-macros.
 163                  return MRAny(~ParserAnyMacro {
 164                      parser: RefCell::new(p),
 165                  })
 166                }

 168                  best_fail_spot = sp;

 173            }
 174            _ => cx.bug("non-matcher found in parsed lhses")

 200      // The grammar for macro_rules! is:

 231      };

 234          name: name,
 235          lhses: lhses,
 236          rhses: rhses,
 237      };

 240          name: token::get_ident(name).to_str(),

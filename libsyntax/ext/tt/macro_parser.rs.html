  26  /* This is an Earley-like parser, without support for in-grammar nonterminals,
  27  only by calling out to the main rust parser for named nonterminals (which it
  28  commits to fully when it hits one in a grammar). This means that there are no
  29  completer or predictor rules, and therefore no need to store one column per
  30  token: instead, there's a set of current Earley items and a set of next
  31  ones. Instead of NTs, we have a special case for Kleene star. The big-O, in
  32  pathological cases, is worse than traditional Earley parsing, but it's an
  33  easier fit for Macro-by-Example-style rules, and I think the overhead is
  34  lower. (In order to prevent the pathological case, we'd need to lazily
  35  construct the resulting `NamedMatch`es at the very end. It'd be a pain,
  36  and require more memory to keep around old items, but it would also save
  37  overhead)*/

 102      sep: Option<Token>,

 104      up: Option<~MatcherPos>,

 106      match_lo: uint, match_hi: uint,

 110  pub fn count_names(ms: &[Matcher]) -> uint {
 111      ms.iter().fold(0, |ct, m| {
 112          ct + match m.node {
 113              MatchTok(_) => 0u,
 114              MatchSeq(ref more_ms, _, _, _, _) => {
 115                  count_names(more_ms.as_slice())
 116              }
 117              MatchNonterminal(_, _, _) => 1u
 118          }})
 119  }

 121  pub fn initial_matcher_pos(ms: Vec<Matcher> , sep: Option<Token>, lo: BytePos)
 122                          -> ~MatcherPos {
 123      let mut match_idx_hi = 0u;
 124      for elt in ms.iter() {
 125          match elt.node {
 126              MatchTok(_) => (),
 127              MatchSeq(_,_,_,_,hi) => {
 128                  match_idx_hi = hi;       // it is monotonic...
 129              }
 130              MatchNonterminal(_,_,pos) => {
 131                  match_idx_hi = pos+1u;  // ...so latest is highest
 132              }
 133          }
 134      }
 135      let matches = Vec::from_fn(count_names(ms.as_slice()), |_i| Vec::new());
 136      ~MatcherPos {
 137          elts: ms,
 138          sep: sep,
 139          idx: 0u,
 140          up: None,
 141          matches: matches,
 142          match_lo: 0u,
 143          match_hi: match_idx_hi,
 144          sp_lo: lo
 145      }
 146  }

 148  // NamedMatch is a pattern-match result for a single ast::MatchNonterminal:
 149  // so it is associated with a single ident in a parse, and all
 150  // MatchedNonterminal's in the NamedMatch have the same nonterminal type
 151  // (expr, item, etc). All the leaves in a single NamedMatch correspond to a
 152  // single matcher_nonterminal in the ast::Matcher that produced it.
 153  //
 154  // It should probably be renamed, it has more or less exact correspondence to
 155  // ast::match nodes, and the in-memory structure of a particular NamedMatch
 156  // represents the match that occurred when a particular subset of an
 157  // ast::match -- those ast::Matcher nodes leading to a single
 158  // MatchNonterminal -- was applied to a particular token tree.
 159  //
 160  // The width of each MatchedSeq in the NamedMatch, and the identity of the
 161  // MatchedNonterminal's, will depend on the token tree it was applied to: each
 162  // MatchedSeq corresponds to a single MatchSeq in the originating
 163  // ast::Matcher. The depth of the NamedMatch structure will therefore depend
 164  // only on the nesting depth of ast::MatchSeq's in the originating
 165  // ast::Matcher it was derived from.
 166  
 167  pub enum NamedMatch {
 168      MatchedSeq(Vec<@NamedMatch> , codemap::Span),
 169      MatchedNonterminal(Nonterminal)
 170  }

 172  pub fn nameize(p_s: &ParseSess, ms: &[Matcher], res: &[@NamedMatch])
 173              -> HashMap<Ident, @NamedMatch> {
 174      fn n_rec(p_s: &ParseSess, m: &Matcher, res: &[@NamedMatch],
 175               ret_val: &mut HashMap<Ident, @NamedMatch>) {
 176          match *m {
 177            codemap::Spanned {node: MatchTok(_), .. } => (),
 178            codemap::Spanned {node: MatchSeq(ref more_ms, _, _, _, _), .. } => {
 179              for next_m in more_ms.iter() {
 180                  n_rec(p_s, next_m, res, ret_val)
 181              };
 182            }
 183            codemap::Spanned {
 184                  node: MatchNonterminal(bind_name, _, idx),
 185                  span
 186            } => {
 187              if ret_val.contains_key(&bind_name) {
 188                  let string = token::get_ident(bind_name);
 189                  p_s.span_diagnostic
 190                     .span_fatal(span, "duplicated bind name: " + string.get())
 191              }
 192              ret_val.insert(bind_name, res[idx]);
 193            }
 194          }
 195      }
 196      let mut ret_val = HashMap::new();
 197      for m in ms.iter() { n_rec(p_s, m, res, &mut ret_val) }
 198      ret_val
 199  }

 219  // perform a token equality check, ignoring syntax context (that is, an unhygienic comparison)
 220  pub fn token_name_eq(t1 : &Token, t2 : &Token) -> bool {
 221      match (t1,t2) {
 222          (&token::IDENT(id1,_),&token::IDENT(id2,_))
 223          | (&token::LIFETIME(id1),&token::LIFETIME(id2)) =>
 224              id1.name == id2.name,
 225          _ => *t1 == *t2
 226      }
 227  }

 229  pub fn parse<R: Reader>(sess: &ParseSess,
 230                          cfg: ast::CrateConfig,
 231                          rdr: R,
 232                          ms: &[Matcher])
 233                          -> ParseResult {
 234      let mut cur_eis = Vec::new();
 235      cur_eis.push(initial_matcher_pos(ms.iter()
 236                                         .map(|x| (*x).clone())
 237                                         .collect(),
 238                                       None,
 239                                       rdr.peek().sp.lo));
 240  
 241      loop {
 242          let mut bb_eis = Vec::new(); // black-box parsed by parser.rs
 243          let mut next_eis = Vec::new(); // or proceed normally
 244          let mut eof_eis = Vec::new();
 245  
 246          let TokenAndSpan {tok: tok, sp: sp} = rdr.peek();
 247  
 248          /* we append new items to this while we go */
 249          loop {
 250              let ei = match cur_eis.pop() {
 251                  None => break, /* for each Earley Item */
 252                  Some(ei) => ei,
 253              };
 254  
 255              let idx = ei.idx;
 256              let len = ei.elts.len();
 257  
 258              /* at end of sequence */
 259              if idx >= len {
 260                  // can't move out of `match`es, so:
 261                  if ei.up.is_some() {
 262                      // hack: a matcher sequence is repeating iff it has a
 263                      // parent (the top level is just a container)
 264  
 265  
 266                      // disregard separator, try to go up
 267                      // (remove this condition to make trailing seps ok)
 268                      if idx == len {
 269                          // pop from the matcher position
 270  
 271                          let mut new_pos = ei.up.clone().unwrap();
 272  
 273                          // update matches (the MBE "parse tree") by appending
 274                          // each tree as a subtree.
 275  
 276                          // I bet this is a perf problem: we're preemptively
 277                          // doing a lot of array work that will get thrown away
 278                          // most of the time.
 279  
 280                          // Only touch the binders we have actually bound
 281                          for idx in range(ei.match_lo, ei.match_hi) {
 282                              let sub = (*ei.matches.get(idx)).clone();
 283                              new_pos.matches
 284                                     .get_mut(idx)
 285                                     .push(@MatchedSeq(sub, mk_sp(ei.sp_lo,
 286                                                                  sp.hi)));
 287                          }
 288  
 289                          new_pos.idx += 1;
 290                          cur_eis.push(new_pos);
 291                      }
 292  
 293                      // can we go around again?
 294  
 295                      // the *_t vars are workarounds for the lack of unary move
 296                      match ei.sep {
 297                        Some(ref t) if idx == len => { // we need a separator
 298                          // i'm conflicted about whether this should be hygienic....
 299                          // though in this case, if the separators are never legal
 300                          // idents, it shouldn't matter.
 301                          if token_name_eq(&tok, t) { //pass the separator
 302                              let mut ei_t = ei.clone();
 303                              ei_t.idx += 1;
 304                              next_eis.push(ei_t);
 305                          }
 306                        }
 307                        _ => { // we don't need a separator
 308                          let mut ei_t = ei;
 309                          ei_t.idx = 0;
 310                          cur_eis.push(ei_t);
 311                        }
 312                      }
 313                  } else {
 314                      eof_eis.push(ei);
 315                  }
 316              } else {
 317                  match ei.elts.get(idx).node.clone() {
 318                    /* need to descend into sequence */
 319                    MatchSeq(ref matchers, ref sep, zero_ok,
 320                             match_idx_lo, match_idx_hi) => {
 321                      if zero_ok {
 322                          let mut new_ei = ei.clone();
 323                          new_ei.idx += 1u;
 324                          //we specifically matched zero repeats.
 325                          for idx in range(match_idx_lo, match_idx_hi) {
 326                              new_ei.matches
 327                                    .get_mut(idx)
 328                                    .push(@MatchedSeq(Vec::new(), sp));
 329                          }
 330  
 331                          cur_eis.push(new_ei);
 332                      }
 333  
 334                      let matches = Vec::from_elem(ei.matches.len(), Vec::new());
 335                      let ei_t = ei;
 336                      cur_eis.push(~MatcherPos {
 337                          elts: (*matchers).clone(),
 338                          sep: (*sep).clone(),
 339                          idx: 0u,
 340                          up: Some(ei_t),
 341                          matches: matches,
 342                          match_lo: match_idx_lo, match_hi: match_idx_hi,
 343                          sp_lo: sp.lo
 344                      });
 345                    }
 346                    MatchNonterminal(_,_,_) => { bb_eis.push(ei) }
 347                    MatchTok(ref t) => {
 348                      let mut ei_t = ei.clone();
 349                      //if (token_name_eq(t,&tok)) {
 350                      if token::mtwt_token_eq(t,&tok) {
 351                          ei_t.idx += 1;
 352                          next_eis.push(ei_t);
 353                      }
 354                    }
 355                  }
 356              }
 357          }
 358  
 359          /* error messages here could be improved with links to orig. rules */
 360          if token_name_eq(&tok, &EOF) {
 361              if eof_eis.len() == 1u {
 362                  let mut v = Vec::new();
 363                  for dv in eof_eis.get_mut(0).matches.mut_iter() {
 364                      v.push(dv.pop().unwrap());
 365                  }
 366                  return Success(nameize(sess, ms, v.as_slice()));
 367              } else if eof_eis.len() > 1u {
 368                  return Error(sp, ~"ambiguity: multiple successful parses");
 369              } else {
 370                  return Failure(sp, ~"unexpected end of macro invocation");
 371              }
 372          } else {
 373              if (bb_eis.len() > 0u && next_eis.len() > 0u)
 374                  || bb_eis.len() > 1u {
 375                  let nts = bb_eis.map(|ei| {
 376                      match ei.elts.get(ei.idx).node {
 377                        MatchNonterminal(bind, name, _) => {
 378                          format!("{} ('{}')",
 379                                  token::get_ident(name),
 380                                  token::get_ident(bind))
 381                        }
 382                        _ => fail!()
 383                      } }).connect(" or ");
 384                  return Error(sp, format!(
 385                      "local ambiguity: multiple parsing options: \
 386                       built-in NTs {} or {} other options.",
 387                      nts, next_eis.len()));
 388              } else if bb_eis.len() == 0u && next_eis.len() == 0u {
 389                  return Failure(sp, format!("no rules expected the token `{}`",
 390                              token::to_str(&tok)));
 391              } else if next_eis.len() > 0u {
 392                  /* Now process the next token */
 393                  while next_eis.len() > 0u {
 394                      cur_eis.push(next_eis.pop().unwrap());
 395                  }
 396                  rdr.next_token();
 397              } else /* bb_eis.len() == 1 */ {
 398                  let mut rust_parser = Parser(sess, cfg.clone(), rdr.dup());
 399

  27  /* This is an Earley-like parser, without support for in-grammar nonterminals,
  28  only by calling out to the main rust parser for named nonterminals (which it
  29  commits to fully when it hits one in a grammar). This means that there are no
  30  completer or predictor rules, and therefore no need to store one column per
  31  token: instead, there's a set of current Earley items and a set of next
  32  ones. Instead of NTs, we have a special case for Kleene star. The big-O, in
  33  pathological cases, is worse than traditional Earley parsing, but it's an
  34  easier fit for Macro-by-Example-style rules, and I think the overhead is
  35  lower. (In order to prevent the pathological case, we'd need to lazily
  36  construct the resulting `named_match`es at the very end. It'd be a pain,
  37  and require more memory to keep around old items, but it would also save
  38  overhead)*/

  40  /* Quick intro to how the parser works:
  41  
  42  A 'position' is a dot in the middle of a matcher, usually represented as a
  43  dot. For example `Â· a $( a )* a b` is a position, as is `a $( Â· a )* a b`.
  44  
  45  The parser walks through the input a character at a time, maintaining a list
  46  of items consistent with the current position in the input string: `cur_eis`.
  47  
  48  As it processes them, it fills up `eof_eis` with items that would be valid if
  49  the macro invocation is now over, `bb_eis` with items that are waiting on
  50  a Rust nonterminal like `$e:expr`, and `next_eis` with items that are waiting

 111  #[deriving(Clone)]
 112  pub struct MatcherPos {
 113      elts: ~[ast::matcher], // maybe should be <'>? Need to understand regions.
 114      sep: Option<Token>,
 115      idx: uint,
 116      up: matcher_pos_up, // mutable for swapping only
 117      matches: ~[~[@named_match]],
 118      match_lo: uint, match_hi: uint,
 119      sp_lo: BytePos,
 120  }

 122  pub fn copy_up(mpu: &matcher_pos_up) -> ~MatcherPos {
 123      match *mpu {
 124        matcher_pos_up(Some(ref mp)) => (*mp).clone(),
 125        _ => fail!()
 126      }
 127  }

 129  pub fn count_names(ms: &[matcher]) -> uint {
 130      do ms.iter().fold(0) |ct, m| {
 131          ct + match m.node {
 132            match_tok(_) => 0u,
 133            match_seq(ref more_ms, _, _, _, _) => count_names((*more_ms)),
 134            match_nonterminal(_,_,_) => 1u
 135          }}
 136  }

 138  pub fn initial_matcher_pos(ms: ~[matcher], sep: Option<Token>, lo: BytePos)
 139                          -> ~MatcherPos {
 140      let mut match_idx_hi = 0u;
 141      for elt in ms.iter() {
 142          match elt.node {
 143            match_tok(_) => (),
 144            match_seq(_,_,_,_,hi) => {
 145              match_idx_hi = hi;       // it is monotonic...
 146            }
 147            match_nonterminal(_,_,pos) => {
 148              match_idx_hi = pos+1u;  // ...so latest is highest
 149            }
 150          }
 151      }
 152      let matches = vec::from_fn(count_names(ms), |_i| ~[]);
 153      ~MatcherPos {
 154          elts: ms,
 155          sep: sep,
 156          idx: 0u,
 157          up: matcher_pos_up(None),
 158          matches: matches,
 159          match_lo: 0u,
 160          match_hi: match_idx_hi,
 161          sp_lo: lo
 162      }
 163  }

 165  // named_match is a pattern-match result for a single ast::match_nonterminal:
 166  // so it is associated with a single ident in a parse, and all
 167  // matched_nonterminals in the named_match have the same nonterminal type
 168  // (expr, item, etc). All the leaves in a single named_match correspond to a
 169  // single matcher_nonterminal in the ast::matcher that produced it.
 170  //
 171  // It should probably be renamed, it has more or less exact correspondence to
 172  // ast::match nodes, and the in-memory structure of a particular named_match
 173  // represents the match that occurred when a particular subset of an
 174  // ast::match -- those ast::matcher nodes leading to a single
 175  // match_nonterminal -- was applied to a particular token tree.
 176  //
 177  // The width of each matched_seq in the named_match, and the identity of the
 178  // matched_nonterminals, will depend on the token tree it was applied to: each
 179  // matched_seq corresponds to a single match_seq in the originating
 180  // ast::matcher. The depth of the named_match structure will therefore depend
 181  // only on the nesting depth of ast::match_seqs in the originating
 182  // ast::matcher it was derived from.
 183  
 184  pub enum named_match {
 185      matched_seq(~[@named_match], codemap::span),
 186      matched_nonterminal(nonterminal)
 187  }

 191  pub fn nameize(p_s: @mut ParseSess, ms: &[matcher], res: &[@named_match])
 192              -> HashMap<ident,@named_match> {
 193      fn n_rec(p_s: @mut ParseSess, m: &matcher, res: &[@named_match],
 194               ret_val: &mut HashMap<ident, @named_match>) {
 195          match *m {
 196            codemap::spanned {node: match_tok(_), _} => (),
 197            codemap::spanned {node: match_seq(ref more_ms, _, _, _, _), _} => {
 198              for next_m in more_ms.iter() {
 199                  n_rec(p_s, next_m, res, ret_val)
 200              };
 201            }
 202            codemap::spanned {
 203                  node: match_nonterminal(ref bind_name, _, idx), span: sp
 204            } => {
 205              if ret_val.contains_key(bind_name) {
 206                  p_s.span_diagnostic.span_fatal(sp, ~"Duplicated bind name: "+
 207                                                 ident_to_str(bind_name))
 208              }
 209              ret_val.insert(*bind_name, res[idx]);
 210            }
 211          }
 212      }
 213      let mut ret_val = HashMap::new();
 214      for m in ms.iter() { n_rec(p_s, m, res, &mut ret_val) }
 215      ret_val
 216  }

 224  pub fn parse_or_else(
 225      sess: @mut ParseSess,
 226      cfg: ast::CrateConfig,
 227      rdr: @mut reader,
 228      ms: ~[matcher]
 229  ) -> HashMap<ident, @named_match> {
 230      match parse(sess, cfg, rdr, ms) {
 231        success(m) => m,
 232        failure(sp, str) => sess.span_diagnostic.span_fatal(sp, str),
 233        error(sp, str) => sess.span_diagnostic.span_fatal(sp, str)
 234      }
 235  }

 237  pub fn parse(
 238      sess: @mut ParseSess,
 239      cfg: ast::CrateConfig,
 240      rdr: @mut reader,
 241      ms: &[matcher]
 242  ) -> parse_result {
 243      let mut cur_eis = ~[];
 244      cur_eis.push(initial_matcher_pos(ms.to_owned(), None, rdr.peek().sp.lo));
 245  
 246      loop {
 247          let mut bb_eis = ~[]; // black-box parsed by parser.rs
 248          let mut next_eis = ~[]; // or proceed normally
 249          let mut eof_eis = ~[];
 250  
 251          let TokenAndSpan {tok: tok, sp: sp} = rdr.peek();
 252  
 253          /* we append new items to this while we go */
 254          while !cur_eis.is_empty() { /* for each Earley Item */
 255              let ei = cur_eis.pop();
 256  
 257              let idx = ei.idx;
 258              let len = ei.elts.len();
 259  
 260              /* at end of sequence */
 261              if idx >= len {
 262                  // can't move out of `match`es, so:
 263                  if is_some(&ei.up) {
 264                      // hack: a matcher sequence is repeating iff it has a
 265                      // parent (the top level is just a container)
 266  
 267  
 268                      // disregard separator, try to go up
 269                      // (remove this condition to make trailing seps ok)
 270                      if idx == len {
 271                          // pop from the matcher position
 272  
 273                          let mut new_pos = copy_up(&ei.up);
 274  
 275                          // update matches (the MBE "parse tree") by appending
 276                          // each tree as a subtree.
 277  
 278                          // I bet this is a perf problem: we're preemptively
 279                          // doing a lot of array work that will get thrown away
 280                          // most of the time.
 281  
 282                          // Only touch the binders we have actually bound
 283                          for idx in range(ei.match_lo, ei.match_hi) {
 284                              let sub = ei.matches[idx].clone();
 285                              new_pos.matches[idx]
 286                                  .push(@matched_seq(sub,
 287                                                     mk_sp(ei.sp_lo,
 288                                                           sp.hi)));
 289                          }
 290  
 291                          new_pos.idx += 1;
 292                          cur_eis.push(new_pos);
 293                      }
 294  
 295                      // can we go around again?
 296  
 297                      // the *_t vars are workarounds for the lack of unary move
 298                      match ei.sep {
 299                        Some(ref t) if idx == len => { // we need a separator
 300                          if tok == (*t) { //pass the separator
 301                              let mut ei_t = ei.clone();
 302                              ei_t.idx += 1;
 303                              next_eis.push(ei_t);
 304                          }
 305                        }
 306                        _ => { // we don't need a separator
 307                          let mut ei_t = ei;
 308                          ei_t.idx = 0;
 309                          cur_eis.push(ei_t);
 310                        }
 311                      }
 312                  } else {
 313                      eof_eis.push(ei);
 314                  }
 315              } else {
 316                  match ei.elts[idx].node.clone() {
 317                    /* need to descend into sequence */
 318                    match_seq(ref matchers, ref sep, zero_ok,
 319                              match_idx_lo, match_idx_hi) => {
 320                      if zero_ok {
 321                          let mut new_ei = ei.clone();
 322                          new_ei.idx += 1u;
 323                          //we specifically matched zero repeats.
 324                          for idx in range(match_idx_lo, match_idx_hi) {
 325                              new_ei.matches[idx].push(@matched_seq(~[], sp));
 326                          }
 327  
 328                          cur_eis.push(new_ei);
 329                      }
 330  
 331                      let matches = vec::from_elem(ei.matches.len(), ~[]);
 332                      let ei_t = ei;
 333                      cur_eis.push(~MatcherPos {
 334                          elts: (*matchers).clone(),
 335                          sep: (*sep).clone(),
 336                          idx: 0u,
 337                          up: matcher_pos_up(Some(ei_t)),
 338                          matches: matches,
 339                          match_lo: match_idx_lo, match_hi: match_idx_hi,
 340                          sp_lo: sp.lo
 341                      });
 342                    }
 343                    match_nonterminal(_,_,_) => { bb_eis.push(ei) }
 344                    match_tok(ref t) => {
 345                      let mut ei_t = ei.clone();
 346                      if (*t) == tok {
 347                          ei_t.idx += 1;
 348                          next_eis.push(ei_t);
 349                      }
 350                    }
 351                  }
 352              }
 353          }
 354  
 355          /* error messages here could be improved with links to orig. rules */
 356          if tok == EOF {
 357              if eof_eis.len() == 1u {
 358                  let mut v = ~[];
 359                  for dv in eof_eis[0u].matches.mut_iter() {
 360                      v.push(dv.pop());
 361                  }
 362                  return success(nameize(sess, ms, v));
 363              } else if eof_eis.len() > 1u {
 364                  return error(sp, ~"Ambiguity: multiple successful parses");
 365              } else {
 366                  return failure(sp, ~"Unexpected end of macro invocation");
 367              }
 368          } else {
 369              if (bb_eis.len() > 0u && next_eis.len() > 0u)
 370                  || bb_eis.len() > 1u {
 371                  let nts = bb_eis.map(|ei| {
 372                      match ei.elts[ei.idx].node {
 373                        match_nonterminal(ref bind,ref name,_) => {
 374                          fmt!("%s ('%s')", ident_to_str(name),
 375                               ident_to_str(bind))
 376                        }
 377                        _ => fail!()
 378                      } }).connect(" or ");
 379                  return error(sp, fmt!(
 380                      "Local ambiguity: multiple parsing options: \
 381                       built-in NTs %s or %u other options.",
 382                      nts, next_eis.len()));
 383              } else if (bb_eis.len() == 0u && next_eis.len() == 0u) {
 384                  return failure(sp, ~"No rules expected the token: "
 385                              + to_str(get_ident_interner(), &tok));
 386              } else if (next_eis.len() > 0u) {
 387                  /* Now process the next token */
 388                  while(next_eis.len() > 0u) {
 389                      cur_eis.push(next_eis.pop());
 390                  }
 391                  rdr.next_token();
 392              } else /* bb_eis.len() == 1 */ {
 393                  let rust_parser = Parser(sess, cfg.clone(), rdr.dup());
 394  
 395                  let mut ei = bb_eis.pop();
 396                  match ei.elts[ei.idx].node {
 397                    match_nonterminal(_, ref name, idx) => {
 398                      ei.matches[idx].push(@matched_nonterminal(
 399                          parse_nt(&rust_parser, ident_to_str(name))));
 400                      ei.idx += 1u;

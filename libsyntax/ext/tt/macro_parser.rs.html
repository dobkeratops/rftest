  27  /* This is an Earley-like parser, without support for in-grammar nonterminals,
  28  only by calling out to the main rust parser for named nonterminals (which it
  29  commits to fully when it hits one in a grammar). This means that there are no
  30  completer or predictor rules, and therefore no need to store one column per
  31  token: instead, there's a set of current Earley items and a set of next
  32  ones. Instead of NTs, we have a special case for Kleene star. The big-O, in
  33  pathological cases, is worse than traditional Earley parsing, but it's an
  34  easier fit for Macro-by-Example-style rules, and I think the overhead is
  35  lower. (In order to prevent the pathological case, we'd need to lazily
  36  construct the resulting `named_match`es at the very end. It'd be a pain,
  37  and require more memory to keep around old items, but it would also save
  38  overhead)*/

 112  pub struct MatcherPos {
 113      elts: ~[ast::matcher], // maybe should be <'>? Need to understand regions.
 114      sep: Option<Token>,
 115      idx: uint,

 122  pub fn copy_up(mpu: &matcher_pos_up) -> ~MatcherPos {
 123      match *mpu {
 124        matcher_pos_up(Some(ref mp)) => (*mp).clone(),
 125        _ => fail!()
 126      }
 127  }

 129  pub fn count_names(ms: &[matcher]) -> uint {
 130      do ms.iter().fold(0) |ct, m| {
 131          ct + match m.node {
 132            match_tok(_) => 0u,
 133            match_seq(ref more_ms, _, _, _, _) => count_names((*more_ms)),
 134            match_nonterminal(_,_,_) => 1u
 135          }}
 136  }

 138  pub fn initial_matcher_pos(ms: ~[matcher], sep: Option<Token>, lo: BytePos)
 139                          -> ~MatcherPos {
 140      let mut match_idx_hi = 0u;
 141      for elt in ms.iter() {
 142          match elt.node {
 143            match_tok(_) => (),
 144            match_seq(_,_,_,_,hi) => {
 145              match_idx_hi = hi;       // it is monotonic...
 146            }
 147            match_nonterminal(_,_,pos) => {
 148              match_idx_hi = pos+1u;  // ...so latest is highest
 149            }
 150          }
 151      }
 152      let matches = vec::from_fn(count_names(ms), |_i| ~[]);
 153      ~MatcherPos {
 154          elts: ms,
 155          sep: sep,
 156          idx: 0u,
 157          up: matcher_pos_up(None),
 158          matches: matches,
 159          match_lo: 0u,
 160          match_hi: match_idx_hi,
 161          sp_lo: lo
 162      }
 163  }

 165  // named_match is a pattern-match result for a single ast::match_nonterminal:
 166  // so it is associated with a single ident in a parse, and all
 167  // matched_nonterminals in the named_match have the same nonterminal type
 168  // (expr, item, etc). All the leaves in a single named_match correspond to a
 169  // single matcher_nonterminal in the ast::matcher that produced it.
 170  //
 171  // It should probably be renamed, it has more or less exact correspondence to
 172  // ast::match nodes, and the in-memory structure of a particular named_match
 173  // represents the match that occurred when a particular subset of an
 174  // ast::match -- those ast::matcher nodes leading to a single
 175  // match_nonterminal -- was applied to a particular token tree.
 176  //
 177  // The width of each matched_seq in the named_match, and the identity of the
 178  // matched_nonterminals, will depend on the token tree it was applied to: each
 179  // matched_seq corresponds to a single match_seq in the originating
 180  // ast::matcher. The depth of the named_match structure will therefore depend
 181  // only on the nesting depth of ast::match_seqs in the originating
 182  // ast::matcher it was derived from.
 183  
 184  pub enum named_match {
 185      matched_seq(~[@named_match], codemap::Span),
 186      matched_nonterminal(nonterminal)
 187  }

 191  pub fn nameize(p_s: @mut ParseSess, ms: &[matcher], res: &[@named_match])
 192              -> HashMap<Ident,@named_match> {
 193      fn n_rec(p_s: @mut ParseSess, m: &matcher, res: &[@named_match],
 194               ret_val: &mut HashMap<Ident, @named_match>) {
 195          match *m {
 196            codemap::Spanned {node: match_tok(_), _} => (),
 197            codemap::Spanned {node: match_seq(ref more_ms, _, _, _, _), _} => {
 198              for next_m in more_ms.iter() {
 199                  n_rec(p_s, next_m, res, ret_val)
 200              };
 201            }
 202            codemap::Spanned {
 203                  node: match_nonterminal(ref bind_name, _, idx), span: sp
 204            } => {
 205              if ret_val.contains_key(bind_name) {
 206                  p_s.span_diagnostic.span_fatal(sp, ~"Duplicated bind name: "+
 207                                                 ident_to_str(bind_name))
 208              }
 209              ret_val.insert(*bind_name, res[idx]);
 210            }
 211          }
 212      }
 213      let mut ret_val = HashMap::new();
 214      for m in ms.iter() { n_rec(p_s, m, res, &mut ret_val) }
 215      ret_val
 216  }

 237  // perform a token equality check, ignoring syntax context (that is, an unhygienic comparison)
 238  pub fn token_name_eq(t1 : &Token, t2 : &Token) -> bool {
 239      match (t1,t2) {
 240          (&token::IDENT(id1,_),&token::IDENT(id2,_)) =>
 241          id1.name == id2.name,
 242          _ => *t1 == *t2
 243      }
 244  }

 246  pub fn parse(
 247      sess: @mut ParseSess,
 248      cfg: ast::CrateConfig,
 249      rdr: @mut reader,
 250      ms: &[matcher]
 251  ) -> parse_result {
 252      let mut cur_eis = ~[];
 253      cur_eis.push(initial_matcher_pos(ms.to_owned(), None, rdr.peek().sp.lo));
 254  
 255      loop {
 256          let mut bb_eis = ~[]; // black-box parsed by parser.rs
 257          let mut next_eis = ~[]; // or proceed normally
 258          let mut eof_eis = ~[];
 259  
 260          let TokenAndSpan {tok: tok, sp: sp} = rdr.peek();
 261  
 262          /* we append new items to this while we go */
 263          while !cur_eis.is_empty() { /* for each Earley Item */
 264              let ei = cur_eis.pop();
 265  
 266              let idx = ei.idx;
 267              let len = ei.elts.len();
 268  
 269              /* at end of sequence */
 270              if idx >= len {
 271                  // can't move out of `match`es, so:
 272                  if is_some(&ei.up) {
 273                      // hack: a matcher sequence is repeating iff it has a
 274                      // parent (the top level is just a container)
 275  
 276  
 277                      // disregard separator, try to go up
 278                      // (remove this condition to make trailing seps ok)
 279                      if idx == len {
 280                          // pop from the matcher position
 281  
 282                          let mut new_pos = copy_up(&ei.up);
 283  
 284                          // update matches (the MBE "parse tree") by appending
 285                          // each tree as a subtree.
 286  
 287                          // I bet this is a perf problem: we're preemptively
 288                          // doing a lot of array work that will get thrown away
 289                          // most of the time.
 290  
 291                          // Only touch the binders we have actually bound
 292                          for idx in range(ei.match_lo, ei.match_hi) {
 293                              let sub = ei.matches[idx].clone();
 294                              new_pos.matches[idx]
 295                                  .push(@matched_seq(sub,
 296                                                     mk_sp(ei.sp_lo,
 297                                                           sp.hi)));
 298                          }
 299  
 300                          new_pos.idx += 1;
 301                          cur_eis.push(new_pos);
 302                      }
 303  
 304                      // can we go around again?
 305  
 306                      // the *_t vars are workarounds for the lack of unary move
 307                      match ei.sep {
 308                        Some(ref t) if idx == len => { // we need a separator
 309                          // i'm conflicted about whether this should be hygienic....
 310                          // though in this case, if the separators are never legal
 311                          // idents, it shouldn't matter.
 312                          if token_name_eq(&tok, t) { //pass the separator
 313                              let mut ei_t = ei.clone();
 314                              ei_t.idx += 1;
 315                              next_eis.push(ei_t);
 316                          }
 317                        }
 318                        _ => { // we don't need a separator
 319                          let mut ei_t = ei;
 320                          ei_t.idx = 0;
 321                          cur_eis.push(ei_t);
 322                        }
 323                      }
 324                  } else {
 325                      eof_eis.push(ei);
 326                  }
 327              } else {
 328                  match ei.elts[idx].node.clone() {
 329                    /* need to descend into sequence */
 330                    match_seq(ref matchers, ref sep, zero_ok,
 331                              match_idx_lo, match_idx_hi) => {
 332                      if zero_ok {
 333                          let mut new_ei = ei.clone();
 334                          new_ei.idx += 1u;
 335                          //we specifically matched zero repeats.
 336                          for idx in range(match_idx_lo, match_idx_hi) {
 337                              new_ei.matches[idx].push(@matched_seq(~[], sp));
 338                          }
 339  
 340                          cur_eis.push(new_ei);
 341                      }
 342  
 343                      let matches = vec::from_elem(ei.matches.len(), ~[]);
 344                      let ei_t = ei;
 345                      cur_eis.push(~MatcherPos {
 346                          elts: (*matchers).clone(),
 347                          sep: (*sep).clone(),
 348                          idx: 0u,
 349                          up: matcher_pos_up(Some(ei_t)),
 350                          matches: matches,
 351                          match_lo: match_idx_lo, match_hi: match_idx_hi,
 352                          sp_lo: sp.lo
 353                      });
 354                    }
 355                    match_nonterminal(_,_,_) => { bb_eis.push(ei) }
 356                    match_tok(ref t) => {
 357                      let mut ei_t = ei.clone();
 358                      //if (token_name_eq(t,&tok)) {
 359                      if (token::mtwt_token_eq(t,&tok)) {
 360                          ei_t.idx += 1;
 361                          next_eis.push(ei_t);
 362                      }
 363                    }
 364                  }
 365              }
 366          }
 367  
 368          /* error messages here could be improved with links to orig. rules */
 369          if token_name_eq(&tok, &EOF) {
 370              if eof_eis.len() == 1u {
 371                  let mut v = ~[];
 372                  for dv in eof_eis[0u].matches.mut_iter() {
 373                      v.push(dv.pop());
 374                  }
 375                  return success(nameize(sess, ms, v));
 376              } else if eof_eis.len() > 1u {
 377                  return error(sp, ~"Ambiguity: multiple successful parses");
 378              } else {
 379                  return failure(sp, ~"Unexpected end of macro invocation");
 380              }
 381          } else {
 382              if (bb_eis.len() > 0u && next_eis.len() > 0u)
 383                  || bb_eis.len() > 1u {
 384                  let nts = bb_eis.map(|ei| {
 385                      match ei.elts[ei.idx].node {
 386                        match_nonterminal(ref bind,ref name,_) => {
 387                          fmt!("%s ('%s')", ident_to_str(name),
 388                               ident_to_str(bind))
 389                        }
 390                        _ => fail!()
 391                      } }).connect(" or ");
 392                  return error(sp, fmt!(
 393                      "Local ambiguity: multiple parsing options: \
 394                       built-in NTs %s or %u other options.",
 395                      nts, next_eis.len()));
 396              } else if (bb_eis.len() == 0u && next_eis.len() == 0u) {
 397                  return failure(sp, ~"No rules expected the token: "
 398                              + to_str(get_ident_interner(), &tok));

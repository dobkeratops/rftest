  26  /* This is an Earley-like parser, without support for in-grammar nonterminals,
  27  only by calling out to the main rust parser for named nonterminals (which it
  28  commits to fully when it hits one in a grammar). This means that there are no
  29  completer or predictor rules, and therefore no need to store one column per
  30  token: instead, there's a set of current Earley items and a set of next
  31  ones. Instead of NTs, we have a special case for Kleene star. The big-O, in
  32  pathological cases, is worse than traditional Earley parsing, but it's an
  33  easier fit for Macro-by-Example-style rules, and I think the overhead is
  34  lower. (In order to prevent the pathological case, we'd need to lazily
  35  construct the resulting `named_match`es at the very end. It'd be a pain,
  36  and require more memory to keep around old items, but it would also save
  37  overhead)*/

  39  /* Quick intro to how the parser works:
  40  
  41  A 'position' is a dot in the middle of a matcher, usually represented as a
  42  dot. For example `Â· a $( a )* a b` is a position, as is `a $( Â· a )* a b`.
  43  
  44  The parser walks through the input a character at a time, maintaining a list
  45  of items consistent with the current position in the input string: `cur_eis`.
  46  
  47  As it processes them, it fills up `eof_eis` with items that would be valid if
  48  the macro invocation is now over, `bb_eis` with items that are waiting on
  49  a Rust nonterminal like `$e:expr`, and `next_eis` with items that are waiting

  51  repetitions indicated by Kleene stars. It only advances or calls out to the
  52  real Rust parser when no `cur_eis` items remain
  53

 110  #[deriving(Clone)]
 111  pub struct MatcherPos {
 112      elts: ~[ast::matcher], // maybe should be <'>? Need to understand regions.
 113      sep: Option<Token>,
 114      idx: uint,
 115      up: matcher_pos_up, // mutable for swapping only
 116      matches: ~[~[@named_match]],
 117      match_lo: uint, match_hi: uint,
 118      sp_lo: BytePos,
 119  }

 121  pub fn copy_up(mpu: &matcher_pos_up) -> ~MatcherPos {
 122      match *mpu {
 123        matcher_pos_up(Some(ref mp)) => (*mp).clone(),
 124        _ => fail!()
 125      }
 126  }

 128  pub fn count_names(ms: &[matcher]) -> uint {
 129      do ms.iter().fold(0) |ct, m| {
 130          ct + match m.node {
 131            match_tok(_) => 0u,
 132            match_seq(ref more_ms, _, _, _, _) => count_names((*more_ms)),
 133            match_nonterminal(_,_,_) => 1u
 134          }}
 135  }

 137  pub fn initial_matcher_pos(ms: ~[matcher], sep: Option<Token>, lo: BytePos)
 138                          -> ~MatcherPos {
 139      let mut match_idx_hi = 0u;
 140      for elt in ms.iter() {
 141          match elt.node {
 142            match_tok(_) => (),
 143            match_seq(_,_,_,_,hi) => {
 144              match_idx_hi = hi;       // it is monotonic...
 145            }
 146            match_nonterminal(_,_,pos) => {
 147              match_idx_hi = pos+1u;  // ...so latest is highest
 148            }
 149          }
 150      }
 151      let matches = vec::from_fn(count_names(ms), |_i| ~[]);
 152      ~MatcherPos {
 153          elts: ms,
 154          sep: sep,
 155          idx: 0u,
 156          up: matcher_pos_up(None),
 157          matches: matches,
 158          match_lo: 0u,
 159          match_hi: match_idx_hi,
 160          sp_lo: lo
 161      }
 162  }

 164  // named_match is a pattern-match result for a single ast::match_nonterminal:
 165  // so it is associated with a single ident in a parse, and all
 166  // matched_nonterminals in the named_match have the same nonterminal type
 167  // (expr, item, etc). All the leaves in a single named_match correspond to a
 168  // single matcher_nonterminal in the ast::matcher that produced it.
 169  //
 170  // It should probably be renamed, it has more or less exact correspondence to
 171  // ast::match nodes, and the in-memory structure of a particular named_match
 172  // represents the match that occurred when a particular subset of an
 173  // ast::match -- those ast::matcher nodes leading to a single
 174  // match_nonterminal -- was applied to a particular token tree.
 175  //
 176  // The width of each matched_seq in the named_match, and the identity of the
 177  // matched_nonterminals, will depend on the token tree it was applied to: each
 178  // matched_seq corresponds to a single match_seq in the originating
 179  // ast::matcher. The depth of the named_match structure will therefore depend
 180  // only on the nesting depth of ast::match_seqs in the originating
 181  // ast::matcher it was derived from.
 182  
 183  pub enum named_match {
 184      matched_seq(~[@named_match], codemap::span),
 185      matched_nonterminal(nonterminal)
 186  }

 190  pub fn nameize(p_s: @mut ParseSess, ms: &[matcher], res: &[@named_match])
 191              -> HashMap<ident,@named_match> {
 192      fn n_rec(p_s: @mut ParseSess, m: &matcher, res: &[@named_match],
 193               ret_val: &mut HashMap<ident, @named_match>) {
 194          match *m {
 195            codemap::spanned {node: match_tok(_), _} => (),
 196            codemap::spanned {node: match_seq(ref more_ms, _, _, _, _), _} => {
 197              for next_m in more_ms.iter() {
 198                  n_rec(p_s, next_m, res, ret_val)
 199              };
 200            }
 201            codemap::spanned {
 202                  node: match_nonterminal(ref bind_name, _, idx), span: sp
 203            } => {
 204              if ret_val.contains_key(bind_name) {
 205                  p_s.span_diagnostic.span_fatal(sp, ~"Duplicated bind name: "+
 206                                                 ident_to_str(bind_name))
 207              }
 208              ret_val.insert(*bind_name, res[idx]);
 209            }
 210          }
 211      }
 212      let mut ret_val = HashMap::new();
 213      for m in ms.iter() { n_rec(p_s, m, res, &mut ret_val) }
 214      ret_val
 215  }

 223  pub fn parse_or_else(
 224      sess: @mut ParseSess,
 225      cfg: ast::CrateConfig,
 226      rdr: @reader,
 227      ms: ~[matcher]
 228  ) -> HashMap<ident, @named_match> {
 229      match parse(sess, cfg, rdr, ms) {
 230        success(m) => m,
 231        failure(sp, str) => sess.span_diagnostic.span_fatal(sp, str),
 232        error(sp, str) => sess.span_diagnostic.span_fatal(sp, str)
 233      }
 234  }
 235  
 236  pub fn parse(
 237      sess: @mut ParseSess,
 238      cfg: ast::CrateConfig,
 239      rdr: @reader,
 240      ms: &[matcher]
 241  ) -> parse_result {
 242      let mut cur_eis = ~[];
 243      cur_eis.push(initial_matcher_pos(ms.to_owned(), None, rdr.peek().sp.lo));
 244  
 245      loop {
 246          let mut bb_eis = ~[]; // black-box parsed by parser.rs
 247          let mut next_eis = ~[]; // or proceed normally
 248          let mut eof_eis = ~[];
 249  
 250          let TokenAndSpan {tok: tok, sp: sp} = rdr.peek();
 251  
 252          /* we append new items to this while we go */
 253          while !cur_eis.is_empty() { /* for each Earley Item */
 254              let ei = cur_eis.pop();
 255  
 256              let idx = ei.idx;
 257              let len = ei.elts.len();
 258  
 259              /* at end of sequence */
 260              if idx >= len {
 261                  // can't move out of `match`es, so:
 262                  if is_some(&ei.up) {
 263                      // hack: a matcher sequence is repeating iff it has a
 264                      // parent (the top level is just a container)
 265  
 266  
 267                      // disregard separator, try to go up
 268                      // (remove this condition to make trailing seps ok)
 269                      if idx == len {
 270                          // pop from the matcher position
 271  
 272                          let mut new_pos = copy_up(&ei.up);
 273  
 274                          // update matches (the MBE "parse tree") by appending
 275                          // each tree as a subtree.
 276  
 277                          // I bet this is a perf problem: we're preemptively
 278                          // doing a lot of array work that will get thrown away
 279                          // most of the time.
 280  
 281                          // Only touch the binders we have actually bound
 282                          for idx in range(ei.match_lo, ei.match_hi) {
 283                              let sub = ei.matches[idx].clone();
 284                              new_pos.matches[idx]
 285                                  .push(@matched_seq(sub,
 286                                                     mk_sp(ei.sp_lo,
 287                                                           sp.hi)));
 288                          }
 289  
 290                          new_pos.idx += 1;
 291                          cur_eis.push(new_pos);
 292                      }
 293  
 294                      // can we go around again?
 295  
 296                      // the *_t vars are workarounds for the lack of unary move
 297                      match ei.sep {
 298                        Some(ref t) if idx == len => { // we need a separator
 299                          if tok == (*t) { //pass the separator
 300                              let mut ei_t = ei.clone();
 301                              ei_t.idx += 1;
 302                              next_eis.push(ei_t);
 303                          }
 304                        }
 305                        _ => { // we don't need a separator
 306                          let mut ei_t = ei;
 307                          ei_t.idx = 0;
 308                          cur_eis.push(ei_t);
 309                        }
 310                      }
 311                  } else {
 312                      eof_eis.push(ei);
 313                  }
 314              } else {
 315                  match ei.elts[idx].node.clone() {
 316                    /* need to descend into sequence */
 317                    match_seq(ref matchers, ref sep, zero_ok,
 318                              match_idx_lo, match_idx_hi) => {
 319                      if zero_ok {
 320                          let mut new_ei = ei.clone();
 321                          new_ei.idx += 1u;
 322                          //we specifically matched zero repeats.
 323                          for idx in range(match_idx_lo, match_idx_hi) {
 324                              new_ei.matches[idx].push(@matched_seq(~[], sp));
 325                          }
 326  
 327                          cur_eis.push(new_ei);
 328                      }
 329  
 330                      let matches = vec::from_elem(ei.matches.len(), ~[]);
 331                      let ei_t = ei;
 332                      cur_eis.push(~MatcherPos {
 333                          elts: (*matchers).clone(),
 334                          sep: (*sep).clone(),
 335                          idx: 0u,
 336                          up: matcher_pos_up(Some(ei_t)),
 337                          matches: matches,
 338                          match_lo: match_idx_lo, match_hi: match_idx_hi,
 339                          sp_lo: sp.lo
 340                      });
 341                    }
 342                    match_nonterminal(_,_,_) => { bb_eis.push(ei) }
 343                    match_tok(ref t) => {
 344                      let mut ei_t = ei.clone();
 345                      if (*t) == tok {
 346                          ei_t.idx += 1;
 347                          next_eis.push(ei_t);
 348                      }
 349                    }
 350                  }
 351              }
 352          }
 353  
 354          /* error messages here could be improved with links to orig. rules */
 355          if tok == EOF {
 356              if eof_eis.len() == 1u {
 357                  let mut v = ~[];
 358                  for dv in eof_eis[0u].matches.mut_iter() {
 359                      v.push(dv.pop());
 360                  }
 361                  return success(nameize(sess, ms, v));
 362              } else if eof_eis.len() > 1u {
 363                  return error(sp, ~"Ambiguity: multiple successful parses");
 364              } else {
 365                  return failure(sp, ~"Unexpected end of macro invocation");
 366              }
 367          } else {
 368              if (bb_eis.len() > 0u && next_eis.len() > 0u)
 369                  || bb_eis.len() > 1u {
 370                  let nts = bb_eis.map(|ei| {
 371                      match ei.elts[ei.idx].node {
 372                        match_nonterminal(ref bind,ref name,_) => {
 373                          fmt!("%s ('%s')", ident_to_str(name),
 374                               ident_to_str(bind))
 375                        }
 376                        _ => fail!()
 377                      } }).connect(" or ");
 378                  return error(sp, fmt!(
 379                      "Local ambiguity: multiple parsing options: \
 380                       built-in NTs %s or %u other options.",
 381                      nts, next_eis.len()));
 382              } else if (bb_eis.len() == 0u && next_eis.len() == 0u) {
 383                  return failure(sp, ~"No rules expected the token: "
 384                              + to_str(get_ident_interner(), &tok));
 385              } else if (next_eis.len() > 0u) {
 386                  /* Now process the next token */
 387                  while(next_eis.len() > 0u) {
 388                      cur_eis.push(next_eis.pop());
 389                  }
 390                  rdr.next_token();
 391              } else /* bb_eis.len() == 1 */ {
 392                  let rust_parser = Parser(sess, cfg.clone(), rdr.dup());
 393  
 394                  let mut ei = bb_eis.pop();
 395                  match ei.elts[ei.idx].node {
 396                    match_nonterminal(_, ref name, idx) => {
 397                      ei.matches[idx].push(@matched_nonterminal(
 398                          parse_nt(&rust_parser, ident_to_str(name))));
 399                      ei.idx += 1u;
 400                    }
 401                    _ => fail!()

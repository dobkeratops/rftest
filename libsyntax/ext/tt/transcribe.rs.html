 128  enum LockstepIterSize {
 129      LisUnconstrained,
 130      LisConstraint(uint, Ident),
 131      LisContradiction(~str),

 134  fn lis_merge(lhs: LockstepIterSize, rhs: LockstepIterSize) -> LockstepIterSize {
 135      match lhs {
 136          LisUnconstrained => rhs.clone(),
 137          LisContradiction(_) => lhs.clone(),
 138          LisConstraint(l_len, l_id) => match rhs {
 139              LisUnconstrained => lhs.clone(),
 140              LisContradiction(_) => rhs.clone(),
 141              LisConstraint(r_len, _) if l_len == r_len => lhs.clone(),
 142              LisConstraint(r_len, r_id) => {
 143                  let l_n = token::get_ident(l_id);
 144                  let r_n = token::get_ident(r_id);
 145                  LisContradiction(format!("inconsistent lockstep iteration: \
 146                                            '{}' has {} items, but '{}' has {}",
 147                                            l_n, l_len, r_n, r_len))
 148              }
 149          }
 150      }
 151  }

 153  fn lockstep_iter_size(t: &TokenTree, r: &TtReader) -> LockstepIterSize {
 154      match *t {
 155          TTDelim(ref tts) | TTSeq(_, ref tts, _, _) => {
 156              tts.iter().fold(LisUnconstrained, |lis, tt| {
 157                  lis_merge(lis, lockstep_iter_size(tt, r))
 158              })
 159          }
 160          TTTok(..) => LisUnconstrained,
 161          TTNonterminal(_, name) => match *lookup_cur_matched(r, name) {
 162              MatchedNonterminal(_) => LisUnconstrained,
 163              MatchedSeq(ref ads, _) => LisConstraint(ads.len(), name)
 164          }
 165      }
 166  }

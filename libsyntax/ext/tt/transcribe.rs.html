 134      fn lis_merge(lhs: lis, rhs: lis) -> lis {
 135          match lhs {
 136            lis_unconstrained => rhs.clone(),
 137            lis_contradiction(_) => lhs.clone(),
 138            lis_constraint(l_len, ref l_id) => match rhs {
 139              lis_unconstrained => lhs.clone(),
 140              lis_contradiction(_) => rhs.clone(),
 141              lis_constraint(r_len, _) if l_len == r_len => lhs.clone(),
 142              lis_constraint(r_len, ref r_id) => {
 143                  let l_n = ident_to_str(l_id);
 144                  let r_n = ident_to_str(r_id);
 145                  lis_contradiction(fmt!("Inconsistent lockstep iteration: \
 146                                         '%s' has %u items, but '%s' has %u",
 147                                          l_n, l_len, r_n, r_len))
 148              }
 149            }
 150          }
 151      }
 152      match *t {
 153        tt_delim(ref tts) | tt_seq(_, ref tts, _, _) => {
 154          do tts.iter().fold(lis_unconstrained) |lis, tt| {
 155              let lis2 = lockstep_iter_size(tt, r);
 156              lis_merge(lis, lis2)
 157          }
 158        }
 159        tt_tok(*) => lis_unconstrained,
 160        tt_nonterminal(_, name) => match *lookup_cur_matched(r, name) {
 161          matched_nonterminal(_) => lis_unconstrained,
 162          matched_seq(ref ads, _) => lis_constraint(ads.len(), name)
 163        }
 164      }
 165  }

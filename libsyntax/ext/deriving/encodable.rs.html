  11  /*!
  12  
  13  The compiler code necessary to implement the #[deriving(Encodable)]
  14  (and Decodable, in decodable.rs) extension.  The idea here is that
  15  type-defining items may be tagged with #[deriving(Encodable,
  16  Decodable)].
  17  
  18  For example, a type like:
  19  
  20      #[deriving(Encodable, Decodable)]
  21      struct Node {id: uint}
  22  
  23  would generate two implementations like:
  24  
  25  impl<S:extra::serialize::Encoder> Encodable<S> for Node {
  26      fn encode(&self, s: &S) {
  27          do s.emit_struct("Node", 1) {
  28              s.emit_field("id", 0, || s.emit_uint(self.id))
  29          }
  30      }
  31  }
  32  
  33  impl<D:Decoder> Decodable for node_id {
  34      fn decode(d: &D) -> Node {
  35          do d.read_struct("Node", 1) {
  36              Node {
  37                  id: d.read_field(~"x", 0, || decode(d))
  38              }
  39          }
  40      }
  41  }
  42  
  43  Other interesting scenarios are whe the item has type parameters or
  44  references other non-built-in types.  A type definition like:
  45  
  46      #[deriving(Encodable, Decodable)]
  47      struct spanned<T> {node: T, span: Span}
  48  
  49  would yield functions like:
  50  
  51      impl<
  52          S: Encoder,
  53          T: Encodable<S>
  54      > spanned<T>: Encodable<S> {
  55          fn encode<S:Encoder>(s: &S) {
  56              do s.emit_rec {
  57                  s.emit_field("node", 0, || self.node.encode(s));
  58                  s.emit_field("span", 1, || self.span.encode(s));
  59              }
  60          }
  61      }
  62  
  63      impl<
  64          D: Decoder,
  65          T: Decodable<D>
  66      > spanned<T>: Decodable<D> {
  67          fn decode(d: &D) -> spanned<T> {
  68              do d.read_rec {
  69                  {
  70                      node: d.read_field(~"node", 0, || decode(d)),
  71                      span: d.read_field(~"span", 1, || decode(d)),
  72                  }
  73              }
  74          }
  75      }
  76  */

 113  fn encodable_substructure(cx: @ExtCtxt, span: Span,
 114                            substr: &Substructure) -> @Expr {
 115      let encoder = substr.nonself_args[0];
 116      // throw an underscore in front to suppress unused variable warnings
 117      let blkarg = cx.ident_of("_e");
 118      let blkencoder = cx.expr_ident(span, blkarg);
 119      let encode = cx.ident_of("encode");
 120  
 121      return match *substr.fields {
 122          Struct(ref fields) => {
 123              let emit_struct_field = cx.ident_of("emit_struct_field");
 124              let mut stmts = ~[];
 125              for (i, f) in fields.iter().enumerate() {
 126                  let (name, val) = match *f {
 127                      (Some(id), e, _) => (cx.str_of(id), e),
 128                      (None, e, _) => (fmt!("_field%u", i).to_managed(), e)
 129                  };
 130                  let enc = cx.expr_method_call(span, val, encode, ~[blkencoder]);
 131                  let lambda = cx.lambda_expr_1(span, enc, blkarg);
 132                  let call = cx.expr_method_call(span, blkencoder,
 133                                                 emit_struct_field,
 134                                                 ~[cx.expr_str(span, name),
 135                                                   cx.expr_uint(span, i),
 136                                                   lambda]);
 137                  stmts.push(cx.stmt_expr(call));
 138              }
 139  
 140              let blk = cx.lambda_stmts_1(span, stmts, blkarg);
 141              cx.expr_method_call(span, encoder, cx.ident_of("emit_struct"),
 142                                  ~[cx.expr_str(span, cx.str_of(substr.type_ident)),
 143                                    cx.expr_uint(span, fields.len()),
 144                                    blk])
 145          }
 146  
 147          EnumMatching(idx, variant, ref fields) => {
 148              // We're not generating an AST that the borrow checker is expecting,
 149              // so we need to generate a unique local variable to take the
 150              // mutable loan out on, otherwise we get conflicts which don't
 151              // actually exist.
 152              let me = cx.stmt_let(span, false, blkarg, encoder);
 153              let encoder = cx.expr_ident(span, blkarg);
 154              let emit_variant_arg = cx.ident_of("emit_enum_variant_arg");
 155              let mut stmts = ~[];
 156              for (i, f) in fields.iter().enumerate() {
 157                  let val = match *f { (_, e, _) => e };
 158                  let enc = cx.expr_method_call(span, val, encode, ~[blkencoder]);
 159                  let lambda = cx.lambda_expr_1(span, enc, blkarg);
 160                  let call = cx.expr_method_call(span, blkencoder,
 161                                                 emit_variant_arg,
 162                                                 ~[cx.expr_uint(span, i),
 163                                                   lambda]);
 164                  stmts.push(cx.stmt_expr(call));
 165              }
 166  
 167              let blk = cx.lambda_stmts_1(span, stmts, blkarg);
 168              let name = cx.expr_str(span, cx.str_of(variant.node.name));
 169              let call = cx.expr_method_call(span, blkencoder,
 170                                             cx.ident_of("emit_enum_variant"),
 171                                             ~[name,
 172                                               cx.expr_uint(span, idx),
 173                                               cx.expr_uint(span, fields.len()),
 174                                               blk]);
 175              let blk = cx.lambda_expr_1(span, call, blkarg);
 176              let ret = cx.expr_method_call(span, encoder,
 177                                            cx.ident_of("emit_enum"),
 178                                            ~[cx.expr_str(span,
 179                                              cx.str_of(substr.type_ident)),
 180                                              blk]);
 181              cx.expr_block(cx.block(span, ~[me], Some(ret)))
 182          }
 183  
 184          _ => cx.bug("expected Struct or EnumMatching in deriving(Encodable)")
 185      };
 186  }

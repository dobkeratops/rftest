 45 fn iter_bytes_substructure(cx: @ExtCtxt, span: span, substr: &Substructure) -> @expr {
 46     let (lsb0, f)= match substr.nonself_args {
 47         [l, f] => (l, f),
 48         _ => cx.span_bug(span, "Incorrect number of arguments in `deriving(IterBytes)`")
 49     };
 50     // Build the "explicitly borrowed" stack closure, "|_buf| f(_buf)".
 51     let blk_arg = cx.ident_of("_buf");
 52     let borrowed_f =
 53         cx.lambda_expr_1(span, cx.expr_call(span, f, ~[cx.expr_ident(span, blk_arg)]),
 54                          blk_arg);
 55 
 56     let iter_bytes_ident = substr.method_ident;
 57     let call_iterbytes = |thing_expr| {
 58         cx.expr_method_call(span,
 59                             thing_expr,
 60                             iter_bytes_ident,
 61                             ~[lsb0, borrowed_f])
 62     };
 63     let mut exprs = ~[];
 64     let fields;
 65     match *substr.fields {
 66         Struct(ref fs) => {
 67             fields = fs
 68         }
 69         EnumMatching(index, ref variant, ref fs) => {
 70             // Determine the discriminant. We will feed this value to the byte
 71             // iteration function.
 72             let discriminant = match variant.node.disr_expr {
 73                 Some(d)=> d,
 74                 None => cx.expr_uint(span, index)
 75             };
 76 
 77             exprs.push(call_iterbytes(discriminant));
 78 
 79             fields = fs;
 80         }
 81         _ => cx.span_bug(span, "Impossible substructure in `deriving(IterBytes)`")
 82     }
 83 
 84     for &(_, field, _) in fields.iter() {
 85         exprs.push(call_iterbytes(field));
 86     }
 87 
 88     if exprs.len() == 0 {
 89         cx.span_bug(span, "#[deriving(IterBytes)] needs at least one field");
 90     }
 91 
 92     do exprs.slice(1, exprs.len()).iter().fold(exprs[0]) |prev, me| {
 93         cx.expr_binary(span, and, prev, *me)
 94     }
 95 }

  42 // It used to be the case that this deriving implementation invoked
  43 // std::sys::log_str, but this isn't sufficient because it doesn't invoke the
  44 // to_str() method on each field. Hence we mirror the logic of the log_str()
  45 // method, but with tweaks to call to_str() on sub-fields.
  46 fn to_str_substructure(cx: @ExtCtxt, span: span,
  47                        substr: &Substructure) -> @expr {
  48     let to_str = cx.ident_of("to_str");
  49 
  50     let doit = |start: &str, end: @str, name: ast::ident,
  51                 fields: &[(Option<ast::ident>, @expr, ~[@expr])]| {
  52         if fields.len() == 0 {
  53             cx.expr_str_uniq(span, cx.str_of(name))
  54         } else {
  55             let buf = cx.ident_of("buf");
  56             let start = cx.str_of(name) + start;
  57             let init = cx.expr_str_uniq(span, start.to_managed());
  58             let mut stmts = ~[cx.stmt_let(span, true, buf, init)];
  59             let push_str = cx.ident_of("push_str");
  60 
  61             let push = |s: @expr| {
  62                 let ebuf = cx.expr_ident(span, buf);
  63                 let call = cx.expr_method_call(span, ebuf, push_str, ~[s]);
  64                 stmts.push(cx.stmt_expr(call));
  65             };
  66 
  67             for (i, &(name, e, _)) in fields.iter().enumerate() {
  68                 if i > 0 {
  69                     push(cx.expr_str(span, @", "));
  70                 }
  71                 match name {
  72                     None => {}
  73                     Some(id) => {
  74                         let name = cx.str_of(id) + ": ";
  75                         push(cx.expr_str(span, name.to_managed()));
  76                     }
  77                 }
  78                 push(cx.expr_method_call(span, e, to_str, ~[]));
  79             }
  80             push(cx.expr_str(span, end));
  81 
  82             cx.expr_block(cx.block(span, stmts, Some(cx.expr_ident(span, buf))))
  83         }
  84     };

  87         Struct(ref fields) => {
  88             if fields.len() == 0 || fields[0].n0_ref().is_none() {
  89                 doit("(", @")", substr.type_ident, *fields)
  90             } else {
  91                 doit("{", @"}", substr.type_ident, *fields)
  92             }
  93         }
  94 
  95         EnumMatching(_, variant, ref fields) => {
  96             match variant.node.kind {
  97                 ast::tuple_variant_kind(*) =>
  98                     doit("(", @")", variant.node.name, *fields),
  99                 ast::struct_variant_kind(*) =>
 100                     doit("{", @"}", variant.node.name, *fields),
 101             }
 102         }
 103

 105     };
 106 }

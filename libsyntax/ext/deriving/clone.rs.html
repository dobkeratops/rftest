 45  fn cs_clone(
 46      name: &str,
 47      cx: &mut ExtCtxt, trait_span: Span,
 48      substr: &Substructure) -> @Expr {
 49      let clone_ident = substr.method_ident;
 50      let ctor_ident;
 51      let all_fields;
 52      let subcall = |field: &FieldInfo|
 53          cx.expr_method_call(field.span, field.self_, clone_ident, Vec::new());
 54  
 55      match *substr.fields {
 56          Struct(ref af) => {
 57              ctor_ident = substr.type_ident;
 58              all_fields = af;
 59          }
 60          EnumMatching(_, variant, ref af) => {
 61              ctor_ident = variant.node.name;
 62              all_fields = af;
 63          },
 64          EnumNonMatching(..) => cx.span_bug(trait_span,
 65                                             format!("non-matching enum variants in `deriving({})`",
 66                                                    name)),
 67          StaticEnum(..) | StaticStruct(..) => cx.span_bug(trait_span,
 68                                                           format!("static method in `deriving({})`",
 69                                                                   name))
 70      }
 71  
 72      if all_fields.len() >= 1 && all_fields.get(0).name.is_none() {
 73          // enum-like
 74          let subcalls = all_fields.iter().map(subcall).collect();
 75          cx.expr_call_ident(trait_span, ctor_ident, subcalls)
 76      } else {
 77          // struct-like
 78          let fields = all_fields.iter().map(|field| {
 79              let ident = match field.name {
 80                  Some(i) => i,
 81                  None => cx.span_bug(trait_span,
 82                                      format!("unnamed field in normal struct in `deriving({})`",
 83                                              name))
 84              };
 85              cx.field_imm(field.span, ident, subcall(field))
 86          }).collect::<Vec<_>>();
 87  
 88          if fields.is_empty() {
 89              // no fields, so construct like `None`
 90              cx.expr_ident(trait_span, ctor_ident)
 91          } else {
 92              cx.expr_struct_ident(trait_span, ctor_ident, fields)
 93          }
 94      }
 95  }

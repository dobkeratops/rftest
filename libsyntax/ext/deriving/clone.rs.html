  69  fn cs_clone(
  70      name: &str,
  71      cx: @ExtCtxt, span: Span,
  72      substr: &Substructure) -> @Expr {
  73      let clone_ident = substr.method_ident;
  74      let ctor_ident;
  75      let all_fields;
  76      let subcall = |field|
  77          cx.expr_method_call(span, field, clone_ident, ~[]);
  78  
  79      match *substr.fields {
  80          Struct(ref af) => {
  81              ctor_ident = substr.type_ident;
  82              all_fields = af;
  83          }
  84          EnumMatching(_, variant, ref af) => {
  85              ctor_ident = variant.node.name;
  86              all_fields = af;
  87          },
  88          EnumNonMatching(*) => cx.span_bug(span,
  89                                            fmt!("Non-matching enum variants in `deriving(%s)`",
  90                                                 name)),
  91          StaticEnum(*) | StaticStruct(*) => cx.span_bug(span,
  92                                                         fmt!("Static method in `deriving(%s)`",
  93                                                              name))
  94      }
  95  
  96      match *all_fields {
  97          [(None, _, _), .. _] => {
  98              // enum-like
  99              let subcalls = all_fields.map(|&(_, self_f, _)| subcall(self_f));
 100              cx.expr_call_ident(span, ctor_ident, subcalls)
 101          },
 102          _ => {
 103              // struct-like
 104              let fields = do all_fields.map |&(o_id, self_f, _)| {
 105                  let ident = match o_id {
 106                      Some(i) => i,
 107                      None => cx.span_bug(span,
 108                                          fmt!("unnamed field in normal struct in `deriving(%s)`",
 109                                               name))
 110                  };
 111                  cx.field_imm(span, ident, subcall(self_f))
 112              };
 113  
 114              if fields.is_empty() {
 115                  // no fields, so construct like `None`
 116                  cx.expr_ident(span, ctor_ident)
 117              } else {
 118                  cx.expr_struct_ident(span, ctor_ident, fields)
 119              }
 120          }
 121      }
 122  }

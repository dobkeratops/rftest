 117 // recur along an ExpnInfo chain to find the original expression
 118 fn topmost_expn_info(expn_info: @codemap::ExpnInfo) -> @codemap::ExpnInfo {
 119     match *expn_info {
 120         ExpnInfo { call_site: ref call_site, _ } => {
 121             match call_site.expn_info {
 122                 Some(next_expn_info) => {
 123                     match *next_expn_info {
 124                         ExpnInfo {
 125                             callee: NameAndSpan { name: ref name, _ },
 126                             _
 127                         } => {
 128                             // Don't recurse into file using "include!"
 129                             if "include" == *name  {
 130                                 expn_info
 131                             } else {
 132                                 topmost_expn_info(next_expn_info)
 133                             }
 134                         }
 135                     }
 136                 },
 137                 None => expn_info
 138             }
 139         }
 140     }
 141 }

 143 // resolve a file-system path to an absolute file-system path (if it
 144 // isn't already)
 145 fn res_rel_file(cx: @ExtCtxt, sp: codemap::span, arg: &Path) -> Path {
 146     // NB: relative paths are resolved relative to the compilation unit
 147     if !arg.is_absolute {
 148         let cu = Path(cx.codemap().span_to_filename(sp));
 149         cu.dir_path().push_many(arg.components)
 150     } else {
 151         (*arg).clone()
 152     }
 153 }

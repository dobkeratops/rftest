  257  impl<'a> AstBuilder for ExtCtxt<'a> {
  258      fn path(&self, span: Span, strs: Vec<ast::Ident> ) -> ast::Path {
  259          self.path_all(span, false, strs, Vec::new(), Vec::new())
  260      }
  261      fn path_ident(&self, span: Span, id: ast::Ident) -> ast::Path {
  262          self.path(span, vec!(id))
  263      }
  264      fn path_global(&self, span: Span, strs: Vec<ast::Ident> ) -> ast::Path {
  265          self.path_all(span, true, strs, Vec::new(), Vec::new())
  266      }
  267      fn path_all(&self,
  268                  sp: Span,
  269                  global: bool,
  270                  mut idents: Vec<ast::Ident> ,
  271                  lifetimes: Vec<ast::Lifetime>,
  272                  types: Vec<P<ast::Ty>> )
  273                  -> ast::Path {
  274          let last_identifier = idents.pop().unwrap();
  275          let mut segments: Vec<ast::PathSegment> = idents.move_iter()
  276                                                        .map(|ident| {
  277              ast::PathSegment {
  278                  identifier: ident,
  279                  lifetimes: Vec::new(),
  280                  types: OwnedSlice::empty(),
  281              }
  282          }).collect();
  283          segments.push(ast::PathSegment {
  284              identifier: last_identifier,
  285              lifetimes: lifetimes,
  286              types: OwnedSlice::from_vec(types),
  287          });
  288          ast::Path {
  289              span: sp,
  290              global: global,
  291              segments: segments,
  292          }
  293      }

  661      fn expr_try(&self, sp: Span, head: @ast::Expr) -> @ast::Expr {
  662          let ok = self.ident_of("Ok");
  663          let ok_path = self.path_ident(sp, ok);
  664          let err = self.ident_of("Err");
  665          let err_path = self.path_ident(sp, err);
  666  
  667          let binding_variable = self.ident_of("__try_var");
  668          let binding_pat = self.pat_ident(sp, binding_variable);
  669          let binding_expr = self.expr_ident(sp, binding_variable);
  670  
  671          // Ok(__try_var) pattern
  672          let ok_pat = self.pat_enum(sp, ok_path, vec!(binding_pat));
  673  
  674          // Err(__try_var)  (pattern and expression resp.)
  675          let err_pat = self.pat_enum(sp, err_path, vec!(binding_pat));
  676          let err_inner_expr = self.expr_call_ident(sp, err, vec!(binding_expr));
  677          // return Err(__try_var)
  678          let err_expr = self.expr(sp, ast::ExprRet(Some(err_inner_expr)));
  679  
  680          // Ok(__try_var) => __try_var
  681          let ok_arm = self.arm(sp, vec!(ok_pat), binding_expr);
  682          // Err(__try_var) => return Err(__try_var)
  683          let err_arm = self.arm(sp, vec!(err_pat), err_expr);
  684  
  685          // match head { Ok() => ..., Err() => ... }
  686          self.expr_match(sp, head, vec!(ok_arm, err_arm))
  687      }

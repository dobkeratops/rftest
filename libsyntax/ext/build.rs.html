 376     fn stmt_let(&self, sp: span, mutbl: bool, ident: ast::ident, ex: @ast::expr) -> @ast::stmt {
 377         let pat = self.pat_ident(sp, ident);
 378         let local = @ast::Local {
 379             is_mutbl: mutbl,
 380             ty: self.ty_infer(sp),
 381             pat: pat,
 382             init: Some(ex),
 383             id: self.next_id(),
 384             span: sp,
 385         };
 386         let decl = respan(sp, ast::decl_local(local));
 387         @respan(sp, ast::stmt_decl(@decl, self.next_id()))
 388     }

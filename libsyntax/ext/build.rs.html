 235          self.path_all(span, false, strs, None, ~[])
 236      }
 237      fn path_ident(&self, span: Span, id: ast::Ident) -> ast::Path {
 238          self.path(span, ~[id])
 239      }
 240      fn path_global(&self, span: Span, strs: ~[ast::Ident]) -> ast::Path {
 241          self.path_all(span, true, strs, None, ~[])
 242      }
 243      fn path_all(&self,
 244                  sp: Span,
 245                  global: bool,
 246                  mut idents: ~[ast::Ident],
 247                  rp: Option<ast::Lifetime>,
 248                  types: ~[ast::Ty])
 249                  -> ast::Path {
 250          let last_identifier = idents.pop();
 251          let mut segments: ~[ast::PathSegment] = idents.move_iter()
 252                                                        .map(|ident| {
 253              ast::PathSegment {
 254                  identifier: ident,
 255                  lifetime: None,
 256                  types: opt_vec::Empty,
 257              }
 258          }).collect();
 259          segments.push(ast::PathSegment {
 260              identifier: last_identifier,
 261              lifetime: rp,
 262              types: opt_vec::from(types),
 263          });
 264          ast::Path {
 265              span: sp,
 266              global: global,
 267              segments: segments,
 268          }
 269      }

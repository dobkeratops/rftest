 449  fn mk_token(cx: @ExtCtxt, sp: span, tok: &token::Token) -> @ast::expr {
 450  
 451      match *tok {
 452          BINOP(binop) => {
 453              return cx.expr_call_ident(sp,
 454                                        id_ext("BINOP"),
 455                                        ~[mk_binop(cx, sp, binop)]);
 456          }
 457          BINOPEQ(binop) => {
 458              return cx.expr_call_ident(sp,
 459                                        id_ext("BINOPEQ"),
 460                                        ~[mk_binop(cx, sp, binop)]);
 461          }
 462  
 463          LIT_INT(i, ity) => {
 464              let s_ity = match ity {
 465                  ast::ty_i => ~"ty_i",
 466                  ast::ty_char => ~"ty_char",
 467                  ast::ty_i8 => ~"ty_i8",
 468                  ast::ty_i16 => ~"ty_i16",
 469                  ast::ty_i32 => ~"ty_i32",
 470                  ast::ty_i64 => ~"ty_i64"
 471              };
 472              let e_ity = cx.expr_ident(sp, id_ext(s_ity));
 473  
 474              let e_i64 = cx.expr_lit(sp, ast::lit_int(i, ast::ty_i64));
 475  
 476              return cx.expr_call_ident(sp,

 641  fn expand_tts(cx: @ExtCtxt,
 642                sp: span,
 643                tts: &[ast::token_tree]) -> @ast::expr {
 644  
 645      // NB: It appears that the main parser loses its mind if we consider
 646      // $foo as a tt_nonterminal during the main parse, so we have to re-parse
 647      // under quote_depth > 0. This is silly and should go away; the _guess_ is
 648      // it has to do with transition away from supporting old-style macros, so
 649      // try removing it when enough of them are gone.
 650  
 651      let p = parse::new_parser_from_tts(
 652          cx.parse_sess(),
 653          cx.cfg(),
 654          tts.to_owned()
 655      );
 656      *p.quote_depth += 1u;
 657      let tts = p.parse_all_token_trees();
 658      p.abort_if_errors();
 659  
 660      // We want to emit a block expression that does a sequence of 'use's to
 661      // import the runtime module, followed by a tt-building expression.
 662  
 663      let uses = ~[ cx.view_use_glob(sp, ast::public,
 664                                     ids_ext(~[~"syntax",
 665                                               ~"ext",
 666                                               ~"quote",
 667                                               ~"rt"])) ];
 668  
 669      // We also bind a single value, sp, to ext_cx.call_site()
 670      //
 671      // This causes every span in a token-tree quote to be attributed to the
 672      // call site of the extension using the quote. We can't really do much
 673      // better since the source of the quote may well be in a library that
 674      // was not even parsed by this compilation run, that the user has no
 675      // source code for (eg. in libsyntax, which they're just _using_).
 676      //
 677      // The old quasiquoter had an elaborate mechanism for denoting input
 678      // file locations from which quotes originated; unfortunately this
 679      // relied on feeding the source string of the quote back into the
 680      // compiler (which we don't really want to do) and, in any case, only
 681      // pushed the problem a very small step further back: an error
 682      // resulting from a parse of the resulting quote is still attributed to
 683      // the site the string literal occurred, which was in a source file
 684      // _other_ than the one the user has control over. For example, an
 685      // error in a quote from the protocol compiler, invoked in user code
 686      // using macro_rules! for example, will be attributed to the macro_rules.rs file in
 687      // libsyntax, which the user might not even have source to (unless they
 688      // happen to have a compiler on hand). Over all, the phase distinction
 689      // just makes quotes "hard to attribute". Possibly this could be fixed
 690      // by recreating some of the original qq machinery in the tt regime
 691      // (pushing fake FileMaps onto the parser to account for original sites
 692      // of quotes, for example) but at this point it seems not likely to be
 693      // worth the hassle.
 694  
 695      let e_sp = cx.expr_method_call(sp,
 696                                     cx.expr_ident(sp, id_ext("ext_cx")),
 697                                     id_ext("call_site"),
 698                                     ~[]);
 699  
 700      let stmt_let_sp = cx.stmt_let(sp, false,
 701                                    id_ext("sp"),
 702                                    e_sp);
 703  
 704      let stmt_let_tt = cx.stmt_let(sp, true,
 705                                    id_ext("tt"),
 706                                    cx.expr_vec_uniq(sp, ~[]));
 707  
 708      cx.expr_block(
 709          cx.block_all(sp, uses,
 710                       ~[stmt_let_sp,
 711                         stmt_let_tt] + mk_tts(cx, sp, tts),
 712                       Some(cx.expr_ident(sp, id_ext("tt")))))
 713  }

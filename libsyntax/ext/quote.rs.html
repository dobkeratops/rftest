 383

 386              return cx.expr_call_ident(sp,
 387                                        id_ext("BINOP"),
 388                                        ~[mk_binop(cx, sp, binop)]);
 389          }
 390          BINOPEQ(binop) => {
 391              return cx.expr_call_ident(sp,
 392                                        id_ext("BINOPEQ"),
 393                                        ~[mk_binop(cx, sp, binop)]);
 394          }

 396          LIT_CHAR(i) => {

 402              return cx.expr_call_ident(sp, id_ext("LIT_CHAR"), ~[e_char, e_ity]);

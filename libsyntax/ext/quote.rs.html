 449 fn mk_token(cx: @ExtCtxt, sp: span, tok: &token::Token) -> @ast::expr {
 450 
 451     match *tok {
 452         BINOP(binop) => {
 453             return cx.expr_call_ident(sp,
 454                                       id_ext("BINOP"),
 455                                       ~[mk_binop(cx, sp, binop)]);
 456         }
 457         BINOPEQ(binop) => {
 458             return cx.expr_call_ident(sp,
 459                                       id_ext("BINOPEQ"),
 460                                       ~[mk_binop(cx, sp, binop)]);
 461         }
 462 
 463         LIT_INT(i, ity) => {
 464             let s_ity = match ity {
 465                 ast::ty_i => ~"ty_i",
 466                 ast::ty_char => ~"ty_char",
 467                 ast::ty_i8 => ~"ty_i8",
 468                 ast::ty_i16 => ~"ty_i16",
 469                 ast::ty_i32 => ~"ty_i32",
 470                 ast::ty_i64 => ~"ty_i64"
 471             };
 472             let e_ity = cx.expr_ident(sp, id_ext(s_ity));
 473 
 474             let e_i64 = cx.expr_lit(sp, ast::lit_int(i, ast::ty_i64));
 475 
 476             return cx.expr_call_ident(sp,
 477                                       id_ext("LIT_INT"),
 478                                       ~[e_i64, e_ity]);
 479         }
 480 
 481         LIT_UINT(u, uty) => {
 482             let s_uty = match uty {
 483                 ast::ty_u => ~"ty_u",
 484                 ast::ty_u8 => ~"ty_u8",
 485                 ast::ty_u16 => ~"ty_u16",
 486                 ast::ty_u32 => ~"ty_u32",
 487                 ast::ty_u64 => ~"ty_u64"
 488             };
 489             let e_uty = cx.expr_ident(sp, id_ext(s_uty));
 490 
 491             let e_u64 = cx.expr_lit(sp, ast::lit_uint(u, ast::ty_u64));
 492 
 493             return cx.expr_call_ident(sp,
 494                                       id_ext("LIT_UINT"),
 495                                       ~[e_u64, e_uty]);
 496         }

 498         LIT_INT_UNSUFFIXED(i) => {

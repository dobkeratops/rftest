  37  struct Context<'a, 'b> {
  38      ecx: &'a mut ExtCtxt<'b>,
  39      fmtsp: Span,
  40  
  41      // Parsed argument expressions and the types that we've found so far for
  42      // them.
  43      args: Vec<@ast::Expr>,
  44      arg_types: Vec<Option<ArgumentType>>,
  45      // Parsed named expressions and the types that we've found for them so far.
  46      // Note that we keep a side-array of the ordering of the named arguments
  47      // found to be sure that we can translate them in the same order that they
  48      // were declared in.
  49      names: HashMap<~str, @ast::Expr>,
  50      name_types: HashMap<~str, ArgumentType>,
  51      name_ordering: Vec<~str>,
  52  
  53      // Collection of the compiled `rt::Piece` structures
  54      pieces: Vec<@ast::Expr> ,
  55      name_positions: HashMap<~str, uint>,
  56      method_statics: Vec<@ast::Item> ,
  57  
  58      // Updated as arguments are consumed or methods are entered
  59      nest_level: uint,
  60      next_arg: uint,
  61  }

  63  /// Parses the arguments from the given list of tokens, returning None
  64  /// if there's a parse error so we can continue parsing other format!
  65  /// expressions.
  66  ///
  67  /// If parsing succeeds, the second return value is:
  68  ///
  69  ///     Some((fmtstr, unnamed arguments, ordering of named arguments,
  70  ///           named arguments))
  71  fn parse_args(ecx: &mut ExtCtxt, sp: Span, tts: &[ast::TokenTree])
  72      -> (@ast::Expr, Option<(@ast::Expr, Vec<@ast::Expr>, Vec<~str>,
  73                              HashMap<~str, @ast::Expr>)>) {
  74      let mut args = Vec::new();
  75      let mut names = HashMap::<~str, @ast::Expr>::new();
  76      let mut order = Vec::new();
  77  
  78      let mut p = rsparse::new_parser_from_tts(ecx.parse_sess(),
  79                                               ecx.cfg(),
  80                                               tts.iter()
  81                                                  .map(|x| (*x).clone())
  82                                                  .collect());
  83      // Parse the leading function expression (maybe a block, maybe a path)
  84      let extra = p.parse_expr();
  85      if !p.eat(&token::COMMA) {
  86          ecx.span_err(sp, "expected token: `,`");
  87          return (extra, None);
  88      }
  89  
  90      if p.token == token::EOF {
  91          ecx.span_err(sp, "requires at least a format string argument");
  92          return (extra, None);
  93      }
  94      let fmtstr = p.parse_expr();
  95      let mut named = false;
  96      while p.token != token::EOF {
  97          if !p.eat(&token::COMMA) {
  98              ecx.span_err(sp, "expected token: `,`");
  99              return (extra, None);
 100          }
 101          if p.token == token::EOF { break } // accept trailing commas
 102          if named || (token::is_ident(&p.token) &&
 103                       p.look_ahead(1, |t| *t == token::EQ)) {
 104              named = true;
 105              let ident = match p.token {
 106                  token::IDENT(i, _) => {
 107                      p.bump();
 108                      i
 109                  }
 110                  _ if named => {
 111                      ecx.span_err(p.span,
 112                                   "expected ident, positional arguments \
 113                                   cannot follow named arguments");
 114                      return (extra, None);
 115                  }
 116                  _ => {
 117                      ecx.span_err(p.span,
 118                                   format!("expected ident for named argument, but found `{}`",
 119                                           p.this_token_to_str()));
 120                      return (extra, None);
 121                  }
 122              };
 123              let interned_name = token::get_ident(ident);
 124              let name = interned_name.get();
 125              p.expect(&token::EQ);
 126              let e = p.parse_expr();
 127              match names.find_equiv(&name) {
 128                  None => {}
 129                  Some(prev) => {
 130                      ecx.span_err(e.span, format!("duplicate argument named `{}`", name));
 131                      ecx.parse_sess.span_diagnostic.span_note(prev.span, "previously here");
 132                      continue
 133                  }
 134              }
 135              order.push(name.to_str());
 136              names.insert(name.to_str(), e);
 137          } else {
 138              args.push(p.parse_expr());
 139          }
 140      }
 141      return (extra, Some((fmtstr, args, order, names)));
 142  }

 144  impl<'a, 'b> Context<'a, 'b> {
 145      /// Verifies one piece of a parse string. All errors are not emitted as
 146      /// fatal so we can continue giving errors about this and possibly other
 147      /// format strings.
 148      fn verify_piece(&mut self, p: &parse::Piece) {
 149          match *p {
 150              parse::String(..) => {}
 151              parse::CurrentArgument => {
 152                  if self.nest_level == 0 {
 153                      self.ecx.span_err(self.fmtsp,
 154                                        "`#` reference used with nothing to \
 155                                         reference back to");
 156                  }
 157              }
 158              parse::Argument(ref arg) => {
 159                  // width/precision first, if they have implicit positional
 160                  // parameters it makes more sense to consume them first.
 161                  self.verify_count(arg.format.width);
 162                  self.verify_count(arg.format.precision);
 163  
 164                  // argument second, if it's an implicit positional parameter
 165                  // it's written second, so it should come after width/precision.
 166                  let pos = match arg.position {
 167                      parse::ArgumentNext => {
 168                          let i = self.next_arg;
 169                          if self.check_positional_ok() {
 170                              self.next_arg += 1;
 171                          }
 172                          Exact(i)
 173                      }
 174                      parse::ArgumentIs(i) => Exact(i),
 175                      parse::ArgumentNamed(s) => Named(s.to_str()),
 176                  };
 177  
 178                  // and finally the method being applied
 179                  match arg.method {
 180                      None => {
 181                          let ty = Known(arg.format.ty.to_str());
 182                          self.verify_arg_type(pos, ty);
 183                      }
 184                      Some(ref method) => { self.verify_method(pos, *method); }
 185                  }
 186              }
 187          }
 188      }
 189  
 190      fn verify_pieces(&mut self, pieces: &[parse::Piece]) {
 191          for piece in pieces.iter() {
 192              self.verify_piece(piece);
 193          }
 194      }
 195  
 196      fn verify_count(&mut self, c: parse::Count) {
 197          match c {
 198              parse::CountImplied | parse::CountIs(..) => {}
 199              parse::CountIsParam(i) => {
 200                  self.verify_arg_type(Exact(i), Unsigned);
 201              }
 202              parse::CountIsName(s) => {
 203                  self.verify_arg_type(Named(s.to_str()), Unsigned);
 204              }
 205              parse::CountIsNextParam => {
 206                  if self.check_positional_ok() {
 207                      self.verify_arg_type(Exact(self.next_arg), Unsigned);
 208                      self.next_arg += 1;
 209                  }
 210              }
 211          }
 212      }
 213  
 214      fn check_positional_ok(&mut self) -> bool {
 215          if self.nest_level != 0 {
 216              self.ecx.span_err(self.fmtsp, "cannot use implicit positional \
 217                                             arguments nested inside methods");
 218              false
 219          } else {
 220              true
 221          }
 222      }
 223  
 224      fn verify_method(&mut self, pos: Position, m: &parse::Method) {
 225          self.nest_level += 1;
 226          match *m {
 227              parse::Plural(_, ref arms, ref default) => {
 228                  let mut seen_cases = HashSet::new();
 229                  self.verify_arg_type(pos, Unsigned);
 230                  for arm in arms.iter() {
 231                      if !seen_cases.insert(arm.selector) {
 232                          match arm.selector {
 233                              parse::Keyword(name) => {
 234                                  self.ecx.span_err(self.fmtsp,
 235                                                    format!("duplicate selector \
 236                                                             `{:?}`", name));
 237                              }
 238                              parse::Literal(idx) => {
 239                                  self.ecx.span_err(self.fmtsp,
 240                                                    format!("duplicate selector \
 241                                                             `={}`", idx));
 242                              }
 243                          }
 244                      }
 245                      self.verify_pieces(arm.result);
 246                  }
 247                  self.verify_pieces(*default);
 248              }
 249              parse::Select(ref arms, ref default) => {
 250                  self.verify_arg_type(pos, String);
 251                  let mut seen_cases = HashSet::new();
 252                  for arm in arms.iter() {
 253                      if !seen_cases.insert(arm.selector) {
 254                          self.ecx.span_err(self.fmtsp,
 255                                            format!("duplicate selector `{}`",
 256                                                 arm.selector));
 257                      } else if arm.selector == "" {
 258                          self.ecx.span_err(self.fmtsp,
 259                                            "empty selector in `select`");
 260                      }
 261                      self.verify_pieces(arm.result);
 262                  }
 263                  self.verify_pieces(*default);
 264              }
 265          }
 266          self.nest_level -= 1;
 267      }

 269      fn verify_arg_type(&mut self, arg: Position, ty: ArgumentType) {
 270          match arg {
 271              Exact(arg) => {
 272                  if arg < 0 || self.args.len() <= arg {
 273                      let msg = format!("invalid reference to argument `{}` (there \
 274                                      are {} arguments)", arg, self.args.len());
 275                      self.ecx.span_err(self.fmtsp, msg);
 276                      return;
 277                  }
 278                  {
 279                      let arg_type = match self.arg_types.get(arg) {
 280                          &None => None,
 281                          &Some(ref x) => Some(x)
 282                      };
 283                      self.verify_same(self.args.get(arg).span, &ty, arg_type);
 284                  }
 285                  if self.arg_types.get(arg).is_none() {
 286                      *self.arg_types.get_mut(arg) = Some(ty);
 287                  }
 288              }
 289  
 290              Named(name) => {
 291                  let span = match self.names.find(&name) {
 292                      Some(e) => e.span,
 293                      None => {
 294                          let msg = format!("there is no argument named `{}`", name);
 295                          self.ecx.span_err(self.fmtsp, msg);
 296                          return;
 297                      }
 298                  };
 299                  self.verify_same(span, &ty, self.name_types.find(&name));
 300                  if !self.name_types.contains_key(&name) {
 301                      self.name_types.insert(name.clone(), ty);
 302                  }
 303                  // Assign this named argument a slot in the arguments array if
 304                  // it hasn't already been assigned a slot.
 305                  if !self.name_positions.contains_key(&name) {
 306                      let slot = self.name_positions.len();
 307                      self.name_positions.insert(name, slot);
 308                  }
 309              }
 310          }
 311      }

 313      /// When we're keeping track of the types that are declared for certain
 314      /// arguments, we assume that `None` means we haven't seen this argument
 315      /// yet, `Some(None)` means that we've seen the argument, but no format was
 316      /// specified, and `Some(Some(x))` means that the argument was declared to
 317      /// have type `x`.
 318      ///
 319      /// Obviously `Some(Some(x)) != Some(Some(y))`, but we consider it true
 320      /// that: `Some(None) == Some(Some(x))`
 321      fn verify_same(&self,
 322                     sp: Span,
 323                     ty: &ArgumentType,
 324                     before: Option<&ArgumentType>) {
 325          let cur = match before {
 326              None => return,
 327              Some(t) => t,
 328          };
 329          if *ty == *cur {
 330              return
 331          }
 332          match (cur, ty) {
 333              (&Known(ref cur), &Known(ref ty)) => {
 334                  self.ecx.span_err(sp,
 335                                    format!("argument redeclared with type `{}` when \
 336                                             it was previously `{}`",
 337                                            *ty,
 338                                            *cur));
 339              }
 340              (&Known(ref cur), _) => {
 341                  self.ecx.span_err(sp,
 342                                    format!("argument used to format with `{}` was \
 343                                             attempted to not be used for formatting",
 344                                             *cur));
 345              }
 346              (_, &Known(ref ty)) => {
 347                  self.ecx.span_err(sp,
 348                                    format!("argument previously used as a format \
 349                                             argument attempted to be used as `{}`",
 350                                             *ty));
 351              }
 352              (_, _) => {
 353                  self.ecx.span_err(sp, "argument declared with multiple formats");
 354              }
 355          }
 356      }

 358      /// These attributes are applied to all statics that this syntax extension
 359      /// will generate.
 360      fn static_attrs(&self) -> Vec<ast::Attribute> {
 361          // Flag statics as `address_insignificant` so LLVM can merge duplicate
 362          // globals as much as possible (which we're generating a whole lot of).
 363          let unnamed = self.ecx
 364                            .meta_word(self.fmtsp,
 365                                       InternedString::new(
 366                                           "address_insignificant"));
 367          let unnamed = self.ecx.attribute(self.fmtsp, unnamed);
 368  
 369          // Do not warn format string as dead code
 370          let dead_code = self.ecx.meta_word(self.fmtsp,
 371                                             InternedString::new("dead_code"));
 372          let allow_dead_code = self.ecx.meta_list(self.fmtsp,
 373                                                   InternedString::new("allow"),
 374                                                   vec!(dead_code));
 375          let allow_dead_code = self.ecx.attribute(self.fmtsp, allow_dead_code);
 376          return vec!(unnamed, allow_dead_code);
 377      }

 411      fn trans_count(&self, c: parse::Count) -> @ast::Expr {
 412          let sp = self.fmtsp;
 413          match c {
 414              parse::CountIs(i) => {
 415                  self.ecx.expr_call_global(sp, self.rtpath("CountIs"),
 416                                            vec!(self.ecx.expr_uint(sp, i)))
 417              }
 418              parse::CountIsParam(i) => {
 419                  self.ecx.expr_call_global(sp, self.rtpath("CountIsParam"),
 420                                            vec!(self.ecx.expr_uint(sp, i)))
 421              }
 422              parse::CountImplied => {
 423                  let path = self.ecx.path_global(sp, self.rtpath("CountImplied"));
 424                  self.ecx.expr_path(path)
 425              }
 426              parse::CountIsNextParam => {
 427                  let path = self.ecx.path_global(sp, self.rtpath("CountIsNextParam"));
 428                  self.ecx.expr_path(path)
 429              }
 430              parse::CountIsName(n) => {
 431                  let i = match self.name_positions.find_equiv(&n) {
 432                      Some(&i) => i,
 433                      None => 0, // error already emitted elsewhere
 434                  };
 435                  let i = i + self.args.len();
 436                  self.ecx.expr_call_global(sp, self.rtpath("CountIsParam"),
 437                                            vec!(self.ecx.expr_uint(sp, i)))
 438              }
 439          }
 440      }

 442      fn trans_method(&mut self, method: &parse::Method) -> @ast::Expr {
 443          let sp = self.fmtsp;
 444          let method = match *method {
 445              parse::Select(ref arms, ref default) => {
 446                  let arms = arms.iter().map(|arm| {
 447                          let p = self.ecx.path_global(sp, self.rtpath("SelectArm"));
 448                          let result = arm.result.iter().map(|p| {
 449                              self.trans_piece(p)
 450                          }).collect();
 451                          let s = token::intern_and_get_ident(arm.selector);
 452                          let selector = self.ecx.expr_str(sp, s);
 453                          self.ecx.expr_struct(sp, p, vec!(
 454                                  self.ecx.field_imm(sp,
 455                                                     self.ecx.ident_of("selector"),
 456                                                     selector),
 457                                  self.ecx.field_imm(sp, self.ecx.ident_of("result"),
 458                                                     self.ecx.expr_vec_slice(sp, result))))
 459                      }).collect();
 460                  let default = default.iter().map(|p| {
 461                          self.trans_piece(p)
 462                      }).collect();
 463                  self.ecx.expr_call_global(sp, self.rtpath("Select"), vec!(
 464                          self.ecx.expr_vec_slice(sp, arms),
 465                          self.ecx.expr_vec_slice(sp, default)))
 466              }
 467              parse::Plural(offset, ref arms, ref default) => {
 468                  let offset = match offset {
 469                      Some(i) => { self.some(self.ecx.expr_uint(sp, i)) }
 470                      None => { self.none() }
 471                  };
 472                  let arms = arms.iter().map(|arm| {
 473                          let p = self.ecx.path_global(sp, self.rtpath("PluralArm"));
 474                          let result = arm.result.iter().map(|p| {
 475                                  self.trans_piece(p)
 476                              }).collect();
 477                          let (lr, selarg) = match arm.selector {
 478                              parse::Keyword(t) => {
 479                                  let p = self.ctpath(format!("{:?}", t));
 480                                  let p = self.ecx.path_global(sp, p);
 481                                  (self.rtpath("Keyword"), self.ecx.expr_path(p))
 482                              }
 483                              parse::Literal(i) => {
 484                                  (self.rtpath("Literal"), self.ecx.expr_uint(sp, i))
 485                              }
 486                          };
 487                          let selector = self.ecx.expr_call_global(sp,
 488                                                                   lr, vec!(selarg));
 489                          self.ecx.expr_struct(sp, p, vec!(
 490                                  self.ecx.field_imm(sp,
 491                                                     self.ecx.ident_of("selector"),
 492                                                     selector),
 493                                  self.ecx.field_imm(sp, self.ecx.ident_of("result"),
 494                                                     self.ecx.expr_vec_slice(sp, result))))
 495                      }).collect();
 496                  let default = default.iter().map(|p| {
 497                          self.trans_piece(p)
 498                      }).collect();
 499                  self.ecx.expr_call_global(sp, self.rtpath("Plural"), vec!(
 500                          offset,
 501                          self.ecx.expr_vec_slice(sp, arms),
 502                          self.ecx.expr_vec_slice(sp, default)))
 503              }
 504          };
 505          let life = self.ecx.lifetime(sp, self.ecx.ident_of("static").name);
 506          let ty = self.ecx.ty_path(self.ecx.path_all(
 507                  sp,
 508                  true,
 509                  self.rtpath("Method"),
 510                  vec!(life),
 511                  Vec::new()
 512                      ), None);
 513          let st = ast::ItemStatic(ty, ast::MutImmutable, method);
 514          let static_name = self.ecx.ident_of(format!("__STATIC_METHOD_{}",
 515                                                      self.method_statics.len()));
 516          let item = self.ecx.item(sp, static_name, self.static_attrs(), st);
 517          self.method_statics.push(item);
 518          self.ecx.expr_ident(sp, static_name)
 519      }

 521      /// Translate a `parse::Piece` to a static `rt::Piece`
 522      fn trans_piece(&mut self, piece: &parse::Piece) -> @ast::Expr {
 523          let sp = self.fmtsp;
 524          match *piece {
 525              parse::String(s) => {
 526                  let s = token::intern_and_get_ident(s);
 527                  self.ecx.expr_call_global(sp,
 528                                            self.rtpath("String"),
 529                                            vec!(
 530                      self.ecx.expr_str(sp, s)
 531                  ))
 532              }
 533              parse::CurrentArgument => {
 534                  let nil = self.ecx.expr_lit(sp, ast::LitNil);
 535                  self.ecx.expr_call_global(sp, self.rtpath("CurrentArgument"), vec!(nil))
 536              }
 537              parse::Argument(ref arg) => {
 538                  // Translate the position
 539                  let pos = match arg.position {
 540                      // These two have a direct mapping
 541                      parse::ArgumentNext => {
 542                          let path = self.ecx.path_global(sp,
 543                                                          self.rtpath("ArgumentNext"));
 544                          self.ecx.expr_path(path)
 545                      }
 546                      parse::ArgumentIs(i) => {
 547                          self.ecx.expr_call_global(sp, self.rtpath("ArgumentIs"),
 548                                                    vec!(self.ecx.expr_uint(sp, i)))
 549                      }
 550                      // Named arguments are converted to positional arguments at
 551                      // the end of the list of arguments
 552                      parse::ArgumentNamed(n) => {
 553                          let i = match self.name_positions.find_equiv(&n) {
 554                              Some(&i) => i,
 555                              None => 0, // error already emitted elsewhere
 556                          };
 557                          let i = i + self.args.len();
 558                          self.ecx.expr_call_global(sp, self.rtpath("ArgumentIs"),
 559                                                    vec!(self.ecx.expr_uint(sp, i)))
 560                      }
 561                  };
 562  
 563                  // Translate the format
 564                  let fill = match arg.format.fill { Some(c) => c, None => ' ' };
 565                  let fill = self.ecx.expr_lit(sp, ast::LitChar(fill as u32));
 566                  let align = match arg.format.align {
 567                      parse::AlignLeft => {
 568                          self.ecx.path_global(sp, self.parsepath("AlignLeft"))
 569                      }
 570                      parse::AlignRight => {
 571                          self.ecx.path_global(sp, self.parsepath("AlignRight"))
 572                      }
 573                      parse::AlignUnknown => {
 574                          self.ecx.path_global(sp, self.parsepath("AlignUnknown"))
 575                      }
 576                  };
 577                  let align = self.ecx.expr_path(align);
 578                  let flags = self.ecx.expr_uint(sp, arg.format.flags);
 579                  let prec = self.trans_count(arg.format.precision);
 580                  let width = self.trans_count(arg.format.width);
 581                  let path = self.ecx.path_global(sp, self.rtpath("FormatSpec"));
 582                  let fmt = self.ecx.expr_struct(sp, path, vec!(
 583                      self.ecx.field_imm(sp, self.ecx.ident_of("fill"), fill),
 584                      self.ecx.field_imm(sp, self.ecx.ident_of("align"), align),
 585                      self.ecx.field_imm(sp, self.ecx.ident_of("flags"), flags),
 586                      self.ecx.field_imm(sp, self.ecx.ident_of("precision"), prec),
 587                      self.ecx.field_imm(sp, self.ecx.ident_of("width"), width)));
 588  
 589                  // Translate the method (if any)
 590                  let method = match arg.method {
 591                      None => { self.none() }
 592                      Some(ref m) => {
 593                          let m = self.trans_method(*m);
 594                          self.some(self.ecx.expr_addr_of(sp, m))
 595                      }
 596                  };
 597                  let path = self.ecx.path_global(sp, self.rtpath("Argument"));
 598                  let s = self.ecx.expr_struct(sp, path, vec!(
 599                      self.ecx.field_imm(sp, self.ecx.ident_of("position"), pos),
 600                      self.ecx.field_imm(sp, self.ecx.ident_of("format"), fmt),
 601                      self.ecx.field_imm(sp, self.ecx.ident_of("method"), method)));
 602                  self.ecx.expr_call_global(sp, self.rtpath("Argument"), vec!(s))
 603              }
 604          }
 605      }

 607      /// Actually builds the expression which the iformat! block will be expanded
 608      /// to
 609      fn to_expr(&self, extra: @ast::Expr) -> @ast::Expr {
 610          let mut lets = Vec::new();
 611          let mut locals = Vec::new();
 612          let mut names = slice::from_fn(self.name_positions.len(), |_| None);
 613          let mut pats = Vec::new();
 614          let mut heads = Vec::new();
 615  
 616          // First, declare all of our methods that are statics
 617          for &method in self.method_statics.iter() {
 618              let decl = respan(self.fmtsp, ast::DeclItem(method));
 619              lets.push(@respan(self.fmtsp,
 620                                ast::StmtDecl(@decl, ast::DUMMY_NODE_ID)));
 621          }
 622  
 623          // Next, build up the static array which will become our precompiled
 624          // format "string"
 625          let fmt = self.ecx.expr_vec(self.fmtsp, self.pieces.clone());
 626          let piece_ty = self.ecx.ty_path(self.ecx.path_all(
 627                  self.fmtsp,
 628                  true, vec!(
 629                      self.ecx.ident_of("std"),
 630                      self.ecx.ident_of("fmt"),
 631                      self.ecx.ident_of("rt"),
 632                      self.ecx.ident_of("Piece")),
 633                  vec!(self.ecx.lifetime(self.fmtsp,
 634                                         self.ecx.ident_of("static").name)),
 635                  Vec::new()
 636              ), None);
 637          let ty = ast::TyFixedLengthVec(
 638              piece_ty,
 639              self.ecx.expr_uint(self.fmtsp, self.pieces.len())
 640          );
 641          let ty = self.ecx.ty(self.fmtsp, ty);
 642          let st = ast::ItemStatic(ty, ast::MutImmutable, fmt);
 643          let static_name = self.ecx.ident_of("__STATIC_FMTSTR");
 644          let item = self.ecx.item(self.fmtsp, static_name,
 645                                   self.static_attrs(), st);
 646          let decl = respan(self.fmtsp, ast::DeclItem(item));
 647          lets.push(@respan(self.fmtsp, ast::StmtDecl(@decl, ast::DUMMY_NODE_ID)));
 648  
 649          // Right now there is a bug such that for the expression:
 650          //      foo(bar(&1))
 651          // the lifetime of `1` doesn't outlast the call to `bar`, so it's not
 652          // vald for the call to `foo`. To work around this all arguments to the
 653          // format! string are shoved into locals. Furthermore, we shove the address
 654          // of each variable because we don't want to move out of the arguments
 655          // passed to this function.
 656          for (i, &e) in self.args.iter().enumerate() {
 657              if self.arg_types.get(i).is_none() {
 658                  continue // error already generated
 659              }
 660  
 661              let name = self.ecx.ident_of(format!("__arg{}", i));
 662              pats.push(self.ecx.pat_ident(e.span, name));
 663              heads.push(self.ecx.expr_addr_of(e.span, e));
 664              locals.push(self.format_arg(e.span, Exact(i),
 665                                          self.ecx.expr_ident(e.span, name)));
 666          }
 667          for name in self.name_ordering.iter() {
 668              let e = match self.names.find(name) {
 669                  Some(&e) if self.name_types.contains_key(name) => e,
 670                  Some(..) | None => continue
 671              };
 672  
 673              let lname = self.ecx.ident_of(format!("__arg{}", *name));
 674              pats.push(self.ecx.pat_ident(e.span, lname));
 675              heads.push(self.ecx.expr_addr_of(e.span, e));
 676              names[*self.name_positions.get(name)] =
 677                  Some(self.format_arg(e.span,
 678                                       Named((*name).clone()),
 679                                       self.ecx.expr_ident(e.span, lname)));
 680          }
 681  
 682          // Now create a vector containing all the arguments
 683          let slicename = self.ecx.ident_of("__args_vec");
 684          {
 685              let args = names.move_iter().map(|a| a.unwrap());
 686              let mut args = locals.move_iter().chain(args);
 687              let args = self.ecx.expr_vec_slice(self.fmtsp, args.collect());
 688              lets.push(self.ecx.stmt_let(self.fmtsp, false, slicename, args));
 689          }
 690  
 691          // Now create the fmt::Arguments struct with all our locals we created.
 692          let fmt = self.ecx.expr_ident(self.fmtsp, static_name);
 693          let args_slice = self.ecx.expr_ident(self.fmtsp, slicename);
 694          let result = self.ecx.expr_call_global(self.fmtsp, vec!(
 695                  self.ecx.ident_of("std"),
 696                  self.ecx.ident_of("fmt"),
 697                  self.ecx.ident_of("Arguments"),
 698                  self.ecx.ident_of("new")), vec!(fmt, args_slice));
 699  
 700          // We did all the work of making sure that the arguments
 701          // structure is safe, so we can safely have an unsafe block.
 702          let result = self.ecx.expr_block(P(ast::Block {
 703             view_items: Vec::new(),
 704             stmts: Vec::new(),
 705             expr: Some(result),
 706             id: ast::DUMMY_NODE_ID,
 707             rules: ast::UnsafeBlock(ast::CompilerGenerated),
 708             span: self.fmtsp,
 709          }));
 710          let resname = self.ecx.ident_of("__args");
 711          lets.push(self.ecx.stmt_let(self.fmtsp, false, resname, result));
 712          let res = self.ecx.expr_ident(self.fmtsp, resname);
 713          let result = self.ecx.expr_call(extra.span, extra, vec!(
 714                              self.ecx.expr_addr_of(extra.span, res)));
 715          let body = self.ecx.expr_block(self.ecx.block(self.fmtsp, lets,
 716                                                        Some(result)));
 717  
 718          // Constructs an AST equivalent to:
 719          //
 720          //      match (&arg0, &arg1) {
 721          //          (tmp0, tmp1) => body
 722          //      }
 723          //
 724          // It was:
 725          //
 726          //      let tmp0 = &arg0;
 727          //      let tmp1 = &arg1;
 728          //      body
 729          //
 730          // Because of #11585 the new temporary lifetime rule, the enclosing
 731          // statements for these temporaries become the let's themselves.
 732          // If one or more of them are RefCell's, RefCell borrow() will also
 733          // end there; they don't last long enough for body to use them. The
 734          // match expression solves the scope problem.
 735          //
 736          // Note, it may also very well be transformed to:
 737          //
 738          //      match arg0 {
 739          //          ref tmp0 => {
 740          //              match arg1 => {
 741          //                  ref tmp1 => body } } }
 742          //
 743          // But the nested match expression is proved to perform not as well
 744          // as series of let's; the first approach does.
 745          let pat = self.ecx.pat(self.fmtsp, ast::PatTup(pats));
 746          let arm = self.ecx.arm(self.fmtsp, vec!(pat), body);
 747          let head = self.ecx.expr(self.fmtsp, ast::ExprTup(heads));
 748          self.ecx.expr_match(self.fmtsp, head, vec!(arm))
 749      }

 751      fn format_arg(&self, sp: Span, argno: Position, arg: @ast::Expr)
 752                    -> @ast::Expr {
 753          let ty = match argno {
 754              Exact(ref i) => self.arg_types.get(*i).get_ref(),
 755              Named(ref s) => self.name_types.get(s)
 756          };
 757  
 758          let fmt_fn = match *ty {
 759              Known(ref tyname) => {
 760                  match tyname.as_slice() {
 761                      ""  => "secret_show",
 762                      "?" => "secret_poly",
 763                      "b" => "secret_bool",
 764                      "c" => "secret_char",
 765                      "d" | "i" => "secret_signed",
 766                      "e" => "secret_lower_exp",
 767                      "E" => "secret_upper_exp",
 768                      "f" => "secret_float",
 769                      "o" => "secret_octal",
 770                      "p" => "secret_pointer",
 771                      "s" => "secret_string",
 772                      "t" => "secret_binary",
 773                      "u" => "secret_unsigned",
 774                      "x" => "secret_lower_hex",
 775                      "X" => "secret_upper_hex",
 776                      _ => {
 777                          self.ecx.span_err(sp, format!("unknown format trait `{}`",
 778                                                        *tyname));
 779                          "dummy"
 780                      }
 781                  }
 782              }
 783              String => {
 784                  return self.ecx.expr_call_global(sp, vec!(
 785                          self.ecx.ident_of("std"),
 786                          self.ecx.ident_of("fmt"),
 787                          self.ecx.ident_of("argumentstr")), vec!(arg))
 788              }
 789              Unsigned => {
 790                  return self.ecx.expr_call_global(sp, vec!(
 791                          self.ecx.ident_of("std"),
 792                          self.ecx.ident_of("fmt"),
 793                          self.ecx.ident_of("argumentuint")), vec!(arg))
 794              }
 795          };
 796  
 797          let format_fn = self.ecx.path_global(sp, vec!(
 798                  self.ecx.ident_of("std"),
 799                  self.ecx.ident_of("fmt"),
 800                  self.ecx.ident_of(fmt_fn)));
 801          self.ecx.expr_call_global(sp, vec!(
 802                  self.ecx.ident_of("std"),
 803                  self.ecx.ident_of("fmt"),
 804                  self.ecx.ident_of("argument")), vec!(self.ecx.expr_path(format_fn), arg))
 805      }
 806  }

 808  pub fn expand_args(ecx: &mut ExtCtxt, sp: Span,
 809                     tts: &[ast::TokenTree]) -> base::MacResult {
 810  
 811      match parse_args(ecx, sp, tts) {
 812          (extra, Some((efmt, args, order, names))) => {
 813              MRExpr(expand_preparsed_format_args(ecx, sp, extra, efmt, args,
 814                                                  order, names))
 815          }
 816          (_, None) => MRExpr(ecx.expr_uint(sp, 2))
 817      }
 818  }
 819  
 820  /// Take the various parts of `format_args!(extra, efmt, args...,
 821  /// name=names...)` and construct the appropriate formatting
 822  /// expression.
 823  pub fn expand_preparsed_format_args(ecx: &mut ExtCtxt, sp: Span,
 824                                      extra: @ast::Expr,
 825                                      efmt: @ast::Expr, args: Vec<@ast::Expr>,
 826                                      name_ordering: Vec<~str>,
 827                                      names: HashMap<~str, @ast::Expr>) -> @ast::Expr {
 828      let arg_types = Vec::from_fn(args.len(), |_| None);
 829      let mut cx = Context {
 830          ecx: ecx,
 831          args: args,
 832          arg_types: arg_types,
 833          names: names,
 834          name_positions: HashMap::new(),

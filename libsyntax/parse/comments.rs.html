  58      /// remove whitespace-only lines from the start/end of lines
  59      fn vertical_trim(lines: ~[~str]) -> ~[~str] {
  60          let mut i = 0u;
  61          let mut j = lines.len();
  62          while i < j && lines[i].trim().is_empty() {
  63              i += 1u;
  64          }
  65          while j > i && lines[j - 1u].trim().is_empty() {
  66              j -= 1u;
  67          }
  68          return lines.slice(i, j).to_owned();
  69      }

  71      /// remove a "[ \t]*\*" block from each line, if possible
  72      fn horizontal_trim(lines: ~[~str]) -> ~[~str] {
  73          let mut i = uint::max_value;
  74          let mut can_trim = true;
  75          let mut first = true;
  76          for line in lines.iter() {
  77              for (j, c) in line.iter().enumerate() {
  78                  if j > i || !"* \t".contains_char(c) {
  79                      can_trim = false;
  80                      break;
  81                  }
  82                  if c == '*' {
  83                      if first {
  84                          i = j;
  85                          first = false;
  86                      } else if i != j {
  87                          can_trim = false;
  88                      }
  89                      break;
  90                  }
  91              }
  92              if i > line.len() {
  93                  can_trim = false;
  94              }
  95              if !can_trim {
  96                  break;
  97              }
  98          }
  99  
 100          if can_trim {
 101              do lines.map |line| {
 102                  line.slice(i + 1, line.len()).to_owned()
 103              }
 104          } else {
 105              lines
 106          }
 107      }
 108  
 109      if comment.starts_with("//") {
 110          // FIXME #5475:
 111          // return comment.slice(3u, comment.len()).to_owned();
 112          let r = comment.slice(3u, comment.len()); return r.to_owned();
 113      }
 114  
 115      if comment.starts_with("/*") {
 116          let lines = comment.slice(3u, comment.len() - 2u)
 117              .any_line_iter()
 118              .map(|s| s.to_owned())
 119              .collect::<~[~str]>();
 120  
 121          let lines = vertical_trim(lines);
 122          let lines = horizontal_trim(lines);
 123  
 124          return lines.connect("\n");
 125      }
 126  
 127      fail!("not a doc-comment: %s", comment);
 128  }

 130  fn read_to_eol(rdr: @mut StringReader) -> ~str {
 131      let mut val = ~"";
 132      while rdr.curr != '\n' && !is_eof(rdr) {
 133          val.push_char(rdr.curr);
 134          bump(rdr);
 135      }
 136      if rdr.curr == '\n' { bump(rdr); }
 137      return val;
 138  }

 206  // Returns None if the first col chars of s contain a non-whitespace char.
 207  // Otherwise returns Some(k) where k is first char offset after that leading
 208  // whitespace.  Note k may be outside bounds of s.
 209  fn all_whitespace(s: &str, col: CharPos) -> Option<uint> {
 210      let len = s.len();
 211      let mut col = col.to_uint();
 212      let mut cursor: uint = 0;
 213      while col > 0 && cursor < len {
 214          let r: str::CharRange = s.char_range_at(cursor);
 215          if !r.ch.is_whitespace() {
 216              return None;
 217          }
 218          cursor = r.next;
 219          col -= 1;
 220      }
 221      return Some(cursor);
 222  }

 224  fn trim_whitespace_prefix_and_push_line(lines: &mut ~[~str],
 225                                          s: ~str, col: CharPos) {
 226      let len = s.len();
 227      let s1 = match all_whitespace(s, col) {
 228          Some(col) => {
 229              if col < len {
 230                  s.slice(col, len).to_owned()
 231              } else {  ~"" }
 232          }
 233          None => s,
 234      };
 235      debug!("pushing line: %s", s1);
 236      lines.push(s1);
 237  }

 239  fn read_block_comment(rdr: @mut StringReader,
 240                        code_to_the_left: bool,
 241                        comments: &mut ~[cmnt]) {
 242      debug!(">>> block comment");
 243      let p = rdr.last_pos;
 244      let mut lines: ~[~str] = ~[];
 245      let col: CharPos = rdr.col;
 246      bump(rdr);
 247      bump(rdr);
 248  
 249      let mut curr_line = ~"/*";
 250  
 251      // doc-comments are not really comments, they are attributes
 252      if rdr.curr == '*' || rdr.curr == '!' {
 253          while !(rdr.curr == '*' && nextch(rdr) == '/') && !is_eof(rdr) {
 254              curr_line.push_char(rdr.curr);
 255              bump(rdr);
 256          }
 257          if !is_eof(rdr) {
 258              curr_line.push_str("*/");
 259              bump(rdr);
 260              bump(rdr);
 261          }
 262          if !is_block_non_doc_comment(curr_line) { return; }
 263          assert!(!curr_line.contains_char('\n'));
 264          lines.push(curr_line);
 265      } else {
 266          let mut level: int = 1;
 267          while level > 0 {
 268              debug!("=== block comment level %d", level);
 269              if is_eof(rdr) {
 270                  (rdr as @reader).fatal(~"unterminated block comment");
 271              }
 272              if rdr.curr == '\n' {
 273                  trim_whitespace_prefix_and_push_line(&mut lines, curr_line,
 274                                                       col);
 275                  curr_line = ~"";
 276                  bump(rdr);
 277              } else {
 278                  curr_line.push_char(rdr.curr);
 279                  if rdr.curr == '/' && nextch(rdr) == '*' {
 280                      bump(rdr);
 281                      bump(rdr);
 282                      curr_line.push_char('*');
 283                      level += 1;
 284                  } else {
 285                      if rdr.curr == '*' && nextch(rdr) == '/' {
 286                          bump(rdr);
 287                          bump(rdr);
 288                          curr_line.push_char('/');
 289                          level -= 1;
 290                      } else { bump(rdr); }
 291                  }
 292              }
 293          }
 294          if curr_line.len() != 0 {
 295              trim_whitespace_prefix_and_push_line(&mut lines, curr_line, col);
 296          }
 297      }
 298  
 299      let mut style = if code_to_the_left { trailing } else { isolated };
 300      consume_non_eol_whitespace(rdr);
 301      if !is_eof(rdr) && rdr.curr != '\n' && lines.len() == 1u {
 302          style = mixed;
 303      }
 304      debug!("<<< block comment");
 305      comments.push(cmnt {style: style, lines: lines, pos: p});
 306  }

 314  fn consume_comment(rdr: @mut StringReader,
 315                     code_to_the_left: bool,
 316                     comments: &mut ~[cmnt]) {

 324      } else { fail!(); }

 326  }

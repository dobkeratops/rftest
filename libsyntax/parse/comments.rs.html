  39  pub fn is_doc_comment(s: &str) -> bool {
  40      (s.starts_with("///") && !is_line_non_doc_comment(s)) ||
  41      s.starts_with("//!") ||
  42      (s.starts_with("/**") && !is_block_non_doc_comment(s)) ||
  43      s.starts_with("/*!")
  44  }

  46  pub fn doc_comment_style(comment: &str) -> ast::AttrStyle {
  47      assert!(is_doc_comment(comment));
  48      if comment.starts_with("//!") || comment.starts_with("/*!") {
  49          ast::AttrInner
  50      } else {
  51          ast::AttrOuter
  52      }
  53  }

  55  pub fn strip_doc_comment_decoration(comment: &str) -> ~str {
  56      /// remove whitespace-only lines from the start/end of lines
  57      fn vertical_trim(lines: Vec<~str> ) -> Vec<~str> {
  58          let mut i = 0u;
  59          let mut j = lines.len();
  60          // first line of all-stars should be omitted
  61          if lines.len() > 0 && lines.get(0).chars().all(|c| c == '*') {
  62              i += 1;
  63          }
  64          while i < j && lines.get(i).trim().is_empty() {
  65              i += 1;
  66          }
  67          // like the first, a last line of all stars should be omitted
  68          if j > i && lines.get(j - 1).chars().skip(1).all(|c| c == '*') {
  69              j -= 1;
  70          }
  71          while j > i && lines.get(j - 1).trim().is_empty() {
  72              j -= 1;
  73          }
  74          return lines.slice(i, j).iter().map(|x| (*x).clone()).collect();
  75      }
  76  
  77      /// remove a "[ \t]*\*" block from each line, if possible
  78      fn horizontal_trim(lines: Vec<~str> ) -> Vec<~str> {
  79          let mut i = uint::MAX;
  80          let mut can_trim = true;
  81          let mut first = true;
  82          for line in lines.iter() {
  83              for (j, c) in line.chars().enumerate() {
  84                  if j > i || !"* \t".contains_char(c) {
  85                      can_trim = false;
  86                      break;
  87                  }
  88                  if c == '*' {
  89                      if first {
  90                          i = j;
  91                          first = false;
  92                      } else if i != j {
  93                          can_trim = false;
  94                      }
  95                      break;
  96                  }
  97              }
  98              if i > line.len() {
  99                  can_trim = false;
 100              }
 101              if !can_trim {
 102                  break;
 103              }
 104          }
 105  
 106          if can_trim {
 107              lines.iter().map(|line| line.slice(i + 1, line.len()).to_owned()).collect()
 108          } else {
 109              lines
 110          }
 111      }
 112  
 113      // one-line comments lose their prefix
 114      static ONLINERS: &'static [&'static str] = &["///!", "///", "//!", "//"];
 115      for prefix in ONLINERS.iter() {
 116          if comment.starts_with(*prefix) {
 117              return comment.slice_from(prefix.len()).to_owned();
 118          }
 119      }
 120  
 121      if comment.starts_with("/*") {
 122          let lines = comment.slice(3u, comment.len() - 2u)
 123              .lines_any()
 124              .map(|s| s.to_owned())
 125              .collect::<Vec<~str> >();
 126  
 127          let lines = vertical_trim(lines);
 128          let lines = horizontal_trim(lines);
 129  
 130          return lines.connect("\n");
 131      }
 132  
 133      fail!("not a doc-comment: {}", comment);
 134  }

 136  fn read_to_eol(rdr: &mut StringReader) -> ~str {
 137      let mut val = ~"";
 138      while !rdr.curr_is('\n') && !is_eof(rdr) {
 139          val.push_char(rdr.curr.unwrap());
 140          bump(rdr);
 141      }
 142      if rdr.curr_is('\n') { bump(rdr); }
 143      return val;
 144  }

 146  fn read_one_line_comment(rdr: &mut StringReader) -> ~str {
 147      let val = read_to_eol(rdr);
 148      assert!((val[0] == '/' as u8 && val[1] == '/' as u8) ||
 149                   (val[0] == '#' as u8 && val[1] == '!' as u8));
 150      return val;
 151  }

 153  fn consume_non_eol_whitespace(rdr: &mut StringReader) {
 154      while is_whitespace(rdr.curr) && !rdr.curr_is('\n') && !is_eof(rdr) {
 155          bump(rdr);
 156      }
 157  }

 159  fn push_blank_line_comment(rdr: &StringReader, comments: &mut Vec<Comment>) {
 160      debug!(">>> blank-line comment");
 161      comments.push(Comment {
 162          style: BlankLine,
 163          lines: Vec::new(),
 164          pos: rdr.last_pos,
 165      });
 166  }

 168  fn consume_whitespace_counting_blank_lines(rdr: &mut StringReader,
 169                                             comments: &mut Vec<Comment>) {
 170      while is_whitespace(rdr.curr) && !is_eof(rdr) {
 171          if rdr.col == CharPos(0u) && rdr.curr_is('\n') {
 172              push_blank_line_comment(rdr, &mut *comments);
 173          }
 174          bump(rdr);
 175      }
 176  }

 179  fn read_shebang_comment(rdr: &mut StringReader, code_to_the_left: bool,
 180                          comments: &mut Vec<Comment>) {
 181      debug!(">>> shebang comment");
 182      let p = rdr.last_pos;
 183      debug!("<<< shebang comment");
 184      comments.push(Comment {
 185          style: if code_to_the_left { Trailing } else { Isolated },
 186          lines: vec!(read_one_line_comment(rdr)),
 187          pos: p
 188      });
 189  }

 191  fn read_line_comments(rdr: &mut StringReader, code_to_the_left: bool,
 192                        comments: &mut Vec<Comment>) {
 193      debug!(">>> line comments");
 194      let p = rdr.last_pos;
 195      let mut lines: Vec<~str> = Vec::new();
 196      while rdr.curr_is('/') && nextch_is(rdr, '/') {
 197          let line = read_one_line_comment(rdr);
 198          debug!("{}", line);
 199          if is_doc_comment(line) { // doc-comments are not put in comments
 200              break;
 201          }
 202          lines.push(line);
 203          consume_non_eol_whitespace(rdr);
 204      }
 205      debug!("<<< line comments");
 206      if !lines.is_empty() {
 207          comments.push(Comment {
 208              style: if code_to_the_left { Trailing } else { Isolated },
 209              lines: lines,
 210              pos: p
 211          });
 212      }
 213  }

 215  // Returns None if the first col chars of s contain a non-whitespace char.
 216  // Otherwise returns Some(k) where k is first char offset after that leading
 217  // whitespace.  Note k may be outside bounds of s.
 218  fn all_whitespace(s: &str, col: CharPos) -> Option<uint> {
 219      let len = s.len();
 220      let mut col = col.to_uint();
 221      let mut cursor: uint = 0;
 222      while col > 0 && cursor < len {
 223          let r: str::CharRange = s.char_range_at(cursor);
 224          if !r.ch.is_whitespace() {
 225              return None;
 226          }
 227          cursor = r.next;
 228          col -= 1;
 229      }
 230      return Some(cursor);
 231  }

 233  fn trim_whitespace_prefix_and_push_line(lines: &mut Vec<~str> ,
 234                                          s: ~str, col: CharPos) {
 235      let len = s.len();
 236      let s1 = match all_whitespace(s, col) {
 237          Some(col) => {
 238              if col < len {
 239                  s.slice(col, len).to_owned()
 240              } else {  ~"" }
 241          }
 242          None => s,
 243      };
 244      debug!("pushing line: {}", s1);
 245      lines.push(s1);
 246  }

 248  fn read_block_comment(rdr: &mut StringReader,
 249                        code_to_the_left: bool,
 250                        comments: &mut Vec<Comment> ) {
 251      debug!(">>> block comment");
 252      let p = rdr.last_pos;
 253      let mut lines: Vec<~str> = Vec::new();
 254      let col = rdr.col;
 255      bump(rdr);
 256      bump(rdr);
 257  
 258      let mut curr_line = ~"/*";
 259  
 260      // doc-comments are not really comments, they are attributes
 261      if (rdr.curr_is('*') && !nextch_is(rdr, '*')) || rdr.curr_is('!') {
 262          while !(rdr.curr_is('*') && nextch_is(rdr, '/')) && !is_eof(rdr) {
 263              curr_line.push_char(rdr.curr.unwrap());
 264              bump(rdr);
 265          }
 266          if !is_eof(rdr) {
 267              curr_line.push_str("*/");
 268              bump(rdr);
 269              bump(rdr);
 270          }
 271          if !is_block_non_doc_comment(curr_line) { return; }
 272          assert!(!curr_line.contains_char('\n'));
 273          lines.push(curr_line);
 274      } else {
 275          let mut level: int = 1;
 276          while level > 0 {
 277              debug!("=== block comment level {}", level);
 278              if is_eof(rdr) {
 279                  rdr.fatal(~"unterminated block comment");
 280              }
 281              if rdr.curr_is('\n') {
 282                  trim_whitespace_prefix_and_push_line(&mut lines, curr_line,
 283                                                       col);
 284                  curr_line = ~"";
 285                  bump(rdr);
 286              } else {
 287                  curr_line.push_char(rdr.curr.unwrap());
 288                  if rdr.curr_is('/') && nextch_is(rdr, '*') {
 289                      bump(rdr);
 290                      bump(rdr);
 291                      curr_line.push_char('*');
 292                      level += 1;
 293                  } else {
 294                      if rdr.curr_is('*') && nextch_is(rdr, '/') {
 295                          bump(rdr);
 296                          bump(rdr);
 297                          curr_line.push_char('/');
 298                          level -= 1;
 299                      } else { bump(rdr); }
 300                  }
 301              }
 302          }
 303          if curr_line.len() != 0 {
 304              trim_whitespace_prefix_and_push_line(&mut lines, curr_line, col);
 305          }
 306      }
 307  
 308      let mut style = if code_to_the_left { Trailing } else { Isolated };
 309      consume_non_eol_whitespace(rdr);
 310      if !is_eof(rdr) && !rdr.curr_is('\n') && lines.len() == 1u {
 311          style = Mixed;
 312      }
 313      debug!("<<< block comment");
 314      comments.push(Comment {style: style, lines: lines, pos: p});
 315  }

 317  fn peeking_at_comment(rdr: &StringReader) -> bool {
 318      return (rdr.curr_is('/') && nextch_is(rdr, '/')) ||
 319           (rdr.curr_is('/') && nextch_is(rdr, '*')) ||
 320           // consider shebangs comments, but not inner attributes
 321           (rdr.curr_is('#') && nextch_is(rdr, '!') &&
 322            !lexer::nextnextch_is(rdr, '['));
 323  }
 324  
 325  fn consume_comment(rdr: &mut StringReader,
 326                     code_to_the_left: bool,
 327                     comments: &mut Vec<Comment> ) {
 328      debug!(">>> consume comment");
 329      if rdr.curr_is('/') && nextch_is(rdr, '/') {
 330          read_line_comments(rdr, code_to_the_left, comments);
 331      } else if rdr.curr_is('/') && nextch_is(rdr, '*') {
 332          read_block_comment(rdr, code_to_the_left, comments);
 333      } else if rdr.curr_is('#') && nextch_is(rdr, '!') {
 334          read_shebang_comment(rdr, code_to_the_left, comments);
 335      } else { fail!(); }
 336      debug!("<<< consume comment");
 337  }

 345  // it appears this function is called only from pprust... that's
 346  // probably not a good thing.
 347  pub fn gather_comments_and_literals(span_diagnostic:
 348                                          &diagnostic::SpanHandler,
 349                                      path: ~str,
 350                                      srdr: &mut io::Reader)
 351                                   -> (Vec<Comment>, Vec<Literal>) {
 352      let src = srdr.read_to_end().unwrap();
 353      let src = str::from_utf8(src.as_slice()).unwrap().to_owned();
 354      let cm = CodeMap::new();
 355      let filemap = cm.new_filemap(path, src);
 356      let mut rdr = lexer::new_low_level_string_reader(span_diagnostic, filemap);
 357  
 358      let mut comments: Vec<Comment> = Vec::new();
 359      let mut literals: Vec<Literal> = Vec::new();
 360      let mut first_read: bool = true;
 361      while !is_eof(&rdr) {
 362          loop {
 363              let mut code_to_the_left = !first_read;
 364              consume_non_eol_whitespace(&mut rdr);
 365              if rdr.curr_is('\n') {
 366                  code_to_the_left = false;
 367                  consume_whitespace_counting_blank_lines(&mut rdr, &mut comments);
 368              }
 369              while peeking_at_comment(&rdr) {
 370                  consume_comment(&mut rdr, code_to_the_left, &mut comments);
 371                  consume_whitespace_counting_blank_lines(&mut rdr, &mut comments);
 372              }
 373              break;
 374          }
 375  
 376  
 377          let bstart = rdr.last_pos;
 378          rdr.next_token();
 379          //discard, and look ahead; we're working with internal state
 380          let TokenAndSpan {tok: tok, sp: sp} = rdr.peek();
 381          if token::is_lit(&tok) {
 382              with_str_from(&rdr, bstart, |s| {
 383                  debug!("tok lit: {}", s);
 384                  literals.push(Literal {lit: s.to_owned(), pos: sp.lo});
 385              })
 386          } else {
 387              debug!("tok: {}", token::to_str(&tok));
 388          }
 389          first_read = false;
 390      }
 391  
 392      (comments, literals)
 393  }

 127      // matches inner_attrs* outer_attr?

 137                      self.parse_attribute(true)
 138                  }
 139                  token::POUND => {
 140                      self.parse_attribute(true)
 141                  }
 142                  token::DOC_COMMENT(s) => {
 143                      self.bump();

 145                                                  self.span.lo,

 148                  _ => {
 149                      break;
 150                  }
 151              };

 153                  inner_attrs.push(attr);

  68  /* For comments.rs, which hackily pokes into 'pos' and 'curr' */
  69  pub fn new_low_level_string_reader(span_diagnostic: @mut span_handler,
  70                                     filemap: @codemap::FileMap)
  71                                  -> @mut StringReader {
  72      // Force the initial reader bump to start on a fresh line
  73      let initial_char = '\n';
  74      let r = @mut StringReader {
  75          span_diagnostic: span_diagnostic,
  76          src: filemap.src,
  77          pos: filemap.start_pos,
  78          last_pos: filemap.start_pos,
  79          col: CharPos(0),
  80          curr: initial_char,
  81          filemap: filemap,
  82          /* dummy values; not read */
  83          peek_tok: token::EOF,
  84          peek_span: codemap::dummy_sp()
  85      };
  86      bump(r);
  87      return r;
  88  }

 107  impl reader for StringReader {
 108      fn is_eof(@mut self) -> bool { is_eof(self) }
 109      // return the next token. EFFECT: advances the string_reader.
 110      fn next_token(@mut self) -> TokenAndSpan {
 111          let ret_val = TokenAndSpan {
 112              tok: util::replace(&mut self.peek_tok, token::UNDERSCORE),
 113              sp: self.peek_span,

 115          string_advance_token(self);
 116          ret_val
 117      }
 118      fn fatal(@mut self, m: ~str) -> ! {
 119          self.span_diagnostic.span_fatal(self.peek_span, m)
 120      }
 121      fn span_diag(@mut self) -> @mut span_handler { self.span_diagnostic }
 122      fn peek(@mut self) -> TokenAndSpan {
 123          // XXX(pcwalton): Bad copy!
 124          TokenAndSpan {
 125              tok: self.peek_tok.clone(),
 126              sp: self.peek_span,
 127          }
 128      }

 162  // report a lexical error spanning [`from_pos`, `to_pos`), appending an
 163  // escaped character to the error message
 164  fn fatal_span_char(rdr: @mut StringReader,
 165                     from_pos: BytePos,
 166                     to_pos: BytePos,
 167                     m: ~str,
 168                     c: char)
 169                  -> ! {

 171      m.push_str(": ");
 172      char::escape_default(c, |c| m.push_char(c));
 173      fatal_span(rdr, from_pos, to_pos, m);
 174  }

 176  // report a lexical error spanning [`from_pos`, `to_pos`), appending the
 177  // offending string to the error message
 178  fn fatal_span_verbose(rdr: @mut StringReader,
 179                        from_pos: BytePos,
 180                        to_pos: BytePos,

 182                     -> ! {

 192  // EFFECT: advance peek_tok and peek_span to refer to the next token.
 193  // EFFECT: update the interner, maybe.
 194  fn string_advance_token(r: @mut StringReader) {
 195      match (consume_whitespace_and_comments(r)) {
 196          Some(comment) => {
 197              r.peek_span = comment.sp;
 198              r.peek_tok = comment.tok;
 199          },
 200          None => {
 201              if is_eof(r) {
 202                  r.peek_tok = token::EOF;
 203              } else {
 204                  let start_bytepos = r.last_pos;
 205                  r.peek_tok = next_token_inner(r);
 206                  r.peek_span = codemap::mk_sp(start_bytepos, r.last_pos);
 207              };
 208          }
 209      }
 210  }

 222  // EFFECT: advance the StringReader by one character. If a newline is
 223  // discovered, add it to the FileMap's list of line start offsets.
 224  pub fn bump(rdr: &mut StringReader) {
 225      rdr.last_pos = rdr.pos;
 226      let current_byte_offset = byte_offset(rdr, rdr.pos).to_uint();
 227      if current_byte_offset < (rdr.src).len() {
 228          assert!(rdr.curr != unsafe { transmute(-1u32) }); // FIXME: #8971: unsound
 229          let last_char = rdr.curr;
 230          let next = rdr.src.char_range_at(current_byte_offset);
 231          let byte_offset_diff = next.next - current_byte_offset;
 232          rdr.pos = rdr.pos + BytePos(byte_offset_diff);
 233          rdr.curr = next.ch;
 234          rdr.col = rdr.col + CharPos(1u);
 235          if last_char == '\n' {
 236              rdr.filemap.next_line(rdr.last_pos);
 237              rdr.col = CharPos(0u);
 238          }
 239  
 240          if byte_offset_diff > 1 {
 241              rdr.filemap.record_multibyte_char(
 242                  BytePos(current_byte_offset), byte_offset_diff);
 243          }
 244      } else {
 245          rdr.curr = unsafe { transmute(-1u32) }; // FIXME: #8971: unsound
 246      }
 247  }

 251  pub fn nextch(rdr: @mut StringReader) -> char {
 252      let offset = byte_offset(rdr, rdr.pos).to_uint();
 253      if offset < (rdr.src).len() {
 254          return rdr.src.char_at(offset);
 255      } else { return unsafe { transmute(-1u32) }; } // FIXME: #8971: unsound
 256  }

 260  fn hex_digit_val(c: char) -> int {
 261      if in_range(c, '0', '9') { return (c as int) - ('0' as int); }
 262      if in_range(c, 'a', 'f') { return (c as int) - ('a' as int) + 10; }
 263      if in_range(c, 'A', 'F') { return (c as int) - ('A' as int) + 10; }
 264      fail!();
 265  }

 286  // EFFECT: eats whitespace and comments.

 294  pub fn is_line_non_doc_comment(s: &str) -> bool {
 295      let s = s.trim_right();
 296      s.len() > 3 && s.iter().all(|ch| ch == '/')
 297  }
 298  
 299  // PRECONDITION: rdr.curr is not whitespace
 300  // EFFECT: eats any kind of comment.
 301  // returns a Some(sugared-doc-attr) if one exists, None otherwise
 302  fn consume_any_line_comment(rdr: @mut StringReader)
 303                           -> Option<TokenAndSpan> {
 304      if rdr.curr == '/' {
 305          match nextch(rdr) {
 306            '/' => {
 307              bump(rdr);
 308              bump(rdr);
 309              // line comments starting with "///" or "//!" are doc-comments
 310              if rdr.curr == '/' || rdr.curr == '!' {
 311                  let start_bpos = rdr.pos - BytePos(3u);
 312                  while rdr.curr != '\n' && !is_eof(rdr) {
 313                      bump(rdr);
 314                  }
 315                  let ret = do with_str_from(rdr, start_bpos) |string| {
 316                      // but comments with only more "/"s are not
 317                      if !is_line_non_doc_comment(string) {
 318                          Some(TokenAndSpan{
 319                              tok: token::DOC_COMMENT(str_to_ident(string)),
 320                              sp: codemap::mk_sp(start_bpos, rdr.pos)
 321                          })
 322                      } else {
 323                          None
 324                      }
 325                  };
 326  
 327                  if ret.is_some() {
 328                      return ret;
 329                  }
 330              } else {
 331                  while rdr.curr != '\n' && !is_eof(rdr) { bump(rdr); }
 332              }
 333              // Restart whitespace munch.
 334              return consume_whitespace_and_comments(rdr);
 335            }
 336            '*' => { bump(rdr); bump(rdr); return consume_block_comment(rdr); }
 337            _ => ()
 338          }
 339      } else if rdr.curr == '#' {
 340          if nextch(rdr) == '!' {
 341              // I guess this is the only way to figure out if
 342              // we're at the beginning of the file...
 343              let cmap = @CodeMap::new();
 344              (*cmap).files.push(rdr.filemap);
 345              let loc = cmap.lookup_char_pos_adj(rdr.last_pos);
 346              if loc.line == 1u && loc.col == CharPos(0u) {
 347                  while rdr.curr != '\n' && !is_eof(rdr) { bump(rdr); }
 348                  return consume_whitespace_and_comments(rdr);
 349              }
 350          }
 351      }
 352      return None;
 353  }

 355  pub fn is_block_non_doc_comment(s: &str) -> bool {
 356      assert!(s.len() >= 1u);
 357      s.slice(1u, s.len() - 1u).iter().all(|ch| ch == '*')
 358  }

 360  // might return a sugared-doc-attr
 361  fn consume_block_comment(rdr: @mut StringReader)
 362                        -> Option<TokenAndSpan> {
 363      // block comments starting with "/**" or "/*!" are doc-comments
 364      let res = if rdr.curr == '*' || rdr.curr == '!' {
 365          let start_bpos = rdr.pos - BytePos(3u);
 366          while !(rdr.curr == '*' && nextch(rdr) == '/') && !is_eof(rdr) {
 367              bump(rdr);
 368          }
 369          if is_eof(rdr) {
 370              fatal_span(rdr, start_bpos, rdr.last_pos,
 371                         ~"unterminated block doc-comment");
 372          } else {
 373              bump(rdr);
 374              bump(rdr);
 375              do with_str_from(rdr, start_bpos) |string| {
 376                  // but comments with only "*"s between two "/"s are not
 377                  if !is_block_non_doc_comment(string) {
 378                      Some(TokenAndSpan{
 379                           tok: token::DOC_COMMENT(str_to_ident(string)),
 380                           sp: codemap::mk_sp(start_bpos, rdr.pos)
 381                           })
 382                  } else {
 383                      None
 384                  }
 385              }
 386          }
 387      } else {
 388          let start_bpos = rdr.last_pos - BytePos(2u);
 389          loop {
 390              if is_eof(rdr) {
 391                  fatal_span(rdr, start_bpos, rdr.last_pos,
 392                             ~"unterminated block comment");
 393              }
 394              if rdr.curr == '*' && nextch(rdr) == '/' {
 395                  bump(rdr);
 396                  bump(rdr);
 397                  break;
 398              } else {
 399                  bump(rdr);
 400              }
 401          }
 402          None
 403      };
 404      // restart whitespace munch.
 405  
 406     if res.is_some() { res } else { consume_whitespace_and_comments(rdr) }
 407  }

 409  fn scan_exponent(rdr: @mut StringReader, start_bpos: BytePos) -> Option<~str> {
 410      let mut c = rdr.curr;
 411      let mut rslt = ~"";
 412      if c == 'e' || c == 'E' {
 413          rslt.push_char(c);
 414          bump(rdr);
 415          c = rdr.curr;
 416          if c == '-' || c == '+' {
 417              rslt.push_char(c);
 418              bump(rdr);
 419          }
 420          let exponent = scan_digits(rdr, 10u);
 421          if exponent.len() > 0u {
 422              return Some(rslt + exponent);
 423          } else {
 424              fatal_span(rdr, start_bpos, rdr.last_pos,
 425                         ~"scan_exponent: bad fp literal");
 426          }
 427      } else { return None::<~str>; }
 428  }

 430  fn scan_digits(rdr: @mut StringReader, radix: uint) -> ~str {
 431      let mut rslt = ~"";
 432      loop {
 433          let c = rdr.curr;
 434          if c == '_' { bump(rdr); loop; }
 435          match char::to_digit(c, radix) {
 436            Some(_) => {
 437              rslt.push_char(c);
 438              bump(rdr);
 439            }
 440            _ => return rslt
 441          }
 442      };
 443  }

 445  fn scan_number(c: char, rdr: @mut StringReader) -> token::Token {
 446      let mut num_str;
 447      let mut base = 10u;
 448      let mut c = c;
 449      let mut n = nextch(rdr);
 450      let start_bpos = rdr.last_pos;
 451      if c == '0' && n == 'x' {
 452          bump(rdr);
 453          bump(rdr);
 454          base = 16u;
 455      } else if c == '0' && n == 'b' {
 456          bump(rdr);
 457          bump(rdr);
 458          base = 2u;
 459      }
 460      num_str = scan_digits(rdr, base);
 461      c = rdr.curr;
 462      nextch(rdr);
 463      if c == 'u' || c == 'i' {
 464          let signed = c == 'i';
 465          let mut tp = {
 466              if signed { either::Left(ast::ty_i) }
 467              else { either::Right(ast::ty_u) }
 468          };
 469          bump(rdr);
 470          c = rdr.curr;
 471          if c == '8' {
 472              bump(rdr);
 473              tp = if signed { either::Left(ast::ty_i8) }
 474                        else { either::Right(ast::ty_u8) };
 475          }
 476          n = nextch(rdr);
 477          if c == '1' && n == '6' {
 478              bump(rdr);
 479              bump(rdr);
 480              tp = if signed { either::Left(ast::ty_i16) }
 481                        else { either::Right(ast::ty_u16) };
 482          } else if c == '3' && n == '2' {
 483              bump(rdr);
 484              bump(rdr);
 485              tp = if signed { either::Left(ast::ty_i32) }
 486                        else { either::Right(ast::ty_u32) };
 487          } else if c == '6' && n == '4' {
 488              bump(rdr);
 489              bump(rdr);
 490              tp = if signed { either::Left(ast::ty_i64) }
 491                        else { either::Right(ast::ty_u64) };
 492          }
 493          if num_str.len() == 0u {
 494              fatal_span(rdr, start_bpos, rdr.last_pos,
 495                         ~"no valid digits found for number");
 496          }
 497          let parsed = match from_str_radix::<u64>(num_str, base as uint) {
 498              Some(p) => p,
 499              None => fatal_span(rdr, start_bpos, rdr.last_pos,
 500                                 ~"int literal is too large")
 501          };
 502  
 503          match tp {
 504            either::Left(t) => return token::LIT_INT(parsed as i64, t),
 505            either::Right(t) => return token::LIT_UINT(parsed, t)
 506          }
 507      }
 508      let mut is_float = false;
 509      if rdr.curr == '.' && !(ident_start(nextch(rdr)) || nextch(rdr) == '.') {
 510          is_float = true;
 511          bump(rdr);
 512          let dec_part = scan_digits(rdr, 10u);
 513          num_str.push_char('.');
 514          num_str.push_str(dec_part);
 515      }
 516      if is_float {
 517          match base {
 518            16u => fatal_span(rdr, start_bpos, rdr.last_pos,
 519                              ~"hexadecimal float literal is not supported"),
 520            2u => fatal_span(rdr, start_bpos, rdr.last_pos,
 521                             ~"binary float literal is not supported"),
 522            _ => ()
 523          }
 524      }
 525      match scan_exponent(rdr, start_bpos) {
 526        Some(ref s) => {
 527          is_float = true;
 528          num_str.push_str(*s);
 529        }
 530        None => ()
 531      }
 532  
 533      let mut is_machine_float = false;
 534      if rdr.curr == 'f' {
 535          bump(rdr);
 536          c = rdr.curr;
 537          n = nextch(rdr);
 538          if c == '3' && n == '2' {
 539              bump(rdr);
 540              bump(rdr);
 541              return token::LIT_FLOAT(str_to_ident(num_str),
 542                                   ast::ty_f32);
 543          } else if c == '6' && n == '4' {
 544              bump(rdr);
 545              bump(rdr);
 546              return token::LIT_FLOAT(str_to_ident(num_str),
 547                                   ast::ty_f64);
 548              /* FIXME (#2252): if this is out of range for either a
 549              32-bit or 64-bit float, it won't be noticed till the
 550              back-end.  */
 551          } else {
 552              is_float = true;
 553              is_machine_float = true;
 554          }
 555      }
 556      if is_float {
 557          if is_machine_float {
 558              return token::LIT_FLOAT(str_to_ident(num_str), ast::ty_f);
 559          }
 560          return token::LIT_FLOAT_UNSUFFIXED(str_to_ident(num_str));
 561      } else {
 562          if num_str.len() == 0u {
 563              fatal_span(rdr, start_bpos, rdr.last_pos,
 564                         ~"no valid digits found for number");
 565          }
 566          let parsed = match from_str_radix::<u64>(num_str, base as uint) {
 567              Some(p) => p,
 568              None => fatal_span(rdr, start_bpos, rdr.last_pos,
 569                                 ~"int literal is too large")
 570          };
 571  
 572          debug!("lexing %s as an unsuffixed integer literal",
 573                 num_str);
 574          return token::LIT_INT_UNSUFFIXED(parsed as i64);
 575      }
 576  }

 578  fn scan_numeric_escape(rdr: @mut StringReader, n_hex_digits: uint) -> char {
 579      let mut accum_int = 0;
 580      let mut i = n_hex_digits;
 581      let start_bpos = rdr.last_pos;
 582      while i != 0u {
 583          let n = rdr.curr;
 584          if !is_hex_digit(n) {
 585              fatal_span_char(rdr, rdr.last_pos, rdr.pos,
 586                              ~"illegal character in numeric character escape",
 587                              n);
 588          }
 589          bump(rdr);
 590          accum_int *= 16;
 591          accum_int += hex_digit_val(n);
 592          i -= 1u;
 593      }
 594      match char::from_u32(accum_int as u32) {
 595          Some(x) => x,
 596          None => fatal_span(rdr, start_bpos, rdr.last_pos,
 597                             ~"illegal numeric character escape")
 598      }
 599  }

 601  fn ident_start(c: char) -> bool {
 602      (c >= 'a' && c <= 'z')
 603          || (c >= 'A' && c <= 'Z')
 604          || c == '_'
 605          || (c > '\x7f' && char::is_XID_start(c))
 606  }
 607  
 608  fn ident_continue(c: char) -> bool {
 609      (c >= 'a' && c <= 'z')
 610          || (c >= 'A' && c <= 'Z')
 611          || (c >= '0' && c <= '9')
 612          || c == '_'
 613          || (c > '\x7f' && char::is_XID_continue(c))
 614  }
 615  
 616  // return the next token from the string
 617  // EFFECT: advances the input past that token
 618  // EFFECT: updates the interner
 619  fn next_token_inner(rdr: @mut StringReader) -> token::Token {
 620      let c = rdr.curr;
 621      if ident_start(c) {
 622          let start = rdr.last_pos;
 623          while ident_continue(rdr.curr) {
 624              bump(rdr);
 625          }
 626  
 627          return do with_str_from(rdr, start) |string| {
 628              if string == "_" {
 629                  token::UNDERSCORE
 630              } else {
 631                  let is_mod_name = rdr.curr == ':' && nextch(rdr) == ':';
 632  
 633                  // FIXME: perform NFKC normalization here. (Issue #2253)
 634                  token::IDENT(str_to_ident(string), is_mod_name)
 635              }
 636          }
 637      }
 638      if is_dec_digit(c) {
 639          return scan_number(c, rdr);
 640      }
 641      fn binop(rdr: @mut StringReader, op: token::binop) -> token::Token {
 642          bump(rdr);
 643          if rdr.curr == '=' {
 644              bump(rdr);
 645              return token::BINOPEQ(op);
 646          } else { return token::BINOP(op); }
 647      }
 648      match c {
 649  
 650  
 651  
 652  
 653  
 654        // One-byte tokens.
 655        ';' => { bump(rdr); return token::SEMI; }
 656        ',' => { bump(rdr); return token::COMMA; }
 657        '.' => {
 658          bump(rdr);
 659          if rdr.curr == '.' && nextch(rdr) != '.' {
 660              bump(rdr);
 661              return token::DOTDOT;
 662          }
 663          return token::DOT;
 664        }
 665        '(' => { bump(rdr); return token::LPAREN; }
 666        ')' => { bump(rdr); return token::RPAREN; }
 667        '{' => { bump(rdr); return token::LBRACE; }
 668        '}' => { bump(rdr); return token::RBRACE; }
 669        '[' => { bump(rdr); return token::LBRACKET; }
 670        ']' => { bump(rdr); return token::RBRACKET; }
 671        '@' => { bump(rdr); return token::AT; }
 672        '#' => { bump(rdr); return token::POUND; }
 673        '~' => { bump(rdr); return token::TILDE; }
 674        ':' => {
 675          bump(rdr);
 676          if rdr.curr == ':' {
 677              bump(rdr);
 678              return token::MOD_SEP;
 679          } else { return token::COLON; }
 680        }
 681  
 682        '$' => { bump(rdr); return token::DOLLAR; }
 683  
 684  
 685  
 686  
 687  
 688        // Multi-byte tokens.
 689        '=' => {
 690          bump(rdr);
 691          if rdr.curr == '=' {
 692              bump(rdr);
 693              return token::EQEQ;
 694          } else if rdr.curr == '>' {
 695              bump(rdr);
 696              return token::FAT_ARROW;
 697          } else {
 698              return token::EQ;
 699          }
 700        }
 701        '!' => {
 702          bump(rdr);
 703          if rdr.curr == '=' {
 704              bump(rdr);
 705              return token::NE;
 706          } else { return token::NOT; }
 707        }
 708        '<' => {
 709          bump(rdr);
 710          match rdr.curr {
 711            '=' => { bump(rdr); return token::LE; }
 712            '<' => { return binop(rdr, token::SHL); }
 713            '-' => {
 714              bump(rdr);
 715              match rdr.curr {
 716                '>' => { bump(rdr); return token::DARROW; }
 717                _ => { return token::LARROW; }
 718              }
 719            }
 720            _ => { return token::LT; }
 721          }
 722        }
 723        '>' => {
 724          bump(rdr);
 725          match rdr.curr {
 726            '=' => { bump(rdr); return token::GE; }
 727            '>' => { return binop(rdr, token::SHR); }
 728            _ => { return token::GT; }
 729          }
 730        }
 731        '\'' => {
 732          // Either a character constant 'a' OR a lifetime name 'abc
 733          bump(rdr);
 734          let start = rdr.last_pos;
 735          let mut c2 = rdr.curr;
 736          bump(rdr);
 737  
 738          // If the character is an ident start not followed by another single
 739          // quote, then this is a lifetime name:
 740          if ident_start(c2) && rdr.curr != '\'' {
 741              while ident_continue(rdr.curr) {
 742                  bump(rdr);
 743              }
 744              return do with_str_from(rdr, start) |lifetime_name| {
 745                  token::LIFETIME(str_to_ident(lifetime_name))
 746              }
 747          }
 748  
 749          // Otherwise it is a character constant:
 750          if c2 == '\\' {
 751              // '\X' for some X must be a character constant:
 752              let escaped = rdr.curr;
 753              let escaped_pos = rdr.last_pos;
 754              bump(rdr);
 755              match escaped {
 756                'n' => { c2 = '\n'; }
 757                'r' => { c2 = '\r'; }
 758                't' => { c2 = '\t'; }
 759                '\\' => { c2 = '\\'; }
 760                '\'' => { c2 = '\''; }
 761                '"' => { c2 = '"'; }
 762                '0' => { c2 = '\x00'; }
 763                'x' => { c2 = scan_numeric_escape(rdr, 2u); }
 764                'u' => { c2 = scan_numeric_escape(rdr, 4u); }
 765                'U' => { c2 = scan_numeric_escape(rdr, 8u); }
 766                c2 => {
 767                  fatal_span_char(rdr, escaped_pos, rdr.last_pos,
 768                                  ~"unknown character escape", c2);
 769                }
 770              }
 771          }
 772          if rdr.curr != '\'' {
 773              fatal_span_verbose(rdr,
 774                                 // Byte offsetting here is okay because the
 775                                 // character before position `start` is an
 776                                 // ascii single quote.
 777                                 start - BytePos(1u),
 778                                 rdr.last_pos,
 779                                 ~"unterminated character constant");
 780          }
 781          bump(rdr); // advance curr past token
 782          return token::LIT_CHAR(c2 as u32);
 783        }
 784        '"' => {
 785          let mut accum_str = ~"";
 786          let start_bpos = rdr.last_pos;
 787          bump(rdr);
 788          while rdr.curr != '"' {
 789              if is_eof(rdr) {
 790                  fatal_span(rdr, start_bpos, rdr.last_pos,
 791                             ~"unterminated double quote string");
 792              }
 793  
 794              let ch = rdr.curr;
 795              bump(rdr);
 796              match ch {
 797                '\\' => {
 798                  let escaped = rdr.curr;
 799                  let escaped_pos = rdr.last_pos;
 800                  bump(rdr);
 801                  match escaped {
 802                    'n' => accum_str.push_char('\n'),
 803                    'r' => accum_str.push_char('\r'),
 804                    't' => accum_str.push_char('\t'),
 805                    '\\' => accum_str.push_char('\\'),
 806                    '\'' => accum_str.push_char('\''),
 807                    '"' => accum_str.push_char('"'),
 808                    '\n' => consume_whitespace(rdr),
 809                    '0' => accum_str.push_char('\x00'),
 810                    'x' => {
 811                      accum_str.push_char(scan_numeric_escape(rdr, 2u));
 812                    }
 813                    'u' => {
 814                      accum_str.push_char(scan_numeric_escape(rdr, 4u));
 815                    }
 816                    'U' => {
 817                      accum_str.push_char(scan_numeric_escape(rdr, 8u));
 818                    }
 819                    c2 => {
 820                      fatal_span_char(rdr, escaped_pos, rdr.last_pos,
 821                                      ~"unknown string escape", c2);
 822                    }
 823                  }
 824                }
 825                _ => accum_str.push_char(ch)
 826              }
 827          }
 828          bump(rdr);
 829          return token::LIT_STR(str_to_ident(accum_str));
 830        }
 831        '-' => {
 832          if nextch(rdr) == '>' {
 833              bump(rdr);
 834              bump(rdr);
 835              return token::RARROW;
 836          } else { return binop(rdr, token::MINUS); }
 837        }
 838        '&' => {
 839          if nextch(rdr) == '&' {
 840              bump(rdr);
 841              bump(rdr);
 842              return token::ANDAND;
 843          } else { return binop(rdr, token::AND); }
 844        }
 845        '|' => {
 846          match nextch(rdr) {
 847            '|' => { bump(rdr); bump(rdr); return token::OROR; }
 848            _ => { return binop(rdr, token::OR); }
 849          }
 850        }
 851        '+' => { return binop(rdr, token::PLUS); }
 852        '*' => { return binop(rdr, token::STAR); }
 853        '/' => { return binop(rdr, token::SLASH); }
 854        '^' => { return binop(rdr, token::CARET); }
 855        '%' => { return binop(rdr, token::PERCENT); }
 856        c => {
 857            fatal_span_char(rdr, rdr.last_pos, rdr.pos,
 858                            ~"unknown start of token", c);
 859        }
 860      }

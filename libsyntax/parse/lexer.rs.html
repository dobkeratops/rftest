  59  pub fn new_string_reader(span_diagnostic: @mut span_handler,
  60                           filemap: @codemap::FileMap)
  61                        -> @mut StringReader {
  62      let r = new_low_level_string_reader(span_diagnostic, filemap);

  67  /* For comments.rs, which hackily pokes into 'pos' and 'curr' */
  68  pub fn new_low_level_string_reader(span_diagnostic: @mut span_handler,
  69                                     filemap: @codemap::FileMap)
  70                                  -> @mut StringReader {
  71      // Force the initial reader bump to start on a fresh line
  72      let initial_char = '\n';
  73      let r = @mut StringReader {
  74          span_diagnostic: span_diagnostic,
  75          src: filemap.src,
  76          pos: filemap.start_pos,
  77          last_pos: filemap.start_pos,
  78          col: CharPos(0),
  79          curr: initial_char,
  80          filemap: filemap,
  81          /* dummy values; not read */
  82          peek_tok: token::EOF,
  83          peek_span: codemap::dummy_sp()
  84      };
  85      bump(r);
  86      return r;
  87  }

 106  impl reader for StringReader {
 107      fn is_eof(@mut self) -> bool { is_eof(self) }
 108      // return the next token. EFFECT: advances the string_reader.
 109      fn next_token(@mut self) -> TokenAndSpan {
 110          let ret_val = TokenAndSpan {
 111              tok: util::replace(&mut self.peek_tok, token::UNDERSCORE),
 112              sp: self.peek_span,
 113          };
 114          string_advance_token(self);
 115          ret_val
 116      }
 117      fn fatal(@mut self, m: ~str) -> ! {
 118          self.span_diagnostic.span_fatal(self.peek_span, m)
 119      }
 120      fn span_diag(@mut self) -> @mut span_handler { self.span_diagnostic }
 121      fn peek(@mut self) -> TokenAndSpan {
 122          // XXX(pcwalton): Bad copy!
 123          TokenAndSpan {
 124              tok: self.peek_tok.clone(),
 125              sp: self.peek_span,
 126          }
 127      }
 128      fn dup(@mut self) -> @mut reader { dup_string_reader(self) as @mut reader }
 129  }

 151  // EFFECT: advance peek_tok and peek_span to refer to the next token.
 152  // EFFECT: update the interner, maybe.
 153  fn string_advance_token(r: @mut StringReader) {
 154      match (consume_whitespace_and_comments(r)) {
 155          Some(comment) => {
 156              r.peek_span = comment.sp;
 157              r.peek_tok = comment.tok;
 158          },
 159          None => {
 160              if is_eof(r) {
 161                  r.peek_tok = token::EOF;
 162              } else {
 163                  let start_bytepos = r.last_pos;
 164                  r.peek_tok = next_token_inner(r);
 165                  r.peek_span = codemap::mk_sp(start_bytepos, r.last_pos);
 166              };
 167          }
 168      }
 169  }

 175  pub fn with_str_from<T>(rdr: @mut StringReader, start: BytePos, f: &fn(s: &str) -> T) -> T {
 176      f(rdr.src.slice(

 179  }

 181  // EFFECT: advance the StringReader by one character. If a newline is
 182  // discovered, add it to the FileMap's list of line start offsets.
 183  pub fn bump(rdr: &mut StringReader) {
 184      rdr.last_pos = rdr.pos;
 185      let current_byte_offset = byte_offset(rdr, rdr.pos).to_uint();
 186      if current_byte_offset < (rdr.src).len() {
 187          assert!(rdr.curr != -1 as char);
 188          let last_char = rdr.curr;
 189          let next = rdr.src.char_range_at(current_byte_offset);
 190          let byte_offset_diff = next.next - current_byte_offset;
 191          rdr.pos = rdr.pos + BytePos(byte_offset_diff);
 192          rdr.curr = next.ch;
 193          rdr.col = rdr.col + CharPos(1u);
 194          if last_char == '\n' {
 195              rdr.filemap.next_line(rdr.last_pos);
 196              rdr.col = CharPos(0u);
 197          }
 198  
 199          if byte_offset_diff > 1 {
 200              rdr.filemap.record_multibyte_char(
 201                  BytePos(current_byte_offset), byte_offset_diff);
 202          }
 203      } else {
 204          rdr.curr = -1 as char;
 205      }
 206  }
 207  pub fn is_eof(rdr: @mut StringReader) -> bool {
 208      rdr.curr == -1 as char

 210  pub fn nextch(rdr: @mut StringReader) -> char {
 211      let offset = byte_offset(rdr, rdr.pos).to_uint();
 212      if offset < (rdr.src).len() {
 213          return rdr.src.char_at(offset);
 214      } else { return -1 as char; }
 215  }

 223      fail!();

 245  // EFFECT: eats whitespace and comments.
 246  // returns a Some(sugared-doc-attr) if one exists, None otherwise.
 247  fn consume_whitespace_and_comments(rdr: @mut StringReader)
 248                                  -> Option<TokenAndSpan> {
 249      while is_whitespace(rdr.curr) { bump(rdr); }
 250      return consume_any_line_comment(rdr);
 251  }

 253  pub fn is_line_non_doc_comment(s: &str) -> bool {
 254      let s = s.trim_right();
 255      s.len() > 3 && s.iter().all(|ch| ch == '/')
 256  }
 257  
 258  // PRECONDITION: rdr.curr is not whitespace
 259  // EFFECT: eats any kind of comment.
 260  // returns a Some(sugared-doc-attr) if one exists, None otherwise
 261  fn consume_any_line_comment(rdr: @mut StringReader)
 262                           -> Option<TokenAndSpan> {
 263      if rdr.curr == '/' {
 264          match nextch(rdr) {
 265            '/' => {
 266              bump(rdr);
 267              bump(rdr);
 268              // line comments starting with "///" or "//!" are doc-comments
 269              if rdr.curr == '/' || rdr.curr == '!' {
 270                  let start_bpos = rdr.pos - BytePos(3u);
 271                  while rdr.curr != '\n' && !is_eof(rdr) {
 272                      bump(rdr);
 273                  }
 274                  let ret = do with_str_from(rdr, start_bpos) |string| {
 275                      // but comments with only more "/"s are not
 276                      if !is_line_non_doc_comment(string) {
 277                          Some(TokenAndSpan{
 278                              tok: token::DOC_COMMENT(str_to_ident(string)),
 279                              sp: codemap::mk_sp(start_bpos, rdr.pos)
 280                          })
 281                      } else {
 282                          None
 283                      }
 284                  };
 285  
 286                  if ret.is_some() {
 287                      return ret;
 288                  }
 289              } else {
 290                  while rdr.curr != '\n' && !is_eof(rdr) { bump(rdr); }
 291              }
 292              // Restart whitespace munch.
 293              return consume_whitespace_and_comments(rdr);
 294            }
 295            '*' => { bump(rdr); bump(rdr); return consume_block_comment(rdr); }
 296            _ => ()
 297          }
 298      } else if rdr.curr == '#' {
 299          if nextch(rdr) == '!' {
 300              // I guess this is the only way to figure out if
 301              // we're at the beginning of the file...
 302              let cmap = @CodeMap::new();
 303              (*cmap).files.push(rdr.filemap);
 304              let loc = cmap.lookup_char_pos_adj(rdr.last_pos);
 305              if loc.line == 1u && loc.col == CharPos(0u) {
 306                  while rdr.curr != '\n' && !is_eof(rdr) { bump(rdr); }
 307                  return consume_whitespace_and_comments(rdr);
 308              }
 309          }
 310      }
 311      return None;
 312  }

 314  pub fn is_block_non_doc_comment(s: &str) -> bool {
 315      assert!(s.len() >= 1u);
 316      s.slice(1u, s.len() - 1u).iter().all(|ch| ch == '*')
 317  }

 319  // might return a sugared-doc-attr
 320  fn consume_block_comment(rdr: @mut StringReader)
 321                        -> Option<TokenAndSpan> {
 322      // block comments starting with "/**" or "/*!" are doc-comments
 323      let res = if rdr.curr == '*' || rdr.curr == '!' {
 324          let start_bpos = rdr.pos - BytePos(3u);
 325          while !(rdr.curr == '*' && nextch(rdr) == '/') && !is_eof(rdr) {
 326              bump(rdr);
 327          }
 328          if is_eof(rdr) {
 329              rdr.fatal(~"unterminated block doc-comment");
 330          } else {
 331              bump(rdr);
 332              bump(rdr);
 333              do with_str_from(rdr, start_bpos) |string| {
 334                  // but comments with only "*"s between two "/"s are not
 335                  if !is_block_non_doc_comment(string) {
 336                      Some(TokenAndSpan{
 337                           tok: token::DOC_COMMENT(str_to_ident(string)),
 338                           sp: codemap::mk_sp(start_bpos, rdr.pos)
 339                           })
 340                  } else {
 341                      None
 342                  }
 343              }
 344          }
 345      } else {
 346          loop {
 347              if is_eof(rdr) { rdr.fatal(~"unterminated block comment"); }
 348              if rdr.curr == '*' && nextch(rdr) == '/' {
 349                  bump(rdr);
 350                  bump(rdr);
 351                  break;
 352              } else {
 353                  bump(rdr);
 354              }
 355          }
 356          None
 357      };
 358      // restart whitespace munch.
 359  
 360     if res.is_some() { res } else { consume_whitespace_and_comments(rdr) }
 361  }

 363  fn scan_exponent(rdr: @mut StringReader) -> Option<~str> {
 364      let mut c = rdr.curr;
 365      let mut rslt = ~"";
 366      if c == 'e' || c == 'E' {
 367          rslt.push_char(c);
 368          bump(rdr);
 369          c = rdr.curr;
 370          if c == '-' || c == '+' {
 371              rslt.push_char(c);
 372              bump(rdr);
 373          }
 374          let exponent = scan_digits(rdr, 10u);
 375          if exponent.len() > 0u {
 376              return Some(rslt + exponent);
 377          } else { rdr.fatal(~"scan_exponent: bad fp literal"); }
 378      } else { return None::<~str>; }
 379  }

 381  fn scan_digits(rdr: @mut StringReader, radix: uint) -> ~str {
 382      let mut rslt = ~"";
 383      loop {
 384          let c = rdr.curr;
 385          if c == '_' { bump(rdr); loop; }
 386          match char::to_digit(c, radix) {
 387            Some(_) => {
 388              rslt.push_char(c);
 389              bump(rdr);
 390            }
 391            _ => return rslt
 392          }
 393      };
 394  }

 396  fn scan_number(c: char, rdr: @mut StringReader) -> token::Token {
 397      let mut num_str;
 398      let mut base = 10u;
 399      let mut c = c;
 400      let mut n = nextch(rdr);
 401      if c == '0' && n == 'x' {
 402          bump(rdr);
 403          bump(rdr);
 404          base = 16u;
 405      } else if c == '0' && n == 'b' {
 406          bump(rdr);
 407          bump(rdr);
 408          base = 2u;
 409      }
 410      num_str = scan_digits(rdr, base);
 411      c = rdr.curr;
 412      nextch(rdr);
 413      if c == 'u' || c == 'i' {
 414          let signed = c == 'i';
 415          let mut tp = {
 416              if signed { either::Left(ast::ty_i) }
 417              else { either::Right(ast::ty_u) }
 418          };
 419          bump(rdr);
 420          c = rdr.curr;
 421          if c == '8' {
 422              bump(rdr);
 423              tp = if signed { either::Left(ast::ty_i8) }
 424                        else { either::Right(ast::ty_u8) };
 425          }
 426          n = nextch(rdr);
 427          if c == '1' && n == '6' {
 428              bump(rdr);
 429              bump(rdr);
 430              tp = if signed { either::Left(ast::ty_i16) }
 431                        else { either::Right(ast::ty_u16) };
 432          } else if c == '3' && n == '2' {
 433              bump(rdr);
 434              bump(rdr);
 435              tp = if signed { either::Left(ast::ty_i32) }
 436                        else { either::Right(ast::ty_u32) };
 437          } else if c == '6' && n == '4' {
 438              bump(rdr);
 439              bump(rdr);
 440              tp = if signed { either::Left(ast::ty_i64) }
 441                        else { either::Right(ast::ty_u64) };
 442          }
 443          if num_str.len() == 0u {
 444              rdr.fatal(~"no valid digits found for number");
 445          }
 446          let parsed = match u64::from_str_radix(num_str, base as uint) {
 447              Some(p) => p,
 448              None => rdr.fatal(~"int literal is too large")
 449          };
 450  
 451          match tp {
 452            either::Left(t) => return token::LIT_INT(parsed as i64, t),
 453            either::Right(t) => return token::LIT_UINT(parsed, t)
 454          }
 455      }
 456      let mut is_float = false;
 457      if rdr.curr == '.' && !(ident_start(nextch(rdr)) || nextch(rdr) == '.') {
 458          is_float = true;
 459          bump(rdr);
 460          let dec_part = scan_digits(rdr, 10u);
 461          num_str.push_char('.');
 462          num_str.push_str(dec_part);
 463      }
 464      if is_float {
 465          match base {
 466            16u => rdr.fatal(~"hexadecimal float literal is not supported"),
 467            2u => rdr.fatal(~"binary float literal is not supported"),
 468            _ => ()
 469          }
 470      }
 471      match scan_exponent(rdr) {
 472        Some(ref s) => {
 473          is_float = true;
 474          num_str.push_str(*s);
 475        }
 476        None => ()
 477      }
 478  
 479      let mut is_machine_float = false;
 480      if rdr.curr == 'f' {
 481          bump(rdr);
 482          c = rdr.curr;
 483          n = nextch(rdr);
 484          if c == '3' && n == '2' {
 485              bump(rdr);
 486              bump(rdr);
 487              return token::LIT_FLOAT(str_to_ident(num_str),
 488                                   ast::ty_f32);
 489          } else if c == '6' && n == '4' {
 490              bump(rdr);
 491              bump(rdr);
 492              return token::LIT_FLOAT(str_to_ident(num_str),
 493                                   ast::ty_f64);
 494              /* FIXME (#2252): if this is out of range for either a
 495              32-bit or 64-bit float, it won't be noticed till the
 496              back-end.  */
 497          } else {
 498              is_float = true;
 499              is_machine_float = true;
 500          }
 501      }
 502      if is_float {
 503          if is_machine_float {
 504              return token::LIT_FLOAT(str_to_ident(num_str), ast::ty_f);
 505          }
 506          return token::LIT_FLOAT_UNSUFFIXED(str_to_ident(num_str));
 507      } else {
 508          if num_str.len() == 0u {
 509              rdr.fatal(~"no valid digits found for number");
 510          }
 511          let parsed = match u64::from_str_radix(num_str, base as uint) {
 512              Some(p) => p,
 513              None => rdr.fatal(~"int literal is too large")
 514          };
 515  
 516          debug!("lexing %s as an unsuffixed integer literal",
 517                 num_str);
 518          return token::LIT_INT_UNSUFFIXED(parsed as i64);
 519      }
 520  }

 522  fn scan_numeric_escape(rdr: @mut StringReader, n_hex_digits: uint) -> char {
 523      let mut accum_int = 0;
 524      let mut i = n_hex_digits;
 525      while i != 0u {
 526          let n = rdr.curr;
 527          bump(rdr);
 528          if !is_hex_digit(n) {
 529              rdr.fatal(fmt!("illegal numeric character escape: %d", n as int));
 530          }
 531          accum_int *= 16;
 532          accum_int += hex_digit_val(n);
 533          i -= 1u;
 534      }
 535      return accum_int as char;
 536  }

 538  fn ident_start(c: char) -> bool {
 539      (c >= 'a' && c <= 'z')
 540          || (c >= 'A' && c <= 'Z')
 541          || c == '_'
 542          || (c > '\x7f' && char::is_XID_start(c))
 543  }
 544  
 545  fn ident_continue(c: char) -> bool {
 546      (c >= 'a' && c <= 'z')
 547          || (c >= 'A' && c <= 'Z')
 548          || (c >= '0' && c <= '9')
 549          || c == '_'
 550          || (c > '\x7f' && char::is_XID_continue(c))
 551  }
 552  
 553  // return the next token from the string
 554  // EFFECT: advances the input past that token
 555  // EFFECT: updates the interner
 556  fn next_token_inner(rdr: @mut StringReader) -> token::Token {
 557      let c = rdr.curr;
 558      if ident_start(c) {
 559          let start = rdr.last_pos;
 560          while ident_continue(rdr.curr) {
 561              bump(rdr);
 562          }
 563  
 564          return do with_str_from(rdr, start) |string| {
 565              if string == "_" {
 566                  token::UNDERSCORE
 567              } else {
 568                  let is_mod_name = rdr.curr == ':' && nextch(rdr) == ':';
 569  
 570                  // FIXME: perform NFKC normalization here. (Issue #2253)
 571                  token::IDENT(str_to_ident(string), is_mod_name)
 572              }
 573          }
 574      }
 575      if is_dec_digit(c) {
 576          return scan_number(c, rdr);
 577      }
 578      fn binop(rdr: @mut StringReader, op: token::binop) -> token::Token {
 579          bump(rdr);
 580          if rdr.curr == '=' {
 581              bump(rdr);
 582              return token::BINOPEQ(op);
 583          } else { return token::BINOP(op); }
 584      }
 585      match c {
 586  
 587  
 588  
 589  
 590  
 591        // One-byte tokens.
 592        ';' => { bump(rdr); return token::SEMI; }
 593        ',' => { bump(rdr); return token::COMMA; }
 594        '.' => {
 595          bump(rdr);
 596          if rdr.curr == '.' && nextch(rdr) != '.' {
 597              bump(rdr);
 598              return token::DOTDOT;
 599          }
 600          return token::DOT;
 601        }
 602        '(' => { bump(rdr); return token::LPAREN; }
 603        ')' => { bump(rdr); return token::RPAREN; }
 604        '{' => { bump(rdr); return token::LBRACE; }
 605        '}' => { bump(rdr); return token::RBRACE; }
 606        '[' => { bump(rdr); return token::LBRACKET; }
 607        ']' => { bump(rdr); return token::RBRACKET; }
 608        '@' => { bump(rdr); return token::AT; }
 609        '#' => { bump(rdr); return token::POUND; }
 610        '~' => { bump(rdr); return token::TILDE; }
 611        ':' => {
 612          bump(rdr);
 613          if rdr.curr == ':' {
 614              bump(rdr);
 615              return token::MOD_SEP;
 616          } else { return token::COLON; }
 617        }
 618  
 619        '$' => { bump(rdr); return token::DOLLAR; }
 620  
 621  
 622  
 623  
 624  
 625        // Multi-byte tokens.
 626        '=' => {
 627          bump(rdr);
 628          if rdr.curr == '=' {
 629              bump(rdr);
 630              return token::EQEQ;
 631          } else if rdr.curr == '>' {
 632              bump(rdr);
 633              return token::FAT_ARROW;
 634          } else {
 635              return token::EQ;
 636          }
 637        }
 638        '!' => {
 639          bump(rdr);
 640          if rdr.curr == '=' {
 641              bump(rdr);
 642              return token::NE;
 643          } else { return token::NOT; }
 644        }
 645        '<' => {
 646          bump(rdr);
 647          match rdr.curr {
 648            '=' => { bump(rdr); return token::LE; }
 649            '<' => { return binop(rdr, token::SHL); }
 650            '-' => {
 651              bump(rdr);
 652              match rdr.curr {
 653                '>' => { bump(rdr); return token::DARROW; }
 654                _ => { return token::LARROW; }
 655              }
 656            }
 657            _ => { return token::LT; }
 658          }
 659        }
 660        '>' => {
 661          bump(rdr);
 662          match rdr.curr {
 663            '=' => { bump(rdr); return token::GE; }
 664            '>' => { return binop(rdr, token::SHR); }
 665            _ => { return token::GT; }
 666          }
 667        }
 668        '\'' => {
 669          // Either a character constant 'a' OR a lifetime name 'abc
 670          bump(rdr);
 671          let start = rdr.last_pos;
 672          let mut c2 = rdr.curr;
 673          bump(rdr);
 674  
 675          // If the character is an ident start not followed by another single
 676          // quote, then this is a lifetime name:
 677          if ident_start(c2) && rdr.curr != '\'' {
 678              while ident_continue(rdr.curr) {
 679                  bump(rdr);
 680              }
 681              return do with_str_from(rdr, start) |lifetime_name| {
 682                  token::LIFETIME(str_to_ident(lifetime_name))
 683              }
 684          }
 685  
 686          // Otherwise it is a character constant:
 687          if c2 == '\\' {
 688              // '\X' for some X must be a character constant:
 689              let escaped = rdr.curr;
 690              bump(rdr);
 691              match escaped {
 692                'n' => { c2 = '\n'; }
 693                'r' => { c2 = '\r'; }
 694                't' => { c2 = '\t'; }
 695                '\\' => { c2 = '\\'; }
 696                '\'' => { c2 = '\''; }
 697                '"' => { c2 = '"'; }
 698                'x' => { c2 = scan_numeric_escape(rdr, 2u); }
 699                'u' => { c2 = scan_numeric_escape(rdr, 4u); }
 700                'U' => { c2 = scan_numeric_escape(rdr, 8u); }
 701                c2 => {
 702                  rdr.fatal(fmt!("unknown character escape: %d", c2 as int));
 703                }
 704              }
 705          }
 706          if rdr.curr != '\'' {
 707              rdr.fatal(~"unterminated character constant");
 708          }
 709          bump(rdr); // advance curr past token
 710          return token::LIT_INT(c2 as i64, ast::ty_char);
 711        }
 712        '"' => {
 713          let mut accum_str = ~"";
 714          let n = rdr.last_pos;
 715          bump(rdr);
 716          while rdr.curr != '"' {
 717              if is_eof(rdr) {
 718                  do with_str_from(rdr, n) |s| {
 719                      rdr.fatal(fmt!("unterminated double quote string: %s", s));
 720                  }
 721              }
 722  
 723              let ch = rdr.curr;
 724              bump(rdr);
 725              match ch {
 726                '\\' => {
 727                  let escaped = rdr.curr;
 728                  bump(rdr);
 729                  match escaped {
 730                    'n' => accum_str.push_char('\n'),
 731                    'r' => accum_str.push_char('\r'),
 732                    't' => accum_str.push_char('\t'),
 733                    '\\' => accum_str.push_char('\\'),
 734                    '\'' => accum_str.push_char('\''),
 735                    '"' => accum_str.push_char('"'),
 736                    '\n' => consume_whitespace(rdr),
 737                    'x' => {
 738                      accum_str.push_char(scan_numeric_escape(rdr, 2u));
 739                    }
 740                    'u' => {
 741                      accum_str.push_char(scan_numeric_escape(rdr, 4u));
 742                    }
 743                    'U' => {
 744                      accum_str.push_char(scan_numeric_escape(rdr, 8u));
 745                    }
 746                    c2 => {
 747                      rdr.fatal(fmt!("unknown string escape: %d", c2 as int));
 748                    }
 749                  }
 750                }
 751                _ => accum_str.push_char(ch)
 752              }
 753          }
 754          bump(rdr);
 755          return token::LIT_STR(str_to_ident(accum_str));
 756        }
 757        '-' => {
 758          if nextch(rdr) == '>' {
 759              bump(rdr);
 760              bump(rdr);
 761              return token::RARROW;
 762          } else { return binop(rdr, token::MINUS); }
 763        }
 764        '&' => {
 765          if nextch(rdr) == '&' {
 766              bump(rdr);
 767              bump(rdr);
 768              return token::ANDAND;
 769          } else { return binop(rdr, token::AND); }
 770        }
 771        '|' => {
 772          match nextch(rdr) {
 773            '|' => { bump(rdr); bump(rdr); return token::OROR; }
 774            _ => { return binop(rdr, token::OR); }
 775          }
 776        }
 777        '+' => { return binop(rdr, token::PLUS); }
 778        '*' => { return binop(rdr, token::STAR); }
 779        '/' => { return binop(rdr, token::SLASH); }
 780        '^' => { return binop(rdr, token::CARET); }
 781        '%' => { return binop(rdr, token::PERCENT); }
 782        c => {
 783            // So the error span points to the unrecognized character
 784            rdr.peek_span = codemap::mk_sp(rdr.last_pos, rdr.pos);
 785            rdr.fatal(fmt!("unknown start of token: %d", c as int));
 786        }
 787      }
 788  }

 790  fn consume_whitespace(rdr: @mut StringReader) {
 791      while is_whitespace(rdr.curr) && !is_eof(rdr) { bump(rdr); }
 792  }

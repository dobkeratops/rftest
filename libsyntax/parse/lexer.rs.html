  42 pub struct StringReader {

  55     peek_tok: token::Token,

  59 pub fn new_string_reader(span_diagnostic: @span_handler,
  60                          filemap: @codemap::FileMap)
  61                       -> @mut StringReader {
  62     let r = new_low_level_string_reader(span_diagnostic, filemap);
  63     string_advance_token(r); /* fill in peek_* */
  64     return r;
  65 }

  67 /* For comments.rs, which hackily pokes into 'pos' and 'curr' */
  68 pub fn new_low_level_string_reader(span_diagnostic: @span_handler,
  69                                    filemap: @codemap::FileMap)
  70                                 -> @mut StringReader {
  71     // Force the initial reader bump to start on a fresh line
  72     let initial_char = '\n';
  73     let r = @mut StringReader {
  74         span_diagnostic: span_diagnostic, src: filemap.src,
  75         pos: filemap.start_pos,
  76         last_pos: filemap.start_pos,
  77         col: CharPos(0),
  78         curr: initial_char,
  79         filemap: filemap,
  80         /* dummy values; not read */
  81         peek_tok: token::EOF,
  82         peek_span: codemap::dummy_sp()
  83     };
  84     bump(r);
  85     return r;
  86 }

  91 fn dup_string_reader(r: @mut StringReader) -> @mut StringReader {

  99         filemap: r.filemap,

 101         peek_span: r.peek_span
 102     }
 103 }

 105 impl reader for StringReader {
 106     fn is_eof(@mut self) -> bool { is_eof(self) }
 107     // return the next token. EFFECT: advances the string_reader.
 108     fn next_token(@mut self) -> TokenAndSpan {
 109         let ret_val = TokenAndSpan {
 110             tok: util::replace(&mut self.peek_tok, token::UNDERSCORE),
 111             sp: self.peek_span,
 112         };
 113         string_advance_token(self);
 114         ret_val
 115     }
 116     fn fatal(@mut self, m: ~str) -> ! {
 117         self.span_diagnostic.span_fatal(self.peek_span, m)
 118     }
 119     fn span_diag(@mut self) -> @span_handler { self.span_diagnostic }
 120     fn peek(@mut self) -> TokenAndSpan {
 121         // XXX(pcwalton): Bad copy!
 122         TokenAndSpan {
 123             tok: self.peek_tok.clone(),
 124             sp: self.peek_span,
 125         }
 126     }
 127     fn dup(@mut self) -> @reader { dup_string_reader(self) as @reader }
 128 }

 146 // EFFECT: advance peek_tok and peek_span to refer to the next token.
 147 // EFFECT: update the interner, maybe.
 148 fn string_advance_token(r: @mut StringReader) {
 149     match (consume_whitespace_and_comments(r)) {
 150         Some(comment) => {
 151             r.peek_span = comment.sp;
 152             r.peek_tok = comment.tok;
 153         },
 154         None => {
 155             if is_eof(r) {
 156                 r.peek_tok = token::EOF;
 157             } else {
 158                 let start_bytepos = r.last_pos;
 159                 r.peek_tok = next_token_inner(r);
 160                 r.peek_span = codemap::mk_sp(start_bytepos, r.last_pos);
 161             };
 162         }
 163     }
 164 }

 170 pub fn with_str_from<T>(rdr: @mut StringReader, start: BytePos, f: &fn(s: &str) -> T) -> T {
 171     f(rdr.src.slice(

 173             byte_offset(rdr, rdr.last_pos).to_uint()))
 174 }

 176 // EFFECT: advance the StringReader by one character. If a newline is
 177 // discovered, add it to the FileMap's list of line start offsets.
 178 pub fn bump(rdr: &mut StringReader) {
 179     rdr.last_pos = rdr.pos;
 180     let current_byte_offset = byte_offset(rdr, rdr.pos).to_uint();
 181     if current_byte_offset < (rdr.src).len() {
 182         assert!(rdr.curr != -1 as char);
 183         let last_char = rdr.curr;
 184         let next = rdr.src.char_range_at(current_byte_offset);
 185         let byte_offset_diff = next.next - current_byte_offset;
 186         rdr.pos = rdr.pos + BytePos(byte_offset_diff);
 187         rdr.curr = next.ch;
 188         rdr.col = rdr.col + CharPos(1u);
 189         if last_char == '\n' {
 190             rdr.filemap.next_line(rdr.last_pos);
 191             rdr.col = CharPos(0u);
 192         }
 193 
 194         if byte_offset_diff > 1 {
 195             rdr.filemap.record_multibyte_char(
 196                 BytePos(current_byte_offset), byte_offset_diff);
 197         }
 198     } else {
 199         rdr.curr = -1 as char;
 200     }
 201 }
 202 pub fn is_eof(rdr: @mut StringReader) -> bool {
 203     rdr.curr == -1 as char
 204 }
 205 pub fn nextch(rdr: @mut StringReader) -> char {
 206     let offset = byte_offset(rdr, rdr.pos).to_uint();
 207     if offset < (rdr.src).len() {
 208         return rdr.src.char_at(offset);
 209     } else { return -1 as char; }
 210 }

 214 fn hex_digit_val(c: char) -> int {
 215     if in_range(c, '0', '9') { return (c as int) - ('0' as int); }
 216     if in_range(c, 'a', 'f') { return (c as int) - ('a' as int) + 10; }
 217     if in_range(c, 'A', 'F') { return (c as int) - ('A' as int) + 10; }
 218     fail!();
 219 }

 240 // EFFECT: eats whitespace and comments.
 241 // returns a Some(sugared-doc-attr) if one exists, None otherwise.
 242 fn consume_whitespace_and_comments(rdr: @mut StringReader)
 243                                 -> Option<TokenAndSpan> {
 244     while is_whitespace(rdr.curr) { bump(rdr); }
 245     return consume_any_line_comment(rdr);
 246 }

 248 pub fn is_line_non_doc_comment(s: &str) -> bool {
 249     let s = s.trim_right();
 250     s.len() > 3 && s.iter().all(|ch| ch == '/')
 251 }
 252 
 253 // PRECONDITION: rdr.curr is not whitespace
 254 // EFFECT: eats any kind of comment.
 255 // returns a Some(sugared-doc-attr) if one exists, None otherwise
 256 fn consume_any_line_comment(rdr: @mut StringReader)
 257                          -> Option<TokenAndSpan> {
 258     if rdr.curr == '/' {
 259         match nextch(rdr) {
 260           '/' => {
 261             bump(rdr);
 262             bump(rdr);
 263             // line comments starting with "///" or "//!" are doc-comments
 264             if rdr.curr == '/' || rdr.curr == '!' {
 265                 let start_bpos = rdr.pos - BytePos(3u);
 266                 while rdr.curr != '\n' && !is_eof(rdr) {
 267                     bump(rdr);
 268                 }
 269                 let ret = do with_str_from(rdr, start_bpos) |string| {
 270                     // but comments with only more "/"s are not
 271                     if !is_line_non_doc_comment(string) {
 272                         Some(TokenAndSpan{
 273                             tok: token::DOC_COMMENT(str_to_ident(string)),
 274                             sp: codemap::mk_sp(start_bpos, rdr.pos)
 275                         })
 276                     } else {
 277                         None
 278                     }
 279                 };
 280 
 281                 if ret.is_some() {
 282                     return ret;
 283                 }
 284             } else {
 285                 while rdr.curr != '\n' && !is_eof(rdr) { bump(rdr); }
 286             }
 287             // Restart whitespace munch.
 288             return consume_whitespace_and_comments(rdr);
 289           }
 290           '*' => { bump(rdr); bump(rdr); return consume_block_comment(rdr); }
 291           _ => ()
 292         }
 293     } else if rdr.curr == '#' {
 294         if nextch(rdr) == '!' {
 295             // I guess this is the only way to figure out if
 296             // we're at the beginning of the file...
 297             let cmap = @CodeMap::new();
 298             (*cmap).files.push(rdr.filemap);
 299             let loc = cmap.lookup_char_pos_adj(rdr.last_pos);
 300             if loc.line == 1u && loc.col == CharPos(0u) {
 301                 while rdr.curr != '\n' && !is_eof(rdr) { bump(rdr); }
 302                 return consume_whitespace_and_comments(rdr);
 303             }
 304         }
 305     }
 306     return None;
 307 }

 309 pub fn is_block_non_doc_comment(s: &str) -> bool {
 310     assert!(s.len() >= 1u);
 311     s.slice(1u, s.len() - 1u).iter().all(|ch| ch == '*')
 312 }

 314 // might return a sugared-doc-attr
 315 fn consume_block_comment(rdr: @mut StringReader)
 316                       -> Option<TokenAndSpan> {
 317     // block comments starting with "/**" or "/*!" are doc-comments
 318     let res = if rdr.curr == '*' || rdr.curr == '!' {
 319         let start_bpos = rdr.pos - BytePos(3u);
 320         while !(rdr.curr == '*' && nextch(rdr) == '/') && !is_eof(rdr) {
 321             bump(rdr);
 322         }
 323         if is_eof(rdr) {
 324             rdr.fatal(~"unterminated block doc-comment");
 325         } else {
 326             bump(rdr);
 327             bump(rdr);
 328             do with_str_from(rdr, start_bpos) |string| {
 329                 // but comments with only "*"s between two "/"s are not
 330                 if !is_block_non_doc_comment(string) {
 331                     Some(TokenAndSpan{
 332                          tok: token::DOC_COMMENT(str_to_ident(string)),
 333                          sp: codemap::mk_sp(start_bpos, rdr.pos)
 334                          })
 335                 } else {
 336                     None
 337                 }
 338             }
 339         }
 340     } else {
 341         loop {
 342             if is_eof(rdr) { rdr.fatal(~"unterminated block comment"); }
 343             if rdr.curr == '*' && nextch(rdr) == '/' {
 344                 bump(rdr);
 345                 bump(rdr);
 346                 break;
 347             } else {
 348                 bump(rdr);
 349             }
 350         }
 351         None
 352     };
 353     // restart whitespace munch.
 354 
 355    if res.is_some() { res } else { consume_whitespace_and_comments(rdr) }
 356 }

 358 fn scan_exponent(rdr: @mut StringReader) -> Option<~str> {
 359     let mut c = rdr.curr;
 360     let mut rslt = ~"";
 361     if c == 'e' || c == 'E' {
 362         rslt.push_char(c);
 363         bump(rdr);
 364         c = rdr.curr;
 365         if c == '-' || c == '+' {
 366             rslt.push_char(c);
 367             bump(rdr);
 368         }
 369         let exponent = scan_digits(rdr, 10u);
 370         if exponent.len() > 0u {
 371             return Some(rslt + exponent);
 372         } else { rdr.fatal(~"scan_exponent: bad fp literal"); }
 373     } else { return None::<~str>; }
 374 }

 376 fn scan_digits(rdr: @mut StringReader, radix: uint) -> ~str {
 377     let mut rslt = ~"";
 378     loop {
 379         let c = rdr.curr;
 380         if c == '_' { bump(rdr); loop; }
 381         match char::to_digit(c, radix) {
 382           Some(_) => {
 383             rslt.push_char(c);
 384             bump(rdr);
 385           }
 386           _ => return rslt
 387         }
 388     };
 389 }

 391 fn scan_number(c: char, rdr: @mut StringReader) -> token::Token {
 392     let mut num_str;
 393     let mut base = 10u;
 394     let mut c = c;
 395     let mut n = nextch(rdr);
 396     if c == '0' && n == 'x' {
 397         bump(rdr);
 398         bump(rdr);
 399         base = 16u;
 400     } else if c == '0' && n == 'b' {
 401         bump(rdr);
 402         bump(rdr);
 403         base = 2u;
 404     }
 405     num_str = scan_digits(rdr, base);
 406     c = rdr.curr;
 407     nextch(rdr);
 408     if c == 'u' || c == 'i' {
 409         let signed = c == 'i';
 410         let mut tp = {
 411             if signed { either::Left(ast::ty_i) }
 412             else { either::Right(ast::ty_u) }
 413         };
 414         bump(rdr);
 415         c = rdr.curr;
 416         if c == '8' {
 417             bump(rdr);
 418             tp = if signed { either::Left(ast::ty_i8) }
 419                       else { either::Right(ast::ty_u8) };
 420         }
 421         n = nextch(rdr);
 422         if c == '1' && n == '6' {
 423             bump(rdr);
 424             bump(rdr);
 425             tp = if signed { either::Left(ast::ty_i16) }
 426                       else { either::Right(ast::ty_u16) };
 427         } else if c == '3' && n == '2' {
 428             bump(rdr);
 429             bump(rdr);
 430             tp = if signed { either::Left(ast::ty_i32) }
 431                       else { either::Right(ast::ty_u32) };
 432         } else if c == '6' && n == '4' {
 433             bump(rdr);
 434             bump(rdr);
 435             tp = if signed { either::Left(ast::ty_i64) }
 436                       else { either::Right(ast::ty_u64) };
 437         }
 438         if num_str.len() == 0u {
 439             rdr.fatal(~"no valid digits found for number");
 440         }
 441         let parsed = match u64::from_str_radix(num_str, base as uint) {
 442             Some(p) => p,
 443             None => rdr.fatal(~"int literal is too large")
 444         };
 445 
 446         match tp {
 447           either::Left(t) => return token::LIT_INT(parsed as i64, t),
 448           either::Right(t) => return token::LIT_UINT(parsed, t)
 449         }
 450     }
 451     let mut is_float = false;
 452     if rdr.curr == '.' && !(ident_start(nextch(rdr)) || nextch(rdr) == '.') {
 453         is_float = true;
 454         bump(rdr);
 455         let dec_part = scan_digits(rdr, 10u);
 456         num_str.push_char('.');
 457         num_str.push_str(dec_part);
 458     }
 459     if is_float {
 460         match base {
 461           16u => rdr.fatal(~"hexadecimal float literal is not supported"),
 462           2u => rdr.fatal(~"binary float literal is not supported"),
 463           _ => ()
 464         }
 465     }
 466     match scan_exponent(rdr) {
 467       Some(ref s) => {
 468         is_float = true;
 469         num_str.push_str(*s);
 470       }
 471       None => ()
 472     }
 473 
 474     let mut is_machine_float = false;
 475     if rdr.curr == 'f' {
 476         bump(rdr);
 477         c = rdr.curr;
 478         n = nextch(rdr);
 479         if c == '3' && n == '2' {
 480             bump(rdr);
 481             bump(rdr);
 482             return token::LIT_FLOAT(str_to_ident(num_str),
 483                                  ast::ty_f32);
 484         } else if c == '6' && n == '4' {
 485             bump(rdr);
 486             bump(rdr);
 487             return token::LIT_FLOAT(str_to_ident(num_str),
 488                                  ast::ty_f64);
 489             /* FIXME (#2252): if this is out of range for either a
 490             32-bit or 64-bit float, it won't be noticed till the
 491             back-end.  */
 492         } else {
 493             is_float = true;
 494             is_machine_float = true;
 495         }
 496     }
 497     if is_float {
 498         if is_machine_float {
 499             return token::LIT_FLOAT(str_to_ident(num_str), ast::ty_f);
 500         }
 501         return token::LIT_FLOAT_UNSUFFIXED(str_to_ident(num_str));
 502     } else {
 503         if num_str.len() == 0u {
 504             rdr.fatal(~"no valid digits found for number");
 505         }
 506         let parsed = match u64::from_str_radix(num_str, base as uint) {
 507             Some(p) => p,
 508             None => rdr.fatal(~"int literal is too large")
 509         };
 510 
 511         debug!("lexing %s as an unsuffixed integer literal",
 512                num_str);
 513         return token::LIT_INT_UNSUFFIXED(parsed as i64);
 514     }
 515 }

 517 fn scan_numeric_escape(rdr: @mut StringReader, n_hex_digits: uint) -> char {
 518     let mut accum_int = 0;
 519     let mut i = n_hex_digits;
 520     while i != 0u {
 521         let n = rdr.curr;
 522         bump(rdr);
 523         if !is_hex_digit(n) {
 524             rdr.fatal(fmt!("illegal numeric character escape: %d", n as int));
 525         }
 526         accum_int *= 16;
 527         accum_int += hex_digit_val(n);
 528         i -= 1u;
 529     }
 530     return accum_int as char;
 531 }

 533 fn ident_start(c: char) -> bool {
 534     (c >= 'a' && c <= 'z')
 535         || (c >= 'A' && c <= 'Z')
 536         || c == '_'
 537         || (c > '\x7f' && char::is_XID_start(c))
 538 }
 539 
 540 fn ident_continue(c: char) -> bool {
 541     (c >= 'a' && c <= 'z')
 542         || (c >= 'A' && c <= 'Z')
 543         || (c >= '0' && c <= '9')
 544         || c == '_'
 545         || (c > '\x7f' && char::is_XID_continue(c))
 546 }

 548 // return the next token from the string
 549 // EFFECT: advances the input past that token
 550 // EFFECT: updates the interner
 551 fn next_token_inner(rdr: @mut StringReader) -> token::Token {
 552     let c = rdr.curr;
 553     if ident_start(c) {
 554         let start = rdr.last_pos;
 555         while ident_continue(rdr.curr) {
 556             bump(rdr);
 557         }
 558 
 559         return do with_str_from(rdr, start) |string| {
 560             if string == "_" {
 561                 token::UNDERSCORE
 562             } else {
 563                 let is_mod_name = rdr.curr == ':' && nextch(rdr) == ':';
 564 
 565                 // FIXME: perform NFKC normalization here. (Issue #2253)
 566                 token::IDENT(str_to_ident(string), is_mod_name)
 567             }
 568         }
 569     }
 570     if is_dec_digit(c) {
 571         return scan_number(c, rdr);
 572     }
 573     fn binop(rdr: @mut StringReader, op: token::binop) -> token::Token {
 574         bump(rdr);
 575         if rdr.curr == '=' {
 576             bump(rdr);
 577             return token::BINOPEQ(op);
 578         } else { return token::BINOP(op); }
 579     }
 580     match c {
 581 
 582 
 583 
 584 
 585 
 586       // One-byte tokens.
 587       ';' => { bump(rdr); return token::SEMI; }
 588       ',' => { bump(rdr); return token::COMMA; }
 589       '.' => {
 590         bump(rdr);
 591         if rdr.curr == '.' && nextch(rdr) != '.' {
 592             bump(rdr);
 593             return token::DOTDOT;
 594         }
 595         return token::DOT;
 596       }
 597       '(' => { bump(rdr); return token::LPAREN; }
 598       ')' => { bump(rdr); return token::RPAREN; }
 599       '{' => { bump(rdr); return token::LBRACE; }
 600       '}' => { bump(rdr); return token::RBRACE; }
 601       '[' => { bump(rdr); return token::LBRACKET; }
 602       ']' => { bump(rdr); return token::RBRACKET; }
 603       '@' => { bump(rdr); return token::AT; }
 604       '#' => { bump(rdr); return token::POUND; }
 605       '~' => { bump(rdr); return token::TILDE; }
 606       ':' => {
 607         bump(rdr);
 608         if rdr.curr == ':' {
 609             bump(rdr);
 610             return token::MOD_SEP;
 611         } else { return token::COLON; }
 612       }
 613 
 614       '$' => { bump(rdr); return token::DOLLAR; }
 615 
 616 
 617 
 618 
 619 
 620       // Multi-byte tokens.
 621       '=' => {
 622         bump(rdr);
 623         if rdr.curr == '=' {
 624             bump(rdr);
 625             return token::EQEQ;
 626         } else if rdr.curr == '>' {
 627             bump(rdr);
 628             return token::FAT_ARROW;
 629         } else {
 630             return token::EQ;
 631         }
 632       }
 633       '!' => {
 634         bump(rdr);
 635         if rdr.curr == '=' {
 636             bump(rdr);
 637             return token::NE;
 638         } else { return token::NOT; }
 639       }
 640       '<' => {
 641         bump(rdr);
 642         match rdr.curr {
 643           '=' => { bump(rdr); return token::LE; }
 644           '<' => { return binop(rdr, token::SHL); }
 645           '-' => {
 646             bump(rdr);
 647             match rdr.curr {
 648               '>' => { bump(rdr); return token::DARROW; }
 649               _ => { return token::LARROW; }
 650             }
 651           }
 652           _ => { return token::LT; }
 653         }
 654       }
 655       '>' => {
 656         bump(rdr);
 657         match rdr.curr {
 658           '=' => { bump(rdr); return token::GE; }
 659           '>' => { return binop(rdr, token::SHR); }
 660           _ => { return token::GT; }
 661         }
 662       }
 663       '\'' => {
 664         // Either a character constant 'a' OR a lifetime name 'abc
 665         bump(rdr);
 666         let start = rdr.last_pos;
 667         let mut c2 = rdr.curr;
 668         bump(rdr);
 669 
 670         // If the character is an ident start not followed by another single
 671         // quote, then this is a lifetime name:
 672         if ident_start(c2) && rdr.curr != '\'' {
 673             while ident_continue(rdr.curr) {
 674                 bump(rdr);
 675             }
 676             return do with_str_from(rdr, start) |lifetime_name| {
 677                 token::LIFETIME(str_to_ident(lifetime_name))
 678             }
 679         }
 680 
 681         // Otherwise it is a character constant:
 682         if c2 == '\\' {
 683             // '\X' for some X must be a character constant:
 684             let escaped = rdr.curr;
 685             bump(rdr);
 686             match escaped {
 687               'n' => { c2 = '\n'; }
 688               'r' => { c2 = '\r'; }
 689               't' => { c2 = '\t'; }
 690               '\\' => { c2 = '\\'; }
 691               '\'' => { c2 = '\''; }
 692               '"' => { c2 = '"'; }
 693               'x' => { c2 = scan_numeric_escape(rdr, 2u); }
 694               'u' => { c2 = scan_numeric_escape(rdr, 4u); }
 695               'U' => { c2 = scan_numeric_escape(rdr, 8u); }
 696               c2 => {
 697                 rdr.fatal(fmt!("unknown character escape: %d", c2 as int));
 698               }
 699             }
 700         }
 701         if rdr.curr != '\'' {
 702             rdr.fatal(~"unterminated character constant");
 703         }
 704         bump(rdr); // advance curr past token
 705         return token::LIT_INT(c2 as i64, ast::ty_char);
 706       }
 707       '"' => {
 708         let mut accum_str = ~"";
 709         let n = rdr.last_pos;
 710         bump(rdr);
 711         while rdr.curr != '"' {
 712             if is_eof(rdr) {
 713                 do with_str_from(rdr, n) |s| {
 714                     rdr.fatal(fmt!("unterminated double quote string: %s", s));
 715                 }
 716             }
 717 
 718             let ch = rdr.curr;
 719             bump(rdr);
 720             match ch {
 721               '\\' => {
 722                 let escaped = rdr.curr;
 723                 bump(rdr);
 724                 match escaped {
 725                   'n' => accum_str.push_char('\n'),
 726                   'r' => accum_str.push_char('\r'),
 727                   't' => accum_str.push_char('\t'),
 728                   '\\' => accum_str.push_char('\\'),
 729                   '\'' => accum_str.push_char('\''),
 730                   '"' => accum_str.push_char('"'),
 731                   '\n' => consume_whitespace(rdr),
 732                   'x' => {
 733                     accum_str.push_char(scan_numeric_escape(rdr, 2u));
 734                   }
 735                   'u' => {
 736                     accum_str.push_char(scan_numeric_escape(rdr, 4u));
 737                   }
 738                   'U' => {
 739                     accum_str.push_char(scan_numeric_escape(rdr, 8u));
 740                   }
 741                   c2 => {
 742                     rdr.fatal(fmt!("unknown string escape: %d", c2 as int));
 743                   }
 744                 }
 745               }
 746               _ => accum_str.push_char(ch)
 747             }
 748         }
 749         bump(rdr);
 750         return token::LIT_STR(str_to_ident(accum_str));
 751       }
 752       '-' => {
 753         if nextch(rdr) == '>' {
 754             bump(rdr);
 755             bump(rdr);
 756             return token::RARROW;
 757         } else { return binop(rdr, token::MINUS); }
 758       }
 759       '&' => {
 760         if nextch(rdr) == '&' {
 761             bump(rdr);
 762             bump(rdr);
 763             return token::ANDAND;
 764         } else { return binop(rdr, token::AND); }
 765       }
 766       '|' => {
 767         match nextch(rdr) {
 768           '|' => { bump(rdr); bump(rdr); return token::OROR; }
 769           _ => { return binop(rdr, token::OR); }
 770         }
 771       }
 772       '+' => { return binop(rdr, token::PLUS); }
 773       '*' => { return binop(rdr, token::STAR); }
 774       '/' => { return binop(rdr, token::SLASH); }
 775       '^' => { return binop(rdr, token::CARET); }
 776       '%' => { return binop(rdr, token::PERCENT); }
 777       c => {
 778           // So the error span points to the unrecognized character
 779           rdr.peek_span = codemap::mk_sp(rdr.last_pos, rdr.pos);
 780           rdr.fatal(fmt!("unknown start of token: %d", c as int));
 781       }
 782     }
 783 }

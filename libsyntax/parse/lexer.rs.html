  225  // EFFECT: advance the StringReader by one character. If a newline is
  226  // discovered, add it to the FileMap's list of line start offsets.
  227  pub fn bump(rdr: &mut StringReader) {
  228      rdr.last_pos = rdr.pos;
  229      let current_byte_offset = byte_offset(rdr, rdr.pos).to_uint();
  230      if current_byte_offset < rdr.filemap.src.len() {
  231          assert!(rdr.curr.is_some());
  232          let last_char = rdr.curr.unwrap();
  233          let next = rdr.filemap.src.char_range_at(current_byte_offset);
  234          let byte_offset_diff = next.next - current_byte_offset;
  235          rdr.pos = rdr.pos + Pos::from_uint(byte_offset_diff);
  236          rdr.curr = Some(next.ch);
  237          rdr.col = rdr.col + CharPos(1u);
  238          if last_char == '\n' {
  239              rdr.filemap.next_line(rdr.last_pos);
  240              rdr.col = CharPos(0u);
  241          }
  242  
  243          if byte_offset_diff > 1 {
  244              rdr.filemap.record_multibyte_char(rdr.last_pos, byte_offset_diff);
  245          }
  246      } else {
  247          rdr.curr = None;
  248      }
  249  }

  324  // PRECONDITION: rdr.curr is not whitespace
  325  // EFFECT: eats any kind of comment.
  326  // returns a Some(sugared-doc-attr) if one exists, None otherwise
  327  fn consume_any_line_comment(rdr: &mut StringReader)
  328                           -> Option<TokenAndSpan> {
  329      if rdr.curr_is('/') {
  330          match nextch(rdr) {
  331              Some('/') => {
  332                  bump(rdr);
  333                  bump(rdr);
  334                  // line comments starting with "///" or "//!" are doc-comments
  335                  if rdr.curr_is('/') || rdr.curr_is('!') {
  336                      let start_bpos = rdr.pos - BytePos(3);
  337                      while !rdr.curr_is('\n') && !is_eof(rdr) {
  338                          bump(rdr);
  339                      }
  340                      let ret = with_str_from(rdr, start_bpos, |string| {
  341                          // but comments with only more "/"s are not
  342                          if !is_line_non_doc_comment(string) {
  343                              Some(TokenAndSpan{
  344                                  tok: token::DOC_COMMENT(str_to_ident(string)),
  345                                  sp: codemap::mk_sp(start_bpos, rdr.pos)
  346                              })
  347                          } else {
  348                              None
  349                          }
  350                      });
  351  
  352                      if ret.is_some() {
  353                          return ret;
  354                      }
  355                  } else {
  356                      while !rdr.curr_is('\n') && !is_eof(rdr) { bump(rdr); }
  357                  }
  358                  // Restart whitespace munch.
  359                  consume_whitespace_and_comments(rdr)
  360              }
  361              Some('*') => { bump(rdr); bump(rdr); consume_block_comment(rdr) }
  362              _ => None
  363          }
  364      } else if rdr.curr_is('#') {
  365          if nextch_is(rdr, '!') {
  366  
  367              // Parse an inner attribute.
  368              if nextnextch_is(rdr, '[') {
  369                  return None;
  370              }
  371  
  372              // I guess this is the only way to figure out if
  373              // we're at the beginning of the file...
  374              let cmap = CodeMap::new();
  375              cmap.files.borrow_mut().push(rdr.filemap.clone());
  376              let loc = cmap.lookup_char_pos_adj(rdr.last_pos);
  377              if loc.line == 1u && loc.col == CharPos(0u) {
  378                  while !rdr.curr_is('\n') && !is_eof(rdr) { bump(rdr); }
  379                  return consume_whitespace_and_comments(rdr);
  380              }
  381          }
  382          None
  383      } else {
  384          None
  385      }
  386  }

  392  // might return a sugared-doc-attr
  393  fn consume_block_comment(rdr: &mut StringReader) -> Option<TokenAndSpan> {
  394      // block comments starting with "/**" or "/*!" are doc-comments
  395      let is_doc_comment = rdr.curr_is('*') || rdr.curr_is('!');
  396      let start_bpos = rdr.pos - BytePos(if is_doc_comment {3} else {2});
  397  
  398      let mut level: int = 1;
  399      while level > 0 {
  400          if is_eof(rdr) {
  401              let msg = if is_doc_comment {
  402                  ~"unterminated block doc-comment"
  403              } else {
  404                  ~"unterminated block comment"
  405              };
  406              fatal_span(rdr, start_bpos, rdr.last_pos, msg);
  407          } else if rdr.curr_is('/') && nextch_is(rdr, '*') {
  408              level += 1;
  409              bump(rdr);
  410              bump(rdr);
  411          } else if rdr.curr_is('*') && nextch_is(rdr, '/') {
  412              level -= 1;

  414              bump(rdr);
  415          } else {
  416              bump(rdr);
  417          }
  418      }

  421          with_str_from(rdr, start_bpos, |string| {

  425                          tok: token::DOC_COMMENT(str_to_ident(string)),

  427                      })
  428              } else {
  429                  None
  430              }
  431          })
  432      } else {
  433          None
  434      };
  435  
  436      // restart whitespace munch.
  437      if res.is_some() { res } else { consume_whitespace_and_comments(rdr) }
  438  }

  440  fn scan_exponent(rdr: &mut StringReader, start_bpos: BytePos) -> Option<~str> {
  441      // \x00 hits the `return None` case immediately, so this is fine.
  442      let mut c = rdr.curr.unwrap_or('\x00');
  443      let mut rslt = ~"";
  444      if c == 'e' || c == 'E' {
  445          rslt.push_char(c);
  446          bump(rdr);
  447          c = rdr.curr.unwrap_or('\x00');
  448          if c == '-' || c == '+' {
  449              rslt.push_char(c);
  450              bump(rdr);
  451          }
  452          let exponent = scan_digits(rdr, 10u);
  453          if exponent.len() > 0u {
  454              return Some(rslt + exponent);
  455          } else {
  456              fatal_span(rdr, start_bpos, rdr.last_pos,
  457                         ~"scan_exponent: bad fp literal");
  458          }
  459      } else { return None::<~str>; }
  460  }

  462  fn scan_digits(rdr: &mut StringReader, radix: uint) -> ~str {
  463      let mut rslt = ~"";
  464      loop {
  465          let c = rdr.curr;
  466          if c == Some('_') { bump(rdr); continue; }
  467          match c.and_then(|cc| char::to_digit(cc, radix)) {
  468            Some(_) => {
  469              rslt.push_char(c.unwrap());
  470              bump(rdr);
  471            }
  472            _ => return rslt
  473          }
  474      };
  475  }

  487      }

  490  fn scan_number(c: char, rdr: &mut StringReader) -> token::Token {
  491      let mut num_str;
  492      let mut base = 10u;
  493      let mut c = c;
  494      let mut n = nextch(rdr).unwrap_or('\x00');
  495      let start_bpos = rdr.last_pos;
  496      if c == '0' && n == 'x' {
  497          bump(rdr);
  498          bump(rdr);
  499          base = 16u;
  500      } else if c == '0' && n == 'o' {
  501          bump(rdr);
  502          bump(rdr);
  503          base = 8u;
  504      } else if c == '0' && n == 'b' {
  505          bump(rdr);
  506          bump(rdr);
  507          base = 2u;
  508      }
  509      num_str = scan_digits(rdr, base);
  510      c = rdr.curr.unwrap_or('\x00');
  511      nextch(rdr);
  512      if c == 'u' || c == 'i' {
  513          enum Result { Signed(ast::IntTy), Unsigned(ast::UintTy) }
  514          let signed = c == 'i';
  515          let mut tp = {
  516              if signed { Signed(ast::TyI) }
  517              else { Unsigned(ast::TyU) }
  518          };
  519          bump(rdr);
  520          c = rdr.curr.unwrap_or('\x00');
  521          if c == '8' {
  522              bump(rdr);
  523              tp = if signed { Signed(ast::TyI8) }
  524                        else { Unsigned(ast::TyU8) };
  525          }
  526          n = nextch(rdr).unwrap_or('\x00');
  527          if c == '1' && n == '6' {
  528              bump(rdr);
  529              bump(rdr);
  530              tp = if signed { Signed(ast::TyI16) }
  531                        else { Unsigned(ast::TyU16) };
  532          } else if c == '3' && n == '2' {
  533              bump(rdr);
  534              bump(rdr);
  535              tp = if signed { Signed(ast::TyI32) }
  536                        else { Unsigned(ast::TyU32) };
  537          } else if c == '6' && n == '4' {
  538              bump(rdr);
  539              bump(rdr);
  540              tp = if signed { Signed(ast::TyI64) }
  541                        else { Unsigned(ast::TyU64) };
  542          }
  543          if num_str.len() == 0u {
  544              fatal_span(rdr, start_bpos, rdr.last_pos,
  545                         ~"no valid digits found for number");
  546          }
  547          let parsed = match from_str_radix::<u64>(num_str, base as uint) {
  548              Some(p) => p,
  549              None => fatal_span(rdr, start_bpos, rdr.last_pos,
  550                                 ~"int literal is too large")
  551          };
  552  
  553          match tp {
  554            Signed(t) => return token::LIT_INT(parsed as i64, t),
  555            Unsigned(t) => return token::LIT_UINT(parsed, t)
  556          }
  557      }
  558      let mut is_float = false;
  559      if rdr.curr_is('.') && !(ident_start(nextch(rdr)) || nextch_is(rdr, '.')) {
  560          is_float = true;
  561          bump(rdr);
  562          let dec_part = scan_digits(rdr, 10u);
  563          num_str.push_char('.');
  564          num_str.push_str(dec_part);
  565      }
  566      match scan_exponent(rdr, start_bpos) {
  567        Some(ref s) => {
  568          is_float = true;
  569          num_str.push_str(*s);
  570        }
  571        None => ()
  572      }
  573  
  574      if rdr.curr_is('f') {
  575          bump(rdr);
  576          c = rdr.curr.unwrap_or('\x00');
  577          n = nextch(rdr).unwrap_or('\x00');
  578          if c == '3' && n == '2' {
  579              bump(rdr);
  580              bump(rdr);
  581              check_float_base(rdr, start_bpos, rdr.last_pos, base);
  582              return token::LIT_FLOAT(str_to_ident(num_str), ast::TyF32);
  583          } else if c == '6' && n == '4' {
  584              bump(rdr);
  585              bump(rdr);
  586              check_float_base(rdr, start_bpos, rdr.last_pos, base);
  587              return token::LIT_FLOAT(str_to_ident(num_str), ast::TyF64);
  588              /* FIXME (#2252): if this is out of range for either a
  589              32-bit or 64-bit float, it won't be noticed till the
  590              back-end.  */
  591          } else {
  592              fatal_span(rdr, start_bpos, rdr.last_pos,
  593                         ~"expected `f32` or `f64` suffix");
  594          }
  595      }
  596      if is_float {
  597          check_float_base(rdr, start_bpos, rdr.last_pos, base);
  598          return token::LIT_FLOAT_UNSUFFIXED(str_to_ident(num_str));
  599      } else {
  600          if num_str.len() == 0u {
  601              fatal_span(rdr, start_bpos, rdr.last_pos,
  602                         ~"no valid digits found for number");
  603          }
  604          let parsed = match from_str_radix::<u64>(num_str, base as uint) {
  605              Some(p) => p,
  606              None => fatal_span(rdr, start_bpos, rdr.last_pos,
  607                                 ~"int literal is too large")
  608          };
  609  
  610          debug!("lexing {} as an unsuffixed integer literal", num_str);
  611          return token::LIT_INT_UNSUFFIXED(parsed as i64);
  612      }
  613  }

  615  fn scan_numeric_escape(rdr: &mut StringReader, n_hex_digits: uint) -> char {

  637          Some(x) => x,
  638          None => fatal_span(rdr, start_bpos, rdr.last_pos,

  640      }
  641  }

  643  fn ident_start(c: Option<char>) -> bool {
  644      let c = match c { Some(c) => c, None => return false };
  645  
  646      (c >= 'a' && c <= 'z')
  647          || (c >= 'A' && c <= 'Z')
  648          || c == '_'
  649          || (c > '\x7f' && char::is_XID_start(c))
  650  }
  651  
  652  fn ident_continue(c: Option<char>) -> bool {
  653      let c = match c { Some(c) => c, None => return false };
  654  
  655      (c >= 'a' && c <= 'z')
  656          || (c >= 'A' && c <= 'Z')
  657          || (c >= '0' && c <= '9')
  658          || c == '_'
  659          || (c > '\x7f' && char::is_XID_continue(c))
  660  }
  661  
  662  // return the next token from the string
  663  // EFFECT: advances the input past that token
  664  // EFFECT: updates the interner
  665  fn next_token_inner(rdr: &mut StringReader) -> token::Token {
  666      let c = rdr.curr;
  667      if ident_start(c) && !nextch_is(rdr, '"') && !nextch_is(rdr, '#') {
  668          // Note: r as in r" or r#" is part of a raw string literal,
  669          // not an identifier, and is handled further down.
  670  
  671          let start = rdr.last_pos;
  672          while ident_continue(rdr.curr) {
  673              bump(rdr);
  674          }
  675  
  676          return with_str_from(rdr, start, |string| {
  677              if string == "_" {
  678                  token::UNDERSCORE
  679              } else {
  680                  let is_mod_name = rdr.curr_is(':') && nextch_is(rdr, ':');
  681  
  682                  // FIXME: perform NFKC normalization here. (Issue #2253)
  683                  token::IDENT(str_to_ident(string), is_mod_name)
  684              }
  685          })
  686      }
  687      if is_dec_digit(c) {
  688          return scan_number(c.unwrap(), rdr);
  689      }
  690      fn binop(rdr: &mut StringReader, op: token::BinOp) -> token::Token {
  691          bump(rdr);
  692          if rdr.curr_is('=') {
  693              bump(rdr);
  694              return token::BINOPEQ(op);
  695          } else { return token::BINOP(op); }
  696      }
  697      match c.expect("next_token_inner called at EOF") {
  698  
  699  
  700  
  701  
  702  
  703        // One-byte tokens.
  704        ';' => { bump(rdr); return token::SEMI; }
  705        ',' => { bump(rdr); return token::COMMA; }
  706        '.' => {
  707            bump(rdr);
  708            return if rdr.curr_is('.') {
  709                bump(rdr);
  710                if rdr.curr_is('.') {
  711                    bump(rdr);
  712                    token::DOTDOTDOT
  713                } else {
  714                    token::DOTDOT
  715                }
  716            } else {
  717                token::DOT
  718            };
  719        }
  720        '(' => { bump(rdr); return token::LPAREN; }
  721        ')' => { bump(rdr); return token::RPAREN; }
  722        '{' => { bump(rdr); return token::LBRACE; }
  723        '}' => { bump(rdr); return token::RBRACE; }
  724        '[' => { bump(rdr); return token::LBRACKET; }
  725        ']' => { bump(rdr); return token::RBRACKET; }
  726        '@' => { bump(rdr); return token::AT; }
  727        '#' => { bump(rdr); return token::POUND; }
  728        '~' => { bump(rdr); return token::TILDE; }
  729        ':' => {
  730          bump(rdr);
  731          if rdr.curr_is(':') {
  732              bump(rdr);
  733              return token::MOD_SEP;
  734          } else { return token::COLON; }
  735        }
  736  
  737        '$' => { bump(rdr); return token::DOLLAR; }
  738  
  739  
  740  
  741  
  742  
  743        // Multi-byte tokens.
  744        '=' => {
  745          bump(rdr);
  746          if rdr.curr_is('=') {
  747              bump(rdr);
  748              return token::EQEQ;
  749          } else if rdr.curr_is('>') {
  750              bump(rdr);
  751              return token::FAT_ARROW;
  752          } else {
  753              return token::EQ;
  754          }
  755        }
  756        '!' => {
  757          bump(rdr);
  758          if rdr.curr_is('=') {
  759              bump(rdr);
  760              return token::NE;
  761          } else { return token::NOT; }
  762        }
  763        '<' => {
  764          bump(rdr);
  765          match rdr.curr.unwrap_or('\x00') {
  766            '=' => { bump(rdr); return token::LE; }
  767            '<' => { return binop(rdr, token::SHL); }
  768            '-' => {
  769              bump(rdr);
  770              match rdr.curr.unwrap_or('\x00') {
  771                '>' => { bump(rdr); return token::DARROW; }
  772                _ => { return token::LARROW; }
  773              }
  774            }
  775            _ => { return token::LT; }
  776          }
  777        }
  778        '>' => {
  779          bump(rdr);
  780          match rdr.curr.unwrap_or('\x00') {
  781            '=' => { bump(rdr); return token::GE; }
  782            '>' => { return binop(rdr, token::SHR); }
  783            _ => { return token::GT; }
  784          }
  785        }
  786        '\'' => {
  787          // Either a character constant 'a' OR a lifetime name 'abc
  788          bump(rdr);
  789          let start = rdr.last_pos;
  790  
  791          // the eof will be picked up by the final `'` check below
  792          let mut c2 = rdr.curr.unwrap_or('\x00');
  793          bump(rdr);
  794  
  795          // If the character is an ident start not followed by another single
  796          // quote, then this is a lifetime name:
  797          if ident_start(Some(c2)) && !rdr.curr_is('\'') {
  798              while ident_continue(rdr.curr) {
  799                  bump(rdr);
  800              }
  801              let ident = with_str_from(rdr, start, |lifetime_name| {
  802                  str_to_ident(lifetime_name)
  803              });
  804              let tok = &token::IDENT(ident, false);
  805  
  806              if token::is_keyword(token::keywords::Self, tok) {
  807                  fatal_span(rdr, start, rdr.last_pos,
  808                             ~"invalid lifetime name: 'self is no longer a special lifetime");
  809              } else if token::is_any_keyword(tok) &&
  810                  !token::is_keyword(token::keywords::Static, tok) {
  811                  fatal_span(rdr, start, rdr.last_pos,
  812                             ~"invalid lifetime name");
  813              } else {
  814                  return token::LIFETIME(ident);
  815              }
  816          }
  817  
  818          // Otherwise it is a character constant:
  819          match c2 {
  820              '\\' => {

 173  // given a function and parsing information (source str,
 174  // filename, crate cfg, and sess), create a parser,
 175  // apply the function, and check that the parser
 176  // consumed all of the input before returning the function's
 177  // result.
 178  pub fn parse_from_source_str<T>(
 179      f: &fn(&Parser) -> T,
 180      name: @str, ss: codemap::FileSubstr,
 181      source: @str,
 182      cfg: ast::CrateConfig,
 183      sess: @mut ParseSess
 184  ) -> T {
 185      let p = new_parser_from_source_substr(
 186          sess,
 187          cfg,
 188          name,
 189          ss,
 190          source
 191      );
 192      let r = f(&p);
 193      if !p.reader.is_eof() {
 194          p.reader.fatal(~"expected end-of-string");
 195      }
 196      maybe_aborted(r,p)
 197  }

 259  /// Given a session and a path and an optional span (for error reporting),
 260  /// add the path to the session's codemap and return the new filemap.
 261  pub fn file_to_filemap(sess: @mut ParseSess, path: &Path, spanopt: Option<Span>)
 262      -> @FileMap {
 263      match io::read_whole_file_str(path) {
 264          Ok(src) => string_to_filemap(sess, src.to_managed(), path.to_str().to_managed()),
 265          Err(e) => {
 266              match spanopt {
 267                  Some(span) => sess.span_diagnostic.span_fatal(span, e),
 268                  None => sess.span_diagnostic.handler().fatal(e)
 269              }
 270          }
 271      }
 272  }

 288  // given a filemap, produce a sequence of token-trees
 289  pub fn filemap_to_tts(sess: @mut ParseSess, filemap: @FileMap)
 290      -> ~[ast::token_tree] {
 291      // it appears to me that the cfg doesn't matter here... indeed,
 292      // parsing tt's probably shouldn't require a parser at all.
 293      let cfg = ~[];
 294      let srdr = lexer::new_string_reader(sess.span_diagnostic, filemap);
 295      let p1 = Parser(sess, cfg, srdr as @mut reader);
 296      p1.parse_all_token_trees()
 297  }

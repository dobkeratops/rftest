 178 // given a function and parsing information (source str,
 179 // filename, crate cfg, and sess), create a parser,
 180 // apply the function, and check that the parser
 181 // consumed all of the input before returning the function's
 182 // result.
 183 pub fn parse_from_source_str<T>(
 184     f: &fn(&Parser) -> T,
 185     name: @str, ss: codemap::FileSubstr,
 186     source: @str,
 187     cfg: ast::CrateConfig,
 188     sess: @mut ParseSess
 189 ) -> T {
 190     let p = new_parser_from_source_substr(
 191         sess,
 192         cfg,
 193         name,
 194         ss,
 195         source
 196     );
 197     let r = f(&p);
 198     if !p.reader.is_eof() {
 199         p.reader.fatal(~"expected end-of-string");
 200     }
 201     maybe_aborted(r,p)
 202 }

 204 // return the next unused node id.
 205 pub fn next_node_id(sess: @mut ParseSess) -> NodeId {
 206     let rv = sess.next_id;
 207     sess.next_id += 1;
 208     // ID 0 is reserved for the crate and doesn't actually exist in the AST
 209     assert!(rv != 0);
 210     return rv;
 211 }

 273 /// Given a session and a path and an optional span (for error reporting),
 274 /// add the path to the session's codemap and return the new filemap.
 275 pub fn file_to_filemap(sess: @mut ParseSess, path: &Path, spanopt: Option<span>)
 276     -> @FileMap {
 277     match io::read_whole_file_str(path) {
 278         Ok(src) => string_to_filemap(sess, src.to_managed(), path.to_str().to_managed()),
 279         Err(e) => {
 280             match spanopt {
 281                 Some(span) => sess.span_diagnostic.span_fatal(span, e),
 282                 None => sess.span_diagnostic.handler().fatal(e)
 283             }
 284         }
 285     }
 286 }

 302 // given a filemap, produce a sequence of token-trees
 303 pub fn filemap_to_tts(sess: @mut ParseSess, filemap: @FileMap)
 304     -> ~[ast::token_tree] {
 305     // it appears to me that the cfg doesn't matter here... indeed,
 306     // parsing tt's probably shouldn't require a parser at all.
 307     let cfg = ~[];
 308     let srdr = lexer::new_string_reader(sess.span_diagnostic, filemap);
 309     let p1 = Parser(sess, cfg, srdr as @reader);
 310     p1.parse_all_token_trees()
 311 }

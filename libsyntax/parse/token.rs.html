 129  pub fn to_str(input: @ident_interner, t: &Token) -> ~str {
 130      match *t {
 131        EQ => ~"=",
 132        LT => ~"<",
 133        LE => ~"<=",
 134        EQEQ => ~"==",
 135        NE => ~"!=",
 136        GE => ~">=",
 137        GT => ~">",
 138        NOT => ~"!",
 139        TILDE => ~"~",
 140        OROR => ~"||",
 141        ANDAND => ~"&&",
 142        BINOP(op) => binop_to_str(op),
 143        BINOPEQ(op) => binop_to_str(op) + "=",
 144  
 145        /* Structural symbols */
 146        AT => ~"@",
 147        DOT => ~".",
 148        DOTDOT => ~"..",
 149        COMMA => ~",",
 150        SEMI => ~";",
 151        COLON => ~":",
 152        MOD_SEP => ~"::",
 153        RARROW => ~"->",
 154        LARROW => ~"<-",
 155        DARROW => ~"<->",
 156        FAT_ARROW => ~"=>",
 157        LPAREN => ~"(",
 158        RPAREN => ~")",
 159        LBRACKET => ~"[",
 160        RBRACKET => ~"]",
 161        LBRACE => ~"{",
 162        RBRACE => ~"}",
 163        POUND => ~"#",
 164        DOLLAR => ~"$",
 165  
 166        /* Literals */
 167        LIT_INT(c, ast::ty_char) => {
 168            let mut res = ~"'";
 169            do (c as char).escape_default |c| {
 170                res.push_char(c);
 171            }
 172            res.push_char('\'');
 173            res
 174        }
 175        LIT_INT(i, t) => {
 176            i.to_str() + ast_util::int_ty_to_str(t)
 177        }
 178        LIT_UINT(u, t) => {
 179            u.to_str() + ast_util::uint_ty_to_str(t)
 180        }
 181        LIT_INT_UNSUFFIXED(i) => { i.to_str() }
 182        LIT_FLOAT(ref s, t) => {
 183          let mut body = ident_to_str(s).to_owned();
 184          if body.ends_with(".") {
 185              body.push_char('0');  // `10.f` is not a float literal
 186          }
 187          body + ast_util::float_ty_to_str(t)
 188        }
 189        LIT_FLOAT_UNSUFFIXED(ref s) => {
 190          let mut body = ident_to_str(s).to_owned();
 191          if body.ends_with(".") {
 192              body.push_char('0');  // `10.f` is not a float literal
 193          }
 194          body
 195        }
 196        LIT_STR(ref s) => { fmt!("\"%s\"", ident_to_str(s).escape_default()) }
 197  
 198        /* Name components */
 199        IDENT(s, _) => input.get(s.name).to_owned(),
 200        LIFETIME(s) => fmt!("'%s", input.get(s.name)),
 201        UNDERSCORE => ~"_",
 202  
 203        /* Other */
 204        DOC_COMMENT(ref s) => ident_to_str(s).to_owned(),
 205        EOF => ~"<eof>",
 206        INTERPOLATED(ref nt) => {
 207          match nt {
 208              &nt_expr(e) => ::print::pprust::expr_to_str(e, input),
 209              &nt_attr(e) => ::print::pprust::attribute_to_str(e, input),
 210              _ => {
 211                  ~"an interpolated " +
 212                      match (*nt) {
 213                        nt_item(*) => ~"item",
 214                        nt_block(*) => ~"block",
 215                        nt_stmt(*) => ~"statement",
 216                        nt_pat(*) => ~"pattern",
 217                        nt_attr(*) => fail!("should have been handled"),
 218                        nt_expr(*) => fail!("should have been handled above"),
 219                        nt_ty(*) => ~"type",
 220                        nt_ident(*) => ~"identifier",
 221                        nt_path(*) => ~"path",
 222                        nt_tt(*) => ~"tt",
 223                        nt_matchers(*) => ~"matcher sequence"
 224                      }
 225              }
 226          }
 227        }
 228      }
 229  }

 231  pub fn can_begin_expr(t: &Token) -> bool {
 232      match *t {
 233        LPAREN => true,
 234        LBRACE => true,
 235        LBRACKET => true,
 236        IDENT(_, _) => true,
 237        UNDERSCORE => true,
 238        TILDE => true,
 239        LIT_INT(_, _) => true,
 240        LIT_UINT(_, _) => true,
 241        LIT_INT_UNSUFFIXED(_) => true,
 242        LIT_FLOAT(_, _) => true,
 243        LIT_FLOAT_UNSUFFIXED(_) => true,
 244        LIT_STR(_) => true,
 245        POUND => true,
 246        AT => true,
 247        NOT => true,
 248        BINOP(MINUS) => true,
 249        BINOP(STAR) => true,
 250        BINOP(AND) => true,
 251        BINOP(OR) => true, // in lambda syntax
 252        OROR => true, // in lambda syntax
 253        MOD_SEP => true,
 254        INTERPOLATED(nt_expr(*))
 255        | INTERPOLATED(nt_ident(*))
 256        | INTERPOLATED(nt_block(*))
 257        | INTERPOLATED(nt_path(*)) => true,
 258        _ => false
 259      }
 260  }

 370        GE             => Some(ast::ge),
 371        GT             => Some(ast::gt),

 128  pub fn to_str(input: @ident_interner, t: &Token) -> ~str {
 129      match *t {
 130        EQ => ~"=",
 131        LT => ~"<",
 132        LE => ~"<=",
 133        EQEQ => ~"==",
 134        NE => ~"!=",
 135        GE => ~">=",
 136        GT => ~">",
 137        NOT => ~"!",
 138        TILDE => ~"~",
 139        OROR => ~"||",
 140        ANDAND => ~"&&",
 141        BINOP(op) => binop_to_str(op),
 142        BINOPEQ(op) => binop_to_str(op) + "=",
 143  
 144        /* Structural symbols */
 145        AT => ~"@",
 146        DOT => ~".",
 147        DOTDOT => ~"..",
 148        COMMA => ~",",
 149        SEMI => ~";",
 150        COLON => ~":",
 151        MOD_SEP => ~"::",
 152        RARROW => ~"->",
 153        LARROW => ~"<-",
 154        DARROW => ~"<->",
 155        FAT_ARROW => ~"=>",
 156        LPAREN => ~"(",
 157        RPAREN => ~")",
 158        LBRACKET => ~"[",
 159        RBRACKET => ~"]",
 160        LBRACE => ~"{",
 161        RBRACE => ~"}",
 162        POUND => ~"#",
 163        DOLLAR => ~"$",
 164  
 165        /* Literals */
 166        LIT_INT(c, ast::ty_char) => {
 167            let mut res = ~"'";
 168            do (c as char).escape_default |c| {
 169                res.push_char(c);
 170            }
 171            res.push_char('\'');
 172            res
 173        }
 174        LIT_INT(i, t) => {
 175            i.to_str() + ast_util::int_ty_to_str(t)
 176        }
 177        LIT_UINT(u, t) => {
 178            u.to_str() + ast_util::uint_ty_to_str(t)
 179        }
 180        LIT_INT_UNSUFFIXED(i) => { i.to_str() }
 181        LIT_FLOAT(ref s, t) => {
 182          let mut body = ident_to_str(s).to_owned();
 183          if body.ends_with(".") {
 184              body.push_char('0');  // `10.f` is not a float literal
 185          }
 186          body + ast_util::float_ty_to_str(t)
 187        }
 188        LIT_FLOAT_UNSUFFIXED(ref s) => {
 189          let mut body = ident_to_str(s).to_owned();
 190          if body.ends_with(".") {
 191              body.push_char('0');  // `10.f` is not a float literal
 192          }
 193          body
 194        }
 195        LIT_STR(ref s) => { fmt!("\"%s\"", ident_to_str(s).escape_default()) }
 196  
 197        /* Name components */
 198        IDENT(s, _) => input.get(s.name).to_owned(),
 199        LIFETIME(s) => fmt!("'%s", input.get(s.name)),
 200        UNDERSCORE => ~"_",
 201  
 202        /* Other */
 203        DOC_COMMENT(ref s) => ident_to_str(s).to_owned(),
 204        EOF => ~"<eof>",
 205        INTERPOLATED(ref nt) => {
 206          match nt {
 207              &nt_expr(e) => ::print::pprust::expr_to_str(e, input),
 208              _ => {
 209                  ~"an interpolated " +
 210                      match (*nt) {
 211                        nt_item(*) => ~"item",
 212                        nt_block(*) => ~"block",
 213                        nt_stmt(*) => ~"statement",
 214                        nt_pat(*) => ~"pattern",
 215                        nt_expr(*) => fail!("should have been handled above"),
 216                        nt_ty(*) => ~"type",
 217                        nt_ident(*) => ~"identifier",
 218                        nt_path(*) => ~"path",
 219                        nt_tt(*) => ~"tt",
 220                        nt_matchers(*) => ~"matcher sequence"
 221                      }
 222              }
 223          }
 224        }
 225      }
 226  }

 228  pub fn can_begin_expr(t: &Token) -> bool {
 229      match *t {
 230        LPAREN => true,
 231        LBRACE => true,
 232        LBRACKET => true,
 233        IDENT(_, _) => true,
 234        UNDERSCORE => true,
 235        TILDE => true,
 236        LIT_INT(_, _) => true,
 237        LIT_UINT(_, _) => true,
 238        LIT_INT_UNSUFFIXED(_) => true,
 239        LIT_FLOAT(_, _) => true,
 240        LIT_FLOAT_UNSUFFIXED(_) => true,
 241        LIT_STR(_) => true,
 242        POUND => true,
 243        AT => true,
 244        NOT => true,
 245        BINOP(MINUS) => true,
 246        BINOP(STAR) => true,
 247        BINOP(AND) => true,
 248        BINOP(OR) => true, // in lambda syntax
 249        OROR => true, // in lambda syntax
 250        MOD_SEP => true,
 251        INTERPOLATED(nt_expr(*))
 252        | INTERPOLATED(nt_ident(*))
 253        | INTERPOLATED(nt_block(*))
 254        | INTERPOLATED(nt_path(*)) => true,
 255        _ => false
 256      }
 257  }

 364        BINOP(OR)      => Some(ast::bitor),

 366        LE             => Some(ast::le),
 367        GE             => Some(ast::ge),
 368        GT             => Some(ast::gt),
 369        EQEQ           => Some(ast::eq),
 370        NE             => Some(ast::ne),

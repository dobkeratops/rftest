  320 pub fn commasep<T>(s: @ps, b: breaks, elts: &[T], op: &fn(@ps, &T)) {
  321     box(s, 0u, b);
  322     let mut first = true;
  323     for elt in elts.iter() {
  324         if first { first = false; } else { word_space(s, ","); }
  325         op(s, elt);
  326     }
  327     end(s);
  328 }

  331 pub fn commasep_cmnt<T>(s: @ps, b: breaks, elts: &[T], op: &fn(@ps, &T),
  332                                get_span: &fn(&T) -> codemap::span) {
  333     box(s, 0u, b);
  334     let len = elts.len();
  335     let mut i = 0u;
  336     for elt in elts.iter() {
  337         maybe_print_comment(s, get_span(elt).hi);
  338         op(s, elt);
  339         i += 1u;

  341             word(s.s, ",");
  342             maybe_print_trailing_comment(s, get_span(elt),
  343                                          Some(get_span(&elts[i]).hi));

  345         }
  346     }
  347     end(s);
  348 }

  378 pub fn print_type(s: @ps, ty: &ast::Ty) {
  379     maybe_print_comment(s, ty.span.lo);

  381     match ty.node {
  382       ast::ty_nil => word(s.s, "()"),
  383       ast::ty_bot => word(s.s, "!"),
  384       ast::ty_box(ref mt) => { word(s.s, "@"); print_mt(s, mt); }
  385       ast::ty_uniq(ref mt) => { word(s.s, "~"); print_mt(s, mt); }
  386       ast::ty_vec(ref mt) => {
  387         word(s.s, "[");
  388         match mt.mutbl {
  389           ast::m_mutbl => word_space(s, "mut"),
  390           ast::m_const => word_space(s, "const"),
  391           ast::m_imm => ()
  392         }
  393         print_type(s, mt.ty);
  394         word(s.s, "]");
  395       }
  396       ast::ty_ptr(ref mt) => { word(s.s, "*"); print_mt(s, mt); }
  397       ast::ty_rptr(ref lifetime, ref mt) => {
  398           word(s.s, "&");
  399           print_opt_lifetime(s, lifetime);
  400           print_mt(s, mt);
  401       }
  402       ast::ty_tup(ref elts) => {
  403         popen(s);
  404         commasep(s, inconsistent, *elts, print_type);
  405         if elts.len() == 1 {
  406             word(s.s, ",");
  407         }
  408         pclose(s);
  409       }
  410       ast::ty_bare_fn(f) => {
  411           let generics = ast::Generics {
  412             lifetimes: f.lifetimes.clone(),
  413             ty_params: opt_vec::Empty
  414           };
  415           print_ty_fn(s, Some(f.abis), None, &None,

  427       }
  428       ast::ty_path(ref path, ref bounds, _) => print_bounded_path(s, path, bounds),

   31  // Each waiting task receives on one of these.
   32  type WaitEnd = Receiver<()>;
   33  type SignalEnd = Sender<()>;
   34  // A doubly-ended queue of waiting tasks.
   35  struct WaitQueue {
   36      head: Receiver<SignalEnd>,
   37      tail: Sender<SignalEnd>,
   38  }
   39  
   40  impl WaitQueue {
   41      fn new() -> WaitQueue {
   42          let (block_tail, block_head) = channel();
   43          WaitQueue { head: block_head, tail: block_tail }
   44      }
   45  
   46      // Signals one live task from the queue.
   47      fn signal(&self) -> bool {
   48          match self.head.try_recv() {
   49              comm::Data(ch) => {
   50                  // Send a wakeup signal. If the waiter was killed, its port will
   51                  // have closed. Keep trying until we get a live task.
   52                  if ch.try_send(()) {
   53                      true
   54                  } else {
   55                      self.signal()
   56                  }
   57              }
   58              _ => false
   59          }
   60      }
   61  
   62      fn broadcast(&self) -> uint {
   63          let mut count = 0;
   64          loop {
   65              match self.head.try_recv() {
   66                  comm::Data(ch) => {
   67                      if ch.try_send(()) {
   68                          count += 1;
   69                      }
   70                  }
   71                  _ => break
   72              }
   73          }
   74          count
   75      }
   76  
   77      fn wait_end(&self) -> WaitEnd {
   78          let (signal_end, wait_end) = channel();
   79          assert!(self.tail.try_send(signal_end));
   80          wait_end
   81      }
   82  }

   84  // The building-block used to make semaphores, mutexes, and rwlocks.
   85  struct Sem<Q> {
   86      lock: mutex::Mutex,
   87      // n.b, we need Sem to be `Share`, but the WaitQueue type is not send/share
   88      //      (for good reason). We have an internal invariant on this semaphore,
   89      //      however, that the queue is never accessed outside of a locked
   90      //      context. For this reason, we shove these behind a pointer which will
   91      //      be inferred to be `Share`.
   92      //
   93      // FIXME: this requires an extra allocation, which is bad.
   94      inner: *()
   95  }

   97  struct SemInner<Q> {
   98      count: int,
   99      waiters: WaitQueue,
  100      // Can be either unit or another waitqueue. Some sems shouldn't come with
  101      // a condition variable attached, others should.
  102      blocked: Q,
  103  }

  110  impl<Q: Send> Sem<Q> {
  111      fn new(count: int, q: Q) -> Sem<Q> {
  112          let inner = unsafe {
  113              cast::transmute(~SemInner {
  114                  waiters: WaitQueue::new(),
  115                  count: count,
  116                  blocked: q,
  117              })
  118          };
  119          Sem {
  120              lock: mutex::Mutex::new(),
  121              inner: inner,
  122          }
  123      }
  124  
  125      unsafe fn with(&self, f: |&mut SemInner<Q>|) {
  126          let _g = self.lock.lock();
  127          f(&mut *(self.inner as *mut SemInner<Q>))
  128      }

  130      pub fn acquire(&self) {
  131          unsafe {
  132              let mut waiter_nobe = None;
  133              self.with(|state| {
  134                  state.count -= 1;
  135                  if state.count < 0 {
  136                      // Create waiter nobe, enqueue ourself, and tell
  137                      // outer scope we need to block.
  138                      waiter_nobe = Some(state.waiters.wait_end());
  139                  }
  140              });
  141              // Uncomment if you wish to test for sem races. Not
  142              // valgrind-friendly.
  143              /* for _ in range(0, 1000) { task::deschedule(); } */
  144              // Need to wait outside the exclusive.
  145              if waiter_nobe.is_some() {
  146                  let _ = waiter_nobe.unwrap().recv();
  147              }
  148          }
  149      }
  150  
  151      pub fn release(&self) {
  152          unsafe {
  153              self.with(|state| {
  154                  state.count += 1;
  155                  if state.count <= 0 {
  156                      state.waiters.signal();
  157                  }
  158              })
  159          }
  160      }

  162      pub fn access<'a>(&'a self) -> SemGuard<'a, Q> {
  163          self.acquire();
  164          SemGuard { sem: self }
  165      }
  166  }

  208  /// A mechanism for atomic-unlock-and-deschedule blocking and signalling.
  209  pub struct Condvar<'a> {
  210      // The 'Sem' object associated with this condvar. This is the one that's
  211      // atomically-unlocked-and-descheduled upon and reacquired during wakeup.
  212      sem: &'a Sem<Vec<WaitQueue> >,
  213      // This is (can be) an extra semaphore which is held around the reacquire
  214      // operation on the first one. This is only used in cvars associated with
  215      // rwlocks, and is needed to ensure that, when a downgrader is trying to
  216      // hand off the access lock (which would be the first field, here), a 2nd
  217      // writer waking up from a cvar wait can't race with a reader to steal it,
  218      // See the comment in write_cond for more detail.
  219      order: ReacquireOrderLock<'a>,
  220      // Make sure condvars are non-copyable.
  221      nocopy: marker::NoCopy,
  222  }
  223  
  224  impl<'a> Condvar<'a> {
  225      /// Atomically drop the associated lock, and block until a signal is sent.
  226      ///
  227      /// # Failure
  228      ///
  229      /// A task which is killed while waiting on a condition variable will wake
  230      /// up, fail, and unlock the associated lock as it unwinds.
  231      pub fn wait(&self) { self.wait_on(0) }
  232  
  233      /// As wait(), but can specify which of multiple condition variables to
  234      /// wait on. Only a signal_on() or broadcast_on() with the same condvar_id
  235      /// will wake this thread.
  236      ///
  237      /// The associated lock must have been initialised with an appropriate
  238      /// number of condvars. The condvar_id must be between 0 and num_condvars-1
  239      /// or else this call will fail.
  240      ///
  241      /// wait() is equivalent to wait_on(0).
  242      pub fn wait_on(&self, condvar_id: uint) {
  243          let mut wait_end = None;
  244          let mut out_of_bounds = None;
  245          // Release lock, 'atomically' enqueuing ourselves in so doing.
  246          unsafe {
  247              self.sem.with(|state| {
  248                  if condvar_id < state.blocked.len() {
  249                      // Drop the lock.
  250                      state.count += 1;
  251                      if state.count <= 0 {
  252                          state.waiters.signal();
  253                      }
  254                      // Create waiter nobe, and enqueue ourself to
  255                      // be woken up by a signaller.
  256                      wait_end = Some(state.blocked.get(condvar_id).wait_end());
  257                  } else {
  258                      out_of_bounds = Some(state.blocked.len());
  259                  }
  260              })
  261          }
  262  
  263          // If deschedule checks start getting inserted anywhere, we can be
  264          // killed before or after enqueueing.
  265          check_cvar_bounds(out_of_bounds, condvar_id, "cond.wait_on()", || {
  266              // Unconditionally "block". (Might not actually block if a
  267              // signaller already sent -- I mean 'unconditionally' in contrast
  268              // with acquire().)
  269              (|| {
  270                  let _ = wait_end.take_unwrap().recv();
  271              }).finally(|| {
  272                  // Reacquire the condvar.
  273                  match self.order {
  274                      Just(lock) => {
  275                          let _g = lock.access();
  276                          self.sem.acquire();
  277                      }
  278                      Nothing => self.sem.acquire(),
  279                  }
  280              })
  281          })
  282      }
  283  
  284      /// Wake up a blocked task. Returns false if there was no blocked task.
  285      pub fn signal(&self) -> bool { self.signal_on(0) }
  286  
  287      /// As signal, but with a specified condvar_id. See wait_on.
  288      pub fn signal_on(&self, condvar_id: uint) -> bool {
  289          unsafe {
  290              let mut out_of_bounds = None;
  291              let mut result = false;
  292              self.sem.with(|state| {
  293                  if condvar_id < state.blocked.len() {
  294                      result = state.blocked.get(condvar_id).signal();
  295                  } else {
  296                      out_of_bounds = Some(state.blocked.len());
  297                  }
  298              });
  299              check_cvar_bounds(out_of_bounds,
  300                                condvar_id,
  301                                "cond.signal_on()",
  302                                || result)
  303          }
  304      }
  305  
  306      /// Wake up all blocked tasks. Returns the number of tasks woken.
  307      pub fn broadcast(&self) -> uint { self.broadcast_on(0) }
  308  
  309      /// As broadcast, but with a specified condvar_id. See wait_on.
  310      pub fn broadcast_on(&self, condvar_id: uint) -> uint {
  311          let mut out_of_bounds = None;
  312          let mut queue = None;
  313          unsafe {
  314              self.sem.with(|state| {
  315                  if condvar_id < state.blocked.len() {
  316                      // To avoid :broadcast_heavy, we make a new waitqueue,
  317                      // swap it out with the old one, and broadcast on the
  318                      // old one outside of the little-lock.
  319                      queue = Some(replace(state.blocked.get_mut(condvar_id),
  320                                                 WaitQueue::new()));
  321                  } else {
  322                      out_of_bounds = Some(state.blocked.len());
  323                  }
  324              });
  325              check_cvar_bounds(out_of_bounds,
  326                                condvar_id,
  327                                "cond.signal_on()",
  328                                || {
  329                  queue.take_unwrap().broadcast()
  330              })
  331          }
  332      }
  333  }

  335  // Checks whether a condvar ID was out of bounds, and fails if so, or does
  336  // something else next on success.
  337  #[inline]
  338  fn check_cvar_bounds<U>(
  339                       out_of_bounds: Option<uint>,
  340                       id: uint,
  341                       act: &str,
  342                       blk: || -> U)
  343                       -> U {
  344      match out_of_bounds {
  345          Some(0) =>
  346              fail!("{} with illegal ID {} - this lock has no condvars!", act, id),
  347          Some(length) =>
  348              fail!("{} with illegal ID {} - ID must be less than {}", act, id, length),
  349          None => blk()
  350      }
  351  }

  431      }

  433      /// Acquires ownership of this mutex, returning an RAII guard which will
  434      /// unlock the mutex when dropped. The associated condition variable can
  435      /// also be accessed through the returned guard.
  436      pub fn lock<'a>(&'a self) -> MutexGuard<'a> {
  437          let SemCondGuard { guard, cvar } = self.sem.access_cond();
  438          MutexGuard { guard: guard, cond: cvar }
  439      }
  440  }

  442  /****************************************************************************
  443   * Reader-writer locks
  444   ****************************************************************************/
  445  
  446  // NB: Wikipedia - Readers-writers_problem#The_third_readers-writers_problem
  447  
  448  /// A blocking, no-starvation, reader-writer lock with an associated condvar.
  449  ///
  450  /// # Failure
  451  ///
  452  /// A task which fails while holding an rwlock will unlock the rwlock as it
  453  /// unwinds.
  454  pub struct RWLock {
  455      order_lock:  Semaphore,
  456      access_lock: Sem<Vec<WaitQueue>>,
  457  
  458      // The only way the count flag is ever accessed is with xadd. Since it is
  459      // a read-modify-write operation, multiple xadds on different cores will
  460      // always be consistent with respect to each other, so a monotonic/relaxed
  461      // consistency ordering suffices (i.e., no extra barriers are needed).
  462      //
  463      // FIXME(#6598): The atomics module has no relaxed ordering flag, so I use
  464      // acquire/release orderings superfluously. Change these someday.
  465      read_count: atomics::AtomicUint,
  466  }
  467  
  468  /// An RAII helper which is created by acquiring a read lock on an RWLock. When
  469  /// dropped, this will unlock the RWLock.
  470  #[must_use]
  471  pub struct RWLockReadGuard<'a> {
  472      lock: &'a RWLock,
  473  }
  474  
  475  /// An RAII helper which is created by acquiring a write lock on an RWLock. When
  476  /// dropped, this will unlock the RWLock.
  477  ///
  478  /// A value of this type can also be consumed to downgrade to a read-only lock.
  479  #[must_use]
  480  pub struct RWLockWriteGuard<'a> {
  481      lock: &'a RWLock,
  482      /// Inner condition variable that is connected to the write-mode of the
  483      /// outer rwlock.
  484      pub cond: Condvar<'a>,
  485  }
  486  
  487  impl RWLock {
  488      /// Create a new rwlock, with one associated condvar.
  489      pub fn new() -> RWLock { RWLock::new_with_condvars(1) }
  490  
  491      /// Create a new rwlock, with a specified number of associated condvars.
  492      /// Similar to mutex_with_condvars.
  493      pub fn new_with_condvars(num_condvars: uint) -> RWLock {
  494          RWLock {
  495              order_lock: Semaphore::new(1),
  496              access_lock: Sem::new_and_signal(1, num_condvars),
  497              read_count: atomics::AtomicUint::new(0),
  498          }
  499      }
  500  
  501      /// Acquires a read-lock, returning an RAII guard that will unlock the lock
  502      /// when dropped. Calls to 'read' from other tasks may run concurrently with
  503      /// this one.
  504      pub fn read<'a>(&'a self) -> RWLockReadGuard<'a> {
  505          let _guard = self.order_lock.access();
  506          let old_count = self.read_count.fetch_add(1, atomics::Acquire);
  507          if old_count == 0 {
  508              self.access_lock.acquire();
  509          }
  510          RWLockReadGuard { lock: self }
  511      }
  512  
  513      /// Acquire a write-lock, returning an RAII guard that will unlock the lock
  514      /// when dropped. No calls to 'read' or 'write' from other tasks will run
  515      /// concurrently with this one.
  516      ///
  517      /// You can also downgrade a write to a read by calling the `downgrade`
  518      /// method on the returned guard. Additionally, the guard will contain a
  519      /// `Condvar` attached to this lock.
  520      ///
  521      /// # Example
  522      ///
  523      /// ```rust
  524      /// use sync::raw::RWLock;
  525      ///
  526      /// let lock = RWLock::new();
  527      /// let write = lock.write();
  528      /// // ... exclusive access ...
  529      /// let read = write.downgrade();
  530      /// // ... shared access ...
  531      /// drop(read);
  532      /// ```
  533      pub fn write<'a>(&'a self) -> RWLockWriteGuard<'a> {
  534          let _g = self.order_lock.access();
  535          self.access_lock.acquire();
  536  
  537          // It's important to thread our order lock into the condvar, so that
  538          // when a cond.wait() wakes up, it uses it while reacquiring the
  539          // access lock. If we permitted a waking-up writer to "cut in line",
  540          // there could arise a subtle race when a downgrader attempts to hand
  541          // off the reader cloud lock to a waiting reader. This race is tested
  542          // in arc.rs (test_rw_write_cond_downgrade_read_race) and looks like:
  543          // T1 (writer)              T2 (downgrader)             T3 (reader)
  544          // [in cond.wait()]
  545          //                          [locks for writing]
  546          //                          [holds access_lock]
  547          // [is signalled, perhaps by
  548          //  downgrader or a 4th thread]
  549          // tries to lock access(!)
  550          //                                                      lock order_lock
  551          //                                                      xadd read_count[0->1]
  552          //                                                      tries to lock access
  553          //                          [downgrade]
  554          //                          xadd read_count[1->2]
  555          //                          unlock access
  556          // Since T1 contended on the access lock before T3 did, it will steal
  557          // the lock handoff. Adding order_lock in the condvar reacquire path
  558          // solves this because T1 will hold order_lock while waiting on access,
  559          // which will cause T3 to have to wait until T1 finishes its write,
  560          // which can't happen until T2 finishes the downgrade-read entirely.
  561          // The astute reader will also note that making waking writers use the
  562          // order_lock is better for not starving readers.
  563          RWLockWriteGuard {
  564              lock: self,
  565              cond: Condvar {
  566                  sem: &self.access_lock,
  567                  order: Just(&self.order_lock),
  568                  nocopy: marker::NoCopy,
  569              }
  570          }
  571      }
  572  }
  573  
  574  impl<'a> RWLockWriteGuard<'a> {
  575      /// Consumes this write lock and converts it into a read lock.
  576      pub fn downgrade(self) -> RWLockReadGuard<'a> {
  577          let lock = self.lock;
  578          // Don't run the destructor of the write guard, we're in charge of
  579          // things from now on
  580          unsafe { cast::forget(self) }
  581  
  582          let old_count = lock.read_count.fetch_add(1, atomics::Release);
  583          // If another reader was already blocking, we need to hand-off
  584          // the "reader cloud" access lock to them.
  585          if old_count != 0 {
  586              // Guaranteed not to let another writer in, because
  587              // another reader was holding the order_lock. Hence they
  588              // must be the one to get the access_lock (because all
  589              // access_locks are acquired with order_lock held). See
  590              // the comment in write_cond for more justification.
  591              lock.access_lock.release();
  592          }
  593          RWLockReadGuard { lock: lock }
  594      }
  595  }
  596  
  597  #[unsafe_destructor]
  598  impl<'a> Drop for RWLockWriteGuard<'a> {
  599      fn drop(&mut self) {
  600          self.lock.access_lock.release();
  601      }
  602  }
  603  
  604  #[unsafe_destructor]
  605  impl<'a> Drop for RWLockReadGuard<'a> {
  606      fn drop(&mut self) {
  607          let old_count = self.lock.read_count.fetch_sub(1, atomics::Release);
  608          assert!(old_count > 0);
  609          if old_count == 1 {
  610              // Note: this release used to be outside of a locked access
  611              // to exclusive-protected state. If this code is ever
  612              // converted back to such (instead of using atomic ops),
  613              // this access MUST NOT go inside the exclusive access.
  614              self.lock.access_lock.release();
  615          }
  616      }
  617  }

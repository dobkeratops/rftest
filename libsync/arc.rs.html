  21  /// An atomically reference counted wrapper for shared state.
  22  ///
  23  /// # Example
  24  ///
  25  /// In this example, a large vector of floats is shared between several tasks.
  26  /// With simple pipes, without `Arc`, a copy would have to be made for each
  27  /// task.
  28  ///

  30  /// use sync::Arc;
  31  ///

  69      #[inline]
  70      pub fn new(data: T) -> Arc<T> {

  72          // held by all the strong pointers (kinda), see std/rc.rs for more info
  73          let x = ~ArcInner {
  74              strong: atomics::AtomicUint::new(1),
  75              weak: atomics::AtomicUint::new(1),
  76              data: data,
  77          };
  78          Arc { x: unsafe { cast::transmute(x) } }
  79      }

  85          // `ArcInner` structure itself is `Share` because the inner data is
  86          // `Share` as well, so we're ok loaning out an immutable pointer to

 136  impl<T: Send + Share + Clone> Arc<T> {
 137      /// Acquires a mutable pointer to the inner contents by guaranteeing that
 138      /// the reference count is one (no sharing is possible).
 139      ///
 140      /// This is also referred to as a copy-on-write operation because the inner
 141      /// data is cloned if the reference count is greater than one.
 142      #[inline]
 143      #[experimental]
 144      pub fn make_unique<'a>(&'a mut self) -> &'a mut T {
 145          if self.inner().strong.load(atomics::SeqCst) != 1 {
 146              *self = Arc::new(self.deref().clone())
 147          }
 148          // This unsafety is ok because we're guaranteed that the pointer
 149          // returned is the *only* pointer that will ever be returned to T. Our
 150          // reference count is guaranteed to be 1 at this point, and we required
 151          // the Arc itself to be `mut`, so we're returning the only possible
 152          // reference to the inner data.
 153          unsafe { cast::transmute_mut(self.deref()) }
 154      }
 155  }

 157  #[unsafe_destructor]
 158  impl<T: Share + Send> Drop for Arc<T> {
 159      fn drop(&mut self) {
 160          // This structure has #[unsafe_no_drop_flag], so this drop glue may run
 161          // more than once (but it is guaranteed to be zeroed after the first if
 162          // it's run more than once)
 163          if self.x.is_null() { return }
 164  
 165          // Because `fetch_sub` is already atomic, we do not need to synchronize
 166          // with other threads unless we are going to delete the object. This
 167          // same logic applies to the below `fetch_sub` to the `weak` count.
 168          if self.inner().strong.fetch_sub(1, atomics::Release) != 0 { return }
 169  
 170          // This fence is needed to prevent reordering of use of the data and
 171          // deletion of the data. Because it is marked `Release`, the
 172          // decreasing of the reference count sychronizes with this `Acquire`
 173          // fence. This means that use of the data happens before decreasing
 174          // the refernce count, which happens before this fence, which
 175          // happens before the deletion of the data.
 176          //
 177          // As explained in the [Boost documentation][1],
 178          //
 179          // It is important to enforce any possible access to the object in
 180          // one thread (through an existing reference) to *happen before*
 181          // deleting the object in a different thread. This is achieved by a
 182          // "release" operation after dropping a reference (any access to the
 183          // object through this reference must obviously happened before),
 184          // and an "acquire" operation before deleting the object.
 185          //
 186          // [1]: (www.boost.org/doc/libs/1_55_0/doc/html/atomic/usage_examples.html)
 187          atomics::fence(atomics::Acquire);
 188  
 189          // Destroy the data at this time, even though we may not free the box
 190          // allocation itself (there may still be weak pointers lying around).
 191          unsafe { drop(ptr::read(&self.inner().data)); }
 192  
 193          if self.inner().weak.fetch_sub(1, atomics::Release) == 0 {
 194              atomics::fence(atomics::Acquire);
 195              unsafe { global_heap::exchange_free(self.x as *u8) }
 196          }
 197      }
 198  }

 200  impl<T: Share + Send> Weak<T> {
 201      /// Attempts to upgrade this weak reference to a strong reference.
 202      ///
 203      /// This method will fail to upgrade this reference if the strong reference
 204      /// count has already reached 0, but if there are still other active strong
 205      /// references this function will return a new strong reference to the data
 206      pub fn upgrade(&self) -> Option<Arc<T>> {
 207          // We use a CAS loop to increment the strong count instead of a
 208          // fetch_add because once the count hits 0 is must never be above 0.
 209          let inner = self.inner();
 210          loop {
 211              let n = inner.strong.load(atomics::SeqCst);
 212              if n == 0 { return None }
 213              let old = inner.strong.compare_and_swap(n, n + 1, atomics::SeqCst);
 214              if old == n { return Some(Arc { x: self.x }) }
 215          }
 216      }

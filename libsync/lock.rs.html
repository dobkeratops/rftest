  11  //! Wrappers for safe, shared, mutable memory between tasks
  12  //!
  13  //! The wrappers in this module build on the primitives from `sync::raw` to
  14  //! provide safe interfaces around using the primitive locks. These primitives
  15  //! implement a technique called "poisoning" where when a task failed with a
  16  //! held lock, all future attempts to use the lock will fail.
  17  //!
  18  //! For example, if two tasks are contending on a mutex and one of them fails
  19  //! after grabbing the lock, the second task will immediately fail because the
  20  //! lock is now poisoned.
  21  
  22  use std::task;
  23  use std::ty::Unsafe;
  24  
  25  use raw;
  26  
  27  /****************************************************************************
  28   * Poisoning helpers
  29   ****************************************************************************/
  30  
  31  struct PoisonOnFail<'a> {
  32      flag: &'a mut bool,
  33      failed: bool,
  34  }
  35  
  36  impl<'a> PoisonOnFail<'a> {
  37      fn check(flag: bool, name: &str) {
  38          if flag {
  39              fail!("Poisoned {} - another task failed inside!", name);
  40          }
  41      }
  42  
  43      fn new<'a>(flag: &'a mut bool, name: &str) -> PoisonOnFail<'a> {
  44          PoisonOnFail::check(*flag, name);
  45          PoisonOnFail {
  46              flag: flag,
  47              failed: task::failing()
  48          }
  49      }
  50  }
  51  
  52  #[unsafe_destructor]
  53  impl<'a> Drop for PoisonOnFail<'a> {
  54      fn drop(&mut self) {
  55          if !self.failed && task::failing() {
  56              *self.flag = true;
  57          }
  58      }
  59  }

  65  enum Inner<'a> {
  66      InnerMutex(raw::MutexGuard<'a>),
  67      InnerRWLock(raw::RWLockWriteGuard<'a>),
  68  }
  69  
  70  impl<'b> Inner<'b> {
  71      fn cond<'a>(&'a self) -> &'a raw::Condvar<'b> {
  72          match *self {
  73              InnerMutex(ref m) => &m.cond,
  74              InnerRWLock(ref m) => &m.cond,
  75          }
  76      }
  77  }

  79  /// A condition variable, a mechanism for unlock-and-descheduling and
  80  /// signaling, for use with the lock types.
  81  pub struct Condvar<'a> {
  82      priv name: &'static str,
  83      // n.b. Inner must be after PoisonOnFail because we must set the poison flag
  84      //      *inside* the mutex, and struct fields are destroyed top-to-bottom
  85      //      (destroy the lock guard last).
  86      priv poison: PoisonOnFail<'a>,
  87      priv inner: Inner<'a>,
  88  }
  89  
  90  impl<'a> Condvar<'a> {
  91      /// Atomically exit the associated lock and block until a signal is sent.
  92      ///
  93      /// wait() is equivalent to wait_on(0).
  94      ///
  95      /// # Failure
  96      ///
  97      /// A task which is killed while waiting on a condition variable will wake
  98      /// up, fail, and unlock the associated lock as it unwinds.
  99      #[inline]
 100      pub fn wait(&self) { self.wait_on(0) }
 101  
 102      /// Atomically exit the associated lock and block on a specified condvar
 103      /// until a signal is sent on that same condvar.
 104      ///
 105      /// The associated lock must have been initialised with an appropriate
 106      /// number of condvars. The condvar_id must be between 0 and num_condvars-1
 107      /// or else this call will fail.
 108      #[inline]
 109      pub fn wait_on(&self, condvar_id: uint) {
 110          assert!(!*self.poison.flag);
 111          self.inner.cond().wait_on(condvar_id);
 112          // This is why we need to wrap sync::condvar.
 113          PoisonOnFail::check(*self.poison.flag, self.name);
 114      }
 115  
 116      /// Wake up a blocked task. Returns false if there was no blocked task.
 117      #[inline]
 118      pub fn signal(&self) -> bool { self.signal_on(0) }
 119  
 120      /// Wake up a blocked task on a specified condvar (as
 121      /// sync::cond.signal_on). Returns false if there was no blocked task.
 122      #[inline]
 123      pub fn signal_on(&self, condvar_id: uint) -> bool {
 124          assert!(!*self.poison.flag);
 125          self.inner.cond().signal_on(condvar_id)
 126      }
 127  
 128      /// Wake up all blocked tasks. Returns the number of tasks woken.
 129      #[inline]
 130      pub fn broadcast(&self) -> uint { self.broadcast_on(0) }
 131  
 132      /// Wake up all blocked tasks on a specified condvar (as
 133      /// sync::cond.broadcast_on). Returns the number of tasks woken.
 134      #[inline]
 135      pub fn broadcast_on(&self, condvar_id: uint) -> uint {
 136          assert!(!*self.poison.flag);
 137          self.inner.cond().broadcast_on(condvar_id)
 138      }
 139  }

 145  /// A wrapper type which provides synchronized access to the underlying data, of
 146  /// type `T`. A mutex always provides exclusive access, and concurrent requests
 147  /// will block while the mutex is already locked.
 148  ///
 149  /// # Example
 150  ///
 151  /// ```
 152  /// use sync::{Mutex, Arc};
 153  ///
 154  /// let mutex = Arc::new(Mutex::new(1));
 155  /// let mutex2 = mutex.clone();
 156  ///
 157  /// spawn(proc() {
 158  ///     let mut val = mutex2.lock();
 159  ///     *val += 1;
 160  ///     val.cond.signal();
 161  /// });
 162  ///
 163  /// let mut value = mutex.lock();
 164  /// while *value != 2 {
 165  ///     value.cond.wait();
 166  /// }
 167  /// ```
 168  pub struct Mutex<T> {
 169      priv lock: raw::Mutex,
 170      priv failed: Unsafe<bool>,
 171      priv data: Unsafe<T>,
 172  }
 173  
 174  /// An guard which is created by locking a mutex. Through this guard the
 175  /// underlying data can be accessed.
 176  pub struct MutexGuard<'a, T> {
 177      priv data: &'a mut T,
 178      /// Inner condition variable connected to the locked mutex that this guard
 179      /// was created from. This can be used for atomic-unlock-and-deschedule.
 180      cond: Condvar<'a>,
 181  }
 182  
 183  impl<T: Send> Mutex<T> {
 184      /// Creates a new mutex to protect the user-supplied data.
 185      pub fn new(user_data: T) -> Mutex<T> {
 186          Mutex::new_with_condvars(user_data, 1)
 187      }
 188  
 189      /// Create a new mutex, with a specified number of associated condvars.
 190      ///
 191      /// This will allow calling wait_on/signal_on/broadcast_on with condvar IDs
 192      /// between 0 and num_condvars-1. (If num_condvars is 0, lock_cond will be
 193      /// allowed but any operations on the condvar will fail.)
 194      pub fn new_with_condvars(user_data: T, num_condvars: uint) -> Mutex<T> {
 195          Mutex {
 196              lock: raw::Mutex::new_with_condvars(num_condvars),
 197              failed: Unsafe::new(false),
 198              data: Unsafe::new(user_data),
 199          }
 200      }
 201  
 202      /// Access the underlying mutable data with mutual exclusion from other
 203      /// tasks. The returned value is an RAII guard which will unlock the mutex
 204      /// when dropped. All concurrent tasks attempting to lock the mutex will
 205      /// block while the returned value is still alive.
 206      ///
 207      /// # Failure
 208      ///
 209      /// Failing while inside the Mutex will unlock the Mutex while unwinding, so
 210      /// that other tasks won't block forever. It will also poison the Mutex:
 211      /// any tasks that subsequently try to access it (including those already
 212      /// blocked on the mutex) will also fail immediately.
 213      #[inline]
 214      pub fn lock<'a>(&'a self) -> MutexGuard<'a, T> {
 215          let guard = self.lock.lock();
 216  
 217          // These two accesses are safe because we're guranteed at this point
 218          // that we have exclusive access to this mutex. We are indeed able to
 219          // promote ourselves from &Mutex to `&mut T`
 220          let poison = unsafe { &mut *self.failed.get() };
 221          let data = unsafe { &mut *self.data.get() };
 222  
 223          MutexGuard {
 224              data: data,
 225              cond: Condvar {
 226                  name: "Mutex",
 227                  poison: PoisonOnFail::new(poison, "Mutex"),
 228                  inner: InnerMutex(guard),
 229              },
 230          }
 231      }
 232  }
 233  
 234  // FIXME(#13042): these should both have T: Send
 235  impl<'a, T> Deref<T> for MutexGuard<'a, T> {
 236      fn deref<'a>(&'a self) -> &'a T { &*self.data }
 237  }
 238  impl<'a, T> DerefMut<T> for MutexGuard<'a, T> {
 239      fn deref_mut<'a>(&'a mut self) -> &'a mut T { &mut *self.data }
 240  }

 246  /// A dual-mode reader-writer lock. The data can be accessed mutably or
 247  /// immutably, and immutably-accessing tasks may run concurrently.
 248  ///
 249  /// # Example
 250  ///
 251  /// ```
 252  /// use sync::{RWLock, Arc};
 253  ///
 254  /// let lock1 = Arc::new(RWLock::new(1));
 255  /// let lock2 = lock1.clone();
 256  ///
 257  /// spawn(proc() {
 258  ///     let mut val = lock2.write();
 259  ///     *val = 3;
 260  ///     let val = val.downgrade();
 261  ///     println!("{}", *val);
 262  /// });
 263  ///
 264  /// let val = lock1.read();
 265  /// println!("{}", *val);
 266  /// ```
 267  pub struct RWLock<T> {
 268      priv lock: raw::RWLock,
 269      priv failed: Unsafe<bool>,
 270      priv data: Unsafe<T>,
 271  }
 272  
 273  /// A guard which is created by locking an rwlock in write mode. Through this
 274  /// guard the underlying data can be accessed.
 275  pub struct RWLockWriteGuard<'a, T> {
 276      priv data: &'a mut T,
 277      /// Inner condition variable that can be used to sleep on the write mode of
 278      /// this rwlock.
 279      cond: Condvar<'a>,
 280  }
 281  
 282  /// A guard which is created by locking an rwlock in read mode. Through this
 283  /// guard the underlying data can be accessed.
 284  pub struct RWLockReadGuard<'a, T> {
 285      priv data: &'a T,
 286      priv guard: raw::RWLockReadGuard<'a>,
 287  }
 288  
 289  impl<T: Send + Share> RWLock<T> {
 290      /// Create a reader/writer lock with the supplied data.
 291      pub fn new(user_data: T) -> RWLock<T> {
 292          RWLock::new_with_condvars(user_data, 1)
 293      }
 294  
 295      /// Create a reader/writer lock with the supplied data and a specified number
 296      /// of condvars (as sync::RWLock::new_with_condvars).
 297      pub fn new_with_condvars(user_data: T, num_condvars: uint) -> RWLock<T> {
 298          RWLock {
 299              lock: raw::RWLock::new_with_condvars(num_condvars),
 300              failed: Unsafe::new(false),
 301              data: Unsafe::new(user_data),
 302          }
 303      }
 304  
 305      /// Access the underlying data mutably. Locks the rwlock in write mode;
 306      /// other readers and writers will block.
 307      ///
 308      /// # Failure
 309      ///
 310      /// Failing while inside the lock will unlock the lock while unwinding, so
 311      /// that other tasks won't block forever. As Mutex.lock, it will also poison
 312      /// the lock, so subsequent readers and writers will both also fail.
 313      #[inline]
 314      pub fn write<'a>(&'a self) -> RWLockWriteGuard<'a, T> {
 315          let guard = self.lock.write();
 316  
 317          // These two accesses are safe because we're guranteed at this point
 318          // that we have exclusive access to this rwlock. We are indeed able to
 319          // promote ourselves from &RWLock to `&mut T`
 320          let poison = unsafe { &mut *self.failed.get() };
 321          let data = unsafe { &mut *self.data.get() };
 322  
 323          RWLockWriteGuard {
 324              data: data,
 325              cond: Condvar {
 326                  name: "RWLock",
 327                  poison: PoisonOnFail::new(poison, "RWLock"),
 328                  inner: InnerRWLock(guard),
 329              },
 330          }
 331      }
 332  
 333      /// Access the underlying data immutably. May run concurrently with other
 334      /// reading tasks.
 335      ///
 336      /// # Failure
 337      ///
 338      /// Failing will unlock the lock while unwinding. However, unlike all other
 339      /// access modes, this will not poison the lock.
 340      pub fn read<'a>(&'a self) -> RWLockReadGuard<'a, T> {
 341          let guard = self.lock.read();
 342          PoisonOnFail::check(unsafe { *self.failed.get() }, "RWLock");
 343          RWLockReadGuard {
 344              guard: guard,
 345              data: unsafe { &*self.data.get() },
 346          }
 347      }
 348  }
 349  
 350  impl<'a, T: Send + Share> RWLockWriteGuard<'a, T> {
 351      /// Consumes this write lock token, returning a new read lock token.
 352      ///
 353      /// This will allow pending readers to come into the lock.
 354      pub fn downgrade(self) -> RWLockReadGuard<'a, T> {
 355          let RWLockWriteGuard { data, cond } = self;
 356          // convert the data to read-only explicitly
 357          let data = &*data;
 358          let guard = match cond.inner {
 359              InnerMutex(..) => unreachable!(),
 360              InnerRWLock(guard) => guard.downgrade()
 361          };
 362          RWLockReadGuard { guard: guard, data: data }
 363      }
 364  }
 365  
 366  // FIXME(#13042): these should all have T: Send + Share
 367  impl<'a, T> Deref<T> for RWLockReadGuard<'a, T> {
 368      fn deref<'a>(&'a self) -> &'a T { self.data }
 369  }
 370  impl<'a, T> Deref<T> for RWLockWriteGuard<'a, T> {
 371      fn deref<'a>(&'a self) -> &'a T { &*self.data }
 372  }
 373  impl<'a, T> DerefMut<T> for RWLockWriteGuard<'a, T> {
 374      fn deref_mut<'a>(&'a mut self) -> &'a mut T { &mut *self.data }
 375  }

 381  /// A barrier enables multiple tasks to synchronize the beginning
 382  /// of some computation.
 383  ///
 384  /// ```rust
 385  /// use sync::{Arc, Barrier};
 386  ///
 387  /// let barrier = Arc::new(Barrier::new(10));
 388  /// for _ in range(0, 10) {
 389  ///     let c = barrier.clone();
 390  ///     // The same messages will be printed together.
 391  ///     // You will NOT see any interleaving.
 392  ///     spawn(proc() {
 393  ///         println!("before wait");
 394  ///         c.wait();
 395  ///         println!("after wait");
 396  ///     });
 397  /// }
 398  /// ```
 399  pub struct Barrier {
 400      priv lock: Mutex<BarrierState>,
 401      priv num_tasks: uint,
 402  }
 403  
 404  // The inner state of a double barrier
 405  struct BarrierState {
 406      count: uint,
 407      generation_id: uint,
 408  }
 409  
 410  impl Barrier {
 411      /// Create a new barrier that can block a given number of tasks.
 412      pub fn new(num_tasks: uint) -> Barrier {
 413          Barrier {
 414              lock: Mutex::new(BarrierState {
 415                  count: 0,
 416                  generation_id: 0,
 417              }),
 418              num_tasks: num_tasks,
 419          }
 420      }
 421  
 422      /// Block the current task until a certain number of tasks is waiting.
 423      pub fn wait(&self) {
 424          let mut lock = self.lock.lock();
 425          let local_gen = lock.generation_id;
 426          lock.count += 1;
 427          if lock.count < self.num_tasks {
 428              // We need a while loop to guard against spurious wakeups.
 429              // http://en.wikipedia.org/wiki/Spurious_wakeup
 430              while local_gen == lock.generation_id &&
 431                    lock.count < self.num_tasks {
 432                  lock.cond.wait();
 433              }
 434          } else {
 435              lock.count = 0;
 436              lock.generation_id += 1;
 437              lock.cond.broadcast();
 438          }
 439      }
 440  }

 659      #[test]
 660      fn test_rw_downgrade() {
 661          // (1) A downgrader gets in write mode and does cond.wait.
 662          // (2) A writer gets in write mode, sets state to 42, and does signal.
 663          // (3) Downgrader wakes, sets state to 31337.
 664          // (4) tells writer and all other readers to contend as it downgrades.
 665          // (5) Writer attempts to set state back to 42, while downgraded task
 666          //     and all reader tasks assert that it's 31337.
 667          let arc = Arc::new(RWLock::new(0));
 668  
 669          // Reader tasks
 670          let mut reader_convos = Vec::new();
 671          for _ in range(0, 10) {
 672              let ((tx1, rx1), (tx2, rx2)) = (channel(), channel());
 673              reader_convos.push((tx1, rx2));
 674              let arcn = arc.clone();
 675              task::spawn(proc() {
 676                  rx1.recv(); // wait for downgrader to give go-ahead
 677                  let lock = arcn.read();
 678                  assert_eq!(*lock, 31337);
 679                  tx2.send(());
 680              });
 681          }
 682  
 683          // Writer task
 684          let arc2 = arc.clone();
 685          let ((tx1, rx1), (tx2, rx2)) = (channel(), channel());
 686          task::spawn(proc() {
 687              rx1.recv();
 688              {
 689                  let mut lock = arc2.write();
 690                  assert_eq!(*lock, 0);
 691                  *lock = 42;
 692                  lock.cond.signal();
 693              }
 694              rx1.recv();
 695              {
 696                  let mut lock = arc2.write();
 697                  // This shouldn't happen until after the downgrade read
 698                  // section, and all other readers, finish.
 699                  assert_eq!(*lock, 31337);
 700                  *lock = 42;
 701              }
 702              tx2.send(());
 703          });
 704  
 705          // Downgrader (us)
 706          let mut lock = arc.write();
 707          tx1.send(()); // send to another writer who will wake us up
 708          while *lock == 0 {
 709              lock.cond.wait();
 710          }
 711          assert_eq!(*lock, 42);
 712          *lock = 31337;
 713          // send to other readers
 714          for &(ref mut rc, _) in reader_convos.mut_iter() {
 715              rc.send(())
 716          }
 717          let lock = lock.downgrade();
 718          // complete handshake with other readers
 719          for &(_, ref mut rp) in reader_convos.mut_iter() {
 720              rp.recv()
 721          }
 722          tx1.send(()); // tell writer to try again
 723          assert_eq!(*lock, 31337);
 724          drop(lock);
 725  
 726          rx2.recv(); // complete handshake with writer
 727      }

 731          // Tests that when a downgrader hands off the "reader cloud" lock
 732          // because of a contending reader, a writer can't race to get it
 733          // instead, which would result in readers_and_writers. This tests
 734          // the raw module rather than this one, but it's here because an

 746              // must involve order_lock, so that it cannot race with a reader

 754          assert!(*lock);
 755          // make writer contend in the cond-reacquire path
 756          lock.cond.signal();

 759          let (tx, rx) = channel();

 769          for _ in range(0, 5) { task::deschedule(); }

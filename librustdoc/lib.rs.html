 252  /// Interprets the input file as a rust source file, passing it through the
 253  /// compiler all the way through the analysis passes. The rustdoc output is then
 254  /// generated from the cleaned AST of the crate.
 255  ///
 256  /// This form of input will run all of the plug/cleaning passes
 257  fn rust_input(cratefile: &str, matches: &getopts::Matches) -> Output {
 258      let mut default_passes = !matches.opt_present("no-defaults");
 259      let mut passes = matches.opt_strs("passes");
 260      let mut plugins = matches.opt_strs("plugins");
 261  
 262      // First, parse the crate and extract all relevant information.
 263      let libs: Vec<Path> = matches.opt_strs("L")
 264                                   .iter()
 265                                   .map(|s| Path::new(s.as_slice()))
 266                                   .collect();
 267      let cfgs = matches.opt_strs("cfg");
 268      let cr = Path::new(cratefile);
 269      info!("starting to run rustc");
 270      let (krate, analysis) = std::task::try(proc() {
 271          let cr = cr;
 272          core::run_core(libs.move_iter().collect(),
 273                         cfgs.move_iter().collect(),
 274                         &cr)
 275      }).unwrap();
 276      info!("finished with rustc");
 277      local_data::set(analysiskey, analysis);
 278  
 279      // Process all of the crate attributes, extracting plugin metadata along
 280      // with the passes which we are supposed to run.
 281      match krate.module.get_ref().doc_list() {
 282          Some(nested) => {
 283              for inner in nested.iter() {
 284                  match *inner {
 285                      clean::Word(ref x) if "no_default_passes" == *x => {
 286                          default_passes = false;
 287                      }
 288                      clean::NameValue(ref x, ref value) if "passes" == *x => {
 289                          for pass in value.words() {
 290                              passes.push(pass.to_owned());
 291                          }
 292                      }
 293                      clean::NameValue(ref x, ref value) if "plugins" == *x => {
 294                          for p in value.words() {
 295                              plugins.push(p.to_owned());
 296                          }
 297                      }
 298                      _ => {}
 299                  }
 300              }
 301          }
 302          None => {}
 303      }
 304      if default_passes {
 305          for name in DEFAULT_PASSES.rev_iter() {
 306              passes.unshift(name.to_owned());
 307          }
 308      }
 309  
 310      // Load all plugins/passes into a PluginManager
 311      let path = matches.opt_str("plugin-path").unwrap_or(~"/tmp/rustdoc/plugins");
 312      let mut pm = plugins::PluginManager::new(Path::new(path));
 313      for pass in passes.iter() {
 314          let plugin = match PASSES.iter().position(|&(p, _, _)| p == *pass) {
 315              Some(i) => PASSES[i].val1(),
 316              None => {
 317                  error!("unknown pass {}, skipping", *pass);
 318                  continue
 319              },
 320          };
 321          pm.add_plugin(plugin);
 322      }
 323      info!("loading plugins...");
 324      for pname in plugins.move_iter() {
 325          pm.load_plugin(pname);
 326      }
 327  
 328      // Run everything!
 329      info!("Executing passes/plugins");
 330      return pm.run_plugins(krate);
 331  }

 333  /// This input format purely deserializes the json output file. No passes are
 334  /// run over the deserialized output.
 335  fn json_input(input: &str) -> Result<Output, ~str> {
 336      let mut input = match File::open(&Path::new(input)) {
 337          Ok(f) => f,
 338          Err(e) => return Err(format!("couldn't open {}: {}", input, e)),
 339      };
 340      match json::from_reader(&mut input) {
 341          Err(s) => Err(s.to_str()),
 342          Ok(json::Object(obj)) => {
 343              let mut obj = obj;
 344              // Make sure the schema is what we expect
 345              match obj.pop(&~"schema") {
 346                  Some(json::String(version)) => {
 347                      if version.as_slice() != SCHEMA_VERSION {
 348                          return Err(format!("sorry, but I only understand \
 349                                              version {}", SCHEMA_VERSION))
 350                      }
 351                  }
 352                  Some(..) => return Err(~"malformed json"),
 353                  None => return Err(~"expected a schema version"),
 354              }
 355              let krate = match obj.pop(&~"crate") {
 356                  Some(json) => {
 357                      let mut d = json::Decoder::new(json);
 358                      Decodable::decode(&mut d).unwrap()
 359                  }
 360                  None => return Err(~"malformed json"),
 361              };
 362              // FIXME: this should read from the "plugins" field, but currently
 363              //      Json doesn't implement decodable...
 364              let plugin_output = Vec::new();

 367          Ok(..) => Err(~"malformed json input: expected an object at the top"),
 368      }
 369  }

  47  /// Progressive construction of a table of contents.

  50      priv top_level: Toc,
  51      /// The current heirachy of parent headings, the levels are
  52      /// strictly increasing (i.e. chain[0].level < chain[1].level <
  53      /// ...) with each entry being the most recent occurance of a
  54      /// heading with that level (it doesn't include the most recent
  55      /// occurences of every level, just, if *is* in `chain` then is is
  56      /// the most recent one).
  57      ///
  58      /// We also have `chain[0].level <= top_level.entries[last]`.
  59      priv chain: Vec<TocEntry>
  60  }

  68      /// Convert into a true `Toc` struct.
  69      pub fn into_toc(mut self) -> Toc {
  70          // we know all levels are >= 1.
  71          self.fold_until(0);
  72          self.top_level
  73      }
  74  
  75      /// Collapse the chain until the first heading more important than
  76      /// `level` (i.e. lower level)
  77      ///
  78      /// Example:
  79      ///
  80      /// ## A
  81      /// # B
  82      /// # C
  83      /// ## D
  84      /// ## E
  85      /// ### F
  86      /// #### G
  87      /// ### H
  88      ///
  89      /// If we are considering H (i.e. level 3), then A and B are in
  90      /// self.top_level, D is in C.children, and C, E, F, G are in
  91      /// self.chain.
  92      ///
  93      /// When we attempt to push H, we realise that first G is not the
  94      /// parent (level is too high) so it is popped from chain and put
  95      /// into F.children, then F isn't the parent (level is equal, aka
  96      /// sibling), so it's also popped and put into E.children.
  97      ///
  98      /// This leaves us looking at E, which does have a smaller level,
  99      /// and, by construction, it's the most recent thing with smaller
 100      /// level, i.e. it's the immediate parent of H.
 101      fn fold_until(&mut self, level: u32) {
 102          let mut this = None;
 103          loop {
 104              match self.chain.pop() {
 105                  Some(mut next) => {
 106                      this.map(|e| next.children.entries.push(e));
 107                      if next.level < level {
 108                          // this is the parent we want, so return it to
 109                          // its rightful place.
 110                          self.chain.push(next);
 111                          return
 112                      } else {
 113                          this = Some(next);
 114                      }
 115                  }
 116                  None => {
 117                      this.map(|e| self.top_level.entries.push(e));
 118                      return
 119                  }
 120              }
 121          }
 122      }
 123  
 124      /// Push a level `level` heading into the appropriate place in the
 125      /// heirarchy, returning a string containing the section number in
 126      /// `<num>.<num>.<num>` format.
 127      pub fn push<'a>(&'a mut self, level: u32, name: ~str, id: ~str) -> &'a str {
 128          assert!(level >= 1);
 129  
 130          // collapse all previous sections into their parents until we
 131          // get to relevant heading (i.e. the first one with a smaller
 132          // level than us)
 133          self.fold_until(level);
 134  
 135          let mut sec_number;
 136          {
 137              let (toc_level, toc) = match self.chain.last() {
 138                  None => {
 139                      sec_number = ~"";
 140                      (0, &self.top_level)
 141                  }
 142                  Some(entry) => {
 143                      sec_number = entry.sec_number.clone();
 144                      sec_number.push_str(".");
 145                      (entry.level, &entry.children)
 146                  }
 147              };
 148              // fill in any missing zeros, e.g. for
 149              // # Foo (1)
 150              // ### Bar (1.0.1)
 151              for _ in range(toc_level, level - 1) {
 152                  sec_number.push_str("0.");
 153              }
 154              let number = toc.count_entries_with_level(level);
 155              sec_number.push_str(format!("{}", number + 1))
 156          }
 157  
 158          self.chain.push(TocEntry {
 159                  level: level,
 160                  name: name,
 161                  sec_number: sec_number,
 162                  id: id,
 163                  children: Toc { entries: Vec::new() }
 164              });
 165  
 166          // get the thing we just pushed, so we can borrow the string
 167          // out of it with the right lifetime
 168          let just_inserted = self.chain.mut_last().unwrap();
 169          just_inserted.sec_number.as_slice()
 170      }
 171  }

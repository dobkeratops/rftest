 104      // Shuffle around a few input and output handles here. We're going to pass
 105      // an explicit handle into rustc to collect output messages, but we also
 106      // want to catch the error message that rustc prints when it fails.
 107      //
 108      // We take our task-local stderr (likely set by the test runner), and move
 109      // it into another task. This helper task then acts as a sink for both the
 110      // stderr of this task and stderr of rustc itself, copying all the info onto
 111      // the stderr channel we originally started with.
 112      //
 113      // The basic idea is to not use a default_handler() for rustc, and then also
 114      // not print things by default to the actual stderr.
 115      let (tx, rx) = channel();
 116      let w1 = io::ChanWriter::new(tx);
 117      let w2 = w1.clone();
 118      let old = io::stdio::set_stderr(~w1);
 119      spawn(proc() {
 120          let mut p = io::ChanReader::new(rx);
 121          let mut err = old.unwrap_or(~io::stderr() as ~Writer);
 122          io::util::copy(&mut p, &mut err).unwrap();
 123      });
 124      let emitter = diagnostic::EmitterWriter::new(~w2);
 125  
 126      // Compile the code
 127      let codemap = CodeMap::new();
 128      let diagnostic_handler = diagnostic::mk_handler(~emitter);
 129      let span_diagnostic_handler =
 130          diagnostic::mk_span_handler(diagnostic_handler, codemap);
 131  
 132      let sess = driver::build_session_(sessopts,
 133                                        None,

 147          Err(e) => fail!("couldn't run the test: {}{}", e,
 148                          if e.kind == io::PermissionDenied {

 150                          } else { "" }),

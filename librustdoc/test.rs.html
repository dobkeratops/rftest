  94  fn runtest(test: &str, cratename: &str, libs: HashSet<Path>, should_fail: bool,
  95             no_run: bool, loose_feature_gating: bool) {
  96      let test = maketest(test, cratename, loose_feature_gating);
  97      let input = driver::StrInput(test);
  98  
  99      let sessopts = session::Options {
 100          maybe_sysroot: Some(os::self_exe_path().unwrap().dir_path()),
 101          addl_lib_search_paths: RefCell::new(libs),
 102          crate_types: vec!(session::CrateTypeExecutable),
 103          output_types: vec!(link::OutputTypeExe),
 104          cg: session::CodegenOptions {
 105              prefer_dynamic: true,
 106              .. session::basic_codegen_options()
 107          },
 108          ..session::basic_options().clone()
 109      };
 110  
 111      // Shuffle around a few input and output handles here. We're going to pass
 112      // an explicit handle into rustc to collect output messages, but we also
 113      // want to catch the error message that rustc prints when it fails.
 114      //
 115      // We take our task-local stderr (likely set by the test runner), and move
 116      // it into another task. This helper task then acts as a sink for both the
 117      // stderr of this task and stderr of rustc itself, copying all the info onto
 118      // the stderr channel we originally started with.
 119      //
 120      // The basic idea is to not use a default_handler() for rustc, and then also
 121      // not print things by default to the actual stderr.
 122      let (tx, rx) = channel();
 123      let w1 = io::ChanWriter::new(tx);
 124      let w2 = w1.clone();
 125      let old = io::stdio::set_stderr(~w1);
 126      spawn(proc() {
 127          let mut p = io::ChanReader::new(rx);
 128          let mut err = old.unwrap_or(~io::stderr() as ~Writer:Send);
 129          io::util::copy(&mut p, &mut err).unwrap();
 130      });
 131      let emitter = diagnostic::EmitterWriter::new(~w2);
 132  
 133      // Compile the code
 134      let codemap = CodeMap::new();
 135      let diagnostic_handler = diagnostic::mk_handler(~emitter);
 136      let span_diagnostic_handler =
 137          diagnostic::mk_span_handler(diagnostic_handler, codemap);
 138  
 139      let sess = driver::build_session_(sessopts,
 140                                        None,
 141                                        span_diagnostic_handler);
 142  
 143      let outdir = TempDir::new("rustdoctest").expect("rustdoc needs a tempdir");
 144      let out = Some(outdir.path().clone());
 145      let cfg = driver::build_configuration(&sess);
 146      driver::compile_input(sess, cfg, &input, &out, &None);
 147  
 148      if no_run { return }
 149  
 150      // Run the code!
 151      let exe = outdir.path().join("rust_out");
 152      let out = Process::output(exe.as_str().unwrap(), []);
 153      match out {
 154          Err(e) => fail!("couldn't run the test: {}{}", e,
 155                          if e.kind == io::PermissionDenied {
 156                              " - maybe your tempdir is mounted with noexec?"
 157                          } else { "" }),
 158          Ok(out) => {
 159              if should_fail && out.status.success() {
 160                  fail!("test executable succeeded when it should have failed");
 161              } else if !should_fail && !out.status.success() {
 162                  fail!("test executable failed:\n{}",
 163                        str::from_utf8(out.error.as_slice()));
 164              }
 165          }
 166      }
 167  }

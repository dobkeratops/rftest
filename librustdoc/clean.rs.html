  237              self.structs.clean().move_iter().collect(),

  239              self.fns.clean().move_iter().collect(),

  241              self.mods.clean().move_iter().collect(),

  243              self.statics.clean().move_iter().collect(),

  251          // the outer `mod` item for the source code.
